70    #[inline]
71    fn clear_cofactor(p: &G2Affine) -> G2Affine {
72        // Based on Section 4.1 of https://eprint.iacr.org/2017/419.pdf
73        // [h(ψ)]P = [x^2 − x − 1]P + [x − 1]ψ(P) + (ψ^2)(2P)
74
75        let x: &'static [u64] = crate::Config::X;
76        let p_projective = p.into_group();
77
78        // [x]P
79        let x_p = Config::mul_affine(p, x);
80        // ψ(P)
81        let psi_p = p_power_endomorphism(p);
82        // (ψ^2)(2P)
83        let mut psi2_p2 = double_p_power_endomorphism(&p_projective.double());
84
85        // tmp = [x]P + ψ(P)
86        let mut tmp = x_p;
87        tmp += &psi_p;
88
89        // tmp2 = [x^2]P + [x]ψ(P)
90        let mut tmp2: Projective<Config> = tmp;
91        tmp2 = tmp2.mul_bigint(x);
92
93        // add up all the terms
94        psi2_p2 += tmp2;
95        psi2_p2 -= x_p;
96        psi2_p2 += &-psi_p;
97        (psi2_p2 - p_projective).into_affine()
98    }

141/// psi(x,y) is the untwist-Frobenius-twist endomorhism on E'(Fq2)
142fn p_power_endomorphism(p: &Affine<Config>) -> Affine<Config> {
143    // The p-power endomorphism for G2 is defined as follows:
144    // 1. Note that G2 is defined on curve E': y^2 = x^3 + 1/u.
145    //    To map a point (x, y) in E' to (s, t) in E,
146    //    one set s = x * (u ^ (1/3)), t = y * (u ^ (1/2)),
147    //    because E: y^2 = x^3 + 1.
148    // 2. Apply the Frobenius endomorphism (s, t) => (s', t'),
149    //    another point on curve E, where s' = s^p, t' = t^p.
150    // 3. Map the point from E back to E'; that is,
151    //    one set x' = s' / ((u) ^ (1/3)), y' = t' / ((u) ^ (1/2)).
152    //
153    // To sum up, it maps
154    // (x,y) -> (x^p * (u ^ ((p-1)/3)), y^p * (u ^ ((p-1)/2)))
155    // as implemented in the code as follows.
156
157    let mut res = *p;
158    res.x.frobenius_map_in_place(1);
159    res.y.frobenius_map_in_place(1);
160
161    res.x *= P_POWER_ENDOMORPHISM_COEFF_0;
162    res.y *= P_POWER_ENDOMORPHISM_COEFF_1;
163
164    res
165}

167/// For a p-power endomorphism psi(P), compute psi(psi(P))
168fn double_p_power_endomorphism(p: &Projective<Config>) -> Projective<Config> {
169    // p_power_endomorphism(&p_power_endomorphism(&p.into_affine())).into()
170    let mut res = *p;
171
172    res.x *= DOUBLE_P_POWER_ENDOMORPHISM_COEFF_0;
173    // u^((p^2 - 1)/2) == -1
174    res.y = res.y.neg();
175
176    res
177}
