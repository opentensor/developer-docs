42/// A connection to an open [Sqlite] database.
43///
44/// Because SQLite is an in-process database accessed by blocking API calls, SQLx uses a background
45/// thread and communicates with it via channels to allow non-blocking access to the database.
46///
47/// Dropping this struct will signal the worker thread to quit and close the database, though
48/// if an error occurs there is no way to pass it back to the user this way.
49///
50/// You can explicitly call [`.close()`][Self::close] to ensure the database is closed successfully
51/// or get an error otherwise.
52pub struct SqliteConnection {
53    optimize_on_close: OptimizeOnClose,
54    pub(crate) worker: ConnectionWorker,
55    pub(crate) row_channel_size: usize,
56}

74impl From<i32> for SqliteOperation {
75    fn from(value: i32) -> Self {
76        match value {
77            SQLITE_INSERT => SqliteOperation::Insert,
78            SQLITE_UPDATE => SqliteOperation::Update,
79            SQLITE_DELETE => SqliteOperation::Delete,
80            code => SqliteOperation::Unknown(code),
81        }
82    }

111impl ConnectionState {
112    /// Drops the `progress_handler_callback` if it exists.
113    pub(crate) fn remove_progress_handler(&mut self) {
114        if let Some(mut handler) = self.progress_handler_callback.take() {
115            unsafe {
116                sqlite3_progress_handler(self.handle.as_ptr(), 0, None, ptr::null_mut());
117                let _ = { Box::from_raw(handler.0.as_mut()) };
118            }
119        }
120    }
121
122    pub(crate) fn remove_update_hook(&mut self) {
123        if let Some(mut handler) = self.update_hook_callback.take() {
124            unsafe {
125                sqlite3_update_hook(self.handle.as_ptr(), None, ptr::null_mut());
126                let _ = { Box::from_raw(handler.0.as_mut()) };
127            }
128        }
129    }

139impl SqliteConnection {
140    pub(crate) async fn establish(options: &SqliteConnectOptions) -> Result<Self, Error> {
141        let params = EstablishParams::from_options(options)?;
142        let worker = ConnectionWorker::establish(params).await?;
143        Ok(Self {
144            optimize_on_close: options.optimize_on_close.clone(),
145            worker,
146            row_channel_size: options.row_channel_size,
147        })
148    }
149
150    /// Lock the SQLite database handle out from the worker thread so direct SQLite API calls can
151    /// be made safely.
152    ///
153    /// Returns an error if the worker thread crashed.
154    pub async fn lock_handle(&mut self) -> Result<LockedSqliteHandle<'_>, Error> {
155        let guard = self.worker.unlock_db().await?;
156
157        Ok(LockedSqliteHandle { guard })
158    }
159}

175    fn close(mut self) -> BoxFuture<'static, Result<(), Error>> {
176        Box::pin(async move {
177            if let OptimizeOnClose::Enabled { analysis_limit } = self.optimize_on_close {
178                let mut pragma_string = String::new();
179                if let Some(limit) = analysis_limit {
180                    write!(pragma_string, "PRAGMA analysis_limit = {limit}; ").ok();
181                }
182                pragma_string.push_str("PRAGMA optimize;");
183                self.execute(&*pragma_string).await?;
184            }
185            let shutdown = self.worker.shutdown();
186            // Drop the statement worker, which should
187            // cover all references to the connection handle outside of the worker thread
188            drop(self);
189            // Ensure the worker thread has terminated
190            shutdown.await
191        })
192    }
193
194    fn close_hard(self) -> BoxFuture<'static, Result<(), Error>> {
195        Box::pin(async move {
196            drop(self);
197            Ok(())
198        })
199    }
200
201    /// Ensure the background worker thread is alive and accepting commands.
202    fn ping(&mut self) -> BoxFuture<'_, Result<(), Error>> {
203        Box::pin(self.worker.ping())
204    }
205
206    fn begin(&mut self) -> BoxFuture<'_, Result<Transaction<'_, Self::Database>, Error>>
207    where
208        Self: Sized,
209    {
210        Transaction::begin(self)
211    }

213    fn cached_statements_size(&self) -> usize {
214        self.worker

216            .cached_statements_size
217            .load(std::sync::atomic::Ordering::Acquire)

220    fn clear_cached_statements(&mut self) -> BoxFuture<'_, Result<(), Error>> {
221        Box::pin(async move {
222            self.worker.clear_cache().await?;
223            Ok(())
224        })
225    }
226
227    #[inline]

229        // No-op.
230    }

232    #[doc(hidden)]
233    fn flush(&mut self) -> BoxFuture<'_, Result<(), Error>> {
234        // For SQLite, FLUSH does effectively nothing...
235        // Well, we could use this to ensure that the command channel has been cleared,
236        // but it would only develop a backlog if a lot of queries are executed and then cancelled
237        // partway through, and then this would only make that situation worse.
238        Box::pin(future::ok(()))
239    }
240
241    #[doc(hidden)]
242    fn should_flush(&self) -> bool {
243        false
244    }

247/// Implements a C binding to a progress callback. The function returns `0` if the
248/// user-provided callback returns `true`, and `1` otherwise to signal an interrupt.
249extern "C" fn progress_callback<F>(callback: *mut c_void) -> c_int
250where
251    F: FnMut() -> bool,
252{
253    unsafe {
254        let r = catch_unwind(|| {
255            let callback: *mut F = callback.cast::<F>();
256            (*callback)()
257        });
258        c_int::from(!r.unwrap_or_default())
259    }
260}
261
262extern "C" fn update_hook<F>(
263    callback: *mut c_void,
264    op_code: c_int,
265    database: *const c_char,
266    table: *const c_char,
267    rowid: i64,
268) where
269    F: FnMut(UpdateHookResult),
270{
271    unsafe {
272        let _ = catch_unwind(|| {
273            let callback: *mut F = callback.cast::<F>();
274            let operation: SqliteOperation = op_code.into();
275            let database = CStr::from_ptr(database).to_str().unwrap_or_default();
276            let table = CStr::from_ptr(table).to_str().unwrap_or_default();
277            (*callback)(UpdateHookResult {
278                operation,
279                database,
280                table,
281                rowid,
282            })
283        });
284    }
285}

287impl LockedSqliteHandle<'_> {
288    /// Returns the underlying sqlite3* connection handle.
289    ///
290    /// As long as this `LockedSqliteHandle` exists, it is guaranteed that the background thread
291    /// is not making FFI calls on this database handle or any of its statements.
292    ///
293    /// ### Note: The `sqlite3` type is semver-exempt.
294    /// This API exposes the `sqlite3` type from `libsqlite3-sys` crate for type safety.
295    /// However, we reserve the right to upgrade `libsqlite3-sys` as necessary.
296    ///
297    /// Thus, if you are making direct calls via `libsqlite3-sys` you should pin the version
298    /// of SQLx that you're using, and upgrade it and `libsqlite3-sys` manually as new
299    /// versions are released.
300    ///
301    /// See [the driver root docs][crate] for details.
302    pub fn as_raw_handle(&mut self) -> NonNull<sqlite3> {
303        self.guard.handle.as_non_null_ptr()
304    }
305
306    /// Apply a collation to the open database.
307    ///
308    /// See [`SqliteConnectOptions::collation()`] for details.
309    pub fn create_collation(
310        &mut self,
311        name: &str,
312        compare: impl Fn(&str, &str) -> Ordering + Send + Sync + 'static,
313    ) -> Result<(), Error> {
314        collation::create_collation(&mut self.guard.handle, name, compare)
315    }
316
317    /// Sets a progress handler that is invoked periodically during long running calls. If the progress callback
318    /// returns `false`, then the operation is interrupted.
319    ///
320    /// `num_ops` is the approximate number of [virtual machine instructions](https://www.sqlite.org/opcode.html)
321    /// that are evaluated between successive invocations of the callback. If `num_ops` is less than one then the
322    /// progress handler is disabled.
323    ///
324    /// Only a single progress handler may be defined at one time per database connection; setting a new progress
325    /// handler cancels the old one.
326    ///
327    /// The progress handler callback must not do anything that will modify the database connection that invoked
328    /// the progress handler. Note that sqlite3_prepare_v2() and sqlite3_step() both modify their database connections
329    /// in this context.
330    pub fn set_progress_handler<F>(&mut self, num_ops: i32, callback: F)
331    where
332        F: FnMut() -> bool + Send + 'static,
333    {
334        unsafe {
335            let callback_boxed = Box::new(callback);
336            // SAFETY: `Box::into_raw()` always returns a non-null pointer.
337            let callback = NonNull::new_unchecked(Box::into_raw(callback_boxed));
338            let handler = callback.as_ptr() as *mut _;
339            self.guard.remove_progress_handler();
340            self.guard.progress_handler_callback = Some(Handler(callback));
341
342            sqlite3_progress_handler(
343                self.as_raw_handle().as_mut(),
344                num_ops,
345                Some(progress_callback::<F>),
346                handler,
347            );
348        }
349    }
350
351    pub fn set_update_hook<F>(&mut self, callback: F)
352    where
353        F: FnMut(UpdateHookResult) + Send + 'static,
354    {
355        unsafe {
356            let callback_boxed = Box::new(callback);
357            // SAFETY: `Box::into_raw()` always returns a non-null pointer.
358            let callback = NonNull::new_unchecked(Box::into_raw(callback_boxed));
359            let handler = callback.as_ptr() as *mut _;
360            self.guard.remove_update_hook();
361            self.guard.update_hook_callback = Some(UpdateHookHandler(callback));
362
363            sqlite3_update_hook(
364                self.as_raw_handle().as_mut(),
365                Some(update_hook::<F>),
366                handler,
367            );
368        }
369    }
370
371    /// Removes the progress handler on a database connection. The method does nothing if no handler was set.
372    pub fn remove_progress_handler(&mut self) {
373        self.guard.remove_progress_handler();
374    }
375
376    pub fn remove_update_hook(&mut self) {
377        self.guard.remove_update_hook();
378    }
379}

381impl Drop for ConnectionState {
382    fn drop(&mut self) {
383        // explicitly drop statements before the connection handle is dropped
384        self.statements.clear();
385        self.remove_progress_handler();
386        self.remove_update_hook();
387    }
388}

390impl Statements {
391    fn new(capacity: usize) -> Self {
392        Statements {
393            cached: StatementCache::new(capacity),
394            temp: None,
395        }
396    }
397
398    fn get(&mut self, query: &str, persistent: bool) -> Result<&mut VirtualStatement, Error> {
399        if !persistent || !self.cached.is_enabled() {
400            return Ok(self.temp.insert(VirtualStatement::new(query, false)?));
401        }
402
403        let exists = self.cached.contains_key(query);
404
405        if !exists {
406            let statement = VirtualStatement::new(query, true)?;
407            self.cached.insert(query, statement);
408        }
409
410        let statement = self.cached.get_mut(query).unwrap();
411
412        if exists {
413            // as this statement has been executed before, we reset before continuing
414            statement.reset()?;
415        }
416
417        Ok(statement)
418    }
419
420    fn len(&self) -> usize {
421        self.cached.len()
422    }
423
424    fn clear(&mut self) {
425        self.cached.clear();
426        self.temp = None;
427    }
