18//! # Utility Pallet
19//! A stateless pallet with helpers for dispatch management which does no re-authentication.
20//!
21//! - [`Config`]
22//! - [`Call`]
23//!
24//! ## Overview
25//!
26//! This pallet contains two basic pieces of functionality:
27//! - Batch dispatch: A stateless operation, allowing any origin to execute multiple calls in a
28//!   single dispatch. This can be useful to amalgamate proposals, combining `set_code` with
29//!   corresponding `set_storage`s, for efficient multiple payouts with just a single signature
30//!   verify, or in combination with one of the other two dispatch functionality.
31//! - Pseudonymal dispatch: A stateless operation, allowing a signed origin to execute a call from
32//!   an alternative signed origin. Each account has 2 * 2**16 possible "pseudonyms" (alternative
33//!   account IDs) and these can be stacked. This can be useful as a key management tool, where you
34//!   need multiple distinct accounts (e.g. as controllers for many staking accounts), but where
35//!   it's perfectly fine to have each of them controlled by the same underlying keypair. Derivative
36//!   accounts are, for the purposes of proxy filtering considered exactly the same as the origin
37//!   and are thus hampered with the origin's filters.
38//!
39//! Since proxy filters are respected in all dispatches of this pallet, it should never need to be
40//! filtered by any proxy.
41//!
42//! ## Interface
43//!
44//! ### Dispatchable Functions
45//!
46//! #### For batch dispatch
47//! * `batch` - Dispatch multiple calls from the sender's origin.
48//!
49//! #### For pseudonymal dispatch
50//! * `as_derivative` - Dispatch a call from a derivative signed origin.
51

109    #[pallet::event]

111    pub enum Event {
112        /// Batch of dispatches did not complete fully. Index of first failing dispatch given, as
113        /// well as the error.
114        BatchInterrupted { index: u32, error: DispatchError },
115        /// Batch of dispatches completed fully with no error.
116        BatchCompleted,
117        /// Batch of dispatches completed but has errors.
118        BatchCompletedWithErrors,
119        /// A single item within a Batch of dispatches has completed with no error.
120        ItemCompleted,
121        /// A single item within a Batch of dispatches has completed with error.
122        ItemFailed { error: DispatchError },
123        /// A call was dispatched.
124        DispatchedAs { result: DispatchResult },
125    }

127    // Align the call size to 1KB. As we are currently compiling the runtime for native/wasm
128    // the `size_of` of the `Call` can be different. To ensure that this don't leads to
129    // mismatches between native/wasm or to different metadata for the same runtime, we
130    // algin the call size. The value is chosen big enough to hopefully never reach it.
131    const CALL_ALIGN: u32 = 1024;
132
133    #[pallet::extra_constants]
134    impl<T: Config> Pallet<T> {
135        /// The limit on the number of batched calls.
136        fn batched_calls_limit() -> u32 {
137            let allocator_limit = sp_core::MAX_POSSIBLE_ALLOCATION;
138            let size = core::mem::size_of::<<T as Config>::RuntimeCall>() as u32;
139
140            let align_up = size.saturating_add(CALL_ALIGN.saturating_sub(1));
141            let call_size = align_up
142                .checked_div(CALL_ALIGN)
143                .unwrap_or(0)
144                .saturating_mul(CALL_ALIGN);
145
146            let margin_factor: u32 = 3;
147
148            let after_margin = allocator_limit.checked_div(margin_factor).unwrap_or(0);
149
150            after_margin.checked_div(call_size).unwrap_or(0)
151        }
152    }

172    #[pallet::call]
173    impl<T: Config> Pallet<T> {
174        /// Send a batch of dispatch calls.
175        ///
176        /// May be called from any origin except `None`.
177        ///
178        /// - `calls`: The calls to be dispatched from the same origin. The number of call must not
179        ///   exceed the constant: `batched_calls_limit` (available in constant metadata).
180        ///
181        /// If origin is root then the calls are dispatched without checking origin filter. (This
182        /// includes bypassing `frame_system::Config::BaseCallFilter`).
183        ///
184        /// ## Complexity
185        /// - O(C) where C is the number of calls to be batched.
186        ///
187        /// This will return `Ok` in all circumstances. To determine the success of the batch, an
188        /// event is deposited. If a call failed and the batch was interrupted, then the
189        /// `BatchInterrupted` event is deposited, along with the number of successful calls made
190        /// and the error of the failed call. If all were successful, then the `BatchCompleted`
191        /// event is deposited.
192        #[pallet::call_index(0)]
193        #[pallet::weight({
194			let (dispatch_weight, dispatch_class) = Pallet::<T>::weight_and_dispatch_class(calls);
195			let dispatch_weight = dispatch_weight.saturating_add(T::WeightInfo::batch(calls.len() as u32));
196			(dispatch_weight, dispatch_class)
197		})]
198        pub fn batch(
199            origin: OriginFor<T>,
200            calls: Vec<<T as Config>::RuntimeCall>,
201        ) -> DispatchResultWithPostInfo {
202            // Do not allow the `None` origin.
203            if ensure_none(origin.clone()).is_ok() {
204                return Err(BadOrigin.into());
205            }
206
207            let is_root = ensure_root(origin.clone()).is_ok();
208            let calls_len = calls.len();
209            ensure!(
210                calls_len <= Self::batched_calls_limit() as usize,
211                Error::<T>::TooManyCalls
212            );
213
214            // Track the actual weight of each of the batch calls.
215            let mut weight = Weight::zero();
216            for (index, call) in calls.into_iter().enumerate() {
217                let info = call.get_dispatch_info();
218                // If origin is root, don't apply any dispatch filters; root can call anything.
219                let result = if is_root {
220                    call.dispatch_bypass_filter(origin.clone())
221                } else {
222                    call.dispatch(origin.clone())
223                };
224                // Add the weight of this call.
225                weight = weight.saturating_add(extract_actual_weight(&result, &info));
226                if let Err(e) = result {
227                    Self::deposit_event(Event::BatchInterrupted {
228                        index: index as u32,
229                        error: e.error,
230                    });
231                    // Take the weight of this function itself into account.
232                    let base_weight = T::WeightInfo::batch(index.saturating_add(1) as u32);
233                    // Return the actual used weight + base_weight of this call.
234                    return Ok(Some(base_weight.saturating_add(weight)).into());
235                }
236                Self::deposit_event(Event::ItemCompleted);
237            }
238            Self::deposit_event(Event::BatchCompleted);
239            let base_weight = T::WeightInfo::batch(calls_len as u32);
240            Ok(Some(base_weight.saturating_add(weight)).into())
241        }
242
243        /// Send a call through an indexed pseudonym of the sender.
244        ///
245        /// Filter from origin are passed along. The call will be dispatched with an origin which
246        /// use the same filter as the origin of this call.
247        ///
248        /// NOTE: If you need to ensure that any account-based filtering is not honored (i.e.
249        /// because you expect `proxy` to have been used prior in the call stack and you do not want
250        /// the call restrictions to apply to any sub-accounts), then use `as_multi_threshold_1`
251        /// in the Multisig pallet instead.
252        ///
253        /// NOTE: Prior to version *12, this was called `as_limited_sub`.
254        ///
255        /// The dispatch origin for this call must be _Signed_.
256        #[pallet::call_index(1)]
257        #[pallet::weight({
258			let dispatch_info = call.get_dispatch_info();
259			(
260				T::WeightInfo::as_derivative()
261					// AccountData for inner call origin accountdata.
262					.saturating_add(T::DbWeight::get().reads_writes(1, 1))
263					.saturating_add(dispatch_info.weight),
264				dispatch_info.class,
265			)
266		})]
267        pub fn as_derivative(
268            origin: OriginFor<T>,
269            index: u16,
270            call: Box<<T as Config>::RuntimeCall>,
271        ) -> DispatchResultWithPostInfo {
272            let mut origin = origin;
273            let who = ensure_signed(origin.clone())?;
274            let pseudonym = Self::derivative_account_id(who, index);
275            origin.set_caller_from(frame_system::RawOrigin::Signed(pseudonym));
276            let info = call.get_dispatch_info();
277            let result = call.dispatch(origin);
278            // Always take into account the base weight of this call.
279            let mut weight = T::WeightInfo::as_derivative()
280                .saturating_add(T::DbWeight::get().reads_writes(1, 1));
281            // Add the real weight of the dispatch.
282            weight = weight.saturating_add(extract_actual_weight(&result, &info));
283            result
284                .map_err(|mut err| {
285                    err.post_info = Some(weight).into();
286                    err
287                })
288                .map(|_| Some(weight).into())
289        }
290
291        /// Send a batch of dispatch calls and atomically execute them.
292        /// The whole transaction will rollback and fail if any of the calls failed.
293        ///
294        /// May be called from any origin except `None`.
295        ///
296        /// - `calls`: The calls to be dispatched from the same origin. The number of call must not
297        ///   exceed the constant: `batched_calls_limit` (available in constant metadata).
298        ///
299        /// If origin is root then the calls are dispatched without checking origin filter. (This
300        /// includes bypassing `frame_system::Config::BaseCallFilter`).
301        ///
302        /// ## Complexity
303        /// - O(C) where C is the number of calls to be batched.
304        #[pallet::call_index(2)]
305        #[pallet::weight({
306			let (dispatch_weight, dispatch_class) = Pallet::<T>::weight_and_dispatch_class(calls);
307			let dispatch_weight = dispatch_weight.saturating_add(T::WeightInfo::batch_all(calls.len() as u32));
308			(dispatch_weight, dispatch_class)
309		})]
310        pub fn batch_all(
311            origin: OriginFor<T>,
312            calls: Vec<<T as Config>::RuntimeCall>,
313        ) -> DispatchResultWithPostInfo {
314            // Do not allow the `None` origin.
315            if ensure_none(origin.clone()).is_ok() {
316                return Err(BadOrigin.into());
317            }
318
319            let is_root = ensure_root(origin.clone()).is_ok();
320            let calls_len = calls.len();
321            ensure!(
322                calls_len <= Self::batched_calls_limit() as usize,
323                Error::<T>::TooManyCalls
324            );
325
326            // Track the actual weight of each of the batch calls.
327            let mut weight = Weight::zero();
328            for (index, call) in calls.into_iter().enumerate() {
329                let info = call.get_dispatch_info();
330                // If origin is root, bypass any dispatch filter; root can call anything.
331                let result = if is_root {
332                    call.dispatch_bypass_filter(origin.clone())
333                } else {
334                    let mut filtered_origin = origin.clone();
335                    // Don't allow users to nest `batch_all` calls.
336                    filtered_origin.add_filter(
337                        move |c: &<T as frame_system::Config>::RuntimeCall| {
338                            let c = <T as Config>::RuntimeCall::from_ref(c);
339                            !matches!(c.is_sub_type(), Some(Call::batch_all { .. }))
340                        },
341                    );
342                    call.dispatch(filtered_origin)
343                };
344                // Add the weight of this call.
345                weight = weight.saturating_add(extract_actual_weight(&result, &info));
346                result.map_err(|mut err| {
347                    // Take the weight of this function itself into account.
348                    let base_weight = T::WeightInfo::batch_all(index.saturating_add(1) as u32);
349                    // Return the actual used weight + base_weight of this call.
350                    err.post_info = Some(base_weight.saturating_add(weight)).into();
351                    err
352                })?;
353                Self::deposit_event(Event::ItemCompleted);
354            }
355            Self::deposit_event(Event::BatchCompleted);
356            let base_weight = T::WeightInfo::batch_all(calls_len as u32);
357            Ok(Some(base_weight.saturating_add(weight)).into())
358        }
359
360        /// Dispatches a function call with a provided origin.
361        ///
362        /// The dispatch origin for this call must be _Root_.
363        ///
364        /// ## Complexity
365        /// - O(1).
366        #[pallet::call_index(3)]
367        #[pallet::weight({
368			let dispatch_info = call.get_dispatch_info();
369			(
370				T::WeightInfo::dispatch_as()
371					.saturating_add(dispatch_info.weight),
372				dispatch_info.class,
373			)
374		})]
375        pub fn dispatch_as(
376            origin: OriginFor<T>,
377            as_origin: Box<T::PalletsOrigin>,
378            call: Box<<T as Config>::RuntimeCall>,
379        ) -> DispatchResult {
380            ensure_root(origin)?;
381
382            let res = call.dispatch_bypass_filter((*as_origin).into());
383
384            Self::deposit_event(Event::DispatchedAs {
385                result: res.map(|_| ()).map_err(|e| e.error),
386            });
387            Ok(())
388        }
389
390        /// Send a batch of dispatch calls.
391        /// Unlike `batch`, it allows errors and won't interrupt.
392        ///
393        /// May be called from any origin except `None`.
394        ///
395        /// - `calls`: The calls to be dispatched from the same origin. The number of call must not
396        ///   exceed the constant: `batched_calls_limit` (available in constant metadata).
397        ///
398        /// If origin is root then the calls are dispatch without checking origin filter. (This
399        /// includes bypassing `frame_system::Config::BaseCallFilter`).
400        ///
401        /// ## Complexity
402        /// - O(C) where C is the number of calls to be batched.
403        #[pallet::call_index(4)]
404        #[pallet::weight({
405			let (dispatch_weight, dispatch_class) = Pallet::<T>::weight_and_dispatch_class(calls);
406			let dispatch_weight = dispatch_weight.saturating_add(T::WeightInfo::force_batch(calls.len() as u32));
407			(dispatch_weight, dispatch_class)
408		})]
409        pub fn force_batch(
410            origin: OriginFor<T>,
411            calls: Vec<<T as Config>::RuntimeCall>,
412        ) -> DispatchResultWithPostInfo {
413            // Do not allow the `None` origin.
414            if ensure_none(origin.clone()).is_ok() {
415                return Err(BadOrigin.into());
416            }
417
418            let is_root = ensure_root(origin.clone()).is_ok();
419            let calls_len = calls.len();
420            ensure!(
421                calls_len <= Self::batched_calls_limit() as usize,
422                Error::<T>::TooManyCalls
423            );
424
425            // Track the actual weight of each of the batch calls.
426            let mut weight = Weight::zero();
427            // Track failed dispatch occur.
428            let mut has_error: bool = false;
429            for call in calls.into_iter() {
430                let info = call.get_dispatch_info();
431                // If origin is root, don't apply any dispatch filters; root can call anything.
432                let result = if is_root {
433                    call.dispatch_bypass_filter(origin.clone())
434                } else {
435                    call.dispatch(origin.clone())
436                };
437                // Add the weight of this call.
438                weight = weight.saturating_add(extract_actual_weight(&result, &info));
439                if let Err(e) = result {
440                    has_error = true;
441                    Self::deposit_event(Event::ItemFailed { error: e.error });
442                } else {
443                    Self::deposit_event(Event::ItemCompleted);
444                }
445            }
446            if has_error {
447                Self::deposit_event(Event::BatchCompletedWithErrors);
448            } else {
449                Self::deposit_event(Event::BatchCompleted);
450            }
451            let base_weight = T::WeightInfo::batch(calls_len as u32);
452            Ok(Some(base_weight.saturating_add(weight)).into())
453        }
454
455        /// Dispatch a function call with a specified weight.
456        ///
457        /// This function does not check the weight of the call, and instead allows the
458        /// Root origin to specify the weight of the call.
459        ///
460        /// The dispatch origin for this call must be _Root_.
461        #[pallet::call_index(5)]
462        #[pallet::weight((*weight, call.get_dispatch_info().class))]
463        pub fn with_weight(
464            origin: OriginFor<T>,
465            call: Box<<T as Config>::RuntimeCall>,
466            weight: Weight,
467        ) -> DispatchResult {
468            ensure_root(origin)?;
469            let _ = weight; // Explicitly don't check the the weight witness.
470
471            let res = call.dispatch_bypass_filter(frame_system::RawOrigin::Root.into());
472            res.map(|_| ()).map_err(|e| e.error)
473        }
474    }

476    impl<T: Config> Pallet<T> {
477        /// Get the accumulated `weight` and the dispatch class for the given `calls`.
478        fn weight_and_dispatch_class(
479            calls: &[<T as Config>::RuntimeCall],
480        ) -> (Weight, DispatchClass) {
481            let dispatch_infos = calls.iter().map(|call| call.get_dispatch_info());
482            let (dispatch_weight, dispatch_class) = dispatch_infos.fold(
483                (Weight::zero(), DispatchClass::Operational),
484                |(total_weight, dispatch_class), di| {
485                    (
486                        if di.pays_fee == Pays::Yes {
487                            total_weight.saturating_add(di.weight)
488                        } else {
489                            total_weight
490                        },
491                        if di.class == DispatchClass::Normal {
492                            di.class
493                        } else {
494                            dispatch_class
495                        },
496                    )
497                },
498            );
499
500            (dispatch_weight, dispatch_class)
501        }
502    }

514impl<T: Config> Pallet<T> {
515    /// Derive a derivative account ID from the owner account and the sub-account index.
516    pub fn derivative_account_id(who: T::AccountId, index: u16) -> T::AccountId {
517        let entropy = (b"modlpy/utilisuba", who, index).using_encoded(blake2_256);
518        Decode::decode(&mut TrailingZeroInput::new(entropy.as_ref()))
519            .expect("infinite length input; no invalid inputs for type; qed")
520    }
