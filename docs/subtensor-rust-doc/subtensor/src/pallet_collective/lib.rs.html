18//! Collective system: Members of a set of account IDs can make their collective feelings known
19//! through dispatched calls from one of two specialized origins.
20//!
21//! The membership can be provided in one of two ways: either directly, using the Root-dispatchable
22//! function `set_members`, or indirectly, through implementing the `ChangeMembers`.
23//! The pallet assumes that the amount of members stays at or below `MaxMembers` for its weight
24//! calculations, but enforces this neither in `set_members` nor in `change_members_sorted`.
25//!
26//! A "prime" member may be set to help determine the default vote behavior based on chain
27//! config. If `PrimeDefaultVote` is used, the prime vote acts as the default vote in case of any
28//! abstentions after the voting period. If `MoreThanMajorityThenPrimeDefaultVote` is used, then
29//! abstentions will first follow the majority of the collective voting, and then the prime
30//! member.
31//!
32//! Voting happens through motions comprising a proposal (i.e. a curried dispatchable) plus a
33//! number of approvals required for it to pass and be called. Motions are open for members to
34//! vote on for a minimum period given by `MotionDuration`. As soon as the needed number of
35//! approvals is given, the motion is closed and executed. If the number of approvals is not reached
36//! during the voting period, then `close` may be called by any account in order to force the end
37//! the motion explicitly. If a prime member is defined then their vote is used in place of any
38//! abstentions and the proposal is executed if there are enough approvals counting the new votes.
39//!
40//! If there are not, or if no prime is set, then the motion is dropped without being executed.
41

76/// A number of members.
77///
78/// This also serves as a number of voting members, and since for motions, each member may
79/// vote exactly once, therefore also the number of votes for any given motion.
80pub type MemberCount = u32;

82/// Default voting strategy when a member is inactive.
83pub trait DefaultVote {
84    /// Get the default voting strategy, given:
85    ///
86    /// - Whether the prime member voted Aye.
87    /// - Raw number of yes votes.
88    /// - Raw number of no votes.
89    /// - Total number of member count.
90    fn default_vote(
91        prime_vote: Option<bool>,
92        yes_votes: MemberCount,
93        no_votes: MemberCount,
94        len: MemberCount,
95    ) -> bool;
96}

98/// Set the prime member's vote as the default vote.
99pub struct PrimeDefaultVote;
100
101impl DefaultVote for PrimeDefaultVote {
102    fn default_vote(
103        prime_vote: Option<bool>,
104        _yes_votes: MemberCount,
105        _no_votes: MemberCount,
106        _len: MemberCount,
107    ) -> bool {
108        prime_vote.unwrap_or(false)
109    }
110}
111
112/// First see if yes vote are over majority of the whole collective. If so, set the default vote
113/// as yes. Otherwise, use the prime member's vote as the default vote.
114pub struct MoreThanMajorityThenPrimeDefaultVote;
115
116impl DefaultVote for MoreThanMajorityThenPrimeDefaultVote {
117    fn default_vote(
118        prime_vote: Option<bool>,
119        yes_votes: MemberCount,
120        _no_votes: MemberCount,
121        len: MemberCount,
122    ) -> bool {
123        let more_than_majority = yes_votes.saturating_mul(2) > len;
124        more_than_majority || prime_vote.unwrap_or(false)
125    }
126}

132pub enum RawOrigin<AccountId, I> {
133    /// It has been condoned by a given number of members of the collective from a given total.
134    Members(MemberCount, MemberCount),
135    /// It has been condoned by a single member of the collective.
136    Member(AccountId),
137    /// Dummy to manage the fact we have instancing.
138    _Phantom(PhantomData<I>),

141impl<AccountId, I> GetBacking for RawOrigin<AccountId, I> {
142    fn get_backing(&self) -> Option<Backing> {
143        match self {
144            RawOrigin::Members(n, d) => Some(Backing {
145                approvals: *n,
146                eligible: *d,
147            }),
148            _ => None,
149        }
150    }

333            /// The hash of the proposal.
334            proposal_hash: T::Hash,
335        },
336        /// A motion was not approved by the required threshold.
337        Disapproved {
338            /// The hash of the proposal.
339            proposal_hash: T::Hash,
340        },
341        /// A motion was executed; result will be `Ok` if it returned without error.
342        Executed {
343            /// The hash of the proposal.
344            proposal_hash: T::Hash,
345            /// The result of the execution.

347        },
348        /// A single member did some action; result will be `Ok` if it returned without error.
349        MemberExecuted {
350            /// The hash of the proposal.
351            proposal_hash: T::Hash,
352            /// The result of the execution.
353            result: DispatchResult,
354        },
355        /// A proposal was closed because its threshold was reached or after its duration was up.
356        Closed {
357            /// The hash of the proposal.
358            proposal_hash: T::Hash,
359            /// Whether the proposal was approved.
360            yes: MemberCount,
361            /// Whether the proposal was rejected.

363        },
364    }

390    // Note that councillor operations are assigned to the operational class.
391    #[pallet::call]
392    impl<T: Config<I>, I: 'static> Pallet<T, I> {
393        /// Set the collective's membership.
394        ///
395        /// - `new_members`: The new member list. Be nice to the chain and provide it sorted.
396        /// - `prime`: The prime member whose vote sets the default.
397        /// - `old_count`: The upper bound for the previous number of members in storage. Used for
398        ///   weight estimation.
399        ///
400        /// The dispatch of this call must be `SetMembersOrigin`.
401        ///
402        /// NOTE: Does not enforce the expected `MaxMembers` limit on the amount of members, but
403        ///       the weight estimations rely on it to estimate dispatchable weight.
404        ///
405        /// # WARNING:
406        ///
407        /// The `pallet-collective` can also be managed by logic outside of the pallet through the
408        /// implementation of the trait [`ChangeMembers`].
409        /// Any call to `set_members` must be careful that the member set doesn't get out of sync
410        /// with other logic managing the member set.
411        ///
412        /// ## Complexity:
413        /// - `O(MP + N)` where:
414        ///   - `M` old-members-count (code- and governance-bounded)
415        ///   - `N` new-members-count (code- and governance-bounded)
416        ///   - `P` proposals-count (code-bounded)
417        #[pallet::call_index(0)]
418        #[pallet::weight((
419			T::WeightInfo::set_members(
420				*old_count, // M
421				new_members.len() as u32, // N
422				T::MaxProposals::get() // P
423			),
424			DispatchClass::Operational
425		))]
426        pub fn set_members(
427            origin: OriginFor<T>,
428            new_members: Vec<T::AccountId>,
429            prime: Option<T::AccountId>,
430            old_count: MemberCount,
431        ) -> DispatchResultWithPostInfo {
432            T::SetMembersOrigin::ensure_origin(origin)?;
433            if new_members.len() > T::MaxMembers::get() as usize {
434                log::error!(
435                    target: LOG_TARGET,
436                    "New members count ({}) exceeds maximum amount of members expected ({}).",
437                    new_members.len(),
438                    T::MaxMembers::get(),
439                );
440            }
441
442            let old = Members::<T, I>::get();
443            if old.len() > old_count as usize {
444                log::warn!(
445                    target: LOG_TARGET,
446                    "Wrong count used to estimate set_members weight. expected ({}) vs actual ({})",
447                    old_count,
448                    old.len(),
449                );
450            }
451            let mut new_members = new_members;
452            new_members.sort();
453            <Self as ChangeMembers<T::AccountId>>::set_members_sorted(&new_members, &old);
454            Prime::<T, I>::set(prime);
455
456            Ok(Some(T::WeightInfo::set_members(
457                old.len() as u32,         // M
458                new_members.len() as u32, // N
459                T::MaxProposals::get(),   // P
460            ))
461            .into())
462        }
463
464        /// Dispatch a proposal from a member using the `Member` origin.
465        ///
466        /// Origin must be a member of the collective.
467        ///
468        /// ## Complexity:
469        /// - `O(B + M + P)` where:
470        /// - `B` is `proposal` size in bytes (length-fee-bounded)
471        /// - `M` members-count (code-bounded)
472        /// - `P` complexity of dispatching `proposal`
473        #[pallet::call_index(1)]
474        #[pallet::weight((
475			T::WeightInfo::execute(
476				*length_bound, // B
477				T::MaxMembers::get(), // M
478			).saturating_add(proposal.get_dispatch_info().weight), // P
479			DispatchClass::Operational
480		))]
481        pub fn execute(
482            origin: OriginFor<T>,
483            proposal: Box<<T as Config<I>>::Proposal>,
484            #[pallet::compact] length_bound: u32,
485        ) -> DispatchResultWithPostInfo {
486            let who = ensure_signed(origin)?;
487            let members = Self::members();
488            ensure!(members.contains(&who), Error::<T, I>::NotMember);
489
490            let proposal_len = proposal.encoded_size();
491            ensure!(
492                proposal_len <= length_bound as usize,
493                Error::<T, I>::ProposalLengthBoundLessThanProposalLength
494            );
495
496            let proposal_hash = T::Hashing::hash_of(&proposal);
497            let result = proposal.dispatch(RawOrigin::Member(who).into());
498            Self::deposit_event(Event::MemberExecuted {
499                proposal_hash,
500                result: result.map(|_| ()).map_err(|e| e.error),
501            });
502
503            Ok(get_result_weight(result)
504                .map(|w| {
505                    T::WeightInfo::execute(
506                        proposal_len as u32,  // B
507                        members.len() as u32, // M
508                    )
509                    .saturating_add(w) // P
510                })
511                .into())
512        }
513
514        /// Add a new proposal to either be voted on or executed directly.
515        ///
516        /// Requires the sender to be member.
517        ///
518        /// `threshold` determines whether `proposal` is executed directly (`threshold < 2`)
519        /// or put up for voting.
520        ///
521        /// ## Complexity
522        /// - `O(B + M + P1)` or `O(B + M + P2)` where:
523        ///   - `B` is `proposal` size in bytes (length-fee-bounded)
524        ///   - `M` is members-count (code- and governance-bounded)
525        ///   - branching is influenced by `threshold` where:
526        ///     - `P1` is proposal execution complexity (`threshold < 2`)
527        ///     - `P2` is proposals-count (code-bounded) (`threshold >= 2`)
528        #[pallet::call_index(2)]
529        #[pallet::weight((
530			T::WeightInfo::propose_proposed(
531				*length_bound, // B
532				T::MaxMembers::get(), // M
533				T::MaxProposals::get(), // P2
534			),
535			DispatchClass::Operational
536		))]
537        pub fn propose(
538            origin: OriginFor<T>,
539            proposal: Box<<T as Config<I>>::Proposal>,
540            #[pallet::compact] length_bound: u32,
541            duration: BlockNumberFor<T>,
542        ) -> DispatchResultWithPostInfo {
543            let who = ensure_signed(origin.clone())?;
544            ensure!(T::CanPropose::can_propose(&who), Error::<T, I>::NotMember);
545
546            ensure!(
547                duration >= T::MotionDuration::get(),
548                Error::<T, I>::DurationLowerThanConfiguredMotionDuration
549            );
550
551            let threshold = T::GetVotingMembers::get_count()
552                .checked_div(2)
553                .unwrap_or(0)
554                .saturating_add(1);
555
556            let members = Self::members();
557            let (proposal_len, active_proposals) =
558                Self::do_propose_proposed(who, threshold, proposal, length_bound, duration)?;
559
560            Ok(Some(T::WeightInfo::propose_proposed(
561                proposal_len,         // B
562                members.len() as u32, // M
563                active_proposals,     // P2
564            ))
565            .into())
566        }
567
568        /// Add an aye or nay vote for the sender to the given proposal.
569        ///
570        /// Requires the sender to be a member.
571        ///
572        /// Transaction fees will be waived if the member is voting on any particular proposal
573        /// for the first time and the call is successful. Subsequent vote changes will charge a
574        /// fee.
575        /// ## Complexity
576        /// - `O(M)` where `M` is members-count (code- and governance-bounded)
577        #[pallet::call_index(3)]
578        #[pallet::weight((T::WeightInfo::vote(T::MaxMembers::get()), DispatchClass::Operational))]
579        pub fn vote(
580            origin: OriginFor<T>,
581            proposal: T::Hash,
582            #[pallet::compact] index: ProposalIndex,
583            approve: bool,
584        ) -> DispatchResultWithPostInfo {
585            let who = ensure_signed(origin.clone())?;
586            ensure!(T::CanVote::can_vote(&who), Error::<T, I>::NotMember);
587
588            let members = Self::members();
589            // Detects first vote of the member in the motion
590            let is_account_voting_first_time = Self::do_vote(who, proposal, index, approve)?;
591
592            if is_account_voting_first_time {
593                Ok((Some(T::WeightInfo::vote(members.len() as u32)), Pays::No).into())
594            } else {
595                Ok((Some(T::WeightInfo::vote(members.len() as u32)), Pays::Yes).into())
596            }
597        }
598
599        // NOTE: call_index(4) was `close_old_weight` and was removed due to weights v1
600        // deprecation
601
602        /// Disapprove a proposal, close, and remove it from the system, regardless of its current
603        /// state.
604        ///
605        /// Must be called by the Root origin.
606        ///
607        /// Parameters:
608        /// * `proposal_hash`: The hash of the proposal that should be disapproved.
609        ///
610        /// ## Complexity
611        /// O(P) where P is the number of max proposals
612        #[pallet::call_index(5)]
613        #[pallet::weight(T::WeightInfo::disapprove_proposal(T::MaxProposals::get()))]
614        pub fn disapprove_proposal(
615            origin: OriginFor<T>,
616            proposal_hash: T::Hash,
617        ) -> DispatchResultWithPostInfo {
618            ensure_root(origin)?;
619            let proposal_count = Self::do_disapprove_proposal(proposal_hash);
620            Ok(Some(T::WeightInfo::disapprove_proposal(proposal_count)).into())
621        }
622
623        /// Close a vote that is either approved, disapproved or whose voting period has ended.
624        ///
625        /// May be called by any signed account in order to finish voting and close the proposal.
626        ///
627        /// If called before the end of the voting period it will only close the vote if it is
628        /// has enough votes to be approved or disapproved.
629        ///
630        /// If called after the end of the voting period abstentions are counted as rejections
631        /// unless there is a prime member set and the prime member cast an approval.
632        ///
633        /// If the close operation completes successfully with disapproval, the transaction fee will
634        /// be waived. Otherwise execution of the approved operation will be charged to the caller.
635        ///
636        /// + `proposal_weight_bound`: The maximum amount of weight consumed by executing the closed
637        /// proposal.
638        /// + `length_bound`: The upper bound for the length of the proposal in storage. Checked via
639        /// `storage::read` so it is `size_of::<u32>() == 4` larger than the pure length.
640        ///
641        /// ## Complexity
642        /// - `O(B + M + P1 + P2)` where:
643        ///   - `B` is `proposal` size in bytes (length-fee-bounded)
644        ///   - `M` is members-count (code- and governance-bounded)
645        ///   - `P1` is the complexity of `proposal` preimage.
646        ///   - `P2` is proposal-count (code-bounded)
647        #[pallet::call_index(6)]
648        #[pallet::weight((
649			{
650				let b = *length_bound;
651				let m = T::MaxMembers::get();
652				let p1 = *proposal_weight_bound;
653				let p2 = T::MaxProposals::get();
654				T::WeightInfo::close_early_approved(b, m, p2)
655					.max(T::WeightInfo::close_early_disapproved(m, p2))
656					.max(T::WeightInfo::close_approved(b, m, p2))
657					.max(T::WeightInfo::close_disapproved(m, p2))
658					.saturating_add(p1)
659			},
660			DispatchClass::Operational
661		))]
662        pub fn close(
663            origin: OriginFor<T>,
664            proposal_hash: T::Hash,
665            #[pallet::compact] index: ProposalIndex,
666            proposal_weight_bound: Weight,
667            #[pallet::compact] length_bound: u32,
668        ) -> DispatchResultWithPostInfo {
669            ensure_root(origin)?;
670
671            Self::do_close(proposal_hash, index, proposal_weight_bound, length_bound)
672        }
673    }
674}

696    /// Add a new proposal to be voted.
697    pub fn do_propose_proposed(
698        who: T::AccountId,
699        threshold: MemberCount,
700        proposal: Box<<T as Config<I>>::Proposal>,
701        length_bound: MemberCount,
702        duration: BlockNumberFor<T>,
703    ) -> Result<(u32, u32), DispatchError> {
704        let proposal_len = proposal.encoded_size();
705        ensure!(
706            proposal_len <= length_bound as usize,
707            Error::<T, I>::ProposalLengthBoundLessThanProposalLength
708        );
709
710        let proposal_hash = T::Hashing::hash_of(&proposal);
711        ensure!(
712            !<ProposalOf<T, I>>::contains_key(proposal_hash),
713            Error::<T, I>::DuplicateProposal
714        );
715
716        let active_proposals =
717            <Proposals<T, I>>::try_mutate(|proposals| -> Result<usize, DispatchError> {
718                proposals
719                    .try_push(proposal_hash)
720                    .map_err(|_| Error::<T, I>::TooManyActiveProposals)?;
721                Ok(proposals.len())
722            })?;
723
724        let index = Self::proposal_count();
725        <ProposalCount<T, I>>::try_mutate(|i| {
726            *i = i
727                .checked_add(1)
728                .ok_or(Error::<T, I>::TooManyActiveProposals)?;
729            Ok::<(), Error<T, I>>(())
730        })?;
731        <ProposalOf<T, I>>::insert(proposal_hash, proposal);
732        let votes = {
733            let end = frame_system::Pallet::<T>::block_number().saturating_add(duration);
734            Votes {
735                index,
736                threshold,
737                ayes: vec![],
738                nays: vec![],
739                end,
740            }
741        };
742        <Voting<T, I>>::insert(proposal_hash, votes);
743
744        Self::deposit_event(Event::Proposed {
745            account: who,
746            proposal_index: index,
747            proposal_hash,
748            threshold,
749        });
750        Ok((proposal_len as u32, active_proposals as u32))
751    }

753    /// Add an aye or nay vote for the member to the given proposal, returns true if it's the first
754    /// vote of the member in the motion
755    pub fn do_vote(
756        who: T::AccountId,
757        proposal: T::Hash,
758        index: ProposalIndex,
759        approve: bool,
760    ) -> Result<bool, DispatchError> {
761        let mut voting = Self::voting(proposal).ok_or(Error::<T, I>::ProposalNotExists)?;
762        ensure!(
763            voting.index == index,
764            Error::<T, I>::IndexMismatchProposalHash
765        );
766
767        let position_yes = voting.ayes.iter().position(|a| a == &who);
768        let position_no = voting.nays.iter().position(|a| a == &who);
769
770        // Detects first vote of the member in the motion
771        let is_account_voting_first_time = position_yes.is_none() && position_no.is_none();
772
773        if approve {
774            if position_yes.is_none() {
775                voting.ayes.push(who.clone());
776            } else {
777                return Err(Error::<T, I>::DuplicateVote.into());
778            }
779            if let Some(pos) = position_no {
780                voting.nays.swap_remove(pos);
781            }
782        } else {
783            if position_no.is_none() {
784                voting.nays.push(who.clone());
785            } else {
786                return Err(Error::<T, I>::DuplicateVote.into());
787            }
788            if let Some(pos) = position_yes {
789                voting.ayes.swap_remove(pos);
790            }
791        }
792
793        let yes_votes = voting.ayes.len() as MemberCount;
794        let no_votes = voting.nays.len() as MemberCount;
795        Self::deposit_event(Event::Voted {
796            account: who,
797            proposal_hash: proposal,
798            voted: approve,
799            yes: yes_votes,
800            no: no_votes,
801        });
802
803        Voting::<T, I>::insert(proposal, voting);
804
805        Ok(is_account_voting_first_time)
806    }

808    /// Close a vote that is either approved, disapproved or whose voting period has ended.
809    pub fn do_close(
810        proposal_hash: T::Hash,
811        index: ProposalIndex,
812        proposal_weight_bound: Weight,
813        length_bound: u32,
814    ) -> DispatchResultWithPostInfo {
815        let voting = Self::voting(proposal_hash).ok_or(Error::<T, I>::ProposalNotExists)?;
816        ensure!(
817            voting.index == index,
818            Error::<T, I>::IndexMismatchProposalHash
819        );
820
821        let mut no_votes = voting.nays.len() as MemberCount;
822        let mut yes_votes = voting.ayes.len() as MemberCount;
823        let seats = T::GetVotingMembers::get_count() as MemberCount;
824        let approved = yes_votes >= voting.threshold;
825        let disapproved = seats.saturating_sub(no_votes) < voting.threshold;
826        // Allow (dis-)approving the proposal as soon as there are enough votes.
827        if approved {
828            let (proposal, len) = Self::validate_and_get_proposal(
829                &proposal_hash,
830                length_bound,
831                proposal_weight_bound,
832            )?;
833            Self::deposit_event(Event::Closed {
834                proposal_hash,
835                yes: yes_votes,
836                no: no_votes,
837            });
838            let (proposal_weight, proposal_count) =
839                Self::do_approve_proposal(seats, yes_votes, proposal_hash, proposal);
840            return Ok((
841                Some(
842                    T::WeightInfo::close_early_approved(len as u32, seats, proposal_count)
843                        .saturating_add(proposal_weight),
844                ),
845                Pays::Yes,
846            )
847                .into());
848        } else if disapproved {
849            Self::deposit_event(Event::Closed {
850                proposal_hash,
851                yes: yes_votes,
852                no: no_votes,
853            });
854            let proposal_count = Self::do_disapprove_proposal(proposal_hash);
855            return Ok((
856                Some(T::WeightInfo::close_early_disapproved(
857                    seats,
858                    proposal_count,
859                )),
860                Pays::No,
861            )
862                .into());
863        }
864
865        // Only allow actual closing of the proposal after the voting period has ended.
866        ensure!(
867            frame_system::Pallet::<T>::block_number() >= voting.end,
868            Error::<T, I>::TooEarlyToCloseProposal
869        );
870
871        let prime_vote = Self::prime().map(|who| voting.ayes.iter().any(|a| a == &who));
872
873        // default voting strategy.
874        let default = T::DefaultVote::default_vote(prime_vote, yes_votes, no_votes, seats);
875
876        let abstentions = seats.saturating_sub(yes_votes.saturating_add(no_votes));
877        match default {
878            true => yes_votes = yes_votes.saturating_add(abstentions),
879            false => no_votes = no_votes.saturating_add(abstentions),
880        }
881        let approved = yes_votes >= voting.threshold;
882
883        if approved {
884            let (proposal, len) = Self::validate_and_get_proposal(
885                &proposal_hash,
886                length_bound,
887                proposal_weight_bound,
888            )?;
889            Self::deposit_event(Event::Closed {
890                proposal_hash,
891                yes: yes_votes,
892                no: no_votes,
893            });
894            let (proposal_weight, proposal_count) =
895                Self::do_approve_proposal(seats, yes_votes, proposal_hash, proposal);
896            Ok((
897                Some(
898                    T::WeightInfo::close_approved(len as u32, seats, proposal_count)
899                        .saturating_add(proposal_weight),
900                ),
901                Pays::Yes,
902            )
903                .into())
904        } else {
905            Self::deposit_event(Event::Closed {
906                proposal_hash,
907                yes: yes_votes,
908                no: no_votes,
909            });
910            let proposal_count = Self::do_disapprove_proposal(proposal_hash);
911            Ok((
912                Some(T::WeightInfo::close_disapproved(seats, proposal_count)),
913                Pays::No,
914            )
915                .into())
916        }
917    }
918
919    /// Ensure that the right proposal bounds were passed and get the proposal from storage.
920    ///
921    /// Checks the length in storage via `storage::read` which adds an extra `size_of::<u32>() == 4`
922    /// to the length.
923    fn validate_and_get_proposal(
924        hash: &T::Hash,
925        length_bound: u32,
926        weight_bound: Weight,
927    ) -> Result<(<T as Config<I>>::Proposal, usize), DispatchError> {
928        let key = ProposalOf::<T, I>::hashed_key_for(hash);
929        // read the length of the proposal storage entry directly
930        let proposal_len =
931            storage::read(&key, &mut [0; 0], 0).ok_or(Error::<T, I>::ProposalNotExists)?;
932        ensure!(
933            proposal_len <= length_bound,
934            Error::<T, I>::ProposalLengthBoundLessThanProposalLength
935        );
936        let proposal = ProposalOf::<T, I>::get(hash).ok_or(Error::<T, I>::ProposalNotExists)?;
937        let proposal_weight = proposal.get_dispatch_info().weight;
938        ensure!(
939            proposal_weight.all_lte(weight_bound),
940            Error::<T, I>::ProposalWeightLessThanDispatchCallWeight
941        );
942        Ok((proposal, proposal_len as usize))
943    }
944
945    /// Weight:
946    /// If `approved`:
947    /// - the weight of `proposal` preimage.
948    /// - two events deposited.
949    /// - two removals, one mutation.
950    /// - computation and i/o `O(P + L)` where:
951    ///   - `P` is number of active proposals,
952    ///   - `L` is the encoded length of `proposal` preimage.
953    ///
954    /// If not `approved`:
955    /// - one event deposited.
956    /// - two removals, one mutation.
957    /// - computation and i/o `O(P)` where:
958    ///   - `P` is number of active proposals
959    fn do_approve_proposal(
960        seats: MemberCount,
961        yes_votes: MemberCount,
962        proposal_hash: T::Hash,
963        proposal: <T as Config<I>>::Proposal,
964    ) -> (Weight, u32) {
965        Self::deposit_event(Event::Approved { proposal_hash });
966
967        let dispatch_weight = proposal.get_dispatch_info().weight;
968        let origin = RawOrigin::Members(yes_votes, seats).into();
969        let result = proposal.dispatch(origin);
970        Self::deposit_event(Event::Executed {
971            proposal_hash,
972            result: result.map(|_| ()).map_err(|e| e.error),
973        });
974        // default to the dispatch info weight for safety
975        let proposal_weight = get_result_weight(result).unwrap_or(dispatch_weight); // P1
976
977        let proposal_count = Self::remove_proposal(proposal_hash);
978        (proposal_weight, proposal_count)
979    }
980
981    /// Removes a proposal from the pallet, and deposit the `Disapproved` event.
982    pub fn do_disapprove_proposal(proposal_hash: T::Hash) -> u32 {
983        // disapproved
984        Self::deposit_event(Event::Disapproved { proposal_hash });
985        Self::remove_proposal(proposal_hash)
986    }
987
988    // Removes a proposal from the pallet, cleaning up votes and the vector of proposals.
989    fn remove_proposal(proposal_hash: T::Hash) -> u32 {
990        // remove proposal and vote
991        ProposalOf::<T, I>::remove(proposal_hash);
992        Voting::<T, I>::remove(proposal_hash);
993        let num_proposals = Proposals::<T, I>::mutate(|proposals| {
994            proposals.retain(|h| h != &proposal_hash);
995            proposals.len().saturating_add(1) // calculate weight based on original length
996        });
997        num_proposals as u32
998    }
999
1000    pub fn remove_votes(who: &T::AccountId) -> Result<bool, DispatchError> {
1001        for h in Self::proposals().into_iter() {
1002            <Voting<T, I>>::mutate(h, |v| {
1003                if let Some(mut votes) = v.take() {
1004                    votes.ayes.retain(|i| i != who);
1005                    votes.nays.retain(|i| i != who);
1006                    *v = Some(votes);
1007                }
1008            });
1009        }
1010
1011        Ok(true)
1012    }

1014    pub fn has_voted(
1015        proposal: T::Hash,
1016        index: ProposalIndex,
1017        who: &T::AccountId,
1018    ) -> Result<bool, DispatchError> {
1019        let voting = Self::voting(proposal).ok_or(Error::<T, I>::ProposalNotExists)?;
1020        ensure!(
1021            voting.index == index,
1022            Error::<T, I>::IndexMismatchProposalHash
1023        );
1024
1025        let position_yes = voting.ayes.iter().position(|a| a == who);
1026        let position_no = voting.nays.iter().position(|a| a == who);
1027
1028        Ok(position_yes.is_some() || position_no.is_some())
1029    }

1032impl<T: Config<I>, I: 'static> ChangeMembers<T::AccountId> for Pallet<T, I> {
1033    /// Update the members of the collective. Votes are updated and the prime is reset.
1034    ///
1035    /// NOTE: Does not enforce the expected `MaxMembers` limit on the amount of members, but
1036    ///       the weight estimations rely on it to estimate dispatchable weight.
1037    ///
1038    /// ## Complexity
1039    /// - `O(MP + N)`
1040    ///   - where `M` old-members-count (governance-bounded)
1041    ///   - where `N` new-members-count (governance-bounded)
1042    ///   - where `P` proposals-count
1043    fn change_members_sorted(
1044        _incoming: &[T::AccountId],
1045        outgoing: &[T::AccountId],
1046        new: &[T::AccountId],
1047    ) {
1048        if new.len() > T::MaxMembers::get() as usize {
1049            log::error!(
1050                target: LOG_TARGET,
1051                "New members count ({}) exceeds maximum amount of members expected ({}).",
1052                new.len(),
1053                T::MaxMembers::get(),
1054            );
1055        }
1056        // remove accounts from all current voting in motions.
1057        let mut outgoing = outgoing.to_vec();
1058        outgoing.sort();
1059        for h in Self::proposals().into_iter() {
1060            <Voting<T, I>>::mutate(h, |v| {
1061                if let Some(mut votes) = v.take() {
1062                    votes.ayes.retain(|i| outgoing.binary_search(i).is_err());
1063                    votes.nays.retain(|i| outgoing.binary_search(i).is_err());
1064                    *v = Some(votes);
1065                }
1066            });
1067        }
1068        Members::<T, I>::put(new);
1069        Prime::<T, I>::kill();
1070    }
1071
1072    fn set_prime(prime: Option<T::AccountId>) {
1073        Prime::<T, I>::set(prime);
1074    }
1075
1076    fn get_prime() -> Option<T::AccountId> {
1077        Prime::<T, I>::get()
1078    }
1079}

1094/// otherwise.

1100    OuterOrigin: Into<result::Result<RawOrigin<AccountId, I>, OuterOrigin>>,
