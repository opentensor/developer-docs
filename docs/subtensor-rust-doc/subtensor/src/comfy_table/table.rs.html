19/// This is the main interface for building a table.
20/// Each table consists of [Rows](Row), which in turn contain [Cells](crate::cell::Cell).
21///
22/// There also exists a representation of a [Column].
23/// Columns are automatically created when adding rows to a table.
24#[derive(Debug)]
25pub struct Table {
26    pub(crate) columns: Vec<Column>,
27    style: HashMap<TableComponent, char>,
28    pub(crate) header: Option<Row>,
29    pub(crate) rows: Vec<Row>,
30    pub(crate) arrangement: ContentArrangement,
31    pub(crate) delimiter: Option<char>,
32    #[cfg(feature = "tty")]
33    no_tty: bool,
34    #[cfg(feature = "tty")]
35    use_stderr: bool,
36    width: Option<u16>,
37    #[cfg(feature = "tty")]
38    enforce_styling: bool,
39    /// Define whether everything in a cells should be styled, including whitespaces
40    /// or whether only the text should be styled.
41    #[cfg(feature = "tty")]
42    pub(crate) style_text_only: bool,
43}

264    ///
265    /// This will be `Some(width)`, if the terminal width can be detected or if the table width is set via [set_width](Table::set_width).
266    ///

399    /// Convenience method to set a [ColumnConstraint] for all columns at once.
400    /// Constraints are used to influence the way the columns will be arranged.
401    /// Check out their docs for more information.
402    ///
403    /// **Attention:**
404    /// This function should be called after at least one row (or the headers) has been added to the table.
405    /// Before that, the columns won't initialized.
406    ///
407    /// If more constraints are passed than there are columns, any superfluous constraints will be ignored.
408    /// ```
409    /// use comfy_table::{Width::*, CellAlignment, ColumnConstraint::*, ContentArrangement, Table};
410    ///
411    /// let mut table = Table::new();
412    /// table.add_row(&vec!["one", "two", "three"])
413    ///     .set_content_arrangement(ContentArrangement::Dynamic)
414    ///     .set_constraints(vec![
415    ///         UpperBoundary(Fixed(15)),
416    ///         LowerBoundary(Fixed(20)),
417    /// ]);
418    /// ```
419    pub fn set_constraints<T: IntoIterator<Item = ColumnConstraint>>(
420        &mut self,
421        constraints: T,
422    ) -> &mut Self {
423        let mut constraints = constraints.into_iter();
424        for column in self.column_iter_mut() {
425            if let Some(constraint) = constraints.next() {
426                column.set_constraint(constraint);
427            } else {
428                break;
429            }
430        }
431
432        self
433    }

435    /// This function creates a TableStyle from a given preset string.\
436    /// Preset strings can be found in `styling::presets::*`.
437    ///
438    /// You can also write your own preset strings and use them with this function.
439    /// There's the convenience method [Table::current_style_as_preset], which prints you a preset
440    /// string from your current style configuration. \
441    /// The function expects the to-be-drawn characters to be in the same order as in the [TableComponent] enum.
442    ///
443    /// If the string isn't long enough, the default [ASCII_FULL] style will be used for all remaining components.
444    ///
445    /// If the string is too long, remaining charaacters will be simply ignored.
446    pub fn load_preset(&mut self, preset: &str) -> &mut Self {
447        let mut components = TableComponent::iter();
448
449        for character in preset.chars() {
450            if let Some(component) = components.next() {
451                // White spaces mean "don't draw this" in presets
452                // If we want to override the default preset, we need to remove
453                // this component from the HashMap in case we find a whitespace.
454                if character == ' ' {
455                    self.remove_style(component);
456                    continue;
457                }
458
459                self.set_style(component, character);
460            } else {
461                break;
462            }
463        }
464
465        self
466    }

468    /// Returns the current style as a preset string.
469    ///
470    /// A pure convenience method, so you're not force to fiddle with those preset strings yourself.
471    ///
472    /// ```
473    /// use comfy_table::Table;
474    /// use comfy_table::presets::UTF8_FULL;
475    ///
476    /// let mut table = Table::new();
477    /// table.load_preset(UTF8_FULL);
478    ///
479    /// assert_eq!(UTF8_FULL, table.current_style_as_preset())
480    /// ```
481    pub fn current_style_as_preset(&mut self) -> String {
482        let components = TableComponent::iter();
483        let mut preset_string = String::new();
484
485        for component in components {
486            match self.style(component) {
487                None => preset_string.push(' '),
488                Some(character) => preset_string.push(character),
489            }
490        }
491
492        preset_string
493    }

495    /// Modify a preset with a modifier string from [modifiers](crate::style::modifiers).
496    ///
497    /// For instance, the [UTF8_ROUND_CORNERS](crate::style::modifiers::UTF8_ROUND_CORNERS) modifies all corners to be round UTF8 box corners.
498    ///
499    /// ```
500    /// use comfy_table::Table;
501    /// use comfy_table::presets::UTF8_FULL;
502    /// use comfy_table::modifiers::UTF8_ROUND_CORNERS;
503    ///
504    /// let mut table = Table::new();
505    /// table.load_preset(UTF8_FULL);
506    /// table.apply_modifier(UTF8_ROUND_CORNERS);
507    /// ```
508    pub fn apply_modifier(&mut self, modifier: &str) -> &mut Self {
509        let mut components = TableComponent::iter();
510
511        for character in modifier.chars() {
512            // Skip spaces while applying modifiers.
513            if character == ' ' {
514                components.next();
515                continue;
516            }
517            if let Some(component) = components.next() {
518                self.set_style(component, character);
519            } else {
520                break;
521            }
522        }
523
524        self
525    }

527    /// Define the char that will be used to draw a specific component.\
528    /// Look at [TableComponent] to see all stylable components
529    ///
530    /// If `None` is supplied, the element won't be displayed.\
531    /// In case of a e.g. *BorderIntersection a whitespace will be used as placeholder,
532    /// unless related borders and and corners are set to `None` as well.
533    ///
534    /// For example, if `TopBorderIntersections` is `None` the first row would look like this:
535    ///
536    /// ```text
537    /// +------ ------+
538    /// | this | test |
539    /// ```
540    ///
541    /// If in addition `TopLeftCorner`,`TopBorder` and `TopRightCorner` would be `None` as well,
542    /// the first line wouldn't be displayed at all.
543    ///
544    /// ```
545    /// use comfy_table::Table;
546    /// use comfy_table::presets::UTF8_FULL;
547    /// use comfy_table::TableComponent::*;
548    ///
549    /// let mut table = Table::new();
550    /// // Load the UTF8_FULL preset
551    /// table.load_preset(UTF8_FULL);
552    /// // Set all outer corners to round UTF8 corners
553    /// // This is basically the same as the UTF8_ROUND_CORNERS modifier
554    /// table.set_style(TopLeftCorner, '╭');
555    /// table.set_style(TopRightCorner, '╮');
556    /// table.set_style(BottomLeftCorner, '╰');
557    /// table.set_style(BottomRightCorner, '╯');
558    /// ```
559    pub fn set_style(&mut self, component: TableComponent, character: char) -> &mut Self {
560        self.style.insert(component, character);
561
562        self
563    }

719    /// Return a vector representing the maximum amount of characters in any line of this column.\
720    ///
721    /// **Attention** This scans the whole current content of the table.
722    pub fn column_max_content_widths(&self) -> Vec<u16> {
723        fn set_max_content_widths(max_widths: &mut [u16], row: &Row) {
724            // Get the max width for each cell of the row
725            let row_max_widths = row.max_content_widths();
726            for (index, width) in row_max_widths.iter().enumerate() {
727                let mut width = (*width).try_into().unwrap_or(u16::MAX);
728                // A column's content is at least 1 char wide.
729                width = std::cmp::max(1, width);
730
731                // Set a new max, if the current cell is the longest for that column.
732                let current_max = max_widths[index];
733                if current_max < width {
734                    max_widths[index] = width;
735                }
736            }
737        }
738        // The vector that'll contain the max widths per column.
739        let mut max_widths = vec![0; self.columns.len()];
740
741        if let Some(header) = &self.header {
742            set_max_content_widths(&mut max_widths, header);
743        }
744        // Iterate through all rows of the table.
745        for row in self.rows.iter() {
746            set_max_content_widths(&mut max_widths, row);
747        }
748
749        max_widths
750    }

763    /// Autogenerate new columns, if a row is added with more cells than existing columns.
764    fn autogenerate_columns(&mut self, row: &Row) {
765        if row.cell_count() > self.columns.len() {
766            for index in self.columns.len()..row.cell_count() {
767                self.columns.push(Column::new(index));
768            }
769        }
770    }

772    /// Calling this might be necessary if you add new cells to rows that're already added to the
773    /// table.
774    ///
775    /// If more cells than're currently know to the table are added to that row,
776    /// the table cannot know about these, since new [Column]s are only
777    /// automatically detected when a new row is added.
778    ///
779    /// To make sure everything works as expected, just call this function if you're adding cells
780    /// to rows that're already added to the table.
781    pub fn discover_columns(&mut self) {
782        for row in self.rows.iter() {
783            if row.cell_count() > self.columns.len() {
784                for index in self.columns.len()..row.cell_count() {
785                    self.columns.push(Column::new(index));
786                }
787            }
788        }
789    }

792/// An iterator over cells of a specific column.
793/// A dedicated struct is necessary, as data is usually handled by rows and thereby stored in
794/// `Table::rows`. This type is returned by [Table::column_cells_iter].
795pub struct ColumnCellIter<'a> {
796    rows: &'a [Row],
797    column_index: usize,
798    row_index: usize,
799}
800
801impl<'a> Iterator for ColumnCellIter<'a> {
802    type Item = Option<&'a Cell>;
803    fn next(&mut self) -> Option<Option<&'a Cell>> {
804        // Check if there's a next row
805        if let Some(row) = self.rows.get(self.row_index) {
806            self.row_index += 1;
807
808            // Return the cell (if it exists).
809            return Some(row.cells.get(self.column_index));
810        }
811
812        None
813    }
814}

816/// An iterator over cells of a specific column.
817/// A dedicated struct is necessary, as data is usually handled by rows and thereby stored in
818/// `Table::rows`. This type is returned by [Table::column_cells_iter].
819pub struct ColumnCellsWithHeaderIter<'a> {
820    header_checked: bool,
821    header: &'a Option<Row>,
822    rows: &'a [Row],
823    column_index: usize,
824    row_index: usize,
825}
826
827impl<'a> Iterator for ColumnCellsWithHeaderIter<'a> {
828    type Item = Option<&'a Cell>;
829    fn next(&mut self) -> Option<Option<&'a Cell>> {
830        // Get the header as the first cell
831        if !self.header_checked {
832            self.header_checked = true;
833
834            return match self.header {
835                Some(header) => {
836                    // Return the cell (if it exists).
837                    Some(header.cells.get(self.column_index))
838                }
839                None => Some(None),
840            };
841        }
842
843        // Check if there's a next row
844        if let Some(row) = self.rows.get(self.row_index) {
845            self.row_index += 1;
846
847            // Return the cell (if it exists).
848            return Some(row.cells.get(self.column_index));
849        }
850
851        None
852    }
853}
