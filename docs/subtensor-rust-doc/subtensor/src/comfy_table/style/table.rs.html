3/// Specify how comfy_table should arrange the content in your table.
4///
5/// ```
6/// use comfy_table::{Table, ContentArrangement};
7///
8/// let mut table = Table::new();
9/// table.set_content_arrangement(ContentArrangement::Dynamic);
10/// ```
11#[derive(Clone, Debug)]
12pub enum ContentArrangement {
13    /// Don't do any content arrangement.\
14    /// Tables with this mode might become wider than your output and look ugly.\
15    /// Constraints on columns are still respected.
16    Disabled,
17    /// Dynamically determine the width of columns in regard to terminal width and content length.\
18    /// With this mode, the content in cells will wrap dynamically to get the  the best column layout
19    /// for the given content.\
20    /// Constraints on columns are still respected.
21    ///
22    /// **Warning:** If terminal width cannot be determined and no table_width is set via
23    /// [Table::set_width](crate::table::Table::set_width),
24    /// this option won't work and [Disabled](ContentArrangement::Disabled) will be used as a fallback.
25    Dynamic,
26    /// This is mode is the same as the [ContentArrangement::Dynamic] arrangement, but it will always use as much
27    /// space as it's given. Any surplus space will be distributed between all columns.
28    DynamicFullWidth,
29}

31/// All configurable table components.
32/// A character can be assigned to each component via [Table::set_style](crate::table::Table::set_style).
33/// This is then used to draw character of the respective component to the commandline.
34///
35/// I hope that most component names are self-explanatory. Just in case:
36/// BorderIntersections are Intersections, where rows/columns lines meet outer borders.
37/// E.g.:
38/// ```text
39///        ---------
40///        v       |
41/// +---+---+---+  |
42/// | a | b | c |  |
43/// +===+===+===+<-|
44/// |   |   |   |  |
45/// +---+---+---+<-- These "+" chars are Borderintersections.
46/// |   |   |   |    The inner "+" chars are MiddleIntersections
47/// +---+---+---+
48/// ```
49#[derive(Debug, PartialEq, Eq, Hash, EnumIter, Clone, Copy)]
50pub enum TableComponent {
51    LeftBorder,
52    RightBorder,
53    TopBorder,
54    BottomBorder,
55    LeftHeaderIntersection,
56    HeaderLines,
57    MiddleHeaderIntersections,
58    RightHeaderIntersection,
59    VerticalLines,
60    HorizontalLines,
61    MiddleIntersections,
62    LeftBorderIntersections,
63    RightBorderIntersections,
64    TopBorderIntersections,
65    BottomBorderIntersections,
66    TopLeftCorner,
67    TopRightCorner,
68    BottomLeftCorner,
69    BottomRightCorner,
70}
