17/// Splits a long word at a given character width.
18/// This needs some special logic, as we have to take multi-character UTF-8 symbols into account.
19/// When simply splitting at a certain char position, we might end up with a string that's has a
20/// wider display width than allowed.
21pub fn split_long_word(allowed_width: usize, word: &str) -> (String, String) {
22    let mut current_width = 0;
23    let mut parts = String::new();
24
25    let mut char_iter = word.chars().peekable();
26    // Check if the string might be too long, one character at a time.
27    // Peek into the next char and check the exit condition.
28    // That is, pushing the next character would result in the string being too long.
29    while let Some(c) = char_iter.peek() {
30        if (current_width + c.width().unwrap_or(1)) > allowed_width {
31            break;
32        }
33
34        // We can unwrap, as we just checked that a suitable character is next in line.
35        let c = char_iter.next().unwrap();
36
37        // We default to 1 char, if the character length cannot be determined.
38        // The user has to live with this, if they decide to add control characters or some fancy
39        // stuff into their tables. This is considered undefined behavior and we try to handle this
40        // to the best of our capabilities.
41        let character_width = c.width().unwrap_or(1);
42
43        current_width += character_width;
44        parts.push(c);
45    }
46
47    // Collect the remaining characters.
48    let remaining = char_iter.collect();
49    (parts, remaining)
50}
