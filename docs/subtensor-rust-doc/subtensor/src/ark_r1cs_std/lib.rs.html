76/// This trait describes some core functionality that is common to high-level
77/// variables, such as `Boolean`s, `FieldVar`s, `GroupVar`s, etc.
78pub trait R1CSVar<F: Field> {
79    /// The type of the "native" value that `Self` represents in the constraint
80    /// system.
81    type Value: core::fmt::Debug + Eq + Clone;
82
83    /// Returns the underlying `ConstraintSystemRef`.
84    ///
85    /// If `self` is a constant value, then this *must* return
86    /// `ark_relations::r1cs::ConstraintSystemRef::None`.
87    fn cs(&self) -> ark_relations::r1cs::ConstraintSystemRef<F>;
88
89    /// Returns `true` if `self` is a circuit-generation-time constant.
90    fn is_constant(&self) -> bool {
91        self.cs().is_none()
92    }
93
94    /// Returns the value that is assigned to `self` in the underlying
95    /// `ConstraintSystem`.
96    fn value(&self) -> Result<Self::Value, ark_relations::r1cs::SynthesisError>;
97}
98
99impl<F: Field, T: R1CSVar<F>> R1CSVar<F> for [T] {
100    type Value = Vec<T::Value>;
101
102    fn cs(&self) -> ark_relations::r1cs::ConstraintSystemRef<F> {
103        let mut result = ark_relations::r1cs::ConstraintSystemRef::None;
104        for var in self {
105            result = var.cs().or(result);
106        }
107        result
108    }
109
110    fn value(&self) -> Result<Self::Value, ark_relations::r1cs::SynthesisError> {
111        let mut result = Vec::new();
112        for var in self {
113            result.push(var.value()?);
114        }
115        Ok(result)
116    }
117}

119impl<'a, F: Field, T: 'a + R1CSVar<F>> R1CSVar<F> for &'a T {
120    type Value = T::Value;
121
122    fn cs(&self) -> ark_relations::r1cs::ConstraintSystemRef<F> {
123        (*self).cs()
124    }
125
126    fn value(&self) -> Result<Self::Value, ark_relations::r1cs::SynthesisError> {
127        (*self).value()
128    }
129}

131/// A utility trait to convert `Self` to `Result<T, SynthesisErrorA`.>
132pub trait Assignment<T> {
133    /// Converts `self` to `Result`.
134    fn get(self) -> Result<T, ark_relations::r1cs::SynthesisError>;
135}
136
137impl<T> Assignment<T> for Option<T> {
138    fn get(self) -> Result<T, ark_relations::r1cs::SynthesisError> {
139        self.ok_or(ark_relations::r1cs::SynthesisError::AssignmentMissing)
140    }
