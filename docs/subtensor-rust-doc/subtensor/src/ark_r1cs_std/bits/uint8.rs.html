12/// Represents an interpretation of 8 `Boolean` objects as an
13/// unsigned integer.
14#[derive(Clone, Debug)]
15pub struct UInt8<F: Field> {
16    /// Little-endian representation: least significant bit first
17    pub(crate) bits: [Boolean<F>; 8],
18    pub(crate) value: Option<u8>,
19}
20
21impl<F: Field> R1CSVar<F> for UInt8<F> {
22    type Value = u8;
23
24    fn cs(&self) -> ConstraintSystemRef<F> {
25        self.bits.as_ref().cs()
26    }
27
28    fn value(&self) -> Result<Self::Value, SynthesisError> {
29        let mut value = None;
30        for (i, bit) in self.bits.iter().enumerate() {
31            let b = u8::from(bit.value()?);
32            value = match value {
33                Some(value) => Some(value + (b << i)),
34                None => Some(b << i),
35            };
36        }
37        debug_assert_eq!(self.value, value);
38        value.get()
39    }
40}

42impl<F: Field> UInt8<F> {
43    /// Construct a constant vector of `UInt8` from a vector of `u8`
44    ///
45    /// This *does not* create any new variables or constraints.
46    /// ```
47    /// # fn main() -> Result<(), ark_relations::r1cs::SynthesisError> {
48    /// // We'll use the BLS12-381 scalar field for our constraints.
49    /// use ark_test_curves::bls12_381::Fr;
50    /// use ark_relations::r1cs::*;
51    /// use ark_r1cs_std::prelude::*;
52    ///
53    /// let cs = ConstraintSystem::<Fr>::new_ref();
54    /// let var = vec![UInt8::new_witness(cs.clone(), || Ok(2))?];
55    ///
56    /// let constant = UInt8::constant_vec(&[2]);
57    /// var.enforce_equal(&constant)?;
58    /// assert!(cs.is_satisfied().unwrap());
59    /// # Ok(())
60    /// # }
61    /// ```
62    pub fn constant_vec(values: &[u8]) -> Vec<Self> {
63        let mut result = Vec::new();
64        for value in values {
65            result.push(UInt8::constant(*value));
66        }
67        result
68    }

70    /// Construct a constant `UInt8` from a `u8`
71    ///
72    /// This *does not* create new variables or constraints.
73    ///
74    /// ```
75    /// # fn main() -> Result<(), ark_relations::r1cs::SynthesisError> {
76    /// // We'll use the BLS12-381 scalar field for our constraints.
77    /// use ark_test_curves::bls12_381::Fr;
78    /// use ark_relations::r1cs::*;
79    /// use ark_r1cs_std::prelude::*;
80    ///
81    /// let cs = ConstraintSystem::<Fr>::new_ref();
82    /// let var = UInt8::new_witness(cs.clone(), || Ok(2))?;
83    ///
84    /// let constant = UInt8::constant(2);
85    /// var.enforce_equal(&constant)?;
86    /// assert!(cs.is_satisfied().unwrap());
87    /// # Ok(())
88    /// # }
89    /// ```
90    pub fn constant(value: u8) -> Self {
91        let mut bits = [Boolean::FALSE; 8];
92
93        let mut tmp = value;
94        for i in 0..8 {
95            // If last bit is one, push one.
96            bits[i] = Boolean::constant((tmp & 1) == 1);
97            tmp >>= 1;
98        }
99
100        Self {
101            bits,
102            value: Some(value),
103        }
104    }

106    /// Allocates a slice of `u8`'s as private witnesses.
107    pub fn new_witness_vec(
108        cs: impl Into<Namespace<F>>,
109        values: &[impl Into<Option<u8>> + Copy],
110    ) -> Result<Vec<Self>, SynthesisError> {
111        let ns = cs.into();
112        let cs = ns.cs();
113        let mut output_vec = Vec::with_capacity(values.len());
114        for value in values {
115            let byte: Option<u8> = Into::into(*value);
116            output_vec.push(Self::new_witness(cs.clone(), || byte.get())?);
117        }
118        Ok(output_vec)
119    }

121    /// Allocates a slice of `u8`'s as public inputs by first packing them into
122    /// elements of `F`, (thus reducing the number of input allocations),
123    /// allocating these elements as public inputs, and then converting
124    /// these field variables `FpVar<F>` variables back into bytes.
125    ///
126    /// From a user perspective, this trade-off adds constraints, but improves
127    /// verifier time and verification key size.
128    ///
129    /// ```
130    /// # fn main() -> Result<(), ark_relations::r1cs::SynthesisError> {
131    /// // We'll use the BLS12-381 scalar field for our constraints.
132    /// use ark_test_curves::bls12_381::Fr;
133    /// use ark_relations::r1cs::*;
134    /// use ark_r1cs_std::prelude::*;
135    ///
136    /// let cs = ConstraintSystem::<Fr>::new_ref();
137    /// let two = UInt8::new_witness(cs.clone(), || Ok(2))?;
138    /// let var = vec![two.clone(); 32];
139    ///
140    /// let c = UInt8::new_input_vec(cs.clone(), &[2; 32])?;
141    /// var.enforce_equal(&c)?;
142    /// assert!(cs.is_satisfied().unwrap());
143    /// # Ok(())
144    /// # }
145    /// ```
146    pub fn new_input_vec(
147        cs: impl Into<Namespace<F>>,
148        values: &[u8],
149    ) -> Result<Vec<Self>, SynthesisError>
150    where
151        F: PrimeField,
152    {
153        let ns = cs.into();
154        let cs = ns.cs();
155        let values_len = values.len();
156        let field_elements: Vec<F> = ToConstraintField::<F>::to_field_elements(values).unwrap();
157
158        let max_size = 8 * ((F::MODULUS_BIT_SIZE - 1) / 8) as usize;
159        let mut allocated_bits = Vec::new();
160        for field_element in field_elements.into_iter() {
161            let fe = AllocatedFp::new_input(cs.clone(), || Ok(field_element))?;
162            let fe_bits = fe.to_bits_le()?;
163
164            // Remove the most significant bit, because we know it should be zero
165            // because `values.to_field_elements()` only
166            // packs field elements up to the penultimate bit.
167            // That is, the most significant bit (`ConstraintF::NUM_BITS`-th bit) is
168            // unset, so we can just pop it off.
169            allocated_bits.extend_from_slice(&fe_bits[0..max_size]);
170        }
171
172        // Chunk up slices of 8 bit into bytes.
173        Ok(allocated_bits[0..(8 * values_len)]
174            .chunks(8)
175            .map(Self::from_bits_le)
176            .collect())
177    }

179    /// Converts a little-endian byte order representation of bits into a
180    /// `UInt8`.
181    ///
182    /// ```
183    /// # fn main() -> Result<(), ark_relations::r1cs::SynthesisError> {
184    /// // We'll use the BLS12-381 scalar field for our constraints.
185    /// use ark_test_curves::bls12_381::Fr;
186    /// use ark_relations::r1cs::*;
187    /// use ark_r1cs_std::prelude::*;
188    ///
189    /// let cs = ConstraintSystem::<Fr>::new_ref();
190    /// let var = UInt8::new_witness(cs.clone(), || Ok(128))?;
191    ///
192    /// let f = Boolean::FALSE;
193    /// let t = Boolean::TRUE;
194    ///
195    /// // Construct [0, 0, 0, 0, 0, 0, 0, 1]
196    /// let mut bits = vec![f.clone(); 7];
197    /// bits.push(t);
198    ///
199    /// let mut c = UInt8::from_bits_le(&bits);
200    /// var.enforce_equal(&c)?;
201    /// assert!(cs.is_satisfied().unwrap());
202    /// # Ok(())
203    /// # }
204    /// ```
205    #[tracing::instrument(target = "r1cs")]
206    pub fn from_bits_le(bits: &[Boolean<F>]) -> Self {
207        assert_eq!(bits.len(), 8);
208        let bits = <&[Boolean<F>; 8]>::try_from(bits).unwrap().clone();
209
210        let mut value = Some(0u8);
211        for (i, b) in bits.iter().enumerate() {
212            value = match b.value().ok() {
213                Some(b) => value.map(|v| v + (u8::from(b) << i)),
214                None => None,
215            }
216        }
217
218        Self { value, bits }
219    }

221    /// Outputs `self ^ other`.
222    ///
223    /// If at least one of `self` and `other` are constants, then this method
224    /// *does not* create any constraints or variables.
225    ///
226    /// ```
227    /// # fn main() -> Result<(), ark_relations::r1cs::SynthesisError> {
228    /// // We'll use the BLS12-381 scalar field for our constraints.
229    /// use ark_test_curves::bls12_381::Fr;
230    /// use ark_relations::r1cs::*;
231    /// use ark_r1cs_std::prelude::*;
232    ///
233    /// let cs = ConstraintSystem::<Fr>::new_ref();
234    /// let a = UInt8::new_witness(cs.clone(), || Ok(16))?;
235    /// let b = UInt8::new_witness(cs.clone(), || Ok(17))?;
236    /// let c = UInt8::new_witness(cs.clone(), || Ok(1))?;
237    ///
238    /// a.xor(&b)?.enforce_equal(&c)?;
239    /// assert!(cs.is_satisfied().unwrap());
240    /// # Ok(())
241    /// # }
242    /// ```
243    #[tracing::instrument(target = "r1cs")]
244    pub fn xor(&self, other: &Self) -> Result<Self, SynthesisError> {
245        let mut result = self.clone();
246        result.value = match (self.value, other.value) {
247            (Some(a), Some(b)) => Some(a ^ b),
248            _ => None,
249        };
250
251        let new_bits = self.bits.iter().zip(&other.bits).map(|(a, b)| a.xor(b));
252
253        for (res, new) in result.bits.iter_mut().zip(new_bits) {
254            *res = new?;
255        }
256
257        Ok(result)
258    }
259}

287impl<ConstraintF: Field> CondSelectGadget<ConstraintF> for UInt8<ConstraintF> {
288    #[tracing::instrument(target = "r1cs", skip(cond, true_value, false_value))]
289    fn conditionally_select(
290        cond: &Boolean<ConstraintF>,
291        true_value: &Self,
292        false_value: &Self,
293    ) -> Result<Self, SynthesisError> {
294        let selected_bits = true_value
295            .bits
296            .iter()
297            .zip(&false_value.bits)
298            .map(|(t, f)| cond.select(t, f));
299        let mut bits = [Boolean::FALSE; 8];
300        for (result, new) in bits.iter_mut().zip(selected_bits) {
301            *result = new?;
302        }
303
304        let value = cond.value().ok().and_then(|cond| {
305            if cond {
306                true_value.value().ok()
307            } else {
308                false_value.value().ok()
309            }
310        });
311        Ok(Self { bits, value })
312    }
313}

315impl<ConstraintF: Field> AllocVar<u8, ConstraintF> for UInt8<ConstraintF> {
316    fn new_variable<T: Borrow<u8>>(
317        cs: impl Into<Namespace<ConstraintF>>,
318        f: impl FnOnce() -> Result<T, SynthesisError>,
319        mode: AllocationMode,
320    ) -> Result<Self, SynthesisError> {
321        let ns = cs.into();
322        let cs = ns.cs();
323        let value = f().map(|f| *f.borrow()).ok();
324
325        let mut values = [None; 8];
326        if let Some(val) = value {
327            values
328                .iter_mut()
329                .enumerate()
330                .for_each(|(i, v)| *v = Some((val >> i) & 1 == 1));
331        }
332
333        let mut bits = [Boolean::FALSE; 8];
334        for (b, v) in bits.iter_mut().zip(&values) {
335            *b = Boolean::new_variable(cs.clone(), || v.get(), mode)?;
336        }
337        Ok(Self { bits, value })
338    }
339}

341/// Parses the `Vec<UInt8<ConstraintF>>` in fixed-sized
342/// `ConstraintF::MODULUS_BIT_SIZE - 1` chunks and converts each chunk, which is
343/// assumed to be little-endian, to its `FpVar<ConstraintF>` representation.
344/// This is the gadget counterpart to the `[u8]` implementation of
345/// [ToConstraintField](ark_ff::ToConstraintField).
346impl<ConstraintF: PrimeField> ToConstraintFieldGadget<ConstraintF> for [UInt8<ConstraintF>] {
347    #[tracing::instrument(target = "r1cs")]
348    fn to_constraint_field(&self) -> Result<Vec<FpVar<ConstraintF>>, SynthesisError> {
349        let max_size = ((ConstraintF::MODULUS_BIT_SIZE - 1) / 8) as usize;
350        self.chunks(max_size)
351            .map(|chunk| Boolean::le_bits_to_fp_var(chunk.to_bits_le()?.as_slice()))
352            .collect::<Result<Vec<_>, SynthesisError>>()
353    }
354}
