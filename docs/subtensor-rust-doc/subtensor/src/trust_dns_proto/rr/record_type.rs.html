25/// The type of the resource record.
26///
27/// This specifies the type of data in the RData field of the Resource Record
28#[cfg_attr(feature = "serde-config", derive(Deserialize, Serialize))]
29#[derive(Debug, PartialEq, Eq, Hash, Copy, Clone)]
30#[allow(dead_code)]
31#[non_exhaustive]
32pub enum RecordType {
33    /// [RFC 1035](https://tools.ietf.org/html/rfc1035) IPv4 Address record
34    A,
35    /// [RFC 3596](https://tools.ietf.org/html/rfc3596) IPv6 address record
36    AAAA,
37    /// [ANAME draft-ietf-dnsop-aname](https://tools.ietf.org/html/draft-ietf-dnsop-aname-04)
38    ANAME,
39    //  AFSDB,      //	18	RFC 1183	AFS database record
40    /// [RFC 1035](https://tools.ietf.org/html/rfc1035) All cached records, aka ANY
41    ANY,
42    //  APL,        //	42	RFC 3123	Address Prefix List
43    /// [RFC 1035](https://tools.ietf.org/html/rfc1035) Authoritative Zone Transfer
44    AXFR,
45    /// [RFC 6844](https://tools.ietf.org/html/rfc6844) Certification Authority Authorization
46    CAA,
47    /// [RFC 7344](https://tools.ietf.org/html/rfc7344) Child DS
48    CDS,
49    /// [RFC 7344](https://tools.ietf.org/html/rfc7344) Child DNSKEY
50    CDNSKEY,
51    //  CERT,       // 37 RFC 4398 Certificate record
52    /// [RFC 1035](https://tools.ietf.org/html/rfc1035) Canonical name record
53    CNAME,
54    //  DHCID,      // 49 RFC 4701 DHCP identifier
55    //  DLV,        //	32769	RFC 4431	DNSSEC Lookaside Validation record
56    //  DNAME,      // 39 RFC 2672 Delegation Name
57    /// [RFC 7477](https://tools.ietf.org/html/rfc4034) Child-to-parent synchronization record
58    CSYNC,
59    /// [RFC 4034](https://tools.ietf.org/html/rfc4034) DNS Key record: RSASHA256 and RSASHA512, RFC5702
60    DNSKEY,
61    /// [RFC 4034](https://tools.ietf.org/html/rfc4034) Delegation signer: RSASHA256 and RSASHA512, RFC5702
62    DS,
63    /// [RFC 1035](https://tools.ietf.org/html/rfc1035) host information
64    HINFO,
65    //  HIP,        // 55 RFC 5205 Host Identity Protocol
66    /// [RFC draft-ietf-dnsop-svcb-https-03](https://tools.ietf.org/html/draft-ietf-dnsop-svcb-httpssvc-03) DNS SVCB and HTTPS RRs
67    HTTPS,
68    //  IPSECKEY,   // 45 RFC 4025 IPsec Key
69    /// [RFC 1996](https://tools.ietf.org/html/rfc1996) Incremental Zone Transfer
70    IXFR,
71    //  KX,         // 36 RFC 2230 Key eXchanger record
72    /// [RFC 2535](https://tools.ietf.org/html/rfc2535) and [RFC 2930](https://tools.ietf.org/html/rfc2930) Key record
73    KEY,
74    //  LOC,        // 29 RFC 1876 Location record
75    /// [RFC 1035](https://tools.ietf.org/html/rfc1035) Mail exchange record
76    MX,
77    /// [RFC 3403](https://tools.ietf.org/html/rfc3403) Naming Authority Pointer
78    NAPTR,
79    /// [RFC 1035](https://tools.ietf.org/html/rfc1035) Name server record
80    NS,
81    /// [RFC 4034](https://tools.ietf.org/html/rfc4034) Next-Secure record
82    NSEC,
83    /// [RFC 5155](https://tools.ietf.org/html/rfc5155) NSEC record version 3
84    NSEC3,
85    /// [RFC 5155](https://tools.ietf.org/html/rfc5155) NSEC3 parameters
86    NSEC3PARAM,
87    /// [RFC 1035](https://tools.ietf.org/html/rfc1035) Null server record, for testing
88    NULL,
89    /// [RFC 7929](https://tools.ietf.org/html/rfc7929) OpenPGP public key
90    OPENPGPKEY,
91    /// [RFC 6891](https://tools.ietf.org/html/rfc6891) Option
92    OPT,
93    /// [RFC 1035](https://tools.ietf.org/html/rfc1035) Pointer record
94    PTR,
95    //  RP,         // 17 RFC 1183 Responsible person
96    /// [RFC 4034](https://tools.ietf.org/html/rfc4034) DNSSEC signature: RSASHA256 and RSASHA512, RFC5702
97    RRSIG,
98    /// [RFC 2535](https://tools.ietf.org/html/rfc2535) (and [RFC 2931](https://tools.ietf.org/html/rfc2931)) Signature, to support [RFC 2137](https://tools.ietf.org/html/rfc2137) Update.
99    SIG,
100    /// [RFC 1035](https://tools.ietf.org/html/rfc1035) and [RFC 2308](https://tools.ietf.org/html/rfc2308) Start of [a zone of] authority record
101    SOA,
102    /// [RFC 2782](https://tools.ietf.org/html/rfc2782) Service locator
103    SRV,
104    /// [RFC 4255](https://tools.ietf.org/html/rfc4255) SSH Public Key Fingerprint
105    SSHFP,
106    /// [RFC draft-ietf-dnsop-svcb-https-03](https://tools.ietf.org/html/draft-ietf-dnsop-svcb-httpssvc-03) DNS SVCB and HTTPS RRs
107    SVCB,
108    //  TA,         // 32768 N/A DNSSEC Trust Authorities
109    //  TKEY,       // 249 RFC 2930 Secret key record
110    /// [RFC 6698](https://tools.ietf.org/html/rfc6698) TLSA certificate association
111    TLSA,
112    /// [RFC 8945](https://tools.ietf.org/html/rfc8945) Transaction Signature
113    TSIG,
114    /// [RFC 1035](https://tools.ietf.org/html/rfc1035) Text record
115    TXT,
116    /// Unknown Record type, or unsupported
117    Unknown(u16),
118
119    /// This corresponds to a record type of 0, unspecified
120    ZERO,
121}

123impl RecordType {
124    /// Returns true if this is an ANY
125    #[inline]
126    pub fn is_any(self) -> bool {
127        self == Self::ANY
128    }
129
130    /// Returns true if this is a CNAME
131    #[inline]
132    pub fn is_cname(self) -> bool {
133        self == Self::CNAME
134    }
135
136    /// Returns true if this is an NS
137    #[inline]
138    pub fn is_ns(self) -> bool {
139        self == Self::NS
140    }
141
142    /// Returns true if this is an SOA
143    #[inline]
144    pub fn is_soa(self) -> bool {
145        self == Self::SOA
146    }
147
148    /// Returns true if this is an SRV
149    #[inline]
150    pub fn is_srv(self) -> bool {
151        self == Self::SRV
152    }
153
154    /// Returns true if this is an A or an AAAA record
155    #[inline]
156    pub fn is_ip_addr(self) -> bool {
157        matches!(self, Self::A | Self::AAAA)
158    }
159
160    /// Returns true if this is a DNSSEC RecordType
161    #[inline]
162    pub fn is_dnssec(self) -> bool {
163        matches!(
164            self,
165            Self::DNSKEY
166                | Self::CDNSKEY
167                | Self::CDS
168                | Self::DS
169                | Self::KEY
170                | Self::NSEC
171                | Self::NSEC3
172                | Self::NSEC3PARAM
173                | Self::RRSIG
174                | Self::SIG
175                | Self::TSIG
176        )
177    }
178
179    /// Returns true if this is a Zero (unspecified) RecordType
180    #[inline]
181    pub fn is_zero(self) -> bool {
182        self == Self::ZERO
183    }
184}

214            "HTTPS" => Ok(Self::HTTPS),

300impl<'r> BinDecodable<'r> for RecordType {
301    fn read(decoder: &mut BinDecoder<'_>) -> ProtoResult<Self> {

303            .read_u16()

421/// [Canonical DNS Name Order](https://tools.ietf.org/html/rfc4034#section-6)

492    /// Check that all record type names parse into unique `RecordType` instances,
493    /// and can be converted back into the same name.
494    #[test]
495    fn test_record_type_parse() {
496        let record_names = &[
497            "A",
498            "AAAA",
499            "ANAME",
500            "CAA",
501            "CNAME",
502            "CSYNC",
503            "HINFO",
504            "NULL",
505            "MX",
506            "NAPTR",
507            "NS",
508            "OPENPGPKEY",
509            "PTR",
510            "SOA",
511            "SRV",
512            "SSHFP",
513            "TLSA",
514            "TXT",
515            "ANY",
516            "AXFR",
517        ];
518
519        #[cfg(feature = "dnssec")]
520        let dnssec_record_names = &[
521            "CDNSKEY",
522            "CDS",
523            "DNSKEY",
524            "DS",
525            "KEY",
526            "NSEC",
527            "NSEC3",
528            "NSEC3PARAM",
529            "RRSIG",
530            "SIG",
531            "TSIG",
532        ];
533        #[cfg(not(feature = "dnssec"))]
534        let dnssec_record_names = &[];
535
536        let mut rtypes = std::collections::HashSet::new();
537        for name in record_names.iter().chain(dnssec_record_names) {
538            let rtype: RecordType = name.parse().unwrap();
539            assert_eq!(rtype.to_string().to_ascii_uppercase().as_str(), *name);
540            assert!(rtypes.insert(rtype));
541        }
542    }
