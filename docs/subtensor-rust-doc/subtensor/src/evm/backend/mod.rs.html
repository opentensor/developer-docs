10/// Basic account information.
11#[derive(Clone, Eq, PartialEq, Debug, Default)]
12#[cfg_attr(
13	feature = "with-codec",
14	derive(scale_codec::Encode, scale_codec::Decode, scale_info::TypeInfo)
15)]
16#[cfg_attr(feature = "with-serde", derive(serde::Serialize, serde::Deserialize))]
17pub struct Basic {
18	/// Account balance.
19	pub balance: U256,
20	/// Account nonce.
21	pub nonce: U256,
22}

24pub use ethereum::Log;
25
26/// Apply state operation.
27#[derive(Clone, Debug)]
28pub enum Apply<I> {
29	/// Modify or create at address.
30	Modify {
31		/// Address.
32		address: H160,
33		/// Basic information of the address.
34		basic: Basic,
35		/// Code. `None` means leaving it unchanged.
36		code: Option<Vec<u8>>,
37		/// Storage iterator.
38		storage: I,
39		/// Whether storage should be wiped empty before applying the storage
40		/// iterator.
41		reset_storage: bool,
42	},
43	/// Delete address.
44	Delete {
45		/// Address.
46		address: H160,
47	},
48}

50/// EVM backend.
51#[auto_impl::auto_impl(&, Arc, Box)]
52pub trait Backend {
53	/// Gas price. Unused for London.
54	fn gas_price(&self) -> U256;
55	/// Origin.
56	fn origin(&self) -> H160;
57	/// Environmental block hash.
58	fn block_hash(&self, number: U256) -> H256;
59	/// Environmental block number.
60	fn block_number(&self) -> U256;
61	/// Environmental coinbase.
62	fn block_coinbase(&self) -> H160;
63	/// Environmental block timestamp.
64	fn block_timestamp(&self) -> U256;
65	/// Environmental block difficulty.
66	fn block_difficulty(&self) -> U256;
67	/// Get environmental block randomness.
68	fn block_randomness(&self) -> Option<H256>;
69	/// Environmental block gas limit.
70	fn block_gas_limit(&self) -> U256;
71	/// Environmental block base fee.
72	fn block_base_fee_per_gas(&self) -> U256;
73	/// Environmental chain ID.
74	fn chain_id(&self) -> U256;
75
76	/// Whether account at address exists.
77	fn exists(&self, address: H160) -> bool;
78	/// Get basic account information.
79	fn basic(&self, address: H160) -> Basic;
80	/// Get account code.
81	fn code(&self, address: H160) -> Vec<u8>;
82	/// Get storage value of address at index.
83	fn storage(&self, address: H160, index: H256) -> H256;
84	/// Get original storage value of address at index, if available.
85	fn original_storage(&self, address: H160, index: H256) -> Option<H256>;
86}

88/// EVM backend that can apply changes.
89pub trait ApplyBackend {
90	/// Apply given values and logs at backend.
91	fn apply<A, I, L>(&mut self, values: A, logs: L, delete_empty: bool)
92	where
93		A: IntoIterator<Item = Apply<I>>,
94		I: IntoIterator<Item = (H256, H256)>,
95		L: IntoIterator<Item = Log>;
96}
