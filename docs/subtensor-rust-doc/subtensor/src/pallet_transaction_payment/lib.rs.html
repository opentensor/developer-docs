18//! # Transaction Payment Pallet
19//!
20//! This pallet provides the basic logic needed to pay the absolute minimum amount needed for a
21//! transaction to be included. This includes:
22//!   - _base fee_: This is the minimum amount a user pays for a transaction. It is declared
23//! 	as a base _weight_ in the runtime and converted to a fee using `WeightToFee`.
24//!   - _weight fee_: A fee proportional to amount of weight a transaction consumes.
25//!   - _length fee_: A fee proportional to the encoded length of the transaction.
26//!   - _tip_: An optional tip. Tip increases the priority of the transaction, giving it a higher
27//!     chance to be included by the transaction queue.
28//!
29//! The base fee and adjusted weight and length fees constitute the _inclusion fee_, which is
30//! the minimum fee for a transaction to be included in a block.
31//!
32//! The formula of final fee:
33//!   ```ignore
34//!   inclusion_fee = base_fee + length_fee + [targeted_fee_adjustment * weight_fee];
35//!   final_fee = inclusion_fee + tip;
36//!   ```
37//!
38//!   - `targeted_fee_adjustment`: This is a multiplier that can tune the final fee based on
39//! 	the congestion of the network.
40//!
41//! Additionally, this pallet allows one to configure:
42//!   - The mapping between one unit of weight to one unit of fee via [`Config::WeightToFee`].
43//!   - A means of updating the fee for the next block, via defining a multiplier, based on the
44//!     final state of the chain at the end of the previous block. This can be configured via
45//!     [`Config::FeeMultiplierUpdate`]
46//!   - How the fees are paid via [`Config::OnChargeTransaction`].
47

87/// A struct to update the weight multiplier per block. It implements `Convert<Multiplier,
88/// Multiplier>`, meaning that it can convert the previous multiplier to the next one. This should
89/// be called on `on_finalize` of a block, prior to potentially cleaning the weight data from the
90/// system pallet.
91///
92/// given:
93/// 	s = previous block weight
94/// 	s'= ideal block weight
95/// 	m = maximum block weight
96/// 		diff = (s - s')/m
97/// 		v = 0.00001
98/// 		t1 = (v * diff)
99/// 		t2 = (v * diff)^2 / 2
100/// 	then:
101/// 	next_multiplier = prev_multiplier * (1 + t1 + t2)
102///
103/// Where `(s', v)` must be given as the `Get` implementation of the `T` generic type. Moreover, `M`
104/// must provide the minimum allowed value for the multiplier. Note that a runtime should ensure
105/// with tests that the combination of this `M` and `V` is not such that the multiplier can drop to
106/// zero and never recover.
107///
108/// Note that `s'` is interpreted as a portion in the _normal transaction_ capacity of the block.
109/// For example, given `s' == 0.25` and `AvailableBlockRatio = 0.75`, then the target fullness is
110/// _0.25 of the normal capacity_ and _0.1875 of the entire block_.
111///
112/// Since block weight is multi-dimension, we use the scarcer resource, referred as limiting
113/// dimension, for calculation of fees. We determine the limiting dimension by comparing the
114/// dimensions using the ratio of `dimension_value / max_dimension_value` and selecting the largest
115/// ratio. For instance, if a block is 30% full based on `ref_time` and 25% full based on
116/// `proof_size`, we identify `ref_time` as the limiting dimension, indicating that the block is 30%
117/// full.
118///
119/// This implementation implies the bound:
120/// - `v ≤ p / k * (s − s')`
121/// - or, solving for `p`: `p >= v * k * (s - s')`
122///
123/// where `p` is the amount of change over `k` blocks.
124///
125/// Hence:
126/// - in a fully congested chain: `p >= v * k * (1 - s')`.
127/// - in an empty chain: `p >= v * k * (-s')`.
128///
129/// For example, when all blocks are full and there are 28800 blocks per day (default in
130/// `substrate-node`) and v == 0.00001, s' == 0.1875, we'd have:
131///
132/// p >= 0.00001 * 28800 * 0.8125
133/// p >= 0.234
134///
135/// Meaning that fees can change by around ~23% per day, given extreme congestion.
136///
137/// More info can be found at:
138/// <https://research.web3.foundation/en/latest/polkadot/overview/2-token-economics.html>
139pub struct TargetedFeeAdjustment<T, S, V, M, X>(core::marker::PhantomData<(T, S, V, M, X)>);
140
141/// Something that can convert the current multiplier to the next one.
142pub trait MultiplierUpdate: Convert<Multiplier, Multiplier> {
143	/// Minimum multiplier. Any outcome of the `convert` function should be at least this.
144	fn min() -> Multiplier;
145	/// Maximum multiplier. Any outcome of the `convert` function should be less or equal this.
146	fn max() -> Multiplier;
147	/// Target block saturation level
148	fn target() -> Perquintill;
149	/// Variability factor
150	fn variability() -> Multiplier;
151}
152
153impl MultiplierUpdate for () {
154	fn min() -> Multiplier {
155		Default::default()
156	}
157	fn max() -> Multiplier {
158		<Multiplier as sp_runtime::traits::Bounded>::max_value()
159	}
160	fn target() -> Perquintill {
161		Default::default()
162	}
163	fn variability() -> Multiplier {
164		Default::default()
165	}
166}

190impl<T, S, V, M, X> Convert<Multiplier, Multiplier> for TargetedFeeAdjustment<T, S, V, M, X>
191where
192	T: frame_system::Config,
193	S: Get<Perquintill>,
194	V: Get<Multiplier>,
195	M: Get<Multiplier>,
196	X: Get<Multiplier>,
197{
198	fn convert(previous: Multiplier) -> Multiplier {
199		// Defensive only. The multiplier in storage should always be at most positive. Nonetheless
200		// we recover here in case of errors, because any value below this would be stale and can
201		// never change.
202		let min_multiplier = M::get();
203		let max_multiplier = X::get();
204		let previous = previous.max(min_multiplier);
205
206		let weights = T::BlockWeights::get();
207		// the computed ratio is only among the normal class.
208		let normal_max_weight =
209			weights.get(DispatchClass::Normal).max_total.unwrap_or(weights.max_block);
210		let current_block_weight = frame_system::Pallet::<T>::block_weight();
211		let normal_block_weight =
212			current_block_weight.get(DispatchClass::Normal).min(normal_max_weight);
213
214		// Normalize dimensions so they can be compared. Ensure (defensive) max weight is non-zero.
215		let normalized_ref_time = Perbill::from_rational(
216			normal_block_weight.ref_time(),
217			normal_max_weight.ref_time().max(1),
218		);
219		let normalized_proof_size = Perbill::from_rational(
220			normal_block_weight.proof_size(),
221			normal_max_weight.proof_size().max(1),
222		);
223
224		// Pick the limiting dimension. If the proof size is the limiting dimension, then the
225		// multiplier is adjusted by the proof size. Otherwise, it is adjusted by the ref time.
226		let (normal_limiting_dimension, max_limiting_dimension) =
227			if normalized_ref_time < normalized_proof_size {
228				(normal_block_weight.proof_size(), normal_max_weight.proof_size())
229			} else {
230				(normal_block_weight.ref_time(), normal_max_weight.ref_time())
231			};
232
233		let target_block_fullness = S::get();
234		let adjustment_variable = V::get();
235
236		let target_weight = (target_block_fullness * max_limiting_dimension) as u128;
237		let block_weight = normal_limiting_dimension as u128;
238
239		// determines if the first_term is positive
240		let positive = block_weight >= target_weight;
241		let diff_abs = block_weight.max(target_weight) - block_weight.min(target_weight);
242
243		// defensive only, a test case assures that the maximum weight diff can fit in Multiplier
244		// without any saturation.
245		let diff = Multiplier::saturating_from_rational(diff_abs, max_limiting_dimension.max(1));
246		let diff_squared = diff.saturating_mul(diff);
247
248		let v_squared_2 = adjustment_variable.saturating_mul(adjustment_variable) /
249			Multiplier::saturating_from_integer(2);
250
251		let first_term = adjustment_variable.saturating_mul(diff);
252		let second_term = v_squared_2.saturating_mul(diff_squared);
253
254		if positive {
255			let excess = first_term.saturating_add(second_term).saturating_mul(previous);
256			previous.saturating_add(excess).clamp(min_multiplier, max_multiplier)
257		} else {
258			// Defensive-only: first_term > second_term. Safe subtraction.
259			let negative = first_term.saturating_sub(second_term).saturating_mul(previous);
260			previous.saturating_sub(negative).clamp(min_multiplier, max_multiplier)
261		}
262	}
263}

340	#[pallet::config(with_default)]
341	pub trait Config: frame_system::Config {
342		/// The overarching event type.
343		#[pallet::no_default_bounds]
344		type RuntimeEvent: From<Event<Self>> + IsType<<Self as frame_system::Config>::RuntimeEvent>;

346		/// Handler for withdrawing, refunding and depositing the transaction fee.
347		/// Transaction fees are withdrawn before the transaction is executed.
348		/// After the transaction was executed the transaction weight can be
349		/// adjusted, depending on the used resources by the transaction. If the
350		/// transaction weight is lower than expected, parts of the transaction fee
351		/// might be refunded. In the end the fees can be deposited.
352		#[pallet::no_default]
353		type OnChargeTransaction: OnChargeTransaction<Self>;
354
355		/// Convert a weight value into a deductible fee based on the currency type.
356		#[pallet::no_default]
357		type WeightToFee: WeightToFee<Balance = BalanceOf<Self>>;
358
359		/// Convert a length value into a deductible fee based on the currency type.
360		#[pallet::no_default]
361		type LengthToFee: WeightToFee<Balance = BalanceOf<Self>>;
362
363		/// Update the multiplier of the next block, based on the previous block's weight.
364		type FeeMultiplierUpdate: MultiplierUpdate;
365
366		/// A fee multiplier for `Operational` extrinsics to compute "virtual tip" to boost their
367		/// `priority`
368		///
369		/// This value is multiplied by the `final_fee` to obtain a "virtual tip" that is later
370		/// added to a tip component in regular `priority` calculations.
371		/// It means that a `Normal` transaction can front-run a similarly-sized `Operational`
372		/// extrinsic (with no tip), by including a tip value greater than the virtual tip.
373		///
374		/// ```rust,ignore
375		/// // For `Normal`
376		/// let priority = priority_calc(tip);
377		///
378		/// // For `Operational`
379		/// let virtual_tip = (inclusion_fee + tip) * OperationalFeeMultiplier;
380		/// let priority = priority_calc(tip + virtual_tip);
381		/// ```
382		///
383		/// Note that since we use `final_fee` the multiplier applies also to the regular `tip`
384		/// sent with the transaction. So, not only does the transaction get a priority bump based
385		/// on the `inclusion_fee`, but we also amplify the impact of tips applied to `Operational`
386		/// transactions.
387		#[pallet::constant]
388		type OperationalFeeMultiplier: Get<u8>;
389	}

442			// given weight == u64, we build multipliers from `diff` of two weight values, which can
443			// at most be maximum block weight. Make sure that this can fit in a multiplier without
444			// loss.
445			assert!(
446				<Multiplier as sp_runtime::traits::Bounded>::max_value() >=
447					Multiplier::checked_from_integer::<u128>(
448						T::BlockWeights::get().max_block.ref_time().try_into().unwrap()
449					)

451			);

454				T::BlockWeights::get().get(DispatchClass::Normal).max_total.expect(
455					"Setting `max_total` for `Normal` dispatch class is not compatible with \
456					`transaction-payment` pallet.",
457				);
458			// add 1 percent;
459			let addition = target / 100;
460			if addition == Weight::zero() {
461				// this is most likely because in a test setup we set everything to ()
462				// or to `ConstFeeMultiplier`.
463				return
464			}
465
466			// This is the minimum value of the multiplier. Make sure that if we collapse to this
467			// value, we can recover with a reasonable amount of traffic. For this test we assert
468			// that if we collapse to minimum, the trend will be positive with a weight value which
469			// is 1% more than the target.
470			let min_value = T::FeeMultiplierUpdate::min();
471			let target = target + addition;
472
473			frame_system::Pallet::<T>::set_block_consumed_resources(target, 0);
474			let next = T::FeeMultiplierUpdate::convert(min_value);
475			assert!(
476				next > min_value,
477				"The minimum bound of the multiplier is too low. When \
478				block saturation is more than target by 1% and multiplier is minimal then \
479				the multiplier doesn't increase."
480			);
481		}
482	}

491	/// Query the data that we know about the fee of a given `call`.
492	///
493	/// This pallet is not and cannot be aware of the internals of a signed extension, for example
494	/// a tip. It only interprets the extrinsic as some encoded value and accounts for its weight
495	/// and length, the runtime's extrinsic base weight, and the current fee multiplier.
496	///
497	/// All dispatchables must be annotated with weight and will have some fee info. This function
498	/// always returns.
499	pub fn query_info<Extrinsic: sp_runtime::traits::Extrinsic + GetDispatchInfo>(
500		unchecked_extrinsic: Extrinsic,
501		len: u32,
502	) -> RuntimeDispatchInfo<BalanceOf<T>>
503	where
504		T::RuntimeCall: Dispatchable<Info = DispatchInfo>,
505	{
506		// NOTE: we can actually make it understand `ChargeTransactionPayment`, but would be some
507		// hassle for sure. We have to make it aware of the index of `ChargeTransactionPayment` in
508		// `Extra`. Alternatively, we could actually execute the tx's per-dispatch and record the
509		// balance of the sender before and after the pipeline.. but this is way too much hassle for
510		// a very very little potential gain in the future.
511		let dispatch_info = <Extrinsic as GetDispatchInfo>::get_dispatch_info(&unchecked_extrinsic);
512
513		let partial_fee = if unchecked_extrinsic.is_signed().unwrap_or(false) {
514			Self::compute_fee(len, &dispatch_info, 0u32.into())
515		} else {
516			// Unsigned extrinsics have no partial fee.
517			0u32.into()
518		};
519
520		let DispatchInfo { weight, class, .. } = dispatch_info;
521
522		RuntimeDispatchInfo { weight, class, partial_fee }
523	}

601		info: &DispatchInfoOf<T::RuntimeCall>,
602		post_info: &PostDispatchInfoOf<T::RuntimeCall>,
603		tip: BalanceOf<T>,
604	) -> BalanceOf<T>
605	where
606		T::RuntimeCall: Dispatchable<Info = DispatchInfo, PostInfo = PostDispatchInfo>,
607	{
608		Self::compute_actual_fee_details(len, info, post_info, tip).final_fee()
609	}
610
611	/// Compute the actual post dispatch fee details for a particular transaction.
612	pub fn compute_actual_fee_details(
613		len: u32,
614		info: &DispatchInfoOf<T::RuntimeCall>,
615		post_info: &PostDispatchInfoOf<T::RuntimeCall>,
616		tip: BalanceOf<T>,
617	) -> FeeDetails<BalanceOf<T>>
618	where
619		T::RuntimeCall: Dispatchable<Info = DispatchInfo, PostInfo = PostDispatchInfo>,
620	{
621		Self::compute_fee_raw(
622			len,
623			post_info.calc_actual_weight(info),
624			tip,
625			post_info.pays_fee(info),
626			info.class,
627		)
628	}
629
630	fn compute_fee_raw(
631		len: u32,
632		weight: Weight,
633		tip: BalanceOf<T>,
634		pays_fee: Pays,
635		class: DispatchClass,
636	) -> FeeDetails<BalanceOf<T>> {
637		if pays_fee == Pays::Yes {
638			// the adjustable part of the fee.
639			let unadjusted_weight_fee = Self::weight_to_fee(weight);
640			let multiplier = NextFeeMultiplier::<T>::get();
641			// final adjusted weight fee.
642			let adjusted_weight_fee = multiplier.saturating_mul_int(unadjusted_weight_fee);
643
644			// length fee. this is adjusted via `LengthToFee`.
645			let len_fee = Self::length_to_fee(len);
646
647			let base_fee = Self::weight_to_fee(T::BlockWeights::get().get(class).base_extrinsic);
648			FeeDetails {
649				inclusion_fee: Some(InclusionFee { base_fee, len_fee, adjusted_weight_fee }),
650				tip,
651			}
652		} else {
653			FeeDetails { inclusion_fee: None, tip }
654		}
655	}
656
657	/// Compute the length portion of a fee by invoking the configured `LengthToFee` impl.
658	pub fn length_to_fee(length: u32) -> BalanceOf<T> {
659		T::LengthToFee::weight_to_fee(&Weight::from_parts(length as u64, 0))
660	}
661
662	/// Compute the unadjusted portion of the weight fee by invoking the configured `WeightToFee`
663	/// impl. Note that the input `weight` is capped by the maximum block weight before computation.
664	pub fn weight_to_fee(weight: Weight) -> BalanceOf<T> {
665		// cap the weight to the maximum defined in runtime, otherwise it will be the
666		// `Bounded` maximum of its data type, which is not desired.
667		let capped_weight = weight.min(T::BlockWeights::get().max_block);
668		T::WeightToFee::weight_to_fee(&capped_weight)
669	}
670
671	/// Deposit the [`Event::TransactionFeePaid`] event.
672	pub fn deposit_fee_paid_event(who: T::AccountId, actual_fee: BalanceOf<T>, tip: BalanceOf<T>) {
673		Self::deposit_event(Event::TransactionFeePaid { who, actual_fee, tip });
674	}

691/// Require the transactor pay for themselves and maybe include a tip to gain additional priority
692/// in the queue.
693///
694/// # Transaction Validity
695///
696/// This extension sets the `priority` field of `TransactionValidity` depending on the amount
697/// of tip being paid per weight unit.
698///
699/// Operational transactions will receive an additional priority bump, so that they are normally
700/// considered before regular transactions.
701#[derive(Encode, Decode, Clone, Eq, PartialEq, TypeInfo)]
702#[scale_info(skip_type_params(T))]
703pub struct ChargeTransactionPayment<T: Config>(#[codec(compact)] BalanceOf<T>);
704
705impl<T: Config> ChargeTransactionPayment<T>
706where
707	T::RuntimeCall: Dispatchable<Info = DispatchInfo, PostInfo = PostDispatchInfo>,
708	BalanceOf<T>: Send + Sync,
709{
710	/// utility constructor. Used only in client/factory code.
711	pub fn from(fee: BalanceOf<T>) -> Self {
712		Self(fee)
713	}
714
715	/// Returns the tip as being chosen by the transaction sender.
716	pub fn tip(&self) -> BalanceOf<T> {
717		self.0
718	}
719
720	fn withdraw_fee(
721		&self,
722		who: &T::AccountId,
723		call: &T::RuntimeCall,
724		info: &DispatchInfoOf<T::RuntimeCall>,
725		len: usize,
726	) -> Result<
727		(
728			BalanceOf<T>,
729			<<T as Config>::OnChargeTransaction as OnChargeTransaction<T>>::LiquidityInfo,
730		),
731		TransactionValidityError,
732	> {
733		let tip = self.0;
734		let fee = Pallet::<T>::compute_fee(len as u32, info, tip);
735
736		<<T as Config>::OnChargeTransaction as OnChargeTransaction<T>>::withdraw_fee(
737			who, call, info, fee, tip,
738		)
739		.map(|i| (fee, i))
740	}
741
742	/// Get an appropriate priority for a transaction with the given `DispatchInfo`, encoded length
743	/// and user-included tip.
744	///
745	/// The priority is based on the amount of `tip` the user is willing to pay per unit of either
746	/// `weight` or `length`, depending which one is more limiting. For `Operational` extrinsics
747	/// we add a "virtual tip" to the calculations.
748	///
749	/// The formula should simply be `tip / bounded_{weight|length}`, but since we are using
750	/// integer division, we have no guarantees it's going to give results in any reasonable
751	/// range (might simply end up being zero). Hence we use a scaling factor:
752	/// `tip * (max_block_{weight|length} / bounded_{weight|length})`, since given current
753	/// state of-the-art blockchains, number of per-block transactions is expected to be in a
754	/// range reasonable enough to not saturate the `Balance` type while multiplying by the tip.
755	pub fn get_priority(
756		info: &DispatchInfoOf<T::RuntimeCall>,
757		len: usize,
758		tip: BalanceOf<T>,
759		final_fee: BalanceOf<T>,
760	) -> TransactionPriority {
761		// Calculate how many such extrinsics we could fit into an empty block and take the
762		// limiting factor.
763		let max_block_weight = T::BlockWeights::get().max_block;
764		let max_block_length = *T::BlockLength::get().max.get(info.class) as u64;
765
766		// bounded_weight is used as a divisor later so we keep it non-zero.
767		let bounded_weight = info.weight.max(Weight::from_parts(1, 1)).min(max_block_weight);
768		let bounded_length = (len as u64).clamp(1, max_block_length);
769
770		// returns the scarce resource, i.e. the one that is limiting the number of transactions.
771		let max_tx_per_block_weight = max_block_weight
772			.checked_div_per_component(&bounded_weight)
773			.defensive_proof("bounded_weight is non-zero; qed")
774			.unwrap_or(1);
775		let max_tx_per_block_length = max_block_length / bounded_length;
776		// Given our current knowledge this value is going to be in a reasonable range - i.e.
777		// less than 10^9 (2^30), so multiplying by the `tip` value is unlikely to overflow the
778		// balance type. We still use saturating ops obviously, but the point is to end up with some
779		// `priority` distribution instead of having all transactions saturate the priority.
780		let max_tx_per_block = max_tx_per_block_length
781			.min(max_tx_per_block_weight)
782			.saturated_into::<BalanceOf<T>>();
783		let max_reward = |val: BalanceOf<T>| val.saturating_mul(max_tx_per_block);
784
785		// To distribute no-tip transactions a little bit, we increase the tip value by one.
786		// This means that given two transactions without a tip, smaller one will be preferred.
787		let tip = tip.saturating_add(One::one());
788		let scaled_tip = max_reward(tip);
789
790		match info.class {
791			DispatchClass::Normal => {
792				// For normal class we simply take the `tip_per_weight`.
793				scaled_tip
794			},
795			DispatchClass::Mandatory => {
796				// Mandatory extrinsics should be prohibited (e.g. by the [`CheckWeight`]
797				// extensions), but just to be safe let's return the same priority as `Normal` here.
798				scaled_tip
799			},
800			DispatchClass::Operational => {
801				// A "virtual tip" value added to an `Operational` extrinsic.
802				// This value should be kept high enough to allow `Operational` extrinsics
803				// to get in even during congestion period, but at the same time low
804				// enough to prevent a possible spam attack by sending invalid operational
805				// extrinsics which push away regular transactions from the pool.
806				let fee_multiplier = T::OperationalFeeMultiplier::get().saturated_into();
807				let virtual_tip = final_fee.saturating_mul(fee_multiplier);
808				let scaled_virtual_tip = max_reward(virtual_tip);
809
810				scaled_tip.saturating_add(scaled_virtual_tip)
811			},
812		}
813		.saturated_into::<TransactionPriority>()
814	}
815}

828impl<T: Config> SignedExtension for ChargeTransactionPayment<T>
829where
830	BalanceOf<T>: Send + Sync + From<u64>,
831	T::RuntimeCall: Dispatchable<Info = DispatchInfo, PostInfo = PostDispatchInfo>,
832{
833	const IDENTIFIER: &'static str = "ChargeTransactionPayment";
834	type AccountId = T::AccountId;
835	type Call = T::RuntimeCall;
836	type AdditionalSigned = ();
837	type Pre = (
838		// tip
839		BalanceOf<T>,
840		// who paid the fee - this is an option to allow for a Default impl.
841		Self::AccountId,
842		// imbalance resulting from withdrawing the fee
843		<<T as Config>::OnChargeTransaction as OnChargeTransaction<T>>::LiquidityInfo,
844	);
845	fn additional_signed(&self) -> core::result::Result<(), TransactionValidityError> {
846		Ok(())
847	}
848
849	fn validate(
850		&self,
851		who: &Self::AccountId,
852		call: &Self::Call,
853		info: &DispatchInfoOf<Self::Call>,
854		len: usize,
855	) -> TransactionValidity {
856		let (final_fee, _) = self.withdraw_fee(who, call, info, len)?;
857		let tip = self.0;
858		Ok(ValidTransaction {
859			priority: Self::get_priority(info, len, tip, final_fee),
860			..Default::default()
861		})
862	}
863
864	fn pre_dispatch(
865		self,
866		who: &Self::AccountId,
867		call: &Self::Call,
868		info: &DispatchInfoOf<Self::Call>,
869		len: usize,
870	) -> Result<Self::Pre, TransactionValidityError> {
871		let (_fee, imbalance) = self.withdraw_fee(who, call, info, len)?;
872		Ok((self.0, who.clone(), imbalance))
873	}
874
875	fn post_dispatch(
876		maybe_pre: Option<Self::Pre>,
877		info: &DispatchInfoOf<Self::Call>,
878		post_info: &PostDispatchInfoOf<Self::Call>,
879		len: usize,
880		_result: &DispatchResult,
881	) -> Result<(), TransactionValidityError> {
882		if let Some((tip, who, imbalance)) = maybe_pre {
883			let actual_fee = Pallet::<T>::compute_actual_fee(len as u32, info, post_info, tip);
884			T::OnChargeTransaction::correct_and_deposit_fee(
885				&who, info, post_info, actual_fee, tip, imbalance,
886			)?;
887			Pallet::<T>::deposit_event(Event::<T>::TransactionFeePaid { who, actual_fee, tip });
888		}
889		Ok(())
890	}
891}
