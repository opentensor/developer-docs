5/// Transfer from source to target, with given value.
6#[derive(Clone, Debug)]
7pub struct Transfer {
8	/// Source address.
9	pub source: H160,
10	/// Target address.
11	pub target: H160,
12	/// Transfer value.
13	pub value: U256,
14}

16/// EVM context handler.
17#[auto_impl::auto_impl(&mut, Box)]
18pub trait Handler {
19	/// Type of `CREATE` interrupt.
20	type CreateInterrupt;
21	/// Feedback value for `CREATE` interrupt.
22	type CreateFeedback;
23	/// Type of `CALL` interrupt.
24	type CallInterrupt;
25	/// Feedback value of `CALL` interrupt.
26	type CallFeedback;
27
28	/// Get balance of address.
29	fn balance(&self, address: H160) -> U256;
30	/// Get code size of address.
31	fn code_size(&self, address: H160) -> U256;
32	/// Get code hash of address.
33	fn code_hash(&self, address: H160) -> H256;
34	/// Get code of address.
35	fn code(&self, address: H160) -> Vec<u8>;
36	/// Get storage value of address at index.
37	fn storage(&self, address: H160, index: H256) -> H256;
38	/// Get original storage value of address at index.
39	fn original_storage(&self, address: H160, index: H256) -> H256;
40
41	/// Get the gas left value.
42	fn gas_left(&self) -> U256;
43	/// Get the gas price value.
44	fn gas_price(&self) -> U256;
45	/// Get execution origin.
46	fn origin(&self) -> H160;
47	/// Get environmental block hash.
48	fn block_hash(&self, number: U256) -> H256;
49	/// Get environmental block number.
50	fn block_number(&self) -> U256;
51	/// Get environmental coinbase.
52	fn block_coinbase(&self) -> H160;
53	/// Get environmental block timestamp.
54	fn block_timestamp(&self) -> U256;
55	/// Get environmental block difficulty.
56	fn block_difficulty(&self) -> U256;
57	/// Get environmental block randomness.
58	fn block_randomness(&self) -> Option<H256>;
59	/// Get environmental gas limit.
60	fn block_gas_limit(&self) -> U256;
61	/// Environmental block base fee.
62	fn block_base_fee_per_gas(&self) -> U256;
63	/// Get environmental chain ID.
64	fn chain_id(&self) -> U256;
65
66	/// Check whether an address exists.
67	fn exists(&self, address: H160) -> bool;
68	/// Check whether an address has already been deleted.
69	fn deleted(&self, address: H160) -> bool;
70	/// Checks if the address or (address, index) pair has been previously accessed
71	/// (or set in `accessed_addresses` / `accessed_storage_keys` via an access list
72	/// transaction).
73	/// References:
74	/// * <https://eips.ethereum.org/EIPS/eip-2929>
75	/// * <https://eips.ethereum.org/EIPS/eip-2930>
76	fn is_cold(&mut self, address: H160, index: Option<H256>) -> Result<bool, ExitError>;
77
78	/// Set storage value of address at index.
79	fn set_storage(&mut self, address: H160, index: H256, value: H256) -> Result<(), ExitError>;
80	/// Create a log owned by address with given topics and data.
81	fn log(&mut self, address: H160, topics: Vec<H256>, data: Vec<u8>) -> Result<(), ExitError>;
82	/// Mark an address to be deleted, with funds transferred to target.
83	fn mark_delete(&mut self, address: H160, target: H160) -> Result<(), ExitError>;
84	/// Invoke a create operation.
85	fn create(
86		&mut self,
87		caller: H160,
88		scheme: CreateScheme,
89		value: U256,
90		init_code: Vec<u8>,
91		target_gas: Option<u64>,
92	) -> Capture<(ExitReason, Option<H160>, Vec<u8>), Self::CreateInterrupt>;
93	/// Feed in create feedback.
94	fn create_feedback(&mut self, _feedback: Self::CreateFeedback) -> Result<(), ExitError> {
95		Ok(())
96	}
97	/// Invoke a call operation.
98	fn call(
99		&mut self,
100		code_address: H160,
101		transfer: Option<Transfer>,
102		input: Vec<u8>,
103		target_gas: Option<u64>,
104		is_static: bool,
105		context: Context,
106	) -> Capture<(ExitReason, Vec<u8>), Self::CallInterrupt>;
107	/// Feed in call feedback.
108	fn call_feedback(&mut self, _feedback: Self::CallFeedback) -> Result<(), ExitError> {
109		Ok(())
110	}
111
112	/// Pre-validation step for the runtime.
113	fn pre_validate(
114		&mut self,
115		context: &Context,
116		opcode: Opcode,
117		stack: &Stack,
118	) -> Result<(), ExitError>;
119	/// Handle other unknown external opcodes.
120	fn other(&mut self, opcode: Opcode, _stack: &mut Machine) -> Result<(), ExitError> {
121		Err(ExitError::InvalidCode(opcode))
122	}
123
124	/// Records some associated `ExternalOperation`.
125	fn record_external_operation(&mut self, op: crate::ExternalOperation) -> Result<(), ExitError>;
126}
