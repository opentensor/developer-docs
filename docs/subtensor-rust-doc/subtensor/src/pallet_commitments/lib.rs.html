196    #[pallet::call]
197    impl<T: Config> Pallet<T> {
198        /// Set the commitment for a given netuid
199        #[pallet::call_index(0)]
200        #[pallet::weight((
201            Weight::from_parts(38_000_000, 0)
202			.saturating_add(T::DbWeight::get().reads(5_u64))
203			.saturating_add(T::DbWeight::get().writes(4_u64)),
204            DispatchClass::Operational,
205            Pays::No
206        ))]
207        pub fn set_commitment(
208            origin: OriginFor<T>,
209            netuid: u16,
210            info: Box<CommitmentInfo<T::MaxFields>>,
211        ) -> DispatchResult {
212            let who = ensure_signed(origin.clone())?;
213            ensure!(
214                T::CanCommit::can_commit(netuid, &who),
215                Error::<T>::AccountNotAllowedCommit
216            );
217
218            let extra_fields = info.fields.len() as u32;
219            ensure!(
220                extra_fields <= T::MaxFields::get(),
221                Error::<T>::TooManyFieldsInCommitmentInfo
222            );
223
224            let cur_block = <frame_system::Pallet<T>>::block_number();
225
226            let min_used_space: u64 = 100;
227            let required_space: u64 = info
228                .fields
229                .iter()
230                .map(|field| field.len_for_rate_limit())
231                .sum::<u64>()
232                .max(min_used_space);
233
234            let mut usage = UsedSpaceOf::<T>::get(netuid, &who).unwrap_or_default();
235            let cur_block_u64 = cur_block.saturated_into::<u64>();
236            let current_epoch = T::TempoInterface::get_epoch_index(netuid, cur_block_u64);
237
238            if usage.last_epoch != current_epoch {
239                usage.last_epoch = current_epoch;
240                usage.used_space = 0;
241            }
242
243            // check if ResetBondsFlag is set in the fields
244            for field in info.fields.iter() {
245                if let Data::ResetBondsFlag = field {
246                    // track when bonds reset was last triggered
247                    <LastBondsReset<T>>::insert(netuid, &who, cur_block);
248                    T::OnMetadataCommitment::on_metadata_commitment(netuid, &who);
249                    break;
250                }
251            }
252
253            let max_allowed = MaxSpace::<T>::get() as u64;
254            ensure!(
255                usage.used_space.saturating_add(required_space) <= max_allowed,
256                Error::<T>::SpaceLimitExceeded
257            );
258
259            usage.used_space = usage.used_space.saturating_add(required_space);
260
261            UsedSpaceOf::<T>::insert(netuid, &who, usage);
262
263            let mut id = match <CommitmentOf<T>>::get(netuid, &who) {
264                Some(mut id) => {
265                    id.info = *info.clone();
266                    id.block = cur_block;
267                    id
268                }
269                None => Registration {
270                    info: *info.clone(),
271                    block: cur_block,
272                    deposit: Zero::zero(),
273                },
274            };
275
276            let old_deposit = id.deposit;
277            let fd = <BalanceOf<T>>::from(extra_fields).saturating_mul(T::FieldDeposit::get());
278            id.deposit = T::InitialDeposit::get().saturating_add(fd);
279            if id.deposit > old_deposit {
280                T::Currency::reserve(&who, id.deposit.saturating_sub(old_deposit))?;
281            }
282            if old_deposit > id.deposit {
283                let err_amount =
284                    T::Currency::unreserve(&who, old_deposit.saturating_sub(id.deposit));
285                if !err_amount.is_zero() {
286                    return Err(Error::<T>::UnexpectedUnreserveLeftover.into());
287                }
288            }
289
290            <CommitmentOf<T>>::insert(netuid, &who, id);
291            <LastCommitment<T>>::insert(netuid, &who, cur_block);
292
293            if let Some(Data::TimelockEncrypted { reveal_round, .. }) = info
294                .fields
295                .iter()
296                .find(|data| matches!(data, Data::TimelockEncrypted { .. }))
297            {
298                Self::deposit_event(Event::TimelockCommitment {
299                    netuid,
300                    who: who.clone(),
301                    reveal_round: *reveal_round,
302                });
303
304                TimelockedIndex::<T>::mutate(|index| {
305                    index.insert((netuid, who.clone()));
306                });
307            } else {
308                Self::deposit_event(Event::Commitment {
309                    netuid,
310                    who: who.clone(),
311                });
312
313                TimelockedIndex::<T>::mutate(|index| {
314                    index.remove(&(netuid, who.clone()));
315                });
316            }
317
318            Ok(())
319        }
320
321        /// *DEPRECATED* Sudo-set the commitment rate limit
322        #[pallet::call_index(1)]
323        #[pallet::weight((
324            Weight::from_parts(3_596_000, 0)
325        	.saturating_add(T::DbWeight::get().reads(0_u64))
326        	.saturating_add(T::DbWeight::get().writes(1_u64)),
327        	DispatchClass::Operational,
328        	Pays::No
329        ))]
330        pub fn set_rate_limit(origin: OriginFor<T>, _rate_limit_blocks: u32) -> DispatchResult {
331            ensure_root(origin)?;
332            // RateLimit::<T>::set(rate_limit_blocks.into());
333            Ok(())
334        }
335
336        /// Sudo-set MaxSpace
337        #[pallet::call_index(2)]
338        #[pallet::weight((
339            Weight::from_parts(3_556_000, 0)
340			.saturating_add(T::DbWeight::get().reads(0_u64))
341			.saturating_add(T::DbWeight::get().writes(1_u64)),
342            DispatchClass::Operational,
343            Pays::No
344        ))]
345        pub fn set_max_space(origin: OriginFor<T>, new_limit: u32) -> DispatchResult {
346            ensure_root(origin)?;
347            MaxSpace::<T>::set(new_limit);
348            Ok(())
349        }

505impl<T: Config> Pallet<T> {
506    pub fn reveal_timelocked_commitments() -> DispatchResult {
507        let index = TimelockedIndex::<T>::get();
508        for (netuid, who) in index.clone() {
509            let Some(mut registration) = <CommitmentOf<T>>::get(netuid, &who) else {
510                TimelockedIndex::<T>::mutate(|idx| {
511                    idx.remove(&(netuid, who.clone()));
512                });
513                continue;
514            };
515
516            let original_fields = registration.info.fields.clone();
517            let mut remain_fields = Vec::new();
518            let mut revealed_fields = Vec::new();
519
520            for data in original_fields {
521                match data {
522                    Data::TimelockEncrypted {
523                        encrypted,
524                        reveal_round,
525                    } => {
526                        let pulse = match pallet_drand::Pulses::<T>::get(reveal_round) {
527                            Some(p) => p,
528                            None => {
529                                remain_fields.push(Data::TimelockEncrypted {
530                                    encrypted,
531                                    reveal_round,
532                                });
533                                continue;
534                            }
535                        };
536
537                        let signature_bytes = pulse
538                            .signature
539                            .strip_prefix(b"0x")
540                            .unwrap_or(&pulse.signature);
541                        let sig_reader = &mut &signature_bytes[..];
542                        let sig =
543                            <TinyBLS381 as EngineBLS>::SignatureGroup::deserialize_compressed(
544                                sig_reader,
545                            )
546                            .map_err(|e| {
547                                log::warn!(
548                                    "Failed to deserialize drand signature for {:?}: {:?}",
549                                    who,
550                                    e
551                                )
552                            })
553                            .ok();
554
555                        let Some(sig) = sig else {
556                            log::warn!("No sig after deserialization");
557                            continue;
558                        };
559
560                        let reader = &mut &encrypted[..];
561                        let commit = TLECiphertext::<TinyBLS381>::deserialize_compressed(reader)
562                            .map_err(|e| {
563                                log::warn!(
564                                    "Failed to deserialize TLECiphertext for {:?}: {:?}",
565                                    who,
566                                    e
567                                )
568                            })
569                            .ok();
570
571                        let Some(commit) = commit else {
572                            log::warn!("No commit after deserialization");
573                            continue;
574                        };
575
576                        let decrypted_bytes: Vec<u8> =
577                            tld::<TinyBLS381, AESGCMStreamCipherProvider>(commit, sig)
578                                .map_err(|e| {
579                                    log::warn!("Failed to decrypt timelock for {:?}: {:?}", who, e)
580                                })
581                                .ok()
582                                .unwrap_or_default();
583
584                        if decrypted_bytes.is_empty() {
585                            log::warn!("Bytes were decrypted for {:?} but they are empty", who);
586                            continue;
587                        }
588
589                        revealed_fields.push(decrypted_bytes);
590                    }
591
592                    other => remain_fields.push(other),
593                }
594            }
595
596            if !revealed_fields.is_empty() {
597                let mut existing_reveals =
598                    RevealedCommitments::<T>::get(netuid, &who).unwrap_or_default();
599
600                let current_block = <frame_system::Pallet<T>>::block_number();
601                let block_u64 = current_block.saturated_into::<u64>();
602
603                // Push newly revealed items onto the tail of existing_reveals and emit the event
604                for revealed_bytes in revealed_fields {
605                    existing_reveals.push((revealed_bytes, block_u64));
606
607                    Self::deposit_event(Event::CommitmentRevealed {
608                        netuid,
609                        who: who.clone(),
610                    });
611                }
612
613                const MAX_REVEALS: usize = 10;
614                if existing_reveals.len() > MAX_REVEALS {
615                    let remove_count = existing_reveals.len().saturating_sub(MAX_REVEALS);
616                    existing_reveals.drain(0..remove_count);
617                }
618
619                RevealedCommitments::<T>::insert(netuid, &who, existing_reveals);
620            }
621
622            registration.info.fields = BoundedVec::try_from(remain_fields)
623                .map_err(|_| "Failed to build BoundedVec for remain_fields")?;
624
625            match registration.info.fields.is_empty() {
626                true => {
627                    <CommitmentOf<T>>::remove(netuid, &who);
628                    TimelockedIndex::<T>::mutate(|idx| {
629                        idx.remove(&(netuid, who.clone()));
630                    });
631                }
632                false => {
633                    <CommitmentOf<T>>::insert(netuid, &who, &registration);
634                    let has_timelock = registration
635                        .info
636                        .fields
637                        .iter()
638                        .any(|f| matches!(f, Data::TimelockEncrypted { .. }));
639                    if !has_timelock {
640                        TimelockedIndex::<T>::mutate(|idx| {
641                            idx.remove(&(netuid, who.clone()));
642                        });
643                    }
644                }
645            }
646        }
647
648        Ok(())
649    }
