34/// Represents stored data which can be:
35/// - `Raw`: a direct blob up to 128 bytes
36/// - `BigRaw`: a larger blob up to 512 bytes
37/// - A cryptographic hash (BlakeTwo256, Sha256, Keccak256, ShaThree256)
38/// - A timelock-encrypted blob with a reveal round
39/// - A reset flag (`ResetBondsFlag`)
40///   Can also be `None`.
41#[derive(Clone, Eq, PartialEq, RuntimeDebug, MaxEncodedLen)]
42pub enum Data {
43    /// No data here.
44    None,
45    /// The data is stored directly (up to 128 bytes).
46    Raw(BoundedVec<u8, ConstU32<128>>),
47    /// Only the Blake2 hash of the data is stored. The preimage of the hash may be retrieved
48    /// through some hash-lookup service.
49    BlakeTwo256([u8; 32]),
50    /// Only the SHA2-256 hash of the data is stored. The preimage of the hash may be retrieved
51    /// through some hash-lookup service.
52    Sha256([u8; 32]),
53    /// Only the Keccak-256 hash of the data is stored. The preimage of the hash may be retrieved
54    /// through some hash-lookup service.
55    Keccak256([u8; 32]),
56    /// Only the SHA3-256 hash of the data is stored. The preimage of the hash may be retrieved
57    /// through some hash-lookup service.
58    ShaThree256([u8; 32]),
59    /// A timelock-encrypted commitment with a reveal round.
60    TimelockEncrypted {
61        encrypted: BoundedVec<u8, ConstU32<MAX_TIMELOCK_COMMITMENT_SIZE_BYTES>>,
62        reveal_round: u64,
63    },
64    /// Flag to trigger bonds reset for subnet
65    ResetBondsFlag,
66    /// The data is stored directly (up to 512 bytes).
67    BigRaw(BoundedVec<u8, ConstU32<MAX_BIGRAW_COMMITMENT_SIZE_BYTES>>),
68}
69
70impl Data {
71    pub fn is_none(&self) -> bool {
72        self == &Data::None
73    }
74
75    /// Check if this is a timelock-encrypted commitment.
76    pub fn is_timelock_encrypted(&self) -> bool {
77        matches!(self, Data::TimelockEncrypted { .. })
78    }
79
80    pub fn len_for_rate_limit(&self) -> u64 {
81        match self {
82            Data::None => 0,
83            Data::Raw(bytes) => bytes.len() as u64,
84            Data::BlakeTwo256(arr)
85            | Data::Sha256(arr)
86            | Data::Keccak256(arr)
87            | Data::ShaThree256(arr) => arr.len() as u64,
88            Data::TimelockEncrypted { encrypted, .. } => encrypted.len() as u64,
89            Data::ResetBondsFlag => 0,
90            Data::BigRaw(bytes) => bytes.len() as u64,
91        }
92    }
93}

95impl Decode for Data {
96    fn decode<I: codec::Input>(input: &mut I) -> sp_std::result::Result<Self, codec::Error> {
97        let b = input.read_byte()?;
98        Ok(match b {
99            0 => Data::None,
100            n @ 1..=129 => {
101                let mut r: BoundedVec<_, _> = vec![0u8; (n as usize).saturating_sub(1)]
102                    .try_into()
103                    .expect("bound checked in match arm condition; qed");
104                input.read(&mut r[..])?;
105                Data::Raw(r)
106            }
107            130 => Data::BlakeTwo256(<[u8; 32]>::decode(input)?),
108            131 => Data::Sha256(<[u8; 32]>::decode(input)?),
109            132 => Data::Keccak256(<[u8; 32]>::decode(input)?),
110            133 => Data::ShaThree256(<[u8; 32]>::decode(input)?),
111            134 => {
112                let encrypted =
113                    BoundedVec::<u8, ConstU32<MAX_TIMELOCK_COMMITMENT_SIZE_BYTES>>::decode(input)?;
114                let reveal_round = u64::decode(input)?;
115                Data::TimelockEncrypted {
116                    encrypted,
117                    reveal_round,
118                }
119            }
120            135 => Data::ResetBondsFlag,
121            136 => {
122                let bigvec =
123                    BoundedVec::<u8, ConstU32<MAX_BIGRAW_COMMITMENT_SIZE_BYTES>>::decode(input)?;
124                Data::BigRaw(bigvec)
125            }
126            _ => return Err(codec::Error::from("invalid leading byte")),
127        })
128    }
129}

131impl Encode for Data {
132    fn encode(&self) -> Vec<u8> {
133        match self {
134            Data::None => vec![0u8; 1],
135            Data::Raw(x) => {
136                let l = x.len().min(128) as u8;
137                let mut r = vec![l.saturating_add(1)];
138                r.extend_from_slice(&x[..]);
139                r
140            }
141            Data::BlakeTwo256(h) => once(130).chain(h.iter().cloned()).collect(),
142            Data::Sha256(h) => once(131).chain(h.iter().cloned()).collect(),
143            Data::Keccak256(h) => once(132).chain(h.iter().cloned()).collect(),
144            Data::ShaThree256(h) => once(133).chain(h.iter().cloned()).collect(),
145            Data::TimelockEncrypted {
146                encrypted,
147                reveal_round,
148            } => {
149                let mut r = vec![134];
150                r.extend_from_slice(&encrypted.encode());
151                r.extend_from_slice(&reveal_round.encode());
152                r
153            }
154            Data::ResetBondsFlag => vec![135],
155            Data::BigRaw(bigvec) => {
156                let mut r = vec![136];
157                r.extend_from_slice(&bigvec.encode());
158                r
159            }
160        }
161    }
162}

399/// Information concerning the identity of the controller of an account.
400///
401/// NOTE: This is stored separately primarily to facilitate the addition of extra fields in a
402/// backwards compatible way through a specialized `Decode` impl.
403#[freeze_struct("632f12850e51c420")]
404#[derive(
405    CloneNoBound, Encode, Eq, MaxEncodedLen, PartialEqNoBound, RuntimeDebugNoBound, TypeInfo,
406)]
407#[codec(mel_bound())]
408#[scale_info(skip_type_params(MaxFields))]
409pub struct Registration<
410    Balance: Encode + Decode + MaxEncodedLen + Copy + Clone + Debug + Eq + PartialEq,
411    MaxFields: Get<u32>,
412    BlockNumber: Codec + Clone + Ord + Eq + AtLeast32BitUnsigned + MaxEncodedLen + Debug,
413> {
414    /// Amount held on deposit for this information.
415    pub deposit: Balance,
416
417    pub block: BlockNumber,
418
419    /// Information on the identity.
420    pub info: CommitmentInfo<MaxFields>,
421}

424//         Balance: Encode + Decode + MaxEncodedLen + Copy + Clone + Debug + Eq + PartialEq + Zero + Add,
425//         MaxFields: Get<u32>,
426//         Block: Codec + Clone + Ord + Eq + AtLeast32BitUnsigned + MaxEncodedLen + Debug,
427//     > Registration<Balance, MaxFields, Block>
428// {

430//         self.deposit

434impl<
435    Balance: Encode + Decode + MaxEncodedLen + Copy + Clone + Debug + Eq + PartialEq,
436    MaxFields: Get<u32>,
437    Block: Codec + Clone + Ord + Eq + AtLeast32BitUnsigned + MaxEncodedLen + Debug,
438> Decode for Registration<Balance, MaxFields, Block>
439{
440    fn decode<I: codec::Input>(input: &mut I) -> sp_std::result::Result<Self, codec::Error> {
441        let (deposit, block, info) = Decode::decode(&mut AppendZerosInput::new(input))?;
442        Ok(Self {
443            deposit,
444            block,
445            info,
446        })
447    }
