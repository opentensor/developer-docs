1/*!
2  `linregress` provides an easy to use implementation of ordinary
3  least squared linear regression with some basic statistics.
4  See [`RegressionModel`] for details.
5
6  The builder [`FormulaRegressionBuilder`] is used to construct a model from a
7  table of data and an R-style formula or a list of columns to use.
8  Currently only very simple formulae are supported,
9  see [`FormulaRegressionBuilder.formula`] for details.
10
11  # Example
12
13  ```
14  use linregress::{FormulaRegressionBuilder, RegressionDataBuilder};
15
16  # use linregress::Error;
17  # fn main() -> Result<(), Error> {
18  let y = vec![1., 2. ,3. , 4., 5.];
19  let x1 = vec![5., 4., 3., 2., 1.];
20  let x2 = vec![729.53, 439.0367, 42.054, 1., 0.];
21  let x3 = vec![258.589, 616.297, 215.061, 498.361, 0.];
22  let data = vec![("Y", y), ("X1", x1), ("X2", x2), ("X3", x3)];
23  let data = RegressionDataBuilder::new().build_from(data)?;
24  let formula = "Y ~ X1 + X2 + X3";
25  let model = FormulaRegressionBuilder::new()
26      .data(&data)
27      .formula(formula)
28      .fit()?;
29  let parameters: Vec<_> = model.iter_parameter_pairs().collect();
30  let pvalues: Vec<_> = model.iter_p_value_pairs().collect();
31  let standard_errors: Vec<_> = model.iter_se_pairs().collect();
32  assert_eq!(
33      parameters,
34      vec![
35          ("X1", -0.9999999999999745),
36          ("X2", 1.5872719805187785e-15),
37          ("X3", -1.4246416546459528e-15),
38      ]
39  );
40  assert_eq!(
41      standard_errors,
42      vec![
43          ("X1", 9.799066977595267e-13),
44          ("X2", 4.443774660560714e-15),
45          ("X3", 2.713389610740135e-15),
46      ]
47  );
48  assert_eq!(
49      pvalues,
50      vec![
51          ("X1", 6.238279788691533e-13),
52          ("X2", 0.7815975465725482),
53          ("X3", 0.6922074604135647),
54      ]
55  );
56  # Ok(())
57  # }
58  ```

114/// Only exposed for use in doc comments. This function is not considered part of this crate's stable API.
115#[doc(hidden)]
116pub fn almost_equal(a: f64, b: f64, precision: f64) -> bool {
117    if a.is_infinite() || b.is_infinite() || a.is_nan() || b.is_nan() {
118        false
119    } else {
120        (a - b).abs() <= precision
121    }
122}
123
124/// Only exposed for use in doc comments. This function is not considered part of this crate's stable API.
125#[doc(hidden)]
126pub fn slices_almost_equal(a: &[f64], b: &[f64], precision: f64) -> bool {
127    if a.len() != b.len() {
128        return false;
129    }
130    for (&x, &y) in a.iter().zip(b.iter()) {
131        if !almost_equal(x, y, precision) {
132            return false;
133        }
134    }
135    true
136}

138/// Compares `a` and `b` approximately.
139///
140/// They are considered equal if
141/// `(a-b).abs() <= epsilon` or they differ by at most `max_ulps`
142/// `units of least precision` i.e. there are at most `max_ulps`
143/// other representable floating point numbers between `a` and `b`
144fn ulps_eq(a: f64, b: f64, epsilon: f64, max_ulps: u32) -> bool {
145    if (a - b).abs() <= epsilon {
146        return true;
147    }
148    if a.signum() != b.signum() {
149        return false;
150    }
151    let a: u64 = a.to_bits();
152    let b: u64 = b.to_bits();
153    a.abs_diff(b) <= max_ulps as u64
154}

156/// A builder to create and fit a linear regression model.
157///
158/// Given a dataset and a set of columns to use this builder
159/// will produce an ordinary least squared linear regression model.
160///
161/// See [`formula`] and [`data`] for details on how to configure this builder.
162///
163/// The pseudo inverse method is used to fit the model.
164///
165/// # Usage
166///
167/// ```
168/// use linregress::{FormulaRegressionBuilder, RegressionDataBuilder, assert_almost_eq};
169///
170/// # use linregress::Error;
171/// # fn main() -> Result<(), Error> {
172/// let y = vec![1., 2. ,3., 4.];
173/// let x = vec![4., 3., 2., 1.];
174/// let data = vec![("Y", y), ("X", x)];
175/// let data = RegressionDataBuilder::new().build_from(data)?;
176/// let model = FormulaRegressionBuilder::new().data(&data).formula("Y ~ X").fit()?;
177/// // Alternatively
178/// let model = FormulaRegressionBuilder::new().data(&data).data_columns("Y", ["X"]).fit()?;
179/// let params = model.parameters();
180/// assert_almost_eq!(params[0], 4.999999999999998);
181/// assert_almost_eq!(params[1], -0.9999999999999989);
182/// assert_eq!(model.regressor_names()[0], "X");
183/// # Ok(())
184/// # }
185/// ```
186///
187/// [`formula`]: struct.FormulaRegressionBuilder.html#method.formula
188/// [`data`]: struct.FormulaRegressionBuilder.html#method.data
189#[derive(Debug, Clone)]
190pub struct FormulaRegressionBuilder<'a> {
191    data: Option<&'a RegressionData<'a>>,
192    formula: Option<Cow<'a, str>>,
193    columns: Option<(Cow<'a, str>, Vec<Cow<'a, str>>)>,
194}
195
196impl<'a> Default for FormulaRegressionBuilder<'a> {
197    fn default() -> Self {
198        FormulaRegressionBuilder::new()
199    }
200}

202impl<'a> FormulaRegressionBuilder<'a> {
203    /// Create as new FormulaRegressionBuilder with no data or formula set.
204    pub fn new() -> Self {
205        FormulaRegressionBuilder {
206            data: None,
207            formula: None,
208            columns: None,
209        }
210    }
211
212    /// Set the data to be used for the regression.
213    ///
214    /// The data has to be given as a reference to a [`RegressionData`] struct.
215    /// See [`RegressionDataBuilder`] for details.
216    ///
217    /// [`RegressionData`]: struct.RegressionData.html
218    /// [`RegressionDataBuilder`]: struct.RegressionDataBuilder.html
219    pub fn data(mut self, data: &'a RegressionData<'a>) -> Self {
220        self.data = Some(data);
221        self
222    }
223
224    /// Set the formula to use for the regression.
225    ///
226    /// The expected format is `<regressand> ~ <regressor 1> + <regressor 2>`.
227    ///
228    /// E.g. for a regressand named Y and three regressors named A, B and C
229    /// the correct format would be `Y ~ A + B + C`.
230    ///
231    /// Note that there is currently no special support for categorical variables.
232    /// So if you have a categorical variable with more than two distinct values
233    /// or values that are not `0` and `1` you will need to perform "dummy coding" yourself.
234    ///
235    /// Alternatively you can use [`data_columns`][Self::data_columns].
236    pub fn formula<T: Into<Cow<'a, str>>>(mut self, formula: T) -> Self {
237        self.formula = Some(formula.into());
238        self
239    }
240
241    /// Set the columns to be used as regressand and regressors for the regression.
242    ///
243    /// Note that there is currently no special support for categorical variables.
244    /// So if you have a categorical variable with more than two distinct values
245    /// or values that are not `0` and `1` you will need to perform "dummy coding" yourself.
246    ///
247    /// Alternatively you can use [`formula`][Self::formula].
248    pub fn data_columns<I, S1, S2>(mut self, regressand: S1, regressors: I) -> Self
249    where
250        I: IntoIterator<Item = S2>,
251        S1: Into<Cow<'a, str>>,
252        S2: Into<Cow<'a, str>>,
253    {
254        let regressand = regressand.into();
255        let regressors: Vec<_> = regressors.into_iter().map(|i| i.into()).collect();
256        self.columns = Some((regressand, regressors));
257        self
258    }
259
260    /// Fits the model and returns a [`RegressionModel`] if successful.
261    /// You need to set the data with [`data`] and a formula with [`formula`]
262    /// before you can use it.
263    ///
264    /// [`RegressionModel`]: struct.RegressionModel.html
265    /// [`data`]: struct.FormulaRegressionBuilder.html#method.data
266    /// [`formula`]: struct.FormulaRegressionBuilder.html#method.formula
267    pub fn fit(self) -> Result<RegressionModel, Error> {
268        let FittingData(input_vector, output_matrix, outputs) =
269            Self::get_matrices_and_regressor_names(self)?;
270        RegressionModel::try_from_matrices_and_regressor_names(input_vector, output_matrix, outputs)
271    }
272
273    /// Like [`fit`] but does not perfom any statistics on the resulting model.
274    /// Returns a [`Vec`] containing the model parameters
275    /// (in the order `intercept, column 1, column 2, â€¦`) if successfull.
276    ///
277    /// This is usefull if you do not care about the statistics or the model and data
278    /// you want to fit result in too few residual degrees of freedom to perform
279    /// statistics.
280    ///
281    /// [`fit`]: struct.FormulaRegressionBuilder.html#method.fit
282    pub fn fit_without_statistics(self) -> Result<Vec<f64>, Error> {
283        let FittingData(input_vector, output_matrix, _output_names) =
284            Self::get_matrices_and_regressor_names(self)?;
285        let low_level_result = fit_ols_pinv(input_vector, output_matrix)?;
286        let parameters = low_level_result.params;
287        Ok(parameters.iter().copied().collect())
288    }
289
290    fn get_matrices_and_regressor_names(self) -> Result<FittingData, Error> {
291        let (input, outputs) = self.get_data_columns()?;
292        let data = &self.data.ok_or(Error::NoData)?.data;
293        let input_vector: Vec<f64> = data
294            .get(&input)
295            .cloned()
296            .ok_or_else(|| Error::ColumnNotInData(input.into()))?;
297        let mut output_matrix = Vec::new();
298        // Add column of all ones as the first column of the matrix
299        output_matrix.resize(input_vector.len(), 1.);
300        // Add each input as a new column of the matrix
301        for output in &outputs {
302            let output_vec = data
303                .get(output.as_ref())
304                .ok_or_else(|| Error::ColumnNotInData(output.to_string()))?;
305            ensure!(
306                output_vec.len() == input_vector.len(),
307                Error::RegressorRegressandDimensionMismatch(output.to_string())
308            );
309            output_matrix.extend_from_slice(output_vec);
310        }
311        let output_matrix = DMatrix::from_vec(input_vector.len(), outputs.len() + 1, output_matrix);
312        let outputs: Vec<_> = outputs.iter().map(|x| x.to_string()).collect();
313        Ok(FittingData(input_vector, output_matrix, outputs))
314    }
315
316    fn get_data_columns(&self) -> Result<(Cow<'_, str>, Vec<Cow<'_, str>>), Error> {
317        match (self.formula.as_ref(), self.columns.as_ref()) {
318            (Some(..), Some(..)) => Err(Error::BothFormulaAndDataColumnsGiven),
319            (Some(formula), None) => Self::parse_formula(formula),
320            (None, Some((regressand, regressors))) => {
321                ensure!(!regressors.is_empty(), Error::InvalidDataColumns);
322                Ok((regressand.clone(), regressors.clone()))
323            }
324            (None, None) => Err(Error::NoFormula),
325        }
326    }
327
328    fn parse_formula(formula: &str) -> Result<(Cow<'_, str>, Vec<Cow<'_, str>>), Error> {
329        let (input, outputs) = formula.split_once('~').ok_or(Error::InvalidFormula)?;
330        let input = input.trim();
331        let outputs: Vec<_> = outputs
332            .split('+')
333            .map(str::trim)
334            .filter(|x| !x.is_empty())
335            .map(|i| i.into())
336            .collect();
337        ensure!(!outputs.is_empty(), Error::InvalidFormula);
338        Ok((input.into(), outputs))
339    }
340}

346/// A container struct for the regression data.
347///
348/// This struct is obtained using a [`RegressionDataBuilder`].
349///
350/// [`RegressionDataBuilder`]: struct.RegressionDataBuilder.html
351#[derive(Debug, Clone)]
352pub struct RegressionData<'a> {
353    data: HashMap<Cow<'a, str>, Vec<f64>>,
354}
355
356impl<'a> RegressionData<'a> {
357    /// Constructs a new `RegressionData` struct from any collection that
358    /// implements the `IntoIterator` trait.
359    ///
360    /// The iterator must consist of tupels of the form `(S, Vec<f64>)` where
361    /// `S` is a type that can be converted to a `Cow<'a, str>`.
362    ///
363    /// `invalid_value_handling` specifies what to do if invalid data is encountered.
364    fn new<I, S>(
365        data: I,
366        invalid_value_handling: InvalidValueHandling,
367    ) -> Result<RegressionData<'a>, Error>
368    where
369        I: IntoIterator<Item = (S, Vec<f64>)>,
370        S: Into<Cow<'a, str>>,
371    {
372        let temp: HashMap<_, _> = data
373            .into_iter()
374            .map(|(key, value)| (key.into(), value))
375            .collect();
376        let first_key = temp.keys().next();
377        ensure!(
378            first_key.is_some(),
379            Error::RegressionDataError("The data contains no columns.".into())
380        );
381        let first_key = first_key.unwrap();
382        let first_len = temp[first_key].len();
383        ensure!(
384            first_len > 0,
385            Error::RegressionDataError("The data contains an empty column.".into())
386        );
387        for key in temp.keys() {
388            let this_len = temp[key].len();
389            ensure!(
390                this_len == first_len,
391                Error::RegressionDataError(
392                    "The lengths of the columns in the given data are inconsistent.".into()
393                )
394            );
395            ensure!(
396                !key.contains('~') && !key.contains('+'),
397                Error::RegressionDataError(
398                    "The column names may not contain `~` or `+`, because they are used \
399                             as separators in the formula."
400                        .into()
401                )
402            );
403        }
404        if Self::check_if_all_columns_are_equal(&temp) {
405            return Err(Error::RegressionDataError(
406                "All input columns contain only equal values. Fitting this model would lead \
407                     to invalid statistics."
408                    .into(),
409            ));
410        }
411        if Self::check_if_data_is_valid(&temp) {
412            return Ok(Self { data: temp });
413        }
414        match invalid_value_handling {
415            InvalidValueHandling::ReturnError => Err(Error::RegressionDataError(
416                "The data contains a non real value (NaN or infinity or negative infinity). \
417                 If you would like to silently drop these values configure the builder with \
418                 InvalidValueHandling::DropInvalid."
419                    .into(),
420            )),
421            InvalidValueHandling::DropInvalid => {
422                let temp = Self::drop_invalid_values(temp);
423                let first_key = temp.keys().next().expect("Cleaned data has no columns.");
424                let first_len = temp[first_key].len();
425                ensure!(
426                    first_len > 0,
427                    Error::RegressionDataError("The cleaned data is empty.".into())
428                );
429                Ok(Self { data: temp })
430            }
431        }
432    }
433
434    fn check_if_all_columns_are_equal(data: &HashMap<Cow<'a, str>, Vec<f64>>) -> bool {
435        for column in data.values() {
436            if column.is_empty() {
437                return false;
438            }
439            let first_iter = iter::repeat(&column[0]).take(column.len());
440            if !first_iter.eq(column.iter()) {
441                return false;
442            }
443        }
444        true
445    }
446
447    fn check_if_data_is_valid(data: &HashMap<Cow<'a, str>, Vec<f64>>) -> bool {
448        for column in data.values() {
449            if column.iter().any(|x| !x.is_finite()) {
450                return false;
451            }
452        }
453        true
454    }
455
456    fn drop_invalid_values(
457        data: HashMap<Cow<'a, str>, Vec<f64>>,
458    ) -> HashMap<Cow<'a, str>, Vec<f64>> {
459        let mut invalid_rows: BTreeSet<usize> = BTreeSet::new();
460        for column in data.values() {
461            for (index, value) in column.iter().enumerate() {
462                if !value.is_finite() {
463                    invalid_rows.insert(index);
464                }
465            }
466        }
467        let mut cleaned_data = HashMap::new();
468        for (key, mut column) in data {
469            for index in invalid_rows.iter().rev() {
470                column.remove(*index);
471            }
472            cleaned_data.insert(key, column);
473        }
474        cleaned_data
475    }
476}

518    /// Build a [`RegressionData`] struct from the given data.
519    ///
520    /// Any type that implements the [`IntoIterator`] trait can be used for the data.
521    /// This could for example be a [`Hashmap`] or a [`Vec`].
522    ///
523    /// The iterator must consist of tupels of the form `(S, Vec<f64>)` where
524    /// `S` is a type that implements `Into<Cow<str>>`, such as [`String`] or [`str`].
525    ///
526    /// You can think of this format as the representation of a table of data where
527    /// each tuple `(S, Vec<f64>)` represents a column. The `S` is the header or label of the
528    /// column and the `Vec<f64>` contains the data of the column.
529    ///
530    /// Because `~` and `+` are used as separators in the formula they may not be used in the name
531    /// of a data column.
532    ///
533    /// # Example
534    ///
535    /// ```
536    /// use std::collections::HashMap;
537    /// use linregress::RegressionDataBuilder;
538    ///
539    /// # use linregress::Error;
540    /// # fn main() -> Result<(), Error> {
541    /// let mut data1 = HashMap::new();
542    /// data1.insert("Y", vec![1., 2., 3., 4.]);
543    /// data1.insert("X", vec![4., 3., 2., 1.]);
544    /// let regression_data1 = RegressionDataBuilder::new().build_from(data1)?;
545    ///
546    /// let y = vec![1., 2., 3., 4.];
547    /// let x = vec![4., 3., 2., 1.];
548    /// let data2 = vec![("X", x), ("Y", y)];
549    /// let regression_data2 = RegressionDataBuilder::new().build_from(data2)?;
550    /// # Ok(())
551    /// # }
552    /// ```
553    ///
554    /// [`RegressionData`]: struct.RegressionData.html
555    /// [`IntoIterator`]: https://doc.rust-lang.org/std/iter/trait.IntoIterator.html
556    /// [`Hashmap`]: https://doc.rust-lang.org/std/collections/struct.HashMap.html
557    /// [`Vec`]: https://doc.rust-lang.org/std/vec/struct.Vec.html
558    /// [`String`]: https://doc.rust-lang.org/std/string/struct.String.html
559    /// [`str`]: https://doc.rust-lang.org/std/primitive.str.html
560    pub fn build_from<'a, I, S>(self, data: I) -> Result<RegressionData<'a>, Error>
561    where
562        I: IntoIterator<Item = (S, Vec<f64>)>,
563        S: Into<Cow<'a, str>>,
564    {
565        RegressionData::new(data, self.handle_invalid_values)
566    }
567}

754    /// Evaluates the model on given new input data and returns the predicted values.
755    ///
756    /// The new data is expected to have the same columns as the original data.
757    /// See [`RegressionDataBuilder.build`] for details on the type of the `new_data` parameter.
758    ///
759    /// ## Note
760    ///
761    /// This function does *no* special handling of non real values (NaN or infinity or negative infinity).
762    /// Such a value in `new_data` will result in a corresponding meaningless prediction.
763    ///
764    /// ## Example
765    ///
766    /// ```
767    /// # use linregress::{RegressionDataBuilder, FormulaRegressionBuilder, assert_slices_almost_eq};
768    /// # use linregress::Error;
769    /// # fn main() -> Result<(), Error> {
770    /// let y = vec![1., 2., 3., 4., 5.];
771    /// let x1 = vec![5., 4., 3., 2., 1.];
772    /// let x2 = vec![729.53, 439.0367, 42.054, 1., 0.];
773    /// let x3 = vec![258.589, 616.297, 215.061, 498.361, 0.];
774    /// let data = vec![("Y", y), ("X1", x1), ("X2", x2), ("X3", x3)];
775    /// let data = RegressionDataBuilder::new().build_from(data).unwrap();
776    /// let formula = "Y ~ X1 + X2 + X3";
777    /// let model = FormulaRegressionBuilder::new()
778    ///     .data(&data)
779    ///     .formula(formula)
780    ///     .fit()?;
781    /// let new_data = vec![
782    ///     ("X1", vec![2.5, 3.5]),
783    ///     ("X2", vec![2.0, 8.0]),
784    ///     ("X3", vec![2.0, 1.0]),
785    /// ];
786    /// let prediction: Vec<f64> = model.predict(new_data)?;
787    /// assert_slices_almost_eq!(&prediction, &[3.500000000000028, 2.5000000000000644]);
788    /// # Ok(())
789    /// # }
790    /// ```
791    ///
792    /// [`RegressionDataBuilder.build`]: struct.RegressionDataBuilder.html#method.build_from
793    pub fn predict<'a, I, S>(&self, new_data: I) -> Result<Vec<f64>, Error>
794    where
795        I: IntoIterator<Item = (S, Vec<f64>)>,
796        S: Into<Cow<'a, str>>,
797    {
798        let new_data: HashMap<Cow<'_, _>, Vec<f64>> = new_data
799            .into_iter()
800            .map(|(key, value)| (key.into(), value))
801            .collect();
802        self.check_variables(&new_data)?;
803        let input_len = new_data.values().next().unwrap().len();
804        let mut new_data_values: Vec<f64> = vec![];
805        for key in &self.regressor_names {
806            new_data_values.extend_from_slice(new_data[&Cow::from(key)].as_slice());
807        }
808
809        let num_regressors = self.model.parameters.len() - 1;
810        let new_data_matrix = DMatrix::from_vec(input_len, num_regressors, new_data_values);
811        let param_matrix = DMatrix::from_iterator(
812            num_regressors,
813            1,
814            self.model.parameters.iter().skip(1).copied(),
815        );
816        let intercept = self.model.parameters[0];
817        let intercept_matrix =
818            DMatrix::from_iterator(input_len, 1, std::iter::repeat(intercept).take(input_len));
819        let predictions = (new_data_matrix * param_matrix) + intercept_matrix;
820        let predictions: Vec<f64> = predictions.into_iter().copied().collect();
821        Ok(predictions)
822    }
823
824    fn check_variables(
825        &self,
826        given_parameters: &HashMap<Cow<'_, str>, Vec<f64>>,
827    ) -> Result<(), Error> {
828        ensure!(!given_parameters.is_empty(), Error::NoData);
829        let first_len = given_parameters.values().next().unwrap().len();
830        ensure!(first_len > 0, Error::NoData);
831        let model_parameters: HashSet<_> = self.regressor_names.iter().map(Cow::from).collect();
832        for param in &model_parameters {
833            if !given_parameters.contains_key(param) {
834                return Err(Error::ColumnNotInData(param.to_string()));
835            }
836        }
837        for (param, values) in given_parameters {
838            ensure!(values.len() == first_len, Error::InconsistentVectors);
839            if !model_parameters.contains(param) {
840                return Err(Error::ModelColumnNotInData(param.to_string()));
841            }
842        }
843        Ok(())
844    }
845
846    fn try_from_matrices_and_regressor_names<I: IntoIterator<Item = String>>(
847        inputs: Vec<f64>,
848        outputs: DMatrix<f64>,
849        output_names: I,
850    ) -> Result<Self, Error> {
851        let low_level_result = fit_ols_pinv(inputs, outputs)?;
852        let model = LowLevelRegressionModel::from_low_level_regression(low_level_result)?;
853        let regressor_names: Vec<String> = output_names.into_iter().collect();
854        let num_slopes = model.parameters.len() - 1;
855        ensure!(
856            regressor_names.len() == num_slopes,
857            Error::InconsistentSlopes(InconsistentSlopes::new(regressor_names.len(), num_slopes))
858        );
859        Ok(Self {
860            regressor_names,
861            model,
862        })
863    }

866/// A fitted regression model
867///
868/// Is the result of [`fit_low_level_regression_model`].
869///
870#[derive(Debug, Clone)]
871pub struct LowLevelRegressionModel {
872    /// The model's intercept and slopes (also known as betas).
873    parameters: Vec<f64>,
874    /// The standard errors of the parameter estimates.
875    se: Vec<f64>,
876    /// Sum of squared residuals.
877    ssr: f64,
878    /// R-squared of the model.
879    rsquared: f64,
880    /// Adjusted R-squared of the model.
881    rsquared_adj: f64,
882    /// The two-tailed p-values for the t-statistics of the params.
883    pvalues: Vec<f64>,
884    /// The residuals of the model.
885    residuals: Vec<f64>,
886    ///  A scale factor for the covariance matrix.
887    ///
888    ///  Note that the square root of `scale` is often
889    ///  called the standard error of the regression.
890    scale: f64,
891}
892
893impl LowLevelRegressionModel {
894    fn from_low_level_regression(
895        low_level_result: InternalLowLevelRegressionResult,
896    ) -> Result<Self, Error> {
897        let parameters = low_level_result.params;
898        let singular_values = low_level_result.singular_values;
899        let normalized_cov_params = low_level_result.normalized_cov_params;
900        let diag = DMatrix::from_diagonal(&singular_values);
901        let rank = &diag.rank(0.0);
902        let input_vec = low_level_result.inputs.to_vec();
903        let input_matrix = DMatrix::from_vec(low_level_result.inputs.len(), 1, input_vec);
904        let residuals = &input_matrix - (low_level_result.outputs * parameters.to_owned());
905        let ssr = residuals.dot(&residuals);
906        let n = low_level_result.inputs.len();
907        let df_resid = n - rank;
908        ensure!(
909            df_resid >= 1,
910            Error::ModelFittingError(
911                "There are not enough residual degrees of freedom to perform statistics on this model".into()));
912        let scale = residuals.dot(&residuals) / df_resid as f64;
913        let cov_params = normalized_cov_params * scale;
914        let se = get_se_from_cov_params(&cov_params);
915        let mean = input_matrix.mean();
916        let mut centered_input_matrix = input_matrix;
917        subtract_value_from_matrix(&mut centered_input_matrix, mean);
918        let centered_tss = centered_input_matrix.dot(&centered_input_matrix);
919        let rsquared = 1. - (ssr / centered_tss);
920        let rsquared_adj = 1. - ((n - 1) as f64 / df_resid as f64 * (1. - rsquared));
921        let tvalues = parameters
922            .iter()
923            .zip(se.iter())
924            .map(|(&x, &y)| x / y.max(std::f64::EPSILON));
925        let pvalues: Vec<f64> = tvalues
926            .map(|x| students_t_cdf(x.abs().neg(), df_resid as i64).map(|i| i * 2.))
927            .collect::<Option<_>>()
928            .ok_or_else(|| {
929                Error::ModelFittingError(
930                    "Failed to calculate p-values: students_t_cdf failed due to invalid parameters"
931                        .into(),
932                )
933            })?;
934        // Convert these from internal Matrix types to user facing types
935        let parameters: Vec<f64> = parameters.iter().copied().collect();
936        let residuals: Vec<f64> = residuals.iter().copied().collect();
937        Ok(Self {
938            parameters,
939            se,
940            ssr,
941            rsquared,
942            rsquared_adj,
943            pvalues,
944            residuals,
945            scale,
946        })
947    }

1001/// Fit a regression model directly on a matrix of input data
1002///
1003/// Expects a matrix in the format
1004///
1005/// | regressand | intercept | regressor 1 | regressor 2 | â€¦   |
1006/// |------------|-----------|-------------|-------------|-----|
1007/// | value      | 1.0       | value       | value       | â€¦   |
1008/// | â‹®          | â‹®         | â‹®           | â‹®           | â‹®   |
1009///
1010/// in row major order.
1011///
1012/// # Note
1013/// - The matrix should already contain the `intercept` column consisting of only the value `1.0`.
1014/// - No validation of the data is performed, except for a simple dimension consistency check.
1015///
1016/// # Example
1017/// ```
1018/// # fn main() -> Result<(), linregress::Error> {
1019/// use linregress::{fit_low_level_regression_model, assert_slices_almost_eq};
1020///
1021/// let data_row_major: Vec<f64> = vec![
1022///     1., 1.0, 1., 7.,
1023///     3., 1.0, 2., 6.,
1024///     4., 1.0, 3., 5.,
1025///     5., 1.0, 4., 4.,
1026///     2., 1.0, 5., 3.,
1027///     3., 1.0, 6., 2.,
1028///     4., 1.0, 7., 1.,
1029/// ];
1030/// let model = fit_low_level_regression_model(&data_row_major, 7, 4)?;
1031/// let params = [
1032///     0.09523809523809518f64,
1033///     0.5059523809523807,
1034///     0.2559523809523811,
1035/// ];
1036/// assert_slices_almost_eq!(model.parameters(), &params);
1037/// # Ok(())
1038/// # }
1039/// ```
1040pub fn fit_low_level_regression_model(
1041    data_row_major: &[f64],
1042    num_rows: usize,
1043    num_columns: usize,
1044) -> Result<LowLevelRegressionModel, Error> {
1045    let regression = get_low_level_regression(data_row_major, num_rows, num_columns)?;
1046    let model = LowLevelRegressionModel::from_low_level_regression(regression)?;
1047    Ok(model)
1048}

1063fn get_low_level_regression(
1064    data_row_major: &[f64],
1065    num_rows: usize,
1066    num_columns: usize,
1067) -> Result<InternalLowLevelRegressionResult, Error> {
1068    ensure!(
1069        !data_row_major.is_empty() && num_rows * num_columns == data_row_major.len(),
1070        Error::InconsistentVectors
1071    );
1072    let data = DMatrix::from_row_slice(num_rows, num_columns, data_row_major);
1073    let inputs = data.view((0, 0), (num_rows, 1));
1074    let inputs: Vec<f64> = inputs.iter().copied().collect();
1075    let outputs: DMatrix<f64> = data.view((0, 1), (num_rows, num_columns - 1)).into_owned();
1076    fit_ols_pinv(inputs, outputs)
1077}

1079/// Result of fitting a low level matrix based model
1080#[derive(Debug, Clone)]
1081struct InternalLowLevelRegressionResult {
1082    inputs: Vec<f64>,
1083    outputs: DMatrix<f64>,
1084    params: DMatrix<f64>,
1085    singular_values: DVector<f64>,
1086    normalized_cov_params: DMatrix<f64>,
1087}
1088
1089/// Performs ordinary least squared linear regression using the pseudo inverse method
1090fn fit_ols_pinv(
1091    inputs: Vec<f64>,
1092    outputs: DMatrix<f64>,
1093) -> Result<InternalLowLevelRegressionResult, Error> {
1094    ensure!(
1095        !inputs.is_empty(),
1096        Error::ModelFittingError(
1097            "Fitting the model failed because the input vector is empty".into()
1098        )
1099    );
1100    ensure!(
1101        outputs.nrows() >= 1 && outputs.ncols() >= 1,
1102        Error::ModelFittingError(
1103            "Fitting the model failed because the output matrix is empty".into()
1104        )
1105    );
1106    let singular_values = outputs
1107        .to_owned()
1108        .try_svd(false, false, std::f64::EPSILON, 0)
1109        .ok_or_else(|| {
1110            Error::ModelFittingError(
1111                "Computing the singular-value decomposition of the output matrix failed".into(),
1112            )
1113        })?
1114        .singular_values;
1115    let pinv = outputs.clone().pseudo_inverse(0.).map_err(|_| {
1116        Error::ModelFittingError("Taking the pinv of the output matrix failed".into())
1117    });
1118    let pinv = pinv?;
1119    let normalized_cov_params = &pinv * &pinv.transpose();
1120    let params = get_sum_of_products(&pinv, &inputs);
1121    ensure!(
1122        params.len() >= 2,
1123        Error::ModelFittingError("Invalid parameter matrix".into())
1124    );
1125    Ok(InternalLowLevelRegressionResult {
1126        inputs,
1127        outputs,
1128        params,
1129        singular_values,
1130        normalized_cov_params,
1131    })
1132}

1134fn subtract_value_from_matrix(matrix: &mut DMatrix<f64>, sub: f64) {
1135    for i in matrix.iter_mut() {
1136        *i -= sub;
1137    }

1140/// Calculates the standard errors given a model's covariate parameters
1141fn get_se_from_cov_params(matrix: &DMatrix<f64>) -> Vec<f64> {
1142    matrix
1143        .row_iter()
1144        .enumerate()
1145        .map(|(n, row)| row.get(n).expect("BUG: Matrix is not square").sqrt())
1146        .collect()
1147}
1148
1149fn get_sum_of_products(matrix: &DMatrix<f64>, vector: &[f64]) -> DMatrix<f64> {
1150    DMatrix::from_iterator(
1151        matrix.nrows(),
1152        1,
1153        matrix
1154            .row_iter()
1155            .map(|row| row.iter().zip(vector.iter()).map(|(x, y)| x * y).sum()),
1156    )
1157}
