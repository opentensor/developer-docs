11// ln(pi)
12const LN_PI: f64 = 1.144_729_885_849_400_2;
13// ln(2 * sqrt(e / pi))
14const LN_2_SQRT_E_OVER_PI: f64 = 0.620_782_237_635_245_2;
15// Auxiliary variable when evaluating the `gamma_ln` function
16const GAMMA_R: f64 = 10.900511;
17
18// Polynomial coefficients for approximating the `gamma_ln` function
19const GAMMA_DK: [f64; 11] = [
20    2.485_740_891_387_535_5e-5,
21    1.051_423_785_817_219_7,
22    -3.456_870_972_220_162_5,
23    4.512_277_094_668_948,
24    -2.982_852_253_235_766_4,
25    1.056_397_115_771_267,

31];

33// Standard epsilon, maximum relative precision of IEEE 754 double-precision
34// floating point numbers (64 bit) e.g. `2^-53`
35const F64_PREC: f64 = 1.1102230246251565e-16;

37/// Calculates the cumulative distribution function for the student's at `x`
38/// with location `0` and scale `1`.
39///
40/// # Formula
41///
42/// ```ignore
43/// if x < μ {
44///     (1 / 2) * I(t, v / 2, 1 / 2)
45/// } else {
46///     1 - (1 / 2) * I(t, v / 2, 1 / 2)
47/// }
48/// ```
49///
50/// where `t = v / (v + k^2)`, `k = (x - μ) / σ`, `μ` is the location,
51/// `σ` is the scale, `v` is the freedom, and `I` is the regularized
52/// incomplete
53/// beta function
54pub fn students_t_cdf(x: f64, freedom: i64) -> Option<f64> {
55    if freedom <= 0 {
56        return None;
57    }
58    let location: f64 = 0.;
59    let scale: f64 = 1.0;
60    let freedom = freedom as f64;
61    let k = (x - location) / scale;
62    let h = freedom / (freedom + k * k);
63    let ib = 0.5 * checked_beta_reg(freedom / 2.0, 0.5, h)?;
64    if x <= location {
65        Some(ib)
66    } else {
67        Some(1.0 - ib)
68    }
69}

71/// Computes the regularized lower incomplete beta function
72/// `I_x(a,b) = 1/Beta(a,b) * int(t^(a-1)*(1-t)^(b-1), t=0..x)`
73/// `a > 0`, `b > 0`, `1 >= x >= 0` where `a` is the first beta parameter,
74/// `b` is the second beta parameter, and `x` is the upper limit of the
75/// integral.
76///
77/// Returns `None` if `a <= 0.0`, `b <= 0.0`, `x < 0.0`, or `x > 1.0`
78fn checked_beta_reg(a: f64, b: f64, x: f64) -> Option<f64> {
79    if a <= 0. || b <= 0. || !(0.0..=1.).contains(&x) {
80        return None;
81    }
82    let bt = if x == 0. || ulps_eq(x, 1.0, f64::EPSILON, 4) {
83        0.0
84    } else {
85        (ln_gamma(a + b) - ln_gamma(a) - ln_gamma(b) + a * x.ln() + b * (1.0 - x).ln()).exp()
86    };
87    let symm_transform = x >= (a + 1.0) / (a + b + 2.0);
88    let eps = F64_PREC;
89    let fpmin = f64::MIN_POSITIVE / eps;
90
91    let mut a = a;
92    let mut b = b;
93    let mut x = x;
94    if symm_transform {
95        let swap = a;
96        x = 1.0 - x;
97        a = b;
98        b = swap;
99    }
100
101    let qab = a + b;
102    let qap = a + 1.0;
103    let qam = a - 1.0;
104    let mut c = 1.0;
105    let mut d = 1.0 - qab * x / qap;
106
107    if d.abs() < fpmin {
108        d = fpmin;
109    }
110    d = 1.0 / d;
111    let mut h = d;
112
113    for m in 1..141 {
114        let m = f64::from(m);
115        let m2 = m * 2.0;
116        let mut aa = m * (b - m) * x / ((qam + m2) * (a + m2));
117        d = 1.0 + aa * d;
118
119        if d.abs() < fpmin {
120            d = fpmin;
121        }
122
123        c = 1.0 + aa / c;
124        if c.abs() < fpmin {
125            c = fpmin;
126        }
127
128        d = 1.0 / d;
129        h = h * d * c;
130        aa = -(a + m) * (qab + m) * x / ((a + m2) * (qap + m2));
131        d = 1.0 + aa * d;
132
133        if d.abs() < fpmin {
134            d = fpmin;
135        }
136
137        c = 1.0 + aa / c;
138
139        if c.abs() < fpmin {
140            c = fpmin;
141        }
142
143        d = 1.0 / d;
144        let del = d * c;
145        h *= del;
146
147        if (del - 1.0).abs() <= eps {
148            return if symm_transform {
149                Some(1.0 - bt * h / a)
150            } else {
151                Some(bt * h / a)
152            };
153        }
154    }
155
156    if symm_transform {
157        Some(1.0 - bt * h / a)
158    } else {
159        Some(bt * h / a)
160    }
161}

163/// Computes the logarithm of the gamma function
164/// with an accuracy of 16 floating point digits.
165/// The implementation is derived from
166/// "An Analysis of the Lanczos Gamma Approximation",
167/// Glendon Ralph Pugh, 2004 p. 116
168fn ln_gamma(x: f64) -> f64 {
169    if x < 0.5 {
170        let s = GAMMA_DK
171            .iter()
172            .enumerate()
173            .skip(1)
174            .fold(GAMMA_DK[0], |s, t| s + t.1 / (t.0 as f64 - x));
175
176        LN_PI
177            - (f64::consts::PI * x).sin().ln()
178            - s.ln()
179            - LN_2_SQRT_E_OVER_PI
180            - (0.5 - x) * ((0.5 - x + GAMMA_R) / f64::consts::E).ln()
181    } else {
182        let s = GAMMA_DK
183            .iter()
184            .enumerate()
185            .skip(1)
186            .fold(GAMMA_DK[0], |s, t| s + t.1 / (x + t.0 as f64 - 1.0));
187
188        s.ln() + LN_2_SQRT_E_OVER_PI + (x - 0.5) * ((x - 0.5 + GAMMA_R) / f64::consts::E).ln()
189    }
190}
