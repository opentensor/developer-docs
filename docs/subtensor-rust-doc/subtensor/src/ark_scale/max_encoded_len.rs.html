12/// An orphan rules helper which provides
13/// `impl scale::MaxEncodedLen for ArkScale<T,WIRE>`
14/// 
15/// We suggest wrapper types in preference to this trait becuase
16/// on-chain verifier code becomes substrate aware anyways whenever
17/// you choose curves, via a conditional ark-substrate dependence. 
18/// Also wrappers can improve documentation and errors.
19pub trait ArkScaleMaxEncodedLen {
20    /// Upper bound, in bytes, of the maximum encoded size of this item.
21    fn max_encoded_len(compress: Compress) -> usize;
22}
23
24impl ArkScaleMaxEncodedLen for () {
25    #[inline]
26    fn max_encoded_len(_: Compress) -> usize { 0 }
27}
28
29impl<T, const U: Usage> MaxEncodedLen for ArkScale<T, U> 
30    where T: CanonicalSerialize+ArkScaleMaxEncodedLen,
31{
32    #[inline]
33    fn max_encoded_len() -> usize {
34        <T as ArkScaleMaxEncodedLen>::max_encoded_len(is_compressed(U))
35    }
36}
37
38impl<'a, T, const U: Usage> MaxEncodedLen for ArkScaleRef<'a, T, U> 
39    where T: CanonicalSerialize+ArkScaleMaxEncodedLen,
40{
41    #[inline]
42    fn max_encoded_len() -> usize {
43        <T as ArkScaleMaxEncodedLen>::max_encoded_len(is_compressed(U))
44    }
45}

63/// Arkworks type wrapped for serialization by Scale
64#[derive(Clone, Eq, PartialEq, Debug)] // CanonicalSerialize, CanonicalDeserialize
65#[repr(transparent)]
66pub struct ArkScaleLen<T, const L: usize>(pub T);
67
68impl<T, const L: usize> From<T> for ArkScaleLen<T,L> {
69    fn from(t: T) -> ArkScaleLen<T,L> {
70        ArkScaleLen(t)
71    }
72}
73
74impl<T: CanonicalDeserialize, const L: usize> Decode for ArkScaleLen<T, L> {
75    fn decode<I: Input>(input: &mut I) -> Result<Self, scale::Error> {
76        <T as CanonicalDeserialize>::deserialize_with_mode(
77            crate::InputAsRead(input),
78            Compress::Yes,
79            Validate::Yes,
80        )
81        .map(|v| ArkScaleLen(v))
82        .map_err(crate::ark_error_to_scale_error)
83    }

92impl<T: CanonicalSerialize, const L: usize> Encode for ArkScaleLen<T, L> {
93    fn size_hint(&self) -> usize {
94        self.encoded_size()
95    }
96
97    fn encode_to<O: Output + ?Sized>(&self, dest: &mut O) {
98        self.0
99            .serialize_with_mode(crate::OutputAsWrite(dest), Compress::Yes)
100            .expect(crate::OOPS);
101    }
102
103    // TODO:  Arkworks wants an io::Write, so we ignre the rule that
104    // value types override using_encoded.
105    // fn using_encoded<R, F: FnOnce(&[u8]) -> R>(&self, f: F) -> R;
106
107    fn encoded_size(&self) -> usize {
108        let l = self.0.serialized_size(Compress::Yes);
109        debug_assert!(l <= L, "ArkScaleLen has incorrect length specified.");
110        l
111    }
112}
