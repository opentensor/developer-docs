22/// Hooks for *BLS12-381* curve.
23pub trait CurveHooks: 'static + Sized {
24    /// Pairing multi Miller loop.
25    fn bls12_381_multi_miller_loop(
26        g1: impl Iterator<Item = <Bls12_381<Self> as Pairing>::G1Prepared>,
27        g2: impl Iterator<Item = <Bls12_381<Self> as Pairing>::G2Prepared>,
28    ) -> Result<<Bls12_381<Self> as Pairing>::TargetField, ()>;
29
30    /// Pairing final exponentiation.
31    fn bls12_381_final_exponentiation(
32        target: <Bls12_381<Self> as Pairing>::TargetField,
33    ) -> Result<<Bls12_381<Self> as Pairing>::TargetField, ()>;
34
35    /// Multi scalar multiplication on G1.
36    fn bls12_381_msm_g1(
37        bases: &[g1::G1Affine<Self>],
38        scalars: &[<g1::Config<Self> as CurveConfig>::ScalarField],
39    ) -> Result<G1Projective<Self>, ()>;
40
41    /// Multi scalar multiplication on G2.
42    fn bls12_381_msm_g2(
43        bases: &[g2::G2Affine<Self>],
44        scalars: &[<g2::Config<Self> as CurveConfig>::ScalarField],
45    ) -> Result<G2Projective<Self>, ()>;
46
47    /// Projective multiplication on G1.
48    fn bls12_381_mul_projective_g1(
49        base: &G1Projective<Self>,
50        scalar: &[u64],
51    ) -> Result<G1Projective<Self>, ()>;
52
53    /// Projective multiplication on G2.
54    fn bls12_381_mul_projective_g2(
55        base: &G2Projective<Self>,
56        scalar: &[u64],
57    ) -> Result<G2Projective<Self>, ()>;
58}
59
60#[derive(Clone, Copy)]
61pub struct Config<H: CurveHooks>(PhantomData<fn() -> H>);
62
63pub type Bls12_381<H> = Bls12<Config<H>>;
64
65impl<H: CurveHooks> Bls12Config for Config<H> {
66    const X: &'static [u64] = <ArkConfig as ArkBls12Config>::X;
67    const X_IS_NEGATIVE: bool = <ArkConfig as ArkBls12Config>::X_IS_NEGATIVE;
68    const TWIST_TYPE: TwistType = <ArkConfig as ArkBls12Config>::TWIST_TYPE;
69
70    type Fp = <ArkConfig as ArkBls12Config>::Fp;
71    type Fp2Config = <ArkConfig as ArkBls12Config>::Fp2Config;
72    type Fp6Config = <ArkConfig as ArkBls12Config>::Fp6Config;
73    type Fp12Config = <ArkConfig as ArkBls12Config>::Fp12Config;
74
75    type G1Config = g1::Config<H>;
76    type G2Config = g2::Config<H>;
77
78    /// Multi Miller loop jumping into the user-defined `multi_miller_loop` hook.
79    ///
80    /// For any internal error returns `TargetField::zero()`.
81    #[inline(always)]
82    fn multi_miller_loop(
83        g1: impl IntoIterator<Item = impl Into<G1Prepared<Self>>>,
84        g2: impl IntoIterator<Item = impl Into<G2Prepared<Self>>>,
85    ) -> MillerLoopOutput<Bls12<Self>> {
86        let g1 = g1.into_iter().map(|item| item.into());
87        let g2 = g2.into_iter().map(|item| item.into());
88        let res = H::bls12_381_multi_miller_loop(g1, g2);
89        MillerLoopOutput(res.unwrap_or_default())
90    }
91
92    /// Final exponentiation jumping into the user-defined `final_exponentiation` hook.
93    ///
94    /// For any internal error returns `None`.
95    #[inline(always)]
96    fn final_exponentiation(
97        target: MillerLoopOutput<Bls12<Self>>,
98    ) -> Option<PairingOutput<Bls12<Self>>> {
99        let res = H::bls12_381_final_exponentiation(target.0);
100        res.map(PairingOutput).ok()
101    }
