40impl<H: CurveHooks> SWCurveConfig for Config<H> {
41    const COEFF_A: Self::BaseField = <ArkConfig as SWCurveConfig>::COEFF_A;
42    const COEFF_B: Self::BaseField = <ArkConfig as SWCurveConfig>::COEFF_B;

44    const GENERATOR: G1Affine<H> = G1Affine::<H>::new_unchecked(G1_GENERATOR_X, G1_GENERATOR_Y);

46    /// Multi scalar multiplication jumping into the user-defined `msm_g1` hook.
47    ///
48    /// On any internal error returns `Err(0)`.
49    #[inline(always)]
50    fn msm(bases: &[G1Affine<H>], scalars: &[Self::ScalarField]) -> Result<G1Projective<H>, usize> {
51        if bases.len() != scalars.len() {
52            return Err(bases.len().min(scalars.len()));
53        }
54        H::bls12_381_msm_g1(bases, scalars).map_err(|_| 0)
55    }
56
57    /// Projective multiplication jumping into the user-defined `mul_projective` hook.
58    ///
59    /// On any internal error returns `Projective::zero()`.
60    #[inline(always)]
61    fn mul_projective(base: &G1Projective<H>, scalar: &[u64]) -> G1Projective<H> {
62        H::bls12_381_mul_projective_g1(base, scalar).unwrap_or_default()
63    }
64
65    /// Affine multiplication jumping into the user-defined `mul_projective` hook.
66    ///
67    /// On any internal error returns `Projective::zero()`.
68    #[inline(always)]
69    fn mul_affine(base: &G1Affine<H>, scalar: &[u64]) -> G1Projective<H> {
70        Self::mul_projective(&(*base).into(), scalar)
71    }

78    // Verbatim copy of upstream implementation.
79    //
80    // Can't call it directly because of different `Affine` configuration.
81    #[inline(always)]
82    fn is_in_correct_subgroup_assuming_on_curve(p: &G1Affine<H>) -> bool {
83        let x_times_p = p.mul_bigint(crate::Config::<H>::X);
84        if x_times_p.eq(p) && !p.infinity {
85            return false;
86        }
87
88        let minus_x_squared_times_p = x_times_p.mul_bigint(crate::Config::<H>::X).neg();
89        let endomorphism_p = endomorphism(p);
90        minus_x_squared_times_p.eq(&endomorphism_p)
91    }

93    // Verbatim copy of upstream implementation.
94    //
95    // Can't call it directly because of different `Affine` configuration.
96    #[inline(always)]
97    fn clear_cofactor(p: &G1Affine<H>) -> G1Affine<H> {
98        let h_eff =
99            one_minus_x(crate::Config::<H>::X_IS_NEGATIVE, crate::Config::<H>::X).into_bigint();
100        Self::mul_affine(p, h_eff.as_ref()).into()
101    }

103    #[inline(always)]

105        <ArkConfig as SWCurveConfig>::serialized_size(compress)
106    }

108    // Verbatim copy of upstream implementation.
109    //
110    // Can't call it directly because of different `Affine` configuration.
111    fn serialize_with_mode<W: Write>(
112        item: &G1Affine<H>,
113        mut writer: W,
114        compress: Compress,
115    ) -> Result<(), SerializationError> {
116        let encoding = EncodingFlags {
117            is_compressed: compress == Compress::Yes,
118            is_infinity: item.is_zero(),
119            is_lexographically_largest: item.y > -item.y,
120        };
121        let mut p = *item;
122        if encoding.is_infinity {
123            p = Affine::<Self>::zero();
124        }
125        // need to access the field struct `x` directly, otherwise we get None from xy()
126        // method
127        let x_bytes = serialize_fq(p.x);
128        if encoding.is_compressed {
129            let mut bytes: [u8; G1_SERIALIZED_SIZE] = x_bytes;
130
131            encoding.encode_flags(&mut bytes);
132            writer.write_all(&bytes)?;
133        } else {
134            let mut bytes = [0u8; 2 * G1_SERIALIZED_SIZE];
135            bytes[0..G1_SERIALIZED_SIZE].copy_from_slice(&x_bytes[..]);
136            bytes[G1_SERIALIZED_SIZE..].copy_from_slice(&serialize_fq(p.y)[..]);
137
138            encoding.encode_flags(&mut bytes);
139            writer.write_all(&bytes)?;
140        };
141        Ok(())
142    }
143
144    // Verbatim copy of upstream implementation.
145    //
146    // Can't call it directly because of different `Affine` configuration.
147    fn deserialize_with_mode<R: Read>(
148        mut reader: R,
149        compress: Compress,
150        validate: Validate,
151    ) -> Result<G1Affine<H>, SerializationError> {
152        let p = if compress == Compress::Yes {
153            read_g1_compressed(&mut reader)?
154        } else {
155            read_g1_uncompressed(&mut reader)?
156        };
157
158        if validate == Validate::Yes && !p.is_in_correct_subgroup_assuming_on_curve() {
159            return Err(SerializationError::InvalidData);
160        }
161        Ok(p)
162    }
163}

165fn one_minus_x(
166    x_is_negative: bool,
167    x_value: &'static [u64],
168) -> <ArkConfig as CurveConfig>::ScalarField {
169    let x = <ArkConfig as CurveConfig>::ScalarField::from_sign_and_limbs(!x_is_negative, x_value);
170    <ArkConfig as CurveConfig>::ScalarField::one() - x
171}

173pub fn endomorphism<H: CurveHooks>(p: &G1Affine<H>) -> G1Affine<H> {
174    // Endomorphism of the points on the curve.
175    // endomorphism_p(x,y) = (BETA * x, y)
176    // where BETA is a non-trivial cubic root of unity in fq.
177    let mut res = *p;
178    res.x *= BETA;
179    res
180}
