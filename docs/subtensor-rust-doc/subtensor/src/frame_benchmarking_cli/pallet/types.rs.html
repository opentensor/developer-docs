24/// How the genesis state for benchmarking should be build.
25#[derive(clap::ValueEnum, Debug, Eq, PartialEq, Clone, Copy)]
26#[clap(rename_all = "kebab-case")]
27pub enum GenesisBuilderPolicy {
28	/// Do not provide any genesis state.
29	///
30	/// Benchmarks are advised to function with this, since they should setup their own required
31	/// state. However, to keep backwards compatibility, this is not the default.
32	None,
33	/// Let the runtime build the genesis state through its `BuildGenesisConfig` runtime API.
34	Runtime,
35	// Use the runtime from the Spec file to build the genesis state.
36	SpecRuntime,
37	/// Use the spec file to build the genesis state. This fails when there is no spec.
38	SpecGenesis,
39	/// Same as `SpecGenesis` - only here for backwards compatibility.
40	Spec,
41}

43/// A runtime blob that was either fetched from genesis storage or loaded from a file.
44// NOTE: This enum is only needed for the annoying lifetime bounds on `RuntimeCode`. Otherwise we
45// could just directly return the blob.
46pub struct FetchedCode<'a, B, H> {
47	pub state: sp_state_machine::backend::BackendRuntimeCode<'a, B, H>,
48}
49
50impl<'a, B, H> FetchedCode<'a, B, H>
51where
52	H: Hash,
53	B: sc_client_api::StateBackend<H>,
54{
55	/// The runtime blob.
56	pub fn code(&'a self) -> Result<RuntimeCode<'a>> {
57		self.state.runtime_code().map_err(Into::into)
58	}
59}

61/// Maps a (pallet, benchmark) to its component ranges.
62pub(crate) type ComponentRangeMap =
63	std::collections::HashMap<(String, String), Vec<ComponentRange>>;
64
65/// The inclusive range of a component.
66#[derive(serde::Serialize, Debug, Clone, Eq, PartialEq)]
67pub(crate) struct ComponentRange {
68	/// Name of the component.
69	pub(crate) name: String,
70	/// Minimal valid value of the component.
71	pub(crate) min: u32,
72	/// Maximal valid value of the component.
73	pub(crate) max: u32,
74}
