42/// Benchmark the execution overhead per-block and per-extrinsic.
43#[derive(Debug, Parser)]
44pub struct OverheadCmd {
45	#[allow(missing_docs)]
46	#[clap(flatten)]
47	pub shared_params: SharedParams,
48
49	#[allow(missing_docs)]
50	#[clap(flatten)]
51	pub import_params: ImportParams,
52
53	#[allow(missing_docs)]
54	#[clap(flatten)]
55	pub params: OverheadParams,
56}

58/// Configures the benchmark, the post-processing and weight generation.
59#[derive(Debug, Default, Serialize, Clone, PartialEq, Args)]
60pub struct OverheadParams {
61	#[allow(missing_docs)]
62	#[clap(flatten)]
63	pub weight: WeightParams,
64
65	#[allow(missing_docs)]
66	#[clap(flatten)]
67	pub bench: ExtrinsicBenchmarkParams,
68
69	#[allow(missing_docs)]
70	#[clap(flatten)]
71	pub hostinfo: HostInfoParams,
72
73	/// Add a header to the generated weight output file.
74	///
75	/// Good for adding LICENSE headers.
76	#[arg(long, value_name = "PATH")]
77	pub header: Option<PathBuf>,
78
79	/// Enable the Trie cache.
80	///
81	/// This should only be used for performance analysis and not for final results.
82	#[arg(long)]
83	pub enable_trie_cache: bool,
84}

86/// Type of a benchmark.
87#[derive(Serialize, Clone, PartialEq, Copy)]
88pub(crate) enum BenchmarkType {
89	/// Measure the per-extrinsic execution overhead.
90	Extrinsic,
91	/// Measure the per-block execution overhead.
92	Block,
93}
94
95impl OverheadCmd {
96	/// Measure the per-block and per-extrinsic execution overhead.
97	///
98	/// Writes the results to console and into two instances of the
99	/// `weights.hbs` template, one for each benchmark.
100	pub fn run<Block, C>(
101		&self,
102		cfg: Configuration,
103		client: Arc<C>,
104		inherent_data: sp_inherents::InherentData,
105		digest_items: Vec<DigestItem>,
106		ext_builder: &dyn ExtrinsicBuilder,
107	) -> Result<()>
108	where
109		Block: BlockT<Extrinsic = OpaqueExtrinsic>,
110		C: ProvideRuntimeApi<Block>
111			+ CallApiAt<Block>
112			+ UsageProvider<Block>
113			+ sp_blockchain::HeaderBackend<Block>,
114		C::Api: ApiExt<Block> + BlockBuilderApi<Block>,
115	{
116		if ext_builder.pallet() != "system" || ext_builder.extrinsic() != "remark" {
117			return Err(format!("The extrinsic builder is required to build `System::Remark` extrinsics but builds `{}` extrinsics instead", ext_builder.name()).into());
118		}
119		let bench = Benchmark::new(client, self.params.bench.clone(), inherent_data, digest_items);
120
121		// per-block execution overhead
122		{
123			let stats = bench.bench_block()?;
124			info!("Per-block execution overhead [ns]:\n{:?}", stats);
125			let template = TemplateData::new(BenchmarkType::Block, &cfg, &self.params, &stats)?;
126			template.write(&self.params.weight.weight_path)?;
127		}
128		// per-extrinsic execution overhead
129		{
130			let stats = bench.bench_extrinsic(ext_builder)?;
131			info!("Per-extrinsic execution overhead [ns]:\n{:?}", stats);
132			let template = TemplateData::new(BenchmarkType::Extrinsic, &cfg, &self.params, &stats)?;
133			template.write(&self.params.weight.weight_path)?;
134		}
135
136		Ok(())
137	}
138}
139
140impl BenchmarkType {
141	/// Short name of the benchmark type.
142	pub(crate) fn short_name(&self) -> &'static str {
143		match self {
144			Self::Extrinsic => "extrinsic",
145			Self::Block => "block",
146		}
147	}
148
149	/// Long name of the benchmark type.
150	pub(crate) fn long_name(&self) -> &'static str {
151		match self {
152			Self::Extrinsic => "ExtrinsicBase",
153			Self::Block => "BlockExecution",
154		}
155	}
156}

158// Boilerplate
159impl CliConfiguration for OverheadCmd {
160	fn shared_params(&self) -> &SharedParams {
161		&self.shared_params
162	}
163
164	fn import_params(&self) -> Option<&ImportParams> {
165		Some(&self.import_params)
166	}
167
168	fn trie_cache_maximum_size(&self) -> Result<Option<usize>> {
169		if self.params.enable_trie_cache {
170			Ok(self.import_params().map(|x| x.trie_cache_maximum_size()).unwrap_or_default())
171		} else {
172			Ok(None)
173		}
174	}
