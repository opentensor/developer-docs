18use sc_cli::Result;
19use sc_client_api::{Backend as ClientBackend, StorageProvider, UsageProvider};
20use sp_runtime::traits::{Block as BlockT, Header as HeaderT};
21
22use log::info;
23use rand::prelude::*;
24use std::{fmt::Debug, sync::Arc, time::Instant};
25
26use super::cmd::StorageCmd;
27use crate::shared::{new_rng, BenchRecord};
28
29impl StorageCmd {
30	/// Benchmarks the time it takes to read a single Storage item.
31	/// Uses the latest state that is available for the given client.
32	pub(crate) fn bench_read<B, BA, C>(&self, client: Arc<C>) -> Result<BenchRecord>
33	where
34		C: UsageProvider<B> + StorageProvider<B, BA>,
35		B: BlockT + Debug,
36		BA: ClientBackend<B>,
37		<<B as BlockT>::Header as HeaderT>::Number: From<u32>,
38	{
39		let mut record = BenchRecord::default();
40		let best_hash = client.usage_info().chain.best_hash;
41
42		info!("Preparing keys from block {}", best_hash);
43		// Load all keys and randomly shuffle them.
44		let mut keys: Vec<_> = client.storage_keys(best_hash, None, None)?.collect();
45		let (mut rng, _) = new_rng(None);
46		keys.shuffle(&mut rng);
47
48		let mut child_nodes = Vec::new();
49		// Interesting part here:
50		// Read all the keys in the database and measure the time it takes to access each.
51		info!("Reading {} keys", keys.len());
52		for key in keys.as_slice() {
53			match (self.params.include_child_trees, self.is_child_key(key.clone().0)) {
54				(true, Some(info)) => {
55					// child tree key
56					for ck in client.child_storage_keys(best_hash, info.clone(), None, None)? {
57						child_nodes.push((ck.clone(), info.clone()));
58					}
59				},
60				_ => {
61					// regular key
62					let start = Instant::now();
63					let v = client
64						.storage(best_hash, &key)
65						.expect("Checked above to exist")
66						.ok_or("Value unexpectedly empty")?;
67					record.append(v.0.len(), start.elapsed())?;
68				},
69			}
70		}
71
72		if self.params.include_child_trees {
73			child_nodes.shuffle(&mut rng);
74
75			info!("Reading {} child keys", child_nodes.len());
76			for (key, info) in child_nodes.as_slice() {
77				let start = Instant::now();
78				let v = client
79					.child_storage(best_hash, info, key)
80					.expect("Checked above to exist")
81					.ok_or("Value unexpectedly empty")?;
82				record.append(v.0.len(), start.elapsed())?;
83			}
84		}
85		Ok(record)
86	}
87}
