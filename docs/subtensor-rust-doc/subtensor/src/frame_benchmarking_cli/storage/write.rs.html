39impl StorageCmd {
40	/// Benchmarks the time it takes to write a single Storage item.
41	/// Uses the latest state that is available for the given client.
42	pub(crate) fn bench_write<Block, BA, H, C>(
43		&self,
44		client: Arc<C>,
45		(db, state_col): (Arc<dyn sp_database::Database<DbHash>>, ColumnId),
46		storage: Arc<dyn sp_state_machine::Storage<HashingFor<Block>>>,
47	) -> Result<BenchRecord>
48	where
49		Block: BlockT<Header = H, Hash = DbHash> + Debug,
50		H: HeaderT<Hash = DbHash>,
51		BA: ClientBackend<Block>,
52		C: UsageProvider<Block> + HeaderBackend<Block> + StorageProvider<Block, BA>,
53	{
54		// Store the time that it took to write each value.
55		let mut record = BenchRecord::default();
56
57		let best_hash = client.usage_info().chain.best_hash;
58		let header = client.header(best_hash)?.ok_or("Header not found")?;
59		let original_root = *header.state_root();
60		let trie = DbStateBuilder::<HashingFor<Block>>::new(storage.clone(), original_root).build();
61
62		info!("Preparing keys from block {}", best_hash);
63		// Load all KV pairs and randomly shuffle them.
64		let mut kvs: Vec<_> = trie.pairs(Default::default())?.collect();
65		let (mut rng, _) = new_rng(None);
66		kvs.shuffle(&mut rng);
67		info!("Writing {} keys", kvs.len());
68
69		let mut child_nodes = Vec::new();
70
71		// Generate all random values first; Make sure there are no collisions with existing
72		// db entries, so we can rollback all additions without corrupting existing entries.
73		for key_value in kvs {
74			let (k, original_v) = key_value?;
75			match (self.params.include_child_trees, self.is_child_key(k.to_vec())) {
76				(true, Some(info)) => {
77					let child_keys =
78						client.child_storage_keys(best_hash, info.clone(), None, None)?;
79					for ck in child_keys {
80						child_nodes.push((ck.clone(), info.clone()));
81					}
82				},
83				_ => {
84					// regular key
85					let mut new_v = vec![0; original_v.len()];
86					loop {
87						// Create a random value to overwrite with.
88						// NOTE: We use a possibly higher entropy than the original value,
89						// could be improved but acts as an over-estimation which is fine for now.
90						rng.fill_bytes(&mut new_v[..]);
91						if check_new_value::<Block>(
92							db.clone(),
93							&trie,
94							&k.to_vec(),
95							&new_v,
96							self.state_version(),
97							state_col,
98							None,
99						) {
100							break
101						}
102					}
103
104					// Write each value in one commit.
105					let (size, duration) = measure_write::<Block>(
106						db.clone(),
107						&trie,
108						k.to_vec(),
109						new_v.to_vec(),
110						self.state_version(),
111						state_col,
112						None,
113					)?;
114					record.append(size, duration)?;
115				},
116			}
117		}
118
119		if self.params.include_child_trees {
120			child_nodes.shuffle(&mut rng);
121			info!("Writing {} child keys", child_nodes.len());
122
123			for (key, info) in child_nodes {
124				if let Some(original_v) = client
125					.child_storage(best_hash, &info.clone(), &key)
126					.expect("Checked above to exist")
127				{
128					let mut new_v = vec![0; original_v.0.len()];
129					loop {
130						rng.fill_bytes(&mut new_v[..]);
131						if check_new_value::<Block>(
132							db.clone(),
133							&trie,
134							&key.0,
135							&new_v,
136							self.state_version(),
137							state_col,
138							Some(&info),
139						) {
140							break
141						}
142					}
143
144					let (size, duration) = measure_write::<Block>(
145						db.clone(),
146						&trie,
147						key.0,
148						new_v.to_vec(),
149						self.state_version(),
150						state_col,
151						Some(&info),
152					)?;
153					record.append(size, duration)?;
154				}
155			}
156		}
157
158		Ok(record)
159	}
160}

162/// Converts a Trie transaction into a DB transaction.
163/// Removals are ignored and will not be included in the final tx.
164/// `invert_inserts` replaces all inserts with removals.
165fn convert_tx<B: BlockT>(
166	db: Arc<dyn sp_database::Database<DbHash>>,
167	mut tx: PrefixedMemoryDB<HashingFor<B>>,
168	invert_inserts: bool,
169	col: ColumnId,
170) -> Transaction<DbHash> {
171	let mut ret = Transaction::<DbHash>::default();
172
173	for (mut k, (v, rc)) in tx.drain().into_iter() {
174		if rc > 0 {
175			db.sanitize_key(&mut k);
176			if invert_inserts {
177				ret.remove(col, &k);
178			} else {
179				ret.set(col, &k, &v);
180			}
181		}
182		// < 0 means removal - ignored.
183		// 0 means no modification.
184	}
185	ret
186}

188/// Measures write benchmark
189/// if `child_info` exist then it means this is a child tree key
190fn measure_write<Block: BlockT>(
191	db: Arc<dyn sp_database::Database<DbHash>>,
192	trie: &DbState<HashingFor<Block>>,
193	key: Vec<u8>,
194	new_v: Vec<u8>,
195	version: StateVersion,
196	col: ColumnId,
197	child_info: Option<&ChildInfo>,
198) -> Result<(usize, Duration)> {
199	let start = Instant::now();
200	// Create a TX that will modify the Trie in the DB and
201	// calculate the root hash of the Trie after the modification.
202	let replace = vec![(key.as_ref(), Some(new_v.as_ref()))];
203	let stx = match child_info {
204		Some(info) => trie.child_storage_root(info, replace.iter().cloned(), version).2,
205		None => trie.storage_root(replace.iter().cloned(), version).1,
206	};
207	// Only the keep the insertions, since we do not want to benchmark pruning.
208	let tx = convert_tx::<Block>(db.clone(), stx.clone(), false, col);
209	db.commit(tx).map_err(|e| format!("Writing to the Database: {}", e))?;
210	let result = (new_v.len(), start.elapsed());
211
212	// Now undo the changes by removing what was added.
213	let tx = convert_tx::<Block>(db.clone(), stx.clone(), true, col);
214	db.commit(tx).map_err(|e| format!("Writing to the Database: {}", e))?;
215	Ok(result)
216}

218/// Checks if a new value causes any collision in tree updates
219/// returns true if there is no collision
220/// if `child_info` exist then it means this is a child tree key
221fn check_new_value<Block: BlockT>(
222	db: Arc<dyn sp_database::Database<DbHash>>,
223	trie: &DbState<HashingFor<Block>>,
224	key: &Vec<u8>,
225	new_v: &Vec<u8>,
226	version: StateVersion,
227	col: ColumnId,
228	child_info: Option<&ChildInfo>,
229) -> bool {
230	let new_kv = vec![(key.as_ref(), Some(new_v.as_ref()))];
231	let mut stx = match child_info {
232		Some(info) => trie.child_storage_root(info, new_kv.iter().cloned(), version).2,
233		None => trie.storage_root(new_kv.iter().cloned(), version).1,
234	};
235	for (mut k, (_, rc)) in stx.drain().into_iter() {
236		if rc > 0 {
237			db.sanitize_key(&mut k);
238			if db.get(col, &k).is_some() {
239				trace!("Benchmark-store key creation: Key collision detected, retry");
240				return false
241			}
242		}
243	}
244	true
