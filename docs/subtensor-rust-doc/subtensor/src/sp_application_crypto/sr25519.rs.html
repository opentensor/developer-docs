18//! Sr25519 crypto types.
19
20use crate::{KeyTypeId, RuntimePublic};
21
22use alloc::vec::Vec;
23
24pub use sp_core::sr25519::*;
25
26mod app {
27	crate::app_crypto!(super, sp_core::testing::SR25519);
28}
29
30pub use app::{Pair as AppPair, Public as AppPublic, Signature as AppSignature};
31
32impl RuntimePublic for Public {
33	type Signature = Signature;
34
35	fn all(key_type: KeyTypeId) -> crate::Vec<Self> {
36		sp_io::crypto::sr25519_public_keys(key_type)
37	}
38
39	fn generate_pair(key_type: KeyTypeId, seed: Option<Vec<u8>>) -> Self {
40		sp_io::crypto::sr25519_generate(key_type, seed)
41	}
42
43	fn sign<M: AsRef<[u8]>>(&self, key_type: KeyTypeId, msg: &M) -> Option<Self::Signature> {
44		sp_io::crypto::sr25519_sign(key_type, self, msg.as_ref())
45	}
46
47	fn verify<M: AsRef<[u8]>>(&self, msg: &M, signature: &Self::Signature) -> bool {
48		sp_io::crypto::sr25519_verify(signature, msg.as_ref(), self)
49	}
50
51	fn to_raw_vec(&self) -> Vec<u8> {
52		sp_core::crypto::ByteArray::to_raw_vec(self)
53	}
54}
