45	/// Peer response data does not have requested bits.
46	pub const BAD_RESPONSE: Rep = Rep::new(-(1 << 12), "Incomplete response");

48	/// Reputation change for peers which send us a known bad state.
49	pub const BAD_STATE: Rep = Rep::new(-(1 << 29), "Bad state");

52/// Action that should be performed on [`StateStrategy`]'s behalf.
53pub enum StateStrategyAction<B: BlockT> {
54	/// Send state request to peer.
55	SendStateRequest { peer_id: PeerId, request: OpaqueStateRequest },
56	/// Disconnect and report peer.
57	DropPeer(BadPeer),
58	/// Import blocks.
59	ImportBlocks { origin: BlockOrigin, blocks: Vec<IncomingBlock<B>> },
60	/// State sync has finished.
61	Finished,
62}
63
64enum PeerState {
65	Available,
66	DownloadingState,
67}
68
69impl PeerState {
70	fn is_available(&self) -> bool {
71		matches!(self, PeerState::Available)
72	}
73}
74
75struct Peer<B: BlockT> {
76	best_number: NumberFor<B>,
77	state: PeerState,
78}

80/// Syncing strategy that downloads and imports a recent state directly.
81pub struct StateStrategy<B: BlockT> {
82	state_sync: Box<dyn StateSyncProvider<B>>,
83	peers: HashMap<PeerId, Peer<B>>,
84	disconnected_peers: DisconnectedPeers,
85	actions: Vec<StateStrategyAction<B>>,
86	succeeded: bool,
87}
88
89impl<B: BlockT> StateStrategy<B> {
90	/// Create a new instance.
91	pub fn new<Client>(
92		client: Arc<Client>,
93		target_header: B::Header,
94		target_body: Option<Vec<B::Extrinsic>>,
95		target_justifications: Option<Justifications>,
96		skip_proof: bool,
97		initial_peers: impl Iterator<Item = (PeerId, NumberFor<B>)>,
98	) -> Self
99	where
100		Client: ProofProvider<B> + Send + Sync + 'static,
101	{
102		let peers = initial_peers
103			.map(|(peer_id, best_number)| {
104				(peer_id, Peer { best_number, state: PeerState::Available })
105			})
106			.collect();
107		Self {
108			state_sync: Box::new(StateSync::new(
109				client,
110				target_header,
111				target_body,
112				target_justifications,
113				skip_proof,
114			)),
115			peers,
116			disconnected_peers: DisconnectedPeers::new(),
117			actions: Vec::new(),
118			succeeded: false,
119		}
120	}

142	/// Notify that a new peer has connected.
143	pub fn add_peer(&mut self, peer_id: PeerId, _best_hash: B::Hash, best_number: NumberFor<B>) {
144		self.peers.insert(peer_id, Peer { best_number, state: PeerState::Available });
145	}

147	/// Notify that a peer has disconnected.
148	pub fn remove_peer(&mut self, peer_id: &PeerId) {
149		if let Some(state) = self.peers.remove(peer_id) {
150			if !state.state.is_available() {
151				if let Some(bad_peer) =
152					self.disconnected_peers.on_disconnect_during_request(*peer_id)
153				{
154					self.actions.push(StateStrategyAction::DropPeer(bad_peer));
155				}
156			}
157		}
158	}

160	/// Submit a validated block announcement.
161	///
162	/// Returns new best hash & best number of the peer if they are updated.
163	#[must_use]
164	pub fn on_validated_block_announce(
165		&mut self,
166		is_best: bool,
167		peer_id: PeerId,
168		announce: &BlockAnnounce<B::Header>,
169	) -> Option<(B::Hash, NumberFor<B>)> {
170		is_best.then_some({
171			let best_number = *announce.header.number();
172			let best_hash = announce.header.hash();
173			if let Some(ref mut peer) = self.peers.get_mut(&peer_id) {
174				peer.best_number = best_number;
175			}
176			// Let `SyncingEngine` know that we should update the peer info.
177			(best_hash, best_number)
178		})
179	}

181	/// Process state response.
182	pub fn on_state_response(&mut self, peer_id: PeerId, response: OpaqueStateResponse) {
183		if let Err(bad_peer) = self.on_state_response_inner(peer_id, response) {
184			self.actions.push(StateStrategyAction::DropPeer(bad_peer));
185		}
186	}
187
188	fn on_state_response_inner(
189		&mut self,
190		peer_id: PeerId,
191		response: OpaqueStateResponse,
192	) -> Result<(), BadPeer> {
193		if let Some(peer) = self.peers.get_mut(&peer_id) {
194			peer.state = PeerState::Available;
195		}
196
197		let response: Box<StateResponse> = response.0.downcast().map_err(|_error| {
198			error!(
199				target: LOG_TARGET,
200				"Failed to downcast opaque state response, this is an implementation bug."
201			);
202			debug_assert!(false);
203
204			BadPeer(peer_id, rep::BAD_RESPONSE)
205		})?;
206
207		debug!(
208			target: LOG_TARGET,
209			"Importing state data from {} with {} keys, {} proof nodes.",
210			peer_id,
211			response.entries.len(),
212			response.proof.len(),
213		);
214
215		match self.state_sync.import(*response) {
216			ImportResult::Import(hash, header, state, body, justifications) => {
217				let origin = BlockOrigin::NetworkInitialSync;
218				let block = IncomingBlock {
219					hash,
220					header: Some(header),
221					body,
222					indexed_body: None,
223					justifications,
224					origin: None,
225					allow_missing_state: true,
226					import_existing: true,
227					skip_execution: true,
228					state: Some(state),
229				};
230				debug!(target: LOG_TARGET, "State download is complete. Import is queued");
231				self.actions
232					.push(StateStrategyAction::ImportBlocks { origin, blocks: vec![block] });
233				Ok(())
234			},
235			ImportResult::Continue => Ok(()),
236			ImportResult::BadResponse => {
237				debug!(target: LOG_TARGET, "Bad state data received from {peer_id}");
238				Err(BadPeer(peer_id, rep::BAD_STATE))
239			},
240		}
241	}

243	/// A batch of blocks have been processed, with or without errors.
244	///
245	/// Normally this should be called when target block with state is imported.
246	pub fn on_blocks_processed(
247		&mut self,
248		imported: usize,
249		count: usize,
250		results: Vec<(Result<BlockImportStatus<NumberFor<B>>, BlockImportError>, B::Hash)>,
251	) {
252		trace!(target: LOG_TARGET, "State sync: imported {imported} of {count}.");
253
254		let results = results
255			.into_iter()
256			.filter_map(|(result, hash)| {
257				if hash == self.state_sync.target_hash() {
258					Some(result)
259				} else {
260					debug!(
261						target: LOG_TARGET,
262						"Unexpected block processed: {hash} with result {result:?}.",
263					);
264					None
265				}
266			})
267			.collect::<Vec<_>>();
268
269		if !results.is_empty() {
270			// We processed the target block
271			results.iter().filter_map(|result| result.as_ref().err()).for_each(|e| {
272				error!(
273					target: LOG_TARGET,
274					"Failed to import target block with state: {e:?}."
275				);
276			});
277			self.succeeded |= results.into_iter().any(|result| result.is_ok());
278			self.actions.push(StateStrategyAction::Finished);
279		}
280	}

282	/// Produce state request.
283	fn state_request(&mut self) -> Option<(PeerId, OpaqueStateRequest)> {
284		if self.state_sync.is_complete() {
285			return None
286		}
287
288		if self
289			.peers
290			.values()
291			.any(|peer| matches!(peer.state, PeerState::DownloadingState))
292		{
293			// Only one state request at a time is possible.
294			return None
295		}
296
297		let peer_id =
298			self.schedule_next_peer(PeerState::DownloadingState, self.state_sync.target_number())?;
299		let request = self.state_sync.next_request();
300		trace!(
301			target: LOG_TARGET,
302			"New state request to {peer_id}: {request:?}.",
303		);
304		Some((peer_id, OpaqueStateRequest(Box::new(request))))
305	}
306
307	fn schedule_next_peer(
308		&mut self,
309		new_state: PeerState,
310		min_best_number: NumberFor<B>,
311	) -> Option<PeerId> {
312		let mut targets: Vec<_> = self.peers.values().map(|p| p.best_number).collect();
313		if targets.is_empty() {
314			return None
315		}
316		targets.sort();
317		let median = targets[targets.len() / 2];
318		let threshold = std::cmp::max(median, min_best_number);
319		// Find a random peer that is synced as much as peer majority and is above
320		// `min_best_number`.
321		for (peer_id, peer) in self.peers.iter_mut() {
322			if peer.state.is_available() &&
323				peer.best_number >= threshold &&
324				self.disconnected_peers.is_peer_available(peer_id)
325			{
326				peer.state = new_state;
327				return Some(*peer_id)
328			}
329		}
330		None
331	}

333	/// Returns the current sync status.
334	pub fn status(&self) -> SyncStatus<B> {
335		SyncStatus {
336			state: if self.state_sync.is_complete() {
337				SyncState::Idle
338			} else {
339				SyncState::Downloading { target: self.state_sync.target_number() }
340			},
341			best_seen_block: Some(self.state_sync.target_number()),
342			num_peers: self.peers.len().saturated_into(),
343			queued_blocks: 0,
344			state_sync: Some(self.state_sync.progress()),
345			warp_sync: None,
346		}
347	}

349	/// Get actions that should be performed by the owner on [`WarpSync`]'s behalf
350	#[must_use]
351	pub fn actions(&mut self) -> impl Iterator<Item = StateStrategyAction<B>> {
352		let state_request = self
353			.state_request()
354			.into_iter()
355			.map(|(peer_id, request)| StateStrategyAction::SendStateRequest { peer_id, request });
356		self.actions.extend(state_request);
357
358		std::mem::take(&mut self.actions).into_iter()
359	}

487		let client = Arc::new(TestClientBuilder::new().set_no_genesis().build());
488		let target_block = BlockBuilderBuilder::new(&*client)
489			.on_parent_block(client.chain_info().best_hash)

491			.build()
492			.unwrap()
493			.build()
494			.unwrap()
495			.block;
496
497		let peers = (1..=10)
498			.map(|best_number| (PeerId::random(), best_number))
499			.collect::<Vec<(_, _)>>();
500		let ninth_peer = peers[8].0;
501		let tenth_peer = peers[9].0;
502		let initial_peers = peers.iter().map(|(p, n)| (*p, *n));
503
504		let mut state_strategy = StateStrategy::new(
505			client.clone(),
506			target_block.header().clone(),
507			None,
508			None,
509			false,
510			initial_peers,
511		);
512
513		// Disconnecting a peer without an inflight request has no effect on persistent states.
514		state_strategy.remove_peer(&tenth_peer);
515		assert!(state_strategy.disconnected_peers.is_peer_available(&tenth_peer));
516
517		// Disconnect the peer with an inflight request.
518		state_strategy.add_peer(tenth_peer, H256::random(), 10);
519		let peer_id: Option<PeerId> =
520			state_strategy.schedule_next_peer(PeerState::DownloadingState, 10);
521		assert_eq!(tenth_peer, peer_id.unwrap());
522		state_strategy.remove_peer(&tenth_peer);
523
524		// Peer is backed off.
525		assert!(!state_strategy.disconnected_peers.is_peer_available(&tenth_peer));
526
527		// No peer available for 10'th best block because of the backoff.
528		state_strategy.add_peer(tenth_peer, H256::random(), 10);
529		let peer_id: Option<PeerId> =
530			state_strategy.schedule_next_peer(PeerState::DownloadingState, 10);
531		assert!(peer_id.is_none());
532
533		// Other requests can still happen.
534		let peer_id: Option<PeerId> =
535			state_strategy.schedule_next_peer(PeerState::DownloadingState, 9);
536		assert_eq!(ninth_peer, peer_id.unwrap());
537	}
