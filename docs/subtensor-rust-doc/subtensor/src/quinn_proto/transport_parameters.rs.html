1//! QUIC connection transport parameters
2//!
3//! The `TransportParameters` type is used to represent the transport parameters
4//! negotiated by peers while establishing a QUIC connection. This process
5//! happens as part of the establishment of the TLS session. As such, the types
6//! contained in this modules should generally only be referred to by custom
7//! implementations of the `crypto::Session` trait.
8

27// Apply a given macro to a list of all the transport parameters having integer types, along with
28// their codes and default values. Using this helps us avoid error-prone duplication of the
29// contained information across decoding, encoding, and the `Default` impl. Whenever we want to do
30// something with transport parameters, we'll handle the bulk of cases by writing a macro that
31// takes a list of arguments in this form, then passing it to this macro.
32macro_rules! apply_params {
33    ($macro:ident) => {
34        $macro! {
35            // #[doc] name (id) = default,
36            /// Milliseconds, disabled if zero
37            max_idle_timeout(0x0001) = 0,
38            /// Limits the size of UDP payloads that the endpoint is willing to receive
39            max_udp_payload_size(0x0003) = 65527,
40
41            /// Initial value for the maximum amount of data that can be sent on the connection
42            initial_max_data(0x0004) = 0,
43            /// Initial flow control limit for locally-initiated bidirectional streams
44            initial_max_stream_data_bidi_local(0x0005) = 0,
45            /// Initial flow control limit for peer-initiated bidirectional streams
46            initial_max_stream_data_bidi_remote(0x0006) = 0,
47            /// Initial flow control limit for unidirectional streams
48            initial_max_stream_data_uni(0x0007) = 0,
49
50            /// Initial maximum number of bidirectional streams the peer may initiate
51            initial_max_streams_bidi(0x0008) = 0,
52            /// Initial maximum number of unidirectional streams the peer may initiate
53            initial_max_streams_uni(0x0009) = 0,
54
55            /// Exponent used to decode the ACK Delay field in the ACK frame
56            ack_delay_exponent(0x000a) = 3,
57            /// Maximum amount of time in milliseconds by which the endpoint will delay sending
58            /// acknowledgments
59            max_ack_delay(0x000b) = 25,
60            /// Maximum number of connection IDs from the peer that an endpoint is willing to store
61            active_connection_id_limit(0x000e) = 2,
62        }
63    };
64}

66macro_rules! make_struct {
67    {$($(#[$doc:meta])* $name:ident ($code:expr) = $default:expr,)*} => {
68        /// Transport parameters used to negotiate connection-level preferences between peers
69        #[derive(Debug, Copy, Clone, Eq, PartialEq)]
70        pub struct TransportParameters {
71            $($(#[$doc])* pub(crate) $name : VarInt,)*
72
73            /// Does the endpoint support active connection migration
74            pub(crate) disable_active_migration: bool,
75            /// Maximum size for datagram frames
76            pub(crate) max_datagram_frame_size: Option<VarInt>,
77            /// The value that the endpoint included in the Source Connection ID field of the first
78            /// Initial packet it sends for the connection
79            pub(crate) initial_src_cid: Option<ConnectionId>,
80            /// The endpoint is willing to receive QUIC packets containing any value for the fixed
81            /// bit
82            pub(crate) grease_quic_bit: bool,
83
84            // Server-only
85            /// The value of the Destination Connection ID field from the first Initial packet sent
86            /// by the client
87            pub(crate) original_dst_cid: Option<ConnectionId>,
88            /// The value that the server included in the Source Connection ID field of a Retry
89            /// packet
90            pub(crate) retry_src_cid: Option<ConnectionId>,
91            /// Token used by the client to verify a stateless reset from the server
92            pub(crate) stateless_reset_token: Option<ResetToken>,
93            /// The server's preferred address for communication after handshake completion
94            pub(crate) preferred_address: Option<PreferredAddress>,
95        }
96
97        impl Default for TransportParameters {
98            /// Standard defaults, used if the peer does not supply a given parameter.
99            fn default() -> Self {
100                Self {
101                    $($name: VarInt::from_u32($default),)*
102
103                    disable_active_migration: false,
104                    max_datagram_frame_size: None,
105                    initial_src_cid: None,
106                    grease_quic_bit: false,
107
108                    original_dst_cid: None,
109                    retry_src_cid: None,
110                    stateless_reset_token: None,
111                    preferred_address: None,
112                }
113            }
114        }
115    }
116}

120impl TransportParameters {
121    pub(crate) fn new(
122        config: &TransportConfig,
123        endpoint_config: &EndpointConfig,
124        cid_gen: &dyn ConnectionIdGenerator,
125        initial_src_cid: ConnectionId,
126        server_config: Option<&ServerConfig>,
127    ) -> Self {
128        TransportParameters {
129            initial_src_cid: Some(initial_src_cid),
130            initial_max_streams_bidi: config.max_concurrent_bidi_streams,
131            initial_max_streams_uni: config.max_concurrent_uni_streams,
132            initial_max_data: config.receive_window,
133            initial_max_stream_data_bidi_local: config.stream_receive_window,
134            initial_max_stream_data_bidi_remote: config.stream_receive_window,
135            initial_max_stream_data_uni: config.stream_receive_window,
136            max_udp_payload_size: endpoint_config.max_udp_payload_size,
137            max_idle_timeout: config.max_idle_timeout.unwrap_or(VarInt(0)),
138            disable_active_migration: server_config.map_or(false, |c| !c.migration),
139            active_connection_id_limit: if cid_gen.cid_len() == 0 {
140                2 // i.e. default, i.e. unsent
141            } else {
142                CidQueue::LEN as u32
143            }
144            .into(),
145            max_datagram_frame_size: config
146                .datagram_receive_buffer_size
147                .map(|x| (x.min(u16::max_value().into()) as u16).into()),
148            grease_quic_bit: endpoint_config.grease_quic_bit,
149            ..Self::default()
150        }
151    }
152
153    /// Check that these parameters are legal when resuming from
154    /// certain cached parameters
155    pub(crate) fn validate_resumption_from(
156        &self,
157        cached: &TransportParameters,
158    ) -> Result<(), TransportError> {
159        if cached.active_connection_id_limit > self.active_connection_id_limit
160            || cached.initial_max_data > self.initial_max_data
161            || cached.initial_max_stream_data_bidi_local > self.initial_max_stream_data_bidi_local
162            || cached.initial_max_stream_data_bidi_remote > self.initial_max_stream_data_bidi_remote
163            || cached.initial_max_stream_data_uni > self.initial_max_stream_data_uni
164            || cached.initial_max_streams_bidi > self.initial_max_streams_bidi
165            || cached.initial_max_streams_uni > self.initial_max_streams_uni
166            || cached.max_datagram_frame_size > self.max_datagram_frame_size
167            || cached.grease_quic_bit && !self.grease_quic_bit
168        {
169            return Err(TransportError::PROTOCOL_VIOLATION(
170                "0-RTT accepted with incompatible transport parameters",
171            ));
172        }
173        Ok(())
174    }

176    /// Maximum number of CIDs to issue to this peer
177    ///
178    /// Consider both a) the active_connection_id_limit from the other end; and
179    /// b) LOC_CID_COUNT used locally
180    pub(crate) fn issue_cids_limit(&self) -> u64 {
181        self.active_connection_id_limit.0.min(LOC_CID_COUNT)
182    }

185/// A server's preferred address
186///
187/// This is communicated as a transport parameter during TLS session establishment.
188#[derive(Debug, Copy, Clone, Eq, PartialEq)]
189pub(crate) struct PreferredAddress {
190    pub address_v4: Option<SocketAddrV4>,
191    pub address_v6: Option<SocketAddrV6>,
192    pub connection_id: ConnectionId,
193    pub stateless_reset_token: ResetToken,
194}
195
196impl PreferredAddress {
197    fn wire_size(&self) -> u16 {
198        4 + 2 + 16 + 2 + 1 + self.connection_id.len() as u16 + 16
199    }
200
201    fn write<W: BufMut>(&self, w: &mut W) {
202        w.write(self.address_v4.map_or(Ipv4Addr::UNSPECIFIED, |x| *x.ip()));
203        w.write::<u16>(self.address_v4.map_or(0, |x| x.port()));
204        w.write(self.address_v6.map_or(Ipv6Addr::UNSPECIFIED, |x| *x.ip()));
205        w.write::<u16>(self.address_v6.map_or(0, |x| x.port()));
206        w.write::<u8>(self.connection_id.len() as u8);
207        w.put_slice(&self.connection_id);
208        w.put_slice(&self.stateless_reset_token);
209    }
210
211    fn read<R: Buf>(r: &mut R) -> Result<Self, Error> {
212        let ip_v4 = r.get::<Ipv4Addr>()?;
213        let port_v4 = r.get::<u16>()?;
214        let ip_v6 = r.get::<Ipv6Addr>()?;
215        let port_v6 = r.get::<u16>()?;
216        let cid_len = r.get::<u8>()?;
217        if r.remaining() < cid_len as usize || cid_len > MAX_CID_SIZE as u8 {
218            return Err(Error::Malformed);
219        }
220        let mut stage = [0; MAX_CID_SIZE];
221        r.copy_to_slice(&mut stage[0..cid_len as usize]);
222        let cid = ConnectionId::new(&stage[0..cid_len as usize]);
223        if r.remaining() < 16 {
224            return Err(Error::Malformed);
225        }
226        let mut token = [0; RESET_TOKEN_SIZE];
227        r.copy_to_slice(&mut token);
228        let address_v4 = if ip_v4.is_unspecified() && port_v4 == 0 {
229            None
230        } else {
231            Some(SocketAddrV4::new(ip_v4, port_v4))
232        };
233        let address_v6 = if ip_v6.is_unspecified() && port_v6 == 0 {
234            None
235        } else {
236            Some(SocketAddrV6::new(ip_v6, port_v6, 0, 0))
237        };
238        if address_v4.is_none() && address_v6.is_none() {
239            return Err(Error::IllegalValue);
240        }
241        Ok(Self {
242            address_v4,
243            address_v6,
244            connection_id: cid,
245            stateless_reset_token: token.into(),
246        })
247    }
248}

250/// Errors encountered while decoding `TransportParameters`
251#[derive(Debug, Copy, Clone, Eq, PartialEq, Error)]
252pub enum Error {
253    /// Parameters that are semantically invalid
254    #[error("parameter had illegal value")]
255    IllegalValue,
256    /// Catch-all error for problems while decoding transport parameters
257    #[error("parameters were malformed")]
258    Malformed,
259}
260
261impl From<Error> for TransportError {
262    fn from(e: Error) -> Self {
263        match e {
264            Error::IllegalValue => TransportError::TRANSPORT_PARAMETER_ERROR("illegal value"),
265            Error::Malformed => TransportError::TRANSPORT_PARAMETER_ERROR("malformed"),
266        }
267    }
268}
269
270impl From<UnexpectedEnd> for Error {
271    fn from(_: UnexpectedEnd) -> Self {
272        Error::Malformed
273    }
274}

276impl TransportParameters {
277    /// Encode `TransportParameters` into buffer
278    pub fn write<W: BufMut>(&self, w: &mut W) {
279        macro_rules! write_params {
280            {$($(#[$doc:meta])* $name:ident ($code:expr) = $default:expr,)*} => {
281                $(
282                    if self.$name.0 != $default {
283                        w.write_var($code);
284                        w.write(VarInt::try_from(self.$name.size()).unwrap());
285                        w.write(self.$name);
286                    }
287                )*
288            }
289        }
290        apply_params!(write_params);
291
292        // Add a reserved parameter to keep people on their toes
293        w.write_var(31 * 5 + 27);
294        w.write_var(0);
295
296        if let Some(ref x) = self.stateless_reset_token {
297            w.write_var(0x02);
298            w.write_var(16);
299            w.put_slice(x);
300        }
301
302        if self.disable_active_migration {
303            w.write_var(0x0c);
304            w.write_var(0);
305        }
306
307        if let Some(x) = self.max_datagram_frame_size {
308            w.write_var(0x20);
309            w.write_var(x.size() as u64);
310            w.write(x);
311        }
312
313        if let Some(ref x) = self.preferred_address {
314            w.write_var(0x000d);
315            w.write_var(x.wire_size() as u64);
316            x.write(w);
317        }
318
319        for &(tag, cid) in &[
320            (0x00, &self.original_dst_cid),
321            (0x0f, &self.initial_src_cid),
322            (0x10, &self.retry_src_cid),
323        ] {
324            if let Some(ref cid) = *cid {
325                w.write_var(tag);
326                w.write_var(cid.len() as u64);
327                w.put_slice(cid);
328            }
329        }
330
331        if self.grease_quic_bit {
332            w.write_var(0x2ab2);
333            w.write_var(0);
334        }
335    }

337    /// Decode `TransportParameters` from buffer
338    pub fn read<R: Buf>(side: Side, r: &mut R) -> Result<Self, Error> {
339        // Initialize to protocol-specified defaults
340        let mut params = TransportParameters::default();
341
342        // State to check for duplicate transport parameters.
343        macro_rules! param_state {
344            {$($(#[$doc:meta])* $name:ident ($code:expr) = $default:expr,)*} => {{
345                struct ParamState {
346                    $($name: bool,)*
347                }
348
349                ParamState {
350                    $($name: false,)*
351                }
352            }}
353        }
354        let mut got = apply_params!(param_state);
355
356        while r.has_remaining() {
357            let id = r.get_var()?;
358            let len = r.get_var()?;
359            if (r.remaining() as u64) < len {
360                return Err(Error::Malformed);
361            }
362            let len = len as usize;
363
364            match id {
365                0x00 => decode_cid(len, &mut params.original_dst_cid, r)?,
366                0x02 => {
367                    if len != 16 || params.stateless_reset_token.is_some() {
368                        return Err(Error::Malformed);
369                    }
370                    let mut tok = [0; RESET_TOKEN_SIZE];
371                    r.copy_to_slice(&mut tok);
372                    params.stateless_reset_token = Some(tok.into());
373                }
374                0x0c => {
375                    if len != 0 || params.disable_active_migration {
376                        return Err(Error::Malformed);
377                    }
378                    params.disable_active_migration = true;
379                }
380                0x0d => {
381                    if params.preferred_address.is_some() {
382                        return Err(Error::Malformed);
383                    }
384                    params.preferred_address =
385                        Some(PreferredAddress::read(&mut r.take(len as usize))?);
386                }
387                0x0f => decode_cid(len, &mut params.initial_src_cid, r)?,
388                0x10 => decode_cid(len, &mut params.retry_src_cid, r)?,
389                0x20 => {
390                    if len > 8 || params.max_datagram_frame_size.is_some() {
391                        return Err(Error::Malformed);
392                    }
393                    params.max_datagram_frame_size = Some(r.get().unwrap());
394                }
395                0x2ab2 => match len {
396                    0 => params.grease_quic_bit = true,
397                    _ => return Err(Error::Malformed),
398                },
399                _ => {
400                    macro_rules! parse {
401                        {$($(#[$doc:meta])* $name:ident ($code:expr) = $default:expr,)*} => {
402                            match id {
403                                $($code => {
404                                    let value = r.get::<VarInt>()?;
405                                    if len != value.size() || got.$name { return Err(Error::Malformed); }
406                                    params.$name = value.into();
407                                    got.$name = true;
408                                })*
409                                _ => r.advance(len as usize),
410                            }
411                        }
412                    }
413                    apply_params!(parse);
414                }
415            }
416        }
417
418        // Semantic validation
419        if params.ack_delay_exponent.0 > 20
420            || params.max_ack_delay.0 >= 1 << 14
421            || params.active_connection_id_limit.0 < 2
422            || params.max_udp_payload_size.0 < 1200
423            || params.initial_max_streams_bidi.0 > MAX_STREAM_COUNT
424            || params.initial_max_streams_uni.0 > MAX_STREAM_COUNT
425            || (side.is_server()
426                && (params.stateless_reset_token.is_some() || params.preferred_address.is_some()))
427        {
428            return Err(Error::IllegalValue);
429        }
430
431        Ok(params)
432    }
433}
434
435fn decode_cid(len: usize, value: &mut Option<ConnectionId>, r: &mut impl Buf) -> Result<(), Error> {
436    if len > MAX_CID_SIZE || value.is_some() || r.remaining() < len {
437        return Err(Error::Malformed);
438    }
439
440    *value = Some(ConnectionId::from_buf(r, len));
441    Ok(())
442}
