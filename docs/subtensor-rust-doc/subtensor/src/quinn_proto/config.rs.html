15/// Parameters governing the core QUIC state machine
16///
17/// Default values should be suitable for most internet applications. Applications protocols which
18/// forbid remotely-initiated streams should set `max_concurrent_bidi_streams` and
19/// `max_concurrent_uni_streams` to zero.
20///
21/// In some cases, performance or resource requirements can be improved by tuning these values to
22/// suit a particular application and/or network connection. In particular, data window sizes can be
23/// tuned for a particular expected round trip time, link capacity, and memory availability. Tuning
24/// for higher bandwidths and latencies increases worst-case memory consumption, but does not impair
25/// performance at lower bandwidths and latencies. The default configuration is tuned for a 100Mbps
26/// link with a 100ms round trip time.
27pub struct TransportConfig {
28    pub(crate) max_concurrent_bidi_streams: VarInt,
29    pub(crate) max_concurrent_uni_streams: VarInt,
30    pub(crate) max_idle_timeout: Option<VarInt>,
31    pub(crate) stream_receive_window: VarInt,
32    pub(crate) receive_window: VarInt,
33    pub(crate) send_window: u64,
34
35    pub(crate) max_tlps: u32,
36    pub(crate) packet_threshold: u32,
37    pub(crate) time_threshold: f32,
38    pub(crate) initial_rtt: Duration,
39    pub(crate) initial_max_udp_payload_size: u16,
40
41    pub(crate) persistent_congestion_threshold: u32,
42    pub(crate) keep_alive_interval: Option<Duration>,
43    pub(crate) crypto_buffer_size: usize,
44    pub(crate) allow_spin: bool,
45    pub(crate) datagram_receive_buffer_size: Option<usize>,
46    pub(crate) datagram_send_buffer_size: usize,
47
48    pub(crate) congestion_controller_factory: Box<dyn congestion::ControllerFactory + Send + Sync>,
49
50    pub(crate) enable_segmentation_offload: bool,
51}

53impl TransportConfig {
54    /// Maximum number of incoming bidirectional streams that may be open concurrently
55    ///
56    /// Must be nonzero for the peer to open any bidirectional streams.
57    ///
58    /// Worst-case memory use is directly proportional to `max_concurrent_bidi_streams *
59    /// stream_receive_window`, with an upper bound proportional to `receive_window`.
60    pub fn max_concurrent_bidi_streams(&mut self, value: VarInt) -> &mut Self {
61        self.max_concurrent_bidi_streams = value;
62        self
63    }
64
65    /// Variant of `max_concurrent_bidi_streams` affecting unidirectional streams
66    pub fn max_concurrent_uni_streams(&mut self, value: VarInt) -> &mut Self {
67        self.max_concurrent_uni_streams = value;
68        self
69    }
70
71    /// Maximum duration of inactivity to accept before timing out the connection.
72    ///
73    /// The true idle timeout is the minimum of this and the peer's own max idle timeout. `None`
74    /// represents an infinite timeout.
75    ///
76    /// **WARNING**: If a peer or its network path malfunctions or acts maliciously, an infinite
77    /// idle timeout can result in permanently hung futures!
78    ///
79    /// ```
80    /// # use std::{convert::TryInto, time::Duration};
81    /// # use quinn_proto::{TransportConfig, VarInt, VarIntBoundsExceeded};
82    /// # fn main() -> Result<(), VarIntBoundsExceeded> {
83    /// let mut config = TransportConfig::default();
84    ///
85    /// // Set the idle timeout as `VarInt`-encoded milliseconds
86    /// config.max_idle_timeout(Some(VarInt::from_u32(10_000).into()));
87    ///
88    /// // Set the idle timeout as a `Duration`
89    /// config.max_idle_timeout(Some(Duration::from_secs(10).try_into()?));
90    /// # Ok(())
91    /// # }
92    /// ```
93    pub fn max_idle_timeout(&mut self, value: Option<IdleTimeout>) -> &mut Self {
94        self.max_idle_timeout = value.map(|t| t.0);
95        self
96    }
97
98    /// Maximum number of bytes the peer may transmit without acknowledgement on any one stream
99    /// before becoming blocked.
100    ///
101    /// This should be set to at least the expected connection latency multiplied by the maximum
102    /// desired throughput. Setting this smaller than `receive_window` helps ensure that a single
103    /// stream doesn't monopolize receive buffers, which may otherwise occur if the application
104    /// chooses not to read from a large stream for a time while still requiring data on other
105    /// streams.
106    pub fn stream_receive_window(&mut self, value: VarInt) -> &mut Self {
107        self.stream_receive_window = value;
108        self
109    }
110
111    /// Maximum number of bytes the peer may transmit across all streams of a connection before
112    /// becoming blocked.
113    ///
114    /// This should be set to at least the expected connection latency multiplied by the maximum
115    /// desired throughput. Larger values can be useful to allow maximum throughput within a
116    /// stream while another is blocked.
117    pub fn receive_window(&mut self, value: VarInt) -> &mut Self {
118        self.receive_window = value;
119        self
120    }
121
122    /// Maximum number of bytes to transmit to a peer without acknowledgment
123    ///
124    /// Provides an upper bound on memory when communicating with peers that issue large amounts of
125    /// flow control credit. Endpoints that wish to handle large numbers of connections robustly
126    /// should take care to set this low enough to guarantee memory exhaustion does not occur if
127    /// every connection uses the entire window.
128    pub fn send_window(&mut self, value: u64) -> &mut Self {
129        self.send_window = value;
130        self
131    }
132
133    /// Maximum number of tail loss probes before an RTO fires.
134    pub fn max_tlps(&mut self, value: u32) -> &mut Self {
135        self.max_tlps = value;
136        self
137    }
138
139    /// Maximum reordering in packet number space before FACK style loss detection considers a
140    /// packet lost. Should not be less than 3, per RFC5681.
141    pub fn packet_threshold(&mut self, value: u32) -> &mut Self {
142        self.packet_threshold = value;
143        self
144    }
145
146    /// Maximum reordering in time space before time based loss detection considers a packet lost,
147    /// as a factor of RTT
148    pub fn time_threshold(&mut self, value: f32) -> &mut Self {
149        self.time_threshold = value;
150        self
151    }
152
153    /// The RTT used before an RTT sample is taken
154    pub fn initial_rtt(&mut self, value: Duration) -> &mut Self {
155        self.initial_rtt = value;
156        self
157    }
158
159    /// UDP payload size that the network must be capable of carrying
160    ///
161    /// Effective max UDP payload size may change over the life of the connection in the future due
162    /// to path MTU detection, but will never fall below this value.
163    ///
164    /// Must be at least 1200, which is the default, and known to be safe for typical internet
165    /// applications. Larger values are more efficient, but increase the risk of unpredictable
166    /// catastrophic packet loss due to exceeding the network path's IP MTU.
167    ///
168    /// Real-world MTUs can vary according to ISP, VPN, and properties of intermediate network links
169    /// outside of either endpoint's control. Extreme caution should be used when raising this value
170    /// for connections outside of private networks where these factors are fully controlled.
171    pub fn initial_max_udp_payload_size(&mut self, value: u16) -> &mut Self {
172        self.initial_max_udp_payload_size = value.max(INITIAL_MAX_UDP_PAYLOAD_SIZE);
173        self
174    }
175
176    /// Number of consecutive PTOs after which network is considered to be experiencing persistent congestion.
177    pub fn persistent_congestion_threshold(&mut self, value: u32) -> &mut Self {
178        self.persistent_congestion_threshold = value;
179        self
180    }
181
182    /// Period of inactivity before sending a keep-alive packet
183    ///
184    /// Keep-alive packets prevent an inactive but otherwise healthy connection from timing out.
185    ///
186    /// `None` to disable, which is the default. Only one side of any given connection needs keep-alive
187    /// enabled for the connection to be preserved. Must be set lower than the idle_timeout of both
188    /// peers to be effective.
189    pub fn keep_alive_interval(&mut self, value: Option<Duration>) -> &mut Self {
190        self.keep_alive_interval = value;
191        self
192    }
193
194    /// Maximum quantity of out-of-order crypto layer data to buffer
195    pub fn crypto_buffer_size(&mut self, value: usize) -> &mut Self {
196        self.crypto_buffer_size = value;
197        self
198    }
199
200    /// Whether the implementation is permitted to set the spin bit on this connection
201    ///
202    /// This allows passive observers to easily judge the round trip time of a connection, which can
203    /// be useful for network administration but sacrifices a small amount of privacy.
204    pub fn allow_spin(&mut self, value: bool) -> &mut Self {
205        self.allow_spin = value;
206        self
207    }
208
209    /// Maximum number of incoming application datagram bytes to buffer, or None to disable
210    /// incoming datagrams
211    ///
212    /// The peer is forbidden to send single datagrams larger than this size. If the aggregate size
213    /// of all datagrams that have been received from the peer but not consumed by the application
214    /// exceeds this value, old datagrams are dropped until it is no longer exceeded.
215    pub fn datagram_receive_buffer_size(&mut self, value: Option<usize>) -> &mut Self {
216        self.datagram_receive_buffer_size = value;
217        self
218    }
219
220    /// Maximum number of outgoing application datagram bytes to buffer
221    ///
222    /// While datagrams are sent ASAP, it is possible for an application to generate data faster
223    /// than the link, or even the underlying hardware, can transmit them. This limits the amount of
224    /// memory that may be consumed in that case. When the send buffer is full and a new datagram is
225    /// sent, older datagrams are dropped until sufficient space is available.
226    pub fn datagram_send_buffer_size(&mut self, value: usize) -> &mut Self {
227        self.datagram_send_buffer_size = value;
228        self
229    }
230
231    /// How to construct new `congestion::Controller`s
232    ///
233    /// Typically the refcounted configuration of a `congestion::Controller`,
234    /// e.g. a `congestion::NewRenoConfig`.
235    ///
236    /// # Example
237    /// ```
238    /// # use quinn_proto::*; use std::sync::Arc;
239    /// let mut config = TransportConfig::default();
240    /// config.congestion_controller_factory(Arc::new(congestion::NewRenoConfig::default()));
241    /// ```
242    pub fn congestion_controller_factory(
243        &mut self,
244        factory: impl congestion::ControllerFactory + Send + Sync + 'static,
245    ) -> &mut Self {
246        self.congestion_controller_factory = Box::new(factory);
247        self
248    }
249
250    /// Whether to use "Generic Segmentation Offload" to accelerate transmits, when supported by the
251    /// environment
252    ///
253    /// Defaults to `true`.
254    ///
255    /// GSO dramatically reduces CPU consumption when sending large numbers of packets with the same
256    /// headers, such as when transmitting bulk data on a connection. However, it is not supported
257    /// by all network interface drivers or packet inspection tools. `quinn-udp` will attempt to
258    /// disable GSO automatically when unavailable, but this can lead to spurious packet loss at
259    /// startup, temporarily degrading performance.
260    pub fn enable_segmentation_offload(&mut self, enabled: bool) -> &mut Self {
261        self.enable_segmentation_offload = enabled;
262        self
263    }

339///
340/// Default values should be suitable for most internet applications.
341#[derive(Clone)]
342pub struct EndpointConfig {
343    pub(crate) reset_key: Arc<dyn HmacKey>,
344    pub(crate) max_udp_payload_size: VarInt,
345    /// CID generator factory
346    ///
347    /// Create a cid generator for local cid in Endpoint struct
348    pub(crate) connection_id_generator_factory:
349        Arc<dyn Fn() -> Box<dyn ConnectionIdGenerator> + Send + Sync>,
350    pub(crate) supported_versions: Vec<u32>,
351    pub(crate) grease_quic_bit: bool,
352}
353
354impl EndpointConfig {
355    /// Create a default config with a particular `reset_key`
356    pub fn new(reset_key: Arc<dyn HmacKey>) -> Self {
357        let cid_factory: fn() -> Box<dyn ConnectionIdGenerator> =
358            || Box::new(RandomConnectionIdGenerator::default());
359        Self {
360            reset_key,
361            max_udp_payload_size: 1480u32.into(), // Typical internet MTU minus IPv4 and UDP overhead, rounded up to a multiple of 8
362            connection_id_generator_factory: Arc::new(cid_factory),
363            supported_versions: DEFAULT_SUPPORTED_VERSIONS.to_vec(),
364            grease_quic_bit: true,
365        }
366    }
367
368    /// Supply a custom connection ID generator factory
369    ///
370    /// Called once by each `Endpoint` constructed from this configuration to obtain the CID
371    /// generator which will be used to generate the CIDs used for incoming packets on all
372    /// connections involving that  `Endpoint`. A custom CID generator allows applications to embed
373    /// information in local connection IDs, e.g. to support stateless packet-level load balancers.
374    ///
375    /// `EndpointConfig::new()` applies a default random CID generator factory. This functions
376    /// accepts any customized CID generator to reset CID generator factory that implements
377    /// the `ConnectionIdGenerator` trait.
378    pub fn cid_generator<F: Fn() -> Box<dyn ConnectionIdGenerator> + Send + Sync + 'static>(
379        &mut self,
380        factory: F,
381    ) -> &mut Self {
382        self.connection_id_generator_factory = Arc::new(factory);
383        self
384    }
385
386    /// Private key used to send authenticated connection resets to peers who were
387    /// communicating with a previous instance of this endpoint.
388    pub fn reset_key(&mut self, key: Arc<dyn HmacKey>) -> &mut Self {
389        self.reset_key = key;
390        self
391    }
392
393    /// Maximum UDP payload size accepted from peers. Excludes UDP and IP overhead.
394    ///
395    /// The default is suitable for typical internet applications. Applications which expect to run
396    /// on networks supporting Ethernet jumbo frames or similar should set this appropriately.
397    pub fn max_udp_payload_size(&mut self, value: u64) -> Result<&mut Self, ConfigError> {
398        self.max_udp_payload_size = value.try_into()?;
399        Ok(self)
400    }
401
402    /// Get the current value of `max_udp_payload_size`
403    ///
404    /// While most parameters don't need to be readable, this must be exposed to allow higher-level
405    /// layers, e.g. the `quinn` crate, to determine how large a receive buffer to allocate to
406    /// support an externally-defined `EndpointConfig`.
407    ///
408    /// While `get_` accessors are typically unidiomatic in Rust, we favor concision for setters,
409    /// which will be used far more heavily.
410    #[doc(hidden)]
411    pub fn get_max_udp_payload_size(&self) -> u64 {
412        self.max_udp_payload_size.into()
413    }
414
415    /// Override supported QUIC versions
416    pub fn supported_versions(&mut self, supported_versions: Vec<u32>) -> &mut Self {
417        self.supported_versions = supported_versions;
418        self
419    }
420
421    /// Whether to accept QUIC packets containing any value for the fixed bit
422    ///
423    /// Enabled by default. Helps protect against protocol ossification and makes traffic less
424    /// identifiable to observers. Disable if helping observers identify this traffic as QUIC is
425    /// desired.
426    pub fn grease_quic_bit(&mut self, value: bool) -> &mut Self {
427        self.grease_quic_bit = value;
428        self
429    }
430}

570impl ServerConfig {
571    /// Create a server config with the given [`crypto::ServerConfig`]
572    ///
573    /// Uses a randomized handshake token key.
574    pub fn with_crypto(crypto: Arc<dyn crypto::ServerConfig>) -> Self {
575        let rng = &mut rand::thread_rng();
576        let mut master_key = [0u8; 64];
577        rng.fill_bytes(&mut master_key);
578        let master_key = ring::hkdf::Salt::new(ring::hkdf::HKDF_SHA256, &[]).extract(&master_key);
579
580        Self::new(crypto, Arc::new(master_key))
581    }

632        self.version = version;
633        self
634    }
635}

637#[cfg(feature = "rustls")]
638impl ClientConfig {
639    /// Create a client configuration that trusts the platform's native roots
640    #[cfg(feature = "native-certs")]
641    pub fn with_native_roots() -> Self {
642        let mut roots = rustls::RootCertStore::empty();
643        match rustls_native_certs::load_native_certs() {
644            Ok(certs) => {
645                for cert in certs {
646                    if let Err(e) = roots.add(&rustls::Certificate(cert.0)) {
647                        tracing::warn!("failed to parse trust anchor: {}", e);
648                    }
649                }
650            }
651            Err(e) => {
652                tracing::warn!("couldn't load any default trust roots: {}", e);
653            }
654        };
655
656        Self::with_root_certificates(roots)
657    }
658
659    /// Create a client configuration that trusts specified trust anchors
660    pub fn with_root_certificates(roots: rustls::RootCertStore) -> Self {
661        Self::new(Arc::new(crypto::rustls::client_config(roots)))
662    }
663}

696/// Maximum duration of inactivity to accept before timing out the connection.
697///
698/// This wraps an underlying [`VarInt`], representing the duration in milliseconds. Values can be
699/// constructed by converting directly from `VarInt`, or using `TryFrom<Duration>`.
700///
701/// ```
702/// # use std::{convert::TryFrom, time::Duration};
703/// # use quinn_proto::{IdleTimeout, VarIntBoundsExceeded, VarInt};
704/// # fn main() -> Result<(), VarIntBoundsExceeded> {
705/// // A `VarInt`-encoded value in milliseconds
706/// let timeout = IdleTimeout::from(VarInt::from_u32(10_000));
707///
708/// // Try to convert a `Duration` into a `VarInt`-encoded timeout
709/// let timeout = IdleTimeout::try_from(Duration::from_secs(10))?;
710/// # Ok(())
711/// # }
712/// ```
713#[derive(Default, Copy, Clone, Eq, Hash, Ord, PartialEq, PartialOrd)]
714pub struct IdleTimeout(VarInt);
715
716impl From<VarInt> for IdleTimeout {
717    fn from(inner: VarInt) -> Self {
718        Self(inner)
719    }
720}
721
722impl std::convert::TryFrom<Duration> for IdleTimeout {
723    type Error = VarIntBoundsExceeded;
724
725    fn try_from(timeout: Duration) -> Result<Self, Self::Error> {
726        let inner = VarInt::try_from(timeout.as_millis())?;
727        Ok(Self(inner))
728    }
729}
