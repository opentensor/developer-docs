1use std::ops::Range;
2
3use tinyvec::TinyVec;
4
5/// A set of u64 values optimized for long runs and random insert/delete/contains
6///
7/// `ArrayRangeSet` uses an array representation, where each array entry represents
8/// a range.
9///
10/// The array-based RangeSet provides 2 benefits:
11/// - There exists an inline representation, which avoids the need of heap
12///   allocating ACK ranges for SentFrames for small ranges.
13/// - Iterating over ranges should usually be faster since there is only
14///   a single cache-friendly contiguous range.
15///
16/// `ArrayRangeSet` is especially useful for tracking ACK ranges where the amount
17/// of ranges is usually very low (since ACK numbers are in consecutive fashion
18/// unless reordering or packet loss occur).
19#[derive(Debug, Default)]
20pub struct ArrayRangeSet(TinyVec<[Range<u64>; ARRAY_RANGE_SET_INLINE_CAPACITY]>);
21
22/// The capacity of elements directly stored in [`ArrayRangeSet`]
23///
24/// An inline capacity of 2 is chosen to keep `SentFrame` below 128 bytes.
25const ARRAY_RANGE_SET_INLINE_CAPACITY: usize = 2;
26
27impl Clone for ArrayRangeSet {
28    fn clone(&self) -> Self {
29        // tinyvec keeps the heap representation after clones.
30        // We rather prefer the inline representation for clones if possible,
31        // since clones (e.g. for storage in `SentFrames`) are rarely mutated
32        if self.0.is_inline() || self.0.len() > ARRAY_RANGE_SET_INLINE_CAPACITY {
33            return Self(self.0.clone());
34        }
35
36        let mut vec = TinyVec::new();
37        vec.extend_from_slice(self.0.as_slice());
38        Self(vec)
39    }
40}
41
42impl ArrayRangeSet {
43    pub fn new() -> Self {
44        Default::default()
45    }
46
47    pub fn iter(&self) -> impl DoubleEndedIterator<Item = Range<u64>> + '_ {
48        self.0.iter().cloned()
49    }
50
51    pub fn elts(&self) -> impl Iterator<Item = u64> + '_ {
52        self.iter().flatten()
53    }
54
55    pub fn len(&self) -> usize {
56        self.0.len()
57    }
58
59    pub fn contains(&self, x: u64) -> bool {
60        for range in self.0.iter() {
61            if range.start > x {
62                // We only get here if there was no prior range that contained x
63                return false;
64            } else if range.contains(&x) {
65                return true;
66            }
67        }
68        false
69    }
70
71    pub fn subtract(&mut self, other: &ArrayRangeSet) {
72        // TODO: This can potentially be made more efficient, since the we know
73        // individual ranges are not overlapping, and the next range must start
74        // after the last one finished
75        for range in &other.0 {
76            self.remove(range.clone());
77        }
78    }
79
80    pub fn insert_one(&mut self, x: u64) -> bool {
81        self.insert(x..x + 1)
82    }
83
84    pub fn insert(&mut self, x: Range<u64>) -> bool {
85        let mut result = false;
86
87        if x.is_empty() {
88            // Don't try to deal with ranges where x.end <= x.start
89            return false;
90        }
91
92        let mut idx = 0;
93        while idx != self.0.len() {
94            let range = &mut self.0[idx];
95
96            if range.start > x.end {
97                // The range is fully before this range and therefore not extensible.
98                // Add a new range to the left
99                self.0.insert(idx, x);
100                return true;
101            } else if range.start > x.start {
102                // The new range starts before this range but overlaps.
103                // Extend the current range to the left
104                // Note that we don't have to merge a potential left range, since
105                // this case would have been captured by merging the right range
106                // in the previous loop iteration
107                result = true;
108                range.start = x.start;
109            }
110
111            // At this point we have handled all parts of the new range which
112            // are in front of the current range. Now we handle everything from
113            // the start of the current range
114
115            if x.end <= range.end {
116                // Fully contained
117                return result;
118            } else if x.start <= range.end {
119                // Extend the current range to the end of the new range.
120                // Since it's not contained it must be bigger
121                range.end = x.end;
122
123                // Merge all follow-up ranges which overlap
124                while idx != self.0.len() - 1 {
125                    let curr = self.0[idx].clone();
126                    let next = self.0[idx + 1].clone();
127                    if curr.end >= next.start {
128                        self.0[idx].end = next.end.max(curr.end);
129                        self.0.remove(idx + 1);
130                    } else {
131                        break;
132                    }
133                }
134
135                return true;
136            }
137
138            idx += 1;
139        }
140
141        // Insert a range at the end
142        self.0.push(x);
143        true
144    }
145
146    pub fn remove(&mut self, x: Range<u64>) -> bool {
147        let mut result = false;
148
149        if x.is_empty() {
150            // Don't try to deal with ranges where x.end <= x.start
151            return false;
152        }
153
154        let mut idx = 0;
155        while idx != self.0.len() && x.start != x.end {
156            let range = self.0[idx].clone();
157
158            if x.end <= range.start {
159                // The range is fully before this range
160                return result;
161            } else if x.start >= range.end {
162                // The range is fully after this range
163                idx += 1;
164                continue;
165            }
166
167            // The range overlaps with this range
168            result = true;
169
170            let left = range.start..x.start;
171            let right = x.end..range.end;
172            if left.is_empty() && right.is_empty() {
173                self.0.remove(idx);
174            } else if left.is_empty() {
175                self.0[idx] = right;
176                idx += 1;
177            } else if right.is_empty() {
178                self.0[idx] = left;
179                idx += 1;
180            } else {
181                self.0[idx] = right;
182                self.0.insert(idx, left);
183                idx += 2;
184            }
185        }
186
187        result
188    }
189
190    pub fn is_empty(&self) -> bool {
191        self.0.is_empty()
192    }
193
194    pub fn pop_min(&mut self) -> Option<Range<u64>> {
195        if !self.0.is_empty() {
196            Some(self.0.remove(0))
197        } else {
198            None
199        }
200    }
201
202    pub fn min(&self) -> Option<u64> {
203        self.iter().next().map(|x| x.start)
204    }
205
206    pub fn max(&self) -> Option<u64> {
207        self.iter().rev().next().map(|x| x.end - 1)
208    }
