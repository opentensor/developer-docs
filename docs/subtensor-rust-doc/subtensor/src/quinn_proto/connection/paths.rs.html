6/// Description of a particular network path
7pub struct PathData {
8    pub remote: SocketAddr,
9    pub rtt: RttEstimator,
10    /// Whether we're enabling ECN on outgoing packets
11    pub sending_ecn: bool,
12    /// Congestion controller state
13    pub congestion: Box<dyn congestion::Controller>,
14    /// Pacing state
15    pub pacing: Pacer,
16    pub challenge: Option<u64>,
17    pub challenge_pending: bool,
18    /// Whether we're certain the peer can both send and receive on this address
19    ///
20    /// Initially equal to `use_stateless_retry` for servers, and becomes false again on every
21    /// migration. Always true for clients.
22    pub validated: bool,
23    /// Total size of all UDP datagrams sent on this path
24    pub total_sent: u64,
25    /// Total size of all UDP datagrams received on this path
26    pub total_recvd: u64,
27    pub max_udp_payload_size: u16,
28    /// Packet number of the first packet sent after an RTT sample was collected on this path
29    ///
30    /// Used in persistent congestion determination.
31    pub first_packet_after_rtt_sample: Option<(SpaceId, u64)>,
32}

35    pub fn new(

64    pub fn from_previous(remote: SocketAddr, prev: &PathData, now: Instant) -> Self {

66        let smoothed_rtt = prev.rtt.get();
67        PathData {

88    /// Indicates whether we're a server that hasn't validated the peer's address and hasn't
89    /// received enough data from the peer to permit sending `bytes_to_send` additional bytes
90    pub fn anti_amplification_blocked(&self, bytes_to_send: u64) -> bool {
91        !self.validated && self.total_recvd * 3 < self.total_sent + bytes_to_send

95/// RTT estimation for a particular network path
96#[derive(Copy, Clone)]
97pub struct RttEstimator {
98    /// The most recent RTT measurement made when receiving an ack for a previously unacked packet
99    latest: Duration,
100    /// The smoothed RTT of the connection, computed as described in RFC6298
101    smoothed: Option<Duration>,
102    /// The RTT variance, computed as described in RFC6298
103    var: Duration,
104    /// The minimum RTT seen in the connection, ignoring ack delay.
105    min: Duration,
106}
107
108impl RttEstimator {
109    fn new(initial_rtt: Duration) -> Self {
110        Self {
111            latest: initial_rtt,
112            smoothed: None,
113            var: initial_rtt / 2,
114            min: initial_rtt,
115        }
116    }
117
118    /// The current best RTT estimation.
119    pub fn get(&self) -> Duration {
120        self.smoothed.unwrap_or(self.latest)
121    }
122
123    /// Conservative estimate of RTT
124    ///
125    /// Takes the maximum of smoothed and latest RTT, as recommended
126    /// in 6.1.2 of the recovery spec (draft 29).
127    pub fn conservative(&self) -> Duration {
128        self.get().max(self.latest)
129    }
130
131    /// Minimum RTT registered so far for this estimator.
132    pub fn min(&self) -> Duration {
133        self.min
134    }
135
136    // PTO computed as described in RFC9002#6.2.1
137    pub(crate) fn pto_base(&self) -> Duration {
138        self.get() + cmp::max(4 * self.var, TIMER_GRANULARITY)
139    }
140
141    pub(crate) fn update(&mut self, ack_delay: Duration, rtt: Duration) {
142        self.latest = rtt;
143        // min_rtt ignores ack delay.
144        self.min = cmp::min(self.min, self.latest);
145        // Based on RFC6298.
146        if let Some(smoothed) = self.smoothed {
147            let adjusted_rtt = if self.min + ack_delay <= self.latest {
148                self.latest - ack_delay
149            } else {
150                self.latest
151            };
152            let var_sample = if smoothed > adjusted_rtt {
153                smoothed - adjusted_rtt
154            } else {
155                adjusted_rtt - smoothed
156            };
157            self.var = (3 * self.var + var_sample) / 4;
158            self.smoothed = Some((7 * smoothed + adjusted_rtt) / 8);
159        } else {
160            self.smoothed = Some(self.latest);
161            self.var = self.latest / 2;
162            self.min = self.latest;
163        }
164    }
165}
