20impl Recv {
21    pub(super) fn new(initial_max_data: u64) -> Self {
22        Self {
23            state: RecvState::default(),
24            assembler: Assembler::new(),
25            sent_max_stream_data: initial_max_data,
26            end: 0,
27            stopped: false,
28        }
29    }

31    /// Process a STREAM frame
32    ///
33    /// Return value is `(number_of_new_bytes_ingested, stream_is_closed)`
34    pub(super) fn ingest(
35        &mut self,
36        frame: frame::Stream,
37        payload_len: usize,
38        received: u64,
39        max_data: u64,
40    ) -> Result<(u64, bool), TransportError> {
41        let end = frame.offset + frame.data.len() as u64;
42        if end >= 2u64.pow(62) {
43            return Err(TransportError::FLOW_CONTROL_ERROR(
44                "maximum stream offset too large",
45            ));
46        }
47
48        if let Some(final_offset) = self.final_offset() {
49            if end > final_offset || (frame.fin && end != final_offset) {
50                debug!(end, final_offset, "final size error");
51                return Err(TransportError::FINAL_SIZE_ERROR(""));
52            }
53        }
54
55        let new_bytes = self.credit_consumed_by(end, received, max_data)?;
56
57        // Stopped streams don't need to wait for the actual data, they just need to know
58        // how much there was.
59        if frame.fin && !self.stopped {
60            if let RecvState::Recv { ref mut size } = self.state {
61                *size = Some(end);
62            }
63        }
64
65        self.end = self.end.max(end);
66        if !self.stopped {
67            self.assembler.insert(frame.offset, frame.data, payload_len);
68        } else {
69            self.assembler.set_bytes_read(end);
70        }
71
72        Ok((new_bytes, frame.fin && self.stopped))
73    }
74
75    pub(super) fn stop(&mut self) -> Result<(u64, ShouldTransmit), UnknownStream> {
76        if self.stopped {
77            return Err(UnknownStream { _private: () });
78        }
79
80        self.stopped = true;
81        self.assembler.clear();
82        // Issue flow control credit for unread data
83        let read_credits = self.end - self.assembler.bytes_read();
84        // This may send a spurious STOP_SENDING if we've already received all data, but it's a bit
85        // fiddly to distinguish that from the case where we've received a FIN but are missing some
86        // data that the peer might still be trying to retransmit, in which case a STOP_SENDING is
87        // still useful.
88        Ok((read_credits, ShouldTransmit(self.is_receiving())))
89    }

91    /// Returns the window that should be advertised in a `MAX_STREAM_DATA` frame
92    ///
93    /// The method returns a tuple which consists of the window that should be
94    /// announced, as well as a boolean parameter which indicates if a new
95    /// transmission of the value is recommended. If the boolean value is
96    /// `false` the new window should only be transmitted if a previous transmission
97    /// had failed.
98    pub(super) fn max_stream_data(&mut self, stream_receive_window: u64) -> (u64, ShouldTransmit) {
99        let max_stream_data = self.assembler.bytes_read() + stream_receive_window;
100
101        // Only announce a window update if it's significant enough
102        // to make it worthwhile sending a MAX_STREAM_DATA frame.
103        // We use here a fraction of the configured stream receive window to make
104        // the decision, and accommodate for streams using bigger windows requiring
105        // less updates. A fixed size would also work - but it would need to be
106        // smaller than `stream_receive_window` in order to make sure the stream
107        // does not get stuck.
108        let diff = max_stream_data - self.sent_max_stream_data;
109        let transmit = self.receiving_unknown_size() && diff >= (stream_receive_window / 8);
110        (max_stream_data, ShouldTransmit(transmit))
111    }
112
113    /// Records that a `MAX_STREAM_DATA` announcing a certain window was sent
114    ///
115    /// This will suppress enqueuing further `MAX_STREAM_DATA` frames unless
116    /// either the previous transmission was not acknowledged or the window
117    /// further increased.
118    pub fn record_sent_max_stream_data(&mut self, sent_value: u64) {
119        if sent_value > self.sent_max_stream_data {
120            self.sent_max_stream_data = sent_value;
121        }
122    }
123
124    pub(super) fn receiving_unknown_size(&self) -> bool {
125        matches!(self.state, RecvState::Recv { size: None })
126    }
127
128    /// Whether data is still being accepted from the peer
129    pub(super) fn is_receiving(&self) -> bool {
130        matches!(self.state, RecvState::Recv { .. })
131    }
132
133    fn final_offset(&self) -> Option<u64> {
134        match self.state {
135            RecvState::Recv { size } => size,
136            RecvState::ResetRecvd { size, .. } => Some(size),
137        }
138    }
139
140    /// Returns `false` iff the reset was redundant
141    pub(super) fn reset(
142        &mut self,
143        error_code: VarInt,
144        final_offset: VarInt,
145        received: u64,
146        max_data: u64,
147    ) -> Result<bool, TransportError> {
148        // Validate final_offset
149        if let Some(offset) = self.final_offset() {
150            if offset != final_offset.into_inner() {
151                return Err(TransportError::FINAL_SIZE_ERROR("inconsistent value"));
152            }
153        } else if self.end > final_offset.into() {
154            return Err(TransportError::FINAL_SIZE_ERROR(
155                "lower than high water mark",
156            ));
157        }
158        self.credit_consumed_by(final_offset.into(), received, max_data)?;
159
160        if matches!(self.state, RecvState::ResetRecvd { .. }) {
161            return Ok(false);
162        }
163        self.state = RecvState::ResetRecvd {
164            size: final_offset.into(),
165            error_code,
166        };
167        // Nuke buffers so that future reads fail immediately, which ensures future reads don't
168        // issue flow control credit redundant to that already issued. We could instead special-case
169        // reset streams during read, but it's unclear if there's any benefit to retaining data for
170        // reset streams.
171        self.assembler.clear();
172        Ok(true)
173    }
174
175    /// Compute the amount of flow control credit consumed, or return an error if more was consumed
176    /// than issued
177    fn credit_consumed_by(
178        &self,
179        offset: u64,
180        received: u64,
181        max_data: u64,
182    ) -> Result<u64, TransportError> {
183        let prev_end = self.end;
184        let new_bytes = offset.saturating_sub(prev_end);
185        if offset > self.sent_max_stream_data || received + new_bytes > max_data {
186            debug!(
187                received,
188                new_bytes,
189                max_data,
190                offset,
191                stream_max_data = self.sent_max_stream_data,
192                "flow control error"
193            );
194            return Err(TransportError::FLOW_CONTROL_ERROR(""));
195        }
196
197        Ok(new_bytes)
198    }
199}

201/// Chunks
202pub struct Chunks<'a> {
203    id: StreamId,
204    ordered: bool,
205    streams: &'a mut StreamsState,
206    pending: &'a mut Retransmits,
207    state: ChunksState,
208    read: u64,
209}
210
211impl<'a> Chunks<'a> {
212    pub(super) fn new(
213        id: StreamId,
214        ordered: bool,
215        streams: &'a mut StreamsState,
216        pending: &'a mut Retransmits,
217    ) -> Result<Self, ReadableError> {
218        let entry = match streams.recv.entry(id) {
219            Entry::Occupied(entry) => entry,
220            Entry::Vacant(_) => return Err(ReadableError::UnknownStream),
221        };
222
223        let mut recv = match entry.get().stopped {
224            true => return Err(ReadableError::UnknownStream),
225            false => entry.remove(),
226        };
227
228        recv.assembler.ensure_ordering(ordered)?;
229        Ok(Self {
230            id,
231            ordered,
232            streams,
233            pending,
234            state: ChunksState::Readable(recv),
235            read: 0,
236        })
237    }
238
239    /// Next
240    ///
241    /// Should call finalize() when done calling this.
242    pub fn next(&mut self, max_length: usize) -> Result<Option<Chunk>, ReadError> {
243        let rs = match self.state {
244            ChunksState::Readable(ref mut rs) => rs,
245            ChunksState::Reset(error_code) => {
246                return Err(ReadError::Reset(error_code));
247            }
248            ChunksState::Finished => {
249                return Ok(None);
250            }
251            ChunksState::Finalized => panic!("must not call next() after finalize()"),
252        };
253
254        if let Some(chunk) = rs.assembler.read(max_length, self.ordered) {
255            self.read += chunk.bytes.len() as u64;
256            return Ok(Some(chunk));
257        }
258
259        match rs.state {
260            RecvState::ResetRecvd { error_code, .. } => {
261                debug_assert_eq!(self.read, 0, "reset streams have empty buffers");
262                self.streams.stream_freed(self.id, StreamHalf::Recv);
263                self.state = ChunksState::Reset(error_code);
264                Err(ReadError::Reset(error_code))
265            }
266            RecvState::Recv { size } => {
267                if size == Some(rs.end) && rs.assembler.bytes_read() == rs.end {
268                    self.streams.stream_freed(self.id, StreamHalf::Recv);
269                    self.state = ChunksState::Finished;
270                    Ok(None)
271                } else {
272                    // We don't need a distinct `ChunksState` variant for a blocked stream because
273                    // retrying a read harmlessly re-traces our steps back to returning
274                    // `Err(Blocked)` again. The buffers can't refill and the stream's own state
275                    // can't change so long as this `Chunks` exists.
276                    Err(ReadError::Blocked)
277                }
278            }
279        }
280    }

282    /// Finalize
283    pub fn finalize(mut self) -> ShouldTransmit {
284        self.finalize_inner(false)
285    }
286
287    fn finalize_inner(&mut self, drop: bool) -> ShouldTransmit {
288        let state = mem::replace(&mut self.state, ChunksState::Finalized);
289        debug_assert!(
290            !drop || matches!(state, ChunksState::Finalized),
291            "finalize must be called before drop"
292        );
293        if let ChunksState::Finalized = state {
294            // Noop on repeated calls
295            return ShouldTransmit(false);
296        }
297
298        let mut should_transmit = false;
299        // We issue additional stream ID credit after the application is notified that a previously
300        // open stream has finished or been reset and we've therefore disposed of its state.
301        if matches!(state, ChunksState::Finished | ChunksState::Reset(_))
302            && self.streams.side != self.id.initiator()
303        {
304            self.pending.max_stream_id[self.id.dir() as usize] = true;
305            should_transmit = true;
306        }
307
308        // If the stream hasn't finished, we may need to issue stream-level flow control credit
309        if let ChunksState::Readable(mut rs) = state {
310            let (_, max_stream_data) = rs.max_stream_data(self.streams.stream_receive_window);
311            should_transmit |= max_stream_data.0;
312            if max_stream_data.0 {
313                self.pending.max_stream_data.insert(self.id);
314            }
315            // Return the stream to storage for future use
316            self.streams.recv.insert(self.id, rs);
317        }
318
319        // Issue connection-level flow control credit for any data we read regardless of state
320        let max_data = self.streams.add_read_credits(self.read);
321        self.pending.max_data |= max_data.0;
322        should_transmit |= max_data.0;
323        ShouldTransmit(should_transmit)
324    }
325}

327impl<'a> Drop for Chunks<'a> {
328    fn drop(&mut self) {
329        let _ = self.finalize_inner(true);
330    }
331}

340/// Errors triggered when reading from a recv stream
341#[derive(Debug, Error, Clone, Eq, PartialEq, Ord, PartialOrd, Hash)]
342pub enum ReadError {
343    /// No more data is currently available on this stream.
344    ///
345    /// If more data on this stream is received from the peer, an `Event::StreamReadable` will be
346    /// generated for this stream, indicating that retrying the read might succeed.
347    #[error("blocked")]
348    Blocked,
349    /// The peer abandoned transmitting data on this stream.
350    ///
351    /// Carries an application-defined error code.
352    #[error("reset by peer: code {0}")]
353    Reset(VarInt),
354}
355
356/// Errors triggered when opening a recv stream for reading
357#[derive(Debug, Error, Clone, Eq, PartialEq, Ord, PartialOrd, Hash)]
358pub enum ReadableError {
359    /// The stream has not been opened or was already stopped, finished, or reset
360    #[error("unknown stream")]
361    UnknownStream,
362    /// Attempted an ordered read following an unordered read
363    ///
364    /// Performing an unordered read allows discontinuities to arise in the receive buffer of a
365    /// stream which cannot be recovered, making further ordered reads impossible.
366    #[error("ordered read after unordered read")]
367    IllegalOrderedRead,
368}
369
370impl From<IllegalOrderedRead> for ReadableError {
371    fn from(_: IllegalOrderedRead) -> Self {
372        ReadableError::IllegalOrderedRead
373    }
374}
