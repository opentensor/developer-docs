27impl PacketBuilder {
28    /// Write a new packet header to `buffer` and determine the packet's properties
29    ///
30    /// Marks the connection drained and returns `None` if the confidentiality limit would be
31    /// violated.
32    pub fn new(
33        now: Instant,
34        space_id: SpaceId,
35        buffer: &mut Vec<u8>,
36        buffer_capacity: usize,
37        datagram_start: usize,
38        ack_eliciting: bool,
39        conn: &mut Connection,
40        version: u32,
41    ) -> Option<PacketBuilder> {
42        // Initiate key update if we're approaching the confidentiality limit
43        let confidentiality_limit = conn.spaces[space_id]
44            .crypto
45            .as_ref()
46            .map_or_else(
47                || &conn.zero_rtt_crypto.as_ref().unwrap().packet,
48                |keys| &keys.packet.local,
49            )
50            .confidentiality_limit();
51        let sent_with_keys = conn.spaces[space_id].sent_with_keys;
52        if space_id == SpaceId::Data {
53            if sent_with_keys.saturating_add(KEY_UPDATE_MARGIN) >= confidentiality_limit {
54                conn.initiate_key_update();
55            }
56        } else if sent_with_keys.saturating_add(1) == confidentiality_limit {
57            // We still have time to attempt a graceful close
58            conn.close_inner(
59                now,
60                Close::Connection(frame::ConnectionClose {
61                    error_code: TransportErrorCode::AEAD_LIMIT_REACHED,
62                    frame_type: None,
63                    reason: Bytes::from_static(b"confidentiality limit reached"),
64                }),
65            )
66        } else if sent_with_keys > confidentiality_limit {
67            // Confidentiality limited violated and there's nothing we can do
68            conn.kill(TransportError::AEAD_LIMIT_REACHED("confidentiality limit reached").into());
69            return None;
70        }
71
72        let space = &mut conn.spaces[space_id];
73
74        space.loss_probes = space.loss_probes.saturating_sub(1);
75        let exact_number = space.get_tx_number();
76
77        let span = trace_span!("send", space = ?space_id, pn = exact_number);
78        span.with_subscriber(|(id, dispatch)| dispatch.enter(id));
79
80        let number = PacketNumber::new(exact_number, space.largest_acked_packet.unwrap_or(0));
81        let header = match space_id {
82            SpaceId::Data if space.crypto.is_some() => Header::Short {
83                dst_cid: conn.rem_cids.active(),
84                number,
85                spin: if conn.spin_enabled {
86                    conn.spin
87                } else {
88                    conn.rng.gen()
89                },
90                key_phase: conn.key_phase,
91            },
92            SpaceId::Data => Header::Long {
93                ty: LongType::ZeroRtt,
94                src_cid: conn.handshake_cid,
95                dst_cid: conn.rem_cids.active(),
96                number,
97                version,
98            },
99            SpaceId::Handshake => Header::Long {
100                ty: LongType::Handshake,
101                src_cid: conn.handshake_cid,
102                dst_cid: conn.rem_cids.active(),
103                number,
104                version,
105            },
106            SpaceId::Initial => Header::Initial {
107                src_cid: conn.handshake_cid,
108                dst_cid: conn.rem_cids.active(),
109                token: conn.retry_token.clone(),
110                number,
111                version,
112            },
113        };
114        let partial_encode = header.encode(buffer);
115        if conn.peer_params.grease_quic_bit && conn.rng.gen() {
116            buffer[partial_encode.start] ^= FIXED_BIT;
117        }
118
119        let (sample_size, tag_len) = if let Some(ref crypto) = space.crypto {
120            (
121                crypto.header.local.sample_size(),
122                crypto.packet.local.tag_len(),
123            )
124        } else if space_id == SpaceId::Data {
125            let zero_rtt = conn.zero_rtt_crypto.as_ref().unwrap();
126            (zero_rtt.header.sample_size(), zero_rtt.packet.tag_len())
127        } else {
128            unreachable!("tried to send {:?} packet without keys", space_id);
129        };
130
131        // Each packet must be large enough for header protection sampling, i.e. the combined
132        // lengths of the encoded packet number and protected payload must be at least 4 bytes
133        // longer than the sample required for header protection. Further, each packet should be at
134        // least tag_len + 6 bytes larger than the destination CID on incoming packets so that the
135        // peer may send stateless resets that are indistinguishable from regular traffic.
136
137        // pn_len + payload_len + tag_len >= sample_size + 4
138        // payload_len >= sample_size + 4 - pn_len - tag_len
139        let min_size = Ord::max(
140            buffer.len() + (sample_size + 4).saturating_sub(number.len() + tag_len),
141            partial_encode.start + conn.rem_cids.active().len() + 6,
142        );
143        let max_size = buffer_capacity - partial_encode.start - partial_encode.header_len - tag_len;
144
145        Some(PacketBuilder {
146            datagram_start,
147            space: space_id,
148            partial_encode,
149            exact_number,
150            short_header: header.is_short(),
151            min_size,
152            max_size,
153            span,
154            tag_len,
155            ack_eliciting,
156        })
157    }

159    pub fn pad_to(&mut self, min_size: u16) {
160        let prev = self.min_size;
161        self.min_size = self.datagram_start + (min_size as usize) - self.tag_len;
162        debug_assert!(self.min_size >= prev, "padding must not shrink datagram");
163    }
164
165    pub fn finish_and_track(
166        self,
167        now: Instant,
168        conn: &mut Connection,
169        sent: Option<SentFrames>,
170        buffer: &mut Vec<u8>,
171    ) {
172        let ack_eliciting = self.ack_eliciting;
173        let exact_number = self.exact_number;
174        let space_id = self.space;
175        let (size, padded) = self.finish(conn, buffer);
176        let sent = match sent {
177            Some(sent) => sent,
178            None => return,
179        };
180
181        let size = match padded || ack_eliciting {
182            true => size as u16,
183            false => 0,
184        };
185
186        let packet = SentPacket {
187            largest_acked: sent.largest_acked,
188            time_sent: now,
189            size,
190            ack_eliciting,
191            retransmits: sent.retransmits,
192            stream_frames: sent.stream_frames,
193        };
194
195        conn.in_flight.insert(&packet);
196        conn.spaces[space_id].sent(exact_number, packet);
197        conn.stats.path.sent_packets += 1;
198        conn.reset_keep_alive(now);
199        if size != 0 {
200            if ack_eliciting {
201                conn.spaces[space_id].time_of_last_ack_eliciting_packet = Some(now);
202                if conn.permit_idle_reset {
203                    conn.reset_idle_timeout(now, space_id);
204                }
205                conn.permit_idle_reset = false;
206            }
207            conn.set_loss_detection_timer(now);
208            conn.path.pacing.on_transmit(size);
209        }
210    }

212    /// Encrypt packet, returning the length of the packet and whether padding was added
213    pub fn finish(
214        self: PacketBuilder,
215        conn: &mut Connection,
216        buffer: &mut Vec<u8>,
217    ) -> (usize, bool) {
218        let pad = buffer.len() < self.min_size;
219        if pad {
220            trace!("PADDING * {}", self.min_size - buffer.len());
221            buffer.resize(self.min_size, 0);
222        }
223
224        let space = &conn.spaces[self.space];
225        let (header_crypto, packet_crypto) = if let Some(ref crypto) = space.crypto {
226            (&*crypto.header.local, &*crypto.packet.local)
227        } else if self.space == SpaceId::Data {
228            let zero_rtt = conn.zero_rtt_crypto.as_ref().unwrap();
229            (&*zero_rtt.header, &*zero_rtt.packet)
230        } else {
231            unreachable!("tried to send {:?} packet without keys", self.space);
232        };
233
234        debug_assert_eq!(
235            packet_crypto.tag_len(),
236            self.tag_len,
237            "Mismatching crypto tag len"
238        );
239
240        buffer.resize(buffer.len() + packet_crypto.tag_len(), 0);
241        debug_assert!(
242            buffer.len() <= self.datagram_start + conn.path.max_udp_payload_size as usize
243        );
244        let encode_start = self.partial_encode.start;
245        let packet_buf = &mut buffer[encode_start..];
246        self.partial_encode.finish(
247            packet_buf,
248            header_crypto,
249            Some((self.exact_number, packet_crypto)),
250        );
251        self.span
252            .with_subscriber(|(id, dispatch)| dispatch.exit(id));
253
254        (buffer.len() - encode_start, pad)
255    }
256}

258/// Perform key updates this many packets before the AEAD confidentiality limit.
259///
260/// Chosen arbitrarily, intended to be large enough to prevent spurious connection loss.
261const KEY_UPDATE_MARGIN: u64 = 10000;
