95    /// Queue data for a tail loss probe (or anti-amplification deadlock prevention) packet
96    ///
97    /// Probes are sent similarly to normal packets when an expect ACK has not arrived. We never
98    /// deem a packet lost until we receive an ACK that should have included it, but if a trailing
99    /// run of packets (or their ACKs) are lost, this might not happen in a timely fashion. We send
100    /// probe packets to force an ACK, and exempt them from congestion control to prevent a deadlock
101    /// when the congestion window is filled with lost tail packets.
102    ///
103    /// We prefer to send new data, to make the most efficient use of bandwidth. If there's no data
104    /// waiting to be sent, then we retransmit in-flight data to reduce odds of loss. If there's no
105    /// in-flight data either, we're probably a client guarding against a handshake
106    /// anti-amplification deadlock and we just make something up.
107    pub(crate) fn maybe_queue_probe(&mut self, streams: &StreamsState) {
108        if self.loss_probes == 0 {
109            return;
110        }
111
112        // Retransmit the data of the oldest in-flight packet
113        if !self.pending.is_empty(streams) {
114            // There's real data to send here, no need to make something up
115            return;
116        }
117
118        for packet in self.sent_packets.values_mut() {
119            if !packet.retransmits.is_empty(streams) {
120                // Remove retransmitted data from the old packet so we don't end up retransmitting
121                // it *again* even if the copy we're sending now gets acknowledged.
122                self.pending |= mem::take(&mut packet.retransmits);
123                return;
124            }
125        }
126
127        // Nothing new to send and nothing to retransmit, so fall back on a ping. This should only
128        // happen in rare cases during the handshake when the server becomes blocked by
129        // anti-amplification.
130        self.ping_pending = true;
131    }
132
133    pub(crate) fn get_tx_number(&mut self) -> u64 {
134        // TODO: Handle packet number overflow gracefully
135        assert!(self.next_packet_number < 2u64.pow(62));
136        let x = self.next_packet_number;
137        self.next_packet_number += 1;
138        self.sent_with_keys += 1;
139        x
140    }

142    pub(crate) fn can_send(&self, streams: &StreamsState) -> SendableFrames {
143        let acks = self.pending_acks.can_send();
144        let other = !self.pending.is_empty(streams) || self.ping_pending;
145
146        SendableFrames { acks, other }
147    }

149    /// Verifies sanity of an ECN block and returns whether congestion was encountered.
150    pub(crate) fn detect_ecn(
151        &mut self,
152        newly_acked: u64,
153        ecn: frame::EcnCounts,
154    ) -> Result<bool, &'static str> {
155        let ect0_increase = ecn
156            .ect0
157            .checked_sub(self.ecn_feedback.ect0)
158            .ok_or("peer ECT(0) count regression")?;
159        let ect1_increase = ecn
160            .ect1
161            .checked_sub(self.ecn_feedback.ect1)
162            .ok_or("peer ECT(1) count regression")?;
163        let ce_increase = ecn
164            .ce
165            .checked_sub(self.ecn_feedback.ce)
166            .ok_or("peer CE count regression")?;
167        let total_increase = ect0_increase + ect1_increase + ce_increase;
168        if total_increase < newly_acked {
169            return Err("ECN bleaching");
170        }
171        if (ect0_increase + ce_increase) < newly_acked || ect1_increase != 0 {
172            return Err("ECN corruption");
173        }
174        // If total_increase > newly_acked (which happens when ACKs are lost), this is required by
175        // the draft so that long-term drift does not occur. If =, then the only question is whether
176        // to count CE packets as CE or ECT0. Recording them as CE is more consistent and keeps the
177        // congestion check obvious.
178        self.ecn_feedback = ecn;
179        Ok(ce_increase != 0)
180    }
181
182    pub(crate) fn sent(&mut self, number: u64, packet: SentPacket) {
183        self.in_flight += u64::from(packet.size);
184        self.sent_packets.insert(number, packet);
185    }

230    pub(crate) stop_sending: Vec<frame::StopSending>,

238impl Retransmits {
239    pub fn is_empty(&self, streams: &StreamsState) -> bool {
240        !self.max_data
241            && !self.max_stream_id.into_iter().any(|x| x)
242            && self.reset_stream.is_empty()
243            && self.stop_sending.is_empty()
244            && self
245                .max_stream_data
246                .iter()
247                .all(|&id| !streams.can_send_flow_control(id))
248            && self.crypto.is_empty()
249            && self.new_cids.is_empty()
250            && self.retire_cids.is_empty()
251            && !self.handshake_done
252    }
253}

255impl ::std::ops::BitOrAssign for Retransmits {
256    fn bitor_assign(&mut self, rhs: Self) {
257        // We reduce in-stream head-of-line blocking by queueing retransmits before other data for
258        // STREAM and CRYPTO frames.
259        self.max_data |= rhs.max_data;
260        for dir in Dir::iter() {
261            self.max_stream_id[dir as usize] |= rhs.max_stream_id[dir as usize];
262        }
263        self.reset_stream.extend_from_slice(&rhs.reset_stream);
264        self.stop_sending.extend_from_slice(&rhs.stop_sending);
265        self.max_stream_data.extend(&rhs.max_stream_data);
266        for crypto in rhs.crypto.into_iter().rev() {
267            self.crypto.push_front(crypto);
268        }
269        self.new_cids.extend(&rhs.new_cids);
270        self.retire_cids.extend(rhs.retire_cids);
271        self.handshake_done |= rhs.handshake_done;
272    }
273}

327/// RFC4303-style sliding window packet number deduplicator.
328///
329/// A contiguous bitfield, where each bit corresponds to a packet number and the rightmost bit is
330/// always set. A set bit represents a packet that has been successfully authenticated. Bits left of
331/// the window are assumed to be set.
332///
333/// ```text
334/// ...xxxxxxxxx 1 0
335///     ^        ^ ^
336/// window highest next
337/// ```
338pub struct Dedup {
339    window: Window,
340    /// Lowest packet number higher than all yet authenticated.
341    next: u64,
342}
343
344/// Inner bitfield type.
345///
346/// Because QUIC never reuses packet numbers, this only needs to be large enough to deal with
347/// packets that are reordered but still delivered in a timely manner.
348type Window = u128;
349
350/// Number of packets tracked by `Dedup`.
351const WINDOW_SIZE: u64 = 1 + mem::size_of::<Window>() as u64 * 8;
352
353impl Dedup {
354    /// Construct an empty window positioned at the start.
355    pub fn new() -> Self {
356        Self { window: 0, next: 0 }
357    }
358
359    /// Highest packet number authenticated.
360    fn highest(&self) -> u64 {
361        self.next - 1
362    }
363
364    /// Record a newly authenticated packet number.
365    ///
366    /// Returns whether the packet might be a duplicate.
367    pub fn insert(&mut self, packet: u64) -> bool {
368        if let Some(diff) = packet.checked_sub(self.next) {
369            // Right of window
370            self.window = (self.window << 1 | 1)
371                .checked_shl(cmp::min(diff, u64::from(u32::max_value())) as u32)
372                .unwrap_or(0);
373            self.next = packet + 1;
374            false
375        } else if self.highest() - packet < WINDOW_SIZE {
376            // Within window
377            if let Some(bit) = (self.highest() - packet).checked_sub(1) {
378                // < highest
379                let mask = 1 << bit;
380                let duplicate = self.window & mask != 0;
381                self.window |= mask;
382                duplicate
383            } else {
384                // == highest
385                true
386            }
387        } else {
388            // Left of window
389            true
390        }
391    }
392}

416#[derive(Debug, Default)]
417pub(crate) struct PendingAcks {
418    permit_ack_only: bool,
419    ranges: ArrayRangeSet,
420    /// This value will be used for calculating ACK delay once it is implemented
421    ///
422    /// ACK delay will be the delay between when a packet arrived (`latest_incoming`)
423    /// and between it will be allowed to be acknowledged (`can_send() == true`).
424    latest_incoming: Option<Instant>,
425    ack_delay: Duration,
426}
427
428impl PendingAcks {
429    /// Whether any ACK frames can be sent
430    pub fn can_send(&self) -> bool {
431        self.permit_ack_only && !self.ranges.is_empty()
432    }
433
434    /// Returns the duration the acknowledgement of the latest incoming packet has been delayed
435    pub fn ack_delay(&self) -> Duration {
436        self.ack_delay
437    }
438
439    /// Handle receipt of a new packet
440    pub fn packet_received(&mut self, ack_eliciting: bool) {
441        self.permit_ack_only |= ack_eliciting;
442    }
443
444    /// Should be called whenever ACKs have been sent
445    ///
446    /// This will suppress sending further ACKs until additional ACK eliciting frames arrive
447    pub fn acks_sent(&mut self) {
448        // If we sent any acks, don't immediately resend them. Setting this even if ack_only is
449        // false needlessly prevents us from ACKing the next packet if it's ACK-only, but saves
450        // the need for subtler logic to avoid double-transmitting acks all the time.
451        // This reset needs to happen before we check whether more data
452        // is available in this space - because otherwise it would return
453        // `true` purely due to the ACKs
454        self.permit_ack_only = false;
455    }
456
457    /// Insert one packet that needs to be acknowledged
458    pub fn insert_one(&mut self, packet: u64, now: Instant) {
459        self.ranges.insert_one(packet);
460        self.latest_incoming = Some(now);
461
462        if self.ranges.len() > MAX_ACK_BLOCKS {
463            self.ranges.pop_min();
464        }
465    }
466
467    /// Remove ACKs of packets numbered at or below `max` from the set of pending ACKs
468    pub fn subtract_below(&mut self, max: u64) {
469        self.ranges.remove(0..(max + 1));
470    }
471
472    /// Returns the set of currently pending ACK ranges
473    pub fn ranges(&self) -> &ArrayRangeSet {
474        &self.ranges
475    }
476}
