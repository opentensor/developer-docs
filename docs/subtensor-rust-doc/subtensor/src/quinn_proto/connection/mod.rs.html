82/// Protocol state and logic for a single QUIC connection
83///
84/// Objects of this type receive [`ConnectionEvent`]s and emit [`EndpointEvent`]s and application
85/// [`Event`]s to make progress. To handle timeouts, a `Connection` returns timer updates and
86/// expects timeouts through various methods. A number of simple getter methods are exposed
87/// to allow callers to inspect some of the connection state.
88///
89/// `Connection` has roughly 4 types of methods:
90///
91/// - A. Simple getters, taking `&self`
92/// - B. Handlers for incoming events from the network or system, named `handle_*`.
93/// - C. State machine mutators, for incoming commands from the application. For convenience we
94///   refer to this as "performing I/O" below, however as per the design of this library none of the
95///   functions actually perform system-level I/O. For example, [`read`](RecvStream::read) and
96///   [`write`](SendStream::write), but also things like [`reset`](SendStream::reset).
97/// - D. Polling functions for outgoing events or actions for the caller to
98///   take, named `poll_*`.
99///
100/// The simplest way to use this API correctly is to call (B) and (C) whenever
101/// appropriate, then after each of those calls, as soon as feasible call all
102/// polling methods (D) and deal with their outputs appropriately, e.g. by
103/// passing it to the application or by making a system-level I/O call. You
104/// should call the polling functions in this order:
105///
106/// 1. [`poll_transmit`](Self::poll_transmit)
107/// 2. [`poll_timeout`](Self::poll_timeout)
108/// 3. [`poll_endpoint_events`](Self::poll_endpoint_events)
109/// 4. [`poll`](Self::poll)
110///
111/// Currently the only actual dependency is from (2) to (1), however additional
112/// dependencies may be added in future, so the above order is recommended.
113///
114/// (A) may be called whenever desired.
115///
116/// Care should be made to ensure that the input events represent monotonically
117/// increasing time. Specifically, calling [`handle_timeout`](Self::handle_timeout)
118/// with events of the same [`Instant`] may be interleaved in any order with a
119/// call to [`handle_event`](Self::handle_event) at that same instant; however
120/// events or timeouts with different instants must not be interleaved.

342    /// Returns application-facing events

344    /// Connections should be polled for events after:
345    /// - a call was made to `handle_event`
346    /// - a call was made to `handle_timeout`
347    #[must_use]
348    pub fn poll(&mut self) -> Option<Event> {
349        if let Some(x) = self.events.pop_front() {
350            return Some(x);
351        }
352
353        if let Some(event) = self.streams.poll() {
354            return Some(Event::Stream(event));
355        }
356
357        if let Some(err) = self.error.take() {
358            return Some(Event::ConnectionLost { reason: err });
359        }
360
361        None
362    }

402    /// Returns packets to transmit
403    ///
404    /// Connections should be polled for transmit after:
405    /// - the application performed some I/O on the connection
406    /// - a call was made to `handle_event`
407    /// - a call was made to `handle_timeout`
408    ///
409    /// `max_datagrams` specifies how many datagrams can be returned inside a
410    /// single Transmit using GSO. This must be at least 1.
411    #[must_use]
412    pub fn poll_transmit(&mut self, now: Instant, max_datagrams: usize) -> Option<Transmit> {
413        assert!(max_datagrams != 0);
414        let max_datagrams = match self.config.enable_segmentation_offload {
415            false => 1,
416            true => max_datagrams.min(MAX_TRANSMIT_SEGMENTS),
417        };
418
419        let mut num_datagrams = 0;
420
421        // Send PATH_CHALLENGE for a previous path if necessary
422        if let Some(ref mut prev_path) = self.prev_path {
423            if prev_path.challenge_pending {
424                prev_path.challenge_pending = false;
425                let token = prev_path
426                    .challenge
427                    .expect("previous path challenge pending without token");
428                let destination = prev_path.remote;
429                debug_assert_eq!(
430                    self.highest_space,
431                    SpaceId::Data,
432                    "PATH_CHALLENGE queued without 1-RTT keys"
433                );
434                let mut buf = Vec::with_capacity(self.path.max_udp_payload_size as usize);
435                let buf_capacity = self.path.max_udp_payload_size as usize;
436
437                let mut builder = PacketBuilder::new(
438                    now,
439                    SpaceId::Data,
440                    &mut buf,
441                    buf_capacity,
442                    0,
443                    false,
444                    self,
445                    self.version,
446                )?;
447                trace!("validating previous path with PATH_CHALLENGE {:08x}", token);
448                buf.write(frame::Type::PATH_CHALLENGE);
449                buf.write(token);
450                self.stats.frame_tx.path_challenge += 1;
451
452                // An endpoint MUST expand datagrams that contain a PATH_CHALLENGE frame
453                // to at least the smallest allowed maximum datagram size of 1200 bytes,
454                // unless the anti-amplification limit for the path does not permit
455                // sending a datagram of this size
456                builder.pad_to(MIN_INITIAL_SIZE);
457
458                builder.finish(self, &mut buf);
459                self.stats.udp_tx.datagrams += 1;
460                self.stats.udp_tx.transmits += 1;
461                self.stats.udp_tx.bytes += buf.len() as u64;
462                return Some(Transmit {
463                    destination,
464                    contents: buf,
465                    ecn: None,
466                    segment_size: None,
467                    src_ip: self.local_ip,
468                });
469            }
470        }
471
472        // If we need to send a probe, make sure we have something to send.
473        for space in SpaceId::iter() {
474            self.spaces[space].maybe_queue_probe(&self.streams);
475        }
476
477        // Check whether we need to send a close message
478        let close = match self.state {
479            State::Drained => {
480                self.app_limited = true;
481                return None;
482            }
483            State::Draining | State::Closed(_) => {
484                // self.close is only reset once the associated packet had been
485                // encoded successfully
486                if !self.close {
487                    self.app_limited = true;
488                    return None;
489                }
490                true
491            }
492            _ => false,
493        };
494
495        let mut buf = Vec::new();
496        // Reserving capacity can provide more capacity than we asked for.
497        // However we are not allowed to write more than MTU size. Therefore
498        // the maximum capacity is tracked separately.
499        let mut buf_capacity = 0;
500
501        let mut coalesce = true;
502        let mut builder: Option<PacketBuilder> = None;
503        let mut sent_frames = None;
504        let mut pad_datagram = false;
505        let mut congestion_blocked = false;
506
507        // Iterate over all spaces and find data to send
508        let mut space_idx = 0;
509        let spaces = [SpaceId::Initial, SpaceId::Handshake, SpaceId::Data];
510        // This loop will potentially spend multiple iterations in the same `SpaceId`,
511        // so we cannot trivially rewrite it to take advantage of `SpaceId::iter()`.
512        while space_idx < spaces.len() {
513            let space_id = spaces[space_idx];
514
515            if close && space_id != self.highest_space {
516                // We ignore data in this space, since the close message
517                // has higher priority
518                space_idx += 1;
519                continue;
520            }
521
522            // Is there data or a close message to send in this space?
523            let can_send = self.space_can_send(space_id);
524            if can_send.is_empty() && !close {
525                space_idx += 1;
526                continue;
527            }
528
529            let mut ack_eliciting = !self.spaces[space_id].pending.is_empty(&self.streams)
530                || self.spaces[space_id].ping_pending;
531            if space_id == SpaceId::Data {
532                ack_eliciting |= self.can_send_1rtt();
533            }
534
535            // Can we append more data into the current buffer?
536            // It is not safe to assume that `buf.len()` is the end of the data,
537            // since the last packet might not have been finished.
538            let buf_end = if let Some(builder) = &builder {
539                buf.len().max(builder.min_size) + builder.tag_len
540            } else {
541                buf.len()
542            };
543
544            if !coalesce || buf_capacity - buf_end < MIN_PACKET_SPACE {
545                // We need to send 1 more datagram and extend the buffer for that.
546
547                // Is 1 more datagram allowed?
548                if buf_capacity >= self.path.max_udp_payload_size as usize * max_datagrams {
549                    // No more datagrams allowed
550                    break;
551                }
552
553                // Anti-amplification is only based on `total_sent`, which gets
554                // updated at the end of this method. Therefore we pass the amount
555                // of bytes for datagrams that are already created, as well as 1 byte
556                // for starting another datagram. If there is any anti-amplification
557                // budget left, we always allow a full MTU to be sent
558                // (see https://github.com/quinn-rs/quinn/issues/1082)
559                if self.path.anti_amplification_blocked(
560                    self.path.max_udp_payload_size as u64 * num_datagrams as u64 + 1,
561                ) {
562                    trace!("blocked by anti-amplification");
563                    break;
564                }
565
566                // Congestion control and pacing checks
567                // Tail loss probes must not be blocked by congestion, or a deadlock could arise
568                if ack_eliciting && self.spaces[space_id].loss_probes == 0 {
569                    // Assume the current packet will get padded to fill the full MTU
570                    let untracked_bytes = if let Some(builder) = &builder {
571                        buf_capacity - builder.partial_encode.start
572                    } else {
573                        0
574                    } as u64;
575                    debug_assert!(untracked_bytes <= self.path.max_udp_payload_size as u64);
576
577                    let bytes_to_send = u64::from(self.path.max_udp_payload_size) + untracked_bytes;
578                    if self.in_flight.bytes + bytes_to_send >= self.path.congestion.window() {
579                        space_idx += 1;
580                        congestion_blocked = true;
581                        // We continue instead of breaking here in order to avoid
582                        // blocking loss probes queued for higher spaces.
583                        continue;
584                    }
585
586                    // Check whether the next datagram is blocked by pacing
587                    let smoothed_rtt = self.path.rtt.get();
588                    if let Some(delay) = self.path.pacing.delay(
589                        smoothed_rtt,
590                        bytes_to_send,
591                        self.path.max_udp_payload_size,
592                        self.path.congestion.window(),
593                        now,
594                    ) {
595                        self.timers.set(Timer::Pacing, delay);
596                        congestion_blocked = true;
597                        // Loss probes should be subject to pacing, even though
598                        // they are not congestion controlled.
599                        break;
600                    }
601                }
602
603                // Finish current packet
604                if let Some(mut builder) = builder.take() {
605                    // Pad the packet to make it suitable for sending with GSO
606                    // which will always send the maximum PDU.
607                    builder.pad_to(self.path.max_udp_payload_size);
608
609                    builder.finish_and_track(now, self, sent_frames.take(), &mut buf);
610
611                    debug_assert_eq!(buf.len(), buf_capacity, "Packet must be padded");
612                }
613
614                // Allocate space for another datagram
615                buf_capacity += self.path.max_udp_payload_size as usize;
616                if buf.capacity() < buf_capacity {
617                    // We reserve the maximum space for sending `max_datagrams` upfront
618                    // to avoid any reallocations if more datagrams have to be appended later on.
619                    // Benchmarks have shown shown a 5-10% throughput improvement
620                    // compared to continuously resizing the datagram buffer.
621                    // While this will lead to over-allocation for small transmits
622                    // (e.g. purely containing ACKs), modern memory allocators
623                    // (e.g. mimalloc and jemalloc) will pool certain allocation sizes
624                    // and therefore this is still rather efficient.
625                    buf.reserve(
626                        max_datagrams * self.path.max_udp_payload_size as usize - buf.capacity(),
627                    );
628                }
629                num_datagrams += 1;
630                coalesce = true;
631                pad_datagram = false;
632            } else {
633                // We can append/coalesce the next packet into the current
634                // datagram.
635                // Finish current packet without adding extra padding
636                if let Some(builder) = builder.take() {
637                    builder.finish_and_track(now, self, sent_frames.take(), &mut buf);
638                }
639            }
640
641            debug_assert!(buf_capacity - buf.len() >= MIN_PACKET_SPACE);
642
643            //
644            // From here on, we've determined that a packet will definitely be sent.
645            //
646
647            if self.spaces[SpaceId::Initial].crypto.is_some()
648                && space_id == SpaceId::Handshake
649                && self.side.is_client()
650            {
651                // A client stops both sending and processing Initial packets when it
652                // sends its first Handshake packet.
653                self.discard_space(now, SpaceId::Initial);
654            }
655            if let Some(ref mut prev) = self.prev_crypto {
656                prev.update_unacked = false;
657            }
658
659            debug_assert!(
660                builder.is_none() && sent_frames.is_none(),
661                "Previous packet must have been finished"
662            );
663
664            // This should really be `builder.insert()`, but `Option::insert`
665            // is not stable yet. Since we `debug_assert!(builder.is_none())` it
666            // doesn't make any functional difference.
667            let builder = builder.get_or_insert(PacketBuilder::new(
668                now,
669                space_id,
670                &mut buf,
671                buf_capacity,
672                (num_datagrams - 1) * (self.path.max_udp_payload_size as usize),
673                ack_eliciting,
674                self,
675                self.version,
676            )?);
677            coalesce = coalesce && !builder.short_header;
678
679            // https://tools.ietf.org/html/draft-ietf-quic-transport-34#section-14.1
680            pad_datagram |=
681                space_id == SpaceId::Initial && (self.side.is_client() || ack_eliciting);
682
683            if close {
684                trace!("sending CONNECTION_CLOSE");
685                // Encode ACKs before the ConnectionClose message, to give the receiver
686                // a better approximate on what data has been processed. This is
687                // especially important with ack delay, since the peer might not
688                // have gotten any other ACK for the data earlier on.
689                if !self.spaces[space_id].pending_acks.ranges().is_empty() {
690                    Self::populate_acks(
691                        self.receiving_ecn,
692                        &mut SentFrames::default(),
693                        &mut self.spaces[space_id],
694                        &mut buf,
695                        &mut self.stats,
696                    );
697                }
698
699                // Since there only 64 ACK frames there will always be enough space
700                // to encode the ConnectionClose frame too. However we still have the
701                // check here to prevent crashes if something changes.
702                debug_assert!(
703                    buf.len() + frame::ConnectionClose::SIZE_BOUND < builder.max_size,
704                    "ACKs should leave space for ConnectionClose"
705                );
706                if buf.len() + frame::ConnectionClose::SIZE_BOUND < builder.max_size {
707                    match self.state {
708                        State::Closed(state::Closed { ref reason }) => {
709                            if space_id == SpaceId::Data {
710                                reason.encode(&mut buf, builder.max_size)
711                            } else {
712                                frame::ConnectionClose {
713                                    error_code: TransportErrorCode::APPLICATION_ERROR,
714                                    frame_type: None,
715                                    reason: Bytes::new(),
716                                }
717                                .encode(&mut buf, builder.max_size)
718                            }
719                        }
720                        State::Draining => frame::ConnectionClose {
721                            error_code: TransportErrorCode::NO_ERROR,
722                            frame_type: None,
723                            reason: Bytes::new(),
724                        }
725                        .encode(&mut buf, builder.max_size),
726                        _ => unreachable!(
727                            "tried to make a close packet when the connection wasn't closed"
728                        ),
729                    }
730                }
731                // Don't send another close packet
732                self.close = false;
733                // `CONNECTION_CLOSE` is the final packet
734                break;
735            }
736
737            let sent = self.populate_packet(space_id, &mut buf, buf_capacity - builder.tag_len);
738
739            // ACK-only packets should only be sent when explicitly allowed. If we write them due
740            // to any other reason, there is a bug which leads to one component announcing write
741            // readiness while not writing any data. This degrades performance. The condition is
742            // only checked if the full MTU is available, so that lack of space in the datagram isn't
743            // the reason for just writing ACKs.
744            debug_assert!(
745                !(sent.is_ack_only(&self.streams)
746                    && !can_send.acks
747                    && can_send.other
748                    && (buf_capacity - builder.datagram_start)
749                        == self.path.max_udp_payload_size as usize),
750                "SendableFrames was {:?}, but only ACKs have been written",
751                can_send
752            );
753            pad_datagram |= sent.requires_padding;
754
755            if sent.largest_acked.is_some() {
756                self.spaces[space_id].pending_acks.acks_sent();
757            }
758
759            // Keep information about the packet around until it gets finalized
760            sent_frames = Some(sent);
761
762            // Don't increment space_idx.
763            // We stay in the current space and check if there is more data to send.
764        }
765
766        // Finish the last packet
767        if let Some(mut builder) = builder {
768            if pad_datagram {
769                builder.pad_to(MIN_INITIAL_SIZE);
770            }
771            let last_packet_number = builder.exact_number;
772            builder.finish_and_track(now, self, sent_frames, &mut buf);
773            self.path
774                .congestion
775                .on_sent(now, buf.len() as u64, last_packet_number);
776        }
777
778        self.app_limited = buf.is_empty() && !congestion_blocked;
779
780        if buf.is_empty() {
781            return None;
782        }
783
784        trace!("sending {} bytes in {} datagrams", buf.len(), num_datagrams);
785        self.path.total_sent = self.path.total_sent.saturating_add(buf.len() as u64);
786
787        self.stats.udp_tx.datagrams += num_datagrams as u64;
788        self.stats.udp_tx.bytes += buf.len() as u64;
789        self.stats.udp_tx.transmits += 1;
790
791        Some(Transmit {
792            destination: self.path.remote,
793            contents: buf,
794            ecn: if self.path.sending_ecn {
795                Some(EcnCodepoint::Ect0)
796            } else {
797                None
798            },
799            segment_size: match num_datagrams {
800                1 => None,
801                _ => Some(self.path.max_udp_payload_size as usize),
802            },
803            src_ip: self.local_ip,
804        })
805    }

807    /// Indicate what types of frames are ready to send for the given space
808    fn space_can_send(&self, space_id: SpaceId) -> SendableFrames {
809        if self.spaces[space_id].crypto.is_some() {
810            let can_send = self.spaces[space_id].can_send(&self.streams);
811            if !can_send.is_empty() {
812                return can_send;
813            }
814        }
815
816        if space_id != SpaceId::Data {
817            return SendableFrames::empty();
818        }
819
820        if self.spaces[space_id].crypto.is_some() && self.can_send_1rtt() {
821            return SendableFrames {
822                other: true,
823                acks: false,
824            };
825        }
826
827        if self.zero_rtt_crypto.is_some() && self.side.is_client() {
828            let mut can_send = self.spaces[space_id].can_send(&self.streams);
829            can_send.other |= self.can_send_1rtt();
830            if !can_send.is_empty() {
831                return can_send;
832            }
833        }
834
835        SendableFrames::empty()
836    }

838    /// Process `ConnectionEvent`s generated by the associated `Endpoint`
839    ///
840    /// Will execute protocol logic upon receipt of a connection event, in turn preparing signals
841    /// (including application `Event`s, `EndpointEvent`s and outgoing datagrams) that should be
842    /// extracted through the relevant methods.
843    pub fn handle_event(&mut self, event: ConnectionEvent) {
844        use self::ConnectionEventInner::*;
845        match event.0 {
846            Datagram {
847                now,
848                remote,
849                ecn,
850                first_decode,
851                remaining,
852            } => {
853                // If this packet could initiate a migration and we're a client or a server that
854                // forbids migration, drop the datagram. This could be relaxed to heuristically
855                // permit NAT-rebinding-like migration.
856                if remote != self.path.remote
857                    && self.server_config.as_ref().map_or(true, |x| !x.migration)
858                {
859                    trace!("discarding packet from unrecognized peer {}", remote);
860                    return;
861                }
862
863                let was_anti_amplification_blocked = self.path.anti_amplification_blocked(1);
864
865                self.stats.udp_rx.datagrams += 1;
866                self.stats.udp_rx.bytes += first_decode.len() as u64;
867                let data_len = first_decode.len();
868
869                self.handle_decode(now, remote, ecn, first_decode);
870                // The current `path` might have changed inside `handle_decode`,
871                // since the packet could have triggered a migration. Make sure
872                // the data received is accounted for the most recent path by accessing
873                // `path` after `handle_decode`.
874                self.path.total_recvd = self.path.total_recvd.saturating_add(data_len as u64);
875
876                if let Some(data) = remaining {
877                    self.stats.udp_rx.bytes += data.len() as u64;
878                    self.handle_coalesced(now, remote, ecn, data);
879                }
880
881                if was_anti_amplification_blocked {
882                    // A prior attempt to set the loss detection timer may have failed due to
883                    // anti-amplification, so ensure it's set now. Prevents a handshake deadlock if
884                    // the server's first flight is lost.
885                    self.set_loss_detection_timer(now);
886                }
887            }
888            NewIdentifiers(ids, now) => {
889                self.local_cid_state.new_cids(&ids, now);
890                ids.into_iter().rev().for_each(|frame| {
891                    self.spaces[SpaceId::Data].pending.new_cids.push(frame);
892                });
893                // Update Timer::PushNewCid
894                if self
895                    .timers
896                    .get(Timer::PushNewCid)
897                    .map_or(true, |x| x <= now)
898                {
899                    self.reset_cid_retirement();
900                }
901            }
902        }
903    }

905    /// Process timer expirations
906    ///
907    /// Executes protocol logic, potentially preparing signals (including application `Event`s,
908    /// `EndpointEvent`s and outgoing datagrams) that should be extracted through the relevant
909    /// methods.
910    ///
911    /// It is most efficient to call this immediately after the system clock reaches the latest
912    /// `Instant` that was output by `poll_timeout`; however spurious extra calls will simply
913    /// no-op and therefore are safe.
914    pub fn handle_timeout(&mut self, now: Instant) {
915        for &timer in &Timer::VALUES {
916            if !self.timers.is_expired(timer, now) {
917                continue;
918            }
919            self.timers.stop(timer);
920            trace!(timer = ?timer, "timeout");
921            match timer {
922                Timer::Close => {
923                    self.state = State::Drained;
924                    self.endpoint_events.push_back(EndpointEventInner::Drained);
925                }
926                Timer::Idle => {
927                    self.kill(ConnectionError::TimedOut);
928                }
929                Timer::KeepAlive => {
930                    trace!("sending keep-alive");
931                    self.ping();
932                }
933                Timer::LossDetection => {
934                    self.on_loss_detection_timeout(now);
935                }
936                Timer::KeyDiscard => {
937                    self.zero_rtt_crypto = None;
938                    self.prev_crypto = None;
939                }
940                Timer::PathValidation => {
941                    debug!("path validation failed");
942                    if let Some(prev) = self.prev_path.take() {
943                        self.path = prev;
944                    }
945                    self.path.challenge = None;
946                    self.path.challenge_pending = false;
947                }
948                Timer::Pacing => trace!("pacing timer expired"),
949                Timer::PushNewCid => {
950                    // Update `retire_prior_to` field in NEW_CONNECTION_ID frame
951                    let num_new_cid = self.local_cid_state.on_cid_timeout().into();
952                    if !self.state.is_closed() {
953                        trace!(
954                            "push a new cid to peer RETIRE_PRIOR_TO field {}",
955                            self.local_cid_state.retire_prior_to()
956                        );
957                        self.endpoint_events
958                            .push_back(EndpointEventInner::NeedIdentifiers(now, num_new_cid));
959                    }
960                }
961            }
962        }
963    }

1121    fn on_ack_received(
1122        &mut self,
1123        now: Instant,
1124        space: SpaceId,
1125        ack: frame::Ack,
1126    ) -> Result<(), TransportError> {
1127        if ack.largest >= self.spaces[space].next_packet_number {
1128            return Err(TransportError::PROTOCOL_VIOLATION("unsent packet acked"));
1129        }
1130        let new_largest = {
1131            let space = &mut self.spaces[space];
1132            if space
1133                .largest_acked_packet
1134                .map_or(true, |pn| ack.largest > pn)
1135            {
1136                space.largest_acked_packet = Some(ack.largest);
1137                if let Some(info) = space.sent_packets.get(&ack.largest) {
1138                    // This should always succeed, but a misbehaving peer might ACK a packet we
1139                    // haven't sent. At worst, that will result in us spuriously reducing the
1140                    // congestion window.
1141                    space.largest_acked_packet_sent = info.time_sent;
1142                }
1143                true
1144            } else {
1145                false
1146            }
1147        };
1148
1149        // Avoid DoS from unreasonably huge ack ranges by filtering out just the new acks.
1150        let mut newly_acked = ArrayRangeSet::new();
1151        for range in ack.iter() {
1152            for (&pn, _) in self.spaces[space].sent_packets.range(range) {
1153                newly_acked.insert_one(pn);
1154            }
1155        }
1156
1157        if newly_acked.is_empty() {
1158            return Ok(());
1159        }
1160
1161        let mut ack_eliciting_acked = false;
1162        for packet in newly_acked.elts() {
1163            if let Some(info) = self.spaces[space].sent_packets.remove(&packet) {
1164                if let Some(acked) = info.largest_acked {
1165                    // Assume ACKs for all packets below the largest acknowledged in `packet` have
1166                    // been received. This can cause the peer to spuriously retransmit if some of
1167                    // our earlier ACKs were lost, but allows for simpler state tracking. See
1168                    // discussion at
1169                    // https://www.rfc-editor.org/rfc/rfc9000.html#name-limiting-ranges-by-tracking
1170                    self.spaces[space].pending_acks.subtract_below(acked);
1171                }
1172                ack_eliciting_acked |= info.ack_eliciting;
1173                self.on_packet_acked(now, space, info);
1174            }
1175        }
1176
1177        self.path.congestion.on_end_acks(
1178            now,
1179            self.in_flight.bytes,
1180            self.app_limited,
1181            self.spaces[space].largest_acked_packet,
1182        );
1183
1184        if new_largest && ack_eliciting_acked {
1185            let ack_delay = if space != SpaceId::Data {
1186                Duration::from_micros(0)
1187            } else {
1188                cmp::min(
1189                    self.max_ack_delay(),
1190                    Duration::from_micros(ack.delay << self.peer_params.ack_delay_exponent.0),
1191                )
1192            };
1193            let rtt = instant_saturating_sub(now, self.spaces[space].largest_acked_packet_sent);
1194            self.path.rtt.update(ack_delay, rtt);
1195            if self.path.first_packet_after_rtt_sample.is_none() {
1196                self.path.first_packet_after_rtt_sample =
1197                    Some((space, self.spaces[space].next_packet_number));
1198            }
1199        }
1200
1201        // Must be called before crypto/pto_count are clobbered
1202        self.detect_lost_packets(now, space, true);
1203
1204        if self.peer_completed_address_validation() {
1205            self.pto_count = 0;
1206        }
1207
1208        // Explicit congestion notification
1209        if self.path.sending_ecn {
1210            if let Some(ecn) = ack.ecn {
1211                // We only examine ECN counters from ACKs that we are certain we received in transmit
1212                // order, allowing us to compute an increase in ECN counts to compare against the number
1213                // of newly acked packets that remains well-defined in the presence of arbitrary packet
1214                // reordering.
1215                if new_largest {
1216                    let sent = self.spaces[space].largest_acked_packet_sent;
1217                    self.process_ecn(now, space, newly_acked.len() as u64, ecn, sent);
1218                }
1219            } else {
1220                // We always start out sending ECN, so any ack that doesn't acknowledge it disables it.
1221                debug!("ECN not acknowledged by peer");
1222                self.path.sending_ecn = false;
1223            }
1224        }
1225
1226        self.set_loss_detection_timer(now);
1227        Ok(())
1228    }

1230    /// Process a new ECN block from an in-order ACK
1231    fn process_ecn(
1232        &mut self,
1233        now: Instant,
1234        space: SpaceId,
1235        newly_acked: u64,
1236        ecn: frame::EcnCounts,
1237        largest_sent_time: Instant,
1238    ) {
1239        match self.spaces[space].detect_ecn(newly_acked, ecn) {
1240            Err(e) => {
1241                debug!("halting ECN due to verification failure: {}", e);
1242                self.path.sending_ecn = false;
1243                // Wipe out the existing value because it might be garbage and could interfere with
1244                // future attempts to use ECN on new paths.
1245                self.spaces[space].ecn_feedback = frame::EcnCounts::ZERO;
1246            }
1247            Ok(false) => {}
1248            Ok(true) => {
1249                self.stats.path.congestion_events += 1;
1250                self.path
1251                    .congestion
1252                    .on_congestion_event(now, largest_sent_time, false, 0);
1253            }
1254        }
1255    }

1257    // Not timing-aware, so it's safe to call this for inferred acks, such as arise from
1258    // high-latency handshakes
1259    fn on_packet_acked(&mut self, now: Instant, space: SpaceId, info: SentPacket) {
1260        self.remove_in_flight(space, &info);
1261        if info.ack_eliciting && self.path.challenge.is_none() {
1262            // Only pass ACKs to the congestion controller if we are not validating the current
1263            // path, so as to ignore any ACKs from older paths still coming in.
1264            self.path.congestion.on_ack(
1265                now,
1266                info.time_sent,
1267                info.size.into(),
1268                self.app_limited,
1269                &self.path.rtt,
1270            );
1271        }
1272
1273        // Update state for confirmed delivery of frames
1274        if let Some(retransmits) = info.retransmits.get() {
1275            for (id, _) in retransmits.reset_stream.iter() {
1276                self.streams.reset_acked(*id);
1277            }
1278        }
1279
1280        for frame in info.stream_frames {
1281            self.streams.received_ack_of(frame);
1282        }
1283    }

1301    fn on_loss_detection_timeout(&mut self, now: Instant) {
1302        if let Some((_, pn_space)) = self.loss_time_and_space() {
1303            // Time threshold loss Detection
1304            self.detect_lost_packets(now, pn_space, false);
1305            self.set_loss_detection_timer(now);
1306            return;
1307        }
1308
1309        let (_, space) = match self.pto_time_and_space(now) {
1310            Some(x) => x,
1311            None => {

1313                return;
1314            }
1315        };
1316        trace!(
1317            in_flight = self.in_flight.bytes,
1318            count = self.pto_count,
1319            ?space,

1321        );

1327                debug_assert!(!self.peer_completed_address_validation());

1338    fn detect_lost_packets(&mut self, now: Instant, pn_space: SpaceId, due_to_ack: bool) {
1339        let mut lost_packets = Vec::<u64>::new();
1340        let rtt = self.path.rtt.conservative();
1341        let loss_delay = cmp::max(rtt.mul_f32(self.config.time_threshold), TIMER_GRANULARITY);
1342
1343        // Packets sent before this time are deemed lost.
1344        let lost_send_time = now - loss_delay;
1345        let largest_acked_packet = self.spaces[pn_space].largest_acked_packet.unwrap();
1346        let packet_threshold = self.config.packet_threshold as u64;
1347        let mut size_of_lost_packets = 0u64;
1348
1349        // InPersistentCongestion: Determine if all packets in the time period before the newest
1350        // lost packet, including the edges, are marked lost. PTO computation must always
1351        // include max ACK delay, i.e. operate as if in Data space (see RFC9001 7.6.1).
1352        let congestion_period =
1353            self.pto(SpaceId::Data) * self.config.persistent_congestion_threshold;
1354        let mut persistent_congestion_start: Option<Instant> = None;
1355        let mut prev_packet = None;
1356        let mut in_persistent_congestion = false;
1357
1358        let space = &mut self.spaces[pn_space];
1359        space.loss_time = None;
1360
1361        for (&packet, info) in space.sent_packets.range(0..largest_acked_packet) {
1362            if prev_packet != Some(packet.wrapping_sub(1)) {
1363                // An intervening packet was acknowledged
1364                persistent_congestion_start = None;
1365            }
1366
1367            if info.time_sent <= lost_send_time || largest_acked_packet >= packet + packet_threshold
1368            {
1369                lost_packets.push(packet);
1370                size_of_lost_packets += info.size as u64;
1371                if info.ack_eliciting && due_to_ack {
1372                    match persistent_congestion_start {
1373                        // Two ACK-eliciting packets lost more than congestion_period apart, with no
1374                        // ACKed packets in between
1375                        Some(start) if info.time_sent - start > congestion_period => {
1376                            in_persistent_congestion = true;
1377                        }
1378                        // Persistent congestion must start after the first RTT sample
1379                        None if self
1380                            .path
1381                            .first_packet_after_rtt_sample
1382                            .map_or(false, |x| x < (pn_space, packet)) =>
1383                        {
1384                            persistent_congestion_start = Some(info.time_sent);
1385                        }
1386                        _ => {}
1387                    }
1388                }
1389            } else {
1390                let next_loss_time = info.time_sent + loss_delay;
1391                space.loss_time = Some(
1392                    space
1393                        .loss_time
1394                        .map_or(next_loss_time, |x| cmp::min(x, next_loss_time)),
1395                );
1396                persistent_congestion_start = None;
1397            }
1398
1399            prev_packet = Some(packet);
1400        }
1401
1402        // OnPacketsLost
1403        if let Some(largest_lost) = lost_packets.last().cloned() {
1404            let old_bytes_in_flight = self.in_flight.bytes;
1405            let largest_lost_sent = self.spaces[pn_space].sent_packets[&largest_lost].time_sent;
1406            self.lost_packets += lost_packets.len() as u64;
1407            self.stats.path.lost_packets += lost_packets.len() as u64;
1408            self.stats.path.lost_bytes += size_of_lost_packets as u64;
1409            trace!(
1410                "packets lost: {:?}, bytes lost: {}",
1411                lost_packets,
1412                size_of_lost_packets
1413            );
1414            for packet in &lost_packets {
1415                let info = self.spaces[pn_space].sent_packets.remove(packet).unwrap(); // safe: lost_packets is populated just above
1416                self.remove_in_flight(pn_space, &info);
1417                for frame in info.stream_frames {
1418                    self.streams.retransmit(frame);
1419                }
1420                self.spaces[pn_space].pending |= info.retransmits;
1421            }
1422            // Don't apply congestion penalty for lost ack-only packets
1423            let lost_ack_eliciting = old_bytes_in_flight != self.in_flight.bytes;
1424
1425            if lost_ack_eliciting {
1426                self.stats.path.congestion_events += 1;
1427                self.path.congestion.on_congestion_event(
1428                    now,
1429                    largest_lost_sent,
1430                    in_persistent_congestion,
1431                    size_of_lost_packets,
1432                );
1433            }
1434        }
1435    }

1443    fn pto_time_and_space(&self, now: Instant) -> Option<(Instant, SpaceId)> {
1444        let backoff = 2u32.pow(self.pto_count.min(MAX_BACKOFF_EXPONENT));
1445        let mut duration = self.path.rtt.pto_base() * backoff;
1446
1447        if self.in_flight.ack_eliciting == 0 {
1448            debug_assert!(!self.peer_completed_address_validation());
1449            let space = match self.highest_space {
1450                SpaceId::Handshake => SpaceId::Handshake,
1451                _ => SpaceId::Initial,
1452            };
1453            return Some((now + duration, space));
1454        }
1455
1456        let mut result = None;
1457        for space in SpaceId::iter() {
1458            if self.spaces[space].in_flight == 0 {
1459                continue;
1460            }
1461            if space == SpaceId::Data {
1462                // Skip ApplicationData until handshake completes.
1463                if self.is_handshaking() {
1464                    return result;
1465                }
1466                // Include max_ack_delay and backoff for ApplicationData.
1467                duration += self.max_ack_delay() * backoff;
1468            }
1469            let last_ack_eliciting = match self.spaces[space].time_of_last_ack_eliciting_packet {
1470                Some(time) => time,
1471                None => continue,
1472            };
1473            let pto = last_ack_eliciting + duration;
1474            if result.map_or(true, |(earliest_pto, _)| pto < earliest_pto) {
1475                result = Some((pto, space));
1476            }
1477        }
1478        result
1479    }

1481    #[allow(clippy::suspicious_operation_groupings)]
1482    fn peer_completed_address_validation(&self) -> bool {
1483        if self.side.is_server() || self.state.is_closed() {
1484            return true;
1485        }
1486        // The server is guaranteed to have validated our address if any of our handshake or 1-RTT
1487        // packets are acknowledged or we've seen HANDSHAKE_DONE and discarded handshake keys.
1488        self.spaces[SpaceId::Handshake]
1489            .largest_acked_packet
1490            .is_some()
1491            || self.spaces[SpaceId::Data].largest_acked_packet.is_some()
1492            || (self.spaces[SpaceId::Data].crypto.is_some()
1493                && self.spaces[SpaceId::Handshake].crypto.is_none())
1494    }

1496    fn set_loss_detection_timer(&mut self, now: Instant) {
1497        if let Some((loss_time, _)) = self.loss_time_and_space() {
1498            // Time threshold loss detection.
1499            self.timers.set(Timer::LossDetection, loss_time);
1500            return;
1501        }
1502
1503        if self.path.anti_amplification_blocked(1) {
1504            // We wouldn't be able to send anything, so don't bother.
1505            self.timers.stop(Timer::LossDetection);
1506            return;
1507        }
1508
1509        if self.in_flight.ack_eliciting == 0 && self.peer_completed_address_validation() {
1510            // There is nothing to detect lost, so no timer is set. However, the client needs to arm
1511            // the timer if the server might be blocked by the anti-amplification limit.
1512            self.timers.stop(Timer::LossDetection);
1513            return;
1514        }
1515
1516        // Determine which PN space to arm PTO for.
1517        // Calculate PTO duration
1518        if let Some((timeout, _)) = self.pto_time_and_space(now) {
1519            self.timers.set(Timer::LossDetection, timeout);
1520        } else {
1521            self.timers.stop(Timer::LossDetection);
1522        }
1523    }

1534    fn on_packet_authenticated(
1535        &mut self,
1536        now: Instant,
1537        space_id: SpaceId,
1538        ecn: Option<EcnCodepoint>,
1539        packet: Option<u64>,
1540        spin: bool,
1541        is_1rtt: bool,
1542    ) {
1543        self.total_authed_packets += 1;
1544        self.reset_keep_alive(now);
1545        self.reset_idle_timeout(now, space_id);
1546        self.permit_idle_reset = true;
1547        self.receiving_ecn |= ecn.is_some();
1548        if let Some(x) = ecn {
1549            self.spaces[space_id].ecn_counters += x;
1550        }
1551
1552        let packet = match packet {
1553            Some(x) => x,
1554            None => return,
1555        };
1556        if self.side.is_server() {
1557            if self.spaces[SpaceId::Initial].crypto.is_some() && space_id == SpaceId::Handshake {
1558                // A server stops sending and processing Initial packets when it receives its first Handshake packet.
1559                self.discard_space(now, SpaceId::Initial);
1560            }
1561            if self.zero_rtt_crypto.is_some() && is_1rtt {
1562                // Discard 0-RTT keys soon after receiving a 1-RTT packet
1563                self.set_key_discard_timer(now, space_id)
1564            }
1565        }
1566        let space = &mut self.spaces[space_id];
1567        space.pending_acks.insert_one(packet, now);
1568        if packet >= space.rx_packet {
1569            space.rx_packet = packet;
1570            // Update outgoing spin bit, inverting iff we're the client
1571            self.spin = self.side.is_client() ^ spin;
1572        }
1573    }

1608        now: Instant,
1609        remote: SocketAddr,

1611        packet_number: u64,
1612        packet: Packet,
1613        remaining: Option<BytesMut>,

1615        let span = trace_span!("first recv");
1616        let _guard = span.enter();
1617        debug_assert!(self.side.is_server());

1680    fn read_crypto(
1681        &mut self,
1682        space: SpaceId,
1683        crypto: &frame::Crypto,
1684        payload_len: usize,
1685    ) -> Result<(), TransportError> {
1686        let expected = if !self.state.is_handshake() {
1687            SpaceId::Data
1688        } else if self.highest_space == SpaceId::Initial {
1689            SpaceId::Initial
1690        } else {
1691            // On the server, self.highest_space can be Data after receiving the client's first
1692            // flight, but we expect Handshake CRYPTO until the handshake is complete.
1693            SpaceId::Handshake
1694        };
1695        // We can't decrypt Handshake packets when highest_space is Initial, CRYPTO frames in 0-RTT
1696        // packets are illegal, and we don't process 1-RTT packets until the handshake is
1697        // complete. Therefore, we will never see CRYPTO data from a later-than-expected space.
1698        debug_assert!(space <= expected, "received out-of-order CRYPTO data");
1699
1700        let end = crypto.offset + crypto.data.len() as u64;
1701        if space < expected && end > self.spaces[space].crypto_stream.bytes_read() {
1702            warn!(
1703                "received new {:?} CRYPTO data when expecting {:?}",
1704                space, expected
1705            );
1706            return Err(TransportError::PROTOCOL_VIOLATION(
1707                "new data at unexpected encryption level",
1708            ));
1709        }
1710
1711        let space = &mut self.spaces[space];
1712        let max = end.saturating_sub(space.crypto_stream.bytes_read());
1713        if max > self.config.crypto_buffer_size as u64 {
1714            return Err(TransportError::CRYPTO_BUFFER_EXCEEDED(""));
1715        }
1716
1717        space
1718            .crypto_stream
1719            .insert(crypto.offset, crypto.data.clone(), payload_len);
1720        while let Some(chunk) = space.crypto_stream.read(usize::MAX, true) {
1721            trace!("consumed {} CRYPTO bytes", chunk.bytes.len());
1722            if self.crypto.read_handshake(&chunk.bytes)? {
1723                self.events.push_back(Event::HandshakeDataReady);
1724            }
1725        }
1726
1727        Ok(())
1728    }

1730    fn write_crypto(&mut self) {
1731        loop {
1732            let space = self.highest_space;
1733            let mut outgoing = Vec::new();
1734            if let Some(crypto) = self.crypto.write_handshake(&mut outgoing) {
1735                match space {
1736                    SpaceId::Initial => {
1737                        self.upgrade_crypto(SpaceId::Handshake, crypto);
1738                    }
1739                    SpaceId::Handshake => {
1740                        self.upgrade_crypto(SpaceId::Data, crypto);
1741                    }
1742                    _ => unreachable!("got updated secrets during 1-RTT"),
1743                }
1744            }
1745            if outgoing.is_empty() {
1746                if space == self.highest_space {
1747                    break;
1748                } else {
1749                    // Keys updated, check for more data to send
1750                    continue;
1751                }
1752            }
1753            let offset = self.spaces[space].crypto_offset;
1754            let outgoing = Bytes::from(outgoing);
1755            if let State::Handshake(ref mut state) = self.state {
1756                if space == SpaceId::Initial && offset == 0 && self.side.is_client() {
1757                    state.client_hello = Some(outgoing.clone());
1758                }
1759            }
1760            self.spaces[space].crypto_offset += outgoing.len() as u64;
1761            trace!("wrote {} {:?} CRYPTO bytes", outgoing.len(), space);
1762            self.spaces[space].pending.crypto.push_back(frame::Crypto {
1763                offset,
1764                data: outgoing,
1765            });
1766        }
1767    }

1769    /// Switch to stronger cryptography during handshake
1770    fn upgrade_crypto(&mut self, space: SpaceId, crypto: Keys) {
1771        debug_assert!(
1772            self.spaces[space].crypto.is_none(),
1773            "already reached packet space {:?}",
1774            space
1775        );
1776        trace!("{:?} keys ready", space);
1777        if space == SpaceId::Data {
1778            // Precompute the first key update
1779            self.next_crypto = Some(
1780                self.crypto
1781                    .next_1rtt_keys()
1782                    .expect("handshake should be complete"),
1783            );
1784        }
1785
1786        self.spaces[space].crypto = Some(crypto);
1787        debug_assert!(space as usize > self.highest_space as usize);
1788        self.highest_space = space;
1789        if space == SpaceId::Data && self.side.is_client() {
1790            // Discard 0-RTT keys because 1-RTT keys are available.
1791            self.zero_rtt_crypto = None;
1792        }
1793    }

1797        trace!("discarding {:?} keys", space_id);

1813    fn handle_coalesced(
1814        &mut self,
1815        now: Instant,
1816        remote: SocketAddr,
1817        ecn: Option<EcnCodepoint>,
1818        data: BytesMut,
1819    ) {
1820        self.path.total_recvd = self.path.total_recvd.saturating_add(data.len() as u64);
1821        let mut remaining = Some(data);

1827                self.endpoint_config.grease_quic_bit,
1828            ) {
1829                Ok((partial_decode, rest)) => {
1830                    remaining = rest;

1832                }
1833                Err(e) => {

1835                    return;

1841    fn handle_decode(
1842        &mut self,
1843        now: Instant,
1844        remote: SocketAddr,
1845        ecn: Option<EcnCodepoint>,
1846        partial_decode: PartialDecode,
1847    ) {
1848        let header_crypto = if partial_decode.is_0rtt() {
1849            if let Some(ref crypto) = self.zero_rtt_crypto {
1850                Some(&*crypto.header)
1851            } else {
1852                debug!("dropping unexpected 0-RTT packet");
1853                return;
1854            }
1855        } else if let Some(space) = partial_decode.space() {
1856            if let Some(ref crypto) = self.spaces[space].crypto {
1857                Some(&*crypto.header.remote)
1858            } else {
1859                debug!(
1860                    "discarding unexpected {:?} packet ({} bytes)",
1861                    space,
1862                    partial_decode.len(),
1863                );
1864                return;
1865            }
1866        } else {
1867            // Unprotected packet
1868            None
1869        };
1870
1871        let packet = partial_decode.data();
1872        let stateless_reset = packet.len() >= RESET_TOKEN_SIZE + 5
1873            && self.peer_params.stateless_reset_token.as_deref()
1874                == Some(&packet[packet.len() - RESET_TOKEN_SIZE..]);
1875
1876        match partial_decode.finish(header_crypto) {
1877            Ok(packet) => self.handle_packet(now, remote, ecn, Some(packet), stateless_reset),
1878            Err(_) if stateless_reset => self.handle_packet(now, remote, ecn, None, true),
1879            Err(e) => {
1880                trace!("unable to complete packet decoding: {}", e);
1881            }
1882        }
1883    }

1885    fn handle_packet(
1886        &mut self,
1887        now: Instant,
1888        remote: SocketAddr,
1889        ecn: Option<EcnCodepoint>,
1890        packet: Option<Packet>,
1891        stateless_reset: bool,
1892    ) {
1893        if let Some(ref packet) = packet {
1894            trace!(
1895                "got {:?} packet ({} bytes) from {} using id {}",
1896                packet.header.space(),
1897                packet.payload.len() + packet.header_data.len(),
1898                remote,
1899                packet.header.dst_cid(),
1900            );
1901        }
1902
1903        if self.is_handshaking() && remote != self.path.remote {
1904            debug!("discarding packet with unexpected remote during handshake");
1905            return;
1906        }
1907
1908        let was_closed = self.state.is_closed();
1909        let was_drained = self.state.is_drained();
1910
1911        let decrypted = match packet {
1912            None => Err(None),
1913            Some(mut packet) => self
1914                .decrypt_packet(now, &mut packet)
1915                .map(move |number| (packet, number)),
1916        };
1917        let result = match decrypted {
1918            Err(Some(e)) => {
1919                warn!("illegal packet: {}", e);
1920                Err(e.into())
1921            }
1922            Err(None) => {
1923                if stateless_reset {
1924                    debug!("got stateless reset");
1925                    Err(ConnectionError::Reset)
1926                } else {
1927                    debug!("failed to authenticate packet");
1928                    self.authentication_failures += 1;
1929                    let integrity_limit = self.spaces[self.highest_space]
1930                        .crypto
1931                        .as_ref()
1932                        .unwrap()
1933                        .packet
1934                        .local
1935                        .integrity_limit();
1936                    if self.authentication_failures > integrity_limit {
1937                        Err(TransportError::AEAD_LIMIT_REACHED("integrity limit violated").into())
1938                    } else {
1939                        return;
1940                    }
1941                }
1942            }
1943            Ok((packet, number)) => {
1944                let span = match number {
1945                    Some(pn) => trace_span!("recv", space = ?packet.header.space(), pn),
1946                    None => trace_span!("recv", space = ?packet.header.space()),
1947                };
1948                let _guard = span.enter();
1949
1950                let is_duplicate = |n| self.spaces[packet.header.space()].dedup.insert(n);
1951                if number.map_or(false, is_duplicate) {
1952                    if stateless_reset {
1953                        Err(ConnectionError::Reset)
1954                    } else {
1955                        warn!("discarding possible duplicate packet");
1956                        return;
1957                    }
1958                } else if self.state.is_handshake() && packet.header.is_short() {
1959                    // TODO: SHOULD buffer these to improve reordering tolerance.
1960                    trace!("dropping short packet during handshake");
1961                    return;
1962                } else {
1963                    if let Header::Initial { ref token, .. } = packet.header {
1964                        if let State::Handshake(ref hs) = self.state {
1965                            if self.side.is_server() && token != &hs.expected_token {
1966                                // Clients must send the same retry token in every Initial. Initial
1967                                // packets can be spoofed, so we discard rather than killing the
1968                                // connection.
1969                                warn!("discarding Initial with invalid retry token");
1970                                return;
1971                            }
1972                        }
1973                    }
1974
1975                    if !self.state.is_closed() {
1976                        let spin = match packet.header {
1977                            Header::Short { spin, .. } => spin,
1978                            _ => false,
1979                        };
1980                        self.on_packet_authenticated(
1981                            now,
1982                            packet.header.space(),
1983                            ecn,
1984                            number,
1985                            spin,
1986                            packet.header.is_1rtt(),
1987                        );
1988                    }
1989                    self.process_decrypted_packet(now, remote, number, packet)
1990                }
1991            }
1992        };
1993
1994        // State transitions for error cases
1995        if let Err(conn_err) = result {
1996            self.error = Some(conn_err.clone());
1997            self.state = match conn_err {
1998                ConnectionError::ApplicationClosed(reason) => State::closed(reason),
1999                ConnectionError::ConnectionClosed(reason) => State::closed(reason),
2000                ConnectionError::Reset
2001                | ConnectionError::TransportError(TransportError {
2002                    code: TransportErrorCode::AEAD_LIMIT_REACHED,
2003                    ..
2004                }) => State::Drained,
2005                ConnectionError::TimedOut => {
2006                    unreachable!("timeouts aren't generated by packet processing");
2007                }
2008                ConnectionError::TransportError(err) => {
2009                    debug!("closing connection due to transport error: {}", err);
2010                    State::closed(err)
2011                }
2012                ConnectionError::VersionMismatch => State::Draining,
2013                ConnectionError::LocallyClosed => {
2014                    unreachable!("LocallyClosed isn't generated by packet processing")
2015                }
2016            };
2017        }
2018
2019        if !was_closed && self.state.is_closed() {
2020            self.close_common();
2021            if !self.state.is_drained() {
2022                self.set_close_timer(now);
2023            }
2024        }
2025        if !was_drained && self.state.is_drained() {
2026            self.endpoint_events.push_back(EndpointEventInner::Drained);
2027            // Close timer may have been started previously, e.g. if we sent a close and got a
2028            // stateless reset in response
2029            self.timers.stop(Timer::Close);
2030        }
2031
2032        // Transmit CONNECTION_CLOSE if necessary
2033        if let State::Closed(_) = self.state {
2034            self.close = remote == self.path.remote;
2035        }
2036    }

2038    fn process_decrypted_packet(
2039        &mut self,
2040        now: Instant,
2041        remote: SocketAddr,
2042        number: Option<u64>,
2043        packet: Packet,
2044    ) -> Result<(), ConnectionError> {
2045        let state = match self.state {
2046            State::Established => {
2047                match packet.header.space() {
2048                    SpaceId::Data => {
2049                        self.process_payload(now, remote, number.unwrap(), packet.payload.freeze())?
2050                    }
2051                    _ => self.process_early_payload(now, packet)?,
2052                }
2053                return Ok(());
2054            }
2055            State::Closed(_) => {
2056                for result in frame::Iter::new(packet.payload.freeze()) {
2057                    let frame = match result {
2058                        Ok(frame) => frame,
2059                        Err(err) => {
2060                            debug!("frame decoding error: {err:?}");
2061                            continue;
2062                        }
2063                    };
2064
2065                    if let Frame::Padding = frame {
2066                        continue;
2067                    };
2068
2069                    self.stats.frame_rx.record(&frame);
2070
2071                    if let Frame::Close(_) = frame {
2072                        trace!("draining");
2073                        self.state = State::Draining;
2074                        break;
2075                    }
2076                }
2077                return Ok(());
2078            }
2079            State::Draining | State::Drained => return Ok(()),
2080            State::Handshake(ref mut state) => state,
2081        };
2082
2083        match packet.header {
2084            Header::Retry {
2085                src_cid: rem_cid, ..
2086            } => {
2087                if self.side.is_server() {
2088                    return Err(TransportError::PROTOCOL_VIOLATION("client sent Retry").into());
2089                }
2090
2091                if self.total_authed_packets > 1
2092                            || packet.payload.len() <= 16 // token + 16 byte tag
2093                            || !self.crypto.is_valid_retry(
2094                                &self.rem_cids.active(),
2095                                &packet.header_data,
2096                                &packet.payload,
2097                            )
2098                {
2099                    trace!("discarding invalid Retry");
2100                    // - After the client has received and processed an Initial or Retry
2101                    //   packet from the server, it MUST discard any subsequent Retry
2102                    //   packets that it receives.
2103                    // - A client MUST discard a Retry packet with a zero-length Retry Token
2104                    //   field.
2105                    // - Clients MUST discard Retry packets that have a Retry Integrity Tag
2106                    //   that cannot be validated
2107                    return Ok(());
2108                }
2109
2110                trace!("retrying with CID {}", rem_cid);
2111                let client_hello = state.client_hello.take().unwrap();
2112                self.retry_src_cid = Some(rem_cid);
2113                self.rem_cids.update_initial_cid(rem_cid);
2114                self.rem_handshake_cid = rem_cid;
2115
2116                let space = &mut self.spaces[SpaceId::Initial];
2117                if let Some(info) = space.sent_packets.remove(&0) {
2118                    self.on_packet_acked(now, SpaceId::Initial, info);
2119                };
2120
2121                self.discard_space(now, SpaceId::Initial); // Make sure we clean up after any retransmitted Initials
2122                self.spaces[SpaceId::Initial] = PacketSpace {
2123                    crypto: Some(self.crypto.initial_keys(&rem_cid, self.side)),
2124                    next_packet_number: self.spaces[SpaceId::Initial].next_packet_number,
2125                    crypto_offset: client_hello.len() as u64,
2126                    ..PacketSpace::new(now)
2127                };
2128                self.spaces[SpaceId::Initial]
2129                    .pending
2130                    .crypto
2131                    .push_back(frame::Crypto {
2132                        offset: 0,
2133                        data: client_hello,
2134                    });
2135
2136                // Retransmit all 0-RTT data
2137                let zero_rtt = mem::take(&mut self.spaces[SpaceId::Data].sent_packets);
2138                for (_, info) in zero_rtt {
2139                    self.remove_in_flight(SpaceId::Data, &info);
2140                    self.spaces[SpaceId::Data].pending |= info.retransmits;
2141                }
2142                self.streams.retransmit_all_for_0rtt();
2143
2144                let token_len = packet.payload.len() - 16;
2145                self.retry_token = packet.payload.freeze().split_to(token_len);
2146                self.state = State::Handshake(state::Handshake {
2147                    expected_token: Bytes::new(),
2148                    rem_cid_set: false,
2149                    client_hello: None,
2150                });
2151                Ok(())
2152            }
2153            Header::Long {
2154                ty: LongType::Handshake,
2155                src_cid: rem_cid,
2156                ..
2157            } => {
2158                if rem_cid != self.rem_handshake_cid {
2159                    debug!(
2160                        "discarding packet with mismatched remote CID: {} != {}",
2161                        self.rem_handshake_cid, rem_cid
2162                    );
2163                    return Ok(());
2164                }
2165                self.path.validated = true;
2166
2167                self.process_early_payload(now, packet)?;
2168                if self.state.is_closed() {
2169                    return Ok(());
2170                }
2171
2172                if self.crypto.is_handshaking() {
2173                    trace!("handshake ongoing");
2174                    return Ok(());
2175                }
2176
2177                if self.side.is_client() {
2178                    // Client-only beceause server params were set from the client's Initial
2179                    let params =
2180                        self.crypto
2181                            .transport_parameters()?
2182                            .ok_or_else(|| TransportError {
2183                                code: TransportErrorCode::crypto(0x6d),
2184                                frame: None,
2185                                reason: "transport parameters missing".into(),
2186                            })?;
2187
2188                    if self.has_0rtt() {
2189                        if !self.crypto.early_data_accepted().unwrap() {
2190                            debug_assert!(self.side.is_client());
2191                            debug!("0-RTT rejected");
2192                            self.accepted_0rtt = false;
2193                            self.streams.zero_rtt_rejected();
2194
2195                            // Discard already-queued frames
2196                            self.spaces[SpaceId::Data].pending = Retransmits::default();
2197
2198                            // Discard 0-RTT packets
2199                            let sent_packets =
2200                                mem::take(&mut self.spaces[SpaceId::Data].sent_packets);
2201                            for (_, packet) in sent_packets {
2202                                self.remove_in_flight(SpaceId::Data, &packet);
2203                            }
2204                        } else {
2205                            self.accepted_0rtt = true;
2206                            params.validate_resumption_from(&self.peer_params)?;
2207                        }
2208                    }
2209                    if let Some(token) = params.stateless_reset_token {
2210                        self.endpoint_events
2211                            .push_back(EndpointEventInner::ResetToken(self.path.remote, token));
2212                    }
2213                    self.handle_peer_params(params)?;
2214                    self.issue_cids(now);
2215                } else {
2216                    // Server-only
2217                    self.spaces[SpaceId::Data].pending.handshake_done = true;
2218                    self.discard_space(now, SpaceId::Handshake);
2219                }
2220
2221                self.events.push_back(Event::Connected);
2222                self.state = State::Established;
2223                trace!("established");
2224                Ok(())
2225            }
2226            Header::Initial {
2227                src_cid: rem_cid, ..
2228            } => {
2229                if !state.rem_cid_set {
2230                    trace!("switching remote CID to {}", rem_cid);
2231                    let mut state = state.clone();
2232                    self.rem_cids.update_initial_cid(rem_cid);
2233                    self.rem_handshake_cid = rem_cid;
2234                    self.orig_rem_cid = rem_cid;
2235                    state.rem_cid_set = true;
2236                    self.state = State::Handshake(state);
2237                } else if rem_cid != self.rem_handshake_cid {
2238                    debug!(
2239                        "discarding packet with mismatched remote CID: {} != {}",
2240                        self.rem_handshake_cid, rem_cid
2241                    );
2242                    return Ok(());
2243                }
2244
2245                let starting_space = self.highest_space;
2246                self.process_early_payload(now, packet)?;
2247
2248                if self.side.is_server()
2249                    && starting_space == SpaceId::Initial
2250                    && self.highest_space != SpaceId::Initial
2251                {
2252                    let params =
2253                        self.crypto
2254                            .transport_parameters()?
2255                            .ok_or_else(|| TransportError {
2256                                code: TransportErrorCode::crypto(0x6d),
2257                                frame: None,
2258                                reason: "transport parameters missing".into(),
2259                            })?;
2260                    self.handle_peer_params(params)?;
2261                    self.issue_cids(now);
2262                    self.init_0rtt();
2263                }
2264                Ok(())
2265            }
2266            Header::Long {
2267                ty: LongType::ZeroRtt,
2268                ..
2269            } => {
2270                self.process_payload(now, remote, number.unwrap(), packet.payload.freeze())?;
2271                Ok(())
2272            }
2273            Header::VersionNegotiate { .. } => {
2274                if self.total_authed_packets > 1 {
2275                    return Ok(());
2276                }
2277                let supported = packet
2278                    .payload
2279                    .chunks(4)
2280                    .any(|x| match <[u8; 4]>::try_from(x) {
2281                        Ok(version) => self.version == u32::from_be_bytes(version),
2282                        Err(_) => false,
2283                    });
2284                if supported {
2285                    return Ok(());
2286                }
2287                debug!("remote doesn't support our version");
2288                Err(ConnectionError::VersionMismatch)
2289            }
2290            Header::Short { .. } => unreachable!(
2291                "short packets received during handshake are discarded in handle_packet"
2292            ),
2293        }
2294    }

2296    /// Process an Initial or Handshake packet payload
2297    fn process_early_payload(
2298        &mut self,
2299        now: Instant,
2300        packet: Packet,
2301    ) -> Result<(), TransportError> {
2302        debug_assert_ne!(packet.header.space(), SpaceId::Data);
2303        let payload_len = packet.payload.len();
2304        let mut ack_eliciting = false;
2305        for result in frame::Iter::new(packet.payload.freeze()) {
2306            let frame = result?;
2307            let span = match frame {
2308                Frame::Padding => continue,
2309                _ => Some(trace_span!("frame", ty = %frame.ty())),
2310            };
2311
2312            self.stats.frame_rx.record(&frame);
2313
2314            let _guard = span.as_ref().map(|x| x.enter());
2315            ack_eliciting |= frame.is_ack_eliciting();
2316
2317            // Process frames
2318            match frame {
2319                Frame::Padding | Frame::Ping => {}
2320                Frame::Crypto(frame) => {
2321                    self.read_crypto(packet.header.space(), &frame, payload_len)?;
2322                }
2323                Frame::Ack(ack) => {
2324                    self.on_ack_received(now, packet.header.space(), ack)?;
2325                }
2326                Frame::Close(reason) => {
2327                    self.error = Some(reason.into());
2328                    self.state = State::Draining;
2329                    return Ok(());
2330                }
2331                _ => {
2332                    let mut err =
2333                        TransportError::PROTOCOL_VIOLATION("illegal frame type in handshake");
2334                    err.frame = Some(frame.ty());
2335                    return Err(err);
2336                }
2337            }
2338        }
2339        self.spaces[packet.header.space()]
2340            .pending_acks
2341            .packet_received(ack_eliciting);
2342
2343        self.write_crypto();
2344        Ok(())
2345    }

2347    fn process_payload(
2348        &mut self,
2349        now: Instant,
2350        remote: SocketAddr,
2351        number: u64,
2352        payload: Bytes,
2353    ) -> Result<(), TransportError> {
2354        let is_0rtt = self.spaces[SpaceId::Data].crypto.is_none();
2355        let mut is_probing_packet = true;
2356        let mut close = None;
2357        let payload_len = payload.len();
2358        let mut ack_eliciting = false;
2359        for result in frame::Iter::new(payload) {
2360            let frame = result?;
2361            let span = match frame {
2362                Frame::Padding => continue,
2363                _ => Some(trace_span!("frame", ty = %frame.ty())),
2364            };
2365
2366            self.stats.frame_rx.record(&frame);
2367            // Crypto, Stream and Datagram frames are special cased in order no pollute
2368            // the log with payload data
2369            match &frame {
2370                Frame::Crypto(f) => {
2371                    trace!(offset = f.offset, len = f.data.len(), "got crypto frame");
2372                }
2373                Frame::Stream(f) => {
2374                    trace!(id = %f.id, offset = f.offset, len = f.data.len(), fin = f.fin, "got stream frame");
2375                }
2376                Frame::Datagram(f) => {
2377                    trace!(len = f.data.len(), "got datagram frame");
2378                }
2379                f => {
2380                    trace!("got frame {:?}", f);
2381                }
2382            }
2383
2384            let _guard = span.as_ref().map(|x| x.enter());
2385            if is_0rtt {
2386                match frame {
2387                    Frame::Crypto(_) | Frame::Close(Close::Application(_)) => {
2388                        return Err(TransportError::PROTOCOL_VIOLATION(
2389                            "illegal frame type in 0-RTT",
2390                        ));
2391                    }
2392                    _ => {}
2393                }
2394            }
2395            ack_eliciting |= frame.is_ack_eliciting();
2396
2397            // Check whether this could be a probing packet
2398            match frame {
2399                Frame::Padding
2400                | Frame::PathChallenge(_)
2401                | Frame::PathResponse(_)
2402                | Frame::NewConnectionId(_) => {}
2403                _ => {
2404                    is_probing_packet = false;
2405                }
2406            }
2407            match frame {
2408                Frame::Crypto(frame) => {
2409                    self.read_crypto(SpaceId::Data, &frame, payload_len)?;
2410                }
2411                Frame::Stream(frame) => {
2412                    if self.streams.received(frame, payload_len)?.should_transmit() {
2413                        self.spaces[SpaceId::Data].pending.max_data = true;
2414                    }
2415                }
2416                Frame::Ack(ack) => {
2417                    self.on_ack_received(now, SpaceId::Data, ack)?;
2418                }
2419                Frame::Padding | Frame::Ping => {}
2420                Frame::Close(reason) => {
2421                    close = Some(reason);
2422                }
2423                Frame::PathChallenge(token) => {
2424                    if self
2425                        .path_response
2426                        .as_ref()
2427                        .map_or(true, |x| x.packet <= number)
2428                    {
2429                        self.path_response = Some(PathResponse {
2430                            packet: number,
2431                            token,
2432                        });
2433                    }
2434                    if remote == self.path.remote {
2435                        // PATH_CHALLENGE on active path, possible off-path packet forwarding
2436                        // attack. Send a non-probing packet to recover the active path.
2437                        self.ping();
2438                    }
2439                }
2440                Frame::PathResponse(token) => {
2441                    if self.path.challenge == Some(token) && remote == self.path.remote {
2442                        trace!("new path validated");
2443                        self.timers.stop(Timer::PathValidation);
2444                        self.path.challenge = None;
2445                        self.path.validated = true;
2446                        if let Some(ref mut prev_path) = self.prev_path {
2447                            prev_path.challenge = None;
2448                            prev_path.challenge_pending = false;
2449                        }
2450                    } else {
2451                        debug!(token, "ignoring invalid PATH_RESPONSE");
2452                    }
2453                }
2454                Frame::MaxData(bytes) => {
2455                    self.streams.received_max_data(bytes);
2456                }
2457                Frame::MaxStreamData { id, offset } => {
2458                    self.streams.received_max_stream_data(id, offset)?;
2459                }
2460                Frame::MaxStreams { dir, count } => {
2461                    self.streams.received_max_streams(dir, count)?;
2462                }
2463                Frame::ResetStream(frame) => {
2464                    if self.streams.received_reset(frame)?.should_transmit() {
2465                        self.spaces[SpaceId::Data].pending.max_data = true;
2466                    }
2467                }
2468                Frame::DataBlocked { offset } => {
2469                    debug!(offset, "peer claims to be blocked at connection level");
2470                }
2471                Frame::StreamDataBlocked { id, offset } => {
2472                    if id.initiator() == self.side && id.dir() == Dir::Uni {
2473                        debug!("got STREAM_DATA_BLOCKED on send-only {}", id);
2474                        return Err(TransportError::STREAM_STATE_ERROR(
2475                            "STREAM_DATA_BLOCKED on send-only stream",
2476                        ));
2477                    }
2478                    debug!(
2479                        stream = %id,
2480                        offset, "peer claims to be blocked at stream level"
2481                    );
2482                }
2483                Frame::StreamsBlocked { dir, limit } => {
2484                    if limit > MAX_STREAM_COUNT {
2485                        return Err(TransportError::FRAME_ENCODING_ERROR(
2486                            "unrepresentable stream limit",
2487                        ));
2488                    }
2489                    debug!(
2490                        "peer claims to be blocked opening more than {} {} streams",
2491                        limit, dir
2492                    );
2493                }
2494                Frame::StopSending(frame::StopSending { id, error_code }) => {
2495                    if id.initiator() != self.side {
2496                        if id.dir() == Dir::Uni {
2497                            debug!("got STOP_SENDING on recv-only {}", id);
2498                            return Err(TransportError::STREAM_STATE_ERROR(
2499                                "STOP_SENDING on recv-only stream",
2500                            ));
2501                        }
2502                    } else if self.streams.is_local_unopened(id) {
2503                        return Err(TransportError::STREAM_STATE_ERROR(
2504                            "STOP_SENDING on unopened stream",
2505                        ));
2506                    }
2507                    self.streams.received_stop_sending(id, error_code);
2508                }
2509                Frame::RetireConnectionId { sequence } => {
2510                    let allow_more_cids = self
2511                        .local_cid_state
2512                        .on_cid_retirement(sequence, self.peer_params.issue_cids_limit())?;
2513                    self.endpoint_events
2514                        .push_back(EndpointEventInner::RetireConnectionId(
2515                            now,
2516                            sequence,
2517                            allow_more_cids,
2518                        ));
2519                }
2520                Frame::NewConnectionId(frame) => {
2521                    trace!(
2522                        sequence = frame.sequence,
2523                        id = %frame.id,
2524                        retire_prior_to = frame.retire_prior_to,
2525                    );
2526                    if self.rem_cids.active().is_empty() {
2527                        return Err(TransportError::PROTOCOL_VIOLATION(
2528                            "NEW_CONNECTION_ID when CIDs aren't in use",
2529                        ));
2530                    }
2531                    if frame.retire_prior_to > frame.sequence {
2532                        return Err(TransportError::PROTOCOL_VIOLATION(
2533                            "NEW_CONNECTION_ID retiring unissued CIDs",
2534                        ));
2535                    }
2536
2537                    use crate::cid_queue::InsertError;
2538                    match self.rem_cids.insert(frame) {
2539                        Ok(None) => {}
2540                        Ok(Some((retired, reset_token))) => {
2541                            self.spaces[SpaceId::Data]
2542                                .pending
2543                                .retire_cids
2544                                .extend(retired);
2545                            self.set_reset_token(reset_token);
2546                        }
2547                        Err(InsertError::ExceedsLimit) => {
2548                            return Err(TransportError::CONNECTION_ID_LIMIT_ERROR(""));
2549                        }
2550                        Err(InsertError::Retired) => {
2551                            trace!("discarding already-retired");
2552                            // RETIRE_CONNECTION_ID might not have been previously sent if e.g. a
2553                            // range of connection IDs larger than the active connection ID limit
2554                            // was retired all at once via retire_prior_to.
2555                            self.spaces[SpaceId::Data]
2556                                .pending
2557                                .retire_cids
2558                                .push(frame.sequence);
2559                            continue;
2560                        }
2561                    };
2562
2563                    if self.side.is_server() && self.rem_cids.active_seq() == 0 {
2564                        // We're a server still using the initial remote CID for the client, so
2565                        // let's switch immediately to enable clientside stateless resets.
2566                        self.update_rem_cid();
2567                    }
2568                }
2569                Frame::NewToken { token } => {
2570                    if self.side.is_server() {
2571                        return Err(TransportError::PROTOCOL_VIOLATION("client sent NEW_TOKEN"));
2572                    }
2573                    if token.is_empty() {
2574                        return Err(TransportError::FRAME_ENCODING_ERROR("empty token"));
2575                    }
2576                    trace!("got new token");
2577                    // TODO: Cache, or perhaps forward to user?
2578                }
2579                Frame::Datagram(datagram) => {
2580                    if self
2581                        .datagrams
2582                        .received(datagram, &self.config.datagram_receive_buffer_size)?
2583                    {
2584                        self.events.push_back(Event::DatagramReceived);
2585                    }
2586                }
2587                Frame::HandshakeDone => {
2588                    if self.side.is_server() {
2589                        return Err(TransportError::PROTOCOL_VIOLATION(
2590                            "client sent HANDSHAKE_DONE",
2591                        ));
2592                    }
2593                    if self.spaces[SpaceId::Handshake].crypto.is_some() {
2594                        self.discard_space(now, SpaceId::Handshake);
2595                    }
2596                }
2597            }
2598        }
2599
2600        self.spaces[SpaceId::Data]
2601            .pending_acks
2602            .packet_received(ack_eliciting);
2603
2604        // Issue stream ID credit due to ACKs of outgoing finish/resets and incoming finish/resets
2605        // on stopped streams. Incoming finishes/resets on open streams are not handled here as they
2606        // are only freed, and hence only issue credit, once the application has been notified
2607        // during a read on the stream.
2608        let pending = &mut self.spaces[SpaceId::Data].pending;
2609        for dir in Dir::iter() {
2610            if self.streams.take_max_streams_dirty(dir) {
2611                pending.max_stream_id[dir as usize] = true;
2612            }
2613        }
2614
2615        if let Some(reason) = close {
2616            self.error = Some(reason.into());
2617            self.state = State::Draining;
2618            self.close = true;
2619        }
2620
2621        if remote != self.path.remote
2622            && !is_probing_packet
2623            && number == self.spaces[SpaceId::Data].rx_packet
2624        {
2625            debug_assert!(
2626                self.server_config
2627                    .as_ref()
2628                    .expect("packets from unknown remote should be dropped by clients")
2629                    .migration,
2630                "migration-initiating packets should have been dropped immediately"
2631            );
2632            self.migrate(now, remote);
2633            // Break linkability, if possible
2634            self.update_rem_cid();
2635            self.spin = false;
2636        }
2637
2638        Ok(())
2639    }

2641    fn migrate(&mut self, now: Instant, remote: SocketAddr) {
2642        trace!(%remote, "migration initiated");
2643        // Reset rtt/congestion state for new path unless it looks like a NAT rebinding.
2644        // Note that the congestion window will not grow until validation terminates. Helps mitigate
2645        // amplification attacks performed by spoofing source addresses.
2646        let mut new_path = if remote.is_ipv4() && remote.ip() == self.path.remote.ip() {
2647            PathData::from_previous(remote, &self.path, now)
2648        } else {
2649            PathData::new(
2650                remote,
2651                self.config.initial_rtt,
2652                self.config.congestion_controller_factory.build(now),
2653                self.config.initial_max_udp_payload_size.min(
2654                    u16::try_from(self.peer_params.max_udp_payload_size.into_inner())
2655                        .unwrap_or(u16::MAX),
2656                ),
2657                now,
2658                false,
2659            )
2660        };
2661        new_path.challenge = Some(self.rng.gen());
2662        new_path.challenge_pending = true;
2663        let prev_pto = self.pto(SpaceId::Data);
2664
2665        let mut prev = mem::replace(&mut self.path, new_path);
2666        // Don't clobber the original path if the previous one hasn't been validated yet
2667        if prev.challenge.is_none() {
2668            prev.challenge = Some(self.rng.gen());
2669            prev.challenge_pending = true;
2670            self.prev_path = Some(prev);
2671        }
2672
2673        self.timers.set(
2674            Timer::PathValidation,
2675            now + 3 * cmp::max(self.pto(SpaceId::Data), prev_pto),
2676        );
2677    }

2715    fn populate_packet(
2716        &mut self,
2717        space_id: SpaceId,
2718        buf: &mut Vec<u8>,
2719        max_size: usize,
2720    ) -> SentFrames {
2721        let mut sent = SentFrames::default();
2722        let space = &mut self.spaces[space_id];
2723        let is_0rtt = space_id == SpaceId::Data && space.crypto.is_none();
2724
2725        // HANDSHAKE_DONE
2726        if !is_0rtt && mem::replace(&mut space.pending.handshake_done, false) {
2727            buf.write(frame::Type::HANDSHAKE_DONE);
2728            sent.retransmits.get_or_create().handshake_done = true;
2729            // This is just a u8 counter and the frame is typically just sent once
2730            self.stats.frame_tx.handshake_done =
2731                self.stats.frame_tx.handshake_done.saturating_add(1);
2732        }
2733
2734        // PING
2735        if mem::replace(&mut space.ping_pending, false) {
2736            trace!("PING");
2737            buf.write(frame::Type::PING);
2738            sent.non_retransmits = true;
2739            self.stats.frame_tx.ping += 1;
2740        }
2741
2742        // ACK
2743        if space.pending_acks.can_send() {
2744            debug_assert!(!space.pending_acks.ranges().is_empty());
2745            Self::populate_acks(self.receiving_ecn, &mut sent, space, buf, &mut self.stats);
2746        }
2747
2748        // PATH_CHALLENGE
2749        if buf.len() + 9 < max_size && space_id == SpaceId::Data {
2750            // Transmit challenges with every outgoing frame on an unvalidated path
2751            if let Some(token) = self.path.challenge {
2752                // But only send a packet solely for that purpose at most once
2753                self.path.challenge_pending = false;
2754                sent.non_retransmits = true;
2755                sent.requires_padding = true;
2756                trace!("PATH_CHALLENGE {:08x}", token);
2757                buf.write(frame::Type::PATH_CHALLENGE);
2758                buf.write(token);
2759                self.stats.frame_tx.path_challenge += 1;
2760            }
2761        }
2762
2763        // PATH_RESPONSE
2764        if buf.len() + 9 < max_size && space_id == SpaceId::Data {
2765            if let Some(response) = self.path_response.take() {
2766                sent.non_retransmits = true;
2767                sent.requires_padding = true;
2768                trace!("PATH_RESPONSE {:08x}", response.token);
2769                buf.write(frame::Type::PATH_RESPONSE);
2770                buf.write(response.token);
2771                self.stats.frame_tx.path_response += 1;
2772            }
2773        }
2774
2775        // CRYPTO
2776        while buf.len() + frame::Crypto::SIZE_BOUND < max_size && !is_0rtt {
2777            let mut frame = match space.pending.crypto.pop_front() {
2778                Some(x) => x,
2779                None => break,
2780            };
2781
2782            // Calculate the maximum amount of crypto data we can store in the buffer.
2783            // Since the offset is known, we can reserve the exact size required to encode it.
2784            // For length we reserve 2bytes which allows to encode up to 2^14,
2785            // which is more than what fits into normally sized QUIC frames.
2786            let max_crypto_data_size = max_size
2787                - buf.len()
2788                - 1 // Frame Type
2789                - VarInt::size(unsafe { VarInt::from_u64_unchecked(frame.offset) })
2790                - 2; // Maximum encoded length for frame size, given we send less than 2^14 bytes
2791
2792            let len = frame
2793                .data
2794                .len()
2795                .min(2usize.pow(14) - 1)
2796                .min(max_crypto_data_size);
2797
2798            let data = frame.data.split_to(len);
2799            let truncated = frame::Crypto {
2800                offset: frame.offset,
2801                data,
2802            };
2803            trace!(
2804                "CRYPTO: off {} len {}",
2805                truncated.offset,
2806                truncated.data.len()
2807            );
2808            truncated.encode(buf);
2809            self.stats.frame_tx.crypto += 1;
2810            sent.retransmits.get_or_create().crypto.push_back(truncated);
2811            if !frame.data.is_empty() {
2812                frame.offset += len as u64;
2813                space.pending.crypto.push_front(frame);
2814            }
2815        }
2816
2817        if space_id == SpaceId::Data {
2818            self.streams.write_control_frames(
2819                buf,
2820                &mut space.pending,
2821                &mut sent.retransmits,
2822                &mut self.stats.frame_tx,
2823                max_size,
2824            );
2825        }
2826
2827        // NEW_CONNECTION_ID
2828        while buf.len() + 44 < max_size {
2829            let issued = match space.pending.new_cids.pop() {
2830                Some(x) => x,
2831                None => break,
2832            };
2833            trace!(
2834                sequence = issued.sequence,
2835                id = %issued.id,
2836                "NEW_CONNECTION_ID"
2837            );
2838            frame::NewConnectionId {
2839                sequence: issued.sequence,
2840                retire_prior_to: self.local_cid_state.retire_prior_to(),
2841                id: issued.id,
2842                reset_token: issued.reset_token,
2843            }
2844            .encode(buf);
2845            sent.retransmits.get_or_create().new_cids.push(issued);
2846            self.stats.frame_tx.new_connection_id += 1;
2847        }
2848
2849        // RETIRE_CONNECTION_ID
2850        while buf.len() + frame::RETIRE_CONNECTION_ID_SIZE_BOUND < max_size {
2851            let seq = match space.pending.retire_cids.pop() {
2852                Some(x) => x,
2853                None => break,
2854            };
2855            trace!(sequence = seq, "RETIRE_CONNECTION_ID");
2856            buf.write(frame::Type::RETIRE_CONNECTION_ID);
2857            buf.write_var(seq);
2858            sent.retransmits.get_or_create().retire_cids.push(seq);
2859            self.stats.frame_tx.retire_connection_id += 1;
2860        }
2861
2862        // DATAGRAM
2863        while buf.len() + Datagram::SIZE_BOUND < max_size && space_id == SpaceId::Data {
2864            match self.datagrams.write(buf, max_size) {
2865                true => {
2866                    sent.non_retransmits = true;
2867                    self.stats.frame_tx.datagram += 1;
2868                }
2869                false => break,
2870            }
2871        }
2872
2873        // STREAM
2874        if space_id == SpaceId::Data {
2875            sent.stream_frames = self.streams.write_stream_frames(buf, max_size);
2876            self.stats.frame_tx.stream += sent.stream_frames.len() as u64;
2877        }
2878
2879        sent
2880    }

2883    ///
2884    /// This method assumes ACKs are pending, and should only be called if

2966    fn decrypt_packet(
2967        &mut self,
2968        now: Instant,
2969        packet: &mut Packet,
2970    ) -> Result<Option<u64>, Option<TransportError>> {
2971        if !packet.header.is_protected() {
2972            // Unprotected packets also don't have packet numbers
2973            return Ok(None);
2974        }
2975        let space = packet.header.space();
2976        let rx_packet = self.spaces[space].rx_packet;
2977        let number = packet.header.number().ok_or(None)?.expand(rx_packet + 1);
2978        let key_phase = packet.header.key_phase();
2979
2980        let mut crypto_update = false;
2981        let crypto = if packet.header.is_0rtt() {
2982            &self.zero_rtt_crypto.as_ref().unwrap().packet
2983        } else if key_phase == self.key_phase || space != SpaceId::Data {
2984            &self.spaces[space].crypto.as_mut().unwrap().packet.remote
2985        } else if let Some(prev) = self.prev_crypto.as_ref().and_then(|crypto| {
2986            // If this packet comes prior to acknowledgment of the key update by the peer,
2987            if crypto.end_packet.map_or(true, |(pn, _)| number < pn) {
2988                // use the previous keys.
2989                Some(crypto)
2990            } else {
2991                // Otherwise, this must be a remotely-initiated key update, so fall through to the
2992                // final case.
2993                None
2994            }
2995        }) {
2996            &prev.crypto.remote
2997        } else {
2998            // We're in the Data space with a key phase mismatch and either there is no locally
2999            // initiated key update or the locally initiated key update was acknowledged by a
3000            // lower-numbered packet. The key phase mismatch must therefore represent a new
3001            // remotely-initiated key update.
3002            crypto_update = true;
3003            &self.next_crypto.as_ref().unwrap().remote
3004        };
3005
3006        crypto
3007            .decrypt(number, &packet.header_data, &mut packet.payload)
3008            .map_err(|_| {
3009                trace!("decryption failed with packet number {}", number);
3010                None
3011            })?;
3012
3013        if let Some(ref mut prev) = self.prev_crypto {
3014            if prev.end_packet.is_none() && key_phase == self.key_phase {
3015                // Outgoing key update newly acknowledged
3016                prev.end_packet = Some((number, now));
3017                self.set_key_discard_timer(now, space);
3018            }
3019        }
3020
3021        if !packet.reserved_bits_valid() {
3022            return Err(Some(TransportError::PROTOCOL_VIOLATION(
3023                "reserved bits set",
3024            )));
3025        }
3026
3027        if crypto_update {
3028            // Validate and commit incoming key update
3029            if number <= rx_packet
3030                || self
3031                    .prev_crypto
3032                    .as_ref()
3033                    .map_or(false, |x| x.update_unacked)
3034            {
3035                return Err(Some(TransportError::KEY_UPDATE_ERROR("")));
3036            }
3037            trace!("key update authenticated");
3038            self.update_keys(Some((number, now)), true);
3039            self.set_key_discard_timer(now, space);
3040        }
3041
3042        Ok(Some(number))
3043    }

3045    fn update_keys(&mut self, end_packet: Option<(u64, Instant)>, remote: bool) {
3046        // Generate keys for the key phase after the one we're switching to, store them in
3047        // `next_crypto`, make the contents of `next_crypto` current, and move the current keys into
3048        // `prev_crypto`.
3049        let new = self
3050            .crypto
3051            .next_1rtt_keys()
3052            .expect("only called for `Data` packets");
3053        let old = mem::replace(
3054            &mut self.spaces[SpaceId::Data]
3055                .crypto
3056                .as_mut()
3057                .unwrap() // safe because update_keys() can only be triggered by short packets
3058                .packet,
3059            mem::replace(self.next_crypto.as_mut().unwrap(), new),
3060        );
3061        self.spaces[SpaceId::Data].sent_with_keys = 0;
3062        self.prev_crypto = Some(PrevCrypto {
3063            crypto: old,
3064            end_packet,
3065            update_unacked: remote,
3066        });
3067        self.key_phase = !self.key_phase;
