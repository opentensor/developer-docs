8/// A simple, standard congestion controller
9#[derive(Debug, Clone)]
10pub struct NewReno {
11    config: Arc<NewRenoConfig>,
12    /// Maximum number of bytes in flight that may be sent.
13    window: u64,
14    /// Slow start threshold in bytes. When the congestion window is below ssthresh, the mode is
15    /// slow start and the window grows by the number of bytes acknowledged.
16    ssthresh: u64,
17    /// The time when QUIC first detects a loss, causing it to enter recovery. When a packet sent
18    /// after this time is acknowledged, QUIC exits recovery.
19    recovery_start_time: Instant,
20    /// Bytes which had been acked by the peer since leaving slow start
21    bytes_acked: u64,
22}
23
24impl NewReno {
25    /// Construct a state using the given `config` and current time `now`
26    pub fn new(config: Arc<NewRenoConfig>, now: Instant) -> Self {
27        Self {
28            window: config.initial_window,
29            ssthresh: u64::max_value(),
30            recovery_start_time: now,
31            config,
32            bytes_acked: 0,
33        }
34    }
35}
36
37impl Controller for NewReno {
38    fn on_ack(
39        &mut self,
40        _now: Instant,
41        sent: Instant,
42        bytes: u64,
43        app_limited: bool,
44        _rtt: &RttEstimator,
45    ) {
46        if app_limited || sent <= self.recovery_start_time {
47            return;
48        }
49
50        if self.window < self.ssthresh {
51            // Slow start
52            self.window += bytes;
53
54            if self.window >= self.ssthresh {
55                // Exiting slow start
56                // Initialize `bytes_acked` for congestion avoidance. The idea
57                // here is that any bytes over `sshthresh` will already be counted
58                // towards the congestion avoidance phase - independent of when
59                // how close to `sshthresh` the `window` was when switching states,
60                // and independent of datagram sizes.
61                self.bytes_acked = self.window - self.ssthresh;
62            }
63        } else {
64            // Congestion avoidance
65            // This implementation uses the method which does not require
66            // floating point math, which also increases the window by 1 datagram
67            // for every round trip.
68            // This mechanism is called Appropriate Byte Counting in
69            // https://tools.ietf.org/html/rfc3465
70            self.bytes_acked += bytes;
71
72            if self.bytes_acked >= self.window {
73                self.bytes_acked -= self.window;
74                self.window += self.config.max_datagram_size;
75            }
76        }
77    }
78
79    fn on_congestion_event(
80        &mut self,
81        now: Instant,
82        sent: Instant,
83        is_persistent_congestion: bool,
84        _lost_bytes: u64,
85    ) {
86        if sent <= self.recovery_start_time {
87            return;
88        }
89
90        self.recovery_start_time = now;
91        self.window = (self.window as f32 * self.config.loss_reduction_factor) as u64;
92        self.window = self.window.max(self.config.minimum_window);
93        self.ssthresh = self.window;
94
95        if is_persistent_congestion {
96            self.window = self.config.minimum_window;
97        }
98    }
99
100    fn window(&self) -> u64 {
101        self.window
102    }
103
104    fn clone_box(&self) -> Box<dyn Controller> {
105        Box::new(self.clone())
106    }
107
108    fn initial_window(&self) -> u64 {
109        self.config.initial_window
110    }
111
112    fn into_any(self: Box<Self>) -> Box<dyn Any> {
113        self
114    }
115}

117/// Configuration for the `NewReno` congestion controller
118#[derive(Debug, Clone)]
119pub struct NewRenoConfig {
120    max_datagram_size: u64,
121    initial_window: u64,
122    minimum_window: u64,
123    loss_reduction_factor: f32,
124}
125
126impl NewRenoConfig {
127    /// The sender’s maximum UDP payload size. Does not include UDP or IP overhead.
128    ///
129    /// Used for calculating initial and minimum congestion windows.
130    pub fn max_datagram_size(&mut self, value: u64) -> &mut Self {
131        self.max_datagram_size = value;
132        self
133    }
134
135    /// Default limit on the amount of outstanding data in bytes.
136    ///
137    /// Recommended value: `min(10 * max_datagram_size, max(2 * max_datagram_size, 14720))`
138    pub fn initial_window(&mut self, value: u64) -> &mut Self {
139        self.initial_window = value;
140        self
141    }
142
143    /// Default minimum congestion window.
144    ///
145    /// Recommended value: `2 * max_datagram_size`.
146    pub fn minimum_window(&mut self, value: u64) -> &mut Self {
147        self.minimum_window = value;
148        self
149    }
150
151    /// Reduction in congestion window when a new loss event is detected.
152    pub fn loss_reduction_factor(&mut self, value: f32) -> &mut Self {
153        self.loss_reduction_factor = value;
154        self
155    }
156}
157
158impl Default for NewRenoConfig {
159    fn default() -> Self {
160        const MAX_DATAGRAM_SIZE: u64 = 1232;
161        Self {
162            max_datagram_size: MAX_DATAGRAM_SIZE,
163            initial_window: 14720.max(2 * MAX_DATAGRAM_SIZE).min(10 * MAX_DATAGRAM_SIZE),
164            minimum_window: 2 * MAX_DATAGRAM_SIZE,
165            loss_reduction_factor: 0.5,
166        }
167    }
168}
169
170impl ControllerFactory for Arc<NewRenoConfig> {
171    fn build(&self, now: Instant) -> Box<dyn Controller> {
172        Box::new(NewReno::new(self.clone(), now))
173    }
