1use std::fmt::{Debug, Display, Formatter};
2use std::time::{Duration, Instant};
3
4use super::min_max::MinMax;
5
6#[derive(Clone, Debug)]
7pub(crate) struct BandwidthEstimation {
8    total_acked: u64,
9    prev_total_acked: u64,
10    acked_time: Option<Instant>,
11    prev_acked_time: Option<Instant>,
12    total_sent: u64,
13    prev_total_sent: u64,
14    sent_time: Instant,
15    prev_sent_time: Option<Instant>,
16    max_filter: MinMax,
17    acked_at_last_window: u64,
18}
19
20impl BandwidthEstimation {
21    pub fn on_sent(&mut self, now: Instant, bytes: u64) {
22        self.prev_total_sent = self.total_sent;
23        self.total_sent += bytes;
24        self.prev_sent_time = Some(self.sent_time);
25        self.sent_time = now;
26    }
27
28    pub fn on_ack(
29        &mut self,
30        now: Instant,
31        _sent: Instant,
32        bytes: u64,
33        round: u64,
34        app_limited: bool,
35    ) {
36        self.prev_total_acked = self.total_acked;
37        self.total_acked += bytes;
38        self.prev_acked_time = self.acked_time;
39        self.acked_time = Some(now);
40
41        let prev_sent_time = match self.prev_sent_time {
42            Some(prev_sent_time) => prev_sent_time,
43            None => return,
44        };
45
46        let send_rate = if self.sent_time > prev_sent_time {
47            BandwidthEstimation::bw_from_delta(
48                self.total_sent - self.prev_total_sent,
49                self.sent_time - prev_sent_time,
50            )
51            .unwrap_or(0)
52        } else {
53            u64::MAX // will take the min of send and ack, so this is just a skip
54        };
55
56        let ack_rate = match self.prev_acked_time {
57            Some(prev_acked_time) => BandwidthEstimation::bw_from_delta(
58                self.total_acked - self.prev_total_acked,
59                now - prev_acked_time,
60            )
61            .unwrap_or(0),
62            None => 0,
63        };
64
65        let bandwidth = send_rate.min(ack_rate);
66        if !app_limited && self.max_filter.get() < bandwidth {
67            self.max_filter.update_max(round, bandwidth);
68        }
69    }
70
71    pub fn bytes_acked_this_window(&self) -> u64 {
72        self.total_acked - self.acked_at_last_window
73    }
74
75    pub fn end_acks(&mut self, _current_round: u64, _app_limited: bool) {
76        self.acked_at_last_window = self.total_acked;
77    }
78
79    pub fn get_estimate(&self) -> u64 {
80        self.max_filter.get()
81    }
82
83    pub const fn bw_from_delta(bytes: u64, delta: Duration) -> Option<u64> {
84        let window_duration_ns = delta.as_nanos();
85        if window_duration_ns == 0 {
86            return None;
87        }
88        let b_ns = bytes * 1_000_000_000;
89        let bytes_per_second = b_ns / (window_duration_ns as u64);
90        Some(bytes_per_second)
91    }
92}
93
94impl Default for BandwidthEstimation {
95    fn default() -> Self {
96        BandwidthEstimation {
97            total_acked: 0,
98            prev_total_acked: 0,
99            acked_time: None,
100            prev_acked_time: None,
101            total_sent: 0,
102            prev_total_sent: 0,
103            // The `sent_time` value set here is ignored; it is used in `on_ack()`, but will
104            // have been reset by `on_sent()` before that method is called.
105            sent_time: Instant::now(),
106            prev_sent_time: None,
107            max_filter: MinMax::default(),
108            acked_at_last_window: 0,
109        }
110    }
111}
112
113impl Display for BandwidthEstimation {
114    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
115        write!(
116            f,
117            "{:.3} MB/s",
118            self.get_estimate() as f32 / (1024 * 1024) as f32
119        )
120    }
