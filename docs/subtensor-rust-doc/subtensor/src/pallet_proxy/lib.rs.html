18//! # Proxy Pallet
19//! A pallet allowing accounts to give permission to other accounts to dispatch types of calls from
20//! their signed origin.
21//!
22//! The accounts to which permission is delegated may be required to announce the action that they
23//! wish to execute some duration prior to execution happens. In this case, the target account may
24//! reject the announcement and in doing so, veto the execution.
25//!
26//! - [`Config`]
27//! - [`Call`]

65/// The parameters under which a particular account has a proxy relationship with some other
66/// account.
67#[derive(
68    Encode,
69    Decode,
70    Clone,

80#[freeze_struct("a37bb67fe5520678")]
81pub struct ProxyDefinition<AccountId, ProxyType, BlockNumber> {
82    /// The account which may act on behalf of another.
83    pub delegate: AccountId,
84    /// A value defining the subset of calls that it is allowed to make.
85    pub proxy_type: ProxyType,
86    /// The number of blocks that an announcement must be in place for before the corresponding
87    /// call may be dispatched. If zero, then no announcement is needed.
88    pub delay: BlockNumber,
89}

91/// Details surrounding a specific instance of an announcement to make a call.
92#[derive(Encode, Decode, Clone, Copy, Eq, PartialEq, RuntimeDebug, MaxEncodedLen, TypeInfo)]

94pub struct Announcement<AccountId, Hash, BlockNumber> {
95    /// The account which made the announcement.
96    real: AccountId,
97    /// The hash of the call to be made.
98    call_hash: Hash,
99    /// The height at which the announcement was made.
100    height: BlockNumber,
101}

112    /// Configuration trait.
113    #[pallet::config]
114    pub trait Config: frame_system::Config {
115        /// The overarching event type.
116        type RuntimeEvent: From<Event<Self>> + IsType<<Self as frame_system::Config>::RuntimeEvent>;
117
118        /// The overarching call type.
119        type RuntimeCall: Parameter
120            + Dispatchable<RuntimeOrigin = Self::RuntimeOrigin>
121            + GetDispatchInfo
122            + From<frame_system::Call<Self>>
123            + IsSubType<Call<Self>>
124            + IsType<<Self as frame_system::Config>::RuntimeCall>;
125
126        /// The currency mechanism.
127        type Currency: ReservableCurrency<Self::AccountId>;
128
129        /// A kind of proxy; specified with the proxy and passed in to the `IsProxyable` filter.
130        /// The instance filter determines whether a given call may be proxied under this type.
131        ///
132        /// IMPORTANT: `Default` must be provided and MUST BE the the *most permissive* value.
133        type ProxyType: Parameter
134            + Member
135            + Ord
136            + PartialOrd
137            + InstanceFilter<<Self as Config>::RuntimeCall>
138            + Default
139            + MaxEncodedLen;
140
141        /// The base amount of currency needed to reserve for creating a proxy.
142        ///
143        /// This is held for an additional storage item whose value size is
144        /// `sizeof(Balance)` bytes and whose key size is `sizeof(AccountId)` bytes.
145        #[pallet::constant]
146        type ProxyDepositBase: Get<BalanceOf<Self>>;
147
148        /// The amount of currency needed per proxy added.
149        ///
150        /// This is held for adding 32 bytes plus an instance of `ProxyType` more into a
151        /// pre-existing storage value. Thus, when configuring `ProxyDepositFactor` one should take
152        /// into account `32 + proxy_type.encode().len()` bytes of data.
153        #[pallet::constant]
154        type ProxyDepositFactor: Get<BalanceOf<Self>>;
155
156        /// The maximum amount of proxies allowed for a single account.
157        #[pallet::constant]
158        type MaxProxies: Get<u32>;
159
160        /// Weight information for extrinsics in this pallet.
161        type WeightInfo: WeightInfo;
162
163        /// The maximum amount of time-delayed announcements that are allowed to be pending.
164        #[pallet::constant]
165        type MaxPending: Get<u32>;
166
167        /// The type of hash used for hashing the call.
168        type CallHasher: Hash;
169
170        /// The base amount of currency needed to reserve for creating an announcement.
171        ///
172        /// This is held when a new storage item holding a `Balance` is created (typically 16
173        /// bytes).
174        #[pallet::constant]
175        type AnnouncementDepositBase: Get<BalanceOf<Self>>;
176
177        /// The amount of currency needed per announcement made.
178        ///
179        /// This is held for adding an `AccountId`, `Hash` and `BlockNumber` (typically 68 bytes)
180        /// into a pre-existing storage value.
181        #[pallet::constant]
182        type AnnouncementDepositFactor: Get<BalanceOf<Self>>;
183    }

185    #[pallet::call]
186    impl<T: Config> Pallet<T> {
187        /// Dispatch the given `call` from an account that the sender is authorised for through
188        /// `add_proxy`.
189        ///
190        /// The dispatch origin for this call must be _Signed_.
191        ///
192        /// Parameters:
193        /// - `real`: The account that the proxy will make a call on behalf of.
194        /// - `force_proxy_type`: Specify the exact proxy type to be used and checked for this call.
195        /// - `call`: The call to be made by the `real` account.
196        #[pallet::call_index(0)]
197        #[pallet::weight({
198			let di = call.get_dispatch_info();
199			let inner_call_weight = match di.pays_fee {
200				Pays::Yes => di.weight,
201				Pays::No => Weight::zero(),
202			};
203			let base_weight = T::WeightInfo::proxy(T::MaxProxies::get())
204				.saturating_add(T::DbWeight::get().reads_writes(1, 1));
205			(base_weight.saturating_add(inner_call_weight), di.class)
206		})]
207        pub fn proxy(
208            origin: OriginFor<T>,
209            real: AccountIdLookupOf<T>,
210            force_proxy_type: Option<T::ProxyType>,
211            call: Box<<T as Config>::RuntimeCall>,
212        ) -> DispatchResult {
213            let who = ensure_signed(origin)?;
214            let real = T::Lookup::lookup(real)?;
215            let def = Self::find_proxy(&real, &who, force_proxy_type)?;
216            ensure!(def.delay.is_zero(), Error::<T>::Unannounced);
217
218            Self::do_proxy(def, real, *call);
219
220            Ok(())
221        }

223        /// Register a proxy account for the sender that is able to make calls on its behalf.
224        ///
225        /// The dispatch origin for this call must be _Signed_.
226        ///
227        /// Parameters:
228        /// - `proxy`: The account that the `caller` would like to make a proxy.
229        /// - `proxy_type`: The permissions allowed for this proxy account.
230        /// - `delay`: The announcement period required of the initial proxy. Will generally be
231        /// zero.
232        #[pallet::call_index(1)]
233        #[pallet::weight(T::WeightInfo::add_proxy(T::MaxProxies::get()))]
234        pub fn add_proxy(
235            origin: OriginFor<T>,
236            delegate: AccountIdLookupOf<T>,
237            proxy_type: T::ProxyType,
238            delay: BlockNumberFor<T>,
239        ) -> DispatchResult {
240            let who = ensure_signed(origin)?;
241            let delegate = T::Lookup::lookup(delegate)?;
242            Self::add_proxy_delegate(&who, delegate, proxy_type, delay)
243        }
244
245        /// Unregister a proxy account for the sender.
246        ///
247        /// The dispatch origin for this call must be _Signed_.
248        ///
249        /// Parameters:
250        /// - `proxy`: The account that the `caller` would like to remove as a proxy.
251        /// - `proxy_type`: The permissions currently enabled for the removed proxy account.
252        #[pallet::call_index(2)]
253        #[pallet::weight(T::WeightInfo::remove_proxy(T::MaxProxies::get()))]
254        pub fn remove_proxy(
255            origin: OriginFor<T>,
256            delegate: AccountIdLookupOf<T>,
257            proxy_type: T::ProxyType,
258            delay: BlockNumberFor<T>,
259        ) -> DispatchResult {
260            let who = ensure_signed(origin)?;
261            let delegate = T::Lookup::lookup(delegate)?;
262            Self::remove_proxy_delegate(&who, delegate, proxy_type, delay)
263        }
264
265        /// Unregister all proxy accounts for the sender.
266        ///
267        /// The dispatch origin for this call must be _Signed_.
268        ///
269        /// WARNING: This may be called on accounts created by `pure`, however if done, then
270        /// the unreserved fees will be inaccessible. **All access to this account will be lost.**
271        #[pallet::call_index(3)]
272        #[pallet::weight(T::WeightInfo::remove_proxies(T::MaxProxies::get()))]
273        pub fn remove_proxies(origin: OriginFor<T>) -> DispatchResult {
274            let who = ensure_signed(origin)?;
275            Self::remove_all_proxy_delegates(&who);
276            Ok(())
277        }

279        /// Spawn a fresh new account that is guaranteed to be otherwise inaccessible, and
280        /// initialize it with a proxy of `proxy_type` for `origin` sender.
281        ///
282        /// Requires a `Signed` origin.
283        ///
284        /// - `proxy_type`: The type of the proxy that the sender will be registered as over the
285        /// new account. This will almost always be the most permissive `ProxyType` possible to
286        ///   allow for maximum flexibility.
287        /// - `index`: A disambiguation index, in case this is called multiple times in the same
288        ///   transaction (e.g. with `utility::batch`). Unless you're using `batch` you probably just
289        ///   want to use `0`.
290        /// - `delay`: The announcement period required of the initial proxy. Will generally be
291        ///   zero.
292        ///
293        /// Fails with `Duplicate` if this has already been called in this transaction, from the
294        /// same sender, with the same parameters.
295        ///
296        /// Fails if there are insufficient funds to pay for deposit.
297        #[pallet::call_index(4)]
298        #[pallet::weight(T::WeightInfo::create_pure(T::MaxProxies::get()))]
299        pub fn create_pure(
300            origin: OriginFor<T>,
301            proxy_type: T::ProxyType,
302            delay: BlockNumberFor<T>,
303            index: u16,
304        ) -> DispatchResult {
305            let who = ensure_signed(origin)?;
306
307            let pure = Self::pure_account(&who, &proxy_type, index, None);
308            ensure!(!Proxies::<T>::contains_key(&pure), Error::<T>::Duplicate);
309
310            let proxy_def = ProxyDefinition {
311                delegate: who.clone(),
312                proxy_type: proxy_type.clone(),
313                delay,
314            };
315            let bounded_proxies: BoundedVec<_, T::MaxProxies> = vec![proxy_def]
316                .try_into()
317                .map_err(|_| Error::<T>::TooMany)?;
318
319            let deposit = T::ProxyDepositBase::get().saturating_add(T::ProxyDepositFactor::get());
320            T::Currency::reserve(&who, deposit)?;
321
322            Proxies::<T>::insert(&pure, (bounded_proxies, deposit));
323            Self::deposit_event(Event::PureCreated {
324                pure,
325                who,
326                proxy_type,
327                disambiguation_index: index,
328            });
329
330            Ok(())
331        }
332
333        /// Removes a previously spawned pure proxy.
334        ///
335        /// WARNING: **All access to this account will be lost.** Any funds held in it will be
336        /// inaccessible.
337        ///
338        /// Requires a `Signed` origin, and the sender account must have been created by a call to
339        /// `pure` with corresponding parameters.
340        ///
341        /// - `spawner`: The account that originally called `pure` to create this account.
342        /// - `index`: The disambiguation index originally passed to `pure`. Probably `0`.
343        /// - `proxy_type`: The proxy type originally passed to `pure`.
344        /// - `height`: The height of the chain when the call to `pure` was processed.
345        /// - `ext_index`: The extrinsic index in which the call to `pure` was processed.
346        ///
347        /// Fails with `NoPermission` in case the caller is not a previously created pure
348        /// account whose `pure` call has corresponding parameters.
349        #[pallet::call_index(5)]
350        #[pallet::weight(T::WeightInfo::kill_pure(T::MaxProxies::get()))]
351        pub fn kill_pure(
352            origin: OriginFor<T>,
353            spawner: AccountIdLookupOf<T>,
354            proxy_type: T::ProxyType,
355            index: u16,
356            #[pallet::compact] height: BlockNumberFor<T>,
357            #[pallet::compact] ext_index: u32,
358        ) -> DispatchResult {
359            let who = ensure_signed(origin)?;
360            let spawner = T::Lookup::lookup(spawner)?;
361
362            let when = (height, ext_index);
363            let proxy = Self::pure_account(&spawner, &proxy_type, index, Some(when));
364            ensure!(proxy == who, Error::<T>::NoPermission);
365
366            let (_, deposit) = Proxies::<T>::take(&who);
367            T::Currency::unreserve(&spawner, deposit);
368
369            Ok(())
370        }

372        /// Publish the hash of a proxy-call that will be made in the future.
373        ///
374        /// This must be called some number of blocks before the corresponding `proxy` is attempted
375        /// if the delay associated with the proxy relationship is greater than zero.
376        ///
377        /// No more than `MaxPending` announcements may be made at any one time.
378        ///
379        /// This will take a deposit of `AnnouncementDepositFactor` as well as
380        /// `AnnouncementDepositBase` if there are no other pending announcements.
381        ///
382        /// The dispatch origin for this call must be _Signed_ and a proxy of `real`.
383        ///
384        /// Parameters:
385        /// - `real`: The account that the proxy will make a call on behalf of.
386        /// - `call_hash`: The hash of the call to be made by the `real` account.
387        #[pallet::call_index(6)]
388        #[pallet::weight(T::WeightInfo::announce(T::MaxPending::get(), T::MaxProxies::get()))]
389        pub fn announce(
390            origin: OriginFor<T>,
391            real: AccountIdLookupOf<T>,
392            call_hash: CallHashOf<T>,
393        ) -> DispatchResult {
394            let who = ensure_signed(origin)?;
395            let real = T::Lookup::lookup(real)?;
396            Proxies::<T>::get(&real)
397                .0
398                .into_iter()
399                .find(|x| x.delegate == who)
400                .ok_or(Error::<T>::NotProxy)?;
401
402            let announcement = Announcement {
403                real: real.clone(),
404                call_hash,
405                height: system::Pallet::<T>::block_number(),
406            };
407
408            Announcements::<T>::try_mutate(&who, |(ref mut pending, ref mut deposit)| {
409                pending
410                    .try_push(announcement)
411                    .map_err(|_| Error::<T>::TooMany)?;
412                Self::rejig_deposit(
413                    &who,
414                    *deposit,
415                    T::AnnouncementDepositBase::get(),
416                    T::AnnouncementDepositFactor::get(),
417                    pending.len(),
418                )
419                .map(|d| {
420                    d.expect("Just pushed; pending.len() > 0; rejig_deposit returns Some; qed")
421                })
422                .map(|d| *deposit = d)
423            })?;
424            Self::deposit_event(Event::Announced {
425                real,
426                proxy: who,
427                call_hash,
428            });
429
430            Ok(())
431        }
432
433        /// Remove a given announcement.
434        ///
435        /// May be called by a proxy account to remove a call they previously announced and return
436        /// the deposit.
437        ///
438        /// The dispatch origin for this call must be _Signed_.
439        ///
440        /// Parameters:
441        /// - `real`: The account that the proxy will make a call on behalf of.
442        /// - `call_hash`: The hash of the call to be made by the `real` account.
443        #[pallet::call_index(7)]
444        #[pallet::weight(T::WeightInfo::remove_announcement(
445            T::MaxPending::get(),
446            T::MaxProxies::get()
447        ))]
448        pub fn remove_announcement(
449            origin: OriginFor<T>,
450            real: AccountIdLookupOf<T>,
451            call_hash: CallHashOf<T>,
452        ) -> DispatchResult {
453            let who = ensure_signed(origin)?;
454            let real = T::Lookup::lookup(real)?;
455            Self::edit_announcements(&who, |ann| ann.real != real || ann.call_hash != call_hash)?;
456
457            Ok(())
458        }
459
460        /// Remove the given announcement of a delegate.
461        ///
462        /// May be called by a target (proxied) account to remove a call that one of their delegates
463        /// (`delegate`) has announced they want to execute. The deposit is returned.
464        ///
465        /// The dispatch origin for this call must be _Signed_.
466        ///
467        /// Parameters:
468        /// - `delegate`: The account that previously announced the call.
469        /// - `call_hash`: The hash of the call to be made.
470        #[pallet::call_index(8)]
471        #[pallet::weight(T::WeightInfo::reject_announcement(
472            T::MaxPending::get(),
473            T::MaxProxies::get()
474        ))]
475        pub fn reject_announcement(
476            origin: OriginFor<T>,
477            delegate: AccountIdLookupOf<T>,
478            call_hash: CallHashOf<T>,
479        ) -> DispatchResult {
480            let who = ensure_signed(origin)?;
481            let delegate = T::Lookup::lookup(delegate)?;
482            Self::edit_announcements(&delegate, |ann| {
483                ann.real != who || ann.call_hash != call_hash
484            })?;
485
486            Ok(())
487        }
488
489        /// Dispatch the given `call` from an account that the sender is authorized for through
490        /// `add_proxy`.
491        ///
492        /// Removes any corresponding announcement(s).
493        ///
494        /// The dispatch origin for this call must be _Signed_.
495        ///
496        /// Parameters:
497        /// - `real`: The account that the proxy will make a call on behalf of.
498        /// - `force_proxy_type`: Specify the exact proxy type to be used and checked for this call.
499        /// - `call`: The call to be made by the `real` account.
500        #[pallet::call_index(9)]
501        #[pallet::weight({
502			let di = call.get_dispatch_info();
503			(T::WeightInfo::proxy_announced(T::MaxPending::get(), T::MaxProxies::get())
504				 // AccountData for inner call origin accountdata.
505				.saturating_add(T::DbWeight::get().reads_writes(1, 1))
506				.saturating_add(di.weight),
507			di.class)
508		})]
509        pub fn proxy_announced(
510            origin: OriginFor<T>,
511            delegate: AccountIdLookupOf<T>,
512            real: AccountIdLookupOf<T>,
513            force_proxy_type: Option<T::ProxyType>,
514            call: Box<<T as Config>::RuntimeCall>,
515        ) -> DispatchResult {
516            ensure_signed(origin)?;
517            let delegate = T::Lookup::lookup(delegate)?;
518            let real = T::Lookup::lookup(real)?;
519            let def = Self::find_proxy(&real, &delegate, force_proxy_type)?;
520
521            let call_hash = T::CallHasher::hash_of(&call);
522            let now = system::Pallet::<T>::block_number();
523            Self::edit_announcements(&delegate, |ann| {
524                ann.real != real
525                    || ann.call_hash != call_hash
526                    || now.saturating_sub(ann.height) < def.delay
527            })
528            .map_err(|_| Error::<T>::Unannounced)?;
529
530            Self::do_proxy(def, real, *call);
531
532            Ok(())
533        }
534    }

571    #[pallet::error]
572    pub enum Error<T> {
573        /// There are too many proxies registered or too many announcements pending.
574        TooMany,
575        /// Proxy registration not found.
576        NotFound,
577        /// Sender is not a proxy of the account to be proxied.
578        NotProxy,
579        /// A call which is incompatible with the proxy type's filter was attempted.
580        Unproxyable,
581        /// Account is already a proxy.
582        Duplicate,
583        /// Call may not be made by proxy because it may escalate its privileges.
584        NoPermission,
585        /// Announcement, if made at all, was made too recently.
586        Unannounced,
587        /// Cannot add self as proxy.
588        NoSelfProxy,
589    }

643    /// Calculate the address of an pure account.
644    ///
645    /// - `who`: The spawner account.
646    /// - `proxy_type`: The type of the proxy that the sender will be registered as over the
647    ///   new account. This will almost always be the most permissive `ProxyType` possible to
648    ///   allow for maximum flexibility.
649    /// - `index`: A disambiguation index, in case this is called multiple times in the same
650    ///   transaction (e.g. with `utility::batch`). Unless you're using `batch` you probably just
651    ///   want to use `0`.
652    /// - `maybe_when`: The block height and extrinsic index of when the pure account was
653    ///   created. None to use current block height and extrinsic index.
654    pub fn pure_account(
655        who: &T::AccountId,
656        proxy_type: &T::ProxyType,
657        index: u16,
658        maybe_when: Option<(BlockNumberFor<T>, u32)>,
659    ) -> T::AccountId {
660        let (height, ext_index) = maybe_when.unwrap_or_else(|| {
661            (
662                system::Pallet::<T>::block_number(),
663                system::Pallet::<T>::extrinsic_index().unwrap_or_default(),
664            )
665        });
666        let entropy = (
667            b"modlpy/proxy____",
668            who,
669            height,
670            ext_index,
671            proxy_type,
672            index,
673        )
674            .using_encoded(blake2_256);
675        Decode::decode(&mut TrailingZeroInput::new(entropy.as_ref()))
676            .expect("infinite length input; no invalid inputs for type; qed")
677    }

679    /// Register a proxy account for the delegator that is able to make calls on its behalf.
680    ///
681    /// Parameters:
682    /// - `delegator`: The delegator account.
683    /// - `delegatee`: The account that the `delegator` would like to make a proxy.
684    /// - `proxy_type`: The permissions allowed for this proxy account.
685    /// - `delay`: The announcement period required of the initial proxy. Will generally be
686    ///   zero.
687    pub fn add_proxy_delegate(
688        delegator: &T::AccountId,
689        delegatee: T::AccountId,
690        proxy_type: T::ProxyType,
691        delay: BlockNumberFor<T>,
692    ) -> DispatchResult {
693        ensure!(delegator != &delegatee, Error::<T>::NoSelfProxy);
694        Proxies::<T>::try_mutate(delegator, |(ref mut proxies, ref mut deposit)| {
695            let proxy_def = ProxyDefinition {
696                delegate: delegatee.clone(),
697                proxy_type: proxy_type.clone(),
698                delay,
699            };
700            let i = proxies
701                .binary_search(&proxy_def)
702                .err()
703                .ok_or(Error::<T>::Duplicate)?;
704            proxies
705                .try_insert(i, proxy_def)
706                .map_err(|_| Error::<T>::TooMany)?;
707            let new_deposit = Self::deposit(proxies.len() as u32);
708            match new_deposit.cmp(deposit) {
709                Ordering::Greater => {
710                    T::Currency::reserve(delegator, new_deposit.saturating_sub(*deposit))?;
711                }
712                Ordering::Less => {
713                    T::Currency::unreserve(delegator, deposit.saturating_sub(new_deposit));
714                }
715                Ordering::Equal => (),
716            }
717            *deposit = new_deposit;
718            Self::deposit_event(Event::<T>::ProxyAdded {
719                delegator: delegator.clone(),
720                delegatee,
721                proxy_type,
722                delay,
723            });
724            Ok(())
725        })
726    }

728    /// Unregister a proxy account for the delegator.
729    ///
730    /// Parameters:
731    /// - `delegator`: The delegator account.
732    /// - `delegatee`: The account that the `delegator` would like to make a proxy.
733    /// - `proxy_type`: The permissions allowed for this proxy account.
734    /// - `delay`: The announcement period required of the initial proxy. Will generally be
735    ///   zero.
736    pub fn remove_proxy_delegate(
737        delegator: &T::AccountId,
738        delegatee: T::AccountId,
739        proxy_type: T::ProxyType,
740        delay: BlockNumberFor<T>,
741    ) -> DispatchResult {
742        Proxies::<T>::try_mutate_exists(delegator, |x| {
743            let (mut proxies, old_deposit) = x.take().ok_or(Error::<T>::NotFound)?;
744            let proxy_def = ProxyDefinition {
745                delegate: delegatee.clone(),
746                proxy_type: proxy_type.clone(),
747                delay,
748            };
749            let i = proxies
750                .binary_search(&proxy_def)
751                .ok()
752                .ok_or(Error::<T>::NotFound)?;
753            proxies.remove(i);
754            let new_deposit = Self::deposit(proxies.len() as u32);
755            match new_deposit.cmp(&old_deposit) {
756                Ordering::Greater => {
757                    T::Currency::reserve(delegator, new_deposit.saturating_sub(old_deposit))?;
758                }
759                Ordering::Less => {
760                    T::Currency::unreserve(delegator, old_deposit.saturating_sub(new_deposit));
761                }
762                Ordering::Equal => (),
763            }
764            if !proxies.is_empty() {
765                *x = Some((proxies, new_deposit))
766            }
767            Self::deposit_event(Event::<T>::ProxyRemoved {
768                delegator: delegator.clone(),
769                delegatee,
770                proxy_type,
771                delay,
772            });
773            Ok(())
774        })
775    }

777    pub fn deposit(num_proxies: u32) -> BalanceOf<T> {
778        if num_proxies == 0 {
779            Zero::zero()
780        } else {
781            T::ProxyDepositBase::get()
782                .saturating_add(T::ProxyDepositFactor::get().saturating_mul(num_proxies.into()))
783        }
784    }
785
786    fn rejig_deposit(
787        who: &T::AccountId,
788        old_deposit: BalanceOf<T>,
789        base: BalanceOf<T>,
790        factor: BalanceOf<T>,
791        len: usize,
792    ) -> Result<Option<BalanceOf<T>>, DispatchError> {
793        let new_deposit = if len == 0 {
794            BalanceOf::<T>::zero()
795        } else {
796            base.saturating_add(factor.saturating_mul((len as u32).into()))
797        };
798        match new_deposit.cmp(&old_deposit) {
799            Ordering::Greater => {
800                T::Currency::reserve(who, new_deposit.saturating_sub(old_deposit))?;
801            }
802            Ordering::Less => {
803                T::Currency::unreserve(who, old_deposit.saturating_sub(new_deposit));
804            }
805            Ordering::Equal => (),
806        }
807        Ok(if len == 0 { None } else { Some(new_deposit) })
808    }

810    fn edit_announcements<
811        F: FnMut(&Announcement<T::AccountId, CallHashOf<T>, BlockNumberFor<T>>) -> bool,
812    >(
813        delegate: &T::AccountId,
814        mut f: F,
815    ) -> DispatchResult {
816        Announcements::<T>::try_mutate_exists(delegate, |x| {
817            let (mut pending, old_deposit) = x.take().ok_or(Error::<T>::NotFound)?;
818            let orig_pending_len = pending.len();
819            pending.retain(&mut f);
820            ensure!(orig_pending_len > pending.len(), Error::<T>::NotFound);
821            *x = Self::rejig_deposit(
822                delegate,
823                old_deposit,
824                T::AnnouncementDepositBase::get(),
825                T::AnnouncementDepositFactor::get(),
826                pending.len(),
827            )?
828            .map(|deposit| (pending, deposit));
829            Ok(())
830        })
831    }

833    pub fn find_proxy(
834        real: &T::AccountId,
835        delegate: &T::AccountId,
836        force_proxy_type: Option<T::ProxyType>,
837    ) -> Result<ProxyDefinition<T::AccountId, T::ProxyType, BlockNumberFor<T>>, DispatchError> {
838        let f = |x: &ProxyDefinition<T::AccountId, T::ProxyType, BlockNumberFor<T>>| -> bool {
839            &x.delegate == delegate && force_proxy_type.as_ref().is_none_or(|y| &x.proxy_type == y)
840        };
841        Ok(Proxies::<T>::get(real)
842            .0
843            .into_iter()
844            .find(f)
845            .ok_or(Error::<T>::NotProxy)?)
846    }
847
848    fn do_proxy(
849        def: ProxyDefinition<T::AccountId, T::ProxyType, BlockNumberFor<T>>,
850        real: T::AccountId,
851        call: <T as Config>::RuntimeCall,
852    ) {
853        // This is a freshly authenticated new account, the origin restrictions doesn't apply.
854        let mut origin: T::RuntimeOrigin = frame_system::RawOrigin::Signed(real).into();
855        origin.add_filter(move |c: &<T as frame_system::Config>::RuntimeCall| {
856            let c = <T as Config>::RuntimeCall::from_ref(c);
857            // We make sure the proxy call does access this pallet to change modify proxies.
858            match c.is_sub_type() {
859                // Proxy call cannot add or remove a proxy with more permissions than it already
860                // has.
861                Some(Call::add_proxy { ref proxy_type, .. })
862                | Some(Call::remove_proxy { ref proxy_type, .. })
863                    if !def.proxy_type.is_superset(proxy_type) =>
864                {
865                    false
866                }
867                // Proxy call cannot remove all proxies or kill pure proxies unless it has full
868                // permissions.
869                Some(Call::remove_proxies { .. }) | Some(Call::kill_pure { .. })
870                    if def.proxy_type != T::ProxyType::default() =>
871                {
872                    false
873                }
874                _ => def.proxy_type.filter(c),
875            }
876        });
877        let e = call.dispatch(origin);
878        Self::deposit_event(Event::ProxyExecuted {
879            result: e.map(|_| ()).map_err(|e| e.error),
880        });
881    }

883    /// Removes all proxy delegates for a given delegator.
884    ///
885    /// Parameters:
886    /// - `delegator`: The delegator account.
887    pub fn remove_all_proxy_delegates(delegator: &T::AccountId) {
