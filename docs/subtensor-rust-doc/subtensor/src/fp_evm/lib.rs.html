78#[derive(Clone, Copy, Eq, PartialEq, Debug, Encode, Decode, TypeInfo)]
79#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
80pub struct WeightInfo {
81	pub ref_time_limit: Option<u64>,
82	pub proof_size_limit: Option<u64>,
83	pub ref_time_usage: Option<u64>,
84	pub proof_size_usage: Option<u64>,
85}
86
87impl WeightInfo {
88	pub fn new_from_weight_limit(
89		weight_limit: Option<Weight>,
90		proof_size_base_cost: Option<u64>,
91	) -> Result<Option<Self>, &'static str> {
92		Ok(match (weight_limit, proof_size_base_cost) {
93			(None, _) => None,
94			(Some(weight_limit), Some(proof_size_base_cost))
95				if weight_limit.proof_size() >= proof_size_base_cost =>
96			{
97				Some(WeightInfo {
98					ref_time_limit: Some(weight_limit.ref_time()),
99					proof_size_limit: Some(weight_limit.proof_size()),
100					ref_time_usage: Some(0u64),
101					proof_size_usage: Some(proof_size_base_cost),
102				})
103			}
104			(Some(weight_limit), None) => Some(WeightInfo {
105				ref_time_limit: Some(weight_limit.ref_time()),
106				proof_size_limit: None,
107				ref_time_usage: Some(0u64),
108				proof_size_usage: None,
109			}),
110			_ => return Err("must provide Some valid weight limit or None"),
111		})
112	}
113
114	fn try_consume(&self, cost: u64, limit: u64, usage: u64) -> Result<u64, ExitError> {
115		let usage = usage.checked_add(cost).ok_or(ExitError::OutOfGas)?;
116		if usage > limit {
117			return Err(ExitError::OutOfGas);
118		}
119		Ok(usage)
120	}
121
122	pub fn try_record_ref_time_or_fail(&mut self, cost: u64) -> Result<(), ExitError> {
123		if let (Some(ref_time_usage), Some(ref_time_limit)) =
124			(self.ref_time_usage, self.ref_time_limit)
125		{
126			let ref_time_usage = self.try_consume(cost, ref_time_limit, ref_time_usage)?;
127			if ref_time_usage > ref_time_limit {
128				return Err(ExitError::OutOfGas);
129			}
130			self.ref_time_usage = Some(ref_time_usage);
131		}
132		Ok(())
133	}
134
135	pub fn try_record_proof_size_or_fail(&mut self, cost: u64) -> Result<(), ExitError> {
136		if let (Some(proof_size_usage), Some(proof_size_limit)) =
137			(self.proof_size_usage, self.proof_size_limit)
138		{
139			let proof_size_usage = self.try_consume(cost, proof_size_limit, proof_size_usage)?;
140			if proof_size_usage > proof_size_limit {
141				return Err(ExitError::OutOfGas);
142			}
143			self.proof_size_usage = Some(proof_size_usage);
144		}
145		Ok(())
146	}
147
148	pub fn refund_proof_size(&mut self, amount: u64) {
149		if let Some(proof_size_usage) = self.proof_size_usage {
150			let proof_size_usage = proof_size_usage.saturating_sub(amount);
151			self.proof_size_usage = Some(proof_size_usage);
152		}
153	}
154
155	pub fn refund_ref_time(&mut self, amount: u64) {
156		if let Some(ref_time_usage) = self.ref_time_usage {
157			let ref_time_usage = ref_time_usage.saturating_sub(amount);
158			self.ref_time_usage = Some(ref_time_usage);
159		}
160	}
161	pub fn remaining_proof_size(&self) -> Option<u64> {
162		if let (Some(proof_size_usage), Some(proof_size_limit)) =
163			(self.proof_size_usage, self.proof_size_limit)
164		{
165			return Some(proof_size_limit.saturating_sub(proof_size_usage));
166		}
167		None
168	}
169
170	pub fn remaining_ref_time(&self) -> Option<u64> {
171		if let (Some(ref_time_usage), Some(ref_time_limit)) =
172			(self.ref_time_usage, self.ref_time_limit)
173		{
174			return Some(ref_time_limit.saturating_sub(ref_time_usage));
175		}
176		None
177	}
178}

245/// `WeightPerGas` is an approximate ratio of the amount of Weight per Gas.
246/// u64 works for approximations because Weight is a very small unit compared to gas.
247///
248/// `GAS_PER_MILLIS * WEIGHT_MILLIS_PER_BLOCK * TXN_RATIO ~= BLOCK_GAS_LIMIT`
249/// `WEIGHT_PER_GAS = WEIGHT_REF_TIME_PER_MILLIS / GAS_PER_MILLIS
250///                 = WEIGHT_REF_TIME_PER_MILLIS / (BLOCK_GAS_LIMIT / TXN_RATIO / WEIGHT_MILLIS_PER_BLOCK)
251///                 = TXN_RATIO * (WEIGHT_REF_TIME_PER_MILLIS * WEIGHT_MILLIS_PER_BLOCK) / BLOCK_GAS_LIMIT`
252///
253/// For example, given the 2000ms Weight, from which 75% only are used for transactions,
254/// the total EVM execution gas limit is `GAS_PER_MILLIS * 2000 * 75% = BLOCK_GAS_LIMIT`.
255pub fn weight_per_gas(
256	block_gas_limit: u64,
257	txn_ratio: Perbill,
258	weight_millis_per_block: u64,
259) -> u64 {
260	let weight_per_block = WEIGHT_REF_TIME_PER_MILLIS.saturating_mul(weight_millis_per_block);
261	let weight_per_gas = (txn_ratio * weight_per_block).saturating_div(block_gas_limit);
262	assert!(
263		weight_per_gas >= 1,
264		"WeightPerGas must greater than or equal with 1"
265	);
266	weight_per_gas
267}
