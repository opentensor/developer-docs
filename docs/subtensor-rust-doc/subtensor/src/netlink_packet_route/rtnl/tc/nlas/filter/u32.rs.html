3/// U32 filter
4///
5/// In its simplest form the U32 filter is a list of records, each consisting
6/// of two fields: a selector and an action. The selectors, described below,
7/// are compared with the currently processed IP packet until the first match
8/// occurs, and then the associated action is performed.
9use anyhow::Context;
10use byteorder::{ByteOrder, NativeEndian};
11
12use crate::{
13    nlas::{self, DefaultNla, NlaBuffer, NlasIterator},
14    parsers::parse_u32,
15    tc::{constants::*, Action},
16    traits::{Emitable, Parseable},
17    DecodeError,
18};
19
20pub const KIND: &str = "u32";
21
22const U32_SEL_BUF_LEN: usize = 16;
23const U32_KEY_BUF_LEN: usize = 16;
24
25#[derive(Debug, PartialEq, Eq, Clone)]
26pub enum Nla {
27    Unspec(Vec<u8>),
28    ClassId(u32),
29    Hash(u32),
30    Link(u32),
31    Divisor(u32),
32    Sel(Sel),
33    Police(Vec<u8>),
34    Act(Vec<Action>),
35    Indev(Vec<u8>),
36    Pcnt(Vec<u8>),
37    Mark(Vec<u8>),
38    Flags(u32),
39    Other(DefaultNla),
40}
41
42impl nlas::Nla for Nla {
43    fn value_len(&self) -> usize {
44        use self::Nla::*;
45        match self {
46            Unspec(b) | Police(b) | Indev(b) | Pcnt(b) | Mark(b) => b.len(),
47            ClassId(_) | Hash(_) | Link(_) | Divisor(_) | Flags(_) => 4,
48            Sel(s) => s.buffer_len(),
49            Act(acts) => acts.as_slice().buffer_len(),
50            Other(attr) => attr.value_len(),
51        }
52    }
53
54    fn emit_value(&self, buffer: &mut [u8]) {
55        use self::Nla::*;
56        match self {
57            Unspec(b) | Police(b) | Indev(b) | Pcnt(b) | Mark(b) => {
58                buffer.copy_from_slice(b.as_slice())
59            }
60            ClassId(i) | Hash(i) | Link(i) | Divisor(i) | Flags(i) => {
61                NativeEndian::write_u32(buffer, *i)
62            }
63            Sel(s) => s.emit(buffer),
64            Act(acts) => acts.as_slice().emit(buffer),
65            Other(attr) => attr.emit_value(buffer),
66        }
67    }
68
69    fn kind(&self) -> u16 {
70        use self::Nla::*;
71        match self {
72            Unspec(_) => TCA_U32_UNSPEC,
73            ClassId(_) => TCA_U32_CLASSID,
74            Hash(_) => TCA_U32_HASH,
75            Link(_) => TCA_U32_LINK,
76            Divisor(_) => TCA_U32_DIVISOR,
77            Sel(_) => TCA_U32_SEL,
78            Police(_) => TCA_U32_POLICE,
79            Act(_) => TCA_U32_ACT,
80            Indev(_) => TCA_U32_INDEV,
81            Pcnt(_) => TCA_U32_PCNT,
82            Mark(_) => TCA_U32_MARK,
83            Flags(_) => TCA_U32_FLAGS,
84            Other(attr) => attr.kind(),
85        }
86    }
87}

89impl<'a, T: AsRef<[u8]> + ?Sized> Parseable<NlaBuffer<&'a T>> for Nla {
90    fn parse(buf: &NlaBuffer<&'a T>) -> Result<Self, DecodeError> {
91        use self::Nla::*;
92        let payload = buf.value();
93        Ok(match buf.kind() {
94            TCA_U32_UNSPEC => Unspec(payload.to_vec()),
95            TCA_U32_CLASSID => {
96                ClassId(parse_u32(payload).context("failed to parse TCA_U32_UNSPEC")?)
97            }
98            TCA_U32_HASH => Hash(parse_u32(payload).context("failed to parse TCA_U32_HASH")?),
99            TCA_U32_LINK => Link(parse_u32(payload).context("failed to parse TCA_U32_LINK")?),
100            TCA_U32_DIVISOR => {
101                Divisor(parse_u32(payload).context("failed to parse TCA_U32_DIVISOR")?)
102            }
103            TCA_U32_SEL => Sel(self::Sel::parse(
104                &SelBuffer::new_checked(payload).context("invalid TCA_U32_SEL")?,
105            )
106            .context("failed to parse TCA_U32_SEL")?),
107            TCA_U32_POLICE => Police(payload.to_vec()),
108            TCA_U32_ACT => {
109                let mut acts = vec![];
110                for act in NlasIterator::new(payload) {
111                    let act = act.context("invalid TCA_U32_ACT")?;
112                    acts.push(Action::parse(&act).context("failed to parse TCA_U32_ACT")?);
113                }
114                Act(acts)
115            }
116            TCA_U32_INDEV => Indev(payload.to_vec()),
117            TCA_U32_PCNT => Pcnt(payload.to_vec()),
118            TCA_U32_MARK => Mark(payload.to_vec()),
119            TCA_U32_FLAGS => Flags(parse_u32(payload).context("failed to parse TCA_U32_FLAGS")?),
120            _ => Other(DefaultNla::parse(buf).context("failed to parse u32 nla")?),
121        })
122    }
123}

125#[derive(Debug, PartialEq, Eq, Clone, Default)]
126pub struct Sel {
127    pub flags: u8,
128    pub offshift: u8,
129    pub nkeys: u8,
130    pub offmask: u16,
131    pub off: u16,
132    pub offoff: u16,
133    pub hoff: u16,
134    pub hmask: u32,
135    pub keys: Vec<Key>,
136}
137
138buffer!(SelBuffer(U32_SEL_BUF_LEN) {
139    flags: (u8, 0),
140    offshift: (u8, 1),
141    nkeys: (u8, 2),
142    //pad: (u8, 3),
143    offmask: (u16, 4..6),
144    off: (u16, 6..8),
145    offoff: (u16, 8..10),
146    hoff: (u16, 10..12),
147    hmask: (u32, 12..U32_SEL_BUF_LEN),
148    keys: (slice, U32_SEL_BUF_LEN..),
149});
150
151impl Emitable for Sel {
152    fn buffer_len(&self) -> usize {
153        U32_SEL_BUF_LEN + (self.nkeys as usize * U32_KEY_BUF_LEN)
154    }
155
156    fn emit(&self, buffer: &mut [u8]) {
157        let mut packet = SelBuffer::new(buffer);
158        packet.set_flags(self.flags);
159        packet.set_offshift(self.offshift);
160        packet.set_offmask(self.offmask);
161        packet.set_off(self.off);
162        packet.set_offoff(self.offoff);
163        packet.set_hoff(self.hoff);
164        packet.set_hmask(self.hmask);
165        packet.set_nkeys(self.nkeys);
166        assert_eq!(self.nkeys as usize, self.keys.len());
167
168        let key_buf = packet.keys_mut();
169        for (i, k) in self.keys.iter().enumerate() {
170            k.emit(&mut key_buf[(i * U32_KEY_BUF_LEN)..((i + 1) * U32_KEY_BUF_LEN)]);
171        }
172    }
173}
174
175impl<T: AsRef<[u8]> + ?Sized> Parseable<SelBuffer<&T>> for Sel {
176    fn parse(buf: &SelBuffer<&T>) -> Result<Self, DecodeError> {
177        let nkeys = buf.nkeys();
178        let mut keys = Vec::<Key>::with_capacity(nkeys.into());
179        let key_payload = buf.keys();
180        for i in 0..nkeys {
181            let i = i as usize;
182            let keybuf = KeyBuffer::new_checked(
183                &key_payload[(i * U32_KEY_BUF_LEN)..(i + 1) * U32_KEY_BUF_LEN],
184            )
185            .context("invalid u32 key")?;
186            keys.push(Key::parse(&keybuf).context("failed to parse u32 key")?);
187        }
188
189        Ok(Self {
190            flags: buf.flags(),
191            offshift: buf.offshift(),
192            nkeys,
193            offmask: buf.offmask(),
194            off: buf.off(),
195            offoff: buf.offoff(),
196            hoff: buf.hoff(),
197            hmask: buf.hmask(),
198            keys,
199        })
200    }
201}

203#[derive(Debug, PartialEq, Eq, Clone, Default)]
204pub struct Key {
205    pub mask: u32,
206    pub val: u32,
207    pub off: i32,
208    pub offmask: i32,
209}
210
211buffer!(KeyBuffer(U32_KEY_BUF_LEN) {
212    mask: (u32, 0..4),
213    val: (u32, 4..8),
214    off: (i32, 8..12),
215    offmask: (i32, 12..U32_KEY_BUF_LEN),
216});
217
218impl Emitable for Key {
219    fn buffer_len(&self) -> usize {
220        U32_KEY_BUF_LEN
221    }
222    fn emit(&self, buffer: &mut [u8]) {
223        let mut packet = KeyBuffer::new(buffer);
224        packet.set_mask(self.mask);
225        packet.set_val(self.val);
226        packet.set_off(self.off);
227        packet.set_offmask(self.offmask);
228    }
229}
230
231impl<T: AsRef<[u8]>> Parseable<KeyBuffer<T>> for Key {
232    fn parse(buf: &KeyBuffer<T>) -> Result<Self, DecodeError> {
233        Ok(Self {
234            mask: buf.mask(),
235            val: buf.val(),
236            off: buf.off(),
237            offmask: buf.offmask(),
238        })
239    }
