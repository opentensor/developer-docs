1//! Bandwidth estimation.
2
3use crate::{rtp_::Mid, Rtc};
4
5pub use crate::rtp_::Bitrate;
6
7#[derive(Debug, PartialEq)]
8/// Bandwidth estimation kind.
9pub enum BweKind {
10    /// Transport wide congestion control.
11    Twcc(Bitrate),
12    /// REMB (Receiver Estimated Maximum Bitrate)
13    Remb(Mid, Bitrate),
14}
15
16/// Access to the Bandwidth Estimate subsystem.
17pub struct Bwe<'a>(pub(crate) &'a mut Rtc);
18
19impl<'a> Bwe<'a> {
20    /// Configure the current bitrate.
21    ///
22    /// Configure the bandwidth estimation system with the current bitrate.
23    /// **Note:** This only has an effect if BWE has been enabled via
24    /// [`RtcConfig::enable_bwe`][crate::RtcConfig::enable_bwe].
25    ///
26    /// * `current_bitrate` an estimate of the current bitrate being sent. When the media is
27    /// produced by encoders this value should be the sum of all the target bitrates for these
28    /// encoders, when the media originates from another WebRTC client it should be the sum of the
29    /// configure bitrates for all tracks being sent. This value should only account for video i.e.
30    /// audio bitrates should be ignored.
31    ///
32    /// ## Example
33    ///
34    /// Say you have a video track with three ingress simulcast layers: `low` with `maxBitrate` set to 250Kbits/,
35    /// `medium` with `maxBitrate` set to 750Kbits/, and `high` with `maxBitrate` 1.5Mbit/s.
36    /// Staring at the lower layer, call:
37    ///
38    /// ```
39    /// # use str0m::{Rtc, bwe::Bitrate};
40    /// let mut rtc = Rtc::new();
41    ///
42    /// rtc.bwe().set_current_bitrate(Bitrate::kbps(250));
43    /// ````
44    ///
45    /// When a new estimate is made available that indicates a switch to the medium layer is
46    /// possible, make the switch and then update the configuration:
47    ///
48    /// ```
49    /// # use str0m::{Rtc, bwe::Bitrate};
50    /// let mut rtc = Rtc::new();
51    ///
52    /// rtc.bwe().set_current_bitrate(Bitrate::kbps(750));
53    /// ````
54    ///
55    /// ## Accuracy
56    ///
57    /// When the original media is derived from another WebRTC implementation that support BWE it's
58    /// advisable to use the value from `RTCOutboundRtpStreamStats.targetBitrate` from `getStats`
59    /// rather than the `maxBitrate` values from `RTCRtpEncodingParameters`.
60    pub fn set_current_bitrate(&mut self, current_bitrate: Bitrate) {
61        self.0.session.set_bwe_current_bitrate(current_bitrate);
62    }
63
64    /// Configure the desired bitrate.
65    ///
66    /// Configure the bandwidth estimation system with the desired bitrate.
67    /// **Note:** This only has an effect if BWE has been enabled via
68    /// [`RtcConfig::enable_bwe`][crate::RtcConfig::enable_bwe].
69    ///
70    /// * `desired_bitrate` The bitrate you would like to eventually send at. The BWE system will
71    /// try to reach this bitrate by probing with padding packets. You should allocate your media
72    /// bitrate based on the estimated the BWE system produces via
73    /// [`Event::EgressBitrateEstimate`][crate::Event::EgressBitrateEstimate]. This rate might
74    /// not be reached if the network link cannot sustain the desired bitrate.
75    ///
76    /// ## Example
77    ///
78    /// Say you have three simulcast video tracks each with a high layer configured at 1.5Mbit/s.
79    /// You should then set the desired bitrate to 4.5Mbit/s(or slightly higher). If the network
80    /// link can sustain 4.5Mbit/s there will eventually be an
81    /// [`Event::EgressBitrateEstimate`][crate::Event::EgressBitrateEstimate] with this estimate.
82    pub fn set_desired_bitrate(&mut self, desired_bitrate: Bitrate) {
83        self.0.session.set_bwe_desired_bitrate(desired_bitrate);
84    }
85}
