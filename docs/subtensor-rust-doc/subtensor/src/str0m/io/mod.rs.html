63/// An instruction to send an outgoing packet.
64#[derive(Serialize, Deserialize)]
65pub struct Transmit {
66    /// Protocol the transmission should use.
67    ///
68    /// Provided to each of the [`Candidate`][crate::Candidate] constructors.
69    pub proto: Protocol,
70
71    /// The source IP this packet should be sent from.
72    ///
73    /// For ICE it's important to send outgoing packets from the correct IP address.
74    /// The IP could come from a local socket or relayed over a TURN server. Features like
75    /// hole-punching will only work if the packets are routed through the correct interfaces.
76    ///
77    /// This address will either be:
78    /// - The address of a socket you have bound locally, such as with [`UdpSocket::bind`][std::net::UdpSocket].
79    /// - The address of a relay socket that you have
80    ///     [allocated](https://www.rfc-editor.org/rfc/rfc8656#name-allocations-2) using TURN.
81    ///
82    /// To correctly handle an instance of [`Transmit`], you should:
83    ///
84    /// - Check if [`Transmit::source`] corresponds to one of your local sockets,
85    ///     if yes, send it through that.
86    /// - Check if [`Transmit::source`] corresponds to one of your relay sockets (i.e. allocations),
87    ///     if yes, send it via one of:
88    ///     - a [TURN channel data message](https://www.rfc-editor.org/rfc/rfc8656#name-sending-a-channeldata-messa)
89    ///     - a [SEND indication](https://www.rfc-editor.org/rfc/rfc8656#name-send-and-data-methods)
90    ///
91    /// `str0m` learns about the source address using [`Candidate`][crate::Candidate] that are added using
92    /// [`Rtc::add_local_candidate`][crate::Rtc::add_local_candidate].
93    ///
94    /// The different candidate types are:
95    ///
96    /// * [`Candidate::host()`][crate::Candidate::host]: Used for locally bound UDP sockets.
97    /// * [`Candidate::relayed()`][crate::Candidate::relayed]: Used for sockets relayed via some
98    ///     other server (normally TURN).
99    /// * [`Candidate::server_reflexive()`][crate::Candidate::server_reflexive]: Used when a local
100    ///     (host) socket appears as some another IP address to the remote peer (usually due to a
101    ///     NAT firewall on the local side). STUN servers can be used to discover the external address.
102    ///     In this case the `base` parameter to `server_reflexive()` is the local address and
103    ///     used for [`Transmit::source`].
104    /// * `Peer reflexive` is another, internal, type of candidate that str0m infers by using the other
105    ///     types of candidates.
106    pub source: SocketAddr,
107
108    /// The destination address this datagram should be sent to.
109    ///
110    /// This will be one of the [`Candidate`][crate::Candidate] provided explicitly using
111    /// [`Rtc::add_remote_candidate`][crate::Rtc::add_remote_candidate] or via SDP negotiation.
112    pub destination: SocketAddr,
113
114    /// Contents of the datagram.
115    pub contents: DatagramSend,
116}

151impl<'a> Receive<'a> {
152    /// Creates a new instance by trying to parse the contents of `buf`.
153    pub fn new(
154        proto: Protocol,
155        source: SocketAddr,
156        destination: SocketAddr,
157        buf: &'a [u8],
158    ) -> Result<Self, NetError> {
159        let contents = DatagramRecv::try_from(buf)?;
160        Ok(Receive {
161            proto,
162            source,
163            destination,
164            contents,
165        })
166    }

182/// Wrapper for a parsed payload to be received.
183#[derive(Serialize, Deserialize)]
184pub struct DatagramRecv<'a> {
185    #[serde(borrow)]
186    pub(crate) inner: DatagramRecvInner<'a>,
187}
188
189#[allow(clippy::large_enum_variant)] // We purposely don't want to allocate.
190#[derive(Serialize, Deserialize)]
191pub(crate) enum DatagramRecvInner<'a> {
192    Stun(StunMessage<'a>),
193    Dtls(&'a [u8]),
194    Rtp(&'a [u8]),
195    Rtcp(&'a [u8]),
196}
197
198impl<'a> TryFrom<&'a [u8]> for DatagramRecv<'a> {
199    type Error = NetError;
200
201    fn try_from(value: &'a [u8]) -> Result<Self, Self::Error> {
202        use DatagramRecvInner::*;
203
204        let kind = MultiplexKind::try_from(value)?;
205
206        let inner = match kind {
207            MultiplexKind::Stun => Stun(StunMessage::parse(value)?),
208            MultiplexKind::Dtls => Dtls(value),
209            MultiplexKind::Rtp => Rtp(value),
210            MultiplexKind::Rtcp => Rtcp(value),
211        };
212
213        Ok(DatagramRecv { inner })
214    }
215}

217#[derive(Debug, Clone, Copy, PartialEq)]
218pub(crate) enum MultiplexKind {
219    Stun,
220    Dtls,
221    Rtp,
222    Rtcp,
223}
224
225impl<'a> TryFrom<&'a [u8]> for MultiplexKind {
226    type Error = io::Error;
227
228    fn try_from(value: &'a [u8]) -> Result<Self, io::Error> {
229        let byte0 = value[0];
230        let len = value.len();
231
232        if byte0 < 2 && len >= 20 {
233            Ok(MultiplexKind::Stun)
234        } else if byte0 >= 20 && byte0 < 64 {
235            Ok(MultiplexKind::Dtls)
236        } else if byte0 >= 128 && byte0 < 192 && len > 2 {
237            let byte1 = value[1];
238            let payload_type = byte1 & 0x7f;
239
240            Ok(if payload_type < 64 {
241                // This is kinda novel, and probably breaks, but...
242                // we can use the < 64 pt as an escape hatch if we run out
243                // of dynamic numbers >= 96
244                // https://bugs.chromium.org/p/webrtc/issues/detail?id=12194
245                MultiplexKind::Rtp
246            } else if payload_type >= 64 && payload_type < 96 {
247                MultiplexKind::Rtcp
248            } else {
249                MultiplexKind::Rtp
250            })
251        } else {
252            Err(io::Error::new(
253                io::ErrorKind::InvalidData,
254                "Unknown datagram",
255            ))
256        }
257    }
258}

310impl TryFrom<&str> for Protocol {
311    type Error = ();
312
313    fn try_from(proto: &str) -> Result<Self, Self::Error> {
314        let proto = proto.to_lowercase();
315        match proto.as_str() {
316            "udp" => Ok(Protocol::Udp),
317            "tcp" => Ok(Protocol::Tcp),
318            "ssltcp" => Ok(Protocol::SslTcp),
319            "tls" => Ok(Protocol::Tls),
320            _ => Err(()),
321        }
322    }
323}
