55#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
56pub enum CheckState {
57    /// A check has not been sent for this pair.
58    #[default]
59    Waiting,
60
61    /// A check has been sent for this pair, but the
62    /// transaction is in progress.
63    InProgress,
64
65    /// A check has been sent for this pair, and it produced a
66    /// successful result.
67    Succeeded,
68    //
69    // Hey martin, this looks fishy you might say. Why is there no
70    // failed state? The reason is that agent removes the candidatepair
71    // straight away if it is deemed failed.
72    //
73    // /// A check has been sent for this pair, and it failed (a
74    // /// response to the check was never received, or a failure response
75    // /// was received).
76    // Failed,
77}

79#[derive(Debug, Default, Clone, Copy, PartialEq, Eq)]
80pub enum NominationState {
81    /// This pair has not been nominated.
82    #[default]
83    None,
84    /// The pair has been nominated. The binding request for the
85    /// nomination has not been sent.
86    Nominated,
87    /// The binding request is sent.
88    Attempt,
89    /// Successful nomination. Received a reply for the transaction id.
90    Success,
91}

133    pub fn calculate_prio(controlling: bool, remote_prio: u32, local_prio: u32) -> u64 {
134        // The ICE agent computes a priority for each candidate pair.  Let G be
135        // the priority for the candidate provided by the controlling agent.
136        // Let D be the priority for the candidate provided by the controlled
137        // agent.  The priority for a pair is computed as follows:
138        //
139        //    pair priority = 2^32*MIN(G,D) + 2*MAX(G,D) + (G>D?1:0)
140
141        let (g, d) = if controlling {
142            (local_prio, remote_prio)
143        } else {
144            (remote_prio, local_prio)
145        };
146
147        2_u64.pow(32) * g.min(d) as u64 + 2 * g.max(d) as u64 + if g > d { 1 } else { 0 }
148    }

195    pub fn nominate(&mut self, force_success: bool) {
196        assert!(self.nomination_state == NominationState::None);
197        if force_success {
198            self.nomination_state = NominationState::Success;
199            debug!("Force success nominated pair {:?}", self);
200        } else {

202            debug!("Nominated pair: {:?}", self);
203        }

206    /// Records a new binding request attempt.
207    ///
208    /// Returns the transaction id to use in the STUN message.
209    pub fn new_attempt(&mut self, now: Instant) -> TransId {
210        // calculate a new time
211        self.cached_next_attempt_time = None;
212
213        if matches!(self.nomination_state, NominationState::Nominated) {
214            debug!("Nominated attempt STUN binding: {:?}", self);
215            self.nomination_state = NominationState::Attempt;
216        }
217
218        let attempt = BindingAttempt {
219            trans_id: TransId::new(),
220            request_sent: now,
221            respone_recv: None,
222            nominated: self.is_nominated(),
223        };
224
225        self.binding_attempts.push_back(attempt);
226
227        // Never keep more than STUN_MAX_RETRANS attempts.
228        while self.binding_attempts.len() > STUN_MAX_RETRANS {
229            self.binding_attempts.pop_front();
230        }
231
232        if self.state == CheckState::Waiting {
233            trace!(
234                "Check state: {:?} -> {:?}",
235                self.state,
236                CheckState::InProgress
237            );
238            self.state = CheckState::InProgress;
239        }
240
241        let last = self.binding_attempts.back().unwrap();
242
243        last.trans_id
244    }

246    /// Tells if this pair caused the binding request for a STUN transaction id.
247    pub fn has_binding_attempt(&self, trans_id: TransId) -> bool {
248        self.binding_attempts.iter().any(|b| b.trans_id == trans_id)
249    }

251    /// Marks a binding request attempt as having a successful response.
252    ///
253    /// ### Panics
254    ///
255    /// Panics if the trans_id doesn't belong to this pair.
256    pub fn record_binding_response(&mut self, now: Instant, trans_id: TransId, valid_idx: usize) {
257        self.cached_next_attempt_time = None;
258
259        self.valid_idx = Some(valid_idx);
260
261        let attempt = self
262            .binding_attempts
263            .iter_mut()
264            .find(|b| b.trans_id == trans_id)
265            .expect("Binding request attempt");
266
267        attempt.respone_recv = Some(now);
268
269        if attempt.nominated && self.nomination_state == NominationState::Attempt {
270            self.nomination_state = NominationState::Success;
271            debug!("Nomination success: {:?}", self);
272        }
273
274        if self.state == CheckState::InProgress {
275            trace!(
276                "Check state: {:?} -> {:?}",
277                self.state,
278                CheckState::Succeeded
279            );
280            self.state = CheckState::Succeeded;
281        }
282
283        trace!("Recorded binding response: {:?}", self);
284    }

286    /// The time of the last binding request attempt.
287    ///
288    /// `None` means there has been no attempts.
289    fn last_attempt_time(&self) -> Option<Instant> {
290        self.binding_attempts.back().map(|b| b.request_sent)
291    }
292
293    /// From the back of binding_attempts, go through all unanswered and find
294    /// the time when we started having an outage.
295    ///
296    /// The returned value is the number of unanswered attempts and the oldest time.
297    fn unanswered(&self) -> Option<(usize, Instant)> {
298        self.binding_attempts
299            .iter()
300            .rev()
301            .take_while(|b| b.respone_recv.is_none())
302            .enumerate()
303            .last()
304            .map(|(idx, b)| (idx + 1, b.request_sent))
305    }

307    /// When we should do the next retry.
308    ///
309    /// Returns `None` if we are not to attempt this pair anymore.
310    pub fn next_binding_attempt(&mut self, now: Instant) -> Instant {
311        if let Some(cached) = self.cached_next_attempt_time {
312            return cached;
313        }
314
315        let next = if matches!(self.nomination_state, NominationState::Nominated) {
316            // Cheating a bit to make the nomination "skip the queue".
317            now.checked_sub(Duration::from_secs(60)).unwrap()
318        } else if let Some(last) = self.last_attempt_time() {
319            // When we have unanswered for longer than STUN_MAX_RTO_MILLIS / 2, start
320            // checking more often.
321            let unanswered_count = self
322                .unanswered()
323                .filter(|(_, since)| now - *since > Duration::from_millis(STUN_MAX_RTO_MILLIS) / 2)
324                .map(|(count, _)| count);
325
326            let send_count = unanswered_count.unwrap_or(self.binding_attempts.len());
327
328            last + stun_resend_delay(send_count)
329        } else {
330            // No previous attempt, do next retry straight away.
331            now
332        };
333
334        // At least do a check at this time.
335        let min = now + MIN_TIMEOUT;
336
337        let at_least = next.min(min);
338
339        // keep this cached since the calculation can happen very often.
340        self.cached_next_attempt_time = Some(at_least);
341
342        at_least
343    }

345    /// Tells if this candidate pair is still possible to use for connectivity.
346    ///
347    /// Returns `false` if the candidate has failed.
348    pub fn is_still_possible(&self, now: Instant) -> bool {
349        let attempts = self.binding_attempts.len();
350        let unanswered = self.unanswered().map(|b| b.0).unwrap_or(0);
351
352        if attempts < STUN_MAX_RETRANS || unanswered < STUN_MAX_RETRANS {
353            true
354        } else {
355            // check to see if we are still waiting for the last attempt
356            // this unwrap is fine because unanswered count > 0
357            let last = self.last_attempt_time().unwrap();
358            let cutoff = last + stun_resend_delay(STUN_MAX_RETRANS);
359            now < cutoff
360        }
361    }
