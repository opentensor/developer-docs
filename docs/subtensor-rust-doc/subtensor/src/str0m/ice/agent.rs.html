108/// States the ICE connection can be in.
109///
110/// More details on connection states can be found in the [ICE RFC][1].
111///
112/// [1]: https://www.rfc-editor.org/rfc/rfc8445
113#[derive(Debug, Clone, Copy, PartialEq, Eq)]
114pub enum IceConnectionState {
115    /// The ICE agent is gathering addresses.
116    New,
117
118    /// The ICE agent is checking pairs of local and remote candidates against one
119    /// another to try to find a compatible match, but has not yet found a pair
120    /// which will allow the peer connection to be made. It is possible that
121    /// gathering of candidates is also still underway.
122    Checking,
123
124    /// A usable pairing of local and remote candidates has been found, and the
125    /// connection has been established. The agent is not in `Completed` because
126    /// it is still gathering candidates or is still checking candidates against
127    /// one another looking for a better connection to use.
128    Connected,
129
130    /// The ICE agent has finished gathering candidates, has checked all pairs
131    /// against one another, and has found a working connection.
132    Completed,
133
134    /// Connection failed. This is a less stringent test than `failed` and may trigger
135    /// intermittently and resolve just as spontaneously on less reliable networks,
136    /// or during temporary disconnections. When the problem resolves, the connection
137    /// may return to the connected state.
138    Disconnected,
139    //
140    // NB: The failed and closed state doesn't really have a mapping in this implementation.
141    //     We never end trickle ice and it's always possible to "come back" if more remote
142    //     candidates are added.
143    //
144    // The ICE candidate has checked all candidates pairs against one another and has
145    // failed to find compatible matches.
146    // Failed,
147    // The ICE agent has shut down and is no longer handling requests.
148    // Closed,
149}
150
151impl IceConnectionState {
152    /// Tells if this state is a connected state.
153    pub fn is_connected(&self) -> bool {
154        use IceConnectionState::*;
155        matches!(self, Connected | Completed)
156    }
157
158    /// Tells if this state is a disconnected state.
159    pub fn is_disconnected(&self) -> bool {
160        *self == IceConnectionState::Disconnected
161    }

329    /// Credentials for STUN.
330    ///
331    /// The username for the credential is formed by concatenating the
332    /// username fragment provided by the peer with the username fragment of
333    /// the ICE agent sending the request, separated by a colon (":").
334    ///
335    /// The password is equal to the password provided by the peer.
336    ///
337    /// The responses utilize the same usernames and passwords as the requests
338    /// (note that the USERNAME attribute is not present in the response).
339    ///
340    /// ### Panics
341    ///
342    /// Panics if there are no remote credentials set.
343    fn stun_credentials(&self, reply: bool) -> (String, String) {
344        let local = &self.local_credentials;
345
346        let (left, right, peer_pass) = if reply {
347            ("not_used", "not_used", "not_used")
348        } else {
349            let peer = self
350                .remote_credentials
351                .as_ref()
352                .expect("Remote ICE credentials");
353            (&peer.ufrag[..], &local.ufrag[..], &peer.pass[..])
354        };
355
356        let username = format!("{left}:{right}");
357        let password = if reply {
358            local.pass.clone()
359        } else {
360            peer_pass.into()
361        };
362
363        (username, password)
364    }

394    /// Adds a local candidate.
395    ///
396    /// Returns `false` if the candidate was not added because it is redundant.
397    /// Adding loopback addresses or multicast/broadcast addresses causes
398    /// an error.
399    pub fn add_local_candidate(&mut self, mut c: Candidate) -> bool {
400        let ip = c.addr().ip();
401
402        if self.ice_lite {
403            // Reject all non-host candidates.
404            if c.kind() != CandidateKind::Host {
405                debug!("Reject non-host candidate due to ice-lite mode: {:?}", c);
406                return false;
407            }
408        }
409
410        // "Adopt" any incoming candidate by setting our current ufrag.
411        c.set_ufrag(&self.local_credentials.ufrag);
412
413        // https://datatracker.ietf.org/doc/html/rfc8445#section-5.1.2.1
414        // The local preference MUST be an integer from 0 (lowest preference) to
415        // 65535 (highest preference) inclusive.  When there is only a single IP
416        // address, this value SHOULD be set to 65535.  If there are multiple
417        // candidates for a particular component for a particular data stream
418        // that have the same type, the local preference MUST be unique for each
419        // one.
420        // ...
421        // If an ICE agent is multihomed and has multiple IP addresses, the
422        // recommendations in [RFC8421] SHOULD be followed.  If multiple TURN
423        // servers are used, local priorities for the candidates obtained from
424        // the TURN servers are chosen in a similar fashion as for multihomed
425        // local candidates: the local preference value is used to indicate a
426        // preference among different servers, but the preference MUST be unique
427        // for each one.
428        // ================
429        //
430        // The above presupposes that we know all the candidates when we start
431        // the ice agent. That doesn't work for us, so we deliberately do not
432        // follow spec. We assign the following intervals for the different
433        // types of candidates:
434        //
435        // 0     - 16384 => relay
436        // 16384 - 32768 => srflx
437        // 32768 - 49152 => prflx
438        // 49152 - 65536 => host
439        //
440        // And furthermore we subdivide these to interleave IPv6 with IPv4
441        // so that odd numbers are ipv6 and even are ipv4.
442        //
443        // For host candidates this means:
444        // 65535 - first ipv6
445        // 65534 - first ipv4
446        // 65533 - second ipv6
447        // 65432 - second ipv4
448        let counter_start: u32 = {
449            use CandidateKind::*;
450            let x = match c.kind() {
451                Host => 65_535,
452                PeerReflexive => 49_151,
453                ServerReflexive => 32_767,
454                Relayed => 16_383,
455            };
456            x - if ip.is_ipv6() { 0 } else { 1 }
457        };
458
459        // Count the number of existing candidates of the same kind.
460        let same_kind = self
461            .local_candidates
462            .iter()
463            .filter(|v| v.kind() == c.kind())
464            .filter(|v| v.addr().is_ipv6() == ip.is_ipv6())
465            .count() as u32;
466
467        let pref = counter_start - same_kind * 2;
468        trace!("Calculated local preference: {}", pref);
469
470        c.set_local_preference(pref);
471
472        // A candidate is redundant if and only if its transport address and base equal those
473        // of another candidate.  The agent SHOULD eliminate the redundant
474        // candidate with the lower priority.
475        //
476        // NB this must be done _after_ set_local_preference(), since the prio() used in the
477        // elimination is calculated from that preference.
478        if let Some((idx, other)) =
479            self.local_candidates.iter_mut().enumerate().find(|(_, v)| {
480                v.addr() == c.addr() && v.base() == c.base() && v.proto() == c.proto()
481            })
482        {
483            if c.prio() < other.prio() {
484                // The new candidate is not better than what we already got.
485                debug!(
486                    "Reject redundant candidate, current: {:?} rejected: {:?}",
487                    other, c
488                );
489                return false;
490            }
491
492            // Stop using the current candidate in favor of the new one.
493            debug!(
494                "Replace redundant candidate, current: {:?} replaced with: {:?}",
495                other, c
496            );
497            other.set_discarded();
498            self.discard_candidate_pairs_by_local(idx);
499        }
500
501        // Tie this ufrag to this ICE-session.
502        c.set_ufrag(&self.local_credentials.ufrag);
503
504        // These are the indexes of the remote candidates this candidate should be paired with.
505        let remote_idxs: Vec<_> = self
506            .remote_candidates
507            .iter()
508            .enumerate()
509            .filter(|(_, v)| !v.discarded() && v.addr().is_ipv4() == ip.is_ipv4())
510            .map(|(i, _)| i)
511            .collect();
512
513        info!("Add local candidate: {:?}", c);
514
515        self.local_candidates.push(c);
516
517        let local_idxs = [self.local_candidates.len() - 1];
518
519        // We always run in trickle ice mode.
520        //
521        // https://www.rfc-editor.org/rfc/rfc8838.html#section-10
522        // A Trickle ICE agent MUST NOT pair a local candidate until it has been trickled
523        // to the remote party.
524        //
525        // TODO: The trickle ice spec is strange. What does it mean "has been trickled to the
526        // remote party"? Since we don't get a confirmation that the candidate has been received
527        // by the remote party, whether we form local pairs directly or later seems irrelevant.
528        self.form_pairs(&local_idxs, &remote_idxs);
529
530        true
531    }

533    /// Adds a remote candidate.
534    ///
535    /// Returns `false` if the candidate was not added because it is redundant.
536    /// Adding loopback addresses or multicast/broadcast addresses causes
537    /// an error.
538    pub fn add_remote_candidate(&mut self, mut c: Candidate) {
539        // This is a a:rtcp-mux-only implementation. The only component
540        // we accept is 1 for RTP.
541        if c.component_id() != 1 {
542            debug!("Reject candidate for component other than 1: {:?}", c);
543            return;
544        }
545
546        if let Some(creds) = &self.remote_credentials {
547            if let Some(ufrag) = c.ufrag() {
548                if ufrag != creds.ufrag {
549                    debug!(
550                        "Reject candidate with ufrag mismatch: {} != {}",
551                        ufrag, creds.ufrag
552                    );
553                    return;
554                }
555            }
556        }
557
558        // After we accepted the ufrag, don't keep this around since it will look
559        // confusing inspecting the state.
560        c.clear_ufrag();
561
562        let existing_prflx = self
563            .remote_candidates
564            .iter_mut()
565            .enumerate()
566            .find(|(_, v)| {
567                v.foundation() == REMOTE_PEER_REFLEXIVE_TEMP_FOUNDATION
568                    && v.kind() == CandidateKind::PeerReflexive
569                    && v.addr() == c.addr()
570            });
571
572        let ipv4 = c.addr().is_ipv4();
573
574        let remote_idx = if let Some((idx, existing)) = existing_prflx {
575            // If any subsequent candidate exchanges contain this peer-reflexive
576            // candidate, it will signal the actual foundation for the candidate.
577            info!(
578                "Replace peer reflexive candidate, current: {:?} replaced with: {:?}",
579                existing, c
580            );
581            *existing = c;
582            idx
583        } else {
584            info!("Add remote candidate: {:?}", c);
585
586            self.remote_candidates.push(c);
587            self.remote_candidates.len() - 1
588        };
589
590        // These are the indexes of the local candidates this candidate should be paired with.
591        let local_idxs: Vec<_> = self
592            .local_candidates
593            .iter()
594            .enumerate()
595            .filter(|(_, v)| !v.discarded() && v.addr().is_ipv4() == ipv4)
596            .map(|(i, _)| i)
597            .collect();
598
599        let remote_idxs = [remote_idx];
600        self.form_pairs(&local_idxs, &remote_idxs);
601    }

603    /// Form pairs given two slices of indexes into the local_candidates and remote_candidates.
604    fn form_pairs(&mut self, local_idxs: &[usize], remote_idxs: &[usize]) {
605        for local_idx in local_idxs {
606            'outer: for remote_idx in remote_idxs {
607                let local = &self.local_candidates[*local_idx];
608                let remote = &self.remote_candidates[*remote_idx];
609
610                // Candidates in a pair must share the same protocol
611                if local.proto() != remote.proto() {
612                    continue 'outer;
613                }
614
615                let prio =
616                    CandidatePair::calculate_prio(self.controlling, remote.prio(), local.prio());
617                let mut pair = CandidatePair::new(*local_idx, *remote_idx, prio);
618
619                trace!("Form pair local: {:?} remote: {:?}", local, remote);
620
621                // The agent prunes each checklist.  This is done by removing a
622                // candidate pair if it is redundant with a higher-priority candidate
623                // pair in the same checklist.  Two candidate pairs are redundant if
624                // their local candidates have the same base and their remote candidates
625                // are identical.
626
627                for (check_idx, check) in self.candidate_pairs.iter().enumerate() {
628                    let check_local = check.local_candidate(&self.local_candidates);
629                    let check_remote = check.remote_candidate(&self.remote_candidates);
630
631                    let redundant = local.base() == check_local.base()
632                        && remote.addr() == check_remote.addr()
633                        && local.proto() == check_local.proto()
634                        && remote.proto() == check_remote.proto();
635
636                    if redundant {
637                        if check.prio() >= pair.prio() {
638                            // skip this new pair since there is a redundant pair already in the
639                            // list with higher/equal priority.
640                            debug!(
641                                "Reject redundant pair, current: {:?} rejected: {:?}",
642                                check, pair
643                            );
644                        } else {
645                            // replace the existing candidate pair, since the new one got a higher prio.
646                            debug!(
647                                "Replace redundant pair, current: {:?} replaced with: {:?}",
648                                check, pair
649                            );
650
651                            let was_nominated = self.candidate_pairs[check_idx].is_nominated();
652                            pair.nominate(was_nominated);
653
654                            if self.ice_lite {
655                                debug!("Retain incoming binding requests for pair");
656                                pair.copy_remote_binding_requests(&self.candidate_pairs[check_idx]);
657                            }
658
659                            self.candidate_pairs[check_idx] = pair;
660                        }
661
662                        // There can only be one candidate pair per local base / remote addr.
663                        // Since we found that redundant entry, there's no point in checking further
664                        // candidate pairs.
665                        continue 'outer;
666                    }
667                }
668
669                debug!("Add new pair {:?}", pair);
670
671                // This is not a redundant pair, add it.
672                self.candidate_pairs.push(pair);
673            }
674        }
675
676        // NB it would be nicer to have BTreeSet, but that makes it impossible to
677        // get mut references to the elements in the list.
678        self.candidate_pairs.sort();
679
680        // an ICE agent MUST limit the total number of connectivity checks
681        // the agent performs across all checklists in the checklist set.
682        // This is done by limiting the total number of candidate pairs in the
683        // checklist set. The default limit of candidate pairs for the checklist
684        // set is 100, but the value MUST be configurable.
685        //
686        // TODO: How does this work with trickle ice?
687        let max = self.max_candidate_pairs.unwrap_or(100);
688        while self.candidate_pairs.len() > max {
689            let pair = self.candidate_pairs.pop();
690            debug!("Remove overflow pair {:?}", pair);
691        }
692    }

694    /// Invalidate a candidate and remove it from the connection.
695    ///
696    /// This is done for host candidates disappearing due to changes in the network
697    /// interfaces like a WiFi disconnecting or changing IPs.
698    ///
699    /// It can also be used to invalidate _remote_ candidates, i.e. if the remote
700    /// has signalled us that they have invalidated one of their candidates.
701    ///
702    /// Returns `true` if the candidate was found and invalidated.
703    #[allow(unused)]
704    pub fn invalidate_candidate(&mut self, c: &Candidate) -> bool {
705        if let Some((idx, other)) = self.local_candidates.iter_mut().enumerate().find(|(_, v)| {
706            v.addr() == c.addr()
707                && v.base() == c.base()
708                && v.raddr() == c.raddr()
709                && v.kind() == c.kind()
710        }) {
711            if !other.discarded() {
712                info!("Local candidate to discard {:?}", other);
713                other.set_discarded();
714                self.discard_candidate_pairs_by_local(idx);
715                return true;
716            }
717        }
718
719        if let Some((idx, other)) = self
720            .remote_candidates
721            .iter_mut()
722            .enumerate()
723            .find(|(_, v)| {
724                v.addr() == c.addr()
725                    && v.base() == c.base()
726                    && v.raddr() == c.raddr()
727                    && v.kind() == c.kind()
728            })
729        {
730            if !other.discarded() {
731                info!("Remote candidate to discard {:?}", other);
732                other.set_discarded();
733                self.discard_candidate_pairs_by_remote(idx);
734                return true;
735            }
736        }
737
738        debug!("No local or remote candidate found: {:?}", c);
739        false
740    }

742    /// Restart ICE.
743    ///
744    /// This is useful when detecting a change in network interfaces, such as
745    /// current session running off a 4G, and we connect to a WiFi. The session
746    /// should continue sending data over the 4G until we redone the ICE gathering
747    /// process.
748    #[allow(unused)]
749    pub fn ice_restart(&mut self, local_credentials: IceCreds, keep_local_candidates: bool) {
750        // An ICE agent MAY restart ICE for existing data streams.  An ICE
751        // restart causes all previous states of the data streams, excluding the
752        // roles of the agents, to be flushed.  The only difference between an
753        // ICE restart and a brand new data session is that during the restart,
754        // data can continue to be sent using existing data sessions, and a new
755        // data session always requires the roles to be determined.
756
757        self.remote_credentials = None;
758        self.remote_candidates.clear();
759        self.candidate_pairs.clear();
760        self.transmit.clear();
761        self.events.clear();
762        self.discovered_recv.clear();
763
764        if keep_local_candidates {
765            // If we're keeping the candidates, we must update the ufrag to the new credentials.
766            // This is so anyone inspecting `.local_candidates()` will get the correct ufrag.
767            for c in &mut self.local_candidates {
768                c.set_ufrag(&local_credentials.ufrag)
769            }
770        } else {
771            self.local_candidates.clear();
772        }
773
774        self.local_credentials = local_credentials;
775
776        self.emit_event(IceAgentEvent::IceRestart(self.local_credentials.clone()));
777        self.set_connection_state(IceConnectionState::Checking, "ice restart");
778    }

792    /// Tells whether the message is for this agent instance.
793    ///
794    /// This is used to multiplex multiple ice agents on a server sharing the same UDP socket.
795    ///
796    /// For binding requests, if no remote credentials have been set using
797    /// `set_remote_credentials`, the remote ufrag is not checked.
798    pub fn accepts_message(&self, message: &StunMessage<'_>) -> bool {
799        trace!("Check if accepts message: {:?}", message);
800
801        let do_integrity_check = |is_request: bool| -> bool {
802            let (_, password) = self.stun_credentials(is_request);
803            let integrity_passed = message.check_integrity(&password);
804
805            // The integrity is always the last thing we check
806            if integrity_passed {
807                trace!("Message accepted");
808            } else {
809                trace!("Message rejected, integrity check failed");
810            }
811            integrity_passed
812        };
813
814        let method = message.method();
815        let class = message.class();
816        match (method, class) {
817            (StunMethod::Binding, StunClass::Request | StunClass::Indication) => {
818                // The username for the credential is formed by concatenating the
819                // username fragment provided by the peer with the username fragment of
820                // the ICE agent sending the request, separated by a colon (":").
821                // The existence of this username is checked in the STUN parser.
822                let (local, remote) = message.split_username().unwrap();
823
824                let local_creds = self.local_credentials();
825                if local != local_creds.ufrag {
826                    trace!(
827                        "Message rejected, local user mismatch: {} != {}",
828                        local,
829                        local_creds.ufrag
830                    );
831                    return false;
832                }
833
834                if let Some(remote_creds) = &self.remote_credentials {
835                    if remote != remote_creds.ufrag {
836                        trace!(
837                            "Message rejected, remote user mismatch: {} != {}",
838                            remote,
839                            remote_creds.ufrag
840                        );
841                        return false;
842                    }
843                }
844
845                do_integrity_check(true)
846            }
847            (StunMethod::Binding, StunClass::Success | StunClass::Failure) => {
848                let belongs_to_a_candidate_pair = self
849                    .candidate_pairs
850                    .iter()
851                    .any(|pair| pair.has_binding_attempt(message.trans_id()));
852
853                if !belongs_to_a_candidate_pair {
854                    trace!("Message rejected, transaction ID does not belong to any of our candidate pairs");
855                    return false;
856                }
857
858                do_integrity_check(false)
859            }
860            (StunMethod::Binding, StunClass::Unknown) => {
861                // Without a known class, it's impossible to know how to validate the message
862                trace!("Message rejected, unknown STUN class");
863                false
864            }
865            (StunMethod::Unknown, _) => {
866                // Without a known method, it's impossible to know how to validate the message
867                trace!("Message rejected, unknown STUN method");
868                false
869            }
870        }
871    }

873    /// Handles an incoming STUN message.
874    ///
875    /// Will not be used if [`IceAgent::accepts_message`] returns false.
876    pub fn handle_packet(&mut self, now: Instant, packet: StunPacket) -> bool {
877        trace!("Handle receive: {:?}", &packet.message);
878
879        // Regardless of whether we have remote_creds at this point, we can
880        // at least check the message integrity.
881        if !self.accepts_message(&packet.message) {
882            debug!("Message not accepted");
883            return false;
884        }
885
886        if packet.message.is_binding_request() {
887            self.stun_server_handle_message(now, &packet);
888        } else if packet.message.is_successful_binding_response() {
889            self.stun_client_handle_response(now, packet.message);
890        }
891
892        self.emit_event(IceAgentEvent::DiscoveredRecv {
893            proto: packet.proto,
894            source: packet.source,
895        });
896
897        // TODO handle unsuccessful responses.
898
899        true
900    }

902    /// Provide the current time to the [`IceAgent`].
903    ///
904    /// Typically, you will want to call [`IceAgent::poll_timeout`] and "wake-up" the agent once that time is reached.
905    pub fn handle_timeout(&mut self, now: Instant) {
906        // This happens exactly once because evaluate_state() below will
907        // switch away from New -> Checking.
908        if self.state == IceConnectionState::New {
909            self.emit_event(IceAgentEvent::IceRestart(self.local_credentials.clone()));
910        }
911
912        self.evaluate_state(now);
913
914        // First we try to empty the queue of saved STUN requests.
915        if self.remote_credentials.is_some() {
916            let queue = &mut self.stun_server_queue;
917
918            // No need hanging on to very old requests.
919            while let Some(peek) = queue.front() {
920                if now - peek.now >= STUN_TIMEOUT {
921                    let r = queue.pop_front();
922                    trace!("Drop too old enqueued STUN request: {:?}", r.unwrap());
923                } else {
924                    break;
925                }
926            }
927
928            if let Some(req) = self.stun_server_queue.pop_front() {
929                debug!("Handle enqueued STUN request: {:?}", req);
930                self.stun_server_handle_request(req);
931                return;
932            }
933        }
934
935        self.last_now = Some(now);
936
937        self.evaluate_nomination();
938
939        // prune failed candidates.
940        let mut any_pruned = false;
941        self.candidate_pairs.retain(|p| {
942            let keep = if self.ice_lite {
943                p.has_recent_remote_binding_request(now)
944            } else {
945                p.is_still_possible(now)
946            };
947            if !keep {
948                debug!("Remove failed pair: {:?}", p);
949                any_pruned = true;
950            }
951            keep
952        });
953        if any_pruned {
954            self.evaluate_nomination();
955            self.evaluate_state(now);
956        }
957
958        if self.remote_credentials.is_none() {
959            trace!("Stop timeout due to missing remote credentials");
960            return;
961        }
962
963        if self.ice_lite {
964            // Remote binding request time is the timestamp in the CandidatePair that
965            // is used to decide whether something is timed out or not. We need all
966            // pairs to have this time set, so that pairs that don't receive any
967            // STUN binding requests eventually times out.
968            for p in &mut self.candidate_pairs {
969                if p.remote_binding_request_time().is_none() {
970                    p.increase_remote_binding_requests(now);
971                }
972            }
973
974            trace!("Stop timeout since ice-lite do no checks");
975            return;
976        }
977
978        // when do we need to handle the next candidate pair?
979        let next = self
980            .candidate_pairs
981            .iter_mut()
982            .enumerate()
983            .map(|(i, c)| (i, c.next_binding_attempt(now)))
984            .min_by_key(|(_, t)| *t);
985
986        if let Some((idx, deadline)) = next {
987            if now >= deadline {
988                let pair = &self.candidate_pairs[idx];
989                trace!("Handle next triggered pair: {:?}", pair);
990                self.stun_client_binding_request(now, idx);
991            } else {
992                // trace!("Next triggered pair is in the future: {:?}", deadline - now);
993            }
994        }
995    }

998    pub fn poll_transmit(&mut self) -> Option<Transmit> {
999        let x = self.transmit.pop_front();
1000        if let Some(x) = &x {
1001            if x.contents.len() > DATAGRAM_MTU_WARN {
1002                warn!("ICE above MTU {}: {}", DATAGRAM_MTU_WARN, x.contents.len());
1003            }
1004            trace!("Poll transmit: {:?}", x);
1005        }
1006        x
1007    }

1009    /// Poll for the next time to call [`IceAgent::handle_timeout`].
1010    ///
1011    /// For optimal performance, you should call this every time the [`IceAgent`]s state changes.
1012    /// For example, after you call [`IceAgent::add_local_candidate`] or [`IceAgent::add_remote_candidate`].
1013    ///
1014    /// Returns `None` until the first ever `handle_timeout` is called.
1015    pub fn poll_timeout(&mut self) -> Option<Instant> {
1016        // if we never called handle_timeout, there will be no current time.
1017        let last_now = self.last_now?;
1018
1019        let has_request = !self.stun_server_queue.is_empty();
1020        let has_transmit = !self.transmit.is_empty();
1021
1022        // We must empty the queued replies or stuff to send as soon as possible.
1023        if has_request || has_transmit {
1024            return Some(last_now + TIMING_ADVANCE);
1025        }
1026
1027        // when do we need to handle the next candidate pair?
1028        let maybe_next = if self.ice_lite {
1029            // ice-lite doesn't do checks.
1030            None
1031        } else {
1032            self.candidate_pairs
1033                .iter_mut()
1034                .map(|c| c.next_binding_attempt(last_now))
1035                .min()
1036        };
1037
1038        // Time must advance with at least Ta.
1039        let next = if let Some(next) = maybe_next {
1040            if next < last_now + TIMING_ADVANCE {
1041                last_now + TIMING_ADVANCE
1042            } else {
1043                next
1044            }
1045        } else {
1046            // IDLE for a while.
1047            last_now + Duration::from_secs(3)
1048        };
1049
1050        Some(next)
1051    }

1077    fn stun_server_handle_message(&mut self, now: Instant, packet: &StunPacket) {
1078        let message = &packet.message;
1079        let prio = message
1080            .prio()
1081            // this should be guarded in the parsing
1082            .expect("STUN request prio");
1083        let use_candidate = message.use_candidate();
1084
1085        if use_candidate {
1086            trace!("Binding request sent USE-CANDIDATE");
1087        }
1088
1089        let trans_id = message.trans_id();
1090
1091        // The existence of USERNAME is checked by the STUN parser.
1092        let (_, remote_ufrag) = message.split_username().unwrap();
1093
1094        // Because we might have to delay stun requests until we receive the remote
1095        // credentials, we extract all relevant bits of information so it can be owned.
1096        let req = StunRequest {
1097            now,
1098            proto: packet.proto,
1099            source: packet.source,
1100            destination: packet.destination,
1101            trans_id,
1102            prio,
1103            use_candidate,
1104            remote_ufrag: remote_ufrag.into(),
1105        };
1106
1107        if self.remote_credentials.is_some() {
1108            self.stun_server_handle_request(req);
1109        } else {
1110            debug!(
1111                "Enqueue STUN request due to missing remote credentials: {:?}",
1112                req
1113            );
1114
1115            let queue = &mut self.stun_server_queue;
1116
1117            // It is possible (and in fact very likely) that the
1118            // initiating agent will receive a Binding request prior to receiving
1119            // the candidates from its peer.
1120            queue.push_back(req);
1121
1122            // This is some denial-of-service attack protection.
1123            while queue.len() > 100 {
1124                let r = queue.pop_front();
1125                debug!("Remove overflow STUN request {:?}", r);
1126            }
1127
1128            // If this happens, the agent MUST
1129            // immediately generate a response.  The agent has sufficient
1130            // information at this point to generate the response; the password from
1131            // the peer is not required.
1132
1133            // TODO: The spec seems to indicate we can generate a reply, but that seems
1134            // to fly in the face of logic. A reply requires a fingerprint message integrity
1135            // and we can't construct that until we get the remote password.
1136        }
1137    }

1139    fn stun_server_handle_request(&mut self, req: StunRequest) {
1140        let remote_creds = self.remote_credentials.as_ref().expect("Remote ICE creds");
1141        if req.remote_ufrag != remote_creds.ufrag {
1142            // this check can be delayed due to receiving STUN bind requests before we
1143            // get the exchange on the signal level.
1144            debug!(
1145                "STUN request rejected, remote user mismatch (enqueued): {} != {}",
1146                req.remote_ufrag, remote_creds.ufrag
1147            );
1148            return;
1149        }
1150
1151        if req.use_candidate && self.controlling {
1152            // the other side is not controlling, and it sent USE-CANDIDATE. that's wrong.
1153            debug!("STUN request rejected, USE-CANDIDATE when local is controlling");
1154            return;
1155        }
1156
1157        self.stats.bind_request_recv += 1;
1158
1159        // If the source transport address of the request does not match any
1160        // existing remote candidates, it represents a new peer-reflexive remote
1161        // candidate.
1162        let found_in_remote = self
1163            .remote_candidates
1164            .iter()
1165            .enumerate()
1166            .find(|(_, c)| !c.discarded() && c.proto() == req.proto && c.addr() == req.source);
1167
1168        let remote_idx = if let Some((idx, _)) = found_in_remote {
1169            trace!("Remote candidate for STUN request found");
1170            idx
1171        } else {
1172            // o  The priority is the value of the PRIORITY attribute in the Binding
1173            //     request.
1174            //
1175            // o  The foundation is an arbitrary value, different from the
1176            //     foundations of all other remote candidates.  If any subsequent
1177            //     candidate exchanges contain this peer-reflexive candidate, it will
1178            //     signal the actual foundation for the candidate.
1179            let c = Candidate::peer_reflexive(
1180                req.proto,
1181                req.source,
1182                req.source,
1183                req.prio,
1184                // TODO: REMOTE_PEER_REFLEXIVE_TEMP_FOUNDATION should probably have
1185                // a counter to really make it "different from the foundations of
1186                // all other remote candidates".
1187                // In practice it might no matter since we don't do the frozen-waiting
1188                // dance for candidate pairs.
1189                Some(REMOTE_PEER_REFLEXIVE_TEMP_FOUNDATION.into()),
1190                self.local_credentials.ufrag.clone(),
1191            );
1192
1193            info!(
1194                "Created peer reflexive remote candidate from STUN request: {:?}",
1195                c
1196            );
1197
1198            // This candidate is added to the list of remote candidates.  However,
1199            // the ICE agent does not pair this candidate with any local candidates.
1200            self.remote_candidates.push(c);
1201
1202            self.remote_candidates.len() - 1
1203        };
1204
1205        let local_idx = match self.local_candidates.iter().enumerate().find(|(_, v)| {
1206            // The local candidate will be
1207            // either a host candidate (for cases where the request was not received
1208            // through a relay) or a relayed candidate (for cases where it is
1209            // received through a relay).  The local candidate can never be a
1210            // server-reflexive candidate.
1211            matches!(v.kind(), CandidateKind::Host | CandidateKind::Relayed)
1212                && v.addr() == req.destination
1213                && v.proto() == req.proto
1214        }) {
1215            Some((i, _)) => i,
1216            None => {
1217                // Receiving traffic for an IP address that neither is a HOST nor RELAY is most likely a configuration fault where the user forgot to add a candidate for the local interface.
1218                // We are network-connected application so we need to handle this gracefully: Log a message and discard the packet.
1219
1220                debug!(
1221                    "Discarding STUN request on unknown interface: {}",
1222                    req.destination
1223                );
1224                return;
1225            }
1226        };
1227
1228        let maybe_pair = self
1229            .candidate_pairs
1230            .iter_mut()
1231            .find(|p| p.local_idx() == local_idx && p.remote_idx() == remote_idx);
1232
1233        if let Some(pair) = maybe_pair {
1234            // When the pair is already on the checklist:
1235            trace!("Found existing pair for STUN request: {:?}", pair);
1236
1237            // TODO: The spec has all these ideas about resetting to Waiting state
1238            // for the candidate pair. I think that's to do speeding up the triggered
1239            // checks if a nomination comes through. It doesn't seem to make much
1240            // sense in this implementation, where a nomination jumps the queue.
1241            // https://datatracker.ietf.org/doc/html/rfc8445#section-7.3.1.4
1242        } else {
1243            // If the pair is not already on the checklist:
1244            let local = &self.local_candidates[local_idx];
1245            let remote = &self.remote_candidates[remote_idx];
1246
1247            if local.discarded() {
1248                return; // Ignore STUN requests to discarded candidates
1249            }
1250
1251            let prio = CandidatePair::calculate_prio(self.controlling, remote.prio(), local.prio());
1252
1253            // *  Its state is set to Waiting. (this is the default)
1254            // *  The pair is inserted into the checklist based on its priority.
1255            // *  The pair is enqueued into the triggered-check queue.
1256            let pair = CandidatePair::new(local_idx, remote_idx, prio);
1257
1258            debug!("Created new pair for STUN request: {:?}", pair);
1259
1260            self.candidate_pairs.push(pair);
1261            self.candidate_pairs.sort();
1262        }
1263
1264        let pair = self
1265            .candidate_pairs
1266            .iter_mut()
1267            .find(|p| p.local_idx() == local_idx && p.remote_idx() == remote_idx)
1268            // unwrap is fine since we have inserted a pair if it was missing.
1269            .unwrap();
1270
1271        let local = pair.local_candidate(&self.local_candidates);
1272        let proto = local.proto();
1273        let local_addr = local.base();
1274        let remote = pair.remote_candidate(&self.remote_candidates);
1275        let remote_addr = remote.addr();
1276
1277        pair.increase_remote_binding_requests(req.now);
1278
1279        if !self.controlling && !pair.is_nominated() && req.use_candidate {
1280            // We need to answer a nomination request with a binding request
1281            // in the other direction.
1282            //
1283            // If this is ice-lite, we make it successful straight away.
1284            pair.nominate(self.ice_lite);
1285        }
1286
1287        if self.controlling && pair.state() == CheckState::Succeeded {
1288            // See if we can nominate something now.
1289            self.evaluate_nomination();
1290        }
1291
1292        let (_, password) = self.stun_credentials(true);
1293
1294        let reply = StunMessage::reply(req.trans_id, req.source);
1295
1296        debug!(
1297            "Send STUN reply: {} -> {} {:?}",
1298            local_addr, remote_addr, reply
1299        );
1300
1301        let mut buf = vec![0_u8; DATAGRAM_MTU];
1302
1303        let n = reply
1304            .to_bytes(&password, &mut buf)
1305            .expect("IO error writing STUN reply");
1306        buf.truncate(n);
1307
1308        let trans = Transmit {
1309            proto,
1310            source: local_addr,
1311            destination: remote_addr,
1312            contents: buf.into(),
1313        };
1314
1315        self.transmit.push_back(trans);
1316    }

1318    fn stun_client_binding_request(&mut self, now: Instant, pair_idx: usize) {
1319        let (username, password) = self.stun_credentials(false);
1320
1321        let pair = &mut self.candidate_pairs[pair_idx];
1322        let local = pair.local_candidate(&self.local_candidates);
1323        let remote = pair.remote_candidate(&self.remote_candidates);
1324        let prio = local.prio_prflx();
1325        // Only the controlling side sends USE-CANDIDATE.
1326        let use_candidate = self.controlling && pair.is_nominated();
1327
1328        let trans_id = pair.new_attempt(now);
1329
1330        self.stats.bind_request_sent += 1;
1331
1332        let binding = StunMessage::binding_request(
1333            &username,
1334            trans_id,
1335            self.controlling,
1336            self.control_tie_breaker,
1337            prio,
1338            use_candidate,
1339        );
1340
1341        debug!(
1342            "Send STUN request: {} -> {} {:?}",
1343            local.base(),
1344            remote.addr(),
1345            binding
1346        );
1347
1348        let mut buf = vec![0_u8; DATAGRAM_MTU];
1349
1350        let n = binding
1351            .to_bytes(&password, &mut buf)
1352            .expect("IO error writing STUN reply");
1353        buf.truncate(n);
1354
1355        let trans = Transmit {
1356            proto: local.proto(),
1357            source: local.base(),
1358            destination: remote.addr(),
1359            contents: buf.into(),
1360        };
1361
1362        self.transmit.push_back(trans);
1363    }

1365    fn stun_client_handle_response(&mut self, now: Instant, message: StunMessage<'_>) {
1366        // Find the candidate pair that this trans_id was sent for.
1367        let trans_id = message.trans_id();
1368        let maybe_pair = self
1369            .candidate_pairs
1370            .iter_mut()
1371            .find(|p| p.has_binding_attempt(trans_id));
1372
1373        self.stats.bind_success_recv += 1;
1374
1375        let pair = match maybe_pair {
1376            Some(v) => v,
1377            // Not finding the candidate pair is fine. That might mean the
1378            // binding response came in "too late", after we discarded the
1379            // pair for some reason.
1380            None => {
1381                debug!("No pair found for STUN response: {:?}", message);
1382                return;
1383            }
1384        };
1385
1386        // The ICE agent MUST check the mapped address from the STUN response.
1387        // If the transport address does not match any of the local candidates
1388        // that the agent knows about, the mapped address represents a new
1389        // candidate: a peer-reflexive candidate.
1390        let mapped_address = message
1391            .mapped_address()
1392            // This should be caught in the parsing.
1393            .expect("Mapped address in STUN response");
1394
1395        let found_in_local = self
1396            .local_candidates
1397            .iter()
1398            .enumerate()
1399            .find(|(_, c)| c.addr() == mapped_address);
1400
1401        let (pair, valid_idx) = if let Some((valid_idx, _)) = found_in_local {
1402            // Note, the valid_idx might not be the same as the local_idx that we
1403            // sent the request from. This might happen for hosts with asymmetric
1404            // routing, traffic leaving on one interface and responses coming back
1405            // on another.
1406            trace!(
1407                "Found local candidate for mapped address: {}",
1408                mapped_address
1409            );
1410            (pair, valid_idx)
1411        } else {
1412            let local_sent_from = pair.local_candidate(&self.local_candidates);
1413
1414            // Like other candidates, a peer-reflexive candidate has a type, base, priority,
1415            // and foundation. They are computed as follows:
1416
1417            // o  The priority is the value of the PRIORITY attribute in the Binding
1418            //     request.
1419            //
1420            // The PRIORITY attribute MUST be included in a Binding request and be
1421            // set to the value computed by the algorithm in Section 5.1.2 for the
1422            // local candidate, but with the candidate type preference of peer-
1423            // reflexive candidates.
1424            let prio = local_sent_from.prio_prflx();
1425
1426            // o  The base is the local candidate of the candidate pair from which
1427            //     the Binding request was sent.
1428            let base = local_sent_from.base();
1429
1430            // o  The type is peer reflexive.
1431            let candidate = Candidate::peer_reflexive(
1432                local_sent_from.proto(),
1433                mapped_address,
1434                base,
1435                prio,
1436                None,
1437                self.local_credentials.ufrag.clone(),
1438            );
1439
1440            debug!(
1441                "Created local peer reflexive candidate for mapped address: {}",
1442                mapped_address
1443            );
1444
1445            // The ICE agent does not need to pair the peer-reflexive candidate with
1446            // remote candidates.
1447            // If an agent wishes to pair the peer-reflexive candidate with remote
1448            // candidates other than the one in the valid pair that will be generated,
1449            // the agent MAY provide updated candidate information to the peer that includes
1450            // the peer-reflexive candidate.  This will cause the peer-reflexive candidate
1451            // to be paired with all other remote candidates.
1452
1453            // For now we do not tell the other side about discovered peer-reflexive candidates.
1454            // We just include it in our list of local candidates and use it for the "valid pair".
1455            self.local_candidates.push(candidate);
1456
1457            let idx = self.local_candidates.len() - 1;
1458
1459            (pair, idx)
1460        };
1461
1462        pair.record_binding_response(now, trans_id, valid_idx);
1463
1464        if self.controlling {
1465            self.evaluate_nomination();
1466        }
1467
1468        // State might change when we get a response.
1469        self.evaluate_state(now);
1470    }

1472    fn evaluate_nomination(&mut self) {
1473        let nominated_pair_priority = self.nominated_pair_priority();
1474
1475        let best_prio = if self.controlling {
1476            // For controlling agents, we pick the best candidate pair using
1477            // this strategy.
1478            self.candidate_pairs
1479                .iter_mut()
1480                .filter(|p| p.state() == CheckState::Succeeded)
1481                .max_by_key(|p| p.prio())
1482        } else {
1483            // For controlled agents, we pick the best pair from what the controlling
1484            // agent has indicated with USE-CANDIDATE stun attribute.
1485            self.candidate_pairs
1486                .iter_mut()
1487                .filter(|p| p.is_nominated())
1488                .max_by_key(|p| p.prio())
1489        };
1490
1491        if let Some(best_prio) = best_prio {
1492            if let Some(nominated) = nominated_pair_priority {
1493                if nominated == best_prio.prio() {
1494                    // The best prio is also the current nominated prio. Make
1495                    // no changes since there can be multiple pairs having the
1496                    // same best_prio.
1497                    return;
1498                }
1499            }
1500            trace!("Nominating best candidate");
1501
1502            if !best_prio.is_nominated() && (self.controlling || self.ice_lite) {
1503                // ice lite progresses pair to success straight away.
1504                best_prio.nominate(self.ice_lite);
1505            }
1506
1507            let local = best_prio.local_candidate(&self.local_candidates);
1508            let remote = best_prio.remote_candidate(&self.remote_candidates);
1509
1510            self.nominated_send = Some(best_prio.id());
1511            self.emit_event(IceAgentEvent::NominatedSend {
1512                proto: local.proto(),
1513                source: local.base(),
1514                destination: remote.addr(),
1515            })
1516        }
1517    }

1519    fn nominated_pair_priority(&self) -> Option<u64> {
1520        let id = self.nominated_send?;
1521
1522        self.candidate_pairs
1523            .iter()
1524            .find_map(|p| (p.id() == id).then_some(p.prio()))
1525    }

1527    fn set_connection_state(&mut self, state: IceConnectionState, reason: &'static str) {
1528        if self.state != state {
1529            info!("State change ({}): {:?} -> {:?}", reason, self.state, state);
1530            self.state = state;
1531            self.emit_event(IceAgentEvent::IceConnectionStateChange(state));
1532        }
1533    }

1535    fn evaluate_state(&mut self, now: Instant) {
1536        use IceConnectionState::*;
1537
1538        let mut any_nomination = false;
1539        let mut any_still_possible = false;
1540
1541        for p in &self.candidate_pairs {
1542            if p.is_nominated() {
1543                any_nomination = true;
1544            } else if p.is_still_possible(now) {
1545                any_still_possible = true;
1546            }
1547        }
1548
1549        // As a special case, before the ice agent has received any add_remote_candidate() or
1550        // discovered a peer reflexive via a STUN message, the agent is still viable. This is
1551        // also the case for ice_restart.
1552        if self.remote_candidates.is_empty() {
1553            any_still_possible = true;
1554        }
1555
1556        match self.state {
1557            New => {
1558                self.set_connection_state(Checking, "new connection");
1559            }
1560            Checking | Disconnected => {
1561                if any_nomination {
1562                    if self.ice_lite {
1563                        self.set_connection_state(Completed, "got nomination in ice lite");
1564                        return;
1565                    }
1566                    if any_still_possible {
1567                        self.set_connection_state(Connected, "got nomination, still trying others");
1568                    } else {
1569                        self.set_connection_state(Completed, "got nomination, no others to try");
1570                    }
1571                } else if !any_still_possible {
1572                    self.set_connection_state(Disconnected, "no possible pairs");
1573                }
1574            }
1575            Connected => {
1576                if any_nomination {
1577                    if !any_still_possible {
1578                        self.set_connection_state(Completed, "no more possible to try");
1579                    }
1580                } else {
1581                    self.set_connection_state(Disconnected, "none nominated");
1582                }
1583            }
1584            Completed => {
1585                if any_nomination {
1586                    if any_still_possible && !self.ice_lite {
1587                        self.set_connection_state(Connected, "got new possible");
1588                    }
1589                } else {
1590                    self.set_connection_state(Disconnected, "none nominated");
1591                }
1592            }
1593        }
1594    }
