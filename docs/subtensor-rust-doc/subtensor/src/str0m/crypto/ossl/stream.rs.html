15pub struct TlsStream<S> {
16    active: Option<bool>,
17    state: State<S>,
18    keying_mat: Option<(KeyingMaterial, SrtpProfile, Fingerprint)>,
19    exported: bool,
20}
21
22pub enum State<S> {
23    Init(Ssl, S),
24    Handshaking(MidHandshakeSslStream<S>),
25    Established(SslStream<S>),
26    Empty,
27}
28
29/// This is okay because there is no way for a user of Rtc to interact with the Dtls subsystem
30/// in a way that would allow them to observe a potentially broken invariant when catching a panic.
31impl<S> UnwindSafe for State<S> {}
32
33impl<S> TlsStream<S>
34where
35    S: io::Read + io::Write + UnwindSafe,
36{
37    pub fn new(ssl: Ssl, stream: S) -> Self {
38        TlsStream {
39            active: None,
40            state: State::Init(ssl, stream),
41            keying_mat: None,
42            exported: false,
43        }
44    }
45
46    pub fn is_active(&self) -> Option<bool> {
47        self.active
48    }
49
50    pub fn is_connected(&self) -> bool {
51        matches!(self.state, State::Established(_))
52    }
53
54    pub fn set_active(&mut self, active: bool) {
55        assert!(
56            self.active.is_none(),
57            "set_active should called exactly once"
58        );
59        self.active = Some(active);
60    }
61
62    pub fn complete_handshake_until_block(&mut self) -> Result<bool, CryptoError> {
63        if let Err(e) = self.handshaken() {
64            if e.kind() == io::ErrorKind::WouldBlock {
65                Ok(false)
66            } else {
67                Err(e.into())
68            }
69        } else {
70            Ok(true)
71        }
72    }
73
74    pub fn is_handshaken(&self) -> bool {
75        matches!(self.state, State::Established(_))
76    }
77
78    pub fn handshaken(&mut self) -> Result<&mut SslStream<S>, io::Error> {
79        let active = self.is_active().expect("set_active must be called");
80        let v = self.state.handshaken(active)?;
81
82        // first time we complete the handshake, we extract the keying material for SRTP.
83        if !self.exported {
84            let keying_mat = export_srtp_keying_material(v)?;
85            self.exported = true;
86            self.keying_mat = Some(keying_mat);
87        }
88
89        Ok(v)
90    }
91
92    pub fn take_srtp_keying_material(
93        &mut self,
94    ) -> Option<(KeyingMaterial, SrtpProfile, Fingerprint)> {
95        self.keying_mat.take()
96    }
97
98    pub fn inner_mut(&mut self) -> &mut S {
99        match &mut self.state {
100            State::Init(_, s) => s,
101            State::Handshaking(v) => v.get_mut(),
102            State::Established(v) => v.get_mut(),
103            State::Empty => panic!("inner_mut on empty dtls state"),
104        }
105    }
106}

108impl<S> State<S>
109where
110    S: io::Read + io::Write + UnwindSafe,
111{
112    fn handshaken(&mut self, active: bool) -> Result<&mut SslStream<S>, io::Error> {
113        if let State::Established(v) = self {
114            return Ok(v);
115        }
116
117        let taken = mem::replace(self, State::Empty);
118
119        let result = match taken {
120            State::Empty | State::Established(_) => unreachable!(),
121            State::Init(ssl, stream) => {
122                if active {
123                    debug!("Connect");
124                    ssl.connect(stream)
125                } else {
126                    debug!("Accept");
127                    ssl.accept(stream)
128                }
129            }
130            State::Handshaking(mid) => mid.handshake(),
131        };
132
133        match result {
134            Ok(v) => {
135                debug!("Established version: {:}", v.ssl().version_str());
136
137                let _ = mem::replace(self, State::Established(v));
138
139                // recursively return the &mut SslStream.
140                self.handshaken(active)
141            }
142            Err(e) => Err(match e {
143                HandshakeError::WouldBlock(e) => {
144                    let _ = mem::replace(self, State::Handshaking(e));
145                    io::Error::new(io::ErrorKind::WouldBlock, "WouldBlock")
146                }
147                HandshakeError::SetupFailure(e) => {
148                    debug!("DTLS setup failed: {:?}", e);
149                    io::Error::new(io::ErrorKind::InvalidInput, e)
150                }
151                HandshakeError::Failure(e) => {
152                    let e = e.into_error();
153                    debug!("DTLS failure: {:?}", e);
154                    io::Error::new(io::ErrorKind::InvalidData, e)
155                }
156            }),
157        }
158    }
159}

161fn export_srtp_keying_material<S>(
162    stream: &mut SslStream<S>,
163) -> Result<(KeyingMaterial, SrtpProfile, Fingerprint), io::Error> {
164    let ssl = stream.ssl();
165
166    // remote peer certificate fingerprint
167    let x509 = ssl
168        .peer_certificate()
169        .ok_or_else(|| io::Error::new(io::ErrorKind::Other, "No remote X509 cert"))?;
170    let digest: &[u8] = &x509.digest(MessageDigest::sha256())?;
171
172    let fp = Fingerprint {
173        hash_func: "sha-256".into(),
174        bytes: digest.to_vec(),
175    };
176
177    let srtp_profile_id = ssl
178        .selected_srtp_profile()
179        .map(|s| s.id())
180        .ok_or_else(|| io::Error::new(io::ErrorKind::Other, "Failed to negotiate SRTP profile"))?;
181    let srtp_profile: SrtpProfile = srtp_profile_id.try_into()?;
182
183    // extract SRTP keying material
184    let mut buf = vec![0_u8; srtp_profile.keying_material_len()];
185    ssl.export_keying_material(&mut buf, DTLS_KEY_LABEL, None)?;
186
187    let mat = KeyingMaterial::new(buf);
188
189    Ok((mat, srtp_profile, fp))
190}

214impl TryFrom<SrtpProfileId> for SrtpProfile {
215    type Error = io::Error;
216
217    fn try_from(value: SrtpProfileId) -> Result<Self, Self::Error> {
218        match value {
219            SrtpProfileId::SRTP_AES128_CM_SHA1_80 => Ok(SrtpProfile::Aes128CmSha1_80),
220            SrtpProfileId::SRTP_AEAD_AES_128_GCM => Ok(SrtpProfile::AeadAes128Gcm),
221            x => Err(io::Error::new(
222                io::ErrorKind::Other,
223                format!("Unsupported SRTP profile {:x}", x.as_raw()),
224            )),
225        }
226    }
