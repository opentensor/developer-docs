9pub struct OsslSrtpCryptoImpl;
10
11impl SrtpCryptoImpl for OsslSrtpCryptoImpl {
12    type Aes128CmSha1_80 = OsslAes128CmSha1_80;
13    type AeadAes128Gcm = OsslAeadAes128Gcm;
14
15    fn srtp_aes_128_ecb_round(key: &[u8], input: &[u8], output: &mut [u8]) {
16        let mut aes =
17            Crypter::new(Cipher::aes_128_ecb(), Mode::Encrypt, key, None).expect("AES deriver");
18
19        // Run AES
20        let count = aes.update(input, output).expect("AES update");
21        let rest = aes.finalize(&mut output[count..]).expect("AES finalize");
22
23        assert_eq!(count + rest, 16 + 16); // input len + block size
24    }
25}

27pub struct OsslAes128CmSha1_80(CipherCtx);
28
29impl aes_128_cm_sha1_80::CipherCtx for OsslAes128CmSha1_80 {
30    fn new(key: aes_128_cm_sha1_80::AesKey, encrypt: bool) -> Self
31    where
32        Self: Sized,
33    {
34        let t = cipher::Cipher::aes_128_ctr();
35        let mut ctx = CipherCtx::new().expect("a reusable cipher context");
36
37        if encrypt {
38            ctx.encrypt_init(Some(t), Some(&key[..]), None)
39                .expect("enc init");
40        } else {
41            ctx.decrypt_init(Some(t), Some(&key[..]), None)
42                .expect("enc init");
43        }
44
45        OsslAes128CmSha1_80(ctx)
46    }
47
48    fn encrypt(
49        &mut self,
50        iv: &aes_128_cm_sha1_80::RtpIv,
51        input: &[u8],
52        output: &mut [u8],
53    ) -> Result<(), CryptoError> {
54        self.0.encrypt_init(None, None, Some(iv))?;
55        let count = self.0.cipher_update(input, Some(output))?;
56        self.0.cipher_final(&mut output[count..])?;
57        Ok(())
58    }
59
60    fn decrypt(
61        &mut self,
62        iv: &aes_128_cm_sha1_80::RtpIv,
63        input: &[u8],
64        output: &mut [u8],
65    ) -> Result<(), CryptoError> {
66        self.0.decrypt_init(None, None, Some(iv))?;
67        let count = self.0.cipher_update(input, Some(output))?;
68        self.0.cipher_final(&mut output[count..])?;
69        Ok(())
70    }
71}

73pub struct OsslAeadAes128Gcm(CipherCtx);
74
75impl aead_aes_128_gcm::CipherCtx for OsslAeadAes128Gcm {
76    fn new(key: aead_aes_128_gcm::AeadKey, encrypt: bool) -> Self
77    where
78        Self: Sized,
79    {
80        let t = cipher::Cipher::aes_128_gcm();
81        let mut ctx = CipherCtx::new().expect("a reusable cipher context");
82
83        if encrypt {
84            ctx.encrypt_init(Some(t), Some(&key), None)
85                .expect("enc init");
86            ctx.set_iv_length(aead_aes_128_gcm::IV_LEN)
87                .expect("IV length");
88            ctx.set_padding(false);
89        } else {
90            ctx.decrypt_init(Some(t), Some(&key), None)
91                .expect("dec init");
92        }
93
94        OsslAeadAes128Gcm(ctx)
95    }
96
97    fn encrypt(
98        &mut self,
99        iv: &[u8; aead_aes_128_gcm::IV_LEN],
100        aad: &[u8],
101        input: &[u8],
102        output: &mut [u8],
103    ) -> Result<(), CryptoError> {
104        assert!(
105            aad.len() >= 12,
106            "Associated data length MUST be at least 12 octets"
107        );
108
109        // Set the IV
110        self.0.encrypt_init(None, None, Some(iv))?;
111
112        // Add the additional authenticated data, omitting the output argument informs
113        // OpenSSL that we are providing AAD.
114        let aad_c = self.0.cipher_update(aad, None)?;
115        // TODO: This should maybe be an error
116        assert!(aad_c == aad.len());
117
118        let count = self.0.cipher_update(input, Some(output))?;
119        let final_count = self.0.cipher_final(&mut output[count..])?;
120
121        // Get the authentication tag and append it to the output
122        let tag_offset = count + final_count;
123        self.0
124            .tag(&mut output[tag_offset..tag_offset + aead_aes_128_gcm::TAG_LEN])?;
125
126        Ok(())
127    }
128
129    fn decrypt(
130        &mut self,
131        iv: &[u8; aead_aes_128_gcm::IV_LEN],
132        aads: &[&[u8]],
133        input: &[u8],
134        output: &mut [u8],
135    ) -> Result<usize, CryptoError> {
136        // This needs to be converted to an error maybe
137        assert!(input.len() >= aead_aes_128_gcm::TAG_LEN);
138
139        let (cipher_text, tag) = input.split_at(input.len() - aead_aes_128_gcm::TAG_LEN);
140
141        self.0.decrypt_init(None, None, Some(iv))?;
142
143        // Add the additional authenticated data, omitting the output argument informs
144        // OpenSSL that we are providing AAD.
145        // With this the authentication tag will be verified.
146        for aad in aads {
147            self.0.cipher_update(aad, None)?;
148        }
149
150        self.0.set_tag(tag)?;
151
152        let count = self.0.cipher_update(cipher_text, Some(output))?;
153
154        let final_count = self.0.cipher_final(&mut output[count..])?;
155
156        Ok(count + final_count)
157    }
158}
