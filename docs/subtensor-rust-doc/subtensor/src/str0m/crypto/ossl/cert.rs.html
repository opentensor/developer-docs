17const RSA_F4: u32 = 0x10001;

19/// Certificate used for DTLS.
20#[derive(Debug, Clone)]
21pub struct OsslDtlsCert {
22    pub(crate) pkey: PKey<Private>,
23    pub(crate) x509: X509,
24}
25
26impl OsslDtlsCert {
27    /// Creates a new (self signed) DTLS certificate.
28    pub fn new() -> Self {
29        Self::self_signed().expect("create dtls cert")
30    }
31
32    // The libWebRTC code we try to match is at:
33    // https://webrtc.googlesource.com/src/+/1568f1b1330f94494197696fe235094e6293b258/rtc_base/openssl_certificate.cc#58
34    fn self_signed() -> Result<Self, CryptoError> {
35        let f4 = BigNum::from_u32(RSA_F4).unwrap();
36        let key = Rsa::generate_with_e(2048, &f4)?;
37        let pkey = PKey::from_rsa(key)?;
38
39        let mut x509b = X509::builder()?;
40        x509b.set_version(2)?; // X509.V3 (zero indexed)
41
42        // For firefox, we must increase the serial number for each generated certificate.
43        // See https://github.com/versatica/mediasoup/issues/127#issuecomment-474460153
44        static SERIAL: AtomicU32 = AtomicU32::new(1);
45        let serial = SERIAL.fetch_add(1, Ordering::SeqCst);
46
47        let serial_bn = BigNum::from_u32(serial)?;
48        let serial = Asn1Integer::from_bn(&serial_bn)?;
49        x509b.set_serial_number(&serial)?;
50        let before = Asn1Time::from_unix(unix_time() - 3600)?;
51        x509b.set_not_before(&before)?;
52        let after = Asn1Time::days_from_now(7)?;
53        x509b.set_not_after(&after)?;
54        x509b.set_pubkey(&pkey)?;
55
56        // The libWebRTC code for this is:
57        //
58        // !X509_NAME_add_entry_by_NID(name.get(), NID_commonName, MBSTRING_UTF8,
59        // (unsigned char*)params.common_name.c_str(), -1, -1, 0) ||
60        //
61        // libWebRTC allows this name to be configured by the user of the library.
62        // That's a future TODO for str0m.
63        let mut nameb = X509Name::builder()?;
64        nameb.append_entry_by_nid_with_type(
65            Nid::COMMONNAME,
66            DTLS_CERT_IDENTITY,
67            Asn1Type::UTF8STRING,
68        )?;
69
70        let name = nameb.build();
71
72        x509b.set_subject_name(&name)?;
73        x509b.set_issuer_name(&name)?;
74
75        x509b.sign(&pkey, MessageDigest::sha1())?;
76        let x509 = x509b.build();
77
78        Ok(OsslDtlsCert { pkey, x509 })
79    }
80
81    /// Produce a (public) fingerprint of the cert.
82    ///
83    /// This is sent via SDP to the other peer to lock down the DTLS
84    /// to this specific certificate.
85    pub fn fingerprint(&self) -> Fingerprint {
86        let digest: &[u8] = &self
87            .x509
88            .digest(MessageDigest::sha256())
89            .expect("digest to fingerprint");
90
91        Fingerprint {
92            hash_func: "sha-256".into(),
93            bytes: digest.to_vec(),
94        }
95    }
96}

98// TODO: Refactor away this use of System::now, to instead go via InstantExt

102// This is not a super high priority since it's only used for setting a before

104pub fn unix_time() -> i64 {
105    SystemTime::now()
106        .duration_since(SystemTime::UNIX_EPOCH)
107        .unwrap()
108        .as_secs() as i64
