42    /// Accept an [`SdpOffer`] from the remote peer. If this call returns successfully, the
43    /// changes will have been made to the session. The resulting [`SdpAnswer`] should be
44    /// sent to the remote peer.
45    ///
46    /// <b>Note. Pending changes from a previous non-completed [`SdpApi`][super::SdpApi] will be
47    /// considered rolled back when calling this function.</b>
48    ///
49    /// The incoming SDP is validated in various ways which can cause this call to fail.
50    /// Example of such problems would be an SDP without any m-lines, missing `a=fingerprint`
51    /// or if `a=group` doesn't match the number of m-lines.
52    ///
53    /// ```no_run
54    /// # use str0m::Rtc;
55    /// # use str0m::change::{SdpOffer};
56    /// // obtain offer from remote peer.
57    /// let json_offer: &[u8] = todo!();
58    /// let offer: SdpOffer = serde_json::from_slice(json_offer).unwrap();
59    ///
60    /// let mut rtc = Rtc::new();
61    /// let answer = rtc.sdp_api().accept_offer(offer).unwrap();
62    ///
63    /// // send json_answer to remote peer.
64    /// let json_answer = serde_json::to_vec(&answer).unwrap();
65    /// ```
66    pub fn accept_offer(self, offer: SdpOffer) -> Result<SdpAnswer, RtcError> {
67        debug!("Accept offer");
68
69        // Invalidate any outstanding PendingOffer.
70        self.rtc.next_change_id();
71
72        if offer.media_lines.is_empty() {
73            return Err(RtcError::RemoteSdp("No m-lines in offer".into()));
74        }
75
76        add_ice_details(self.rtc, &offer, None)?;
77
78        if self.rtc.remote_fingerprint.is_none() {
79            if let Some(f) = offer.fingerprint() {
80                self.rtc.remote_fingerprint = Some(f);
81            } else {
82                self.rtc.disconnect();
83                return Err(RtcError::RemoteSdp("missing a=fingerprint".into()));
84            }
85        }
86
87        if !self.rtc.dtls.is_inited() {
88            // The side that makes the first offer is the controlling side.
89            self.rtc.ice.set_controlling(false);
90        }
91
92        // Ensure setup=active/passive is corresponding remote and init dtls.
93        init_dtls(self.rtc, &offer)?;
94
95        // Modify session with offer
96        apply_offer(&mut self.rtc.session, offer)?;
97
98        // Handle potentially new m=application line.
99        let client = self.rtc.dtls.is_active().expect("DTLS active to be set");
100        if self.rtc.session.app().is_some() {
101            self.rtc.init_sctp(client);
102        }
103
104        let params = AsSdpParams::new(self.rtc, None);
105        let sdp = as_sdp(&self.rtc.session, params);
106
107        debug!("Create answer");
108        Ok(sdp.into())
109    }

111    /// Accept an answer to a previously created [`SdpOffer`].
112    ///
113    /// This function returns an [`RtcError::ChangesOutOfOrder`] if we have created and applied another
114    /// [`SdpApi`][super::SdpApi] before calling this. The same also happens if we use
115    /// [`SdpApi::accept_offer()`] before using this pending instance.
116    ///
117    /// ```no_run
118    /// # use str0m::Rtc;
119    /// # use str0m::media::{MediaKind, Direction};
120    /// # use str0m::change::SdpAnswer;
121    /// let mut rtc = Rtc::new();
122    ///
123    /// let mut changes = rtc.sdp_api();
124    /// let mid = changes.add_media(MediaKind::Audio, Direction::SendOnly, None, None);
125    /// let (offer, pending) = changes.apply().unwrap();
126    ///
127    /// // send offer to remote peer, receive answer back
128    /// let answer: SdpAnswer = todo!();
129    ///
130    /// rtc.sdp_api().accept_answer(pending, answer).unwrap();
131    /// ```
132    pub fn accept_answer(
133        self,
134        mut pending: SdpPendingOffer,
135        answer: SdpAnswer,
136    ) -> Result<(), RtcError> {
137        debug!("Accept answer");
138
139        // Ensure we don't use the wrong changes below. We must use that of pending.
140        drop(self.changes);
141
142        if !self.rtc.is_correct_change_id(pending.change_id) {
143            return Err(RtcError::ChangesOutOfOrder);
144        }
145
146        add_ice_details(self.rtc, &answer, Some(&pending))?;
147
148        // Ensure setup=active/passive is corresponding remote and init dtls.
149        init_dtls(self.rtc, &answer)?;
150
151        if self.rtc.remote_fingerprint.is_none() {
152            if let Some(f) = answer.fingerprint() {
153                self.rtc.remote_fingerprint = Some(f);
154            } else {
155                self.rtc.disconnect();
156                return Err(RtcError::RemoteSdp("missing a=fingerprint".into()));
157            }
158        }
159
160        // Split out new channels, since that is not handled by the Session.
161        let new_channels = pending.changes.take_new_channels();
162
163        // Modify session with answer
164        apply_answer(&mut self.rtc.session, pending.changes, answer)?;
165
166        // Handle potentially new m=application line.
167        let client = self.rtc.dtls.is_active().expect("DTLS to be inited");
168        if self.rtc.session.app().is_some() {
169            self.rtc.init_sctp(client);
170        }
171
172        for (id, config) in new_channels {
173            self.rtc.chan.confirm(id, config);
174        }
175
176        Ok(())
177    }

197    /// Add audio or video media and get the `mid` that will be used.
198    ///
199    /// Each call will result in a new m-line in the offer identified by the [`Mid`].
200    ///
201    /// The mid is not valid to use until the SDP offer-answer dance is complete and
202    /// the mid been advertised via [`Event::MediaAdded`][crate::Event::MediaAdded].
203    ///
204    /// * `stream_id` is used to synchronize media. It is `a=msid-semantic: WMS <streamId>` line in SDP.
205    /// * `track_id` is becomes both the track id in `a=msid <streamId> <trackId>` as well as the
206    ///   CNAME in the RTP SDES.
207    ///
208    /// ```
209    /// # use str0m::{Rtc, media::MediaKind, media::Direction};
210    /// let mut rtc = Rtc::new();
211    ///
212    /// let mut changes = rtc.sdp_api();
213    ///
214    /// let mid = changes.add_media(MediaKind::Audio, Direction::SendRecv, None, None);
215    /// ```
216    pub fn add_media(
217        &mut self,
218        kind: MediaKind,
219        dir: Direction,
220        stream_id: Option<String>,
221        track_id: Option<String>,
222    ) -> Mid {
223        let mid = self.rtc.new_mid();
224
225        // https://www.rfc-editor.org/rfc/rfc8830
226        // msid-id = 1*64token-char
227        fn is_token_char(c: &char) -> bool {
228            // token-char = %x21 / %x23-27 / %x2A-2B / %x2D-2E / %x30-39
229            // / %x41-5A / %x5E-7E
230            let u = *c as u32;
231            u == 0x21
232                || (0x23..=0x27).contains(&u)
233                || (0x2a..=0x2b).contains(&u)
234                || (0x2d..=0x2e).contains(&u)
235                || (0x30..=0x39).contains(&u)
236                || (0x41..=0x5a).contains(&u)
237                || (0x5e..0x7e).contains(&u)
238        }
239
240        let stream_id = if let Some(stream_id) = stream_id {
241            stream_id.chars().filter(is_token_char).take(64).collect()
242        } else {
243            Id::<20>::random().to_string()
244        };
245
246        let track_id = if let Some(track_id) = track_id {
247            track_id.chars().filter(is_token_char).take(64).collect()
248        } else {
249            Id::<20>::random().to_string()
250        };
251
252        let rtx = kind.is_video().then(|| self.rtc.session.streams.new_ssrc());
253        let ssrcs = vec![(self.rtc.session.streams.new_ssrc(), rtx)];
254
255        // TODO: let user configure stream/track name.
256        let msid = Msid {
257            stream_id,
258            track_id: track_id.clone(),
259        };
260
261        let add = AddMedia {
262            mid,
263            cname: track_id,
264            msid,
265            kind,
266            dir,
267            ssrcs,
268
269            // Added later
270            pts: vec![],
271            exts: ExtensionMap::empty(),
272            index: 0,
273        };
274
275        self.changes.0.push(Change::AddMedia(add));
276        mid
277    }

281    /// All media have a direction. The media can be added by this side via

293        if changed {
294            self.changes.0.push(Change::Direction(mid, dir));

298    /// Add a new data channel and get the `id` that will be used.
299    ///
300    /// The first ever data channel added to a WebRTC session results in a media
301    /// of a special "application" type in the SDP. The m-line is for a SCTP association over
302    /// DTLS, and all data channels are multiplexed over this single association.
303    ///
304    /// That means only the first ever `add_channel` will result in an [`SdpOffer`].
305    /// Consecutive channels will be opened without needing a negotiation.
306    ///
307    /// The label is used to identify the data channel to the remote peer. This is mostly
308    /// useful whe multiple channels are in use at the same time.
309    ///
310    /// ```
311    /// # use str0m::Rtc;
312    /// let mut rtc = Rtc::new();
313    ///
314    /// let mut changes = rtc.sdp_api();
315    ///
316    /// let cid = changes.add_channel("my special channel".to_string());
317    /// ```
318    pub fn add_channel(&mut self, label: String) -> ChannelId {
319        let has_media = self.rtc.session.app().is_some();
320        let changes_contains_add_app = self.changes.contains_add_app();
321
322        if !has_media && !changes_contains_add_app {
323            let mid = self.rtc.new_mid();
324            self.changes.0.push(Change::AddApp(mid));
325        }
326
327        let config = ChannelConfig {
328            label,
329            ..Default::default()
330        };
331
332        let id = self.rtc.chan.new_channel(&config);
333
334        self.changes.0.push(Change::AddChannel((id, config)));
335
336        id
337    }

339    /// Perform an ICE restart.
340    ///
341    /// Only one ICE restart can be pending at the time. Calling this repeatedly removes any other
342    /// pending ICE restart.
343    ///
344    /// The local ICE candidates can be kept as is, or be cleared out, in which case new ice
345    /// candidates must be added via [`Rtc::add_local_candidate`] before connectivity can be
346    /// re-established.
347    ///
348    /// Returns the new ICE credentials that will be used going forward.
349    pub fn ice_restart(&mut self, keep_local_candidates: bool) -> IceCreds {
350        self.changes
351            .retain(|c| !matches!(c, Change::IceRestart(_, _)));
352
353        let new_creds = IceCreds::new();
354        self.changes
355            .push(Change::IceRestart(new_creds.clone(), keep_local_candidates));
356
357        new_creds
358    }

360    /// Attempt to apply the changes made.
361    ///
362    /// If this returns [`SdpOffer`], the caller the changes are
363    /// not happening straight away, and the caller is expected to do a negotiation with the remote
364    /// peer and apply the answer using [`SdpPendingOffer`].
365    ///
366    /// In case this returns `None`, there either were no changes, or the changes could be applied
367    /// without doing a negotiation. Specifically for additional [`SdpApi::add_channel()`]
368    /// after the first, there is no negotiation needed.
369    ///
370    /// The [`SdpPendingOffer`] is valid until the next time we call this function, at which
371    /// point using it will raise an error. Using [`SdpApi::accept_offer()`] will also invalidate
372    /// the current [`SdpPendingOffer`].
373    ///
374    /// ```
375    /// # use str0m::Rtc;
376    /// let mut rtc = Rtc::new();
377    ///
378    /// let changes = rtc.sdp_api();
379    /// assert!(changes.apply().is_none());
380    /// ```
381    pub fn apply(self) -> Option<(SdpOffer, SdpPendingOffer)> {
382        if self.changes.is_empty() {
383            return None;
384        }
385
386        let change_id = self.rtc.next_change_id();
387
388        let requires_negotiation = self.changes.0.iter().any(requires_negotiation);
389
390        if requires_negotiation {
391            let offer = create_offer(self.rtc, &self.changes);
392            let pending = SdpPendingOffer {
393                change_id,
394                changes: self.changes,
395            };
396            debug!("Create offer");
397            Some((offer, pending))
398        } else {
399            debug!("Apply direct changes");
400            apply_direct_changes(self.rtc, self.changes);
401            None
402        }
403    }

405    /// Combines the modifications made in [`SdpApi`] with those in [`SdpPendingOffer`].
406    ///
407    /// This function merges the changes present in [`SdpApi`] with the changes
408    /// in [`SdpPendingOffer`]. In result this [`SdpApi`] will incorporate modifications
409    /// from both the previous [`SdpPendingOffer`] and any newly added changes.
410    ///
411    /// ## Example
412    ///
413    /// ```no_run
414    /// # use str0m::media::{Direction, MediaKind};
415    /// # use str0m::Rtc;
416    /// let mut rtc = Rtc::new();
417    /// let mut changes = rtc.sdp_api();
418    /// changes.add_media(MediaKind::Audio, Direction::SendOnly, None, None);
419    /// let (_offer, pending) = changes.apply().unwrap();
420    ///
421    /// let mut changes = rtc.sdp_api();
422    /// changes.add_media(MediaKind::Video, Direction::SendOnly, None, None);
423    /// changes.merge(pending);
424    ///
425    /// // This `SdpOffer` will have changes from the first `SdpPendingChanges`
426    /// // and new changes from `SdpApi`
427    /// let (_offer, pending) = changes.apply().unwrap();
428    /// ```
429    pub fn merge(&mut self, mut pending_offer: SdpPendingOffer) {
430        pending_offer.retain_relevant(self.rtc);
431        self.changes.extend(pending_offer.changes.drain(..));
432    }
433}

435/// Pending offer from a previous [`Rtc::sdp_api()`] call.
436///
437/// This allows us to accept a remote answer. No changes have been made to the session
438/// before we call [`SdpApi::accept_answer()`], which means that rolling back a
439/// change is as simple as dropping this instance.
440///
441/// ```no_run
442/// # use str0m::Rtc;
443/// # use str0m::media::{MediaKind, Direction};
444/// # use str0m::change::SdpAnswer;
445/// let mut rtc = Rtc::new();
446///
447/// let mut changes = rtc.sdp_api();
448/// let mid = changes.add_media(MediaKind::Audio, Direction::SendOnly, None, None);
449/// let (offer, pending) = changes.apply().unwrap();
450///
451/// // send offer to remote peer, receive answer back
452/// let answer: SdpAnswer = todo!();
453///
454/// rtc.sdp_api().accept_answer(pending, answer).unwrap();
455/// ```
456pub struct SdpPendingOffer {
457    change_id: usize,
458    changes: Changes,
459}
460
461impl SdpPendingOffer {
462    /// Retains only the relevant changes in the `changes` vector based on the provided `Rtc` instance.
463    ///
464    /// This function filters the vector of `Change` instances stored in the current object and retains
465    /// only those changes that are considered relevant with respect to the provided `Rtc` instance.
466    fn retain_relevant(&mut self, rtc: &Rtc) {
467        fn is_relevant(rtc: &Rtc, c: &Change) -> bool {
468            match c {
469                Change::AddMedia(v) => rtc.media(v.mid).is_none(),
470                Change::AddApp(_) => rtc.session.app().is_none(),
471                Change::AddChannel(v) => rtc.chan.stream_id_by_channel_id(v.0).is_none(),
472                Change::Direction(m, d) => {
473                    // If mid is missing, this is not relevant.
474                    rtc.media(*m).map(|m| m.direction() != *d).unwrap_or(false)
475                }
476                Change::IceRestart(v, _) => rtc.ice.local_credentials() != v,
477            }
478        }
479
480        self.changes.retain(|c| is_relevant(rtc, c));
481    }
482}

487impl Changes {
488    /// Details of the active ICE restart, if any.
489    ///
490    /// Returns the new local ICE credentials and the whether to keep local ICE candidates if an
491    /// ICE restart has been initiated in the offer, otherwise [`None`].
492    fn ice_restart(&self) -> Option<(IceCreds, bool)> {
493        self.iter().find_map(|c| match c {
494            Change::IceRestart(creds, keep_local_candidates) => {
495                Some((creds.clone(), *keep_local_candidates))
496            }
497            _ => None,
498        })
499    }

543fn requires_negotiation(c: &Change) -> bool {
544    match c {
545        Change::IceRestart(_, _) => true,
546        Change::AddMedia(_) => true,
547        Change::AddApp(_) => true,
548        Change::AddChannel(_) => false,
549        Change::Direction(_, _) => true,
550    }
551}

553fn apply_direct_changes(rtc: &mut Rtc, mut changes: Changes) {
554    // Split out new channels, since that is not handled by the Session.
555    let new_channels = changes.take_new_channels();
556
557    for (id, config) in new_channels {
558        rtc.chan.confirm(id, config);
559    }
560}

562fn create_offer(rtc: &mut Rtc, changes: &Changes) -> SdpOffer {
563    if !rtc.dtls.is_inited() {
564        // The side that makes the first offer is the controlling side.
565        rtc.ice.set_controlling(true);
566    }
567
568    let params = AsSdpParams::new(rtc, Some(changes));
569    let sdp = as_sdp(&rtc.session, params);
570
571    sdp.into()
572}

574fn add_ice_details(
575    rtc: &mut Rtc,
576    sdp: &Sdp,
577    pending: Option<&SdpPendingOffer>,
578) -> Result<(), RtcError> {
579    let Some(creds) = sdp.ice_creds() else {
580        return Err(RtcError::RemoteSdp("missing a=ice-ufrag/pwd".into()));
581    };
582
583    // If we are handling an **offer** from the remote, differing ICE credentials indicate an ICE
584    // restart initiated by the remote.
585    //
586    // If we are handling an **answer** from the remote, differing ICE credentials indicate an
587    // acceptance of an ICE restart we requested.
588    let ice_restart = match rtc.ice.remote_credentials() {
589        Some(v) => *v != creds,
590        None => false,
591    };
592    if ice_restart {
593        let (new_local_creds, keep_local_candidates) = if let Some(pending) = pending {
594            // Since we have a pending, this is an answer to our offer.
595            pending.changes.ice_restart().ok_or_else(||
596                // Answer contained changed remote creds, indicating an ice restart
597                // but since we have no pending ice-creds, we didn't initiate it
598                // Ice restart in an ANSWER breaks spec.
599                    RtcError::RemoteSdp(
600                    "Ice restart in answer without one in the preceeding offer".into(),
601                ))?
602        } else {
603            // The remote OFFER had an ice restart, and we need to respond with
604            // new credentials in the ANSWER.
605            (IceCreds::new(), true)
606        };
607
608        rtc.ice
609            .ice_restart(new_local_creds.clone(), keep_local_candidates);
610    }
611
612    rtc.ice.set_remote_credentials(creds);
613
614    for r in sdp.ice_candidates() {
615        rtc.ice.add_remote_candidate(r.clone());
616    }
617
618    Ok(())
619}

621fn init_dtls(rtc: &mut Rtc, remote_sdp: &Sdp) -> Result<(), RtcError> {
622    let setup = match remote_sdp.setup() {
623        Some(v) => match v {
624            // Remote being ActPass, we take Passive role.
625            Setup::ActPass => Setup::Passive,
626            _ => v.invert(),
627        },
628
629        None => {
630            warn!("Missing a=setup line");
631            Setup::Passive
632        }
633    };
634
635    let active = setup == Setup::Active;
636    rtc.init_dtls(active)?;
637
638    Ok(())
639}

641fn as_sdp(session: &Session, params: AsSdpParams) -> Sdp {
642    let (media_lines, mids, stream_ids) = {
643        let mut v = as_media_lines(session);
644
645        let mut new_lines = vec![];
646
647        // When creating new m-lines from the pending changes, the m-line index starts from this.
648        let new_index_start = v.len();
649
650        // If there are additions in the pending changes, prepend them now.
651        if let Some(pending) = params.pending {
652            new_lines = pending
653                .as_new_medias(new_index_start, &session.codec_config, &session.exts)
654                .collect();
655        }
656
657        // Add potentially new m-lines to the existing ones.
658        v.extend(new_lines.iter().map(|n| n as &dyn AsSdpMediaLine));
659
660        // Turn into sdp::MediaLine (m-line).
661        let mut lines = v
662            .iter()
663            .map(|m| {
664                // Candidates should only be in the first BUNDLE mid
665                let include_candidates = m.index() == 0;
666
667                let attrs = params.media_attributes(include_candidates);
668
669                // Already made send stream SSRCs
670                let mut ssrcs = session.streams.ssrcs_tx(m.mid());
671
672                // Merged with pending stream SSRCs
673                if let Some(pending) = params.pending {
674                    ssrcs.extend(pending.ssrcs_for_mid(m.mid()))
675                }
676
677                let params: Vec<_> = session
678                    .codec_config
679                    .all_for_kind(m.kind())
680                    .cloned()
681                    .collect();
682
683                m.as_media_line(attrs, &ssrcs, &session.exts, &params)
684            })
685            .collect::<Vec<_>>();
686
687        if let Some(pending) = params.pending {
688            pending.apply_to(&mut lines);
689        }
690
691        // Mids go into the session part of the SDP.
692        let mids = v.iter().map(|m| m.mid()).collect();
693
694        let mut stream_ids = vec![];
695        for msid in v.iter().filter_map(|v| v.msid()) {
696            if !stream_ids.contains(&msid.stream_id) {
697                stream_ids.push(msid.stream_id.clone());
698            }
699        }
700
701        (lines, mids, stream_ids)
702    };
703
704    let mut attrs = vec![
705        SessionAttribute::Group {
706            typ: "BUNDLE".into(),
707            mids,
708        },
709        SessionAttribute::MsidSemantic {
710            semantic: "WMS".to_string(),
711            stream_ids,
712        },
713    ];
714
715    if session.ice_lite {
716        attrs.push(SessionAttribute::IceLite);
717    }
718
719    Sdp {
720        session: sdp::Session {
721            id: session.id(),
722            bw: None,
723            attrs,
724        },
725        media_lines,
726    }
727}

729fn apply_offer(session: &mut Session, offer: SdpOffer) -> Result<(), RtcError> {
730    offer.assert_consistency()?;
731
732    update_session(session, &offer);
733
734    let new_lines = sync_medias(session, &offer).map_err(RtcError::RemoteSdp)?;
735
736    add_new_lines(session, &new_lines, true).map_err(RtcError::RemoteSdp)?;
737
738    ensure_stream_tx(session);
739
740    Ok(())
741}
742
743fn apply_answer(
744    session: &mut Session,
745    pending: Changes,
746    answer: SdpAnswer,
747) -> Result<(), RtcError> {
748    answer.assert_consistency()?;
749
750    update_session(session, &answer);
751
752    let new_lines = sync_medias(session, &answer).map_err(RtcError::RemoteSdp)?;
753
754    // The new_lines from the answer must correspond to what we sent in the offer.
755    if let Some(err) = pending.ensure_correct_answer(&new_lines) {
756        return Err(RtcError::RemoteSdp(err));
757    }
758
759    add_new_lines(session, &new_lines, false).map_err(RtcError::RemoteSdp)?;
760
761    // Add all pending changes (since we pre-allocated SSRC communicated in the Offer).
762    add_pending_changes(session, pending);
763
764    ensure_stream_tx(session);
765
766    Ok(())
767}
768
769fn ensure_stream_tx(session: &mut Session) {
770    for media in &session.medias {
771        // Only make send streams when we have to.
772        if !media.direction().is_sending() {
773            continue;
774        }
775
776        let mut rids: Vec<Option<Rid>> = vec![];
777
778        if let Some(sim) = media.simulcast() {
779            for rid in &*sim.send {
780                let rid: Rid = rid.0.as_str().into();
781                rids.push(Some(rid));
782            }
783        } else {
784            rids.push(None);
785        }
786
787        // If any payload param has RTX, we need to prepare for RTX. This is because we always
788        // communicate a=ssrc lines, which need to be complete with main and RTX SSRC.
789        let has_rtx = session
790            .codec_config
791            .iter()
792            .filter(|p| media.remote_pts().contains(&p.pt))
793            .any(|p| p.resend().is_some());
794
795        for rid in rids {
796            // If we already have the stream, we don't make any new one.
797            let has_stream = session
798                .streams
799                .stream_tx_by_mid_rid(media.mid(), rid)
800                .is_some();
801
802            if has_stream {
803                continue;
804            }
805
806            let (ssrc, rtx) = if has_rtx {
807                let (ssrc, rtx) = session.streams.new_ssrc_pair();
808                (ssrc, Some(rtx))
809            } else {
810                let ssrc = session.streams.new_ssrc();
811                (ssrc, None)
812            };
813
814            let stream = session
815                .streams
816                .declare_stream_tx(ssrc, rtx, media.mid(), rid);
817
818            // Configure cache size
819            let size = if media.kind().is_audio() {
820                session.send_buffer_audio
821            } else {
822                session.send_buffer_video
823            };
824
825            stream.set_rtx_cache(size, DEFAULT_RTX_CACHE_DURATION);
826        }
827    }
828}
829
830fn add_pending_changes(session: &mut Session, pending: Changes) {
831    // For pending AddMedia, we have outgoing SSRC communicated that needs to be added.
832    for change in pending.0 {
833        let add_media = match change {
834            Change::AddMedia(v) => v,
835            _ => continue,
836        };
837
838        let media = session
839            .medias
840            .iter_mut()
841            .find(|m| m.mid() == add_media.mid)
842            .expect("Media to be added for pending mid");
843
844        // the cname/msid has already been communicated in the offer, we need to kep
845        // it the same once the m-line is created.
846        media.set_cname(add_media.cname);
847        media.set_msid(add_media.msid);
848
849        for (ssrc, rtx) in add_media.ssrcs {
850            // TODO: When we allow sending RID, we need to add that here.
851            let stream = session
852                .streams
853                .declare_stream_tx(ssrc, rtx, add_media.mid, None);
854
855            let size = if media.kind().is_audio() {
856                session.send_buffer_audio
857            } else {
858                session.send_buffer_video
859            };
860
861            stream.set_rtx_cache(size, DEFAULT_RTX_CACHE_DURATION);
862        }
863    }
864}

866/// Compares m-lines in Sdp with that already in the session.
867///
868/// * Existing m-lines can apply changes (such as direction change).
869/// * New m-lines are returned to the caller.
870fn sync_medias<'a>(session: &mut Session, sdp: &'a Sdp) -> Result<Vec<&'a MediaLine>, String> {
871    let mut new_lines = Vec::with_capacity(sdp.media_lines.len());
872
873    for (idx, m) in sdp.media_lines.iter().enumerate() {
874        // First, match existing m-lines.
875        match m.typ {
876            MediaType::Application => {
877                if let Some((_, index)) = session.app() {
878                    if idx != *index {
879                        return index_err(m.mid());
880                    }
881                    continue;
882                }
883            }
884            MediaType::Audio | MediaType::Video => {
885                if let Some(media) = session.medias.iter_mut().find(|l| l.mid() == m.mid()) {
886                    if idx != media.index() {
887                        return index_err(m.mid());
888                    }
889
890                    update_media(
891                        media,
892                        m,
893                        &mut session.codec_config,
894                        &session.exts,
895                        &mut session.streams,
896                    );
897
898                    continue;
899                }
900            }
901            _ => {
902                continue;
903            }
904        }
905
906        // Second, discover new m-lines.
907        new_lines.push(m);
908    }
909
910    fn index_err<T>(mid: Mid) -> Result<T, String> {
911        Err(format!("Changed order for m-line with mid: {mid}"))
912    }
913
914    Ok(new_lines)
915}

917/// Adds new m-lines as found in an offer or answer.
918fn add_new_lines(
919    session: &mut Session,
920    new_lines: &[&MediaLine],
921    is_offer: bool,
922) -> Result<(), String> {
923    for m in new_lines {
924        let idx = session.line_count();
925
926        if m.typ.is_media() {
927            let mut media = Media::from_remote_media_line(m, idx, is_offer);
928            media.need_open_event = is_offer;
929
930            // Match/remap remote params.
931            session
932                .codec_config
933                .update_params(&m.rtp_params(), m.direction());
934
935            // Remap the extension to that of the answer.
936            session.exts.remap(&m.extmaps());
937
938            update_media(
939                &mut media,
940                m,
941                &mut session.codec_config,
942                &session.exts,
943                &mut session.streams,
944            );
945
946            session.add_media(media);
947        } else if m.typ.is_channel() {
948            session.set_app(m.mid(), idx)?;
949        } else {
950            return Err(format!(
951                "New m-line is neither media nor channel: {}",
952                m.mid()
953            ));
954        }
955    }
956
957    Ok(())
958}

960/// Update session level properties like
961/// Extensions from offer or answer.
962fn update_session(session: &mut Session, sdp: &Sdp) {
963    // Does any m-line contain a a=rtcp-fb:xx transport-cc?
964    let has_transport_cc = sdp
965        .media_lines
966        .iter()
967        .any(|m| m.rtp_params().iter().any(|p| p.fb_transport_cc));
968
969    // Is the session level sequence number enabled?
970    let has_twcc_header = session
971        .exts
972        .id_of(Extension::TransportSequenceNumber)
973        .is_some();
974
975    // Since twcc feedback is session wide and not per m-line or pt, we enable it if
976    // there are _any_ m-line with a a=rtcp-fb transport-cc parameter and the sequence
977    // number header is enabled.
978    if has_transport_cc && has_twcc_header {
979        session.enable_twcc_feedback();
980    }
981}

983/// Returns all media/channels as `AsMediaLine` trait.
984fn as_media_lines(session: &Session) -> Vec<&dyn AsSdpMediaLine> {
985    let mut v = vec![];
986
987    if let Some(app) = session.app() {
988        v.push(app as &dyn AsSdpMediaLine);
989    }
990    v.extend(session.medias().iter().map(|m| m as &dyn AsSdpMediaLine));
991    v.sort_by_key(|f| f.index());
992    v
993}
994
995fn update_media(
996    media: &mut Media,
997    m: &MediaLine,
998    config: &mut CodecConfig,
999    exts: &ExtensionMap,
1000    streams: &mut Streams,
1001) {
1002    // Direction changes
1003    //
1004    // All changes come from the other side, either via an incoming OFFER
1005    // or a ANSWER from our OFFER. Either way, the direction is inverted to
1006    // how we have it locally.
1007    let new_dir = m.direction().invert();
1008    //
1009    let change_direction_disallowed = !media.remote_created()
1010        && media.direction() == Direction::Inactive
1011        && new_dir == Direction::SendOnly;
1012
1013    if change_direction_disallowed {
1014        info!(
1015            "Ignore attempt to change inactive to recvonly by remote peer for locally created mid: {}",
1016            media.mid()
1017        );
1018    } else {
1019        media.set_direction(new_dir);
1020    }
1021
1022    for rid in m.rids().iter() {
1023        media.expect_rid(*rid);
1024    }
1025
1026    // Narrowing/ordering of of PT
1027    let pts: Vec<Pt> = m
1028        .rtp_params()
1029        .into_iter()
1030        .filter_map(|p| config.sdp_match_remote(p, m.direction()))
1031        .collect();
1032    media.set_remote_pts(pts);
1033
1034    let mut remote_extmap = ExtensionMap::empty();
1035    for (id, ext) in m.extmaps().into_iter() {
1036        // The remapping of extensions should already have happened, which
1037        // means the ID are matching in the session to the remote.
1038        if exts.lookup(id) != Some(ext) {
1039            // Don't set any extensions that aren't enabled in Session.
1040            continue;
1041        }
1042        remote_extmap.set(id, ext.clone());
1043    }
1044    media.set_remote_extmap(remote_extmap);
1045
1046    if new_dir.is_receiving() {
1047        // SSRC changes
1048        // This will always be for ReceiverSource since any incoming a=ssrc line will be
1049        // about the remote side's SSRC.
1050        let infos = m.ssrc_info();
1051        let main = infos.iter().filter(|i| i.repairs.is_none());
1052
1053        if m.simulcast().is_none() {
1054            // Only use pre-communicated SSRC if we are running without simulcast.
1055            // We found a bug in FF where the order of the simulcast lines does not
1056            // correspond to the order of the simulcast declarations. In this case
1057            // it's better to fall back on mid/rid dynamic mapping.
1058
1059            for i in main {
1060                // TODO: If the remote is communicating _BOTH_ rid and a=ssrc this will fail.
1061                info!("Adding pre-communicated SSRC: {:?}", i);
1062                let repair_ssrc = infos
1063                    .iter()
1064                    .find(|r| r.repairs == Some(i.ssrc))
1065                    .map(|r| r.ssrc);
1066
1067                // If remote communicated a main a=ssrc, but no RTX, we will not send nacks.
1068                let suppress_nack = repair_ssrc.is_none();
1069                streams.expect_stream_rx(
1070                    i.ssrc,
1071                    repair_ssrc,
1072                    media.mid(),
1073                    None,
1074                    suppress_nack,
1075                    None,
1076                );
1077            }
1078        }
1079
1080        // Simulcast configuration
1081        if let Some(s) = m.simulcast() {
1082            if s.is_munged {
1083                warn!("Not supporting simulcast via munging SDP");
1084            } else if media.simulcast().is_none() {
1085                // Invert before setting, since it has a recv and send config.
1086                media.set_simulcast(s.invert());
1087            }
1088        }
1089    }
1090}

1094    fn msid(&self) -> Option<&Msid>;
1095    fn index(&self) -> usize;

1124        _params: &[PayloadParams],
1125    ) -> MediaLine {
1126        attrs.push(MediaAttribute::Mid(self.0));
1127        attrs.push(MediaAttribute::SctpPort(5000));
1128        attrs.push(MediaAttribute::MaxMessageSize(262144));
1129
1130        MediaLine {
1131            typ: sdp::MediaType::Application,
1132            disabled: false,

1134            pts: vec![],

1136            attrs,
1137        }

1141impl AsSdpMediaLine for Media {
1142    fn mid(&self) -> Mid {
1143        Media::mid(self)
1144    }
1145    fn msid(&self) -> Option<&Msid> {
1146        Some(Media::msid(self))
1147    }
1148    fn index(&self) -> usize {
1149        Media::index(self)
1150    }
1151    fn kind(&self) -> MediaKind {
1152        Media::kind(self)
1153    }
1154    fn as_media_line(
1155        &self,
1156        mut attrs: Vec<MediaAttribute>,
1157        ssrcs_tx: &[(Ssrc, Option<Ssrc>)],
1158        exts: &ExtensionMap,
1159        params: &[PayloadParams],
1160    ) -> MediaLine {
1161        if self.app_tmp {
1162            let app = (self.mid(), self.index());
1163            return app.as_media_line(attrs, ssrcs_tx, exts, params);
1164        }
1165
1166        attrs.push(MediaAttribute::Mid(self.mid()));
1167
1168        let audio = self.kind() == MediaKind::Audio;
1169        for (id, ext) in self.remote_extmap().iter_by_media_type(audio) {
1170            attrs.push(MediaAttribute::ExtMap {
1171                id,
1172                ext: ext.clone(),
1173            });
1174        }
1175
1176        attrs.push(self.direction().into());
1177        attrs.push(MediaAttribute::Msid(self.msid().clone()));
1178        attrs.push(MediaAttribute::RtcpMux);
1179
1180        // The effective params start from the Session::codec_config to retain the
1181        // user's configured preferred order, however they are narrowed only include
1182        // those the remote peer wants.
1183        let effective_params = params.iter().filter(|p| self.remote_pts().contains(&p.pt));
1184
1185        let mut pts = vec![];
1186
1187        for p in effective_params {
1188            p.as_media_attrs(&mut attrs);
1189
1190            // The pts that will be advertised in the SDP
1191            pts.push(p.pt());
1192            if let Some(rtx) = p.resend() {
1193                pts.push(rtx);
1194            }
1195        }
1196
1197        if let Some(s) = self.simulcast() {
1198            fn to_rids<'a>(
1199                gs: &'a SimulcastGroups,
1200                direction: &'static str,
1201            ) -> impl Iterator<Item = MediaAttribute> + 'a {
1202                gs.iter().map(move |rid| MediaAttribute::Rid {
1203                    id: rid.clone(),
1204                    direction,
1205                    pt: vec![],
1206                    restriction: vec![],
1207                })
1208            }
1209            attrs.extend(to_rids(&s.recv, "recv"));
1210            attrs.extend(to_rids(&s.send, "send"));
1211            attrs.push(MediaAttribute::Simulcast(s.clone()));
1212        }
1213
1214        // Outgoing SSRCs
1215        let msid = format!("{} {}", self.msid().stream_id, self.msid().track_id);
1216        for (ssrc, ssrc_rtx) in ssrcs_tx {
1217            attrs.push(MediaAttribute::Ssrc {
1218                ssrc: *ssrc,
1219                attr: "cname".to_string(),
1220                value: self.cname().to_string(),
1221            });
1222            attrs.push(MediaAttribute::Ssrc {
1223                ssrc: *ssrc,
1224                attr: "msid".to_string(),
1225                value: msid.clone(),
1226            });
1227            if let Some(ssrc_rtx) = ssrc_rtx {
1228                attrs.push(MediaAttribute::Ssrc {
1229                    ssrc: *ssrc_rtx,
1230                    attr: "cname".to_string(),
1231                    value: self.cname().to_string(),
1232                });
1233                attrs.push(MediaAttribute::Ssrc {
1234                    ssrc: *ssrc_rtx,
1235                    attr: "msid".to_string(),
1236                    value: msid.clone(),
1237                });
1238            }
1239        }
1240
1241        let count = ssrcs_tx.len();
1242        #[allow(clippy::comparison_chain)]
1243        if count == 1 {
1244            let (ssrc, ssrc_rtx) = &ssrcs_tx[0];
1245            if let Some(ssrc_rtx) = ssrc_rtx {
1246                attrs.push(MediaAttribute::SsrcGroup {
1247                    semantics: "FID".to_string(),
1248                    ssrcs: vec![*ssrc, *ssrc_rtx],
1249                });
1250            }
1251        } else {
1252            // TODO: handle simulcast
1253        }
1254
1255        MediaLine {
1256            typ: self.kind().into(),
1257            disabled: false,
1258            proto: Proto::Srtp,
1259            pts,
1260            bw: None,
1261            attrs,
1262        }
1263    }
1264}

1275struct AsSdpParams<'a, 'b> {
1276    pub candidates: Vec<Candidate>,
1277    pub creds: IceCreds,
1278    pub fingerprint: &'a Fingerprint,
1279    pub setup: Setup,
1280    pub pending: Option<&'b Changes>,
1281}
1282
1283impl<'a, 'b> AsSdpParams<'a, 'b> {
1284    pub fn new(rtc: &'a Rtc, pending: Option<&'b Changes>) -> Self {
1285        let (creds, candidates) = if let Some((new_creds, keep_local_candidates)) =
1286            pending.and_then(|p| p.ice_restart())
1287        {
1288            if keep_local_candidates {
1289                // If we are performing an ICE restart and we are keeping the same
1290                // candidates we need to use ufrag from the new ICE credentials
1291                // in our offer.
1292                let mut new_candidates = rtc.ice.local_candidates().to_vec();
1293                for c in &mut new_candidates {
1294                    c.set_ufrag(&new_creds.ufrag);
1295                }
1296
1297                (new_creds, new_candidates)
1298            } else {
1299                (new_creds, vec![])
1300            }
1301        } else {
1302            (
1303                rtc.ice.local_credentials().clone(),
1304                rtc.ice.local_candidates().to_vec(),
1305            )
1306        };
1307
1308        AsSdpParams {
1309            candidates,
1310            creds,
1311            fingerprint: rtc.dtls.local_fingerprint(),
1312            setup: match rtc.dtls.is_active() {
1313                Some(true) => Setup::Active,
1314                Some(false) => Setup::Passive,
1315                None => Setup::ActPass,
1316            },
1317            pending,
1318        }
1319    }
1320
1321    fn media_attributes(&self, include_candidates: bool) -> Vec<MediaAttribute> {
1322        use MediaAttribute::*;
1323
1324        let mut v = if include_candidates {
1325            self.candidates
1326                .iter()
1327                .map(|c| Candidate(c.clone()))
1328                .collect()
1329        } else {
1330            vec![]
1331        };
1332
1333        v.push(IceUfrag(self.creds.ufrag.clone()));
1334        v.push(IcePwd(self.creds.pass.clone()));
1335        v.push(IceOptions("trickle".into()));
1336        v.push(Fingerprint(self.fingerprint.clone()));
1337        v.push(Setup(self.setup));
1338
1339        v
1340    }
1341}

1349impl Changes {
1350    pub fn contains_add_app(&self) -> bool {
1351        for i in 0..self.0.len() {
1352            if matches!(&self.0[i], Change::AddApp(_)) {
1353                return true;
1354            }
1355        }
1356        false
1357    }
1358
1359    pub fn take_new_channels(&mut self) -> Vec<(ChannelId, ChannelConfig)> {
1360        let mut v = vec![];
1361
1362        if self.0.is_empty() {
1363            return v;
1364        }
1365
1366        for i in (0..self.0.len()).rev() {
1367            if matches!(&self.0[i], Change::AddChannel(_)) {
1368                if let Change::AddChannel(id) = self.0.remove(i) {
1369                    v.push(id);
1370                }
1371            }
1372        }
1373
1374        v
1375    }
1376
1377    /// Tests the given lines (from answer) corresponds to changes.
1378    fn ensure_correct_answer(&self, lines: &[&MediaLine]) -> Option<String> {
1379        if self.count_new_medias() != lines.len() {
1380            return Some(format!(
1381                "Differing m-line count in offer vs answer: {} != {}",
1382                self.count_new_medias(),
1383                lines.len()
1384            ));
1385        }
1386
1387        'next: for l in lines {
1388            let mid = l.mid();
1389
1390            for m in &self.0 {
1391                use Change::*;
1392                match m {
1393                    AddMedia(v) if v.mid == mid => {
1394                        if !l.typ.is_media() {
1395                            return Some(format!(
1396                                "Answer m-line for mid ({}) is not of media type: {:?}",
1397                                mid, l.typ
1398                            ));
1399                        }
1400                        continue 'next;
1401                    }
1402                    AddApp(v) if *v == mid => {
1403                        if !l.typ.is_channel() {
1404                            return Some(format!(
1405                                "Answer m-line for mid ({}) is not a data channel: {:?}",
1406                                mid, l.typ
1407                            ));
1408                        }
1409                        continue 'next;
1410                    }
1411                    _ => {}
1412                }
1413            }
1414
1415            return Some(format!("Mid in answer is not in offer: {mid}"));
1416        }
1417
1418        None
1419    }
1420
1421    fn count_new_medias(&self) -> usize {
1422        self.0
1423            .iter()
1424            .filter(|c| matches!(c, Change::AddMedia(_) | Change::AddApp(_)))
1425            .count()
1426    }
1427
1428    pub fn as_new_medias<'a, 'b: 'a>(
1429        &'a self,
1430        index_start: usize,
1431        config: &'b CodecConfig,
1432        exts: &'b ExtensionMap,
1433    ) -> impl Iterator<Item = Media> + '_ {
1434        self.0
1435            .iter()
1436            .enumerate()
1437            .filter_map(move |(idx, c)| c.as_new_media(index_start + idx, config, exts))
1438    }
1439
1440    pub(crate) fn apply_to(&self, lines: &mut [MediaLine]) {
1441        for change in &self.0 {
1442            if let Change::Direction(mid, dir) = change {
1443                if let Some(line) = lines.iter_mut().find(|l| l.mid() == *mid) {
1444                    if let Some(dir_pos) = line.attrs.iter().position(|a| a.is_direction()) {
1445                        line.attrs[dir_pos] = (*dir).into();
1446                    }
1447                }
1448            }
1449        }
1450    }
1451
1452    fn ssrcs_for_mid(&self, mid: Mid) -> &[(Ssrc, Option<Ssrc>)] {
1453        let maybe_add_media = self
1454            .0
1455            .iter()
1456            .filter_map(|c| {
1457                if let Change::AddMedia(m) = c {
1458                    Some(m)
1459                } else {
1460                    None
1461                }
1462            })
1463            .find(|m| m.mid == mid);
1464
1465        let Some(m) = maybe_add_media else {
1466            return &[];
1467        };
1468
1469        &m.ssrcs
1470    }
1471}

1473impl Change {
1474    fn as_new_media(
1475        &self,
1476        index: usize,
1477        config: &CodecConfig,
1478        exts: &ExtensionMap,
1479    ) -> Option<Media> {
1480        use Change::*;
1481        match self {
1482            AddMedia(v) => {
1483                // TODO can we avoid all this cloning?
1484                let mut add = v.clone();
1485                add.pts = config.all_for_kind(v.kind).map(|p| p.pt()).collect();
1486                add.exts = exts.cloned_with_type(v.kind.is_audio());
1487                add.index = index;
1488
1489                Some(Media::from_add_media(add))
1490            }
1491            AddApp(mid) => Some(Media::from_app_tmp(*mid, index)),
1492            _ => None,
1493        }
1494    }
