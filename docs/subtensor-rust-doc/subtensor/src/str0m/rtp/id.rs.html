10macro_rules! str_id {
11    ($id:ident, $name:literal, $num:tt, $new_len:tt) => {
12        impl $id {
13            /// Creates a new random id.
14            pub fn new() -> $id {
15                let mut arr = Id::<$num>::random().into_array();
16                for i in $new_len..$num {
17                    arr[i] = b' ';
18                }
19                $id(arr)
20            }
21
22            /// Converts an array of bytes to an id.
23            pub const fn from_array(a: [u8; $num]) -> $id {
24                $id(a)
25            }
26        }
27
28        impl fmt::Display for $id {
29            fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
30                let s: &str = self;
31                write!(f, "{}", s)
32            }
33        }
34
35        impl fmt::Debug for $id {
36            fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
37                let s: &str = self;
38                write!(f, "{}({})", $name, s)
39            }
40        }
41
42        impl Deref for $id {
43            type Target = str;
44
45            fn deref(&self) -> &Self::Target {
46                from_utf8(&self.0).expect("ascii id").trim()
47            }
48        }
49
50        impl<'a> From<&'a str> for $id {
51            fn from(v: &'a str) -> Self {
52                let v = v
53                    .chars()
54                    .map(|c| if c.is_ascii_alphanumeric() { c } else { '_' })
55                    .collect::<String>();
56
57                let bytes = v.as_bytes();
58                let bytes = &bytes[0..$num.min(bytes.len())];
59
60                // pad with space.
61                let mut array = [b' '; $num];
62
63                let max = bytes.len().min(array.len());
64                (&mut array[0..max]).copy_from_slice(bytes);
65
66                $id(array)
67            }
68        }
69
70        impl Default for $id {
71            fn default() -> Self {
72                $id::new()
73            }
74        }
75    };
76}

78macro_rules! num_id {
79    ($id:ident, $t:tt) => {
80        impl $id {
81            /// Creates a new random id.
82            pub fn new() -> Self {
83                $id(NonCryptographicRng::$t())
84            }
85        }
86
87        impl Deref for $id {
88            type Target = $t;
89
90            fn deref(&self) -> &Self::Target {
91                &self.0
92            }
93        }
94
95        impl From<$t> for $id {
96            fn from(v: $t) -> Self {
97                $id(v)
98            }
99        }
100
101        impl fmt::Display for $id {
102            fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
103                write!(f, "{}", self.0)
104            }
105        }
106    };
107}

109/// Media identifier.
110///
111/// In SDP this is found per m-line with the attribute `a=mid:<mid>`.
112///
113/// When using Direct API we still need `Mid`, since they group individual
114/// encoded streams. For example a simulcast of 3 layers would have
115/// 3 incoming StreamRx, but since they belong to the same media,
116/// the have the same `Mid`.
117#[derive(Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
118pub struct Mid([u8; 16]);
119str_id!(Mid, "Mid", 16, 3);

121/// Identifier of a simulcast layer for an encoded stream.

123/// The abbreviation means "RTP Stream Id", which is a very confusing name, because

128#[derive(Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
129pub struct Rid([u8; 8]);
130str_id!(Rid, "Rid", 8, 3);

141/// Paylad type.

158/// Sequence number of an RTP packet.
159///
160/// An RTP packet is identified by: SSRC + SeqNo. However in the RTP header the sequence number
161/// is a `u16`, meaning the value quite quickly "rolls over". To uniquely identify a packet,
162/// str0m keeps track of the roll overs and converts the `u16` to `u64` in this `SeqNo`.
163///
164/// To get the RTP u16 value from a `SeqNo`, use `as_u16()` or cast it to u16.
165///
166/// ```
167/// # use str0m::rtp::SeqNo;
168/// let seq_no: SeqNo = 65_537.into();
169///
170/// // Use `as_u16()`.
171/// let a = seq_no.as_u16();
172/// // Discard upper 48 bits to get RTP u16.
173/// let b = *seq_no as u16;
174///
175/// assert_eq!(a, 1);
176/// assert_eq!(b, 1);
177/// ```
178#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord, Serialize, Deserialize)]
179pub struct SeqNo(u64);
180num_id!(SeqNo, u64);
181
182impl SeqNo {
183    pub(crate) const MAX: SeqNo = SeqNo(u64::MAX);
184
185    /// Check if the `other` sequence number is directly following this.
186    #[inline(always)]
187    pub fn is_next(&self, other: SeqNo) -> bool {
188        if **self >= *other {
189            return false;
190        }
191        *other - **self == 1
192    }
193
194    /// Increase (mutate) this sequence number and return the previous value.
195    #[inline(always)]
196    pub fn inc(&mut self) -> SeqNo {
197        let n = SeqNo(self.0);
198        self.0 += 1;
199        n
200    }
201
202    #[inline(always)]
203    pub(crate) fn is_max(&self) -> bool {
204        self.0 == Self::MAX.0
205    }
206
207    /// The RTP header value (discarding the ROC).
208    ///
209    /// This is the same as discarding the top 48 bits by casting to a u16.
210    ///
211    /// ```
212    /// # use str0m::rtp::SeqNo;
213    /// let seq_no: SeqNo = 65_537.into();
214    ///
215    /// // Use `as_u16()`.
216    /// let a = seq_no.as_u16();
217    ///
218    /// assert_eq!(a, 1);
219    /// ```
220    #[inline(always)]
221    pub fn as_u16(&self) -> u16 {
222        self.0 as u16
223    }
224
225    /// Get the rollover counter (ROC) value.
226    ///
227    /// ```
228    /// # use str0m::rtp::SeqNo;
229    /// // More than 2^16, thus rolled over.
230    /// let seq_no: SeqNo = 95_000.into();
231    ///
232    /// assert_eq!(seq_no.roc(), 1);
233    ///
234    /// // Is the same as shifting 16 bits.
235    /// assert_eq!(seq_no.roc(), 95_000 >> 16);
236    /// ```
237    #[inline(always)]
238    pub fn roc(&self) -> u64 {
239        self.0 >> 16
240    }
241}
