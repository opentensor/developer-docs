53impl RtcpPacket for Descriptions {
54    fn header(&self) -> RtcpHeader {
55        RtcpHeader {
56            rtcp_type: RtcpType::SourceDescription,
57            feedback_message_type: FeedbackMessageType::SourceCount(self.reports.len() as u8),
58            words_less_one: (self.length_words() - 1) as u16,
59        }
60    }
61
62    fn length_words(&self) -> usize {
63        // * header: 1
64        // * size-per-item * items
65        1 + self.reports.iter().map(|r| r.word_size()).sum::<usize>()
66    }
67
68    fn write_to(&self, buf: &mut [u8]) -> usize {
69        self.header().write_to(buf);
70
71        let mut buf = &mut buf[4..];
72        let mut tot = 4;
73
74        for r in &*self.reports {
75            let n = r.write_to(buf);
76            buf = &mut buf[n..];
77            tot += n;
78        }
79
80        tot
81    }
82}

84impl Sdes {
85    fn write_to(&self, buf: &mut [u8]) -> usize {
86        buf[..4].copy_from_slice(&self.ssrc.to_be_bytes());
87        let mut tot = 4;
88
89        let mut buf = &mut buf[4..];
90        for (t, v) in &self.values {
91            let bytes = v.as_bytes();
92            let len = bytes.len();
93
94            buf[0] = *t as u8;
95            buf[1] = len as u8;
96
97            buf = &mut buf[2..];
98            buf[..len].copy_from_slice(bytes);
99
100            buf = &mut buf[len..];
101            tot += 2 + len;
102        }
103
104        buf[0] = SdesType::END as u8;
105        buf = &mut buf[1..];
106        tot += 1;
107
108        let pad = 4 - tot % 4;
109        if pad < 4 {
110            #[allow(clippy::needless_range_loop)]
111            for i in 0..pad {
112                buf[i] = SdesType::END as u8;
113            }
114            tot += pad;
115        }
116
117        assert!(tot % 4 == 0, "Sdes is padded to word boundary");
118
119        tot
120    }
121}

123impl WordSized for Sdes {
124    fn word_size(&self) -> usize {
125        let byte_size = 4 + self
126            .values
127            .iter()
128            // 2 here for 2 byte encoding of type + length
129            .map(|(_, s)| 2 + s.as_bytes().len())
130            .sum::<usize>();
131
132        let padded = pad_bytes_to_word(byte_size);
133
134        padded / 4
135    }
136}

158impl<'a> TryFrom<&'a [u8]> for Descriptions {
159    type Error = &'static str;
160
161    fn try_from(buf: &'a [u8]) -> Result<Self, Self::Error> {
162        let mut reports = ReportList::new();
163
164        let mut buf = buf;
165
166        loop {
167            if reports.len() == 31 {
168                break;
169            }
170            // For some reason FF sends us a full SDES and then [0,0,0,0] at the end.
171            // This can't be interpreted as SDES, so we just ignore it.
172            if buf.len() < 8 {
173                break;
174            }
175            let report: Sdes = buf.try_into()?;
176
177            let len = report.word_size() * 4;
178            buf = &buf[len..];
179
180            reports.push(report);
181        }
182
183        Ok(Descriptions {
184            reports: Box::new(reports),
185        })
186    }
187}

189impl<'a> TryFrom<&'a [u8]> for Sdes {
190    type Error = &'static str;
191
192    fn try_from(buf: &'a [u8]) -> Result<Self, Self::Error> {
193        if buf.len() < 8 {
194            return Err("Less than 8 bytes for Sdes");
195        }
196
197        let ssrc = u32::from_be_bytes([buf[0], buf[1], buf[2], buf[3]]).into();
198        let mut values = ReportList::new();
199
200        let mut buf = &buf[4..];
201        let mut abs = 0;
202
203        loop {
204            if buf.len() < 2 {
205                return Err("Less than 2 bytes for next Sdes value");
206            }
207
208            let stype: SdesType = buf[0].into();
209
210            if matches!(stype, SdesType::END) {
211                // The end of SDES.
212
213                // Each chunk consists of an SSRC/CSRC identifier followed by a list of
214                // zero or more items, which carry information about the SSRC/CSRC.
215                // Each chunk starts on a 32-bit boundary.
216                //
217                // Items are contiguous, i.e., items are not individually padded to a
218                // 32-bit boundary.  Text is not null terminated because some multi-
219                // octet encodings include null octets.
220                //
221                // No length octet follows the null item type octet, but additional null
222                // octets MUST be included if needed to pad until the next 32-bit
223                // boundary.
224
225                let pad = 4 - abs % 4;
226                if pad < 4 && buf.len() < pad {
227                    return Err("Not enough buf.len() for Sdes padding");
228                }
229
230                break;
231            }
232
233            let len = buf[1] as usize;
234
235            if buf.len() < 2 + len {
236                return Err("Not enough buf.len() for Sdes value");
237            }
238            buf = &buf[2..];
239            abs += 2;
240
241            if let Ok(value) = from_utf8(&buf[..len]) {
242                values.push((stype, value.to_string()));
243            } else {
244                // failed to read as utf-8. skip.
245            }
246
247            buf = &buf[len..];
248            abs += len;
249        }
250
251        Ok(Sdes { ssrc, values })
252    }
253}
