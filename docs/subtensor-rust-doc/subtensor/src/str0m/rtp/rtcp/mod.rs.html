94    pub(crate) fn read_packet(buf: &[u8], feedback: &mut VecDeque<Rtcp>) {
95        let mut buf = buf;
96        loop {
97            if buf.is_empty() {
98                break;
99            }
100
101            let header: RtcpHeader = match buf.try_into() {
102                Ok(v) => v,
103                Err(e) => {
104                    debug!("{}", e);
105                    break;
106                }
107            };
108            let has_padding = buf[0] & 0b00_1_00000 > 0;
109            let full_length = header.length_words() * 4;
110
111            if full_length > buf.len() {
112                // this length is incorrect.
113                break;
114            }
115
116            let unpadded_length = if has_padding {
117                let pad = buf[full_length - 1] as usize;
118                if full_length < pad {
119                    debug!("buf.len() is less than padding: {} < {}", full_length, pad);
120                    break;
121                }
122                full_length - pad
123            } else {
124                full_length
125            };
126
127            match (&buf[..unpadded_length]).try_into() {
128                Ok(v) => feedback.push_back(v),
129                Err(e) => debug!("{}", e),
130            }
131
132            buf = &buf[full_length..];
133        }
134    }

136    pub(crate) fn write_packet(
137        feedback: &mut VecDeque<Rtcp>,
138        buf: &mut [u8],
139        mut output: impl FnMut(Rtcp),
140    ) -> usize {
141        if feedback.is_empty() {
142            return 0;
143        }
144
145        // Total length, in bytes, shrunk to be on the pad_to boundary.
146        let total_len = buf.len();
147
148        // Capacity in words
149        let word_capacity = total_len / 4;
150
151        // Pack RTCP feedback packets. Merge together ones of the same type.
152        Rtcp::pack(feedback, word_capacity);
153
154        let mut offset = 0;
155        while let Some(fb) = feedback.front() {
156            // Length of next item.
157            let item_len = fb.length_words() * 4;
158
159            // Capacity left in the buffer.
160            let capacity = total_len - offset;
161            if capacity < item_len {
162                break;
163            }
164
165            // We definitely can fit the next RTCP item.
166            let fb = feedback.pop_front().unwrap();
167            let written = fb.write_to(&mut buf[offset..]);
168
169            assert_eq!(
170                written, item_len,
171                "length_words equals write_to length: {fb:?}"
172            );
173
174            // When debugging we can pass an output to get the serialized packets.
175            output(fb);
176
177            // Move offsets for the amount written.
178            offset += item_len;
179        }
180
181        offset
182    }

184    fn merge(&mut self, other: &mut Rtcp, words_left: usize) -> bool {
185        match (self, other) {
186            // Stack receiver reports into sender reports.
187            (Rtcp::SenderReport(sr), Rtcp::ReceiverReport(rr)) => {
188                let n = sr.reports.append_all_possible(&mut rr.reports, words_left);
189                n > 0
190            }
191
192            // Stack receiver reports.
193            (Rtcp::ReceiverReport(r1), Rtcp::ReceiverReport(r2)) => {
194                let n = r1.reports.append_all_possible(&mut r2.reports, words_left);
195                n > 0
196            }
197
198            // Stack source descriptions.
199            (Rtcp::SourceDescription(s1), Rtcp::SourceDescription(s2)) => {
200                let n = s1.reports.append_all_possible(&mut s2.reports, words_left);
201                n > 0
202            }
203
204            // Stack source descriptions.
205            (Rtcp::Goodbye(g1), Rtcp::Goodbye(g2)) => {
206                let n = g1.reports.append_all_possible(&mut g2.reports, words_left);
207                n > 0
208            }
209
210            // Stack Nack
211            (Rtcp::Nack(n1), Rtcp::Nack(n2)) if n1.ssrc == n2.ssrc => {
212                let n = n1.reports.append_all_possible(&mut n2.reports, words_left);
213                n > 0
214            }
215
216            // Stack source descriptions.
217            (Rtcp::Fir(f1), Rtcp::Fir(f2)) => {
218                let n = f1.reports.append_all_possible(&mut f2.reports, words_left);
219                n > 0
220            }
221
222            // No merge possible
223            _ => false,
224        }
225    }

227    fn is_full(&self) -> bool {
228        match self {
229            Rtcp::SenderReport(v) => v.reports.is_full(),
230            Rtcp::ReceiverReport(v) => v.reports.is_full(),
231            Rtcp::ExtendedReport(_) => true,
232            Rtcp::SourceDescription(v) => v.reports.is_full(),
233            Rtcp::Goodbye(v) => v.reports.is_full(),
234            Rtcp::Nack(v) => v.reports.is_full(),
235            Rtcp::Pli(_) => true,
236            Rtcp::Fir(v) => v.reports.is_full(),
237            Rtcp::Twcc(_) => true,
238            Rtcp::Remb(_) => true,
239        }
240    }

242    /// If this RtcpFb contains no reports (anymore). This can happen after
243    /// merging reports together.
244    fn is_empty(&self) -> bool {
245        match self {
246            // A SenderReport always has, at least, the SenderInfo part.
247            Rtcp::SenderReport(_) => false,
248            // ReceiverReport can become empty.
249            Rtcp::ReceiverReport(v) => v.reports.is_empty(),
250            // ExtendedReport can become empty.
251            Rtcp::ExtendedReport(v) => v.blocks.is_empty(),
252            // SourceDescription can become empty.
253            Rtcp::SourceDescription(v) => v.reports.is_empty(),
254            // Goodbye can become empty,
255            Rtcp::Goodbye(v) => v.reports.is_empty(),
256            // Nack can become empty
257            Rtcp::Nack(v) => v.reports.is_empty(),
258            // Nack is never empty
259            Rtcp::Pli(_) => false,
260            // Fir can be merged to empty.
261            Rtcp::Fir(v) => v.reports.is_empty(),
262            // A twcc report is never empty.
263            Rtcp::Twcc(_) => false,
264            // A REMB report is never empty.
265            Rtcp::Remb(_) => false,
266        }
267    }

269    fn pack(feedback: &mut VecDeque<Self>, mut word_capacity: usize) {
270        // Index into feedback of item we are to pack into.
271        let mut i = 0;
272        let len = feedback.len();
273
274        // Need at least on feedback to pack into, and one to take from.
275        if len < 2 {
276            return;
277        }
278
279        // SenderReport/ReceiveReport first for SRTCP.
280        feedback.make_contiguous().sort_by_key(Self::order_no);
281
282        'outer: loop {
283            // If we reach last element, there is no more packing to do.
284            if i == len - 1 {
285                break;
286            }
287
288            let (pack_into, pack_from) = feedback.make_contiguous().split_at_mut(i + 1);
289            let fb_a = pack_into.last_mut().unwrap();
290
291            // abort if fb_a won't fit in the spare capacity.
292            if word_capacity < fb_a.length_words() {
293                break 'outer;
294            }
295
296            // if we manage to merge anything into fb_a.
297            let mut any_change = false;
298
299            // fb_b goes from the item _after_ i
300            for fb_b in pack_from {
301                // if fb_a is full (or empty), we don't want to move any more elements into fb_a.
302                if fb_a.is_full() || fb_a.is_empty() {
303                    break;
304                }
305
306                // abort if fb_a won't fit in the spare capacity.
307                if word_capacity < fb_a.length_words() {
308                    break 'outer;
309                }
310
311                // amount of capacity (in words) left to fill.
312                let capacity = word_capacity - fb_a.length_words();
313
314                // attempt to merge some elements into fb_a from fb_b.
315                let did_merge = fb_a.merge(fb_b, capacity);
316                any_change |= did_merge;
317            }
318
319            if !any_change {
320                word_capacity -= fb_a.length_words();
321                i += 1;
322            }
323        }
324
325        // Prune empty.
326        feedback.retain(|f| !f.is_empty());
327    }
328
329    fn order_no(&self) -> u8 {
330        use Rtcp::*;
331        match self {
332            // SenderReport/ReceiverReport first since they possibly contain
333            // the SSRC for the SRTCP encryption.
334            SenderReport(_) => 0,
335            ReceiverReport(_) => 1,
336
337            SourceDescription(_) => 2,
338            Nack(_) => 3,
339            Pli(_) => 4,
340            Fir(_) => 5,
341            Twcc(_) => 6,
342            Remb(_) => 7,
343            ExtendedReport(_) => 10,
344
345            // Goodbye last since they remove stuff.
346            Goodbye(_) => 11,
347        }
348    }

398impl<'a> TryFrom<&'a [u8]> for Rtcp {
399    type Error = &'static str;
400
401    fn try_from(buf: &'a [u8]) -> Result<Self, Self::Error> {
402        let header: RtcpHeader = buf.try_into()?;
403
404        // By constraining the length, all subparsing can go
405        // until they exhaust the buffer length. This presupposes
406        // padding is removed from the input.
407        let buf = &buf[4..];
408
409        Ok(match header.rtcp_type() {
410            RtcpType::SenderReport => Rtcp::SenderReport(buf.try_into()?),
411            RtcpType::ReceiverReport => Rtcp::ReceiverReport(buf.try_into()?),
412            RtcpType::SourceDescription => Rtcp::SourceDescription(buf.try_into()?),
413            RtcpType::Goodbye => Rtcp::Goodbye((header.count(), buf).try_into()?),
414            RtcpType::ApplicationDefined => return Err("Ignore RTCP type: ApplicationDefined"),
415            RtcpType::TransportLayerFeedback => {
416                let tlfb = match header.feedback_message_type() {
417                    FeedbackMessageType::TransportFeedback(v) => v,
418                    _ => return Err("Expected TransportFeedback in FeedbackMessageType"),
419                };
420
421                match tlfb {
422                    TransportType::Nack => Rtcp::Nack(buf.try_into()?),
423                    TransportType::TransportWide => Rtcp::Twcc(buf.try_into()?),
424                }
425            }
426            RtcpType::PayloadSpecificFeedback => {
427                let plfb = match header.feedback_message_type() {
428                    FeedbackMessageType::PayloadFeedback(v) => v,
429                    _ => return Err("Expected PayloadFeedback in FeedbackMessageType"),
430                };
431
432                match plfb {
433                    PayloadType::PictureLossIndication => Rtcp::Pli(buf.try_into()?),
434                    PayloadType::SliceLossIndication => return Err("Ignore PayloadType type: SLI"),
435                    PayloadType::ReferencePictureSelectionIndication => {
436                        return Err("Ignore PayloadType type: RPSI")
437                    }
438                    PayloadType::FullIntraRequest => Rtcp::Fir(buf.try_into()?),
439                    PayloadType::ApplicationLayer => {
440                        if header.rtcp_type() == RtcpType::PayloadSpecificFeedback {
441                            if let Ok(remb) = Remb::try_from(buf) {
442                                return Ok(Rtcp::Remb(remb));
443                            }
444                        }
445                        return Err("Ignore PayloadType: ApplicationLayer");
446                    }
447                }
448            }
449            RtcpType::ExtendedReport => Rtcp::ExtendedReport(buf.try_into()?),
450        })
451    }
452}

460/// Pad up to the next word (4 byte) boundary.
461fn pad_bytes_to_word(n: usize) -> usize {
462    let pad = 4 - n % 4;
463    if pad == 4 {
464        n
465    } else {
466        n + pad
467    }
