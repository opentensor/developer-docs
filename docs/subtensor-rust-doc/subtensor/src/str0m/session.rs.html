30/// Minimum time we delay between sending nacks. This should be
31/// set high enough to not cause additional problems in very bad

104    pub fn new(config: &RtcConfig) -> Self {
105        let mut id = SessionId::new();
106        // Max 2^62 - 1: https://bugzilla.mozilla.org/show_bug.cgi?id=861895
107        const MAX_ID: u64 = 2_u64.pow(62) - 1;
108        while *id > MAX_ID {
109            id = (*id >> 1).into();
110        }
111        let (pacer, bwe) = if let Some(rate) = config.bwe_initial_bitrate {
112            let pacer = PacerImpl::LeakyBucket(LeakyBucketPacer::new(rate * PACING_FACTOR * 2.0));
113
114            let send_side_bwe = SendSideBandwithEstimator::new(rate);
115            let bwe = Bwe {
116                bwe: send_side_bwe,
117                desired_bitrate: Bitrate::ZERO,
118                current_bitrate: rate,
119
120                last_emitted_estimate: Bitrate::ZERO,
121            };
122
123            (pacer, Some(bwe))
124        } else {
125            (PacerImpl::Null(NullPacer::default()), None)
126        };
127
128        Session {
129            id,
130            medias: vec![],
131            streams: Streams::default(),
132            app: None,
133            reordering_size_audio: config.reordering_size_audio,
134            reordering_size_video: config.reordering_size_video,
135            send_buffer_audio: config.send_buffer_audio,
136            send_buffer_video: config.send_buffer_video,
137            exts: config.exts.clone(),
138
139            // Both sending and receiving starts from the configured codecs.
140            // These can then be changed in the SDP OFFER/ANSWER dance.
141            codec_config: config.codec_config.clone(),
142
143            srtp_rx: None,
144            srtp_tx: None,
145            last_nack: already_happened(),
146            last_twcc: already_happened(),
147            twcc: 0,
148            twcc_rx_register: TwccRecvRegister::new(100),
149            twcc_tx_register: TwccSendRegister::new(1000),
150            bwe,
151            enable_twcc_feedback: false,
152            pacer,
153            poll_packet_buf: vec![0; 2000],
154            pending_packet: None,
155            ice_lite: config.ice_lite,
156            rtp_mode: config.rtp_mode,
157            feedback_tx: VecDeque::new(),
158            feedback_rx: VecDeque::new(),
159            raw_packets: if config.enable_raw_packets {
160                Some(VecDeque::new())

204    pub fn handle_timeout(&mut self, now: Instant) -> Result<(), RtcError> {
205        // Payload any waiting samples
206        self.do_payload(now)?;
207
208        let sender_ssrc = self.streams.first_ssrc_local();
209
210        let do_nack = now >= self.nack_at();
211
212        self.streams.handle_timeout(
213            now,
214            sender_ssrc,
215            do_nack,
216            &self.medias,
217            &self.codec_config,
218            &mut self.feedback_tx,
219        );
220
221        if do_nack {
222            self.last_nack = now;
223        }
224
225        self.update_queue_state(now);
226
227        if let Some(twcc_at) = self.twcc_at() {
228            if now >= twcc_at {
229                self.create_twcc_feedback(sender_ssrc, now);
230            }
231        }
232
233        if let Some(bwe) = self.bwe.as_mut() {
234            bwe.handle_timeout(now);
235        }
236
237        Ok(())
238    }

240    fn update_queue_state(&mut self, now: Instant) {
241        let iter = self.streams.streams_tx().map(|m| m.queue_state(now));
242
243        let Some(padding_request) = self.pacer.handle_timeout(now, iter) else {
244            return;
245        };
246
247        let stream = self
248            .streams
249            .stream_tx_by_mid_rid(padding_request.mid, None)
250            .expect("pacer to use an existing stream");
251
252        stream.generate_padding(padding_request.padding);
253    }

255    fn create_twcc_feedback(&mut self, sender_ssrc: Ssrc, now: Instant) -> Option<()> {
256        self.last_twcc = now;
257        let mut twcc = self.twcc_rx_register.build_report(DATAGRAM_MTU - 100)?;
258
259        // These SSRC are on medial level, but twcc is on session level,
260        // we fill in the first discovered media SSRC in each direction.
261        twcc.sender_ssrc = sender_ssrc;
262        twcc.ssrc = self.streams.first_ssrc_remote();
263
264        debug!("Created feedback TWCC: {:?}", twcc);
265        self.feedback_tx.push_front(Rtcp::Twcc(twcc));
266        Some(())
267    }

285    fn mid_and_ssrc_for_header(&mut self, header: &RtpHeader) -> Option<(Mid, Ssrc)> {
286        let ssrc_header = header.ssrc;
287
288        if let Some(r) = self.streams.mid_ssrc_rx_by_ssrc_or_rtx(ssrc_header) {
289            return Some(r);
290        }
291
292        // Attempt to dynamically map this header to some Media/ReceiveStream.
293        self.map_dynamic(header);
294
295        // The dynamic mapping might have added an entry by now.
296        self.streams.mid_ssrc_rx_by_ssrc_or_rtx(ssrc_header)
297    }

299    fn map_dynamic(&mut self, header: &RtpHeader) {
300        // There are two strategies for dynamically mapping SSRC. Both use the RTP "mid"
301        // header extension.
302        // A) Mid+Rid - used when doing simulcast. Rid points out which
303        //              simulcast layer is in use. There is a separate header
304        //              to indicate repair (RTX) stream.
305        // B) Mid+PT - when not doing simulcast, the PT identifies whether
306        //             this is a repair stream.
307
308        let Some(mid) = header.ext_vals.mid else {
309            return;
310        };
311        let rid = header.ext_vals.rid.or(header.ext_vals.rid_repair);
312
313        // The media the mid points out. Bail if the mid points to something
314        // we don't know about.
315        let Some(media) = self.medias.iter_mut().find(|m| m.mid() == mid) else {
316            return;
317        };
318
319        // Figure out which payload the PT maps to. Either main or RTX.
320        let maybe_payload = self
321            .codec_config
322            .iter()
323            .find(|p| p.pt() == header.payload_type || p.resend() == Some(header.payload_type));
324
325        // If we don't find it, bail out.
326        let Some(payload) = maybe_payload else {
327            return;
328        };
329
330        if let Some(rid) = rid {
331            // Case A - use the rid_repair header to identify RTX.
332            let is_main = header.ext_vals.rid.is_some();
333
334            self.streams
335                .map_dynamic_by_rid(header.ssrc, mid, rid, media, *payload, is_main);
336        } else {
337            // Case B - the payload type identifies RTX.
338            let is_main = payload.pt() == header.payload_type;
339
340            self.streams
341                .map_dynamic_by_pt(header.ssrc, mid, media, *payload, is_main);
342        }
343    }

345    pub(crate) fn handle_rtp(&mut self, now: Instant, mut header: RtpHeader, buf: &[u8]) {
346        // Rewrite absolute-send-time (if present) to be relative to now.
347        header.ext_vals.update_absolute_send_time(now);
348
349        trace!("Handle RTP: {:?}", header);
350        if let Some(transport_cc) = header.ext_vals.transport_cc {
351            let prev = self.twcc_rx_register.max_seq();
352            let extended = extend_u16(Some(*prev), transport_cc);
353            self.twcc_rx_register.update_seq(extended.into(), now);
354        }
355
356        // The ssrc is the _main_ ssrc (no the rtx, that might be in the header).
357        let Some((mid, ssrc)) = self.mid_and_ssrc_for_header(&header) else {
358            debug!("No mid/SSRC for header: {:?}", header);
359            return;
360        };
361
362        let srtp = match self.srtp_rx.as_mut() {
363            Some(v) => v,
364            None => {
365                trace!("Rejecting SRTP while missing SrtpContext");
366                return;
367            }
368        };
369
370        // Both of these unwraps are fine because mid_and_ssrc_for_header guarantees it.
371        let media = self.medias.iter_mut().find(|m| m.mid() == mid).unwrap();
372        let stream = self.streams.stream_rx(&ssrc).unwrap();
373
374        let params = match main_payload_params(&self.codec_config, header.payload_type) {
375            Some(p) => p,
376            None => {
377                trace!(
378                    "No payload params could be found (main or RTX) for {:?}",
379                    header.payload_type
380                );
381                return;
382            }
383        };
384        let clock_rate = params.spec().clock_rate;
385        let pt = params.pt();
386        let is_repair = pt != header.payload_type;
387
388        // is_repair controls whether update is updating the main register or the RTX register.
389        // Either way we get a seq_no_outer which is used to decrypt the SRTP.
390        let receipt_outer = stream.update(now, &header, clock_rate, is_repair);
391
392        let mut data = match srtp.unprotect_rtp(buf, &header, *receipt_outer.seq_no) {
393            Some(v) => v,
394            None => {
395                trace!("Failed to unprotect SRTP");
396                return;
397            }
398        };
399
400        if header.has_padding && !RtpHeader::unpad_payload(&mut data) {
401            // Unpadding failed. Broken data?
402            trace!("unpadding of unprotected payload failed");
403            return;
404        }
405
406        if let Some(raw_packets) = &mut self.raw_packets {
407            raw_packets.push_back(Box::new(RawPacket::RtpRx(header.clone(), data.clone())));
408        }
409
410        // RTX packets must be rewritten to be a normal packet. This only changes the
411        // the seq_no, however MediaTime might be different when interpreted against the
412        // the "main" register.
413        let receipt = if is_repair {
414            // Drop RTX packets that are just empty padding. The payload here
415            // is empty because we would have done RtpHeader::unpad_payload above.
416            // For unpausing, it's enough with the stream.update() already done above.
417            if data.is_empty() {
418                return;
419            }
420
421            // Rewrite the header, and removes the resent seq_no from the body.
422            stream.un_rtx(&mut header, &mut data, pt);
423
424            // Now update the "main" register with the repaired packet info.
425            // This gives us the extended sequence number of the main stream.
426            stream.update(now, &header, clock_rate, false)
427        } else {
428            // This is not RTX, the outer seq and time is what we use. The first
429            // stream.update will have updated the main register.
430            receipt_outer
431        };
432
433        let Some(packet) = stream.handle_rtp(now, header, data, receipt.seq_no, receipt.time)
434        else {
435            return;
436        };
437
438        if self.rtp_mode {
439            // In RTP mode, we store the packet temporarily here for the next poll_output().
440            // However only if this is a packet not seen before. This filters out spurious resends for padding.
441            if receipt.is_new_packet {
442                self.pending_packet = Some(packet);
443            }
444        } else {
445            // In non-RTP mode, we let the Media use a Depayloader.
446            media.depayload(
447                stream.rid(),
448                packet,
449                self.reordering_size_audio,
450                self.reordering_size_video,
451                &self.codec_config,
452            );
453        }
454    }

456    fn handle_rtcp(&mut self, now: Instant, buf: &[u8]) -> Option<()> {
457        let srtp: &mut SrtpContext = self.srtp_rx.as_mut()?;
458        let unprotected = srtp.unprotect_rtcp(buf)?;
459
460        Rtcp::read_packet(&unprotected, &mut self.feedback_rx);
461        let mut need_configure_pacer = false;
462
463        if let Some(raw_packets) = &mut self.raw_packets {
464            for fb in &self.feedback_rx {
465                raw_packets.push_back(Box::new(RawPacket::RtcpRx(fb.clone())));
466            }
467        }
468
469        for fb in RtcpFb::from_rtcp(self.feedback_rx.drain(..)) {
470            if let RtcpFb::Twcc(twcc) = fb {
471                debug!("Handle TWCC: {:?}", twcc);
472                let range = self.twcc_tx_register.apply_report(twcc, now);
473
474                if let Some(bwe) = &mut self.bwe {
475                    let records = range.and_then(|range| self.twcc_tx_register.send_records(range));
476
477                    if let Some(records) = records {
478                        bwe.update(records, now);
479                    }
480                }
481                need_configure_pacer = true;
482
483                // The funky thing about TWCC reports is that they are never stapled
484                // together with other RTCP packet. If they were though, we want to
485                // handle more packets.
486                continue;
487            }
488
489            if fb.is_for_rx() {
490                let Some(stream) = self.streams.stream_rx(&fb.ssrc()) else {
491                    continue;
492                };
493                stream.handle_rtcp(now, fb);
494            } else {
495                let Some(stream) = self.streams.stream_tx(&fb.ssrc()) else {
496                    continue;
497                };
498                stream.handle_rtcp(now, fb);
499            }
500        }
501
502        // Not in the above if due to lifetime issues, still okay because the method
503        // doesn't do anything when BWE isn't configured.
504        if need_configure_pacer {
505            self.configure_pacer();
506        }
507
508        Some(())
509    }

511    pub fn poll_event(&mut self) -> Option<Event> {
512        if let Some(bitrate_estimate) = self.bwe.as_mut().and_then(|bwe| bwe.poll_estimate()) {
513            return Some(Event::EgressBitrateEstimate(BweKind::Twcc(
514                bitrate_estimate,
515            )));
516        }
517
518        // If we're not ready to flow media, don't send any events.
519        if !self.ready_for_srtp() {
520            return None;
521        }
522
523        if let Some(raw_packets) = &mut self.raw_packets {
524            if let Some(p) = raw_packets.pop_front() {
525                return Some(Event::RawPacket(p));
526            }
527        }
528
529        // This must be before pending_packet.take() since we need to emit the unpaused event
530        // before the first packet causing the unpause.
531        if let Some(paused) = self.streams.poll_stream_paused() {
532            return Some(Event::StreamPaused(paused));
533        }
534
535        if self.rtp_mode {
536            if let Some(packet) = self.pending_packet.take() {
537                return Some(Event::RtpPacket(packet));
538            }
539        }
540
541        if let Some(req) = self.streams.poll_keyframe_request() {
542            return Some(Event::KeyframeRequest(req));
543        }
544
545        if let Some((mid, bitrate)) = self.streams.poll_remb_request() {
546            return Some(Event::EgressBitrateEstimate(BweKind::Remb(mid, bitrate)));
547        }
548
549        for media in &mut self.medias {
550            if media.need_open_event {
551                media.need_open_event = false;
552
553                return Some(Event::MediaAdded(MediaAdded {
554                    mid: media.mid(),
555                    kind: media.kind(),
556                    direction: media.direction(),
557                    simulcast: media.simulcast().map(|s| s.clone().into()),
558                }));
559            }
560
561            if media.need_changed_event {
562                media.need_changed_event = false;
563                return Some(Event::MediaChanged(MediaChanged {
564                    mid: media.mid(),
565                    direction: media.direction(),
566                }));
567            }
568        }
569
570        None
571    }

574        // Not relevant in rtp_mode, where the packets are picked up by poll_event().
575        if self.rtp_mode {

592    pub fn poll_datagram(&mut self, now: Instant) -> Option<net::DatagramSend> {
593        // Time must have progressed forward from start value.
594        if now == already_happened() {
595            return None;
596        }
597
598        let x = None
599            .or_else(|| self.poll_feedback())
600            .or_else(|| self.poll_packet(now));
601
602        if let Some(x) = &x {
603            // In RTP mode we trust the API user feeds the RTP packet sizes they
604            // need for the MTU they are targeting. This warning is only for when
605            // str0m does the RTP packetization.
606            if !self.rtp_mode && x.len() > DATAGRAM_MTU_WARN {
607                warn!("RTP above MTU {}: {}", DATAGRAM_MTU_WARN, x.len());
608            }
609        }
610
611        x
612    }

614    fn poll_feedback(&mut self) -> Option<net::DatagramSend> {
615        if self.feedback_tx.is_empty() {
616            return None;
617        }
618
619        // Round to nearest multiple of 4 bytes.
620        const ENCRYPTABLE_MTU: usize = (DATAGRAM_MTU - SRTCP_OVERHEAD) & !3;
621        assert!(ENCRYPTABLE_MTU % 4 == 0);
622
623        let mut data = vec![0_u8; ENCRYPTABLE_MTU];
624
625        let mut raw_packets = self.raw_packets.as_mut();
626        let output = move |fb| {
627            if let Some(raw_packets) = &mut raw_packets {
628                raw_packets.push_back(Box::new(RawPacket::RtcpTx(fb)));
629            }
630        };
631
632        let len = Rtcp::write_packet(&mut self.feedback_tx, &mut data, output);
633
634        if len == 0 {
635            return None;
636        }
637
638        data.truncate(len);
639
640        let srtp = self.srtp_tx.as_mut()?;
641        let protected = srtp.protect_rtcp(&data);
642
643        assert!(
644            protected.len() < DATAGRAM_MTU,
645            "Encrypted SRTCP should be less than MTU"
646        );
647
648        Some(protected.into())
649    }

651    fn poll_packet(&mut self, now: Instant) -> Option<DatagramSend> {
652        let srtp_tx = self.srtp_tx.as_mut()?;
653
654        // Figure out which, if any, queue to poll
655        let mid = self.pacer.poll_queue()?;
656        let media = self
657            .medias
658            .iter()
659            .find(|m| m.mid() == mid)
660            .expect("index is media");
661
662        let buf = &mut self.poll_packet_buf;
663        let twcc_seq = self.twcc;
664
665        // TODO: allow for sending simulcast
666        let stream = self.streams.stream_tx_by_mid_rid(media.mid(), None)?;
667
668        let params = &self.codec_config;
669        let exts = media.remote_extmap();
670        let receipt = stream.poll_packet(now, exts, &mut self.twcc, params, buf)?;
671
672        let PacketReceipt {
673            header,
674            seq_no,
675            is_padding,
676            payload_size,
677        } = receipt;
678
679        trace!(payload_size, is_padding, "Poll RTP: {:?}", header);
680
681        #[cfg(feature = "_internal_dont_use_log_stats")]
682        {
683            let kind = if is_padding { "padding" } else { "media" };
684
685            crate::log_stat!("PACKET_SENT", header.ssrc, payload_size, kind);
686        }
687
688        self.pacer.register_send(now, payload_size.into(), mid);
689
690        if let Some(raw_packets) = &mut self.raw_packets {
691            raw_packets.push_back(Box::new(RawPacket::RtpTx(header.clone(), buf.clone())));
692        }
693
694        let protected = srtp_tx.protect_rtp(buf, &header, *seq_no);
695
696        self.twcc_tx_register
697            .register_seq(twcc_seq.into(), now, payload_size);
698
699        // Technically we should wait for the next handle_timeout, but this speeds things up a bit
700        // avoiding an extra poll_timeout.
701        self.update_queue_state(now);
702
703        Some(protected.into())
704    }

706    pub fn poll_timeout(&mut self) -> Option<Instant> {
707        let regular_at = Some(self.regular_feedback_at());
708        let nack_at = Some(self.nack_at());
709        let twcc_at = self.twcc_at();
710        let pacing_at = self.pacer.poll_timeout();
711        let packetize_at = self.medias.iter().flat_map(|m| m.poll_timeout()).next();
712        let bwe_at = self.bwe.as_ref().map(|bwe| bwe.poll_timeout());
713        let paused_at = Some(self.paused_at());
714        let timestamp_writes_at = self.streams.timestamp_writes_at();
715
716        let timeout = (regular_at, "regular")
717            .soonest((nack_at, "nack"))
718            .soonest((twcc_at, "twcc"))
719            .soonest((pacing_at, "pacing"))
720            .soonest((packetize_at, "media"))
721            .soonest((bwe_at, "bwe"))
722            .soonest((paused_at, "paused"))
723            .soonest((timestamp_writes_at, "timestamp writes"));
724
725        // trace!("poll_timeout soonest is: {}", timeout.1);
726
727        timeout.0
728    }

768    pub fn visit_stats(&mut self, now: Instant, snapshot: &mut StatsSnapshot) {
769        for stream in self.streams.streams_tx() {
770            stream.visit_stats(snapshot, now);
771        }
772
773        for stream in self.streams.streams_rx() {
774            stream.visit_stats(snapshot, now);
775        }
776
777        snapshot.tx = snapshot.egress.values().map(|s| s.bytes).sum();
778        snapshot.rx = snapshot.ingress.values().map(|s| s.bytes).sum();
779        snapshot.bwe_tx = self.bwe.as_ref().and_then(|bwe| bwe.last_estimate());
780
781        snapshot.egress_loss_fraction = self.twcc_tx_register.loss(Duration::from_secs(1), now);
782        snapshot.ingress_loss_fraction = self.twcc_rx_register.loss();
783    }

816    fn configure_pacer(&mut self) {
817        let Some(bwe) = self.bwe.as_ref() else {
818            return;
819        };
820
821        let padding_rate = bwe
822            .last_estimate()
823            .map(|estimate| estimate.min(bwe.desired_bitrate))
824            .unwrap_or(Bitrate::ZERO);
825
826        self.pacer.set_padding_rate(padding_rate);
827
828        // We pad up to the pacing rate, therefore we need to increase pacing if the estimate, and
829        // thus the padding rate, exceeds the current bitrate adjusted with the pacing factor.
830        // Otherwise we can have a case where the current bitrate is 250Kbit/s resulting in a
831        // pacing rate of 275KBit/s which means we'll only ever pad about 25Kbit/s. If the estimate
832        // is actually 600Kbit/s we need to use that for the pacing rate to ensure we send as much as
833        // we think the link capacity can sustain, if not the estimate is a lie.
834        let pacing_rate = (bwe.current_bitrate * PACING_FACTOR).max(padding_rate);
835        self.pacer.set_pacing_rate(pacing_rate);
836    }

854    pub fn set_direction(&mut self, mid: Mid, direction: Direction) -> bool {
855        let Some(media) = self.media_by_mid_mut(mid) else {
856            return false;
857        };
858        let old_dir = media.direction();
859        if old_dir == direction {
860            return false;
861        }
862
863        media.set_direction(direction);
864
865        if old_dir.is_sending() && !direction.is_sending() {
866            self.streams.reset_buffers_tx(mid);
867        }
868
869        if old_dir.is_receiving() && !direction.is_receiving() {
870            self.streams.reset_buffers_rx(mid);
871        }
872
873        true
874    }

909    fn poll_estimate(&mut self) -> Option<Bitrate> {
910        let estimate = self.bwe.last_estimate()?;
911
912        let min = self.last_emitted_estimate * (1.0 - ESTIMATE_TOLERANCE);
913        let max = self.last_emitted_estimate * (1.0 + ESTIMATE_TOLERANCE);
914
915        if estimate < min || estimate > max {
916            self.last_emitted_estimate = estimate;
917            Some(estimate)
918        } else {
919            // Estimate is within tolerances.
920            None
921        }
922    }

925        self.bwe.poll_timeout()
926    }
