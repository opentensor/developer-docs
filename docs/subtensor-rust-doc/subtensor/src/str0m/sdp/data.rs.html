54    pub(crate) fn ice_candidates(&self) -> impl Iterator<Item = &Candidate> {
55        let mut candidates: HashSet<&Candidate> = HashSet::new();

74    fn do_assert_consistency(&self) -> Option<String> {
75        // TODO: SDP assertions we need to make:
76        // 1. Ensure that every m-line has the same PT configuration for a codec. I.e. if FIR is enabled
77        //    for PT 96 in one m-line it must be in all m-lines.
78        // 2. Compare with previous m-lines that remote isn't narrowing/expanding PT and/or extmaps.
79
80        let group = self
81            .session
82            .attrs
83            .iter()
84            .find(|a| matches!(a, SessionAttribute::Group { .. }));
85
86        if let Some(SessionAttribute::Group { mids, .. }) = group {
87            if !mids.len() == self.media_lines.len() {
88                return Some(format!(
89                    "a=group mid count doesn't match m-line count {} != {}",
90                    mids.len(),
91                    self.media_lines.len()
92                ));
93            }
94            for (media, mid) in self.media_lines.iter().zip(mids.iter()) {
95                media.check_consistent()?;
96                let m = media.mid();
97                if m != *mid {
98                    return Some(format!("Mid order not matching a=group {m} != {mid}"));
99                }
100            }
101        } else {
102            return Some("Session attribute a=group missing".into());
103        }
104
105        None
106    }

269    pub fn rtp_params(&self) -> Vec<PayloadParams> {
270        let rtp_maps: Vec<_> = self
271            .attrs
272            .iter()
273            .filter_map(|a| {
274                if let MediaAttribute::RtpMap { pt, value: c } = a {
275                    Some((*pt, *c))
276                } else {
277                    None
278                }
279            })
280            .collect();
281
282        let fmtps: Vec<_> = self
283            .attrs
284            .iter()
285            .filter_map(|a| {
286                if let MediaAttribute::Fmtp { pt, values } = a {
287                    Some((pt, values))
288                } else {
289                    None
290                }
291            })
292            .collect();
293
294        let fbs: Vec<_> = self
295            .attrs
296            .iter()
297            .filter_map(|a| {
298                if let MediaAttribute::RtcpFb { pt, value } = a {
299                    Some((pt, value))
300                } else {
301                    None
302                }
303            })
304            .collect();
305
306        let mut params: Vec<_> = rtp_maps
307            .iter()
308            .filter(|(_, c)| c.codec.is_audio() | c.codec.is_video())
309            .map(|(pt, c)| PayloadParams::new(*pt, None, (*c).into()))
310            .collect();
311
312        for p in &mut params {
313            for (pt, values) in fmtps.iter() {
314                // find matching a=fmtp line, if it exists.
315                if **pt == p.pt {
316                    for param in values.iter() {
317                        p.spec.format.set_param(param);
318                    }
319                }
320
321                // find resend pt, if there is one.
322                for fp in values.iter() {
323                    if let FormatParam::Apt(v) = fp {
324                        if *v == p.pt {
325                            // ensure this is a rtx
326                            let is_rtx = rtp_maps
327                                .iter()
328                                .any(|(cpt, c)| cpt == *pt && c.codec == Codec::Rtx);
329                            if is_rtx {
330                                p.resend = Some(**pt);
331                            }
332                        }
333                    }
334                }
335            }
336
337            // rtcp feedback mechanisms
338            for (pt, value) in fbs.iter() {
339                if **pt == p.pt {
340                    match &value[..] {
341                        "goog-remb" => {
342                            p.fb_remb = true;
343                        }
344                        "transport-cc" => {
345                            p.fb_transport_cc = true;
346                        }
347                        "ccm fir" => {
348                            p.fb_fir = true;
349                        }
350                        "nack" => {
351                            p.fb_nack = true;
352                        }
353                        "nack pli" => {
354                            p.fb_pli = true;
355                        }
356                        _ => {
357                            //
358                        }
359                    }
360                }
361            }
362        }
363
364        params
365    }

367    pub fn check_consistent(&self) -> Option<String> {
368        use MediaAttribute::*;
369
370        let mid_count = self.attrs.iter().filter(|a| matches!(a, Mid(_))).count();
371
372        if mid_count == 0 {
373            return Some(format!(
374                "Media is missing a=mid: {} {}",
375                self.typ, self.proto
376            ));
377        }
378
379        if mid_count > 1 {
380            return Some(format!(
381                "Media has more than one a=mid: {} {}",
382                self.typ, self.proto
383            ));
384        }
385
386        let setup = self.attrs.iter().filter(|a| matches!(a, Setup(_))).count();
387
388        if setup > 1 {
389            return Some(format!(
390                "Expected 0 or 1 a=setup: line for mid: {}",
391                self.mid()
392            ));
393        }
394
395        let dir_count = self.attrs.iter().filter(|a| is_dir(a)).count();
396
397        if self.proto == Proto::Srtp && dir_count != 1 {
398            return Some(format!(
399                "Expected exactly one of a=sendrecv, a=sendonly, a=recvonly, a=inactive for mid: {}",
400                self.mid()
401            ));
402        }
403
404        if self.proto == Proto::Srtp && self.pts.is_empty() {
405            return Some(format!("Expected at least one PT for mid: {}", self.mid()));
406        }
407
408        for m in &self.pts {
409            let rtp_count = self
410                .attrs
411                .iter()
412                .filter(|a| {
413                    if let MediaAttribute::RtpMap { pt, value: _ } = a {
414                        pt == m
415                    } else {
416                        false
417                    }
418                })
419                .count();
420            if rtp_count == 0 {
421                return Some(format!("Missing a=rtp_map:{} for mid: {}", m, self.mid()));
422            }
423            if rtp_count > 1 {
424                return Some(format!(
425                    "More than one a=rtp_map:{} for mid: {}",
426                    m,
427                    self.mid()
428                ));
429            }
430        }
431        None
432    }

548    pub fn ssrc_info(&self) -> Vec<SsrcInfo> {
549        let mut v = vec![];
550
551        fn by_ssrc(v: &mut Vec<SsrcInfo>, ssrc: Ssrc) -> &mut SsrcInfo {
552            if let Some(pos) = v.iter().position(|i| i.ssrc == ssrc) {
553                &mut v[pos]
554            } else {
555                v.push(SsrcInfo {
556                    ssrc,
557                    ..Default::default()
558                });
559                v.last_mut().unwrap()
560            }
561        }
562
563        for a in &self.attrs {
564            match a {
565                MediaAttribute::Ssrc { ssrc, attr, value } => {
566                    let info = by_ssrc(&mut v, *ssrc);
567
568                    // a=ssrc:2147603131 cname:TbS1Ajv9obq6/63I
569                    // a=ssrc:2147603131 msid:- 7a08dda6-518f-4027-b707-410a6d414176
570                    match attr.to_lowercase().as_str() {
571                        "cname" => info.cname = Some(value.clone()),
572                        "msid" => {
573                            let mut iter = value.split(' ');
574
575                            fn trim_and_no_minus(s: &str) -> Option<String> {
576                                let s = s.trim();
577
578                                if s == "-" {
579                                    None
580                                } else {
581                                    Some(s.into())
582                                }
583                            }
584
585                            if let Some(stream_id) = iter.next() {
586                                info.stream_id = trim_and_no_minus(stream_id);
587                            }
588                            if let Some(track_id) = iter.next() {
589                                info.track_id = trim_and_no_minus(track_id);
590                            }
591                        }
592                        _ => {}
593                    }
594                }
595                _ => {}
596            }
597        }
598        // Match this second to ensure we preserve order of a=ssrc.
599        for a in &self.attrs {
600            match a {
601                MediaAttribute::SsrcGroup { semantics, ssrcs } => {
602                    if semantics.to_lowercase() != "fid" {
603                        continue;
604                    }
605
606                    // a=ssrc-group:FID 659652645 98148385
607                    // Should be two SSRC after FID.
608                    if ssrcs.len() != 2 {
609                        continue;
610                    }
611
612                    let info = by_ssrc(&mut v, ssrcs[1]);
613                    info.repairs = Some(ssrcs[0]);
614                }
615                _ => {}
616            }
617        }
618
619        v
620    }
621}

811        ext: Extension,
812    },

816    Inactive, // a=inactive
817    // a=msid:5UUdwiuY7OML2EkQtF38pJtNP5v7In1LhjEK f78dde68-7055-4e20-bb37-433803dd1ed1

844        id: RestrictionId,       //
845        direction: &'static str, // send or recv

848        restriction: Vec<(String, String)>,
849    },

851    // a=rid:lo send
852    // a=simulcast:send hi;lo

875#[derive(Debug, Clone, Copy, PartialEq, Eq)]
876pub enum FormatParam {
877    /// The minimum duration of media represented by a packet.
878    ///
879    /// Default 3. Max 120.
880    MinPTime(u8),
881
882    /// Specifies that the decoder can do Opus in-band FEC
883    UseInbandFec(bool),
884
885    /// Specifies that the decoder can do Opus DTX
886    UseDtx(bool),
887
888    /// Whether h264 sending media encoded at a different level in the offerer-to-answerer
889    /// direction than the level in the answerer-to-offerer direction, is allowed.
890    LevelAsymmetryAllowed(bool),
891
892    /// What h264 packetization mode is used.
893    ///
894    /// * 0 - single nal.
895    /// * 1 - STAP-A, FU-A is allowed. Non-interleaved.
896    PacketizationMode(u8),
897
898    /// H264 profile level.
899    ///
900    /// * 42 00 1f - 4200=baseline (B)              1f=level 3.1
901    /// * 42 e0 1f - 42e0=constrained baseline (CB) 1f=level 3.1
902    /// * 4d 00 1f - 4d00=main (M)                  1f=level 3.1
903    /// * 64 00 1f - 6400=high (H)                  1f=level 3.1
904    ProfileLevelId(u32),
905
906    /// VP9 profile id
907    ProfileId(u32),
908
909    /// RTX (resend) codecs, which PT it concerns.
910    Apt(Pt),
911
912    /// Unrecognized fmtp.
913    Unknown,
914}
915
916impl FormatParam {
917    pub fn parse(k: &str, v: &str) -> Self {
918        use FormatParam::*;
919        match k {
920            "minptime" => {
921                if let Ok(v) = v.parse() {
922                    MinPTime(v)
923                } else {
924                    trace!("Failed to parse: {}", k);
925                    Unknown
926                }
927            }
928            "useinbandfec" => UseInbandFec(v == "1"),
929            "usedtx" => UseDtx(v == "1"),
930            "level-asymmetry-allowed" => LevelAsymmetryAllowed(v == "1"),
931            "packetization-mode" => {
932                if let Ok(v) = v.parse() {
933                    PacketizationMode(v)
934                } else {
935                    trace!("Failed to parse: {}", k);
936                    Unknown
937                }
938            }
939            "profile-level-id" => {
940                if let Ok(v) = u32::from_str_radix(v, 16).or_else(|_| v.parse()) {
941                    ProfileLevelId(v)
942                } else {
943                    trace!("Failed to parse: {}", k);
944                    Unknown
945                }
946            }
947            "profile-id" => {
948                if let Ok(v) = v.parse() {
949                    ProfileId(v)
950                } else {
951                    trace!("Failed to parse: {}", k);
952                    Unknown
953                }
954            }
955            "apt" => {
956                if let Ok(v) = v.parse::<u8>() {
957                    Apt(v.into())
958                } else {
959                    trace!("Failed to parse: {}", k);
960                    Unknown
961                }
962            }
963            _ => Unknown,
964        }
965    }
966}

987impl PayloadParams {
988    pub(crate) fn as_media_attrs(&self, attrs: &mut Vec<MediaAttribute>) {
989        attrs.push(MediaAttribute::RtpMap {
990            pt: self.pt,
991            value: self.spec.into(),
992        });
993
994        if self.fb_transport_cc {
995            attrs.push(MediaAttribute::RtcpFb {
996                pt: self.pt,
997                value: "transport-cc".into(),
998            });
999        }
1000        if self.fb_remb {
1001            attrs.push(MediaAttribute::RtcpFb {
1002                pt: self.pt,
1003                value: "goog-remb".into(),
1004            });
1005        }
1006        if self.fb_fir {
1007            attrs.push(MediaAttribute::RtcpFb {
1008                pt: self.pt,
1009                value: "ccm fir".into(),
1010            });
1011        }
1012        if self.fb_nack {
1013            attrs.push(MediaAttribute::RtcpFb {
1014                pt: self.pt,
1015                value: "nack".into(),
1016            });
1017        }
1018        if self.fb_pli {
1019            attrs.push(MediaAttribute::RtcpFb {
1020                pt: self.pt,
1021                value: "nack pli".into(),
1022            });
1023        }
1024
1025        let fmtps = self.spec.format.to_format_param();
1026        if !fmtps.is_empty() {
1027            attrs.push(MediaAttribute::Fmtp {
1028                pt: self.pt,
1029                values: fmtps,
1030            });
1031        }
1032
1033        if let Some(pt) = self.resend {
1034            attrs.push(MediaAttribute::RtpMap {
1035                pt,
1036                value: RtpMap {
1037                    codec: Codec::Rtx,
1038                    clock_rate: self.spec.clock_rate,
1039                    channels: None,
1040                },
1041            });
1042            attrs.push(MediaAttribute::Fmtp {
1043                pt,
1044                values: vec![FormatParam::Apt(self.pt)],
1045            });
1046        }
1047    }

1276            }

1278            SendRecv => write!(f, "a=sendrecv\r\n")?,
1279            SendOnly => write!(f, "a=sendonly\r\n")?,

1281            // a=msid:5UUdwiuY7OML2EkQtF38pJtNP5v7In1LhjEK f78dde68-7055-4e20-bb37-433803dd1ed1
1282            // a=msid:- 78dde68-7055-4e20-bb37-433803dd1ed1
1283            Msid(v) => write!(f, "a=msid:{} {}\r\n", v.stream_id, v.track_id)?,
1284            RtcpMux => write!(f, "a=rtcp-mux\r\n")?,
1285            RtcpMuxOnly => write!(f, "a=rtcp-mux-only\r\n")?,
1286            RtcpRsize => write!(f, "a=rtcp-rsize\r\n")?,
1287            Candidate(c) => write!(f, "a={}\r\n", c.to_sdp_string())?,
1288            EndOfCandidates => write!(f, "a=end-of-candidates\r\n")?,
1289            RtpMap { pt, value: c } => {
1290                write!(f, "a=rtpmap:{} {}/{}", pt, c.codec, c.clock_rate)?;
1291                if let Some(e) = c.channels {
1292                    write!(f, "/{e}")?;
1293                }
1294                write!(f, "\r\n")?;
1295            }
1296            RtcpFb { pt, value } => write!(f, "a=rtcp-fb:{pt} {value}\r\n")?,
1297            Fmtp { pt, values } => {
1298                write!(f, "a=fmtp:{pt} ")?;
1299                for (idx, v) in values.iter().enumerate() {
1300                    if idx + 1 < values.len() {
1301                        write!(f, "{v};")?;
1302                    } else {
1303                        write!(f, "{v}\r\n")?;
1304                    }
1305                }
1306            }
1307            // a=rid:hi send pt=111,112;max-br=64000;max-height=360
1308            Rid {
1309                id,
1310                direction,
1311                pt,
1312                restriction,
1313            } => {
1314                write!(f, "a=rid:{} {}", id.0, direction)?;
1315                for (idx, p) in pt.iter().enumerate() {
1316                    if idx == 0 {
1317                        write!(f, " pt=")?;
1318                    }
1319                    if idx + 1 == pt.len() {
1320                        write!(f, "{p}")?;
1321                    } else {
1322                        write!(f, "{p},")?;
1323                    }
1324                }
1325                for (idx, (k, v)) in restriction.iter().enumerate() {
1326                    if idx == 0 {
1327                        if pt.is_empty() {
1328                            write!(f, " ")?;
1329                        } else {
1330                            write!(f, ";")?;
1331                        }
1332                    }
1333                    if idx + 1 == restriction.len() {
1334                        write!(f, "{k}={v}")?;
1335                    } else {
1336                        write!(f, "{k}={v};")?;
1337                    }
1338                }
1339                write!(f, "\r\n")?;
1340            }
1341            // a=simulcast:<send/recv> <alt A>;<alt B>,<or C> <send/recv> [same]
1342            Simulcast(x) => {
1343                let self::Simulcast {
1344                    send,
1345                    recv,
1346                    is_munged,
1347                } = x;
1348                assert!(
1349                    !(send.0.is_empty() && recv.0.is_empty()),
1350                    "Empty a=simulcast"
1351                );
1352                if *is_munged {
1353                    // don't write
1354                    return Ok(());
1355                }
1356                write!(f, "a=simulcast:")?;
1357                if !send.0.is_empty() {
1358                    write!(f, "send {send}")?;
1359                }
1360                if !recv.0.is_empty() {
1361                    if !send.0.is_empty() {
1362                        write!(f, " ")?;
1363                    }
1364                    write!(f, "recv {recv}")?;
1365                }
1366                write!(f, "\r\n")?;
1367            }
1368            SsrcGroup { semantics, ssrcs } => {
1369                write!(f, "a=ssrc-group:{semantics} ")?;
1370                for (idx, ssrc) in ssrcs.iter().enumerate() {
1371                    if idx + 1 < ssrcs.len() {
1372                        write!(f, "{} ", **ssrc)?;
1373                    } else {
1374                        write!(f, "{}\r\n", **ssrc)?;
1375                    }
1376                }
1377            }
1378            Ssrc { ssrc, attr, value } => {
1379                write!(f, "a=ssrc:{} {}:{}\r\n", **ssrc, attr, value)?;
1380            }
1381            Unused(v) => write!(f, "a={v}\r\n")?,
1382        }
1383        Ok(())
1384    }
1385}
