2//!
3//! A synchronous sans I/O WebRTC implementation in Rust.
4//!
5//! This is a [Sans I/O][sansio] implementation meaning the `Rtc` instance itself is not doing any network
6//! talking. Furthermore it has no internal threads or async tasks. All operations are synchronously
7//! happening from the calls of the public API.
8//!
9//! This is deliberately not a standard `RTCPeerConnection` API since that isn't a great fit for Rust.
10//! See more details in below section.

19//! # Usage
20//!
21//! The [`chat`][x-chat] example shows how to connect multiple browsers
22//! together and act as an SFU (Selective Forwarding Unit). The example
23//! multiplexes all traffic over one server UDP socket and uses two threads
24//! (one for the web server, and one for the SFU loop).
25//!
26//! ## TLS
27//!
28//! For the browser to do WebRTC, all traffic must be under TLS. The
29//! project ships with a self-signed certificate that is used for the
30//! examples. The certificate is for hostname `str0m.test` since TLD .test
31//! should never resolve to a real DNS name.
32//!
33//! ```text
34//! cargo run --example chat
35//! ```
36//!
37//! The log should prompt you to connect a browser to https://10.0.0.103:3000 â€“ this will
38//! most likely cause a security warning that you must get the browser to accept.
39//!
40//! The [`http-post`][x-post] example roughly illustrates how to receive
41//! media data from a browser client. The example is single threaded and
42//! is a bit simpler than the chat. It is a good starting point to understand the API.
43//!
44//! ```text
45//! cargo run --example http-post

48//! ## Passive
49//!
50//! For passive connections, i.e. where the media and initial OFFER is
51//! made by a remote peer, we need these steps to open the connection.
52//!
53//! ```no_run
54//! # use str0m::{Rtc, Candidate};
55//! // Instantiate a new Rtc instance.
56//! let mut rtc = Rtc::new();
57//!
58//! //  Add some ICE candidate such as a locally bound UDP port.
59//! let addr = "1.2.3.4:5000".parse().unwrap();
60//! let candidate = Candidate::host(addr, "udp").unwrap();
61//! rtc.add_local_candidate(candidate);
62//!
63//! // Accept an incoming offer from the remote peer
64//! // and get the corresponding answer.
65//! let offer = todo!();
66//! let answer = rtc.sdp_api().accept_offer(offer).unwrap();
67//!
68//! // Forward the answer to the remote peer.
69//!
70//! // Go to _run loop_
71//! ```
72//!
73//! ## Active
74//!
75//! Active connections means we are making the inital OFFER and waiting for a
76//! remote ANSWER to start the connection.
77//!
78//! ```no_run
79//! # use str0m::{Rtc, Candidate};
80//! # use str0m::media::{MediaKind, Direction};
81//! // Instantiate a new Rtc instance.
82//! let mut rtc = Rtc::new();
83//!
84//! // Add some ICE candidate such as a locally bound UDP port.
85//! let addr = "1.2.3.4:5000".parse().unwrap();
86//! let candidate = Candidate::host(addr, "udp").unwrap();
87//! rtc.add_local_candidate(candidate);
88//!
89//! // Create a `SdpApi`. The change lets us make multiple changes
90//! // before sending the offer.
91//! let mut change = rtc.sdp_api();
92//!
93//! // Do some change. A valid OFFER needs at least one "m-line" (media).
94//! let mid = change.add_media(MediaKind::Audio, Direction::SendRecv, None, None);
95//!
96//! // Get the offer.
97//! let (offer, pending) = change.apply().unwrap();
98//!
99//! // Forward the offer to the remote peer and await the answer.
100//! // How to transfer this is outside the scope for this library.
101//! let answer = todo!();
102//!
103//! // Apply answer.
104//! rtc.sdp_api().accept_answer(pending, answer).unwrap();
105//!
106//! // Go to _run loop_
107//! ```

109//! ## Run loop
110//!
111//! Driving the state of the `Rtc` forward is a run loop that, regardless of sync or async,
112//! looks like this.
113//!
114//! ```no_run
115//! # use str0m::{Rtc, Output, IceConnectionState, Event, Input};
116//! # use str0m::net::{Receive, Protocol};
117//! # use std::io::ErrorKind;
118//! # use std::net::UdpSocket;
119//! # use std::time::Instant;
120//! # let rtc = Rtc::new();
121//! // Buffer for reading incoming UDP packets.
122//! let mut buf = vec![0; 2000];
123//!
124//! // A UdpSocket we obtained _somehow_.
125//! let socket: UdpSocket = todo!();
126//!
127//! loop {
128//!     // Poll output until we get a timeout. The timeout means we
129//!     // are either awaiting UDP socket input or the timeout to happen.
130//!     let timeout = match rtc.poll_output().unwrap() {
131//!         // Stop polling when we get the timeout.
132//!         Output::Timeout(v) => v,
133//!
134//!         // Transmit this data to the remote peer. Typically via
135//!         // a UDP socket. The destination IP comes from the ICE
136//!         // agent. It might change during the session.
137//!         Output::Transmit(v) => {
138//!             socket.send_to(&v.contents, v.destination).unwrap();
139//!             continue;
140//!         }
141//!
142//!         // Events are mainly incoming media data from the remote
143//!         // peer, but also data channel data and statistics.
144//!         Output::Event(v) => {
145//!
146//!             // Abort if we disconnect.
147//!             if v == Event::IceConnectionStateChange(IceConnectionState::Disconnected) {
148//!                 return;
149//!             }
150//!
151//!             // TODO: handle more cases of v here, such as incoming media data.
152//!
153//!             continue;
154//!         }
155//!     };
156//!
157//!     // Duration until timeout.
158//!     let duration = timeout - Instant::now();
159//!
160//!     // socket.set_read_timeout(Some(0)) is not ok
161//!     if duration.is_zero() {
162//!         // Drive time forwards in rtc straight away.
163//!         rtc.handle_input(Input::Timeout(Instant::now())).unwrap();
164//!         continue;
165//!     }
166//!
167//!     socket.set_read_timeout(Some(duration)).unwrap();
168//!
169//!     // Scale up buffer to receive an entire UDP packet.
170//!     buf.resize(2000, 0);
171//!
172//!     // Try to receive. Because we have a timeout on the socket,
173//!     // we will either receive a packet, or timeout.
174//!     // This is where having an async loop shines. We can await multiple things to
175//!     // happen such as outgoing media data, the timeout and incoming network traffic.
176//!     // When using async there is no need to set timeout on the socket.
177//!     let input = match socket.recv_from(&mut buf) {
178//!         Ok((n, source)) => {
179//!             // UDP data received.
180//!             buf.truncate(n);
181//!             Input::Receive(
182//!                 Instant::now(),
183//!                 Receive {
184//!                     proto: Protocol::Udp,
185//!                     source,
186//!                     destination: socket.local_addr().unwrap(),
187//!                     contents: buf.as_slice().try_into().unwrap(),
188//!                 },
189//!             )
190//!         }
191//!
192//!         Err(e) => match e.kind() {
193//!             // Expected error for set_read_timeout().
194//!             // One for windows, one for the rest.
195//!             ErrorKind::WouldBlock
196//!                 | ErrorKind::TimedOut => Input::Timeout(Instant::now()),
197//!
198//!             e => {
199//!                 eprintln!("Error: {:?}", e);
200//!                 return; // abort
201//!             }
202//!         },
203//!     };
204//!
205//!     // Input is either a Timeout or Receive of data. Both drive the state forward.
206//!     rtc.handle_input(input).unwrap();
207//! }
208//! ```

210//! ## Sending media data
211//!
212//! When creating the media, we can decide which codecs to support, and they
213//! are negotiated with the remote side. Each codec corresponds to a
214//! "payload type" (PT). To send media data we need to figure out which PT
215//! to use when sending.
216//!
217//! ```no_run
218//! # use str0m::Rtc;
219//! # use str0m::media::Mid;
220//! # let rtc: Rtc = todo!();
221//! // Obtain mid from Event::MediaAdded
222//! let mid: Mid = todo!();
223//!
224//! // Create a media writer for the mid.
225//! let writer = rtc.writer(mid).unwrap();
226//!
227//! // Get the payload type (pt) for the wanted codec.
228//! let pt = writer.payload_params().nth(0).unwrap().pt();
229//!
230//! // Write the data
231//! let wallclock = todo!();   // Absolute time of the data
232//! let media_time = todo!();  // Media time, in RTP time
233//! let data: &[u8] = todo!(); // Actual data
234//! writer.write(pt, wallclock, media_time, data).unwrap();
235//! ```
236//!
237//! ## Media time, wallclock and local time
238//!
239//! str0m has three main concepts of time. "now", media time and wallclock.
240//!
241//! ### Now
242//!
243//! Some calls in str0m, such as `Rtc::handle_input` takes a `now` argument
244//! that is a `std::time::Instant`. These calls "drive the time forward" in
245//! the internal state. This is used for everything like deciding when
246//! to produce various feedback reports (RTCP) to remote peers, to
247//! bandwidth estimation (BWE) and statistics.
248//!
249//! Str0m has _no internal clock_ calls. I.e. str0m never calls
250//! `Instant::now()` itself. All time is external input. That means it's
251//! possible to construct test cases driving an `Rtc` instance faster
252//! than realtime (see the [integration tests][intg]).
253//!
254//! ### Media time
255//!
256//! Each RTP header has a 32 bit number that str0m calls _media time_.
257//! Media time is in some time base that is dependent on the codec,
258//! however all codecs in str0m use 90_000Hz for video and 48_000Hz
259//! for audio.
260//!
261//! For video the `MediaTime` type is `<timestamp>/90_000` str0m extends
262//! the 32 bit number in the RTP header to 64 bit taking into account
263//! "rollover". 64 bit is such a large number the user doesn't need to
264//! think about rollovers.
265//!
266//! ### Wallclock
267//!
268//! With _wallclock_ str0m means the time a sample of media was produced
269//! at an originating source. I.e. if we are talking into a microphone the
270//! wallclock is the NTP time the sound is sampled.
271//!
272//! We can't know the exact wallclock for media from a remote peer since
273//! not every device is synchronized with NTP. Every sender does
274//! periodically produce a Sender Report (SR) that contains the peer's
275//! idea of its wallclock, however this number can be very wrong compared to
276//! "real" NTP time.
277//!
278//! Furthermore, not all remote devices will have a linear idea of
279//! time passing that exactly matches the local time. A minute on the
280//! remote peer might not be exactly one minute locally.
281//!
282//! These timestamps become important when handling simultaneous audio from
283//! multiple peers.
284//!
285//! When writing media we need to provide str0m with an estimated wallclock.
286//! The simplest strategy is to only trust local time and use arrival time
287//! of the incoming UDP packet. Another simple strategy is to lock some
288//! time T at the first UDP packet, and then offset each wallclock using
289//! `MediaTime`, i.e. for video we could have `T + <media time>/90_000`
290//!
291//! A production worthy SFU probably needs an even more sophisticated
292//! strategy weighing in all possible time sources to get a good estimate
293//! of the remote wallclock for a packet.

311//! # Design
312//!
313//! Output from the `Rtc` instance can be grouped into three kinds.
314//!
315//! 1. Events (such as receiving media or data channel data).
316//! 2. Network output. Data to be sent, typically from a UDP socket.
317//! 3. Timeouts. Indicates when the instance next expects a time input.
318//!
319//! Input to the `Rtc` instance is:
320//!
321//! 1. User operations (such as sending media or data channel data).
322//! 2. Network input. Typically read from a UDP socket.
323//! 3. Timeouts. As obtained from the output above.
324//!
325//! The correct use can be seen in the above [Run loop](#run-loop) or in the
326//! examples.
327//!
328//! Sans I/O is a pattern where we turn both network input/output as well
329//! as time passing into external input to the API. This means str0m has
330//! no internal threads, just an enormous state machine that is driven
331//! forward by different kinds of input.
332//!
333//! ## Sample or RTP level?
334//!
335//! Str0m defaults to the "sample level" which treats the RTP as an internal detail. The user
336//! will thus mainly interact with:
337//!
338//! 1. [`Event::MediaData`][evmed] to receive full "samples" (audio frames or video frames).
339//! 2. [`Writer::write`][writer] to write full samples.
340//! 3. [`Writer::request_keyframe`][reqkey] to request keyframes.
341//!
342//! ### Sample level
343//!
344//! All codecs such as h264, vp8, vp9 and opus outputs what we call
345//! "Samples". A sample has a very specific meaning for audio, but this
346//! project uses it in a broader sense, where a sample is either a video
347//! or audio time stamped chunk of encoded data that typically represents
348//! a chunk of audio, or _one single frame for video_.
349//!
350//! Samples are not suitable to use directly in UDP (RTP) packets - for
351//! one they are too big. Samples are therefore further chunked up by
352//! codec specific payloaders into RTP packets.
353//!
354//! ### RTP mode
355//!
356//! Str0m also provides an RTP level API. This would be similar to many other
357//! RTP libraries where the RTP packets themselves are the the API surface
358//! towards the user (when building an SFU one would often talk about "forwarding
359//! RTP packets", while with str0m we can also "forward samples").  Using
360//! this API requires a deeper knowledge of RTP and WebRTC.
361//!
362//! To enable RTP mode
363//!
364//! ```
365//! # use str0m::Rtc;
366//! let rtc = Rtc::builder()
367//!     // Enable RTP mode for this Rtc instance.
368//!     // This disables `MediaEvent` and the `Writer::write` API.
369//!     .set_rtp_mode(true)
370//!     .build();
371//! ```
372//!
373//! RTP mode gives us some new API points.
374//!
375//! 1. [`Event::RtpPacket`][rtppak] emitted for every incoming RTP packet. Empty packets for bandwidth
376//!    estimation are silently discarded.
377//! 2. [`StreamTx::write_rtp`][wrtrtp] to write outgoing RTP packets.
378//! 3. [`StreamRx::request_keyframe`][reqkey2] to request keyframes from remote.
379//!
380//! ## NIC enumeration and TURN (and STUN)
381//!
382//! The [ICE RFC][ice] talks about "gathering ice candidates". This means
383//! inspecting the local network interfaces and potentially binding UDP
384//! sockets on each usable interface. Since str0m is Sans I/O, this part
385//! is outside the scope of what str0m does. How the user figures out
386//! local IP addresses, via config or via looking up local NICs is not
387//! something str0m cares about.
388//!
389//! TURN is a way of obtaining IP addresses that can be used as fallback
390//! in case direct connections fail. We consider TURN similar to
391//! enumerating local network interfaces â€“ it's a way of obtaining
392//! sockets.
393//!
394//! All discovered candidates, be they local (NIC) or remote sockets
395//! (TURN), are added to str0m and str0m will perform the task of ICE
396//! agent, forming "candidate pairs" and figuring out the best connection
397//! while the actual task of sending the network traffic is left to the
398//! user.
399//!
400//! ## The importance of `&mut self`
401//!
402//! Rust shines when we can eschew locks and heavily rely `&mut` for data
403//! write access. Since str0m has no internal threads, we never have to
404//! deal with shared data. Furthermore the the internals of the library is
405//! organized such that we don't need multiple references to the same
406//! entities. In str0m there are no `Rc`, `Mutex`, `mpsc`, `Arc`(*),  or
407//! other locks.
408//!
409//! This means all input to the lib can be modelled as
410//! `handle_something(&mut self, something)`.
411//!
412//! (*) Ok. There is one `Arc` if you use Windows where we also require openssl.
413//!
414//! ## Not a standard WebRTC "Peer Connection" API
415//!
416//! The library deliberately steps away from the "standard" WebRTC API as
417//! seen in JavaScript and/or [webrtc-rs][webrtc-rs] (or [Pion][pion] in Go).
418//! There are few reasons for this.
419//!
420//! First, in the standard API, events are callbacks, which are not a
421//! great fit for Rust. Callbacks require some kind of reference
422//! (ownership?) over the entity the callback is being dispatched
423//! upon. I.e. if in Rust we want `pc.addEventListener(x)`, `x` needs
424//! to be wholly owned by `pc`, or have some shared reference (like
425//! `Arc`). Shared references means shared data, and to get mutable shared
426//! data, we will need some kind of lock. i.e. `Arc<Mutex<EventListener>>`
427//! or similar.
428//!
429//! As an alternative we could turn all events into `mpsc` channels, but
430//! listening to multiple channels is awkward without async.
431//!
432//! Second, in the standard API, entities like `RTCPeerConnection` and
433//! `RTCRtpTransceiver`, are easily clonable and/or long lived
434//! references. I.e. `pc.getTranscievers()` returns objects that can be
435//! retained and owned by the caller. This pattern is fine for garbage
436//! collected or reference counted languages, but not great with Rust.
437//!
438//! ## Panics, Errors and unwraps
439//!
440//! Str0m adheres to [fail-fast][ff]. That means rather than brushing state
441//! bugs under the carpet, it panics. We make a distinction between errors and
442//! bugs.
443//!
444//! * Errors are as a result of incorrect or impossible to understand user input.
445//! * Bugs are broken internal invariants (assumptions).
446//!
447//! If you scan the str0m code you find a few `unwrap()` (or `expect()`). These
448//! will (should) always be accompanied by a code comment that explains why the
449//! unwrap is okay. This is an internal invariant, a state assumption that
450//! str0m is responsible for maintaining.
451//!
452//! We do not believe it's correct to change every `unwrap()`/`expect()` into
453//! `unwrap_or_else()`, `if let Some(x) = x { ... }` etc, because doing so
454//! brushes an actual problem (an incorrect assumption) under the carpet. Trying
455//! to hobble along with an incorrect state would at best result in broken
456//! behavior, at worst a security risk!
457//!
458//! Panics are our friends: *panic means bug*
459//!
460//! And also: str0m should *never* panic on any user input. If you encounter a panic,
461//! please report it!
462//!
463//! ### Catching panics
464//!
465//! Panics should be incredibly rare, or we have a serious problem as a project. For an SFU,
466//! it might not be ideal if str0m encounters a bug and brings the entire server down with it.
467//!
468//! For those who want an extra level of safety, we recommend looking at [`catch_unwind`][catch]
469//! to safely discard a faulty `Rtc` instance. Since `Rtc` has no internal threads, locks or async
470//! tasks, discarding the instance never risk poisoning locks or other issues that can happen
471//! when catching a panic.

805/// Instance that does WebRTC. Main struct of the entire library.
806///
807/// ## Usage
808///
809/// ```no_run
810/// # use str0m::{Rtc, Output, Input};
811/// let mut rtc = Rtc::new();
812///
813/// loop {
814///     let timeout = match rtc.poll_output().unwrap() {
815///         Output::Timeout(v) => v,
816///         Output::Transmit(t) => {
817///             // TODO: Send data to remote peer.
818///             continue; // poll again
819///         }
820///         Output::Event(e) => {
821///             // TODO: Handle event.
822///             continue; // poll again
823///         }
824///     };
825///
826///     // TODO: Wait for one of two events, reaching `timeout`
827///     //       or receiving network input. Both are encapsulated

858/// Events produced by [`Rtc::poll_output()`].
859#[derive(Debug)]
860#[non_exhaustive]
861#[rustfmt::skip]
862pub enum Event {
863    // =================== ICE related events ===================
864
865    /// Emitted when we got ICE connection and established DTLS.
866    Connected,
867
868    /// ICE connection state changes tells us whether the [`Rtc`] instance is
869    /// connected to the peer or not.
870    IceConnectionStateChange(IceConnectionState),
871
872    // =================== Media related events ==================
873
874    /// Upon adding new media to the session. The lines are emitted.
875    ///
876    /// Upon this event, the [`Media`] instance is available via [`Rtc::media()`].
877    MediaAdded(MediaAdded),
878
879    /// Incoming media data sent by the remote peer.
880    MediaData(MediaData),
881
882    /// Changes to the media may be emitted.
883    ///
884    ///. Currently only covers a change of direction.
885    MediaChanged(MediaChanged),
886
887    // =================== Data channel related events ===================
888
889    /// A data channel has opened.
890    ///
891    /// The string is the channel label which is set by the opening peer and can
892    /// be used to identify the purpose of the channel when there are more than one.
893    ///
894    /// The negotiation is to set up an SCTP association via DTLS. Subsequent data
895    /// channels reuse the same association.
896    ///
897    /// Upon this event, the [`Channel`] can be obtained via [`Rtc::channel()`].
898    ///
899    /// For [`SdpApi`]: The first ever data channel results in an SDP
900    /// negotiation, and this events comes at the end of that.
901    ChannelOpen(ChannelId, String),
902
903    /// Incoming data channel data from the remote peer.
904    ChannelData(ChannelData),
905
906    /// A data channel has been closed.
907    ChannelClose(ChannelId),
908
909    // =================== Statistics and BWE related events ===================
910
911    /// Statistics event for the Rtc instance
912    ///
913    /// Includes both media traffic (rtp payload) as well as all traffic
914    PeerStats(PeerStats),
915
916    /// Aggregated statistics for each media (mid, rid) in the ingress direction
917    MediaIngressStats(MediaIngressStats),
918
919    /// Aggregated statistics for each media (mid, rid) in the egress direction
920    MediaEgressStats(MediaEgressStats),
921
922    /// A new estimate from the bandwidth estimation subsystem.
923    EgressBitrateEstimate(BweKind),
924
925    // =================== RTP related events ===================
926
927    /// Incoming keyframe request for media that we are sending to the remote peer.
928    ///
929    /// The request is either PLI (Picture Loss Indication) or FIR (Full Intra Request).
930    KeyframeRequest(KeyframeRequest),
931
932    /// Whether an incoming encoded stream is paused.
933    ///
934    /// This means the stream has not received any data for some time (default 1.5 seconds).
935    StreamPaused(StreamPaused),
936
937    /// Incoming RTP data.
938    RtpPacket(RtpPacket),
939
940    /// Debug output of incoming and outgoing RTCP/RTP packets.
941    ///
942    /// Enable using [`RtcConfig::enable_raw_packets()`].
943    /// This clones data, and is therefore expensive.
944    /// Should not be enabled outside of tests and troubleshooting.
945    RawPacket(Box<RawPacket>),
946}

1257    /// Poll the `Rtc` instance for output. Output can be three things, something to _Transmit_
1258    /// via a UDP socket (maybe via a TURN server). An _Event_, such as receiving media data,
1259    /// or a _Timeout_.
1260    ///
1261    /// The user of the library is expected to continuously call this function and deal with
1262    /// the output until it encounters an [`Output::Timeout`] at which point no further output
1263    /// is produced (if polled again, it will result in just another timeout).
1264    ///
1265    /// After exhausting the `poll_output`, the function will only produce more output again
1266    /// when one of two things happen:
1267    ///
1268    /// 1. The polled timeout is reached.
1269    /// 2. New network input.
1270    ///
1271    /// See [`Rtc`] instance documentation for how this is expected to be used in a loop.
1272    pub fn poll_output(&mut self) -> Result<Output, RtcError> {
1273        let o = self.do_poll_output()?;
1274
1275        match &o {
1276            Output::Event(e) => match e {
1277                Event::ChannelData(_) | Event::MediaData(_) => trace!("{:?}", e),
1278                _ => debug!("{:?}", e),
1279            },
1280            Output::Transmit(t) => {
1281                self.peer_bytes_tx += t.contents.len() as u64;
1282                trace!("OUT {:?}", t)
1283            }
1284            Output::Timeout(_t) => {}
1285        }
1286
1287        Ok(o)
1288    }
1289
1290    fn do_poll_output(&mut self) -> Result<Output, RtcError> {
1291        if !self.alive {
1292            return Ok(Output::Timeout(not_happening()));
1293        }
1294
1295        while let Some(e) = self.ice.poll_event() {
1296            match e {
1297                IceAgentEvent::IceRestart(_) => {
1298                    //
1299                }
1300                IceAgentEvent::IceConnectionStateChange(v) => {
1301                    return Ok(Output::Event(Event::IceConnectionStateChange(v)))
1302                }
1303                IceAgentEvent::DiscoveredRecv { proto, source } => {
1304                    info!("ICE remote address: {:?}/{:?}", source, proto);
1305                    self.remote_addrs.push(source);
1306                    while self.remote_addrs.len() > 20 {
1307                        self.remote_addrs.remove(0);
1308                    }
1309                }
1310                IceAgentEvent::NominatedSend {
1311                    proto,
1312                    source,
1313                    destination,
1314                } => {
1315                    info!(
1316                        "ICE nominated send from: {:?} to: {:?} with protocol {:?}",
1317                        source, destination, proto,
1318                    );
1319                    self.send_addr = Some(SendAddr {
1320                        proto,
1321                        source,
1322                        destination,
1323                    });
1324                }
1325            }
1326        }
1327
1328        let mut dtls_connected = false;
1329
1330        while let Some(e) = self.dtls.poll_event() {
1331            match e {
1332                DtlsEvent::Connected => {
1333                    debug!("DTLS connected");
1334                    dtls_connected = true;
1335                }
1336                DtlsEvent::SrtpKeyingMaterial(mat, srtp_profile) => {
1337                    info!(
1338                        "DTLS set SRTP keying material and profile: {}",
1339                        srtp_profile
1340                    );
1341                    let active = self.dtls.is_active().expect("DTLS must be inited by now");
1342                    self.session.set_keying_material(mat, srtp_profile, active);
1343                }
1344                DtlsEvent::RemoteFingerprint(v1) => {
1345                    debug!("DTLS verify remote fingerprint");
1346                    if let Some(v2) = &self.remote_fingerprint {
1347                        if v1 != *v2 {
1348                            self.disconnect();
1349                            return Err(RtcError::RemoteSdp("remote fingerprint no match".into()));
1350                        }
1351                    } else {
1352                        self.disconnect();
1353                        return Err(RtcError::RemoteSdp("no a=fingerprint before dtls".into()));
1354                    }
1355                }
1356                DtlsEvent::Data(v) => {
1357                    self.sctp.handle_input(self.last_now, &v);
1358                }
1359            }
1360        }
1361
1362        if dtls_connected {
1363            return Ok(Output::Event(Event::Connected));
1364        }
1365
1366        while let Some(e) = self.sctp.poll() {
1367            match e {
1368                SctpEvent::Transmit { mut packets } => {
1369                    if let Some(v) = packets.front() {
1370                        if let Err(e) = self.dtls.handle_input(v) {
1371                            if e.is_would_block() {
1372                                self.sctp.push_back_transmit(packets);
1373                                break;
1374                            } else {
1375                                return Err(e.into());
1376                            }
1377                        }
1378                        packets.pop_front();
1379                        // If there are still packets, they are sent on next
1380                        // poll_output()
1381                        if !packets.is_empty() {
1382                            self.sctp.push_back_transmit(packets);
1383                        }
1384                        break;
1385                    }
1386                }
1387                SctpEvent::Open { id, label } => {
1388                    self.chan.ensure_channel_id_for(id);
1389                    let id = self.chan.channel_id_by_stream_id(id).unwrap();
1390                    return Ok(Output::Event(Event::ChannelOpen(id, label)));
1391                }
1392                SctpEvent::Close { id } => {
1393                    let Some(id) = self.chan.channel_id_by_stream_id(id) else {
1394                        warn!("Drop ChannelClose event for id: {:?}", id);
1395                        continue;
1396                    };
1397                    self.chan.remove_channel(id);
1398                    return Ok(Output::Event(Event::ChannelClose(id)));
1399                }
1400                SctpEvent::Data { id, binary, data } => {
1401                    let Some(id) = self.chan.channel_id_by_stream_id(id) else {
1402                        warn!("Drop ChannelData event for id: {:?}", id);
1403                        continue;
1404                    };
1405                    let cd = ChannelData { id, binary, data };
1406                    return Ok(Output::Event(Event::ChannelData(cd)));
1407                }
1408            }
1409        }
1410
1411        if let Some(ev) = self.session.poll_event() {
1412            return Ok(Output::Event(ev));
1413        }
1414
1415        // Some polling needs to bubble up errors.
1416        if let Some(ev) = self.session.poll_event_fallible()? {
1417            return Ok(Output::Event(ev));
1418        }
1419
1420        if let Some(e) = self.stats.as_mut().and_then(|s| s.poll_output()) {
1421            return Ok(match e {
1422                StatsEvent::Peer(s) => Output::Event(Event::PeerStats(s)),
1423                StatsEvent::MediaIngress(s) => Output::Event(Event::MediaIngressStats(s)),
1424                StatsEvent::MediaEgress(s) => Output::Event(Event::MediaEgressStats(s)),
1425            });
1426        }
1427
1428        if let Some(v) = self.ice.poll_transmit() {
1429            return Ok(Output::Transmit(v));
1430        }
1431
1432        if let Some(send) = &self.send_addr {
1433            // These can only be sent after we got an ICE connection.
1434            let datagram = None
1435                .or_else(|| self.dtls.poll_datagram())
1436                .or_else(|| self.session.poll_datagram(self.last_now));
1437
1438            if let Some(contents) = datagram {
1439                let t = net::Transmit {
1440                    proto: send.proto,
1441                    source: send.source,
1442                    destination: send.destination,
1443                    contents,
1444                };
1445                return Ok(Output::Transmit(t));
1446            }
1447        }
1448
1449        let time_and_reason = (None, "<not happening>")
1450            .soonest((self.ice.poll_timeout(), "ice"))
1451            .soonest((self.session.poll_timeout(), "session"))
1452            .soonest((self.sctp.poll_timeout(), "sctp"))
1453            .soonest((self.chan.poll_timeout(&self.sctp), "chan"))
1454            .soonest((self.stats.as_mut().and_then(|s| s.poll_timeout()), "stats"));
1455
1456        // trace!("poll_output timeout reason: {}", time_and_reason.1);
1457
1458        let time = time_and_reason.0.unwrap_or_else(not_happening);
1459
1460        // We want to guarantee time doesn't go backwards.
1461        let next = if time < self.last_now {
1462            self.last_now
1463        } else {
1464            time
1465        };
1466
1467        Ok(Output::Timeout(next))
1468    }

1470    /// Check if this `Rtc` instance accepts the given input. This is used for demultiplexing
1471    /// several `Rtc` instances over the same UDP server socket.
1472    ///
1473    /// [`Input::Timeout`] is always accepted. [`Input::Receive`] is tested against the nominated
1474    /// ICE candidate. If that doesn't match and the incoming data is a STUN packet, the accept call
1475    /// is delegated to the ICE agent which recognizes the remote peer from `a=ufrag`/`a=password`
1476    /// credentials negotiated in the SDP. If that also doesn't match, all remote ICE candidates are
1477    /// checked for a match.
1478    ///
1479    /// In a server setup, the server would try to find an `Rtc` instances using [`Rtc::accepts()`].
1480    /// The first found instance would be given the input via [`Rtc::handle_input()`].
1481    ///
1482    /// ```no_run
1483    /// # use str0m::{Rtc, Input};
1484    /// // A vec holding the managed rtc instances. One instance per remote peer.
1485    /// let mut rtcs = vec![Rtc::new(), Rtc::new(), Rtc::new()];
1486    ///
1487    /// // Configure instances with local ice candidates etc.
1488    ///
1489    /// loop {
1490    ///     // TODO poll_timeout() and handle the output.
1491    ///
1492    ///     let input: Input = todo!(); // read network data from socket.
1493    ///     for rtc in &mut rtcs {
1494    ///         if rtc.accepts(&input) {
1495    ///             rtc.handle_input(input).unwrap();
1496    ///         }
1497    ///     }
1498    /// }
1499    /// ```
1500    pub fn accepts(&self, input: &Input) -> bool {
1501        let Input::Receive(_, r) = input else {
1502            // always accept the Input::Timeout.
1503            return true;
1504        };
1505
1506        // Fast path: DTLS, RTP, and RTCP traffic coming in from the same socket address
1507        // we've nominated for sending via the ICE agent. This is the typical case
1508        if let Some(send_addr) = &self.send_addr {
1509            if r.source == send_addr.destination {
1510                return true;
1511            }
1512        }
1513
1514        // STUN can use the ufrag/password to identify that a message belongs
1515        // to this Rtc instance.
1516        if let DatagramRecvInner::Stun(v) = &r.contents.inner {
1517            return self.ice.accepts_message(v);
1518        }
1519
1520        // Slow path: Occasionally, traffic comes in on a socket address corresponding
1521        // to a successful candidate pair other than the one we've currently nominated.
1522        // This typically happens at the beginning of the connection
1523        if self.ice.has_viable_remote_candidate(r.source) {
1524            return true;
1525        }
1526
1527        false
1528    }

1530    /// Provide input to this `Rtc` instance. Input is either a [`Input::Timeout`] for some
1531    /// time that was previously obtained from [`Rtc::poll_output()`], or [`Input::Receive`]
1532    /// for network data.
1533    ///
1534    /// Both the timeout and the network data contains a [`std::time::Instant`] which drives
1535    /// time forward in the instance. For network data, the intention is to record the time
1536    /// of receiving the network data as precise as possible. This time is used to calculate
1537    /// things like jitter and bandwidth.
1538    ///
1539    /// It's always okay to call [`Rtc::handle_input()`] with a timeout, also before the
1540    /// time obtained via [`Rtc::poll_output()`].
1541    ///
1542    /// ```no_run
1543    /// # use str0m::{Rtc, Input};
1544    /// # use std::time::Instant;
1545    /// let mut rtc = Rtc::new();
1546    ///
1547    /// loop {
1548    ///     let timeout: Instant = todo!(); // rtc.poll_output() until we get a timeout.
1549    ///
1550    ///     let input: Input = todo!(); // wait for network data or timeout.
1551    ///     rtc.handle_input(input);
1552    /// }
1553    /// ```
1554    pub fn handle_input(&mut self, input: Input) -> Result<(), RtcError> {
1555        if !self.alive {
1556            return Ok(());
1557        }
1558
1559        match input {
1560            Input::Timeout(now) => self.do_handle_timeout(now)?,
1561            Input::Receive(now, r) => {
1562                self.do_handle_receive(now, r)?;
1563                self.do_handle_timeout(now)?;
1564            }
1565        }
1566        Ok(())
1567    }

1569    fn init_time(&mut self, now: Instant) {
1570        // The operation is somewhat expensive, hence we only do it once.
1571        if !self.need_init_time {
1572            return;
1573        }
1574
1575        // We assume this first "now" is a time 0 start point for calculating ntp/unix time offsets.
1576        // This initializes the conversion of Instant -> NTP/Unix time.
1577        let _ = now.to_unix_duration();
1578
1579        self.need_init_time = false;
1580    }
1581
1582    fn do_handle_timeout(&mut self, now: Instant) -> Result<(), RtcError> {
1583        self.init_time(now);
1584
1585        self.last_now = now;
1586        self.ice.handle_timeout(now);
1587        self.sctp.handle_timeout(now);
1588        self.chan.handle_timeout(now, &mut self.sctp);
1589        self.session.handle_timeout(now)?;
1590
1591        if let Some(stats) = &mut self.stats {
1592            if stats.wants_timeout(now) {
1593                let mut snapshot = StatsSnapshot::new(now);
1594                snapshot.peer_rx = self.peer_bytes_rx;
1595                snapshot.peer_tx = self.peer_bytes_tx;
1596                self.session.visit_stats(now, &mut snapshot);
1597                stats.do_handle_timeout(&mut snapshot);
1598            }
1599        }
1600
1601        Ok(())
1602    }
1603
1604    fn do_handle_receive(&mut self, now: Instant, r: net::Receive) -> Result<(), RtcError> {
1605        self.init_time(now);
1606
1607        trace!("IN {:?}", r);
1608        self.last_now = now;
1609        use DatagramRecvInner::*;
1610
1611        let bytes_rx = match r.contents.inner {
1612            // TODO: stun is already parsed (depacketized) here
1613            Stun(_) => 0,
1614            Dtls(v) | Rtp(v) | Rtcp(v) => v.len(),
1615        };
1616
1617        self.peer_bytes_rx += bytes_rx as u64;
1618
1619        match r.contents.inner {
1620            Stun(stun) => {
1621                let packet = io::StunPacket {
1622                    proto: r.proto,
1623                    source: r.source,
1624                    destination: r.destination,
1625                    message: stun,
1626                };
1627                self.ice.handle_packet(now, packet);
1628            }
1629            Dtls(dtls) => self.dtls.handle_receive(dtls)?,
1630            Rtp(rtp) => self.session.handle_rtp_receive(now, rtp),
1631            Rtcp(rtcp) => self.session.handle_rtcp_receive(now, rtcp),
1632        }
1633
1634        Ok(())
1635    }

2191/// Log a CSV like stat to stdout.
2192///
2193/// ```ignore
2194/// log_stat!("MY_STAT", 1, "hello", 3);
2195/// ```
2196///
2197/// will result in the following being printed
2198///
2199/// ```text
2200/// MY_STAT 1, hello, 3, {unix_timestamp_ms}
2201/// ````
2202///
2203/// These logs can be easily grepped for, parsed and graphed, or otherwise analyzed.
2204///
2205/// This macro turns into a NO-OP if the `_internal_dont_use_log_stats` feature is not enabled
2206macro_rules! log_stat {
2207    ($name:expr, $($arg:expr),+) => {
2208        #[cfg(feature = "_internal_dont_use_log_stats")]
2209        {
2210            use std::time::SystemTime;
2211            use std::io::{self, Write};
2212
2213            let now = SystemTime::now();
2214            let since_epoch = now.duration_since(SystemTime::UNIX_EPOCH).unwrap();
2215            let unix_time_ms = since_epoch.as_millis();
2216            let mut lock = io::stdout().lock();
2217            write!(lock, "{} ", $name).expect("Failed to write to stdout");
2218
2219            $(
2220                write!(lock, "{},", $arg).expect("Failed to write to stdout");
2221            )+
2222            writeln!(lock, "{}", unix_time_ms).expect("Failed to write to stdout");
2223        }
2224    };
2225}
