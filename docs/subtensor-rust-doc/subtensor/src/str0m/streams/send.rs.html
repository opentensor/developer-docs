234    /// Write RTP packet to a send stream.
235    ///
236    /// The `payload` argument is expected to be only the RTP payload, not the RTP packet header.
237    ///
238    /// * `pt` Payload type. Declared in the Media this encoded stream belongs to.
239    /// * `seq_no` Sequence number to use for this packet.
240    /// * `time` Time in whatever the clock rate is for the media in question (normally 90_000 for video
241    ///          and 48_000 for audio).
242    /// * `wallclock` Real world time that corresponds to the media time in the RTP packet. For an SFU,
243    ///               this can be hard to know, since RTP packets typically only contain the media
244    ///               time (RTP time). In the simplest SFU setup, the wallclock could simply be the
245    ///               arrival time of the incoming RTP data. For better synchronization the SFU
246    ///               probably needs to weigh in clock drifts and data provided via the statistics, receiver
247    ///               reports etc.
248    /// * `marker` Whether to "mark" this packet. This is usually done for the last packet belonging to
249    ///            a series of RTP packets constituting the same frame in a video stream.
250    /// * `ext_vals` The RTP header extension values to set. The values must be mapped in the session,
251    ///              or they will not be set on the RTP packet.
252    /// * `nackable` Whether we should respond this packet for incoming NACK from the remote peer. For
253    ///              audio this is always false. For temporal encoded video, some packets are discardable
254    ///              and this flag should be set accordingly.
255    /// * `payload` RTP packet payload, without header.
256    #[allow(clippy::too_many_arguments)]
257    pub fn write_rtp(
258        &mut self,
259        pt: Pt,
260        seq_no: SeqNo,
261        time: u32,
262        wallclock: Instant,
263        marker: bool,
264        ext_vals: ExtensionValues,
265        nackable: bool,
266        payload: Vec<u8>,
267    ) -> Result<(), RtcError> {
268        let first_call = self.rtp_and_wallclock.is_none();
269
270        if first_call && seq_no.roc() > 0 {
271            // TODO: make it possible to supress this.
272            warn!("First SeqNo has non-zero ROC ({}), which needs out-of-band signalling to remote peer", seq_no.roc());
273        }
274
275        // This 1 in clock frequency will be fixed in poll_output.
276        let media_time = MediaTime::from_secs(time as i64);
277        self.rtp_and_wallclock = Some((time, wallclock));
278
279        let header = RtpHeader {
280            sequence_number: *seq_no as u16,
281            marker,
282            payload_type: pt,
283            timestamp: time,
284            ssrc: self.ssrc,
285            ext_vals,
286            ..Default::default()
287        };
288
289        let packet = RtpPacket {
290            seq_no,
291            time: media_time,
292            header,
293            payload,
294            nackable,
295            // The overall idea for str0m is to only drive time forward from handle_input. If we
296            // used a "now" argument to write_rtp(), we effectively get a second point that also need
297            // to move time forward _for all of Rtc_ – that's too complicated.
298            //
299            // Instead we set a future timestamp here. When time moves forward in the "regular way",
300            // in handle_timeout() we delegate to self.send_queue.handle_timeout() to mark the enqueued
301            // timestamp of all packets that are about to be sent.
302            timestamp: not_happening(),
303
304            // This is only relevant for incoming RTP packets.
305            last_sender_info: None,
306        };
307
308        self.send_queue.push(packet);
309
310        Ok(())
311    }

317    pub(crate) fn poll_packet(
318        &mut self,
319        now: Instant,
320        exts: &ExtensionMap,
321        twcc: &mut u64,
322        params: &[PayloadParams],
323        buf: &mut Vec<u8>,
324    ) -> Option<PacketReceipt> {
325        let mid = self.mid;
326        let rid = self.rid;
327        let ssrc_rtx = self.rtx;
328
329        let (next, is_padding) = if let Some(next) = self.poll_packet_resend(now) {
330            (next, false)
331        } else if let Some(next) = self.poll_packet_regular(now) {
332            (next, false)
333        } else if let Some(next) = self.poll_packet_padding(now) {
334            (next, true)
335        } else {
336            return None;
337        };
338
339        let pop_send_queue = next.kind == NextPacketKind::Regular;
340
341        // Need the header for the receipt and modifications
342        // TODO: Can we remove this?
343        let header_ref = &mut next.pkt.header;
344
345        // This is true also for RTX.
346        header_ref.ext_vals.mid = Some(mid);
347
348        let pt_main = header_ref.payload_type;
349
350        // The pt in next.pkt is the "main" pt.
351        let Some(param) = params.iter().find(|p| p.pt() == pt_main) else {
352            // PT does not exist in the connected media.
353            warn!("Media is missing PT ({}) used in RTP packet", pt_main);
354
355            // Get rid of this packet we can't send.
356            if pop_send_queue {
357                self.send_queue.pop(now);
358            }
359
360            return None;
361        };
362
363        let mut set_pt_for_padding = None;
364        let mut set_cr = None;
365
366        let mut header = match next.kind {
367            NextPacketKind::Regular => {
368                let rtx_possible = param.resend().is_some();
369
370                if rtx_possible {
371                    // Remember PT We want to set these directly on `self` here, but can't
372                    // because we already have a mutable borrow. We are using pt_main
373                    // since the above loop figuring out param needs to be correct also
374                    // for the NextPacketKind::Blank case.
375                    set_pt_for_padding = Some(pt_main);
376                } else {
377                    // If the PT we're sending on doesn't have a corresponding RTX PT,
378                    // the packet is de-facto not nackable.
379                    //
380                    // This blocks incoming NACK requests and thus ensures there are no
381                    // entries in self.retries without a RTX PT.
382                    next.pkt.nackable = false;
383                }
384
385                let clock_rate = param.spec().clock_rate;
386                set_cr = Some(clock_rate);
387
388                // Modify the cached packet time. This is so write_rtp can use u32 media time without
389                // worrying about lengthening or the clock rate.
390                let time = MediaTime::new(next.pkt.time.numer(), clock_rate);
391                next.pkt.time = time;
392
393                // Modify the original (and also cached) header value.
394                header_ref.ext_vals.rid = rid;
395                header_ref.ext_vals.rid_repair = None;
396
397                header_ref.clone()
398            }
399            NextPacketKind::Resend(_) | NextPacketKind::Blank(_) => {
400                // * For the Resend case, we will not have accepted/cached the packet unless
401                //   we have a RTX PT (see logic setting next.pkt.nackable above).
402                // * For the Blank case, we will only have produced blank packets if we
403                //   got a "real" PTX RT, either via set_pt_for_padding above, or via
404                //   the on_first_timeout() further down.
405                // Either way, unwrapping this optional _should_ be correct.
406                let pt_rtx = param.resend().expect("PT for resend or blank");
407
408                // Clone header to not change the original (cached) header.
409                let mut header = header_ref.clone();
410
411                // Update clone of header (to not change the cached value).
412                header.payload_type = pt_rtx;
413                header.ssrc = ssrc_rtx.expect("Should have RTX SSRC for resends");
414                header.sequence_number = *next.seq_no as u16;
415
416                header.ext_vals.rid = None;
417                header.ext_vals.rid_repair = rid;
418
419                header
420            }
421        };
422
423        // These need to match `Extension::is_supported()` so we are sending what we are
424        // declaring we support.
425        header.ext_vals.abs_send_time = Some(now);
426        header.ext_vals.transport_cc = Some(*twcc as u16);
427        *twcc += 1;
428
429        buf.resize(DATAGRAM_MAX_PACKET_SIZE, 0);
430
431        let header_len = header.write_to(buf, exts);
432        assert!(header_len % 4 == 0, "RTP header must be multiple of 4");
433        header.header_len = header_len;
434
435        let mut body_out = &mut buf[header_len..];
436
437        // For resends, the original seq_no is inserted before the payload.
438        let mut original_seq_len = 0;
439        if let NextPacketKind::Resend(orig_seq_no) = next.kind {
440            original_seq_len = RtpHeader::write_original_sequence_number(body_out, orig_seq_no);
441            body_out = &mut body_out[original_seq_len..];
442        }
443
444        let pkt = &next.pkt;
445
446        let body_len = match next.kind {
447            NextPacketKind::Regular | NextPacketKind::Resend(_) => {
448                let body_len = pkt.payload.len();
449                body_out[..body_len].copy_from_slice(&pkt.payload);
450
451                // pad for SRTP
452                let pad_len = RtpHeader::pad_packet(
453                    &mut buf[..],
454                    header_len,
455                    body_len + original_seq_len,
456                    SRTP_BLOCK_SIZE,
457                );
458
459                body_len + original_seq_len + pad_len
460            }
461            NextPacketKind::Blank(len) => {
462                let len = RtpHeader::create_padding_packet(
463                    &mut buf[..],
464                    header_len,
465                    len,
466                    SRTP_BLOCK_SIZE,
467                );
468
469                if len == 0 {
470                    return None;
471                }
472
473                len
474            }
475        };
476
477        buf.truncate(header_len + body_len);
478
479        #[cfg(feature = "_internal_dont_use_log_stats")]
480        {
481            let queued_at = match next.kind {
482                NextPacketKind::Regular => Some(pkt.timestamp),
483                _ => {
484                    // TODO: We don't have queued at stats for Resends or blank padding.
485                    None
486                }
487            };
488
489            if let Some(delay) = queued_at.map(|i| now.duration_since(i)) {
490                crate::log_stat!("QUEUE_DELAY", header.ssrc, delay.as_secs_f64() * 1000.0);
491            }
492        }
493
494        let seq_no = next.seq_no;
495        self.last_used = now;
496
497        // Padding comes in two forms, "spurious resends" of sent packets where
498        // the remote side didn't ask for a resend. The other variant are blank
499        // packets, containing nothing but zeroes. Such packets must be sent from
500        // _some_ RTX PT. A good pick is the RTX for the PT last used to send
501        // regular media data.
502        //
503        // This is set here due to borrow checker.
504        if set_pt_for_padding.is_some() && self.pt_for_padding != set_pt_for_padding {
505            self.pt_for_padding = set_pt_for_padding;
506        }
507
508        if set_cr.is_some() && self.clock_rate != set_cr {
509            self.clock_rate = set_cr;
510        }
511
512        if pop_send_queue {
513            // poll_packet_regular leaves the packet in the head of the send_queue
514            let pkt = self
515                .send_queue
516                .pop(now)
517                .expect("head of send_queue to be there");
518            if pkt.nackable {
519                self.rtx_cache.cache_sent_packet(pkt, now);
520            }
521        }
522
523        Some(PacketReceipt {
524            header,
525            seq_no,
526            is_padding,
527            payload_size: body_len,
528        })
529    }

531    fn rtx_ratio_downsampled(&mut self, now: Instant) -> f32 {
532        let (value, ts) = self.rtx_ratio;
533        if now - ts < Duration::from_millis(50) {
534            // not worth re-evaluating, return the old value
535            return value;
536        }
537
538        // bytes stats refer to the last second by default
539        let bytes_transmitted = self.stats.bytes_transmitted.sum();
540        let bytes_retransmitted = self.stats.bytes_retransmitted.sum();
541        let ratio = bytes_retransmitted as f32 / (bytes_retransmitted + bytes_transmitted) as f32;
542        let ratio = if ratio.is_finite() { ratio } else { 0_f32 };
543        self.rtx_ratio = (ratio, now);
544        ratio
545    }

547    fn poll_packet_resend(&mut self, now: Instant) -> Option<NextPacket<'_>> {
548        let ratio = self.rtx_ratio_downsampled(now);
549
550        // If we hit the cap, stop doing resends by clearing those we have queued.
551        if ratio > 0.15_f32 {
552            self.resends.clear();
553            return None;
554        }
555
556        let seq_no = loop {
557            let resend = self.resends.pop_front()?;
558
559            let pkt = self.rtx_cache.get_cached_packet_by_seq_no(resend.seq_no);
560
561            // The seq_no could simply be too old to exist in the buffer, in which
562            // case we will not do a resend.
563            let Some(pkt) = pkt else {
564                continue;
565            };
566
567            // Cached packets must be nackable. This is ensured before adding the
568            // entry to the self.rtx_cache.
569            assert!(pkt.nackable);
570
571            break pkt.seq_no;
572        };
573
574        // Borrow checker gymnastics.
575        let pkt = self.rtx_cache.get_cached_packet_by_seq_no(seq_no).unwrap();
576
577        let len = pkt.payload.len() as u64;
578        self.stats.update_packet_counts(len, true);
579        self.stats.bytes_retransmitted.push(now, len);
580
581        let seq_no = self.seq_no_rtx.inc();
582
583        let orig_seq_no = pkt.seq_no;
584
585        Some(NextPacket {
586            kind: NextPacketKind::Resend(orig_seq_no),
587            seq_no,
588            pkt,
589        })
590    }

592    fn poll_packet_regular(&mut self, now: Instant) -> Option<NextPacket<'_>> {
593        // exit via ? here is ok since that means there is nothing to send.
594        // The packet remains in the head of the send queue until we
595        // finish poll_packet, at which point we move it to the cache.
596        let pkt = self.send_queue.peek()?;
597
598        pkt.timestamp = now;
599
600        let len = pkt.payload.len() as u64;
601        self.stats.update_packet_counts(len, false);
602        self.stats.bytes_transmitted.push(now, len);
603
604        let seq_no = pkt.seq_no;
605
606        Some(NextPacket {
607            kind: NextPacketKind::Regular,
608            seq_no,
609            pkt,
610        })
611    }

613    fn poll_packet_padding(&mut self, _now: Instant) -> Option<NextPacket> {
614        if !self.padding_enabled() {
615            self.padding = 0;
616            return None;
617        }
618
619        if self.padding == 0 {
620            return None;
621        }
622
623        #[allow(clippy::unnecessary_operation)]
624        'outer: {
625            if self.padding > MIN_SPURIOUS_PADDING_SIZE {
626                // Find a historic packet that is smaller than this max size. The max size
627                // is a headroom since we can accept slightly larger padding than asked for.
628                let max_size = (self.padding * 2).min(DATAGRAM_MTU_WARN - MAX_RTP_OVERHEAD);
629
630                let Some(pkt) = self.rtx_cache.get_cached_packet_smaller_than(max_size) else {
631                    // Couldn't find spurious packet, try a blank packet instead.
632                    break 'outer;
633                };
634
635                let orig_seq_no = pkt.seq_no;
636                let seq_no = self.seq_no_rtx.inc();
637
638                self.padding = self.padding.saturating_sub(pkt.payload.len());
639
640                return Some(NextPacket {
641                    kind: NextPacketKind::Resend(orig_seq_no),
642                    seq_no,
643                    pkt,
644                });
645            }
646        };
647
648        let seq_no = self.seq_no_rtx.inc();
649
650        let pkt = &mut self.blank_packet;
651        pkt.seq_no = seq_no;
652        // Unwrap here is correct because self.padding_enabled() above checks the we got the PT set.
653        pkt.header.payload_type = self.pt_for_padding.unwrap();
654
655        let len = self
656            .padding
657            .clamp(SRTP_BLOCK_SIZE, MAX_BLANK_PADDING_PAYLOAD_SIZE);
658        assert!(len <= 255); // should fit in a byte
659
660        self.padding = self.padding.saturating_sub(len);
661
662        Some(NextPacket {
663            kind: NextPacketKind::Blank(len as u8),
664            seq_no,
665            pkt,
666        })
667    }

685    pub(crate) fn handle_rtcp(&mut self, now: Instant, fb: RtcpFb) {
686        use RtcpFb::*;
687        match fb {
688            ReceptionReport(r) => self.stats.update_with_rr(now, r),
689            Nack(_, list) => {
690                self.stats.increase_nacks();
691                let entries = list.into_iter();
692                self.handle_nack(entries, now);
693            }
694            Pli(_) => {
695                self.stats.increase_plis();
696                self.pending_request_keyframe = Some(KeyframeRequestKind::Pli);
697            }
698            Fir(_) => {
699                self.stats.increase_firs();
700                self.pending_request_keyframe = Some(KeyframeRequestKind::Fir);
701            }
702            Remb(r) => {
703                self.pending_request_remb = Some(Bitrate::from(r.bitrate as f64));
704            }
705            Twcc(_) => unreachable!("TWCC should be handled on session level"),
706            _ => {}
707        }
708    }

710    pub(crate) fn handle_nack(
711        &mut self,
712        entries: impl Iterator<Item = NackEntry>,
713        now: Instant,
714    ) -> Option<()> {
715        // Turning NackEntry into SeqNo we need to know a SeqNo "close by" to lengthen the 16 bit
716        // sequence number into the 64 bit we have in SeqNo.
717        let seq_no = self.rtx_cache.last_cached_seq_no()?;
718        let iter = entries.flat_map(|n| n.into_iter(seq_no));
719
720        // Schedule all resends. They will be handled on next poll_packet
721        for seq_no in iter {
722            let Some(packet) = self.rtx_cache.get_cached_packet_by_seq_no(seq_no) else {
723                // Packet was not available in RTX cache, it has probably expired.
724                continue;
725            };
726
727            let resend = Resend {
728                seq_no,
729                queued_at: now,
730                payload_size: packet.payload.len(),
731            };
732            self.resends.push_back(resend);
733        }
734
735        Some(())
736    }

792    fn current_rtp_time(&self, now: Instant) -> Option<MediaTime> {
793        // This is the RTP time and the wallclock from the last written media.
794        // We use that as an offset to current time (now), to calculate the
795        // current RTP time.
796        let (t_u32, w) = self.rtp_and_wallclock?;
797
798        let clock_rate = self.clock_rate?;
799        let t = MediaTime::new(t_u32 as i64, clock_rate);
800
801        // Wallclock needs to be in the past.
802        if w > now {
803            let delta = w - now;
804            debug!("write_rtp wallclock is in the future: {:?}", delta);
805            return None;
806        }
807        let offset = now - w;
808
809        // This might be in the wrong base.
810        let rtp_time = t + offset.into();
811
812        Some(rtp_time.rebase(clock_rate))
813    }

831    pub(crate) fn queue_state(&mut self, now: Instant) -> QueueState {
832        // The unpaced flag is set to a default value on first handle_timeout. The
833        // default is to not pace audio. We unwrap default to "true" here to not
834        // apply any pacing until we know what kind of content we are sending.
835        let unpaced = self.unpaced.unwrap_or(true);
836
837        // It's only possible to use this sender for padding if RTX is enabled and
838        // we know a PT to use for it.
839        let use_for_padding = self.padding_enabled();
840
841        let mut snapshot = self.send_queue.snapshot(now);
842
843        if let Some(snapshot_resend) = self.queue_state_resend(now) {
844            snapshot.merge(&snapshot_resend);
845        }
846
847        if let Some(snapshot_padding) = self.queue_state_padding(now) {
848            snapshot.merge(&snapshot_padding);
849        }
850
851        QueueState {
852            mid: self.mid,
853            unpaced,
854            use_for_padding,
855            snapshot,
856        }
857    }
858
859    fn queue_state_resend(&self, now: Instant) -> Option<QueueSnapshot> {
860        if self.resends.is_empty() {
861            return None;
862        }
863
864        // Outstanding resends
865        let mut snapshot = self
866            .resends
867            .iter()
868            .fold(QueueSnapshot::default(), |mut snapshot, r| {
869                snapshot.total_queue_time_origin += now.duration_since(r.queued_at);
870                snapshot.size += r.payload_size;
871                snapshot.packet_count += 1;
872                snapshot.first_unsent = snapshot
873                    .first_unsent
874                    .map(|i| i.min(r.queued_at))
875                    .or(Some(r.queued_at));
876
877                snapshot
878            });
879        snapshot.created_at = now;
880        snapshot.update_priority(QueuePriority::Media);
881
882        Some(snapshot)
883    }
884
885    fn queue_state_padding(&self, now: Instant) -> Option<QueueSnapshot> {
886        if self.padding == 0 {
887            return None;
888        }
889
890        // TODO: Be more scientific about these factors.
891        const AVERAGE_PADDING_PACKET_SIZE: usize = 800;
892        const FAKE_PADDING_DURATION_MILLIS: usize = 5;
893
894        let fake_packets = self.padding / AVERAGE_PADDING_PACKET_SIZE;
895        let fake_millis = fake_packets * FAKE_PADDING_DURATION_MILLIS;
896        let fake_duration = Duration::from_millis(fake_millis as u64);
897
898        Some(QueueSnapshot {
899            created_at: now,
900            size: self.padding,
901            packet_count: fake_packets as u32,
902            total_queue_time_origin: fake_duration,
903            priority: QueuePriority::Padding,
904            ..Default::default()
905        })
906    }

919    pub(crate) fn handle_timeout<'a>(
920        &mut self,
921        now: Instant,
922        get_media: impl FnOnce() -> (&'a Media, &'a CodecConfig),
923    ) {
924        // If kind is None, this is the first time we ever get a handle_timeout.
925        if self.kind.is_none() {
926            let (media, config) = get_media();
927            self.on_first_timeout(media, config);
928        }
929
930        self.send_queue.handle_timeout(now);
931    }
932
933    fn on_first_timeout(&mut self, media: &Media, config: &CodecConfig) {
934        // Always set on first timeout.
935        self.kind = Some(media.kind());
936        self.cname = Some(media.cname().to_string());
937
938        // Set on first timeout, if not set already by configuration.
939        if self.unpaced.is_none() {
940            // Default audio to be unpaced.
941            self.unpaced = Some(media.kind().is_audio());
942        }
943
944        // To allow for sending padding on a newly created StreamTx, before any regular
945        // packet has been sent, we need any main PT that has associated RTX. This is
946        // later be overwritten when we send the first regular packet.
947        if self.rtx.is_some() && self.pt_for_padding.is_none() {
948            if let Some(pt) = media.first_pt_with_rtx(config) {
949                trace!(
950                    "StreamTx Mid {} PT {} before first regular packet",
951                    self.mid,
952                    pt
953                );
954                self.pt_for_padding = Some(pt);
955
956                // Setting the pt_for_rtx should enable RTX.
957                assert!(self.padding_enabled());
958            }
959        }
960    }

992    fn update_with_rr(&mut self, now: Instant, r: ReceptionReport) {
993        let ntp_time = now.to_ntp_duration();
994        let rtt = calculate_rtt_ms(ntp_time, r.last_sr_delay, r.last_sr_time);
995        self.rtt = rtt;
996
997        let ext_seq = {
998            let prev = self.losses.last().map(|s| s.0).unwrap_or(r.max_seq as u64);
999            let next = (r.max_seq & 0xffff) as u16;
1000            extend_u16(Some(prev), next)
1001        };
1002
1003        self.losses
1004            .push((ext_seq, r.fraction_lost as f32 / u8::MAX as f32));
1005    }
1006
1007    pub(crate) fn fill(
1008        &mut self,
1009        snapshot: &mut StatsSnapshot,
1010        mid: Mid,
1011        rid: Option<Rid>,
1012        now: Instant,
1013    ) {
1014        if self.bytes == 0 {
1015            return;
1016        }
1017
1018        let key = (mid, rid);
1019
1020        let loss = {
1021            let mut value = 0_f32;
1022            let mut total_weight = 0_u64;
1023
1024            // just in case we received RRs out of order
1025            self.losses.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap());
1026
1027            // average known RR losses weighted by their number of packets
1028            for it in self.losses.windows(2) {
1029                let [prev, next] = it else { continue };
1030                let weight = next.0.saturating_sub(prev.0);
1031                value += next.1 * weight as f32;
1032                total_weight += weight;
1033            }
1034
1035            let result = value / total_weight as f32;
1036            result.is_finite().then_some(result)
1037        };
1038
1039        self.losses.drain(..self.losses.len().saturating_sub(1));
1040
1041        snapshot.egress.insert(
1042            key,
1043            MediaEgressStats {
1044                mid,
1045                rid,
1046                bytes: self.bytes,
1047                packets: self.packets,
1048                firs: self.firs,
1049                plis: self.plis,
1050                nacks: self.nacks,
1051                rtt: self.rtt,
1052                loss,
1053                timestamp: now,
1054            },
1055        );
1056    }
