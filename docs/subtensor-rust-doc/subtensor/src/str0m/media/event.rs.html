12impl From<SdpSimulcast> for Simulcast {
13    fn from(s: SdpSimulcast) -> Self {
14        let send = s.send.iter().map(|r| r.0.as_ref()).map(Rid::from).collect();
15        let recv = s.recv.iter().map(|r| r.0.as_ref()).map(Rid::from).collect();
16
17        Simulcast { send, recv }
18    }
19}

70/// Video or audio data from the remote peer.
71///
72/// This is obtained via [`Event::MediaData`][crate::Event::MediaData].
73#[derive(PartialEq, Eq)]
74pub struct MediaData {
75    /// Identifier of the media in the session this media belongs to.
76    pub mid: Mid,
77
78    /// Payload type (PT) tells which negotiated codec is being used. Each media
79    /// can carry different codecs, the payload type can theoretically change
80    /// from one packet to the next.
81    pub pt: Pt,
82
83    /// Rtp Stream Id (RID) identifies an RTP stream without referring to its
84    /// Synchronization Source (SSRC).
85    ///
86    /// This is a newer standard that is sometimes used in WebRTC to identify
87    /// a stream. Specifically when using Simulcast in Chrome.
88    pub rid: Option<Rid>,
89
90    /// Parameters for the codec. This is used to match incoming PT to outgoing PT.
91    pub params: PayloadParams,
92
93    /// The RTP media time of this packet. Media time is described as a nominator/denominator
94    /// quantity. The nominator is the timestamp field from the RTP header, the denominator
95    /// depends on whether this is an audio or video packet.
96    ///
97    /// For audio the timebase is 48kHz for video it is 90kHz.
98    pub time: MediaTime,
99
100    /// The time of the [`Input::Receive`][crate::Input::Receive] of the first packet that caused this MediaData.
101    ///
102    /// In simple SFU setups this can be used as wallclock for [`Writer::write`][crate::media::Writer].
103    pub network_time: Instant,
104
105    /// The (RTP) sequence numbers that made up this data.
106    pub seq_range: RangeInclusive<SeqNo>,
107
108    /// Whether the data is contiguous from the one just previously emitted. If this is false,
109    /// we got an interruption in RTP packets, and the data may or may not be usable in a decoder
110    /// without requesting a new keyframe.
111    ///
112    /// For audio this flag most likely doesn't matter.
113    pub contiguous: bool,
114
115    /// The actual packet data a.k.a Sample.
116    ///
117    /// Bigger samples don't fit in one UDP packet, thus WebRTC RTP is chopping up codec
118    /// transmission units into smaller parts.
119    ///
120    /// This data is a full depayloaded Sample.
121    pub data: Vec<u8>,
122
123    /// RTP header extensions for this media data. This is taken from the
124    /// first RTP header.
125    pub ext_vals: ExtensionValues,
126
127    /// Additional codec specific information
128    pub codec_extra: CodecExtra,
129
130    /// Sender information from the most recent Sender Report(SR).
131    ///
132    /// If no Sender Report(SR) has been received this is [`None`].
133    pub last_sender_info: Option<SenderInfo>,
134}

136/// Details for an incoming a keyframe request (PLI or FIR).
137///
138/// This is obtained via the [`Event::KeyframeRequest`][crate::Event::KeyframeRequest].
139///
140/// Sending a keyframe request is done via [`Media::request_keyframe()`][crate::media::Media].
141#[derive(Debug, Clone, Copy, PartialEq, Eq)]
142pub struct KeyframeRequest {
143    /// The media identifier this keyframe request is for.
144    pub mid: Mid,
145
146    /// Rid the keyframe request is for. Relevant when doing simulcast.
147    pub rid: Option<Rid>,
148
149    /// The kind of keyframe request (PLI or FIR).
150    pub kind: KeyframeRequestKind,
151}
152
153/// Type of keyframe request.
154#[derive(Debug, Clone, Copy, PartialEq, Eq)]
155pub enum KeyframeRequestKind {
156    /// Picture Loss Indication (PLI) is a less severe keyframe request that can be
157    /// automatically generated by an SFU or by the end peer.
158    Pli,
159
160    /// Full Intra Request (FIR) is a more severe keyframe request that should only
161    /// be used when it's impossible for an end peer to show a video stream.
162    Fir,
163}

165impl fmt::Debug for MediaData {
166    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
167        f.debug_struct("MediaData")
168            .field("mid", &self.mid)
169            .field("pt", &self.pt)
170            .field("rid", &self.rid)
171            .field("time", &self.time)
172            .field("len", &self.data.len())
173            .finish()
174    }
