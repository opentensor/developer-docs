26const MAX_RTT_HISTORY_WINDOW: usize = 32;
27const INITIAL_BITRATE_WINDOW: Duration = Duration::from_millis(500);
28const BITRATE_WINDOW: Duration = Duration::from_millis(150);
29const UPDATE_INTERVAL: Duration = Duration::from_millis(25);
30/// The maximum time we keep updating our estimate without receiving a TWCC report.
31const MAX_TWCC_GAP: Duration = Duration::from_millis(500);
32
33/// Main entry point for the Googcc inspired BWE implementation.
34///
35/// This takes as input packet statuses recorded at send time and enriched by TWCC reports and produces as its output a periodic
36/// estimate of the available send bitrate.
37pub struct SendSideBandwithEstimator {
38    arrival_group_accumulator: ArrivalGroupAccumulator,
39    trendline_estimator: TrendlineEstimator,
40    rate_control: RateControl,
41    acked_bitrate_estimator: AckedBitrateEstimator,
42    /// Last estimate produced, unlike [`next_estimate`] this will always have a value after the
43    /// first estimate.
44    last_estimate: Option<Bitrate>,
45    /// History of the max RTT derived for each TWCC report.
46    max_rtt_history: VecDeque<Duration>,
47    /// Calculated mean of max_rtt_history.
48    mean_max_rtt: Option<Duration>,
49
50    /// The next time we should poll.
51    next_timeout: Instant,
52    /// The last time we ingested a TWCC report.
53    last_twcc_report: Instant,
54}
55
56impl SendSideBandwithEstimator {
57    pub fn new(initial_bitrate: Bitrate) -> Self {
58        Self {
59            arrival_group_accumulator: ArrivalGroupAccumulator::default(),
60            trendline_estimator: TrendlineEstimator::new(20),
61            acked_bitrate_estimator: AckedBitrateEstimator::new(
62                INITIAL_BITRATE_WINDOW,
63                BITRATE_WINDOW,
64            ),
65            rate_control: RateControl::new(initial_bitrate, Bitrate::kbps(40), Bitrate::gbps(10)),
66            last_estimate: None,
67            max_rtt_history: VecDeque::default(),
68            mean_max_rtt: None,
69            next_timeout: already_happened(),
70            last_twcc_report: already_happened(),
71        }
72    }
73
74    /// Record a packet from a TWCC report.
75    pub(crate) fn update<'t>(
76        &mut self,
77        records: impl Iterator<Item = &'t TwccSendRecord>,
78        now: Instant,
79    ) {
80        let mut acked: Vec<AckedPacket> = Vec::new();
81
82        let mut max_rtt = None;
83        for record in records {
84            let Ok(acked_packet) = record.try_into() else {
85                continue;
86            };
87            acked.push(acked_packet);
88            max_rtt = max_rtt.max(record.rtt());
89        }
90        acked.sort_by(AckedPacket::order_by_receive_time);
91
92        for acked_packet in acked {
93            self.acked_bitrate_estimator
94                .update(acked_packet.remote_recv_time, acked_packet.size);
95
96            if let Some(delay_variation) = self
97                .arrival_group_accumulator
98                .accumulate_packet(acked_packet)
99            {
100                crate::packet::bwe::macros::log_delay_variation!(delay_variation.delay_delta);
101
102                // Got a new delay variation, add it to the trendline
103                self.trendline_estimator
104                    .add_delay_observation(delay_variation, now);
105            }
106        }
107
108        if let Some(rtt) = max_rtt {
109            self.add_max_rtt(rtt);
110        }
111
112        let new_hypothesis = self.trendline_estimator.hypothesis();
113
114        self.update_estimate(
115            new_hypothesis,
116            self.acked_bitrate_estimator.current_estimate(),
117            self.mean_max_rtt,
118            now,
119        );
120        self.last_twcc_report = now;
121    }
122
123    pub(crate) fn poll_timeout(&self) -> Instant {
124        self.next_timeout
125    }
126
127    pub(crate) fn handle_timeout(&mut self, now: Instant) {
128        if !self.trendline_hypothesis_valid(now) {
129            // We haven't received a TWCC report in a while. The trendline hypothesis can
130            // no longer be considered valid. We need another TWCC report before we can update
131            // estimates.
132            let next_timeout_in = self
133                .mean_max_rtt
134                .unwrap_or(MAX_TWCC_GAP)
135                .min(UPDATE_INTERVAL);
136
137            // Set this even if we didn't update, otherwise we get stuck in a poll -> handle loop
138            // that starves the run loop.
139            self.next_timeout = now + next_timeout_in;
140            return;
141        }
142
143        self.update_estimate(
144            self.trendline_estimator.hypothesis(),
145            self.acked_bitrate_estimator.current_estimate(),
146            self.mean_max_rtt,
147            now,
148        );
149    }
150
151    /// Get the latest estimate.
152    pub(crate) fn last_estimate(&self) -> Option<Bitrate> {
153        self.last_estimate
154    }
155
156    fn add_max_rtt(&mut self, max_rtt: Duration) {
157        while self.max_rtt_history.len() > MAX_RTT_HISTORY_WINDOW {
158            self.max_rtt_history.pop_front();
159        }
160        self.max_rtt_history.push_back(max_rtt);
161
162        let sum = self
163            .max_rtt_history
164            .iter()
165            .fold(Duration::ZERO, |acc, rtt| acc + *rtt);
166
167        self.mean_max_rtt = Some(sum / self.max_rtt_history.len() as u32);
168    }
169
170    fn update_estimate(
171        &mut self,
172        hypothesis: BandwithUsage,
173        observed_bitrate: Option<Bitrate>,
174        mean_max_rtt: Option<Duration>,
175        now: Instant,
176    ) {
177        if let Some(observed_bitrate) = observed_bitrate {
178            self.rate_control
179                .update(hypothesis.into(), observed_bitrate, mean_max_rtt, now);
180            let estimated_rate = self.rate_control.estimated_bitrate();
181
182            crate::packet::bwe::macros::log_bitrate_estimate!(estimated_rate.as_f64());
183            self.last_estimate = Some(estimated_rate);
184        }
185
186        // Set this even if we didn't update, otherwise we get stuck in a poll -> handle loop
187        // that starves the run loop.
188        self.next_timeout = now + UPDATE_INTERVAL;
189    }
190
191    /// Whether the current trendline hypothesis is valid i.e. not too old.
192    fn trendline_hypothesis_valid(&self, now: Instant) -> bool {
193        now.duration_since(self.last_twcc_report)
194            <= self
195                .mean_max_rtt
196                .map(|rtt| rtt * 2)
197                .unwrap_or(MAX_TWCC_GAP)
198                .min(UPDATE_INTERVAL * 2)
199    }
200}

202/// A RTP packet that has been sent and acknowledged by the receiver in a TWCC report.
203#[derive(Debug, Copy, Clone)]
204pub struct AckedPacket {
205    /// The TWCC sequence number
206    seq_no: SeqNo,
207    /// The size of the packets in bytes.
208    size: DataSize,
209    /// When we sent the packet
210    local_send_time: Instant,
211    /// When the packet was received at the remote, note this Instant is only usable with other
212    /// instants of the same type i.e. those that represent a TWCC reported receive time for this
213    /// session.
214    remote_recv_time: Instant,
215}
216
217impl AckedPacket {
218    fn order_by_receive_time(lhs: &Self, rhs: &Self) -> Ordering {
219        if lhs.remote_recv_time != rhs.remote_recv_time {
220            lhs.remote_recv_time.cmp(&rhs.remote_recv_time)
221        } else if lhs.local_send_time != rhs.local_send_time {
222            lhs.local_send_time.cmp(&rhs.local_send_time)
223        } else {
224            lhs.seq_no.cmp(&rhs.seq_no)
225        }
226    }
227}
228
229impl TryFrom<&TwccSendRecord> for AckedPacket {
230    type Error = ();
231
232    fn try_from(value: &TwccSendRecord) -> Result<Self, Self::Error> {
233        let Some(remote_recv_time) = value.remote_recv_time() else {
234            return Err(());
235        };
236
237        Ok(Self {
238            seq_no: value.seq(),
239            size: value.size().into(),
240            local_send_time: value.local_send_time(),
241            remote_recv_time,
242        })
243    }
244}

246#[derive(Debug, Clone, Copy, PartialEq, Eq)]
247enum BandwithUsage {
248    Overuse,
249    Normal,
250    Underuse,
251}
252
253impl fmt::Display for BandwithUsage {
254    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
255        match self {
256            BandwithUsage::Overuse => write!(f, "overuse"),
257            BandwithUsage::Normal => write!(f, "normal"),
258            BandwithUsage::Underuse => write!(f, "underuse"),
259        }
260    }
