1use std::time::{Duration, Instant};
2
3use crate::rtp_::DataSize;
4use crate::Bitrate;

6// Ported from libWebRTC's src/modules/congestion_controller/goog_cc/bitrate_estimator.cc at
7// `9f3ccf291e`.
8
9const SMALL_SAMPLE_THRESHOLD: DataSize = DataSize::bytes(2000);
10const SMALL_SAMPLE_UNCERTAINTY: f64 = 25.0;
11const UNCERTAINTY: f64 = 10.0;
12const ESTIMATE_FLOOR: Bitrate = Bitrate::kbps(40);
13
14pub struct AckedBitrateEstimator {
15    /// The initial window to use for the first estimate.
16    initial_window: Duration,
17    /// The window to use for subsequent estimates after achieving the initial estimate.
18    window: Duration,
19    /// The estimate of the acked bitrate.
20    estimate: Option<Bitrate>,
21    /// The estimated variance.
22    estimate_var: f64,
23    /// The sum in the current window.
24    sum: DataSize,
25    /// The size of the current window.
26    current_window: Duration,
27    /// The last time the window was updated.
28    last_update: Option<Instant>,
29}
30
31impl AckedBitrateEstimator {
32    pub(super) fn new(initial_window: Duration, window: Duration) -> Self {
33        Self {
34            initial_window,
35            window,
36            estimate: None,
37            estimate_var: 50.0,
38            sum: DataSize::ZERO,
39            current_window: Duration::ZERO,
40            last_update: None,
41        }
42    }
43
44    pub(super) fn update(&mut self, receive_time: Instant, packet_size: DataSize) {
45        let window = if self.estimate.is_none() {
46            // Use the initial, larger, window at first
47            self.initial_window
48        } else {
49            self.window
50        };
51        let Some((sample_estimate, is_small_sample)) =
52            self.update_window(receive_time, packet_size, window)
53        else {
54            // No update
55            return;
56        };
57
58        let Some(estimate) = self.estimate else {
59            // This is the initial estimate, use it to initialize the estimate.
60            self.estimate = Some(sample_estimate);
61            return;
62        };
63
64        let scale = if is_small_sample && sample_estimate < estimate {
65            SMALL_SAMPLE_UNCERTAINTY
66        } else {
67            UNCERTAINTY
68        };
69
70        let sample_estimate_bps = sample_estimate.as_f64();
71        let estimate_bps = estimate.as_f64();
72        // Define the sample uncertainty as a function of how far away it is from the
73        // current estimate. With low values of uncertainty_symmetry_cap_ we add more
74        // uncertainty to increases than to decreases. For higher values we approach
75        // symmetry.
76        let sample_uncertainty =
77            scale * (estimate_bps - sample_estimate_bps).abs() / (estimate_bps.max(25_000.0));
78        let sample_var = sample_uncertainty.powf(2.0);
79
80        // Update a bayesian estimate of the rate, weighting it lower if the sample
81        // uncertainty is large.
82        // The bitrate estimate uncertainty is increased with each update to model
83        // that the bitrate changes over time.
84        let pred_bitrate_estimate_var = self.estimate_var + 5.0;
85        let mut new_estimate = (sample_var * estimate_bps
86            + pred_bitrate_estimate_var * sample_estimate_bps)
87            / (sample_var + pred_bitrate_estimate_var);
88        new_estimate = new_estimate.max(ESTIMATE_FLOOR.as_f64());
89        self.estimate = Some(Bitrate::bps(new_estimate.ceil() as u64));
90        self.estimate_var =
91            (sample_var * pred_bitrate_estimate_var) / (sample_var + pred_bitrate_estimate_var);
92    }
93
94    pub(super) fn current_estimate(&self) -> Option<Bitrate> {
95        self.estimate
96    }
97
98    fn update_window(
99        &mut self,
100        receive_time: Instant,
101        packet_size: DataSize,
102        window: Duration,
103    ) -> Option<(Bitrate, bool)> {
104        let time_moved_back = Some(receive_time) < self.last_update;
105        if time_moved_back {
106            // Time moved backwards, reset state
107            self.sum = DataSize::ZERO;
108            self.current_window = Duration::ZERO;
109            self.last_update = Some(receive_time);
110
111            return None;
112        }
113
114        if let Some(last) = self.last_update {
115            self.current_window += receive_time - last;
116            if receive_time - last >= window {
117                // No update for a while, reset estimates.
118                self.sum = DataSize::ZERO;
119                self.current_window = Duration::from_micros(
120                    self.window.as_micros() as u64 % window.as_micros() as u64,
121                );
122            }
123        }
124
125        self.last_update = Some(receive_time);
126
127        let mut estimate = None;
128
129        let mut is_small = false;
130        if self.current_window >= window {
131            is_small = self.sum < SMALL_SAMPLE_THRESHOLD;
132            estimate = Some(self.sum / window);
133            self.sum = DataSize::ZERO;
134            self.current_window -= window;
135        }
136
137        self.sum += packet_size;
138
139        estimate.map(|e| (e, false))
140    }
141}

143#[cfg(test)]
144mod test {
145    use super::*;
146
147    #[test]
148    fn test_no_estimate_before_initial_window_has_passed() {
149        let now = Instant::now();
150        let mut estimator =
151            AckedBitrateEstimator::new(Duration::from_millis(500), Duration::from_millis(150));
152
153        estimator.update(now, DataSize::bytes(950));
154        estimator.update(now + Duration::from_millis(250), DataSize::bytes(381));
155        estimator.update(now + Duration::from_millis(499), DataSize::bytes(1110));
156
157        assert!(
158            estimator.current_estimate().is_none(),
159            "AckedBitrateEstiamtor should produce no estimate before the initial window is reached"
160        );
161
162        estimator.update(now + Duration::from_millis(501), DataSize::bytes(1110));
163
164        let estimate = estimator.current_estimate().expect(
165            "After the first window of time has passed AckedBitrateEstimator should produce an estimate"
166        );
167
168        assert_eq!(
169            estimate.as_u64(),
170            39056,
171            "AckedBitrateEstiamtor should produce the correct bitrate"
172        );
173    }
174
175    #[test]
176    fn test_correct_estimate_after_initial_window() {
177        let now = Instant::now();
178        let mut estimator =
179            AckedBitrateEstimator::new(Duration::from_millis(500), Duration::from_millis(150));
180
181        estimator.update(now, DataSize::bytes(2500));
182        estimator.update(now + Duration::from_millis(250), DataSize::bytes(1392));
183        estimator.update(now + Duration::from_millis(499), DataSize::bytes(4021));
184        estimator.update(now + Duration::from_millis(500), DataSize::bytes(0));
185
186        assert!(
187            estimator.current_estimate().is_some(),
188            "After the first window of time has passed AckedBitrateEstimator should produce an estimate"
189        );
190
191        estimator.update(now + Duration::from_millis(550), DataSize::bytes(271));
192        estimator.update(now + Duration::from_millis(558), DataSize::bytes(813));
193        estimator.update(now + Duration::from_millis(648), DataSize::bytes(731));
194        /// Will not be counted, part of next window
195        estimator.update(now + Duration::from_millis(651), DataSize::bytes(900));
196
197        let estimate = estimator.current_estimate().expect(
198            "After the first window of time has passed AckedBitrateEstimator should produce an estimate"
199        );
200
201        assert_eq!(
202            estimate.as_u64(),
203            99530,
204            "AckedBitrateEstiamtor should produce the correct bitrate"
205        );
206    }
