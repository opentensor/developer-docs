8/// Generic packetizer for G711 and G722 packets.
9#[derive(Default, Debug, Copy, Clone)]
10pub struct G7xxPacketizer;
11
12impl Packetizer for G7xxPacketizer {
13    /// Payload fragments an G7xx packet across one or more byte arrays
14    fn packetize(&mut self, mtu: usize, payload: &[u8]) -> Result<Vec<Vec<u8>>, PacketError> {
15        if payload.is_empty() || mtu == 0 {
16            return Ok(vec![]);
17        }
18
19        let mut payload_data_remaining = payload.len();
20        let mut payload_data_index = 0;
21        let mut payloads = Vec::with_capacity(payload_data_remaining / mtu);
22        while payload_data_remaining > 0 {
23            let current_fragment_size = std::cmp::min(mtu, payload_data_remaining);
24            payloads.push(
25                payload[payload_data_index..payload_data_index + current_fragment_size].to_vec(),
26            );
27
28            payload_data_remaining -= current_fragment_size;
29            payload_data_index += current_fragment_size;
30        }
31
32        Ok(payloads)
33    }
34
35    fn is_marker(&mut self, data: &[u8], previous: Option<&[u8]>, last: bool) -> bool {
36        false
37    }
38}

40#[cfg(test)]
41mod test {
42    use super::*;

44    #[test]
45    fn test_g7xx_payload() -> Result<(), PacketError> {
46        let mut pck = G711Packetizer::default();
47
48        const TEST_LEN: usize = 10000;
49        const TEST_MTU: usize = 1500;
50
51        //generate random 8-bit g722 samples
52        let samples: Vec<u8> = (0..TEST_LEN).map(|_| fastrand::u8(..)).collect();
53
54        //make a copy, for packetizer input
55        let mut samples_in = vec![0u8; TEST_LEN];
56        samples_in.clone_from_slice(&samples);
57        let samples_in = samples_in.to_vec();
58
59        //split our samples into payloads
60        let payloads = pck.packetize(TEST_MTU, &samples_in)?;
61
62        let outcnt = ((TEST_LEN as f64) / (TEST_MTU as f64)).ceil() as usize;
63        assert_eq!(
64            outcnt,
65            payloads.len(),
66            "Generated {} payloads instead of {}",
67            payloads.len(),
68            outcnt
69        );
70        assert_eq!(&samples, &samples_in, "Modified input samples");
71
72        let samples_out = payloads.concat();
73        assert_eq!(&samples_out, &samples_in, "Output samples don't match");
74
75        let empty = &[];
76        let payload = &[0x90, 0x90, 0x90];
77
78        // Positive MTU, empty payload
79        let result = pck.packetize(1, empty)?;
80        assert!(result.is_empty(), "Generated payload should be empty");
81
82        // 0 MTU, small payload
83        let result = pck.packetize(0, payload)?;
84        assert_eq!(result.len(), 0, "Generated payload should be empty");
85
86        // Positive MTU, small payload
87        let result = pck.packetize(10, payload)?;
88        assert_eq!(result.len(), 1, "Generated payload should be the 1");
89
90        Ok(())
91    }
