14#[derive(Clone, PartialEq, Eq)]
15/// Holds metadata incoming RTP data.
16pub struct RtpMeta {
17    /// When this RTP packet was received.
18    pub received: Instant,
19    /// Media time translated from the RtpHeader time.
20    pub time: MediaTime,
21    /// Sequence number, extended from the RTPHeader.
22    pub seq_no: SeqNo,
23    /// The actual header.
24    pub header: RtpHeader,
25    /// Sender information from the most recent Sender Report(SR).
26    ///
27    /// If no Sender Report(SR) has been received this is [`None`].
28    pub last_sender_info: Option<SenderInfo>,
29}

40impl Depacketized {
41    pub fn first_network_time(&self) -> Instant {
42        self.meta
43            .iter()
44            .map(|m| m.received)
45            .min()
46            .expect("a depacketized to consist of at least one packet")
47    }
48
49    pub fn first_sender_info(&self) -> Option<SenderInfo> {
50        self.meta
51            .iter()
52            .min_by_key(|m| m.received)
53            .map(|m| m.last_sender_info)
54            .expect("a depacketized to consist of at least one packet")
55    }
56
57    pub fn seq_range(&self) -> RangeInclusive<SeqNo> {
58        let first = self.meta[0].seq_no;
59        let last = self.meta.last().expect("at least one element").seq_no;
60        first..=last
61    }
62
63    pub fn ext_vals(&self) -> &ExtensionValues {
64        // We use the extensions from the last packet because certain extensions, such as video
65        // orientation, are only added on the last packet to save bytes.
66        &self.meta[self.meta.len() - 1].header.ext_vals
67    }

90impl DepacketizingBuffer {
91    pub(crate) fn new(depack: CodecDepacketizer, hold_back: usize) -> Self {
92        let contiguity = match depack {
93            CodecDepacketizer::Vp8(_) => Contiguity::Vp8(Vp8Contiguity::new()),
94            CodecDepacketizer::Vp9(_) => Contiguity::Vp9(Vp9Contiguity::new()),
95            CodecDepacketizer::H264(_)
96            | CodecDepacketizer::H265(_)
97            | CodecDepacketizer::Boxed(_)
98            | CodecDepacketizer::Opus(_)
99            | CodecDepacketizer::Null(_) => Contiguity::None,
100        };
101
102        DepacketizingBuffer {
103            hold_back,
104            depack,
105            queue: VecDeque::new(),
106            segments: Vec::new(),
107            last_emitted: None,
108            max_time: None,
109            depack_cache: None,
110            contiguity,
111        }
112    }
113
114    pub fn push(&mut self, meta: RtpMeta, data: Vec<u8>) {
115        // We're not emitting samples in the wrong order. If we receive
116        // packets that are before the last emitted, we drop.
117        //
118        // As a special case, per popular demand, if hold_back is 0, we do emit
119        // out of order packets.
120        if let Some((last, _)) = self.last_emitted {
121            if meta.seq_no <= last && self.hold_back > 0 {
122                trace!("Drop before emitted: {} <= {}", meta.seq_no, last);
123                return;
124            }
125        }
126
127        // Record that latest seen max time (used for extending time to u64).
128        self.max_time = Some(if let Some(m) = self.max_time {
129            m.max(meta.time)
130        } else {
131            meta.time
132        });
133
134        match self
135            .queue
136            .binary_search_by_key(&meta.seq_no, |r| r.meta.seq_no)
137        {
138            Ok(_) => {
139                // exact same seq_no found. ignore
140                trace!("Drop exactly same packet: {}", meta.seq_no);
141            }
142            Err(i) => {
143                let head = self.depack.is_partition_head(&data);
144                let tail = self.depack.is_partition_tail(meta.header.marker, &data);
145
146                // i is insertion point to maintain order
147                let entry = Entry {
148                    meta,
149                    data,
150                    head,
151                    tail,
152                };
153                self.queue.insert(i, entry);
154            }
155        }
156    }
157
158    pub fn pop(&mut self) -> Option<Result<Depacketized, PacketError>> {
159        self.update_segments();
160
161        // println!(
162        //     "{:?} {:?}",
163        //     self.queue.iter().map(|e| e.meta.seq_no).collect::<Vec<_>>(),
164        //     self.segments
165        // );
166
167        let (start, stop) = *self.segments.first()?;
168
169        let seq = {
170            let last = self.queue.get(stop).expect("entry for stop index");
171            last.meta.seq_no
172        };
173
174        // depack ahead, even if we may not emit right away
175        let mut dep = match self.depacketize(start, stop, seq) {
176            Ok(d) => d,
177            Err(e) => {
178                // this segment cannot be decoded correctly
179                // remove from the queue and return the error
180                self.last_emitted = Some((seq, CodecExtra::None));
181                self.queue.drain(0..=stop);
182                return Some(Err(e));
183            }
184        };
185
186        // If we have contiguity of seq numbers we emit right away,
187        // Otherwise, we wait for retransmissions up to `hold_back` frames
188        // and re-evaluate contiguity based on codec specific information
189
190        let more_than_hold_back = self.segments.len() >= self.hold_back;
191        let contiguous_seq = self.is_following_last(start);
192        let wait_for_contiguity = !contiguous_seq && !more_than_hold_back;
193
194        if wait_for_contiguity {
195            // if we are not sending, cache the depacked
196            self.depack_cache = Some((start..stop, dep));
197            return None;
198        }
199
200        let (can_emit, contiguous_codec) = self.contiguity.check(&dep.codec_extra, contiguous_seq);
201        dep.contiguous = contiguous_codec;
202
203        let last = self
204            .queue
205            .get(stop)
206            .expect("entry for stop index")
207            .meta
208            .seq_no;
209
210        // We're not going to emit samples in the incorrect order, there's no point in keeping
211        // stuff before the emitted range.
212        self.queue.drain(0..=stop);
213
214        if !can_emit {
215            return None;
216        }
217
218        self.last_emitted = Some((last, dep.codec_extra));
219
220        Some(Ok(dep))
221    }

223    fn depacketize(
224        &mut self,
225        start: usize,
226        stop: usize,
227        seq: SeqNo,
228    ) -> Result<Depacketized, PacketError> {
229        if let Some(cached) = self.depack_cache.take() {
230            if cached.0 == (start..stop) {
231                trace!("depack cache hit for segment start {}", start);
232                return Ok(cached.1);
233            }
234        }
235
236        let mut data = Vec::new();
237        let mut codec_extra = CodecExtra::None;
238
239        let time = self.queue.get(start).expect("first index exist").meta.time;
240        let mut meta = Vec::with_capacity(stop - start + 1);
241
242        for entry in self.queue.range_mut(start..=stop) {
243            self.depack
244                .depacketize(&entry.data, &mut data, &mut codec_extra)?;
245            meta.push(entry.meta.clone());
246        }
247
248        Ok(Depacketized {
249            time,
250            contiguous: true, // the caller taking ownership will modify this accordingly
251            meta,
252            data,
253            codec_extra,
254        })
255    }

257    fn update_segments(&mut self) -> Option<(usize, usize)> {
258        self.segments.clear();
259
260        #[derive(Clone, Copy)]
261        struct Start {
262            index: i64,
263            time: MediaTime,
264            offset: i64,
265        }
266
267        let mut start: Option<Start> = None;
268
269        for (index, entry) in self.queue.iter().enumerate() {
270            let index = index as i64;
271            let iseq = *entry.meta.seq_no as i64;
272            let expected_seq = start.map(|s| s.offset.saturating_add(index));
273
274            let is_expected_seq = expected_seq == Some(iseq);
275            let is_same_timestamp = start.map(|s| s.time) == Some(entry.meta.time);
276            let is_defacto_tail = is_expected_seq && !is_same_timestamp;
277
278            if start.is_some() && is_defacto_tail {
279                // We found a segment that ended because the timestamp changed without
280                // a gap in the sequence number. The marker bit in the RTP packet is
281                // just indicative, this is the robust fallback.
282                let segment = (start.unwrap().index as usize, index as usize - 1);
283                self.segments.push(segment);
284                start = None;
285            }
286
287            if start.is_some() && (!is_expected_seq || !is_same_timestamp) {
288                // Not contiguous. Start looking again.
289                start = None;
290            }
291
292            // Each segment can have multiple is_partition_head() == true, record the first.
293            if start.is_none() && entry.head {
294                start = Some(Start {
295                    index,
296                    time: entry.meta.time,
297                    offset: iseq.saturating_sub(index),
298                });
299            }
300
301            if start.is_some() && entry.tail {
302                // We found a contiguous sequence of packets ending with something from
303                // the packet (like the RTP marker bit) indicating it's the tail.
304                let segment = (start.unwrap().index as usize, index as usize);
305                self.segments.push(segment);
306                start = None;
307            }
308        }
309
310        None
311    }
312
313    fn is_following_last(&self, start: usize) -> bool {
314        let Some((last, _)) = self.last_emitted else {
315            // First time we emit something.
316            return true;
317        };
318
319        // track sequence numbers are sequential
320        let mut seq = last;
321
322        // Expect all entries before start to be padding.
323        for entry in self.queue.range(0..start) {
324            if !seq.is_next(entry.meta.seq_no) {
325                // Not a sequence
326                return false;
327            }
328            // for next loop round.
329            seq = entry.meta.seq_no;
330
331            let is_padding = entry.data.is_empty() && !entry.head && !entry.tail;
332            if !is_padding {
333                return false;
334            }
335        }
336
337        let start_entry = self.queue.get(start).expect("entry for start index");
338
339        seq.is_next(start_entry.meta.seq_no)
340    }

343        self.max_time
344    }
