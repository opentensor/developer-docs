6pub(crate) fn not_happening() -> Instant {
7    const YEARS_100: Duration = Duration::from_secs(60 * 60 * 24 * 365 * 100);
8    static FUTURE: Lazy<Instant> = Lazy::new(|| Instant::now() + YEARS_100);
9    *FUTURE
10}

12// The goal here is to make a constant "beginning of time" in both Instant and SystemTime
13// that we can use as relative values for the rest of str0m.
14// This is indeed a bit dodgy, but we want str0m's internal idea of time to be completely
15// driven from the external API using `Instant`. What works against us is that Instant can't
16// represent things like UNIX EPOCH (but SystemTime can).
17static BEGINNING_OF_TIME: Lazy<(Instant, SystemTime)> = Lazy::new(|| {
18    // These two should be "frozen" the same instant. Hopefully they are not differing too much.
19    let now = Instant::now();
20    let now_sys = SystemTime::now();
21
22    // Find an Instant in the past which is up to an hour back.
23    let beginning_of_time = {
24        let mut secs = 3600;
25        loop {
26            let dur = Duration::from_secs(secs);
27            if let Some(v) = now.checked_sub(dur) {
28                break v;
29            }
30            secs -= 1;
31            if secs == 0 {
32                panic!("Failed to find a beginning of time instant");
33            }
34        }
35    };
36
37    // This might be less than 1 hour if the machine uptime is less.
38    let since_beginning_of_time = Instant::now() - beginning_of_time;
39
40    let beginning_of_time_sys = now_sys - since_beginning_of_time;
41
42    // This pair represents our "beginning of time" for the same moment.
43    (beginning_of_time, beginning_of_time_sys)
44});
45
46pub fn epoch_to_beginning() -> Duration {
47    BEGINNING_OF_TIME
48        .1
49        .duration_since(SystemTime::UNIX_EPOCH)
50        .expect("beginning of time to be after epoch")
51}

53pub(crate) fn already_happened() -> Instant {
54    BEGINNING_OF_TIME.0
55}

57pub trait InstantExt {
58    /// Convert an Instant to a Duration for unix time.
59    ///
60    /// First ever time must be "now".
61    ///
62    /// panics if `time` goes backwards, i.e. we use this for one Instant and then an earlier Instant.
63    fn to_unix_duration(&self) -> Duration;
64
65    /// Convert a unix duration to our relative Instant.
66    fn from_unix_duration(d: Duration) -> Self;
67
68    /// Convert an Instant to a Duration for ntp time.
69    fn to_ntp_duration(&self) -> Duration;
70
71    /// Convert an ntp_64 as seen in SR to an Instant.
72    fn from_ntp_64(v: u64) -> Self;
73
74    /// Convert instant to ntp_64
75    fn as_ntp_64(&self) -> u64;
76}

78// RTP spec "wallclock" uses NTP time, which starts at 1900-01-01.
79//
80// https://tools.ietf.org/html/rfc868
81//
82// 365 days * 70 years + 17 leap year days
83// (365 * 70 + 17) * 86400 = 2208988800
84const SECS_1900: u64 = 2_208_988_800;
85const MICROS_1900: u64 = SECS_1900 * 1_000_000;
86
87/// 2^32 as float.
88const F32: f64 = 4_294_967_296.0;
89
90impl InstantExt for Instant {
91    fn to_unix_duration(&self) -> Duration {
92        // This is a bit fishy. We "freeze" a moment in time for Instant and SystemTime,
93        // so we can make relative comparisons of Instant - Instant and translate that to
94        // SystemTime - unix epoch. Hopefully the error is quite small.
95        if *self < BEGINNING_OF_TIME.0 {
96            warn!("Time went backwards from beginning_of_time Instant");
97        }
98
99        let duration_since_time_0 = self.duration_since(BEGINNING_OF_TIME.0);
100        let system_time = BEGINNING_OF_TIME.1 + duration_since_time_0;
101
102        system_time
103            .duration_since(SystemTime::UNIX_EPOCH)
104            .expect("clock to go forwards from unix epoch")
105    }
106
107    fn from_unix_duration(d: Duration) -> Self {
108        let since_beginning_of_time = d.saturating_sub(epoch_to_beginning());
109        BEGINNING_OF_TIME.0 + since_beginning_of_time
110    }
111
112    fn to_ntp_duration(&self) -> Duration {
113        self.to_unix_duration() + Duration::from_micros(MICROS_1900)
114    }
115
116    fn from_ntp_64(v: u64) -> Self {
117        // https://tools.ietf.org/html/rfc3550#section-4
118        // Wallclock time (absolute date and time) is represented using the
119        // timestamp format of the Network Time Protocol (NTP), which is in
120        // seconds relative to 0h UTC on 1 January 1900 [4]. The full
121        // resolution NTP timestamp is a 64-bit unsigned fixed-point number with
122        // the integer part in the first 32 bits and the fractional part in the
123        // last 32 bits.
124        let secs_ntp = (v as f64) / F32;
125
126        // Shift to UNIX EPOCH
127        let secs_epoch = secs_ntp - SECS_1900 as f64;
128
129        // Duration not allowed to be negative
130        let secs_dur = if secs_epoch <= 0.0 {
131            Duration::ZERO
132        } else {
133            Duration::from_secs_f64(secs_epoch)
134        };
135
136        // Time in SystemTime
137        let sys = SystemTime::UNIX_EPOCH + secs_dur;
138
139        // Relative duration from our beginning of time.
140        let since_beginning_of_time = sys
141            .duration_since(BEGINNING_OF_TIME.1)
142            .unwrap_or(Duration::ZERO);
143
144        // Translate relative to Instant
145        BEGINNING_OF_TIME.0 + since_beginning_of_time
146    }
147
148    fn as_ntp_64(&self) -> u64 {
149        let since_beginning_of_time = self.duration_since(BEGINNING_OF_TIME.0);
150
151        let since_epoch = since_beginning_of_time + epoch_to_beginning();
152        let secs_epoch = since_epoch.as_secs_f64();
153
154        let secs_ntp = secs_epoch + SECS_1900 as f64;
155
156        (secs_ntp * F32) as u64
157    }
158}
