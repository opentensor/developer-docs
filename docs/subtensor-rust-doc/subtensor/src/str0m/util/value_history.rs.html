1use std::{
2    collections::VecDeque,
3    iter::Sum,
4    ops::{AddAssign, SubAssign},
5    time::{Duration, Instant},
6};
7
8/// Holds a history values of type T for up to a certain Duration, as well as the
9/// cumulated (total) value.
10#[derive(Debug)]
11pub(crate) struct ValueHistory<T> {
12    value: T,
13    history: VecDeque<(Instant, T)>,
14    max_time: Duration,
15}
16
17const DEFAULT_VALUE_HISTORY_DURATION: Duration = Duration::from_secs(1);
18
19impl<T: Default> Default for ValueHistory<T> {
20    fn default() -> Self {
21        Self {
22            value: Default::default(),
23            history: Default::default(),
24            max_time: DEFAULT_VALUE_HISTORY_DURATION,
25        }
26    }
27}
28
29impl<T> ValueHistory<T>
30where
31    T: Copy + AddAssign + SubAssign + Sum,
32{
33    /// Adds a timed value
34    /// Note: time should always monotonically increase in subsequent calls to add()
35    pub fn push(&mut self, t: Instant, v: T) {
36        self.value += v;
37        self.history.push_back((t, v));
38        self.drain(t);
39    }
40
41    /// Returns the sum of all values in the history up to max_time
42    /// This is more efficient than sum_since() as it does not need to iterate over the history
43    pub fn sum(&self) -> T {
44        self.value
45    }
46
47    fn drain(&mut self, t: Instant) -> Option<()> {
48        while t.duration_since(self.history.front()?.0) > self.max_time {
49            if let Some((_, v)) = self.history.pop_front() {
50                self.value -= v;
51            }
52        }
53
54        Some(())
55    }
56}

62
63    use super::ValueHistory;

65    #[test]
66    fn test() {
67        let now = Instant::now();
68
69        let mut h = ValueHistory {
70            value: 11,
71            max_time: Duration::from_secs(1),
72            ..Default::default()
73        };
74

77        h.push(now - Duration::from_millis(500), 22);
78        assert_eq!(h.sum(), 55);
