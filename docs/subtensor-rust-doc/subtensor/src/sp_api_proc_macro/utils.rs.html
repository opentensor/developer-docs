28/// Generates the access to the `sc_client` crate.
29pub fn generate_crate_access() -> TokenStream {
30	match crate_name("sp-api") {
31		Ok(FoundCrate::Itself) => quote!(sp_api::__private),
32		Ok(FoundCrate::Name(renamed_name)) => {
33			let renamed_name = Ident::new(&renamed_name, Span::call_site());
34			quote!(#renamed_name::__private)
35		},
36		Err(e) =>
37			if let Ok(FoundCrate::Name(name)) =
38				crate_name(&"polkadot-sdk-frame").or_else(|_| crate_name(&"frame"))
39			{
40				let path = format!("{}::deps::sp_api::__private", name);
41				let path = syn::parse_str::<syn::Path>(&path).expect("is a valid path; qed");
42				quote!( #path )
43			} else if let Ok(FoundCrate::Name(name)) = crate_name(&"polkadot-sdk") {
44				let path = format!("{}::sp_api::__private", name);
45				let path = syn::parse_str::<syn::Path>(&path).expect("is a valid path; qed");
46				quote!( #path )
47			} else {
48				let err = Error::new(Span::call_site(), e).to_compile_error();
49				quote!( #err )
50			},
51	}
52}

63pub fn return_type_extract_type(rt: &ReturnType) -> Type {

66		ReturnType::Type(_, ref ty) => *ty.clone(),

70/// Replace the `_` (wild card) parameter names in the given signature with unique identifiers.
71pub fn replace_wild_card_parameter_names(input: &mut Signature) {
72	let mut generated_pattern_counter = 0;
73	input.inputs.iter_mut().for_each(|arg| {
74		if let FnArg::Typed(arg) = arg {
75			arg.pat = Box::new(generate_unique_pattern(
76				(*arg.pat).clone(),
77				&mut generated_pattern_counter,
78			));
79		}
80	});
81}

83/// Fold the given `Signature` to make it usable on the client side.
84pub fn fold_fn_decl_for_client_side(
85	input: &mut Signature,
86	block_hash: &TokenStream,
87	crate_: &TokenStream,
88) {
89	replace_wild_card_parameter_names(input);
90
91	// Add `&self, at:& Block::Hash` as parameters to each function at the beginning.
92	input.inputs.insert(0, parse_quote!( __runtime_api_at_param__: #block_hash ));
93	input.inputs.insert(0, parse_quote!(&self));
94
95	// Wrap the output in a `Result`
96	input.output = {
97		let ty = return_type_extract_type(&input.output);
98		parse_quote!( -> std::result::Result<#ty, #crate_::ApiError> )
99	};
100}

102/// Generate an unique pattern based on the given counter, if the given pattern is a `_`.
103pub fn generate_unique_pattern(pat: Pat, counter: &mut u32) -> Pat {
104	match pat {
105		Pat::Wild(_) => {
106			let generated_name =
107				Ident::new(&format!("__runtime_api_generated_name_{}__", counter), pat.span());
108			*counter += 1;
109
110			parse_quote!( #generated_name )
111		},
112		_ => pat,
113	}
114}

116/// Allow `&self` in parameters of a method.
117pub enum AllowSelfRefInParameters {
118	/// Allows `&self` in parameters, but doesn't return it as part of the parameters.
119	YesButIgnore,
120	No,
121}
122
123/// Extracts the name, the type and `&` or ``(if it is a reference or not)
124/// for each parameter in the given function signature.
125pub fn extract_parameter_names_types_and_borrows(
126	sig: &Signature,
127	allow_self: AllowSelfRefInParameters,
128) -> Result<Vec<(Pat, Type, Option<And>)>> {
129	let mut result = Vec::new();
130	let mut generated_pattern_counter = 0;
131	for input in sig.inputs.iter() {
132		match input {
133			FnArg::Typed(arg) => {
134				let (ty, borrow) = match &*arg.ty {
135					Type::Reference(t) => ((*t.elem).clone(), Some(t.and_token)),
136					t => (t.clone(), None),
137				};
138
139				let name =
140					generate_unique_pattern((*arg.pat).clone(), &mut generated_pattern_counter);
141				result.push((name, ty, borrow));
142			},
143			FnArg::Receiver(_) if matches!(allow_self, AllowSelfRefInParameters::No) =>
144				return Err(Error::new(input.span(), "`self` parameter not supported!")),
145			FnArg::Receiver(recv) =>
146				if recv.mutability.is_some() || recv.reference.is_none() {
147					return Err(Error::new(recv.span(), "Only `&self` is supported!"))
148				},
149		}
150	}
151
152	Ok(result)
153}

160/// Extract all types that appear in signatures in the given `ImplItem`'s.
161///
162/// If a type is a reference, the inner type is extracted (without the reference).
163pub fn extract_all_signature_types(items: &[ImplItem]) -> Vec<Type> {
164	items
165		.iter()
166		.filter_map(|i| match i {
167			ImplItem::Fn(method) => Some(&method.sig),
168			_ => None,
169		})
170		.flat_map(|sig| {
171			let ret_ty = match &sig.output {
172				ReturnType::Default => None,
173				ReturnType::Type(_, ty) => Some((**ty).clone()),
174			};
175
176			sig.inputs
177				.iter()
178				.filter_map(|i| match i {
179					FnArg::Typed(arg) => Some(&arg.ty),
180					_ => None,
181				})
182				.map(|ty| match &**ty {
183					Type::Reference(t) => (*t.elem).clone(),
184					_ => (**ty).clone(),
185				})
186				.chain(ret_ty)
187		})
188		.collect()
189}

191/// Extracts the block type from a trait path.
192///
193/// It is expected that the block type is the first type in the generic arguments.
194pub fn extract_block_type_from_trait_path(trait_: &Path) -> Result<&TypePath> {
195	let span = trait_.span();
196	let generics = trait_
197		.segments
198		.last()
199		.ok_or_else(|| Error::new(span, "Empty path not supported"))?;
200
201	match &generics.arguments {
202		PathArguments::AngleBracketed(ref args) => args
203			.args
204			.first()
205			.and_then(|v| match v {
206				GenericArgument::Type(Type::Path(ref block)) => Some(block),
207				_ => None,
208			})
209			.ok_or_else(|| Error::new(args.span(), "Missing `Block` generic parameter.")),
210		PathArguments::None => {
211			let span = trait_.segments.last().as_ref().unwrap().span();
212			Err(Error::new(span, "Missing `Block` generic parameter."))
213		},
214		PathArguments::Parenthesized(_) =>
215			Err(Error::new(generics.arguments.span(), "Unexpected parentheses in path!")),
216	}
217}

219/// Should a qualified trait path be required?
220///
221/// e.g. `path::Trait` is qualified and `Trait` is not.
222pub enum RequireQualifiedTraitPath {
223	Yes,
224	No,
225}
226
227/// Extract the trait that is implemented by the given `ItemImpl`.
228pub fn extract_impl_trait(impl_: &ItemImpl, require: RequireQualifiedTraitPath) -> Result<&Path> {
229	impl_
230		.trait_
231		.as_ref()
232		.map(|v| &v.1)
233		.ok_or_else(|| Error::new(impl_.span(), "Only implementation of traits are supported!"))
234		.and_then(|p| {
235			if p.segments.len() > 1 || matches!(require, RequireQualifiedTraitPath::No) {
236				Ok(p)
237			} else {
238				Err(Error::new(
239					p.span(),
240					"The implemented trait has to be referenced with a path, \
241					e.g. `impl client::Core for Runtime`.",
242				))
243			}
244		})
245}

247/// Parse the given attribute as `API_VERSION_ATTRIBUTE`.
248pub fn parse_runtime_api_version(version: &Attribute) -> Result<u64> {
249	let version = version.parse_args::<syn::LitInt>().map_err(|_| {
250		Error::new(
251			version.span(),
252			&format!(
253				"Unexpected `{api_version}` attribute. The supported format is `{api_version}(1)`",
254				api_version = API_VERSION_ATTRIBUTE
255			),
256		)
257	})?;
258
259	version.base10_parse()
260}

267/// Extract the documentation from the provided attributes.
268pub fn get_doc_literals(attrs: &[syn::Attribute]) -> Vec<syn::Lit> {
269	use quote::ToTokens;
270	attrs
271		.iter()
272		.filter_map(|attr| {
273			let syn::Meta::NameValue(meta) = &attr.meta else { return None };
274			let Ok(lit) = syn::parse2::<syn::Lit>(meta.value.to_token_stream()) else {
275				unreachable!("non-lit doc attribute values do not exist");
276			};
277			meta.path.get_ident().filter(|ident| *ident == "doc").map(|_| lit)
278		})
279		.collect()
280}

282/// Filters all attributes except the cfg ones.
283pub fn filter_cfg_attributes(attrs: &[syn::Attribute]) -> Vec<syn::Attribute> {
284	attrs.iter().filter(|a| a.path().is_ident("cfg")).cloned().collect()
285}
