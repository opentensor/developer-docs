79pub fn ensure_ethereum_transaction<OuterOrigin>(o: OuterOrigin) -> Result<H160, &'static str>
80where
81	OuterOrigin: Into<Result<RawOrigin, OuterOrigin>>,
82{
83	match o.into() {
84		Ok(RawOrigin::EthereumTransaction(n)) => Ok(n),
85		_ => Err("bad origin: expected to be an Ethereum transaction"),
86	}
87}

106impl<T> Call<T>
107where
108	OriginFor<T>: Into<Result<RawOrigin, OriginFor<T>>>,
109	T: Send + Sync + Config,
110	T::RuntimeCall: Dispatchable<Info = DispatchInfo, PostInfo = PostDispatchInfo>,
111{
112	pub fn is_self_contained(&self) -> bool {
113		matches!(self, Call::transact { .. })
114	}
115
116	pub fn check_self_contained(&self) -> Option<Result<H160, TransactionValidityError>> {
117		if let Call::transact { transaction } = self {
118			let check = || {
119				let origin = Pallet::<T>::recover_signer(transaction).ok_or(
120					InvalidTransaction::Custom(TransactionValidationError::InvalidSignature as u8),
121				)?;
122
123				Ok(origin)
124			};
125
126			Some(check())
127		} else {
128			None
129		}
130	}
131
132	pub fn pre_dispatch_self_contained(
133		&self,
134		origin: &H160,
135		dispatch_info: &DispatchInfoOf<T::RuntimeCall>,
136		len: usize,
137	) -> Option<Result<(), TransactionValidityError>> {
138		if let Call::transact { transaction } = self {
139			if let Err(e) = CheckWeight::<T>::do_pre_dispatch(dispatch_info, len) {
140				return Some(Err(e));
141			}
142
143			Some(Pallet::<T>::validate_transaction_in_block(
144				*origin,
145				transaction,
146			))
147		} else {
148			None
149		}
150	}
151
152	pub fn validate_self_contained(
153		&self,
154		origin: &H160,
155		dispatch_info: &DispatchInfoOf<T::RuntimeCall>,
156		len: usize,
157	) -> Option<TransactionValidity> {
158		if let Call::transact { transaction } = self {
159			if let Err(e) = CheckWeight::<T>::do_validate(dispatch_info, len) {
160				return Some(Err(e));
161			}
162
163			Some(Pallet::<T>::validate_transaction_in_pool(
164				*origin,
165				transaction,
166			))
167		} else {
168			None
169		}
170	}

207	#[pallet::hooks]
208	impl<T: Config> Hooks<BlockNumberFor<T>> for Pallet<T> {
209		fn on_finalize(n: BlockNumberFor<T>) {
210			<Pallet<T>>::store_block(
211				match fp_consensus::find_pre_log(&frame_system::Pallet::<T>::digest()) {
212					Ok(_) => None,
213					Err(_) => Some(T::PostLogContent::get()),
214				},
215				U256::from(UniqueSaturatedInto::<u128>::unique_saturated_into(
216					frame_system::Pallet::<T>::block_number(),
217				)),
218			);
219			// move block hash pruning window by one block
220			let block_hash_count = T::BlockHashCount::get();
221			let to_remove = n
222				.saturating_sub(block_hash_count)
223				.saturating_sub(One::one());
224			// keep genesis hash
225			if !to_remove.is_zero() {
226				<BlockHash<T>>::remove(U256::from(
227					UniqueSaturatedInto::<u32>::unique_saturated_into(to_remove),
228				));
229			}
230			Pending::<T>::kill();
231		}
232
233		fn on_initialize(_: BlockNumberFor<T>) -> Weight {
234			let mut weight = T::SystemWeightInfo::kill_storage(1);
235
236			// If the digest contain an existing ethereum block(encoded as PreLog), If contains,
237			// execute the imported block firstly and disable transact dispatch function.
238			if let Ok(log) = fp_consensus::find_pre_log(&frame_system::Pallet::<T>::digest()) {
239				let PreLog::Block(block) = log;
240
241				for transaction in block.transactions {
242					let source = Self::recover_signer(&transaction).expect(
243						"pre-block transaction signature invalid; the block cannot be built",
244					);
245
246					Self::validate_transaction_in_block(source, &transaction).expect(
247						"pre-block transaction verification failed; the block cannot be built",
248					);
249					let (r, _) = Self::apply_validated_transaction(source, transaction)
250						.expect("pre-block apply transaction failed; the block cannot be built");
251
252					weight = weight.saturating_add(r.actual_weight.unwrap_or_default());
253				}
254			}
255			// Account for `on_finalize` weight:
256			//	- read: frame_system::Pallet::<T>::digest()
257			//	- read: frame_system::Pallet::<T>::block_number()
258			//	- write: <Pallet<T>>::store_block()
259			//	- write: <BlockHash<T>>::remove()
260			weight.saturating_add(T::DbWeight::get().reads_writes(2, 2))
261		}
262
263		fn on_runtime_upgrade() -> Weight {
264			frame_support::storage::unhashed::put::<EthereumStorageSchema>(
265				PALLET_ETHEREUM_SCHEMA,
266				&EthereumStorageSchema::V3,
267			);
268
269			T::DbWeight::get().writes(1)
270		}
271	}

273	#[pallet::call]
274	impl<T: Config> Pallet<T>
275	where
276		OriginFor<T>: Into<Result<RawOrigin, OriginFor<T>>>,
277	{
278		/// Transact an Ethereum transaction.
279		#[pallet::call_index(0)]
280		#[pallet::weight({
281			let without_base_extrinsic_weight = true;
282			<T as pallet_evm::Config>::GasWeightMapping::gas_to_weight({
283				let transaction_data: TransactionData = transaction.into();
284				transaction_data.gas_limit.unique_saturated_into()
285			}, without_base_extrinsic_weight)
286		})]
287		pub fn transact(
288			origin: OriginFor<T>,
289			transaction: Transaction,
290		) -> DispatchResultWithPostInfo {
291			let source = ensure_ethereum_transaction(origin)?;
292			// Disable transact functionality if PreLog exist.
293			assert!(
294				fp_consensus::find_pre_log(&frame_system::Pallet::<T>::digest()).is_err(),
295				"pre log already exists; block is invalid",
296			);
297
298			Self::apply_validated_transaction(source, transaction).map(|(post_info, _)| post_info)
299		}
300	}

377	fn recover_signer(transaction: &Transaction) -> Option<H160> {
378		let mut sig = [0u8; 65];
379		let mut msg = [0u8; 32];
380		match transaction {
381			Transaction::Legacy(t) => {
382				sig[0..32].copy_from_slice(&t.signature.r()[..]);
383				sig[32..64].copy_from_slice(&t.signature.s()[..]);
384				sig[64] = t.signature.standard_v();
385				msg.copy_from_slice(
386					&ethereum::LegacyTransactionMessage::from(t.clone()).hash()[..],
387				);
388			}
389			Transaction::EIP2930(t) => {
390				sig[0..32].copy_from_slice(&t.r[..]);
391				sig[32..64].copy_from_slice(&t.s[..]);
392				sig[64] = t.odd_y_parity as u8;
393				msg.copy_from_slice(
394					&ethereum::EIP2930TransactionMessage::from(t.clone()).hash()[..],
395				);
396			}
397			Transaction::EIP1559(t) => {
398				sig[0..32].copy_from_slice(&t.r[..]);
399				sig[32..64].copy_from_slice(&t.s[..]);
400				sig[64] = t.odd_y_parity as u8;
401				msg.copy_from_slice(
402					&ethereum::EIP1559TransactionMessage::from(t.clone()).hash()[..],
403				);
404			}
405		}
406		let pubkey = sp_io::crypto::secp256k1_ecdsa_recover(&sig, &msg).ok()?;
407		Some(H160::from(H256::from(sp_io::hashing::keccak_256(&pubkey))))
408	}

410	fn store_block(post_log: Option<PostLogContent>, block_number: U256) {
411		let mut transactions = Vec::new();
412		let mut statuses = Vec::new();
413		let mut receipts = Vec::new();
414		let mut logs_bloom = Bloom::default();
415		let mut cumulative_gas_used = U256::zero();
416		for (transaction, status, receipt) in Pending::<T>::get() {
417			transactions.push(transaction);
418			statuses.push(status);
419			receipts.push(receipt.clone());
420			let (logs, used_gas) = match receipt {
421				Receipt::Legacy(d) | Receipt::EIP2930(d) | Receipt::EIP1559(d) => {
422					(d.logs.clone(), d.used_gas)
423				}
424			};
425			cumulative_gas_used = used_gas;
426			Self::logs_bloom(logs, &mut logs_bloom);
427		}
428
429		let ommers = Vec::<ethereum::Header>::new();
430		let receipts_root = ethereum::util::ordered_trie_root(
431			receipts.iter().map(ethereum::EnvelopedEncodable::encode),
432		);
433		let partial_header = ethereum::PartialHeader {
434			parent_hash: if block_number > U256::zero() {
435				BlockHash::<T>::get(block_number - 1)
436			} else {
437				H256::default()
438			},
439			beneficiary: pallet_evm::Pallet::<T>::find_author(),
440			state_root: T::StateRoot::get(),
441			receipts_root,
442			logs_bloom,
443			difficulty: U256::zero(),
444			number: block_number,
445			gas_limit: T::BlockGasLimit::get(),
446			gas_used: cumulative_gas_used,
447			timestamp: T::Timestamp::now().unique_saturated_into(),
448			extra_data: Vec::new(),
449			mix_hash: H256::default(),
450			nonce: H64::default(),
451		};
452		let block = ethereum::Block::new(partial_header, transactions.clone(), ommers);
453
454		CurrentBlock::<T>::put(block.clone());
455		CurrentReceipts::<T>::put(receipts.clone());
456		CurrentTransactionStatuses::<T>::put(statuses.clone());
457		BlockHash::<T>::insert(block_number, block.header.hash());
458
459		match post_log {
460			Some(PostLogContent::BlockAndTxnHashes) => {
461				let digest = DigestItem::Consensus(
462					FRONTIER_ENGINE_ID,
463					PostLog::Hashes(fp_consensus::Hashes::from_block(block)).encode(),
464				);
465				frame_system::Pallet::<T>::deposit_log(digest);
466			}
467			Some(PostLogContent::OnlyBlockHash) => {
468				let digest = DigestItem::Consensus(
469					FRONTIER_ENGINE_ID,
470					PostLog::BlockHash(block.header.hash()).encode(),
471				);
472				frame_system::Pallet::<T>::deposit_log(digest);
473			}
474			None => { /* do nothing*/ }
475		}
476	}

478	fn logs_bloom(logs: Vec<Log>, bloom: &mut Bloom) {
479		for log in logs {
480			bloom.accrue(BloomInput::Raw(&log.address[..]));
481			for topic in log.topics {
482				bloom.accrue(BloomInput::Raw(&topic[..]));
483			}
484		}
485	}

487	// Controls that must be performed by the pool.
488	// The controls common with the State Transition Function (STF) are in
489	// the function `validate_transaction_common`.
490	fn validate_transaction_in_pool(
491		origin: H160,
492		transaction: &Transaction,
493	) -> TransactionValidity {
494		let transaction_data: TransactionData = transaction.into();
495		let transaction_nonce = transaction_data.nonce;
496		let (weight_limit, proof_size_base_cost) = Self::transaction_weight(&transaction_data);
497		let (base_fee, _) = T::FeeCalculator::min_gas_price();
498		let (who, _) = pallet_evm::Pallet::<T>::account_basic(&origin);
499
500		let _ = CheckEvmTransaction::<InvalidTransactionWrapper>::new(
501			CheckEvmTransactionConfig {
502				evm_config: T::config(),
503				block_gas_limit: T::BlockGasLimit::get(),
504				base_fee,
505				chain_id: T::ChainId::get(),
506				is_transactional: true,
507			},
508			transaction_data.clone().into(),
509			weight_limit,
510			proof_size_base_cost,
511		)
512		.validate_in_pool_for(&who)
513		.and_then(|v| v.with_chain_id())
514		.and_then(|v| v.with_base_fee())
515		.and_then(|v| v.with_balance_for(&who))
516		.map_err(|e| e.0)?;
517
518		// EIP-3607: https://eips.ethereum.org/EIPS/eip-3607
519		// Do not allow transactions for which `tx.sender` has any code deployed.
520		//
521		// This check should be done on the transaction validation (here) **and**
522		// on transaction execution, otherwise a contract tx will be included in
523		// the mempool and pollute the mempool forever.
524		if !pallet_evm::AccountCodes::<T>::get(origin).is_empty() {
525			return Err(InvalidTransaction::BadSigner.into());
526		}
527
528		let priority = match (
529			transaction_data.gas_price,
530			transaction_data.max_fee_per_gas,
531			transaction_data.max_priority_fee_per_gas,
532		) {
533			// Legacy or EIP-2930 transaction.
534			// Handle priority here. On legacy transaction everything in gas_price except
535			// the current base_fee is considered a tip to the miner and thus the priority.
536			(Some(gas_price), None, None) => {
537				gas_price.saturating_sub(base_fee).unique_saturated_into()
538			}
539			// EIP-1559 transaction without tip.
540			(None, Some(_), None) => 0,
541			// EIP-1559 transaction with tip.
542			(None, Some(max_fee_per_gas), Some(max_priority_fee_per_gas)) => max_fee_per_gas
543				.saturating_sub(base_fee)
544				.min(max_priority_fee_per_gas)
545				.unique_saturated_into(),
546			// Unreachable because already validated. Gracefully handle.
547			_ => return Err(InvalidTransaction::Payment.into()),
548		};
549
550		// The tag provides and requires must be filled correctly according to the nonce.
551		let mut builder = ValidTransactionBuilder::default()
552			.and_provides((origin, transaction_nonce))
553			.priority(priority);
554
555		// In the context of the pool, a transaction with
556		// too high a nonce is still considered valid
557		if transaction_nonce > who.nonce {
558			if let Some(prev_nonce) = transaction_nonce.checked_sub(1.into()) {
559				builder = builder.and_requires((origin, prev_nonce))
560			}
561		}
562
563		builder.build()
564	}

566	fn apply_validated_transaction(
567		source: H160,
568		transaction: Transaction,
569	) -> Result<(PostDispatchInfo, CallOrCreateInfo), DispatchErrorWithPostInfo> {
570		let (to, _, info) = Self::execute(source, &transaction, None)?;
571
572		let pending = Pending::<T>::get();
573		let transaction_hash = transaction.hash();
574		let transaction_index = pending.len() as u32;
575
576		let (reason, status, weight_info, used_gas, dest, extra_data) = match info.clone() {
577			CallOrCreateInfo::Call(info) => (
578				info.exit_reason.clone(),
579				TransactionStatus {
580					transaction_hash,
581					transaction_index,
582					from: source,
583					to,
584					contract_address: None,
585					logs: info.logs.clone(),
586					logs_bloom: {
587						let mut bloom: Bloom = Bloom::default();
588						Self::logs_bloom(info.logs, &mut bloom);
589						bloom
590					},
591				},
592				info.weight_info,
593				info.used_gas,
594				to,
595				match info.exit_reason {
596					ExitReason::Revert(_) => {
597						const LEN_START: usize = 36;
598						const MESSAGE_START: usize = 68;
599
600						let data = info.value;
601						let data_len = data.len();
602						if data_len > MESSAGE_START {
603							let message_len = U256::from(&data[LEN_START..MESSAGE_START])
604								.saturated_into::<usize>();
605							let message_end = MESSAGE_START.saturating_add(
606								message_len.min(T::ExtraDataLength::get() as usize),
607							);
608
609							if data_len >= message_end {
610								data[MESSAGE_START..message_end].to_vec()
611							} else {
612								data
613							}
614						} else {
615							data
616						}
617					}
618					_ => vec![],
619				},
620			),
621			CallOrCreateInfo::Create(info) => (
622				info.exit_reason,
623				TransactionStatus {
624					transaction_hash,
625					transaction_index,
626					from: source,
627					to,
628					contract_address: Some(info.value),
629					logs: info.logs.clone(),
630					logs_bloom: {
631						let mut bloom: Bloom = Bloom::default();
632						Self::logs_bloom(info.logs, &mut bloom);
633						bloom
634					},
635				},
636				info.weight_info,
637				info.used_gas,
638				Some(info.value),
639				Vec::new(),
640			),
641		};
642
643		let receipt = {
644			let status_code: u8 = match reason {
645				ExitReason::Succeed(_) => 1,
646				_ => 0,
647			};
648			let logs_bloom = status.logs_bloom;
649			let logs = status.clone().logs;
650			let cumulative_gas_used = if let Some((_, _, receipt)) = pending.last() {
651				match receipt {
652					Receipt::Legacy(d) | Receipt::EIP2930(d) | Receipt::EIP1559(d) => {
653						d.used_gas.saturating_add(used_gas.effective)
654					}
655				}
656			} else {
657				used_gas.effective
658			};
659			match &transaction {
660				Transaction::Legacy(_) => Receipt::Legacy(ethereum::EIP658ReceiptData {
661					status_code,
662					used_gas: cumulative_gas_used,
663					logs_bloom,
664					logs,
665				}),
666				Transaction::EIP2930(_) => Receipt::EIP2930(ethereum::EIP2930ReceiptData {
667					status_code,
668					used_gas: cumulative_gas_used,
669					logs_bloom,
670					logs,
671				}),
672				Transaction::EIP1559(_) => Receipt::EIP1559(ethereum::EIP2930ReceiptData {
673					status_code,
674					used_gas: cumulative_gas_used,
675					logs_bloom,
676					logs,
677				}),
678			}
679		};
680
681		Pending::<T>::append((transaction, status, receipt));
682
683		Self::deposit_event(Event::Executed {
684			from: source,
685			to: dest.unwrap_or_default(),
686			transaction_hash,
687			exit_reason: reason,
688			extra_data,
689		});
690
691		Ok((
692			PostDispatchInfo {
693				actual_weight: {
694					let mut gas_to_weight = T::GasWeightMapping::gas_to_weight(
695						core::cmp::max(
696							used_gas.standard.unique_saturated_into(),
697							used_gas.effective.unique_saturated_into(),
698						),
699						true,
700					);
701					if let Some(weight_info) = weight_info {
702						if let Some(proof_size_usage) = weight_info.proof_size_usage {
703							*gas_to_weight.proof_size_mut() = proof_size_usage;
704						}
705					}
706					Some(gas_to_weight)
707				},
708				pays_fee: Pays::No,
709			},
710			info,
711		))
712	}

719	/// Execute an Ethereum transaction.
720	pub fn execute(
721		from: H160,
722		transaction: &Transaction,
723		config: Option<evm::Config>,
724	) -> Result<(Option<H160>, Option<H160>, CallOrCreateInfo), DispatchErrorWithPostInfo> {
725		let transaction_data: TransactionData = transaction.into();
726		let (weight_limit, proof_size_base_cost) = Self::transaction_weight(&transaction_data);
727		let is_transactional = true;
728		let validate = false;
729
730		let (
731			input,
732			value,
733			gas_limit,
734			max_fee_per_gas,
735			max_priority_fee_per_gas,
736			nonce,
737			action,
738			access_list,
739		) = {
740			match transaction {
741				// max_fee_per_gas and max_priority_fee_per_gas in legacy and 2930 transactions is
742				// the provided gas_price.
743				Transaction::Legacy(t) => (
744					t.input.clone(),
745					t.value,
746					t.gas_limit,
747					Some(t.gas_price),
748					Some(t.gas_price),
749					Some(t.nonce),
750					t.action,
751					Vec::new(),
752				),
753				Transaction::EIP2930(t) => {
754					let access_list: Vec<(H160, Vec<H256>)> = t
755						.access_list
756						.iter()
757						.map(|item| (item.address, item.storage_keys.clone()))
758						.collect();
759					(
760						t.input.clone(),
761						t.value,
762						t.gas_limit,
763						Some(t.gas_price),
764						Some(t.gas_price),
765						Some(t.nonce),
766						t.action,
767						access_list,
768					)
769				}
770				Transaction::EIP1559(t) => {
771					let access_list: Vec<(H160, Vec<H256>)> = t
772						.access_list
773						.iter()
774						.map(|item| (item.address, item.storage_keys.clone()))
775						.collect();
776					(
777						t.input.clone(),
778						t.value,
779						t.gas_limit,
780						Some(t.max_fee_per_gas),
781						Some(t.max_priority_fee_per_gas),
782						Some(t.nonce),
783						t.action,
784						access_list,
785					)
786				}
787			}
788		};
789
790		match action {
791			ethereum::TransactionAction::Call(target) => {
792				let res = match T::Runner::call(
793					from,
794					target,
795					input,
796					value,
797					gas_limit.unique_saturated_into(),
798					max_fee_per_gas,
799					max_priority_fee_per_gas,
800					nonce,
801					access_list,
802					is_transactional,
803					validate,
804					weight_limit,
805					proof_size_base_cost,
806					config.as_ref().unwrap_or_else(|| T::config()),
807				) {
808					Ok(res) => res,
809					Err(e) => {
810						return Err(DispatchErrorWithPostInfo {
811							post_info: PostDispatchInfo {
812								actual_weight: Some(e.weight),
813								pays_fee: Pays::Yes,
814							},
815							error: e.error.into(),
816						})
817					}
818				};
819
820				Ok((Some(target), None, CallOrCreateInfo::Call(res)))
821			}
822			ethereum::TransactionAction::Create => {
823				let whitelist = pallet_evm::WhitelistedCreators::<T>::get();
824				let whitelist_disabled = pallet_evm::DisableWhitelistCheck::<T>::get();
825				let res = match T::Runner::create(
826					from,
827					input,
828					value,
829					gas_limit.unique_saturated_into(),
830					max_fee_per_gas,
831					max_priority_fee_per_gas,
832					nonce,
833					access_list,
834					whitelist,
835					whitelist_disabled,
836					is_transactional,
837					validate,
838					weight_limit,
839					proof_size_base_cost,
840					config.as_ref().unwrap_or_else(|| T::config()),
841				) {
842					Ok(res) => res,
843					Err(e) => {
844						return Err(DispatchErrorWithPostInfo {
845							post_info: PostDispatchInfo {
846								actual_weight: Some(e.weight),
847								pays_fee: Pays::Yes,
848							},
849							error: e.error.into(),
850						})
851					}
852				};
853
854				Ok((None, Some(res.value), CallOrCreateInfo::Create(res)))
855			}
856		}
857	}

859	/// Validate an Ethereum transaction already in block
860	///
861	/// This function must be called during the pre-dispatch phase
862	/// (just before applying the extrinsic).
863	pub fn validate_transaction_in_block(
864		origin: H160,
865		transaction: &Transaction,
866	) -> Result<(), TransactionValidityError> {
867		let transaction_data: TransactionData = transaction.into();
868		let (weight_limit, proof_size_base_cost) = Self::transaction_weight(&transaction_data);
869		let (base_fee, _) = T::FeeCalculator::min_gas_price();
870		let (who, _) = pallet_evm::Pallet::<T>::account_basic(&origin);
871
872		let _ = CheckEvmTransaction::<InvalidTransactionWrapper>::new(
873			CheckEvmTransactionConfig {
874				evm_config: T::config(),
875				block_gas_limit: T::BlockGasLimit::get(),
876				base_fee,
877				chain_id: T::ChainId::get(),
878				is_transactional: true,
879			},
880			transaction_data.into(),
881			weight_limit,
882			proof_size_base_cost,
883		)
884		.validate_in_block_for(&who)
885		.and_then(|v| v.with_chain_id())
886		.and_then(|v| v.with_base_fee())
887		.and_then(|v| v.with_balance_for(&who))
888		.map_err(|e| TransactionValidityError::Invalid(e.0))?;
889
890		Ok(())
891	}

893	pub fn migrate_block_v0_to_v2() -> Weight {
894		let db_weights = T::DbWeight::get();
895		let mut weight: Weight = db_weights.reads(1);
896		let item = b"CurrentBlock";
897		let block_v0 = frame_support::storage::migration::get_storage_value::<ethereum::BlockV0>(
898			Self::name().as_bytes(),
899			item,
900			&[],
901		);
902		if let Some(block_v0) = block_v0 {
903			weight = weight.saturating_add(db_weights.writes(1));
904			let block_v2: ethereum::BlockV2 = block_v0.into();
905			frame_support::storage::migration::put_storage_value::<ethereum::BlockV2>(
906				Self::name().as_bytes(),
907				item,
908				&[],
909				block_v2,
910			);
911		}
912		weight
913	}

915	#[cfg(feature = "try-runtime")]
916	pub fn pre_migrate_block_v2() -> Result<Vec<u8>, &'static str> {
917		let item = b"CurrentBlock";
918		let block_v0 = frame_support::storage::migration::get_storage_value::<ethereum::BlockV0>(
919			Self::name().as_bytes(),
920			item,
921			&[],
922		);
923		if let Some(block_v0) = block_v0 {
924			Ok((
925				block_v0.header.number,
926				block_v0.header.parent_hash,
927				block_v0.transactions.len() as u64,
928			)
929				.encode())
930		} else {
931			Ok(Vec::new())
932		}
933	}
934
935	#[cfg(feature = "try-runtime")]
936	pub fn post_migrate_block_v2(v0_data: Vec<u8>) -> Result<(), &'static str> {
937		let (v0_number, v0_parent_hash, v0_transaction_len): (U256, H256, u64) = Decode::decode(
938			&mut v0_data.as_slice(),
939		)
940		.expect("the state parameter should be something that was generated by pre_upgrade");
941		let item = b"CurrentBlock";
942		let block_v2 = frame_support::storage::migration::get_storage_value::<ethereum::BlockV2>(
943			Self::name().as_bytes(),
944			item,
945			&[],
946		);
947
948		assert!(block_v2.is_some());
949
950		let block_v2 = block_v2.unwrap();
951		assert_eq!(block_v2.header.number, v0_number);
952		assert_eq!(block_v2.header.parent_hash, v0_parent_hash);
953		assert_eq!(block_v2.transactions.len() as u64, v0_transaction_len);
954		Ok(())
955	}
