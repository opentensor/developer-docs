44/// Connection role.
45#[derive(Debug, Copy, Clone)]
46pub enum Role {
47    /// Dialer.
48    Dialer,
49
50    /// Listener.
51    Listener,
52}
53
54impl From<Role> for crate::yamux::Mode {
55    fn from(value: Role) -> Self {
56        match value {
57            Role::Dialer => crate::yamux::Mode::Client,
58            Role::Listener => crate::yamux::Mode::Server,
59        }
60    }
61}

63/// Configuration builder for [`Litep2p`](`crate::Litep2p`).
64pub struct ConfigBuilder {
65    // TCP transport configuration.
66    tcp: Option<TcpConfig>,
67
68    /// QUIC transport config.
69    quic: Option<QuicConfig>,
70
71    /// WebRTC transport config.
72    webrtc: Option<WebRtcConfig>,
73
74    /// WebSocket transport config.
75    websocket: Option<WebSocketConfig>,
76
77    /// Keypair.
78    keypair: Option<Keypair>,
79
80    /// Ping protocol config.
81    ping: Option<ping::Config>,
82
83    /// Identify protocol config.
84    identify: Option<identify::Config>,
85
86    /// Kademlia protocol config.
87    kademlia: Option<kademlia::Config>,
88
89    /// Bitswap protocol config.
90    bitswap: Option<bitswap::Config>,
91
92    /// Notification protocols.
93    notification_protocols: HashMap<ProtocolName, notification::Config>,
94
95    /// Request-response protocols.
96    request_response_protocols: HashMap<ProtocolName, request_response::Config>,
97
98    /// User protocols.
99    user_protocols: HashMap<ProtocolName, Box<dyn UserProtocol>>,
100
101    /// mDNS configuration.
102    mdns: Option<MdnsConfig>,
103
104    /// Known addresess.
105    known_addresses: Vec<(PeerId, Vec<Multiaddr>)>,
106
107    /// Executor for running futures.
108    executor: Option<Arc<dyn Executor>>,
109
110    /// Maximum number of parallel dial attempts.
111    max_parallel_dials: usize,
112}
113
114impl Default for ConfigBuilder {
115    fn default() -> Self {
116        Self::new()
117    }
118}
119
120impl ConfigBuilder {
121    /// Create empty [`ConfigBuilder`].
122    pub fn new() -> Self {
123        Self {
124            tcp: None,
125            quic: None,
126            webrtc: None,
127            websocket: None,
128            keypair: None,
129            ping: None,
130            identify: None,
131            kademlia: None,
132            bitswap: None,
133            mdns: None,
134            executor: None,
135            max_parallel_dials: MAX_PARALLEL_DIALS,
136            user_protocols: HashMap::new(),
137            notification_protocols: HashMap::new(),
138            request_response_protocols: HashMap::new(),
139            known_addresses: Vec::new(),
140        }
141    }
142
143    /// Add TCP transport configuration, enabling the transport.
144    pub fn with_tcp(mut self, config: TcpConfig) -> Self {
145        self.tcp = Some(config);
146        self
147    }
148
149    /// Add QUIC transport configuration, enabling the transport.
150    pub fn with_quic(mut self, config: QuicConfig) -> Self {
151        self.quic = Some(config);
152        self
153    }
154
155    /// Add WebRTC transport configuration, enabling the transport.
156    pub fn with_webrtc(mut self, config: WebRtcConfig) -> Self {
157        self.webrtc = Some(config);
158        self
159    }
160
161    /// Add WebSocket transport configuration, enabling the transport.
162    pub fn with_websocket(mut self, config: WebSocketConfig) -> Self {
163        self.websocket = Some(config);
164        self
165    }
166
167    /// Add keypair.
168    ///
169    /// If no keypair is specified, litep2p creates a new keypair.
170    pub fn with_keypair(mut self, keypair: Keypair) -> Self {
171        self.keypair = Some(keypair);
172        self
173    }
174
175    /// Enable notification protocol.
176    pub fn with_notification_protocol(mut self, config: notification::Config) -> Self {
177        self.notification_protocols.insert(config.protocol_name().clone(), config);
178        self
179    }
180
181    /// Enable IPFS Ping protocol.
182    pub fn with_libp2p_ping(mut self, config: ping::Config) -> Self {
183        self.ping = Some(config);
184        self
185    }
186
187    /// Enable IPFS Identify protocol.
188    pub fn with_libp2p_identify(mut self, config: identify::Config) -> Self {
189        self.identify = Some(config);
190        self
191    }
192
193    /// Enable IPFS Kademlia protocol.
194    pub fn with_libp2p_kademlia(mut self, config: kademlia::Config) -> Self {
195        self.kademlia = Some(config);
196        self
197    }
198
199    /// Enable IPFS Bitswap protocol.
200    pub fn with_libp2p_bitswap(mut self, config: bitswap::Config) -> Self {
201        self.bitswap = Some(config);
202        self
203    }
204
205    /// Enable request-response protocol.
206    pub fn with_request_response_protocol(mut self, config: request_response::Config) -> Self {
207        self.request_response_protocols.insert(config.protocol_name().clone(), config);
208        self
209    }
210
211    /// Enable user protocol.
212    pub fn with_user_protocol(mut self, protocol: Box<dyn UserProtocol>) -> Self {
213        self.user_protocols.insert(protocol.protocol(), protocol);
214        self
215    }
216
217    /// Enable mDNS for peer discoveries in the local network.
218    pub fn with_mdns(mut self, config: MdnsConfig) -> Self {
219        self.mdns = Some(config);
220        self
221    }
222
223    /// Add known address(es) for one or more peers.
224    pub fn with_known_addresses(
225        mut self,
226        addresses: impl Iterator<Item = (PeerId, Vec<Multiaddr>)>,
227    ) -> Self {
228        self.known_addresses = addresses.collect();
229        self
230    }
231
232    /// Add executor for running futures spawned by `litep2p`.
233    ///
234    /// If no executor is specified, `litep2p` defaults to calling `tokio::spawn()`.
235    pub fn with_executor(mut self, executor: Arc<dyn Executor>) -> Self {
236        self.executor = Some(executor);
237        self
238    }
239
240    /// How many addresses should litep2p attempt to dial in parallel.
241    pub fn with_max_parallel_dials(mut self, max_parallel_dials: usize) -> Self {
242        self.max_parallel_dials = max_parallel_dials;
243        self
244    }
245
246    /// Build [`Litep2pConfig`].
247    pub fn build(mut self) -> Litep2pConfig {
248        let keypair = match self.keypair {
249            Some(keypair) => keypair,
250            None => Keypair::generate(),
251        };
252
253        Litep2pConfig {
254            keypair,
255            tcp: self.tcp.take(),
256            mdns: self.mdns.take(),
257            quic: self.quic.take(),
258            webrtc: self.webrtc.take(),
259            websocket: self.websocket.take(),
260            ping: self.ping.take(),
261            identify: self.identify.take(),
262            kademlia: self.kademlia.take(),
263            bitswap: self.bitswap.take(),
264            max_parallel_dials: self.max_parallel_dials,
265            executor: self.executor.map_or(Arc::new(DefaultExecutor {}), |executor| executor),
266            user_protocols: self.user_protocols,
267            notification_protocols: self.notification_protocols,
268            request_response_protocols: self.request_response_protocols,
269            known_addresses: self.known_addresses,
270        }
271    }
272}
