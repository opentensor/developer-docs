21//! Behavior defining how futures running in the background should be executed.
22
23use std::{future::Future, pin::Pin};
24
25/// Trait which defines the interface the executor must implement.
26pub trait Executor: Send + Sync {
27    /// Start executing a future in the background.
28    fn run(&self, future: Pin<Box<dyn Future<Output = ()> + Send>>);
29
30    /// Start executing a future in the background and give the future a name;
31    fn run_with_name(&self, name: &'static str, future: Pin<Box<dyn Future<Output = ()> + Send>>);
32}
33
34/// Default executor, defaults to calling `tokio::spawn()`.
35pub(crate) struct DefaultExecutor;
36
37impl Executor for DefaultExecutor {
38    fn run(&self, future: Pin<Box<dyn Future<Output = ()> + Send>>) {
39        tokio::spawn(future);
40    }
41
42    fn run_with_name(&self, _: &'static str, future: Pin<Box<dyn Future<Output = ()> + Send>>) {
43        tokio::spawn(future);
44    }
45}

47#[cfg(test)]
48mod tests {
49    use super::*;
50    use tokio::sync::mpsc::channel;
51
52    #[tokio::test]
53    async fn run_with_name() {
54        let executor = DefaultExecutor;
55        let (tx, mut rx) = channel(1);
56
57        let sender = tx.clone();
58        executor.run(Box::pin(async move {
59            sender.send(1337usize).await.unwrap();
60        }));
61
62        executor.run_with_name(
63            "test",
64            Box::pin(async move {
65                tx.send(1337usize).await.unwrap();
66            }),
67        );
68
69        assert_eq!(rx.recv().await.unwrap(), 1337usize);
70        assert_eq!(rx.recv().await.unwrap(), 1337usize);
71    }
72}
