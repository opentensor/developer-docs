87/// Litep2p events.
88#[derive(Debug)]
89pub enum Litep2pEvent {
90    /// Connection established to peer.
91    ConnectionEstablished {
92        /// Remote peer ID.
93        peer: PeerId,
94
95        /// Endpoint.
96        endpoint: Endpoint,
97    },
98
99    /// Connection closed to remote peer.
100    ConnectionClosed {
101        /// Peer ID.
102        peer: PeerId,
103
104        /// Connection ID.
105        connection_id: ConnectionId,
106    },
107
108    /// Failed to dial peer.
109    DialFailure {
110        /// Address of the peer.
111        address: Multiaddr,
112
113        /// Dial error.
114        error: Error,
115    },
116}

121    local_peer_id: PeerId,
122
123    /// Listen addresses.
124    listen_addresses: Vec<Multiaddr>,

128
129    /// Bandwidth sink.

131}

133impl Litep2p {
134    /// Create new [`Litep2p`].
135    pub fn new(mut litep2p_config: Litep2pConfig) -> crate::Result<Litep2p> {
136        let local_peer_id = PeerId::from_public_key(&litep2p_config.keypair.public().into());
137        let bandwidth_sink = BandwidthSink::new();
138        let mut listen_addresses = vec![];
139
140        let supported_transports = Self::supported_transports(&litep2p_config);
141        let (mut transport_manager, transport_handle) = TransportManager::new(
142            litep2p_config.keypair.clone(),
143            supported_transports,
144            bandwidth_sink.clone(),
145            litep2p_config.max_parallel_dials,
146        );
147
148        // add known addresses to `TransportManager`, if any exist
149        if !litep2p_config.known_addresses.is_empty() {
150            for (peer, addresses) in litep2p_config.known_addresses {
151                transport_manager.add_known_address(peer, addresses.iter().cloned());
152            }
153        }
154
155        // start notification protocol event loops
156        for (protocol, config) in litep2p_config.notification_protocols.into_iter() {
157            tracing::debug!(
158                target: LOG_TARGET,
159                ?protocol,
160                "enable notification protocol",
161            );
162
163            let service = transport_manager.register_protocol(
164                protocol,
165                config.fallback_names.clone(),
166                config.codec,
167            );
168            let executor = Arc::clone(&litep2p_config.executor);
169            litep2p_config.executor.run(Box::pin(async move {
170                NotificationProtocol::new(service, config, executor).run().await
171            }));
172        }
173
174        // start request-response protocol event loops
175        for (protocol, config) in litep2p_config.request_response_protocols.into_iter() {
176            tracing::debug!(
177                target: LOG_TARGET,
178                ?protocol,
179                "enable request-response protocol",
180            );
181
182            let service = transport_manager.register_protocol(
183                protocol,
184                config.fallback_names.clone(),
185                config.codec,
186            );
187            litep2p_config.executor.run(Box::pin(async move {
188                RequestResponseProtocol::new(service, config).run().await
189            }));
190        }
191
192        // start user protocol event loops
193        for (protocol_name, protocol) in litep2p_config.user_protocols.into_iter() {
194            tracing::debug!(target: LOG_TARGET, protocol = ?protocol_name, "enable user protocol");
195
196            let service =
197                transport_manager.register_protocol(protocol_name, Vec::new(), protocol.codec());
198            litep2p_config.executor.run(Box::pin(async move {
199                let _ = protocol.run(service).await;
200            }));
201        }
202
203        // start ping protocol event loop if enabled
204        if let Some(ping_config) = litep2p_config.ping.take() {
205            tracing::debug!(
206                target: LOG_TARGET,
207                protocol = ?ping_config.protocol,
208                "enable ipfs ping protocol",
209            );
210
211            let service = transport_manager.register_protocol(
212                ping_config.protocol.clone(),
213                Vec::new(),
214                ping_config.codec,
215            );
216            litep2p_config.executor.run(Box::pin(async move {
217                Ping::new(service, ping_config).run().await
218            }));
219        }
220
221        // start kademlia protocol event loop if enabled
222        if let Some(kademlia_config) = litep2p_config.kademlia.take() {
223            tracing::debug!(
224                target: LOG_TARGET,
225                protocol_names = ?kademlia_config.protocol_names,
226                "enable ipfs kademlia protocol",
227            );
228
229            let main_protocol =
230                kademlia_config.protocol_names.first().expect("protocol name to exist");
231            let fallback_names = kademlia_config.protocol_names.iter().skip(1).cloned().collect();
232
233            let service = transport_manager.register_protocol(
234                main_protocol.clone(),
235                fallback_names,
236                kademlia_config.codec,
237            );
238            litep2p_config.executor.run(Box::pin(async move {
239                let _ = Kademlia::new(service, kademlia_config).run().await;
240            }));
241        }
242
243        // start identify protocol event loop if enabled
244        let mut identify_info = match litep2p_config.identify.take() {
245            None => None,
246            Some(mut identify_config) => {
247                tracing::debug!(
248                    target: LOG_TARGET,
249                    protocol = ?identify_config.protocol,
250                    "enable ipfs identify protocol",
251                );
252
253                let service = transport_manager.register_protocol(
254                    identify_config.protocol.clone(),
255                    Vec::new(),
256                    identify_config.codec,
257                );
258                identify_config.public = Some(litep2p_config.keypair.public().into());
259
260                Some((service, identify_config))
261            }
262        };
263
264        // start bitswap protocol event loop if enabled
265        if let Some(bitswap_config) = litep2p_config.bitswap.take() {
266            tracing::debug!(
267                target: LOG_TARGET,
268                protocol = ?bitswap_config.protocol,
269                "enable ipfs bitswap protocol",
270            );
271
272            let service = transport_manager.register_protocol(
273                bitswap_config.protocol.clone(),
274                Vec::new(),
275                bitswap_config.codec,
276            );
277            litep2p_config.executor.run(Box::pin(async move {
278                Bitswap::new(service, bitswap_config).run().await
279            }));
280        }
281
282        // enable tcp transport if the config exists
283        if let Some(config) = litep2p_config.tcp.take() {
284            let handle = transport_manager.transport_handle(Arc::clone(&litep2p_config.executor));
285            let (transport, transport_listen_addresses) =
286                <TcpTransport as TransportBuilder>::new(handle, config)?;
287
288            for address in transport_listen_addresses {
289                transport_manager.register_listen_address(address.clone());
290                listen_addresses.push(address.with(Protocol::P2p(
291                    Multihash::from_bytes(&local_peer_id.to_bytes()).unwrap(),
292                )));
293            }
294
295            transport_manager.register_transport(SupportedTransport::Tcp, Box::new(transport));
296        }
297
298        // enable quic transport if the config exists
299        if let Some(config) = litep2p_config.quic.take() {
300            let handle = transport_manager.transport_handle(Arc::clone(&litep2p_config.executor));
301            let (transport, transport_listen_addresses) =
302                <QuicTransport as TransportBuilder>::new(handle, config)?;
303
304            for address in transport_listen_addresses {
305                transport_manager.register_listen_address(address.clone());
306                listen_addresses.push(address.with(Protocol::P2p(
307                    Multihash::from_bytes(&local_peer_id.to_bytes()).unwrap(),
308                )));
309            }
310
311            transport_manager.register_transport(SupportedTransport::Quic, Box::new(transport));
312        }
313
314        // enable webrtc transport if the config exists
315        if let Some(config) = litep2p_config.webrtc.take() {
316            let handle = transport_manager.transport_handle(Arc::clone(&litep2p_config.executor));
317            let (transport, transport_listen_addresses) =
318                <WebRtcTransport as TransportBuilder>::new(handle, config)?;
319
320            for address in transport_listen_addresses {
321                transport_manager.register_listen_address(address.clone());
322                listen_addresses.push(address.with(Protocol::P2p(
323                    Multihash::from_bytes(&local_peer_id.to_bytes()).unwrap(),
324                )));
325            }
326
327            transport_manager.register_transport(SupportedTransport::WebRtc, Box::new(transport));
328        }
329
330        // enable websocket transport if the config exists
331        if let Some(config) = litep2p_config.websocket.take() {
332            let handle = transport_manager.transport_handle(Arc::clone(&litep2p_config.executor));
333            let (transport, transport_listen_addresses) =
334                <WebSocketTransport as TransportBuilder>::new(handle, config)?;
335
336            for address in transport_listen_addresses {
337                transport_manager.register_listen_address(address.clone());
338                listen_addresses.push(address.with(Protocol::P2p(
339                    Multihash::from_bytes(&local_peer_id.to_bytes()).unwrap(),
340                )));
341            }
342
343            transport_manager
344                .register_transport(SupportedTransport::WebSocket, Box::new(transport));
345        }
346
347        // enable mdns if the config exists
348        if let Some(config) = litep2p_config.mdns.take() {
349            let mdns = Mdns::new(transport_handle, config, listen_addresses.clone())?;
350
351            litep2p_config.executor.run(Box::pin(async move {
352                let _ = mdns.start().await;
353            }));
354        }
355
356        // if identify was enabled, give it the enabled protocols and listen addresses and start it
357        if let Some((service, mut identify_config)) = identify_info.take() {
358            identify_config.protocols = transport_manager.protocols().cloned().collect();
359            let identify = Identify::new(service, identify_config, listen_addresses.clone());
360
361            litep2p_config.executor.run(Box::pin(async move {
362                let _ = identify.run().await;
363            }));
364        }
365
366        if transport_manager.installed_transports().count() == 0 {
367            return Err(Error::Other("No transport specified".to_string()));
368        }
369
370        // verify that at least one transport is specified
371        if listen_addresses.is_empty() {
372            tracing::warn!(
373                target: LOG_TARGET,
374                "litep2p started with no listen addresses, cannot accept inbound connections",
375            );
376        }
377
378        Ok(Self {
379            local_peer_id,
380            bandwidth_sink,
381            listen_addresses,
382            transport_manager,
383        })
384    }
385
386    /// Collect supported transports before initializing the transports themselves.
387    ///
388    /// Information of the supported transports is needed to initialize protocols but
389    /// information about protocols must be known to initialize transports so the initialization
390    /// has to be split.
391    fn supported_transports(config: &Litep2pConfig) -> HashSet<SupportedTransport> {
392        let mut supported_transports = HashSet::new();
393
394        config
395            .tcp
396            .is_some()
397            .then(|| supported_transports.insert(SupportedTransport::Tcp));
398        config
399            .quic
400            .is_some()
401            .then(|| supported_transports.insert(SupportedTransport::Quic));
402        config
403            .websocket
404            .is_some()
405            .then(|| supported_transports.insert(SupportedTransport::WebSocket));
406        config
407            .webrtc
408            .is_some()
409            .then(|| supported_transports.insert(SupportedTransport::WebRtc));
410
411        supported_transports
412    }

439    /// Add one ore more known addresses for peer.
440    ///
441    /// Return value denotes how many addresses were added for the peer.
442    // Addresses belonging to disabled/unsupported transports will be ignored.
443    pub fn add_known_address(
444        &mut self,
445        peer: PeerId,
446        address: impl Iterator<Item = Multiaddr>,
447    ) -> usize {
448        self.transport_manager.add_known_address(peer, address)
449    }
450
451    /// Poll next event.
452    ///
453    /// This function must be called in order for litep2p to make progress.
454    pub async fn next_event(&mut self) -> Option<Litep2pEvent> {
455        loop {
456            match self.transport_manager.next().await? {
457                TransportEvent::ConnectionEstablished { peer, endpoint, .. } =>
458                    return Some(Litep2pEvent::ConnectionEstablished { peer, endpoint }),
459                TransportEvent::ConnectionClosed {
460                    peer,
461                    connection_id,
462                } =>
463                    return Some(Litep2pEvent::ConnectionClosed {
464                        peer,
465                        connection_id,
466                    }),
467                TransportEvent::DialFailure { address, error, .. } =>
468                    return Some(Litep2pEvent::DialFailure { address, error }),
469                _ => {}
470            }
471        }
472    }

563    #[tokio::test]
564    async fn dial_same_address_twice() {
565        let _ = tracing_subscriber::fmt()
566            .with_env_filter(tracing_subscriber::EnvFilter::from_default_env())
567            .try_init();
568
569        let (config1, _service1) = NotificationConfig::new(
570            ProtocolName::from("/notificaton/1"),
571            1337usize,
572            vec![1, 2, 3, 4],
573            Vec::new(),
574            false,
575            64,
576            64,
577            true,
578        );
579        let (config2, _service2) = NotificationConfig::new(
580            ProtocolName::from("/notificaton/2"),
581            1337usize,
582            vec![1, 2, 3, 4],
583            Vec::new(),
584            false,
585            64,
586            64,
587            true,
588        );
589        let (ping_config, _ping_event_stream) = ping::Config::default();
590
591        let config = ConfigBuilder::new()
592            .with_tcp(Default::default())
593            .with_quic(Default::default())
594            .with_notification_protocol(config1)
595            .with_notification_protocol(config2)
596            .with_libp2p_ping(ping_config)
597            .build();
598
599        let peer = PeerId::random();
600        let address = Multiaddr::empty()
601            .with(Protocol::Ip4(Ipv4Addr::new(255, 254, 253, 252)))
602            .with(Protocol::Tcp(8888))
603            .with(Protocol::P2p(
604                Multihash::from_bytes(&peer.to_bytes()).unwrap(),
605            ));
606
607        let mut litep2p = Litep2p::new(config).unwrap();
608        litep2p.dial_address(address.clone()).await.unwrap();
609        litep2p.dial_address(address.clone()).await.unwrap();
610
611        match litep2p.next_event().await {
612            Some(Litep2pEvent::DialFailure { .. }) => {}
613            _ => panic!("invalid event received"),
614        }
615
616        // verify that the second same dial was ignored and the dial failure is reported only once
617        match tokio::time::timeout(std::time::Duration::from_secs(20), litep2p.next_event()).await {
618            Err(_) => {}
619            _ => panic!("invalid event received"),
620        }
621    }
