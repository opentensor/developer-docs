41/// QUIC listener.
42pub struct QuicListener {
43    /// Listen addresses.
44    _listen_addresses: Vec<SocketAddr>,
45
46    /// Listeners.
47    listeners: Vec<Endpoint>,
48
49    /// Incoming connections.
50    incoming: FuturesUnordered<BoxFuture<'static, Option<(usize, Connecting)>>>,
51}
52
53impl QuicListener {
54    /// Create new [`QuicListener`].
55    pub fn new(
56        keypair: &Keypair,
57        addresses: Vec<Multiaddr>,
58    ) -> crate::Result<(Self, Vec<Multiaddr>)> {
59        let mut listeners: Vec<Endpoint> = Vec::new();
60        let mut listen_addresses = Vec::new();
61
62        for address in addresses.into_iter() {
63            let (listen_address, _) = Self::get_socket_address(&address)?;
64            let crypto_config = Arc::new(make_server_config(keypair).expect("to succeed"));
65            let server_config = ServerConfig::with_crypto(crypto_config);
66            let listener = Endpoint::server(server_config, listen_address).unwrap();
67
68            let listen_address = listener.local_addr()?;
69            listen_addresses.push(listen_address);
70            listeners.push(listener);
71            // );
72        }
73
74        let listen_multi_addresses = listen_addresses
75            .iter()
76            .cloned()
77            .map(|address| {
78                Multiaddr::empty()
79                    .with(Protocol::from(address.ip()))
80                    .with(Protocol::Udp(address.port()))
81                    .with(Protocol::QuicV1)
82            })
83            .collect();
84
85        Ok((
86            Self {
87                incoming: listeners
88                    .iter_mut()
89                    .enumerate()
90                    .map(|(i, listener)| {
91                        let inner = listener.clone();
92                        async move { inner.accept().await.map(|connecting| (i, connecting)) }
93                            .boxed()
94                    })
95                    .collect(),
96                listeners,
97                _listen_addresses: listen_addresses,
98            },
99            listen_multi_addresses,
100        ))
101    }
102
103    /// Extract socket address and `PeerId`, if found, from `address`.
104    pub fn get_socket_address(address: &Multiaddr) -> crate::Result<(SocketAddr, Option<PeerId>)> {
105        tracing::trace!(target: LOG_TARGET, ?address, "parse multi address");
106
107        let mut iter = address.iter();
108        let socket_address = match iter.next() {
109            Some(Protocol::Ip6(address)) => match iter.next() {
110                Some(Protocol::Udp(port)) => SocketAddr::new(IpAddr::V6(address), port),
111                protocol => {
112                    tracing::error!(
113                        target: LOG_TARGET,
114                        ?protocol,
115                        "invalid transport protocol, expected `QuicV1`",
116                    );
117                    return Err(Error::AddressError(AddressError::InvalidProtocol));
118                }
119            },
120            Some(Protocol::Ip4(address)) => match iter.next() {
121                Some(Protocol::Udp(port)) => SocketAddr::new(IpAddr::V4(address), port),
122                protocol => {
123                    tracing::error!(
124                        target: LOG_TARGET,
125                        ?protocol,
126                        "invalid transport protocol, expected `QuicV1`",
127                    );
128                    return Err(Error::AddressError(AddressError::InvalidProtocol));
129                }
130            },
131            protocol => {
132                tracing::error!(target: LOG_TARGET, ?protocol, "invalid transport protocol");
133                return Err(Error::AddressError(AddressError::InvalidProtocol));
134            }
135        };
136
137        // verify that quic exists
138        match iter.next() {
139            Some(Protocol::QuicV1) => {}
140            _ => return Err(Error::AddressError(AddressError::InvalidProtocol)),
141        }
142
143        let maybe_peer = match iter.next() {
144            Some(Protocol::P2p(multihash)) => Some(PeerId::from_multihash(multihash)?),
145            None => None,
146            protocol => {
147                tracing::error!(
148                    target: LOG_TARGET,
149                    ?protocol,
150                    "invalid protocol, expected `P2p` or `None`"
151                );
152                return Err(Error::AddressError(AddressError::InvalidProtocol));
153            }
154        };
155
156        Ok((socket_address, maybe_peer))
157    }
158}
159
160impl Stream for QuicListener {
161    type Item = Connecting;
162
163    fn poll_next(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
164        if self.incoming.is_empty() {
165            return Poll::Pending;
166        }
167
168        match futures::ready!(self.incoming.poll_next_unpin(cx)) {
169            None => Poll::Ready(None),
170            Some(None) => Poll::Ready(None),
171            Some(Some((listener, future))) => {
172                let inner = self.listeners[listener].clone();
173                self.incoming.push(
174                    async move { inner.accept().await.map(|connecting| (listener, connecting)) }
175                        .boxed(),
176                );
177
178                Poll::Ready(Some(future))
179            }
180        }
181    }
182}

255    #[tokio::test]
256    async fn one_listener() {
257        let address: Multiaddr = "/ip6/::1/udp/0/quic-v1".parse().unwrap();
258        let keypair = Keypair::generate();
259        let peer = PeerId::from_public_key(&keypair.public().into());
260        let (mut listener, listen_addresses) =
261            QuicListener::new(&keypair, vec![address.clone()]).unwrap();
262        let Some(Protocol::Udp(port)) =
263            listen_addresses.iter().next().unwrap().clone().iter().skip(1).next()
264        else {
265            panic!("invalid address");
266        };
267
268        let crypto_config =
269            Arc::new(make_client_config(&Keypair::generate(), Some(peer)).expect("to succeed"));
270        let client_config = ClientConfig::new(crypto_config);
271        let client = Endpoint::client(SocketAddr::new(IpAddr::V6(Ipv6Addr::UNSPECIFIED), 0))
272            .map_err(|error| Error::Other(error.to_string()))
273            .unwrap();
274        let connection = client
275            .connect_with(client_config, format!("[::1]:{port}").parse().unwrap(), "l")
276            .map_err(|error| Error::Other(error.to_string()))
277            .unwrap();
278
279        let (res1, res2) = tokio::join!(
280            listener.next(),
281            Box::pin(async move {
282                match connection.await {
283                    Ok(connection) => Ok(connection),
284                    Err(error) => Err(error),
285                }
286            })
287        );
288
289        assert!(res1.is_some() && res2.is_ok());
290    }
