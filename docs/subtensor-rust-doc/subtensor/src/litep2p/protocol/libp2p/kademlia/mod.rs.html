84/// Peer action.
85#[derive(Debug)]
86enum PeerAction {
87    /// Send `FIND_NODE` message to peer.
88    SendFindNode(QueryId),
89
90    /// Send `PUT_VALUE` message to peer.
91    SendPutValue(Bytes),
92}
93
94/// Peer context.
95#[derive(Default)]
96struct PeerContext {
97    /// Pending action, if any.
98    pending_actions: HashMap<SubstreamId, PeerAction>,
99}
100
101impl PeerContext {
102    /// Create new [`PeerContext`].
103    pub fn new() -> Self {
104        Self {
105            pending_actions: HashMap::new(),
106        }
107    }
108
109    /// Add pending action for peer.
110    pub fn add_pending_action(&mut self, substream_id: SubstreamId, action: PeerAction) {
111        self.pending_actions.insert(substream_id, action);
112    }
113}

164    /// Create new [`Kademlia`].
165    pub(crate) fn new(mut service: TransportService, config: Config) -> Self {
166        let local_peer_id = service.local_peer_id;
167        let local_key = Key::from(service.local_peer_id);
168        let mut routing_table = RoutingTable::new(local_key.clone());
169
170        for (peer, addresses) in config.known_peers {
171            tracing::trace!(target: LOG_TARGET, ?peer, ?addresses, "add bootstrap peer");
172
173            routing_table.add_known_peer(peer, addresses.clone(), ConnectionType::NotConnected);
174            service.add_known_address(&peer, addresses.into_iter());
175        }
176
177        Self {
178            service,
179            routing_table,
180            peers: HashMap::new(),
181            cmd_rx: config.cmd_rx,
182            store: MemoryStore::new(),
183            event_tx: config.event_tx,
184            local_key,
185            pending_dials: HashMap::new(),
186            executor: QueryExecutor::new(),
187            pending_substreams: HashMap::new(),
188            update_mode: config.update_mode,
189            validation_mode: config.validation_mode,
190            record_ttl: config.record_ttl,
191            replication_factor: config.replication_factor,
192            engine: QueryEngine::new(local_peer_id, config.replication_factor, PARALLELISM_FACTOR),
193        }
194    }

196    /// Connection established to remote peer.
197    fn on_connection_established(&mut self, peer: PeerId) -> crate::Result<()> {
198        tracing::trace!(target: LOG_TARGET, ?peer, "connection established");
199
200        match self.peers.entry(peer) {
201            Entry::Vacant(entry) => {
202                if let KBucketEntry::Occupied(entry) = self.routing_table.entry(Key::from(peer)) {
203                    entry.connection = ConnectionType::Connected;
204                }
205
206                let Some(actions) = self.pending_dials.remove(&peer) else {
207                    entry.insert(PeerContext::new());
208                    return Ok(());
209                };
210
211                // go over all pending actions, open substreams and save the state to `PeerContext`
212                // from which it will be later queried when the substream opens
213                let mut context = PeerContext::new();
214
215                for action in actions {
216                    match self.service.open_substream(peer) {
217                        Ok(substream_id) => {
218                            context.add_pending_action(substream_id, action);
219                        }
220                        Err(error) => {
221                            tracing::debug!(
222                                target: LOG_TARGET,
223                                ?peer,
224                                ?action,
225                                ?error,
226                                "connection established to peer but failed to open substream",
227                            );
228
229                            if let PeerAction::SendFindNode(query_id) = action {
230                                self.engine.register_response_failure(query_id, peer);
231                            }
232                        }
233                    }
234                }
235
236                entry.insert(context);
237                Ok(())
238            }
239            Entry::Occupied(_) => Err(Error::PeerAlreadyExists(peer)),
240        }
241    }

243    /// Disconnect peer from `Kademlia`.
244    ///
245    /// Peer is disconnected either because the substream was detected closed
246    /// or because the connection was closed.
247    ///
248    /// The peer is kept in the routing table but its connection state is set
249    /// as `NotConnected`, meaning it can be evicted from a k-bucket if another
250    /// peer that shares the bucket connects.
251    async fn disconnect_peer(&mut self, peer: PeerId, query: Option<QueryId>) {
252        tracing::trace!(target: LOG_TARGET, ?peer, ?query, "disconnect peer");
253
254        if let Some(query) = query {
255            self.engine.register_response_failure(query, peer);
256        }
257
258        if let Some(PeerContext { pending_actions }) = self.peers.remove(&peer) {
259            pending_actions.into_iter().for_each(|(_, action)| {
260                if let PeerAction::SendFindNode(query_id) = action {
261                    self.engine.register_response_failure(query_id, peer);
262                }
263            });
264        }
265
266        if let KBucketEntry::Occupied(entry) = self.routing_table.entry(Key::from(peer)) {
267            entry.connection = ConnectionType::NotConnected;
268        }
269    }

271    /// Local node opened a substream to remote node.
272    async fn on_outbound_substream(
273        &mut self,
274        peer: PeerId,
275        substream_id: SubstreamId,
276        substream: Substream,
277    ) -> crate::Result<()> {
278        tracing::trace!(
279            target: LOG_TARGET,
280            ?peer,
281            ?substream_id,
282            "outbound substream opened",
283        );
284        let _ = self.pending_substreams.remove(&substream_id);
285
286        let pending_action = &mut self
287            .peers
288            .get_mut(&peer)
289            .ok_or(Error::PeerDoesntExist(peer))?
290            .pending_actions
291            .remove(&substream_id);
292
293        match pending_action.take() {
294            None => {
295                tracing::trace!(
296                    target: LOG_TARGET,
297                    ?peer,
298                    ?substream_id,
299                    "pending action doesn't exist for peer, closing substream",
300                );
301
302                let _ = substream.close().await;
303                return Ok(());
304            }
305            Some(PeerAction::SendFindNode(query)) => {
306                match self.engine.next_peer_action(&query, &peer) {
307                    Some(QueryAction::SendMessage {
308                        query,
309                        peer,
310                        message,
311                    }) => {
312                        tracing::trace!(target: LOG_TARGET, ?peer, ?query, "start sending message to peer");
313
314                        self.executor.send_request_read_response(
315                            peer,
316                            Some(query),
317                            message,
318                            substream,
319                        );
320                    }
321                    // query finished while the substream was being opened
322                    None => {
323                        let _ = substream.close().await;
324                    }
325                    action => {
326                        tracing::warn!(target: LOG_TARGET, ?query, ?peer, ?action, "unexpected action for `FIND_NODE`");
327                        let _ = substream.close().await;
328                        debug_assert!(false);
329                    }
330                }
331            }
332            Some(PeerAction::SendPutValue(message)) => {
333                tracing::trace!(target: LOG_TARGET, ?peer, "send `PUT_VALUE` response");
334
335                self.executor.send_message(peer, message, substream);
336            }
337        }
338
339        Ok(())
340    }

342    /// Remote opened a substream to local node.
343    async fn on_inbound_substream(&mut self, peer: PeerId, substream: Substream) {
344        tracing::trace!(target: LOG_TARGET, ?peer, "inbound substream opened");
345
346        self.executor.read_message(peer, None, substream);
347    }

349    /// Update routing table if the routing table update mode was set to automatic.
350    ///
351    /// Inform user about the potential routing table, allowing them to update it manually if
352    /// the mode was set to manual.
353    async fn update_routing_table(&mut self, peers: &[KademliaPeer]) {
354        let peers: Vec<_> =
355            peers.iter().filter(|peer| peer.peer != self.service.local_peer_id).collect();
356
357        // inform user about the routing table update, regardless of what the routing table update
358        // mode is
359        let _ = self
360            .event_tx
361            .send(KademliaEvent::RoutingTableUpdate {
362                peers: peers.iter().map(|peer| peer.peer).collect::<Vec<PeerId>>(),
363            })
364            .await;
365
366        for info in peers {
367            self.service.add_known_address(&info.peer, info.addresses.iter().cloned());
368
369            if std::matches!(self.update_mode, RoutingTableUpdateMode::Automatic) {
370                self.routing_table.add_known_peer(
371                    info.peer,
372                    info.addresses.clone(),
373                    self.peers
374                        .get(&info.peer)
375                        .map_or(ConnectionType::NotConnected, |_| ConnectionType::Connected),
376                );
377            }
378        }
379    }

381    /// Handle received message.
382    async fn on_message_received(
383        &mut self,
384        peer: PeerId,
385        query_id: Option<QueryId>,
386        message: BytesMut,
387        substream: Substream,
388    ) -> crate::Result<()> {
389        tracing::trace!(target: LOG_TARGET, ?peer, ?query_id, "handle message from peer");
390
391        match KademliaMessage::from_bytes(message).ok_or(Error::InvalidData)? {
392            ref message @ KademliaMessage::FindNode {
393                ref target,
394                ref peers,
395            } => {
396                match query_id {
397                    Some(query_id) => {
398                        tracing::trace!(
399                            target: LOG_TARGET,
400                            ?peer,
401                            ?target,
402                            "handle `FIND_NODE` response",
403                        );
404
405                        // update routing table and inform user about the update
406                        self.update_routing_table(peers).await;
407                        self.engine.register_response(query_id, peer, message.clone());
408                    }
409                    None => {
410                        tracing::trace!(
411                            target: LOG_TARGET,
412                            ?peer,
413                            ?target,
414                            "handle `FIND_NODE` request",
415                        );
416
417                        let message = KademliaMessage::find_node_response(
418                            target,
419                            self.routing_table
420                                .closest(Key::from(target.clone()), self.replication_factor),
421                        );
422                        self.executor.send_message(peer, message.into(), substream);
423                    }
424                }
425            }
426            KademliaMessage::PutValue { record } => {
427                tracing::trace!(
428                    target: LOG_TARGET,
429                    ?peer,
430                    record_key = ?record.key,
431                    "handle `PUT_VALUE` message",
432                );
433
434                if let IncomingRecordValidationMode::Automatic = self.validation_mode {
435                    self.store.put(record.clone());
436                }
437
438                let _ = self.event_tx.send(KademliaEvent::IncomingRecord { record }).await;
439            }
440            ref message @ KademliaMessage::GetRecord {
441                ref key,
442                ref record,
443                ref peers,
444            } => {
445                match (query_id, key) {
446                    (Some(query_id), _) => {
447                        tracing::trace!(
448                            target: LOG_TARGET,
449                            ?peer,
450                            ?query_id,
451                            ?peers,
452                            ?record,
453                            "handle `GET_VALUE` response",
454                        );
455
456                        // update routing table and inform user about the update
457                        self.update_routing_table(peers).await;
458                        self.engine.register_response(query_id, peer, message.clone());
459                    }
460                    (None, Some(key)) => {
461                        tracing::trace!(
462                            target: LOG_TARGET,
463                            ?peer,
464                            ?key,
465                            "handle `GET_VALUE` request",
466                        );
467
468                        let value = self.store.get(key).cloned();
469                        let closest_peers = self
470                            .routing_table
471                            .closest(Key::from(key.to_vec()), self.replication_factor);
472
473                        let message = KademliaMessage::get_value_response(
474                            (*key).clone(),
475                            closest_peers,
476                            value,
477                        );
478                        self.executor.send_message(peer, message.into(), substream);
479                    }
480                    (None, None) => tracing::debug!(
481                        target: LOG_TARGET,
482                        ?peer,
483                        ?message,
484                        "both query and record key missing, unable to handle message",
485                    ),
486                }
487            }
488        }
489
490        Ok(())
491    }

493    /// Failed to open substream to remote peer.
494    async fn on_substream_open_failure(&mut self, substream_id: SubstreamId, error: Error) {
495        tracing::trace!(
496            target: LOG_TARGET,
497            ?substream_id,
498            ?error,
499            "failed to open substream"
500        );
501
502        let Some(peer) = self.pending_substreams.remove(&substream_id) else {
503            tracing::debug!(
504                target: LOG_TARGET,
505                ?substream_id,
506                "outbound substream failed for non-existent peer"
507            );
508            return;
509        };
510
511        if let Some(context) = self.peers.get_mut(&peer) {
512            let query = match context.pending_actions.remove(&substream_id) {
513                Some(PeerAction::SendFindNode(query)) => Some(query),
514                _ => None,
515            };
516
517            self.disconnect_peer(peer, query).await;
518        }
519    }

521    /// Handle dial failure.
522    fn on_dial_failure(&mut self, peer: PeerId, address: Multiaddr) {
523        tracing::trace!(target: LOG_TARGET, ?peer, ?address, "failed to dial peer");
524
525        let Some(actions) = self.pending_dials.remove(&peer) else {
526            return;
527        };
528
529        for action in actions {
530            if let PeerAction::SendFindNode(query_id) = action {
531                tracing::trace!(
532                    target: LOG_TARGET,
533                    ?peer,
534                    ?query_id,
535                    ?address,
536                    "report failure for pending query",
537                );
538
539                self.engine.register_response_failure(query_id, peer);
540            }
541        }
542    }

544    /// Handle next query action.
545    async fn on_query_action(&mut self, action: QueryAction) -> Result<(), (QueryId, PeerId)> {
546        match action {
547            QueryAction::SendMessage { query, peer, .. } => match self.service.open_substream(peer)
548            {
549                Err(_) => {
550                    tracing::trace!(target: LOG_TARGET, ?query, ?peer, "dial peer");
551
552                    match self.service.dial(&peer) {
553                        Ok(_) => match self.pending_dials.entry(peer) {
554                            Entry::Occupied(entry) => {
555                                entry.into_mut().push(PeerAction::SendFindNode(query));
556                            }
557                            Entry::Vacant(entry) => {
558                                entry.insert(vec![PeerAction::SendFindNode(query)]);
559                            }
560                        },
561                        Err(error) => {
562                            tracing::trace!(target: LOG_TARGET, ?query, ?peer, ?error, "failed to dial peer");
563                            self.engine.register_response_failure(query, peer);
564                        }
565                    }
566
567                    Ok(())
568                }
569                Ok(substream_id) => {
570                    tracing::trace!(
571                        target: LOG_TARGET,
572                        ?query,
573                        ?peer,
574                        ?substream_id,
575                        "open outbound substream for peer"
576                    );
577
578                    self.pending_substreams.insert(substream_id, peer);
579                    self.peers
580                        .entry(peer)
581                        .or_default()
582                        .pending_actions
583                        .insert(substream_id, PeerAction::SendFindNode(query));
584
585                    Ok(())
586                }
587            },
588            QueryAction::FindNodeQuerySucceeded {
589                target,
590                peers,
591                query,
592            } => {
593                tracing::debug!(
594                    target: LOG_TARGET,
595                    ?query,
596                    peer = ?target,
597                    num_peers = ?peers.len(),
598                    "`FIND_NODE` succeeded",
599                );
600
601                let _ = self
602                    .event_tx
603                    .send(KademliaEvent::FindNodeSuccess {
604                        target,
605                        query_id: query,
606                        peers: peers.into_iter().map(|info| (info.peer, info.addresses)).collect(),
607                    })
608                    .await;
609                Ok(())
610            }
611            QueryAction::PutRecordToFoundNodes { record, peers } => {
612                tracing::trace!(
613                    target: LOG_TARGET,
614                    record_key = ?record.key,
615                    num_peers = ?peers.len(),
616                    "store record to found peers",
617                );
618                let key = record.key.clone();
619                let message = KademliaMessage::put_value(record);
620
621                for peer in peers {
622                    match self.service.open_substream(peer.peer) {
623                        Ok(substream_id) => {
624                            self.pending_substreams.insert(substream_id, peer.peer);
625                            self.peers
626                                .entry(peer.peer)
627                                .or_default()
628                                .pending_actions
629                                .insert(substream_id, PeerAction::SendPutValue(message.clone()));
630                        }
631                        Err(_) => match self.service.dial(&peer.peer) {
632                            Ok(_) => match self.pending_dials.entry(peer.peer) {
633                                Entry::Occupied(entry) => {
634                                    entry
635                                        .into_mut()
636                                        .push(PeerAction::SendPutValue(message.clone()));
637                                }
638                                Entry::Vacant(entry) => {
639                                    entry.insert(vec![PeerAction::SendPutValue(message.clone())]);
640                                }
641                            },
642                            Err(error) => {
643                                tracing::debug!(
644                                    target: LOG_TARGET,
645                                    ?peer,
646                                    ?key,
647                                    ?error,
648                                    "failed to dial peer",
649                                );
650                            }
651                        },
652                    }
653                }
654
655                Ok(())
656            }
657            QueryAction::GetRecordQueryDone { query_id, records } => {
658                // Considering this gives a view of all peers and their records, some peers may have
659                // outdated records. Store only the record which is backed by most
660                // peers.
661                let now = std::time::Instant::now();
662                let rec = records
663                    .iter()
664                    .filter_map(|peer_record| {
665                        if peer_record.record.is_expired(now) {
666                            None
667                        } else {
668                            Some(&peer_record.record)
669                        }
670                    })
671                    .fold(HashMap::new(), |mut acc, rec| {
672                        *acc.entry(rec).or_insert(0) += 1;
673                        acc
674                    })
675                    .into_iter()
676                    .max_by_key(|(_, v)| *v)
677                    .map(|(k, _)| k);
678
679                if let Some(record) = rec {
680                    self.store.put(record.clone());
681                }
682
683                let _ = self
684                    .event_tx
685                    .send(KademliaEvent::GetRecordSuccess {
686                        query_id,
687                        records: RecordsType::Network(records),
688                    })
689                    .await;
690                Ok(())
691            }
692            QueryAction::QueryFailed { query } => {
693                tracing::debug!(target: LOG_TARGET, ?query, "query failed");
694
695                let _ = self.event_tx.send(KademliaEvent::QueryFailed { query_id: query }).await;
696                Ok(())
697            }
698            QueryAction::QuerySucceeded { .. } => unreachable!(),
699        }
700    }

702    /// [`Kademlia`] event loop.
703    pub async fn run(mut self) -> crate::Result<()> {
704        tracing::debug!(target: LOG_TARGET, "starting kademlia event loop");
705
706        loop {
707            // poll `QueryEngine` for next actions.
708            while let Some(action) = self.engine.next_action() {
709                if let Err((query, peer)) = self.on_query_action(action).await {
710                    self.disconnect_peer(peer, Some(query)).await;
711                }
712            }
713
714            tokio::select! {
715                event = self.service.next() => match event {
716                    Some(TransportEvent::ConnectionEstablished { peer, .. }) => {
717                        if let Err(error) = self.on_connection_established(peer) {
718                            tracing::debug!(target: LOG_TARGET, ?error, "failed to handle established connection");
719                        }
720                    }
721                    Some(TransportEvent::ConnectionClosed { peer }) => {
722                        self.disconnect_peer(peer, None).await;
723                    }
724                    Some(TransportEvent::SubstreamOpened { peer, direction, substream, .. }) => {
725                        match direction {
726                            Direction::Inbound => self.on_inbound_substream(peer, substream).await,
727                            Direction::Outbound(substream_id) => {
728                                if let Err(error) = self.on_outbound_substream(peer, substream_id, substream).await {
729                                    tracing::debug!(
730                                        target: LOG_TARGET,
731                                        ?peer,
732                                        ?substream_id,
733                                        ?error,
734                                        "failed to handle outbound substream",
735                                    );
736                                }
737                            }
738                        }
739                    },
740                    Some(TransportEvent::SubstreamOpenFailure { substream, error }) => {
741                        self.on_substream_open_failure(substream, error).await;
742                    }
743                    Some(TransportEvent::DialFailure { peer, address }) => self.on_dial_failure(peer, address),
744                    None => return Err(Error::EssentialTaskClosed),
745                },
746                context = self.executor.next() => {
747                    let QueryContext { peer, query_id, result } = context.unwrap();
748
749                    match result {
750                        QueryResult::SendSuccess { substream } => {
751                            tracing::trace!(target: LOG_TARGET, ?peer, ?query_id, "message sent to peer");
752                            let _ = substream.close().await;
753                        }
754                        QueryResult::ReadSuccess { substream, message } => {
755                            tracing::trace!(target: LOG_TARGET, ?peer, ?query_id, "message read from peer");
756
757                            if let Err(error) = self.on_message_received(peer, query_id, message, substream).await {
758                                tracing::debug!(target: LOG_TARGET, ?peer, ?error, "failed to process message");
759                            }
760                        }
761                        QueryResult::SubstreamClosed | QueryResult::Timeout => {
762                            tracing::debug!(
763                                target: LOG_TARGET,
764                                ?peer,
765                                ?query_id,
766                                ?result,
767                                "failed to read message from substream",
768                            );
769
770                            self.disconnect_peer(peer, query_id).await;
771                        }
772                    }
773                }
774                command = self.cmd_rx.recv() => {
775                    match command {
776                        Some(KademliaCommand::FindNode { peer, query_id }) => {
777                            tracing::debug!(target: LOG_TARGET, ?peer, ?query_id, "starting `FIND_NODE` query");
778
779                            self.engine.start_find_node(
780                                query_id,
781                                peer,
782                                self.routing_table.closest(Key::from(peer), self.replication_factor).into()
783                            );
784                        }
785                        Some(KademliaCommand::PutRecord { mut record, query_id }) => {
786                            tracing::debug!(target: LOG_TARGET, ?query_id, key = ?record.key, "store record to DHT");
787
788                            // For `PUT_VALUE` requests originating locally we are always the publisher.
789                            record.publisher = Some(self.local_key.clone().into_preimage());
790
791                            // Make sure TTL is set.
792                            record.expires = record.expires.or_else(|| Some(Instant::now() + self.record_ttl));
793
794                            let key = Key::new(record.key.clone());
795
796                            self.store.put(record.clone());
797
798                            self.engine.start_put_record(
799                                query_id,
800                                record,
801                                self.routing_table.closest(key, self.replication_factor).into(),
802                            );
803                        }
804                        Some(KademliaCommand::PutRecordToPeers { mut record, query_id, peers, update_local_store }) => {
805                            tracing::debug!(target: LOG_TARGET, ?query_id, key = ?record.key, "store record to DHT to specified peers");
806
807                            // Make sure TTL is set.
808                            record.expires = record.expires.or_else(|| Some(Instant::now() + self.record_ttl));
809
810                            if update_local_store {
811                                self.store.put(record.clone());
812                            }
813
814                            // Put the record to the specified peers.
815                            let peers = peers.into_iter().filter_map(|peer| {
816                                if peer == self.service.local_peer_id {
817                                    return None;
818                                }
819
820                                match self.routing_table.entry(Key::from(peer)) {
821                                    KBucketEntry::Occupied(entry) => Some(entry.clone()),
822                                    KBucketEntry::Vacant(entry) if !entry.addresses.is_empty() => Some(entry.clone()),
823                                    _ => None,
824                                }
825                            }).collect();
826
827                            self.engine.start_put_record_to_peers(
828                                query_id,
829                                record,
830                                peers,
831                            );
832                        }
833                        Some(KademliaCommand::GetRecord { key, quorum, query_id }) => {
834                            tracing::debug!(target: LOG_TARGET, ?key, "get record from DHT");
835
836                            match (self.store.get(&key), quorum) {
837                                (Some(record), Quorum::One) => {
838                                    let _ = self
839                                        .event_tx
840                                        .send(KademliaEvent::GetRecordSuccess { query_id, records: RecordsType::LocalStore(record.clone()) })
841                                        .await;
842                                }
843                                (record, _) => {
844                                    self.engine.start_get_record(
845                                        query_id,
846                                        key.clone(),
847                                        self.routing_table.closest(Key::new(key.clone()), self.replication_factor).into(),
848                                        quorum,
849                                        if record.is_some() { 1 } else { 0 },
850                                    );
851                                }
852                            }
853
854                        }
855                        Some(KademliaCommand::AddKnownPeer { peer, addresses }) => {
856                            tracing::trace!(
857                                target: LOG_TARGET,
858                                ?peer,
859                                ?addresses,
860                                "add known peer",
861                            );
862
863                            self.routing_table.add_known_peer(
864                                peer,
865                                addresses.clone(),
866                                self.peers
867                                    .get(&peer)
868                                    .map_or(ConnectionType::NotConnected, |_| ConnectionType::Connected),
869                            );
870                            self.service.add_known_address(&peer, addresses.into_iter());
871
872                        }
873                        Some(KademliaCommand::StoreRecord { mut record }) => {
874                            tracing::debug!(
875                                target: LOG_TARGET,
876                                key = ?record.key,
877                                "store record in local store",
878                            );
879
880                            // Make sure TTL is set.
881                            record.expires = record.expires.or_else(|| Some(Instant::now() + self.record_ttl));
882
883                            self.store.put(record);
884                        }
885                        None => return Err(Error::EssentialTaskClosed),
886                    }
887                },
888            }
889        }
890    }
891}

910    fn make_kademlia() -> (Kademlia, Context, TransportManager) {
911        let (manager, handle) = TransportManager::new(
912            Keypair::generate(),
913            HashSet::new(),
914            BandwidthSink::new(),
915            8usize,
916        );
917
918        let peer = PeerId::random();
919        let (transport_service, _tx) = TransportService::new(
920            peer,
921            ProtocolName::from("/kad/1"),
922            Vec::new(),
923            Default::default(),
924            handle,
925        );
926        let (event_tx, event_rx) = channel(64);
927        let (_cmd_tx, cmd_rx) = channel(64);
928
929        let config = Config {
930            protocol_names: vec![ProtocolName::from("/kad/1")],
931            known_peers: HashMap::new(),
932            codec: ProtocolCodec::UnsignedVarint(None),
933            replication_factor: 20usize,
934            update_mode: RoutingTableUpdateMode::Automatic,
935            validation_mode: IncomingRecordValidationMode::Automatic,
936            record_ttl: Duration::from_secs(36 * 60 * 60),
937            event_tx,
938            cmd_rx,
939        };
940
941        (
942            Kademlia::new(transport_service, config),
943            Context { _cmd_tx, event_rx },
944            manager,
945        )
946    }

948    #[tokio::test]
949    async fn check_get_records_update() {
950        let (mut kademlia, _context, _manager) = make_kademlia();
951
952        let key = RecordKey::from(vec![1, 2, 3]);
953        let records = vec![
954            // 2 peers backing the same record.

956                peer: PeerId::random(),
957                record: Record::new(key.clone(), vec![0x1]),
958            },
959            PeerRecord {
960                peer: PeerId::random(),
961                record: Record::new(key.clone(), vec![0x1]),

966                record: Record::new(key.clone(), vec![0x2]),

968        ];
969
970        let query_id = QueryId(1);
