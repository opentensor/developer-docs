48//! This crate provides custom de/serialization helpers to use in combination with [serde's with-annotation][with-annotation] and with the improved [`serde_as`][as-annotation]-annotation.
49//! Some common use cases are:
50//!
51//! * De/Serializing a type using the `Display` and `FromStr` traits, e.g., for `u8`, `url::Url`, or `mime::Mime`.
52//!      Check [`DisplayFromStr`] for details.
53//! * Support for arrays larger than 32 elements or using const generics.
54//!     With `serde_as` large arrays are supported, even if they are nested in other types.
55//!     `[bool; 64]`, `Option<[u8; M]>`, and `Box<[[u8; 64]; N]>` are all supported, as [this examples shows](#large-and-const-generic-arrays).
56//! * Skip serializing all empty `Option` types with [`#[skip_serializing_none]`][skip_serializing_none].
57//! * Apply a prefix to each field name of a struct, without changing the de/serialize implementations of the struct using [`with_prefix!`][].
58//! * Deserialize a comma separated list like `#hash,#tags,#are,#great` into a `Vec<String>`.
59//!      Check the documentation for [`serde_with::StringWithSeparator::<CommaSeparator, T>`][StringWithSeparator].
60//!

81//! # Examples
82//!
83//! Annotate your struct or enum to enable the custom de/serializer.
84//! The `#[serde_as]` attribute must be placed *before* the `#[derive]`.
85//!
86//! ## `DisplayFromStr`
87//!
88//! ```rust
89//! # #[cfg(feature = "macros")]
90//! # use serde::{Deserialize, Serialize};
91//! # #[cfg(feature = "macros")]
92//! # use serde_with::{serde_as, DisplayFromStr};
93//! # #[cfg(feature = "macros")]
94//! #[serde_as]
95//! # #[derive(Debug, Eq, PartialEq)]
96//! #[derive(Deserialize, Serialize)]
97//! struct Foo {
98//!     // Serialize with Display, deserialize with FromStr
99//!     #[serde_as(as = "DisplayFromStr")]
100//!     bar: u8,
101//! }
102//!
103//! # #[cfg(all(feature = "macros", feature = "json"))] {
104//! // This will serialize
105//! # let foo =
106//! Foo {bar: 12}
107//! # ;
108//!
109//! // into this JSON
110//! # let json = r#"
111//! {"bar": "12"}
112//! # "#;
113//! # assert_eq!(json.replace(" ", "").replace("\n", ""), serde_json::to_string(&foo).unwrap());
114//! # assert_eq!(foo, serde_json::from_str(&json).unwrap());
115//! # }
116//! ```
117//!
118//! ## Large and const-generic arrays
119//!
120//! serde does not support arrays with more than 32 elements or using const-generics.
121//! The `serde_as` attribute allows circumventing this restriction, even for nested types and nested arrays.
122//!
123//! ```rust
124//! # #[cfg(FALSE)] {
125//! # #[cfg(feature = "macros")]
126//! # use serde::{Deserialize, Serialize};
127//! # #[cfg(feature = "macros")]
128//! # use serde_with::serde_as;
129//! # #[cfg(feature = "macros")]
130//! #[serde_as]
131//! # #[derive(Debug, Eq, PartialEq)]
132//! #[derive(Deserialize, Serialize)]
133//! struct Arrays<const N: usize, const M: usize> {
134//!     #[serde_as(as = "[_; N]")]
135//!     constgeneric: [bool; N],
136//!
137//!     #[serde_as(as = "Box<[[_; 64]; N]>")]
138//!     nested: Box<[[u8; 64]; N]>,
139//!
140//!     #[serde_as(as = "Option<[_; M]>")]
141//!     optional: Option<[u8; M]>,
142//! }
143//!
144//! # #[cfg(all(feature = "macros", feature = "json"))] {
145//! // This allows us to serialize a struct like this
146//! let arrays: Arrays<100, 128> = Arrays {
147//!     constgeneric: [true; 100],
148//!     nested: Box::new([[111; 64]; 100]),
149//!     optional: Some([222; 128])
150//! };
151//! assert!(serde_json::to_string(&arrays).is_ok());
152//! # }
153//! # }
154//! ```
155//!
156//! ## `skip_serializing_none`
157//!
158//! This situation often occurs with JSON, but other formats also support optional fields.
159//! If many fields are optional, putting the annotations on the structs can become tedious.
160//! The `#[skip_serializing_none]` attribute must be placed *before* the `#[derive]`.
161//!
162//! ```rust
163//! # #[cfg(feature = "macros")]
164//! # use serde::{Deserialize, Serialize};
165//! # #[cfg(feature = "macros")]
166//! # use serde_with::skip_serializing_none;
167//! # #[cfg(feature = "macros")]
168//! #[skip_serializing_none]
169//! # #[derive(Debug, Eq, PartialEq)]
170//! #[derive(Deserialize, Serialize)]
171//! struct Foo {
172//!     a: Option<usize>,
173//!     b: Option<usize>,
174//!     c: Option<usize>,
175//!     d: Option<usize>,
176//!     e: Option<usize>,
177//!     f: Option<usize>,
178//!     g: Option<usize>,
179//! }
180//!
181//! # #[cfg(all(feature = "macros", feature = "json"))] {
182//! // This will serialize
183//! # let foo =
184//! Foo {a: None, b: None, c: None, d: Some(4), e: None, f: None, g: Some(7)}
185//! # ;
186//!
187//! // into this JSON
188//! # let json = r#"
189//! {"d": 4, "g": 7}
190//! # "#;
191//! # assert_eq!(json.replace(" ", "").replace("\n", ""), serde_json::to_string(&foo).unwrap());
192//! # assert_eq!(foo, serde_json::from_str(&json).unwrap());
193//! # }
194//! ```
195//!
196//! ## Advanced `serde_as` usage
197//!
198//! This example is mainly supposed to highlight the flexibility of the `serde_as`-annotation compared to [serde's with-annotation][with-annotation].
199//! More details about `serde_as` can be found in the [user guide][].
200//!
201//! ```rust
202//! # #[cfg(all(feature = "macros", feature = "hex"))]
203//! # use {
204//! #     serde::{Deserialize, Serialize},
205//! #     serde_with::{serde_as, DisplayFromStr, DurationSeconds, hex::Hex},
206//! #     std::time::Duration,
207//! #     std::collections::BTreeMap,
208//! # };
209//! # #[cfg(all(feature = "macros", feature = "hex"))]
210//! #[serde_as]
211//! # #[derive(Debug, Eq, PartialEq)]
212//! #[derive(Deserialize, Serialize)]
213//! struct Foo {
214//!      // Serialize them into a list of number as seconds
215//!      #[serde_as(as = "Vec<DurationSeconds>")]
216//!      durations: Vec<Duration>,
217//!      // We can treat a Vec like a map with duplicates.
218//!      // JSON only allows string keys, so convert i32 to strings
219//!      // The bytes will be hex encoded
220//!      #[serde_as(as = "BTreeMap<DisplayFromStr, Hex>")]
221//!      bytes: Vec<(i32, Vec<u8>)>,
222//! }
223//!
224//! # #[cfg(all(feature = "macros", feature = "json", feature = "hex"))] {
225//! // This will serialize
226//! # let foo =
227//! Foo {
228//!     durations: vec![Duration::new(5, 0), Duration::new(3600, 0), Duration::new(0, 0)],
229//!     bytes: vec![
230//!         (1, vec![0, 1, 2]),
231//!         (-100, vec![100, 200, 255]),
232//!         (1, vec![0, 111, 222]),
233//!     ],
234//! }
235//! # ;
236//!
237//! // into this JSON
238//! # let json = r#"
239//! {
240//!     "durations": [5, 3600, 0],
241//!     "bytes": {
242//!         "1": "000102",
243//!         "-100": "64c8ff",
244//!         "1": "006fde"
245//!     }
246//! }
247//! # "#;
248//! # assert_eq!(json.replace(" ", "").replace("\n", ""), serde_json::to_string(&foo).unwrap());
249//! # assert_eq!(foo, serde_json::from_str(&json).unwrap());
250//! # }
251//! ```

362/// Adapter to convert from `serde_as` to the serde traits.
363///
364/// The `As` type adapter allows using types which implement [`DeserializeAs`] or [`SerializeAs`] in place of serde's with-annotation.
365/// The with-annotation allows running custom code when de/serializing, however it is quite inflexible.
366/// The traits [`DeserializeAs`]/[`SerializeAs`] are more flexible, as they allow composition and nesting of types to create more complex de/serialization behavior.
367/// However, they are not directly compatible with serde, as they are not provided by serde.
368/// The `As` type adapter makes them compatible, by forwarding the function calls to `serialize`/`deserialize` to the corresponding functions `serialize_as` and `deserialize_as`.
369///
370/// It is not required to use this type directly.
371/// Instead, it is highly encouraged to use the [`#[serde_as]`][serde_as] attribute since it includes further usability improvements.
372/// If the use of the use of the proc-macro is not acceptable, then `As` can be used directly with serde.
373///
374/// ```rust
375/// # use serde::{Deserialize, Serialize};
376/// # use serde_with::{As, DisplayFromStr};
377/// #
378/// #[derive(Deserialize, Serialize)]
379/// # struct S {
380/// // Serialize numbers as sequence of strings, using Display and FromStr
381/// #[serde(with = "As::<Vec<DisplayFromStr>>")]
382/// field: Vec<u8>,
383/// # }
384/// ```
385/// If the normal `Deserialize`/`Serialize` traits should be used, the placeholder type [`Same`] can be used.
386/// It implements [`DeserializeAs`][]/[`SerializeAs`][], when the underlying type implements `Deserialize`/`Serialize`.
387///
388/// ```rust
389/// # use serde::{Deserialize, Serialize};
390/// # use serde_with::{As, DisplayFromStr, Same};
391/// # use std::collections::BTreeMap;
392/// #
393/// #[derive(Deserialize, Serialize)]
394/// # struct S {
395/// // Serialize map, turn keys into strings but keep type of value
396/// #[serde(with = "As::<BTreeMap<DisplayFromStr, Same>>")]
397/// field: BTreeMap<u8, i32>,
398/// # }
399/// ```
400///
401/// [serde_as]: https://docs.rs/serde_with/2.0.0/serde_with/attr.serde_as.html
402pub struct As<T: ?Sized>(PhantomData<T>);
403
404impl<T: ?Sized> As<T> {
405    /// Serialize type `T` using [`SerializeAs`][]
406    ///
407    /// The function signature is compatible with [serde's with-annotation][with-annotation].
408    ///
409    /// [with-annotation]: https://serde.rs/field-attrs.html#with
410    pub fn serialize<S, I>(value: &I, serializer: S) -> Result<S::Ok, S::Error>
411    where
412        S: Serializer,
413        T: SerializeAs<I>,
414        I: ?Sized,
415    {
416        T::serialize_as(value, serializer)
417    }
418
419    /// Deserialize type `T` using [`DeserializeAs`][]
420    ///
421    /// The function signature is compatible with [serde's with-annotation][with-annotation].
422    ///
423    /// [with-annotation]: https://serde.rs/field-attrs.html#with
424    pub fn deserialize<'de, D, I>(deserializer: D) -> Result<I, D::Error>
425    where
426        T: DeserializeAs<'de, I>,
427        D: Deserializer<'de>,
428    {
429        T::deserialize_as(deserializer)
430    }
431}

440/// De/Serialize using [`Display`] and [`FromStr`] implementation
441///
442/// This allows deserializing a string as a number.
443/// It can be very useful for serialization formats like JSON, which do not support integer
444/// numbers and have to resort to strings to represent them.
445///
446/// Another use case is types with [`Display`] and [`FromStr`] implementations, but without serde
447/// support, which can be found in some crates.
448///
449/// If you control the type you want to de/serialize, you can instead use the two derive macros, [`SerializeDisplay`] and [`DeserializeFromStr`].
450/// They properly implement the traits [`Serialize`] and [`Deserialize`] such that user of the type no longer have to use the `serde_as` system.
451///
452/// # Examples
453///
454/// ```rust
455/// # #[cfg(feature = "macros")] {

460/// #[serde_as]
461/// #[derive(Deserialize, Serialize)]
462/// struct A {
463///     #[serde_as(as = "DisplayFromStr")]
464///     mime: mime::Mime,
465///     #[serde_as(as = "DisplayFromStr")]
466///     number: u32,
467/// }
468///
469/// let v: A = serde_json::from_value(json!({
470///     "mime": "text/plain",
471///     "number": "159",

474/// assert_eq!(159, v.number);
475///
476/// let x = A {
477///     mime: mime::STAR_STAR,
478///     number: 777,
479/// };
480/// assert_eq!(json!({ "mime": "*/*", "number": "777" }), serde_json::to_value(&x).unwrap());
481/// # }
482/// ```

484/// [`Display`]: std::fmt::Display

529/// Deserialize value and return [`Default`] on error
530///
531/// The main use case is ignoring error while deserializing.
532/// Instead of erroring, it simply deserializes the [`Default`] variant of the type.
533/// It is not possible to find the error location, i.e., which field had a deserialization error, with this method.
534/// During serialization this wrapper does nothing.
535/// The serialization behavior of the underlying type is preserved.
536/// The type must implement [`Default`] for this conversion to work.
537///
538/// # Examples
539///
540/// ```
541/// # #[cfg(feature = "macros")] {
542/// # use serde::Deserialize;
543/// # use serde_with::{serde_as, DefaultOnError};
544/// #
545/// #[serde_as]
546/// #[derive(Deserialize, Debug)]
547/// struct A {
548///     #[serde_as(deserialize_as = "DefaultOnError")]
549///     value: u32,
550/// }
551///
552/// let a: A = serde_json::from_str(r#"{"value": 123}"#).unwrap();
553/// assert_eq!(123, a.value);
554///
555/// // null is of invalid type
556/// let a: A = serde_json::from_str(r#"{"value": null}"#).unwrap();
557/// assert_eq!(0, a.value);
558///
559/// // String is of invalid type
560/// let a: A = serde_json::from_str(r#"{"value": "123"}"#).unwrap();
561/// assert_eq!(0, a.value);
562///
563/// // Map is of invalid type
564/// let a: A = dbg!(serde_json::from_str(r#"{"value": {}}"#)).unwrap();
565/// assert_eq!(0, a.value);
566///
567/// // Missing entries still cause errors
568/// assert!(serde_json::from_str::<A>(r#"{  }"#).is_err());
569/// # }
570/// ```
571///
572/// Deserializing missing values can be supported by adding the `default` field attribute:
573///
574/// ```
575/// # #[cfg(feature = "macros")] {
576/// # use serde::Deserialize;
577/// # use serde_with::{serde_as, DefaultOnError};
578/// #
579/// #[serde_as]
580/// #[derive(Deserialize)]
581/// struct B {
582///     #[serde_as(deserialize_as = "DefaultOnError")]
583///     #[serde(default)]
584///     value: u32,
585/// }
586///
587/// let b: B = serde_json::from_str(r#"{  }"#).unwrap();
588/// assert_eq!(0, b.value);
589/// # }
590/// ```
591///
592/// `DefaultOnError` can be combined with other conversion methods.
593/// In this example, we deserialize a `Vec`, each element is deserialized from a string.
594/// If the string does not parse as a number, then we get the default value of 0.
595///
596/// ```rust
597/// # #[cfg(feature = "macros")] {
598/// # use serde::{Deserialize, Serialize};
599/// # use serde_json::json;
600/// # use serde_with::{serde_as, DefaultOnError, DisplayFromStr};
601/// #
602/// #[serde_as]
603/// #[derive(Serialize, Deserialize)]
604/// struct C {
605///     #[serde_as(as = "Vec<DefaultOnError<DisplayFromStr>>")]
606///     value: Vec<u32>,
607/// };
608///
609/// let c: C = serde_json::from_value(json!({
610///     "value": ["1", "2", "a3", "", {}, "6"]
611/// })).unwrap();
612/// assert_eq!(vec![1, 2, 0, 0, 0, 6], c.value);
613/// # }
614/// ```
615#[cfg(feature = "alloc")]
616pub struct DefaultOnError<T = Same>(PhantomData<T>);

722/// De/Serialize Durations as number of seconds.
723///
724/// De/serialize durations as number of seconds with subsecond precision.
725/// Subsecond precision is *only* supported for [`DurationSecondsWithFrac`], but not for [`DurationSeconds`].
726/// You can configure the serialization format between integers, floats, and stringified numbers with the `FORMAT` specifier and configure the deserialization with the `STRICTNESS` specifier.
727///
728/// The `STRICTNESS` specifier can either be [`formats::Strict`] or [`formats::Flexible`] and defaults to [`formats::Strict`].
729/// [`formats::Strict`] means that deserialization only supports the type given in `FORMAT`, e.g., if `FORMAT` is `u64` deserialization from a `f64` will error.
730/// [`formats::Flexible`] means that deserialization will perform a best effort to extract the correct duration and allows deserialization from any type.
731/// For example, deserializing `DurationSeconds<f64, Flexible>` will discard any subsecond precision during deserialization from `f64` and will parse a `String` as an integer number.
732///
733/// This type also supports [`chrono::Duration`] with the `chrono_0_4`-[feature flag].
734/// This type also supports [`time::Duration`][::time_0_3::Duration] with the `time_0_3`-[feature flag].
735///
736/// | Duration Type         | Converter                 | Available `FORMAT`s    |
737/// | --------------------- | ------------------------- | ---------------------- |
738/// | `std::time::Duration` | `DurationSeconds`         | `u64`, `f64`, `String` |
739/// | `std::time::Duration` | `DurationSecondsWithFrac` | `f64`, `String`        |
740/// | `chrono::Duration`    | `DurationSeconds`         | `i64`, `f64`, `String` |
741/// | `chrono::Duration`    | `DurationSecondsWithFrac` | `f64`, `String`        |
742/// | `time::Duration`      | `DurationSeconds`         | `i64`, `f64`, `String` |
743/// | `time::Duration`      | `DurationSecondsWithFrac` | `f64`, `String`        |
744///
745/// # Examples

759///     d_u64: Duration,

762///     #[serde_as(as = "DurationSeconds<String>")]
763///     d_string: Duration,

1374/// Optimized handling of owned and borrowed byte representations.
1375///
1376/// Serialization of byte sequences like `&[u8]` or `Vec<u8>` is quite inefficient since each value will be serialized individually.
1377/// This converter type optimizes the serialization and deserialization.
1378///
1379/// This is a port of the [`serde_bytes`] crate making it compatible with the `serde_as`-annotation, which allows it to be used in more cases than provided by [`serde_bytes`].
1380///
1381/// The type provides de/serialization for these types:
1382///
1383/// * `[u8; N]`, not possible using `serde_bytes`
1384/// * `&[u8; N]`, not possible using `serde_bytes`
1385/// * `&[u8]`
1386/// * `Box<[u8; N]>`, not possible using `serde_bytes`
1387/// * `Box<[u8]>`
1388/// * `Vec<u8>`
1389/// * `Cow<'_, [u8]>`
1390/// * `Cow<'_, [u8; N]>`, not possible using `serde_bytes`

1491/// #     .new_line("\n".into());
1492/// assert_eq!(expected, ron::ser::to_string_pretty(&value, pretty_config).unwrap());
1493/// // RON doesn't support borrowed deserialization of byte arrays
1494/// # }
1495/// ```

1497/// ## Alternative to [`BytesOrString`]
1498///
1499/// The [`Bytes`] can replace [`BytesOrString`].
1500/// [`Bytes`] is implemented for more types, which makes it better.
1501/// The serialization behavior of [`Bytes`] differs from [`BytesOrString`], therefore only `deserialize_as` should be used.

1505/// # use serde::Deserialize;

1521///     "from_bytes": [70,111,111,45,66,97,114],

1525/// // can be deserialized ...

1527///     from_bytes: b"Foo-Bar".to_vec(),
1528///     from_str: b"Foo-Bar".to_vec(),
1529/// };
1530/// assert_eq!(test, serde_json::from_value(j).unwrap());
1531///
1532/// // and serialization will always be a byte sequence
1533/// # assert_eq!(json!(
1534/// {
1535///     "from_bytes": [70,111,111,45,66,97,114],
1536///     "from_str": [70,111,111,45,66,97,114],
1537/// }
1538/// # ), serde_json::to_value(&test).unwrap());
1539/// # }
1540/// ```

1543/// Deserialize one or many elements
1544///
1545/// Sometimes it is desirable to have a shortcut in writing 1-element lists in a config file.
1546/// Usually, this is done by either writing a list or the list element itself.
1547/// This distinction is not semantically important on the Rust side, thus both forms should deserialize into the same `Vec`.
1548///
1549/// The `OneOrMany` adapter achieves exactly this use case.
1550/// The serialization behavior can be tweaked to either always serialize as a list using `PreferMany` or to serialize as the inner element if possible using `PreferOne`.
1551/// By default, `PreferOne` is assumed, which can also be omitted like `OneOrMany<_>`.
1552///
1553/// # Examples
1554///
1555/// ```rust
1556/// # #[cfg(feature = "macros")] {
1557/// # use serde::Deserialize;
1558/// # use serde_json::json;
1559/// # use serde_with::{serde_as, OneOrMany};
1560/// # use serde_with::formats::{PreferOne, PreferMany};
1561/// #
1562/// #[serde_as]
1563/// # #[derive(Debug, PartialEq)]
1564/// #[derive(Deserialize, serde::Serialize)]
1565/// struct Data {
1566///     #[serde_as(deserialize_as = "OneOrMany<_, PreferOne>")]
1567///     countries: Vec<String>,
1568///     #[serde_as(deserialize_as = "OneOrMany<_, PreferMany>")]
1569///     cities: Vec<String>,
1570/// }
1571///
1572/// // The adapter allows deserializing a `Vec` from either
1573/// // a single element
1574/// let j = json!({
1575///     "countries": "Spain",
1576///     "cities": "Berlin",
1577/// });
1578/// assert!(serde_json::from_value::<Data>(j).is_ok());
1579///
1580/// // or from a list.
1581/// let j = json!({
1582///     "countries": ["Germany", "France"],
1583///     "cities": ["Amsterdam"],
1584/// });
1585/// assert!(serde_json::from_value::<Data>(j).is_ok());
1586///
1587/// // For serialization you can choose how a single element should be encoded.
1588/// // Either directly, with `PreferOne` (default), or as a list with `PreferMany`.
1589/// let data = Data {
1590///     countries: vec!["Spain".to_string()],
1591///     cities: vec!["Berlin".to_string()],
1592/// };
1593/// let j = json!({
1594///     "countries": "Spain",
1595///     "cities": ["Berlin"],
1596/// });
1597/// assert_eq!(data, serde_json::from_value(j).unwrap());
1598/// # }
1599/// ```
1600#[cfg(feature = "alloc")]
1601pub struct OneOrMany<T, FORMAT: formats::Format = formats::PreferOne>(PhantomData<(T, FORMAT)>);

1603/// Try multiple deserialization options until one succeeds.
1604///
1605/// This adapter allows you to specify a list of deserialization options.
1606/// They are tried in order and the first one working is applied.
1607/// Serialization always picks the first option.
1608///
1609/// `PickFirst` has one type parameter which must be instantiated with a tuple of two, three, or four elements.
1610/// For example, `PickFirst<(_, DisplayFromStr)>` on a field of type `u32` allows deserializing from a number or from a string via the `FromStr` trait.
1611/// The value will be serialized as a number, since that is what the first type `_` indicates.
1612///
1613/// # Examples
1614///
1615/// Deserialize a number from either a number or a string.
1616///
1617/// ```rust
1618/// # #[cfg(feature = "macros")] {
1619/// # use serde::{Deserialize, Serialize};
1620/// # use serde_json::json;
1621/// # use serde_with::{serde_as, DisplayFromStr, PickFirst};
1622/// #
1623/// #[serde_as]
1624/// # #[derive(Debug, PartialEq)]
1625/// #[derive(Deserialize, Serialize)]
1626/// struct Data {
1627///     #[serde_as(as = "PickFirst<(_, DisplayFromStr)>")]
1628///     as_number: u32,
1629///     #[serde_as(as = "PickFirst<(DisplayFromStr, _)>")]
1630///     as_string: u32,
1631/// }
1632/// let data = Data {
1633///     as_number: 123,
1634///     as_string: 456
1635/// };
1636///
1637/// // Both fields can be deserialized from numbers:
1638/// let j = json!({
1639///     "as_number": 123,
1640///     "as_string": 456,
1641/// });
1642/// assert_eq!(data, serde_json::from_value(j).unwrap());
1643///
1644/// // or from a string:
1645/// let j = json!({
1646///     "as_number": "123",
1647///     "as_string": "456",
1648/// });
1649/// assert_eq!(data, serde_json::from_value(j).unwrap());
1650///
1651/// // For serialization the first type in the tuple determines the behavior.
1652/// // The `as_number` field will use the normal `Serialize` behavior and produce a number,
1653/// // while `as_string` used `Display` to produce a string.
1654/// let expected = json!({
1655///     "as_number": 123,
1656///     "as_string": "456",
1657/// });
1658/// assert_eq!(expected, serde_json::to_value(&data).unwrap());
1659/// # }
1660/// ```
1661#[cfg(feature = "alloc")]
1662pub struct PickFirst<T>(PhantomData<T>);

1664/// Serialize value by converting to/from a proxy type with serde support.
1665///
1666/// This adapter serializes a type `O` by converting it into a second type `T` and serializing `T`.
1667/// Deserializing works analogue, by deserializing a `T` and then converting into `O`.
1668///
1669/// ```rust
1670/// # #[cfg(FALSE)] {
1671/// struct S {
1672///     #[serde_as(as = "FromInto<T>")]
1673///     value: O,
1674/// }
1675/// # }
1676/// ```
1677///
1678/// For serialization `O` needs to be `O: Into<T> + Clone`.
1679/// For deserialization the opposite `T: Into<O>` is required.
1680/// The `Clone` bound is required since `serialize` operates on a reference but `Into` implementations on references are uncommon.
1681///
1682/// **Note**: [`TryFromInto`] is the more generalized version of this adapter which uses the [`TryInto`](std::convert::TryInto) trait instead.
1683///
1684/// # Example
1685///
1686/// ```rust
1687/// # #[cfg(feature = "macros")] {
1688/// # use serde::{Deserialize, Serialize};
1689/// # use serde_json::json;
1690/// # use serde_with::{serde_as, FromInto};
1691/// #
1692/// #[derive(Clone, Debug, PartialEq)]
1693/// struct Rgb {
1694///     red: u8,
1695///     green: u8,
1696///     blue: u8,
1697/// }
1698///
1699/// # /*
1700/// impl From<(u8, u8, u8)> for Rgb { ... }
1701/// impl From<Rgb> for (u8, u8, u8) { ... }
1702/// # */
1703/// #
1704/// # impl From<(u8, u8, u8)> for Rgb {
1705/// #     fn from(v: (u8, u8, u8)) -> Self {
1706/// #         Rgb {
1707/// #             red: v.0,
1708/// #             green: v.1,
1709/// #             blue: v.2,
1710/// #         }
1711/// #     }
1712/// # }
1713/// #
1714/// # impl From<Rgb> for (u8, u8, u8) {
1715/// #     fn from(v: Rgb) -> Self {
1716/// #         (v.red, v.green, v.blue)
1717/// #     }
1718/// # }
1719///
1720/// #[serde_as]
1721/// # #[derive(Debug, PartialEq)]
1722/// #[derive(Deserialize, Serialize)]
1723/// struct Color {
1724///     #[serde_as(as = "FromInto<(u8, u8, u8)>")]
1725///     rgb: Rgb,
1726/// }
1727/// let color = Color {
1728///     rgb: Rgb {
1729///         red: 128,
1730///         green: 64,
1731///         blue: 32,
1732///     },
1733/// };
1734///
1735/// // Define our expected JSON form
1736/// let j = json!({
1737///     "rgb": [128, 64, 32],
1738/// });
1739/// // Ensure serialization and deserialization produce the expected results
1740/// assert_eq!(j, serde_json::to_value(&color).unwrap());
1741/// assert_eq!(color, serde_json::from_value(j).unwrap());
1742/// # }
1743/// ```
1744pub struct FromInto<T>(PhantomData<T>);

1746/// Serialize value by converting to/from a proxy type with serde support.
1747///
1748/// This adapter serializes a type `O` by converting it into a second type `T` and serializing `T`.
1749/// Deserializing works analogue, by deserializing a `T` and then converting into `O`.
1750///
1751/// ```rust
1752/// # #[cfg(FALSE)] {
1753/// struct S {
1754///     #[serde_as(as = "TryFromInto<T>")]
1755///     value: O,
1756/// }
1757/// # }
1758/// ```
1759///
1760/// For serialization `O` needs to be `O: TryInto<T> + Clone`.
1761/// For deserialization the opposite `T: TryInto<O>` is required.
1762/// The `Clone` bound is required since `serialize` operates on a reference but `TryInto` implementations on references are uncommon.
1763/// In both cases the `TryInto::Error` type must implement [`Display`](std::fmt::Display).
1764///
1765/// **Note**: [`FromInto`] is the more specialized version of this adapter which uses the infallible [`Into`] trait instead.
1766/// [`TryFromInto`] is strictly more general and can also be used where [`FromInto`] is applicable.
1767/// The example shows a use case, when only the deserialization behavior is fallible, but not serializing.
1768///
1769/// # Example
1770///
1771/// ```rust
1772/// # #[cfg(feature = "macros")] {
1773/// # use serde::{Deserialize, Serialize};
1774/// # use serde_json::json;
1775/// # use serde_with::{serde_as, TryFromInto};
1776/// # use std::convert::TryFrom;
1777/// #
1778/// #[derive(Clone, Debug, PartialEq)]
1779/// enum Boollike {
1780///     True,
1781///     False,
1782/// }
1783///
1784/// # /*
1785/// impl From<Boollike> for u8 { ... }
1786/// # */
1787/// #
1788/// impl TryFrom<u8> for Boollike {
1789///     type Error = String;
1790///     fn try_from(v: u8) -> Result<Self, Self::Error> {
1791///         match v {
1792///             0 => Ok(Boollike::False),
1793///             1 => Ok(Boollike::True),
1794///             _ => Err(format!("Boolikes can only be constructed from 0 or 1 but found {}", v))
1795///         }
1796///     }
1797/// }
1798/// #
1799/// # impl From<Boollike> for u8 {
1800/// #     fn from(v: Boollike) -> Self {
1801/// #        match v {
1802/// #            Boollike::True => 1,
1803/// #            Boollike::False => 0,
1804/// #        }
1805/// #     }
1806/// # }
1807///
1808/// #[serde_as]
1809/// # #[derive(Debug, PartialEq)]
1810/// #[derive(Deserialize, Serialize)]
1811/// struct Data {
1812///     #[serde_as(as = "TryFromInto<u8>")]
1813///     b: Boollike,
1814/// }
1815/// let data = Data {
1816///     b: Boollike::True,
1817/// };
1818///
1819/// // Define our expected JSON form
1820/// let j = json!({
1821///     "b": 1,
1822/// });
1823/// // Ensure serialization and deserialization produce the expected results
1824/// assert_eq!(j, serde_json::to_value(&data).unwrap());
1825/// assert_eq!(data, serde_json::from_value(j).unwrap());
1826///
1827/// // Numbers besides 0 or 1 should be an error
1828/// let j = json!({
1829///     "b": 2,
1830/// });
1831/// assert_eq!("Boolikes can only be constructed from 0 or 1 but found 2", serde_json::from_value::<Data>(j).unwrap_err().to_string());
1832/// # }
1833/// ```
1834pub struct TryFromInto<T>(PhantomData<T>);

1836/// Borrow `Cow` data during deserialization when possible.
1837///
1838/// The types `Cow<'a, [u8]>`, `Cow<'a, [u8; N]>`, and `Cow<'a, str>` can borrow from the input data during deserialization.
1839/// serde supports this, by annotating the fields with `#[serde(borrow)]`. but does not support borrowing on nested types.

1841///
1842/// Using this adapter with `Cow<'a, [u8]>`/Cow<'a, [u8; N]>` will serialize the value as a sequence of `u8` values.

1845///
1846/// When combined with the [`serde_as`] attribute, the `#[serde(borrow)]` annotation will be added automatically.
1847/// If the annotation is wrong or too broad, for example because of multiple lifetime parameters, a manual annotation is required.
1848///
1849/// # Examples
1850///
1851/// ```rust
1852/// # #[cfg(feature = "macros")] {

1855/// # use std::borrow::Cow;
1856/// #
1857/// #[serde_as]
1858/// # #[derive(Debug, PartialEq)]
1859/// #[derive(Deserialize, Serialize)]
1860/// struct Data<'a, 'b, 'c> {
1861///     #[serde_as(as = "BorrowCow")]
1862///     str: Cow<'a, str>,
1863///     #[serde_as(as = "BorrowCow")]
1864///     slice: Cow<'b, [u8]>,
1865///
1866///     #[serde_as(as = "Option<[BorrowCow; 1]>")]
1867///     nested: Option<[Cow<'c, str>; 1]>,
1868/// }
1869/// let data = Data {
1870///     str: "foobar".into(),
1871///     slice: b"foobar"[..].into(),
1872///     nested: Some(["HelloWorld".into()]),
1873/// };
1874///
1875/// // Define our expected JSON form
1876/// let j = r#"{
1877///   "str": "foobar",
1878///   "slice": [
1879///     102,
1880///     111,
1881///     111,
1882///     98,
1883///     97,
1884///     114
1885///   ],
1886///   "nested": [
1887///     "HelloWorld"
1888///   ]
1889/// }"#;
1890/// // Ensure serialization and deserialization produce the expected results
1891/// assert_eq!(j, serde_json::to_string_pretty(&data).unwrap());
1892/// assert_eq!(data, serde_json::from_str(j).unwrap());

1903pub struct BorrowCow;
