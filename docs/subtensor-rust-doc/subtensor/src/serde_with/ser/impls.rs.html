223impl<T, As, const N: usize> SerializeAs<[T; N]> for [As; N]
224where
225    As: SerializeAs<T>,
226{
227    fn serialize_as<S>(array: &[T; N], serializer: S) -> Result<S::Ok, S::Error>
228    where
229        S: Serializer,
230    {
231        use serde::ser::SerializeTuple;
232        let mut arr = serializer.serialize_tuple(N)?;
233        for elem in array {
234            arr.serialize_element(&SerializeAsWrap::<T, As>::new(elem))?;
235        }
236        arr.end()
237    }
238}

244macro_rules! seq_impl {
245    ($ty:ident < T $(: $tbound1:ident $(+ $tbound2:ident)*)* $(, $typaram:ident : $bound:ident )* >) => {
246        impl<T, U $(, $typaram)*> SerializeAs<$ty<T $(, $typaram)*>> for $ty<U $(, $typaram)*>
247        where
248            U: SerializeAs<T>,
249            $(T: ?Sized + $tbound1 $(+ $tbound2)*,)*
250            $($typaram: ?Sized + $bound,)*
251        {
252            fn serialize_as<S>(source: &$ty<T $(, $typaram)*>, serializer: S) -> Result<S::Ok, S::Error>
253            where
254                S: Serializer,
255            {
256                serializer.collect_seq(source.iter().map(|item| SerializeAsWrap::<T, U>::new(item)))
257            }
258        }
259    }

534impl<SEPARATOR, I, T> SerializeAs<I> for StringWithSeparator<SEPARATOR, T>
535where
536    SEPARATOR: Separator,
537    for<'x> &'x I: IntoIterator<Item = &'x T>,
538    T: Display,
539    // This set of bounds is enough to make the function compile but has inference issues
540    // making it unusable at the moment.
541    // https://github.com/rust-lang/rust/issues/89196#issuecomment-932024770
542    // for<'x> &'x I: IntoIterator,
543    // for<'x> <&'x I as IntoIterator>::Item: Display,
544{
545    fn serialize_as<S>(source: &I, serializer: S) -> Result<S::Ok, S::Error>
546    where
547        S: Serializer,
548    {
549        use core::fmt;
550        pub(crate) struct DisplayWithSeparator<'a, I, SEPARATOR>(&'a I, PhantomData<SEPARATOR>);
551
552        impl<'a, I, SEPARATOR> DisplayWithSeparator<'a, I, SEPARATOR> {
553            pub(crate) fn new(iter: &'a I) -> Self {
554                Self(iter, PhantomData)
555            }
556        }
557
558        impl<'a, I, SEPARATOR> Display for DisplayWithSeparator<'a, I, SEPARATOR>
559        where
560            SEPARATOR: Separator,
561            &'a I: IntoIterator,
562            <&'a I as IntoIterator>::Item: Display,
563        {
564            fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
565                let mut iter = self.0.into_iter();
566
567                if let Some(first) = iter.next() {
568                    first.fmt(f)?;
569                }
570                for elem in iter {
571                    f.write_str(SEPARATOR::separator())?;
572                    elem.fmt(f)?;
573                }
574
575                Ok(())
576            }
577        }
578
579        serializer.collect_str(&DisplayWithSeparator::<I, SEPARATOR>::new(source))
580    }
581}

583macro_rules! use_signed_duration {
584    (
585        $main_trait:ident $internal_trait:ident =>
586        {
587            $ty:ty =>
588            $({
589                $format:ty, $strictness:ty =>
590                $($tbound:ident: $bound:ident $(,)?)*
591            })*
592        }
593    ) => {
594        $(
595            impl<$($tbound,)*> SerializeAs<$ty> for $main_trait<$format, $strictness>
596            where
597                $($tbound: $bound,)*
598            {
599                fn serialize_as<S>(source: &$ty, serializer: S) -> Result<S::Ok, S::Error>
600                where
601                    S: Serializer,
602                {
603                    $internal_trait::<$format, $strictness>::serialize_as(
604                        &DurationSigned::from(source),
605                        serializer,
606                    )
607                }
608            }
609        )*
610    };
611    (
612        $( $main_trait:ident $internal_trait:ident, )+ => $rest:tt
613    ) => {
614        $( use_signed_duration!($main_trait $internal_trait => $rest); )+
615    };
616}
