1use crate::{CurveConfig, CurveGroup};

3/// The GLV parameters for computing the endomorphism and scalar decomposition.
4pub trait GLVConfig: Send + Sync + 'static + CurveConfig {
5    /// A representation of curve points that enables efficient arithmetic by
6    /// avoiding inversions.
7    type Curve: CurveGroup<Config = Self>;
8
9    // Constants that are used to calculate `phi(G) := lambda*G`.
10
11    /// Coefficient `a_1` of `f(y) = a_1 * (y + a_2) * (y + a_3)`.
12    const COEFF_A1: Self::BaseField;
13    /// Coefficient `a_2` of `f(y) = a_1 * (y + a_2) * (y + a_3)`.
14    const COEFF_A2: Self::BaseField;
15    /// Coefficient `a_3` of `f(y) = a_1 * (y + a_2) * (y + a_3)`.
16    const COEFF_A3: Self::BaseField;
17
18    /// Coefficient `b_1` of `g(y) = b_1 * (y + b_2) * (y + b_3)`.
19    const COEFF_B1: Self::BaseField;
20    /// Coefficient `b_2` of `g(y) = b_1 * (y + b_2) * (y + b_3)`.
21    const COEFF_B2: Self::BaseField;
22    /// Coefficient `b_3` of `g(y) = b_1 * (y + b_2) * (y + b_3)`.
23    const COEFF_B3: Self::BaseField;
24
25    /// Coefficient `c_1` of `h(y) = (y + c_1) * (y + c_2)`.
26    const COEFF_C1: Self::BaseField;
27    /// Coefficient `c_2` of `h(y) = (y + c_1) * (y + c_2)`.
28    const COEFF_C2: Self::BaseField;
29
30    // Constants for scalar decomposition.
31    // This is a 2x2 matrix, which is practically the LLL-reduced bases.
32
33    /// The first element of the matrix for scalar decomposition.
34    const COEFF_N11: Self::ScalarField;
35    /// The second element of the matrix for scalar decomposition.
36    const COEFF_N12: Self::ScalarField;
37    /// The third element of the matrix for scalar decomposition.
38    const COEFF_N21: Self::ScalarField;
39    /// The forth element of the matrix for the scalar decomposition.
40    const COEFF_N22: Self::ScalarField;
41
42    /// Maps a point G to phi(G):= lambda G where psi is the endomorphism.
43    // On an affine curve, the function takes the following steps:
44    //  f(y) = a_1 * (y + a_2) * (y + a_3)
45    //  g(y) = b_1 * (y + b_2) * (y + b_3)
46    //  h(y) = (y + c_1) * (y + c_2)
47    //  return (x',y') where
48    //  x' = x * f(y) / y
49    //  y' = g(y) / h(y)
50    fn endomorphism(
51        base: &<Self::Curve as CurveGroup>::Affine,
52    ) -> <Self::Curve as CurveGroup>::Affine;
53
54    /// Decomposes a scalar s into k1, k2, s.t. s = k1 + lambda k2,
55    fn scalar_decomposition(k: &Self::ScalarField) -> (Self::ScalarField, Self::ScalarField);
56
57    /// Performs GLV multiplication.
58    fn glv_mul(
59        base: &<Self::Curve as CurveGroup>::Affine,
60        scalar: &Self::ScalarField,
61    ) -> Self::Curve;
62}
