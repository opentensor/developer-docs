20    pub fn get_window_table<T: ScalarMul>(
21        scalar_size: usize,
22        window: usize,
23        g: T,
24    ) -> Vec<Vec<T::MulBase>> {
25        let in_window = 1 << window;
26        let outerc = (scalar_size + window - 1) / window;
27        let last_in_window = 1 << (scalar_size - (outerc - 1) * window);
28
29        let mut multiples_of_g = vec![vec![T::zero(); in_window]; outerc];
30
31        let mut g_outer = g;
32        let mut g_outers = Vec::with_capacity(outerc);
33        for _ in 0..outerc {
34            g_outers.push(g_outer);
35            for _ in 0..window {
36                g_outer.double_in_place();
37            }
38        }
39        cfg_iter_mut!(multiples_of_g)
40            .enumerate()
41            .take(outerc)
42            .zip(g_outers)
43            .for_each(|((outer, multiples_of_g), g_outer)| {
44                let cur_in_window = if outer == outerc - 1 {
45                    last_in_window
46                } else {
47                    in_window
48                };
49
50                let mut g_inner = T::zero();
51                for inner in multiples_of_g.iter_mut().take(cur_in_window) {
52                    *inner = g_inner;
53                    g_inner += &g_outer;
54                }
55            });
56        cfg_iter!(multiples_of_g)
57            .map(|s| T::batch_convert_to_mul_base(s))
58            .collect()
59    }

61    pub fn windowed_mul<T: ScalarMul>(
62        outerc: usize,
63        window: usize,
64        multiples_of_g: &[Vec<<T as ScalarMul>::MulBase>],
65        scalar: &T::ScalarField,
66    ) -> T {
67        let modulus_size = T::ScalarField::MODULUS_BIT_SIZE as usize;
68        let scalar_val = scalar.into_bigint().to_bits_le();
69
70        let mut res = T::from(multiples_of_g[0][0]);
71        for outer in 0..outerc {
72            let mut inner = 0usize;
73            for i in 0..window {
74                if outer * window + i < modulus_size && scalar_val[outer * window + i] {
75                    inner |= 1 << i;
76                }
77            }
78            res += &multiples_of_g[outer][inner];
79        }
80        res
81    }

85    pub fn msm<T: ScalarMul>(

87        window: usize,
88        table: &[Vec<<T as ScalarMul>::MulBase>],
89        v: &[T::ScalarField],

91        let outerc = (scalar_size + window - 1) / window;
92        assert!(outerc <= table.len());
93
94        cfg_iter!(v)
95            .map(|e| Self::windowed_mul::<T>(outerc, window, table, e))
96            .collect::<Vec<_>>()
97    }
