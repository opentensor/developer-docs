18use crate::pallet::Def;
19
20struct ConstDef {
21	/// Name of the associated type.
22	pub ident: syn::Ident,
23	/// The type in Get, e.g. `u32` in `type Foo: Get<u32>;`, but `Self` is replaced by `T`
24	pub type_: syn::Type,
25	/// The doc associated
26	pub doc: Vec<syn::Expr>,
27	/// default_byte implementation
28	pub default_byte_impl: proc_macro2::TokenStream,
29	/// Constant name for Metadata (optional)
30	pub metadata_name: Option<syn::Ident>,
31}
32
33/// Implement the `pallet_constants_metadata` function for the pallet.
34pub fn expand_constants(def: &mut Def) -> proc_macro2::TokenStream {
35	let frame_support = &def.frame_support;
36	let type_impl_gen = &def.type_impl_generics(proc_macro2::Span::call_site());
37	let type_use_gen = &def.type_use_generics(proc_macro2::Span::call_site());
38	let pallet_ident = &def.pallet_struct.pallet;
39	let trait_use_gen = &def.trait_use_generics(proc_macro2::Span::call_site());
40
41	let mut where_clauses = vec![&def.config.where_clause];
42	where_clauses.extend(def.extra_constants.iter().map(|d| &d.where_clause));
43	let completed_where_clause = super::merge_where_clauses(&where_clauses);
44
45	let config_consts = def.config.consts_metadata.iter().map(|const_| {
46		let ident = &const_.ident;
47		let const_type = &const_.type_;
48
49		ConstDef {
50			ident: const_.ident.clone(),
51			type_: const_.type_.clone(),
52			doc: const_.doc.clone(),
53			default_byte_impl: quote::quote!(
54				let value = <<T as Config #trait_use_gen>::#ident as
55					#frame_support::traits::Get<#const_type>>::get();
56				#frame_support::__private::codec::Encode::encode(&value)
57			),
58			metadata_name: None,
59		}
60	});
61
62	let extra_consts = def.extra_constants.iter().flat_map(|d| &d.extra_constants).map(|const_| {
63		let ident = &const_.ident;
64
65		ConstDef {
66			ident: const_.ident.clone(),
67			type_: const_.type_.clone(),
68			doc: const_.doc.clone(),
69			default_byte_impl: quote::quote!(
70				let value = <Pallet<#type_use_gen>>::#ident();
71				#frame_support::__private::codec::Encode::encode(&value)
72			),
73			metadata_name: const_.metadata_name.clone(),
74		}
75	});
76
77	let consts = config_consts.chain(extra_consts).map(|const_| {
78		let const_type = &const_.type_;
79		let ident_str = format!("{}", const_.metadata_name.unwrap_or(const_.ident));
80
81		let no_docs = vec![];
82		let doc = if cfg!(feature = "no-metadata-docs") { &no_docs } else { &const_.doc };
83
84		let default_byte_impl = &const_.default_byte_impl;
85
86		quote::quote!({
87			#frame_support::__private::metadata_ir::PalletConstantMetadataIR {
88				name: #ident_str,
89				ty: #frame_support::__private::scale_info::meta_type::<#const_type>(),
90				value: { #default_byte_impl },
91				docs: #frame_support::__private::vec![ #( #doc ),* ],
92			}
93		})
94	});
95
96	quote::quote!(
97		impl<#type_impl_gen> #pallet_ident<#type_use_gen> #completed_where_clause{
98
99			#[doc(hidden)]
100			pub fn pallet_constants_metadata()
101				-> #frame_support::__private::Vec<#frame_support::__private::metadata_ir::PalletConstantMetadataIR>
102			{
103				#frame_support::__private::vec![ #( #consts ),* ]
104			}
105		}
106	)
107}
