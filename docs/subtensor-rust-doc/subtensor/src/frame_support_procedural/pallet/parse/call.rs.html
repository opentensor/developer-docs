38/// Definition of dispatchables typically `impl<T: Config> Pallet<T> { ... }`
39pub struct CallDef {
40	/// The where_clause used.
41	pub where_clause: Option<syn::WhereClause>,
42	/// A set of usage of instance, must be check for consistency with trait.
43	pub instances: Vec<helper::InstanceUsage>,
44	/// The index of call item in pallet module.
45	pub index: usize,
46	/// Information on methods (used for expansion).
47	pub methods: Vec<CallVariantDef>,
48	/// The span of the pallet::call attribute.
49	pub attr_span: proc_macro2::Span,
50	/// Docs, specified on the impl Block.
51	pub docs: Vec<syn::Expr>,
52	/// The optional `weight` attribute on the `pallet::call`.
53	pub inherited_call_weight: Option<InheritedCallWeightAttr>,
54}

56/// The weight of a call.
57#[derive(Clone)]
58pub enum CallWeightDef {
59	/// Explicitly set on the call itself with `#[pallet::weight(â€¦)]`. This value is used.
60	Immediate(syn::Expr),
61
62	/// The default value that should be set for dev-mode pallets. Usually zero.
63	DevModeDefault,
64
65	/// Inherits whatever value is configured on the pallet level.
66	///
67	/// The concrete value is not known at this point.
68	Inherited,
69}

71/// Definition of dispatchable typically: `#[weight...] fn foo(origin .., param1: ...) -> ..`
72#[derive(Clone)]
73pub struct CallVariantDef {
74	/// Function name.
75	pub name: syn::Ident,
76	/// Information on args: `(is_compact, name, type)`
77	pub args: Vec<(bool, syn::Ident, Box<syn::Type>)>,
78	/// Weight for the call.
79	pub weight: CallWeightDef,
80	/// Call index of the dispatchable.
81	pub call_index: u8,
82	/// Whether an explicit call index was specified.
83	pub explicit_call_index: bool,
84	/// Docs, used for metadata.
85	pub docs: Vec<syn::Expr>,
86	/// Attributes annotated at the top of the dispatchable function.
87	pub attrs: Vec<syn::Attribute>,
88	/// The `cfg` attributes.
89	pub cfg_attrs: Vec<syn::Attribute>,
90	/// The optional `feeless_if` attribute on the `pallet::call`.
91	pub feeless_check: Option<syn::ExprClosure>,
92	/// The return type of the call: `DispatchInfo` or `DispatchResultWithPostInfo`.
93	pub return_type: helper::CallReturnType,
94}

96/// Attributes for functions in call impl block.
97pub enum FunctionAttr {
98	/// Parse for `#[pallet::call_index(expr)]`
99	CallIndex(u8),
100	/// Parse for `#[pallet::weight(expr)]`
101	Weight(syn::Expr),
102	/// Parse for `#[pallet::feeless_if(expr)]`
103	FeelessIf(Span, syn::ExprClosure),
104}
105
106impl syn::parse::Parse for FunctionAttr {
107	fn parse(input: syn::parse::ParseStream) -> syn::Result<Self> {
108		input.parse::<syn::Token![#]>()?;
109		let content;
110		syn::bracketed!(content in input);
111		content.parse::<keyword::pallet>()?;
112		content.parse::<syn::Token![::]>()?;
113
114		let lookahead = content.lookahead1();
115		if lookahead.peek(keyword::weight) {
116			content.parse::<keyword::weight>()?;
117			let weight_content;
118			syn::parenthesized!(weight_content in content);
119			Ok(FunctionAttr::Weight(weight_content.parse::<syn::Expr>()?))
120		} else if lookahead.peek(keyword::call_index) {
121			content.parse::<keyword::call_index>()?;
122			let call_index_content;
123			syn::parenthesized!(call_index_content in content);
124			let index = call_index_content.parse::<syn::LitInt>()?;
125			if !index.suffix().is_empty() {
126				let msg = "Number literal must not have a suffix";
127				return Err(syn::Error::new(index.span(), msg))
128			}
129			Ok(FunctionAttr::CallIndex(index.base10_parse()?))
130		} else if lookahead.peek(keyword::feeless_if) {
131			content.parse::<keyword::feeless_if>()?;
132			let closure_content;
133			syn::parenthesized!(closure_content in content);
134			Ok(FunctionAttr::FeelessIf(
135				closure_content.span(),
136				closure_content.parse::<syn::ExprClosure>().map_err(|e| {
137					let msg = "Invalid feeless_if attribute: expected a closure";
138					let mut err = syn::Error::new(closure_content.span(), msg);
139					err.combine(e);
140					err
141				})?,
142			))
143		} else {
144			Err(lookahead.error())
145		}
146	}
147}

149/// Attribute for arguments in function in call impl block.
150/// Parse for `#[pallet::compact]|
151pub struct ArgAttrIsCompact;
152
153impl syn::parse::Parse for ArgAttrIsCompact {
154	fn parse(input: syn::parse::ParseStream) -> syn::Result<Self> {
155		input.parse::<syn::Token![#]>()?;
156		let content;
157		syn::bracketed!(content in input);
158		content.parse::<keyword::pallet>()?;
159		content.parse::<syn::Token![::]>()?;
160
161		content.parse::<keyword::compact>()?;
162		Ok(ArgAttrIsCompact)
163	}
164}

166/// Check the syntax is `OriginFor<T>`, `&OriginFor<T>` or `T::RuntimeOrigin`.
167pub fn check_dispatchable_first_arg_type(ty: &syn::Type, is_ref: bool) -> syn::Result<()> {
168	pub struct CheckOriginFor(bool);
169	impl syn::parse::Parse for CheckOriginFor {
170		fn parse(input: syn::parse::ParseStream) -> syn::Result<Self> {
171			let is_ref = input.parse::<syn::Token![&]>().is_ok();
172			input.parse::<keyword::OriginFor>()?;
173			input.parse::<syn::Token![<]>()?;
174			input.parse::<keyword::T>()?;
175			input.parse::<syn::Token![>]>()?;
176
177			Ok(Self(is_ref))
178		}
179	}
180
181	pub struct CheckRuntimeOrigin;
182	impl syn::parse::Parse for CheckRuntimeOrigin {
183		fn parse(input: syn::parse::ParseStream) -> syn::Result<Self> {
184			input.parse::<keyword::T>()?;
185			input.parse::<syn::Token![::]>()?;
186			input.parse::<keyword::RuntimeOrigin>()?;
187
188			Ok(Self)
189		}
190	}
191
192	let result_origin_for = syn::parse2::<CheckOriginFor>(ty.to_token_stream());
193	let result_runtime_origin = syn::parse2::<CheckRuntimeOrigin>(ty.to_token_stream());
194	return match (result_origin_for, result_runtime_origin) {
195		(Ok(CheckOriginFor(has_ref)), _) if is_ref == has_ref => Ok(()),
196		(_, Ok(_)) => Ok(()),
197		(_, _) => {
198			let msg = if is_ref {
199				"Invalid type: expected `&OriginFor<T>`"
200			} else {
201				"Invalid type: expected `OriginFor<T>` or `T::RuntimeOrigin`"
202			};
203			return Err(syn::Error::new(ty.span(), msg))
204		},
205	}
206}

208impl CallDef {
209	pub fn try_from(
210		attr_span: proc_macro2::Span,
211		index: usize,
212		item: &mut syn::Item,
213		dev_mode: bool,
214		inherited_call_weight: Option<InheritedCallWeightAttr>,
215	) -> syn::Result<Self> {
216		let item_impl = if let syn::Item::Impl(item) = item {
217			item
218		} else {
219			return Err(syn::Error::new(item.span(), "Invalid pallet::call, expected item impl"))
220		};
221
222		let instances = vec![
223			helper::check_impl_gen(&item_impl.generics, item_impl.impl_token.span())?,
224			helper::check_pallet_struct_usage(&item_impl.self_ty)?,
225		];
226
227		if let Some((_, _, for_)) = item_impl.trait_ {
228			let msg = "Invalid pallet::call, expected no trait ident as in \
229				`impl<..> Pallet<..> { .. }`";
230			return Err(syn::Error::new(for_.span(), msg))
231		}
232
233		let mut methods = vec![];
234		let mut indices = HashMap::new();
235		let mut last_index: Option<u8> = None;
236		for item in &mut item_impl.items {
237			if let syn::ImplItem::Fn(method) = item {
238				if !matches!(method.vis, syn::Visibility::Public(_)) {
239					let msg = "Invalid pallet::call, dispatchable function must be public: \
240						`pub fn`";
241
242					let span = match method.vis {
243						syn::Visibility::Inherited => method.sig.span(),
244						_ => method.vis.span(),
245					};
246
247					return Err(syn::Error::new(span, msg))
248				}
249
250				match method.sig.inputs.first() {
251					None => {
252						let msg = "Invalid pallet::call, must have at least origin arg";
253						return Err(syn::Error::new(method.sig.span(), msg))
254					},
255					Some(syn::FnArg::Receiver(_)) => {
256						let msg = "Invalid pallet::call, first argument must be a typed argument, \
257							e.g. `origin: OriginFor<T>`";
258						return Err(syn::Error::new(method.sig.span(), msg))
259					},
260					Some(syn::FnArg::Typed(arg)) => {
261						check_dispatchable_first_arg_type(&arg.ty, false)?;
262					},
263				}
264
265				let return_type = helper::check_pallet_call_return_type(&method.sig)?;
266
267				let cfg_attrs: Vec<syn::Attribute> = helper::get_item_cfg_attrs(&method.attrs);
268				let mut call_idx_attrs = vec![];
269				let mut weight_attrs = vec![];
270				let mut feeless_attrs = vec![];
271				for attr in helper::take_item_pallet_attrs(&mut method.attrs)?.into_iter() {
272					match attr {
273						FunctionAttr::CallIndex(_) => {
274							call_idx_attrs.push(attr);
275						},
276						FunctionAttr::Weight(_) => {
277							weight_attrs.push(attr);
278						},
279						FunctionAttr::FeelessIf(span, _) => {
280							feeless_attrs.push((span, attr));
281						},
282					}
283				}
284
285				if weight_attrs.is_empty() && dev_mode {
286					// inject a default O(1) weight when dev mode is enabled and no weight has
287					// been specified on the call
288					let empty_weight: syn::Expr = syn::parse_quote!(0);
289					weight_attrs.push(FunctionAttr::Weight(empty_weight));
290				}
291
292				let weight = match weight_attrs.len() {
293					0 if inherited_call_weight.is_some() => CallWeightDef::Inherited,
294					0 if dev_mode => CallWeightDef::DevModeDefault,
295					0 => return Err(syn::Error::new(
296						method.sig.span(),
297						"A pallet::call requires either a concrete `#[pallet::weight($expr)]` or an
298						inherited weight from the `#[pallet:call(weight($type))]` attribute, but
299						none were given.",
300					)),
301					1 => match weight_attrs.pop().unwrap() {
302						FunctionAttr::Weight(w) => CallWeightDef::Immediate(w),
303						_ => unreachable!("checked during creation of the let binding"),
304					},
305					_ => {
306						let msg = "Invalid pallet::call, too many weight attributes given";
307						return Err(syn::Error::new(method.sig.span(), msg))
308					},
309				};
310
311				if call_idx_attrs.len() > 1 {
312					let msg = "Invalid pallet::call, too many call_index attributes given";
313					return Err(syn::Error::new(method.sig.span(), msg))
314				}
315				let call_index = call_idx_attrs.pop().map(|attr| match attr {
316					FunctionAttr::CallIndex(idx) => idx,
317					_ => unreachable!("checked during creation of the let binding"),
318				});
319				let explicit_call_index = call_index.is_some();
320
321				let final_index = match call_index {
322					Some(i) => i,
323					None =>
324						last_index.map_or(Some(0), |idx| idx.checked_add(1)).ok_or_else(|| {
325							let msg = "Call index doesn't fit into u8, index is 256";
326							syn::Error::new(method.sig.span(), msg)
327						})?,
328				};
329				last_index = Some(final_index);
330
331				if let Some(used_fn) = indices.insert(final_index, method.sig.ident.clone()) {
332					let msg = format!(
333						"Call indices are conflicting: Both functions {} and {} are at index {}",
334						used_fn, method.sig.ident, final_index,
335					);
336					let mut err = syn::Error::new(used_fn.span(), &msg);
337					err.combine(syn::Error::new(method.sig.ident.span(), msg));
338					return Err(err)
339				}
340
341				let mut args = vec![];
342				for arg in method.sig.inputs.iter_mut().skip(1) {
343					let arg = if let syn::FnArg::Typed(arg) = arg {
344						arg
345					} else {
346						unreachable!("Only first argument can be receiver");
347					};
348
349					let arg_attrs: Vec<ArgAttrIsCompact> =
350						helper::take_item_pallet_attrs(&mut arg.attrs)?;
351
352					if arg_attrs.len() > 1 {
353						let msg = "Invalid pallet::call, argument has too many attributes";
354						return Err(syn::Error::new(arg.span(), msg))
355					}
356
357					let arg_ident = if let syn::Pat::Ident(pat) = &*arg.pat {
358						pat.ident.clone()
359					} else {
360						let msg = "Invalid pallet::call, argument must be ident";
361						return Err(syn::Error::new(arg.pat.span(), msg))
362					};
363
364					args.push((!arg_attrs.is_empty(), arg_ident, arg.ty.clone()));
365				}
366
367				let docs = get_doc_literals(&method.attrs);
368
369				if feeless_attrs.len() > 1 {
370					let msg = "Invalid pallet::call, there can only be one feeless_if attribute";
371					return Err(syn::Error::new(feeless_attrs[1].0, msg))
372				}
373				let feeless_check: Option<ExprClosure> =
374					feeless_attrs.pop().map(|(_, attr)| match attr {
375						FunctionAttr::FeelessIf(_, closure) => closure,
376						_ => unreachable!("checked during creation of the let binding"),
377					});
378
379				if let Some(ref feeless_check) = feeless_check {
380					if feeless_check.inputs.len() != args.len() + 1 {
381						let msg = "Invalid pallet::call, feeless_if closure must have same \
382							number of arguments as the dispatchable function";
383						return Err(syn::Error::new(feeless_check.span(), msg))
384					}
385
386					match feeless_check.inputs.first() {
387						None => {
388							let msg = "Invalid pallet::call, feeless_if closure must have at least origin arg";
389							return Err(syn::Error::new(feeless_check.span(), msg))
390						},
391						Some(syn::Pat::Type(arg)) => {
392							check_dispatchable_first_arg_type(&arg.ty, true)?;
393						},
394						_ => {
395							let msg = "Invalid pallet::call, feeless_if closure first argument must be a typed argument, \
396								e.g. `origin: OriginFor<T>`";
397							return Err(syn::Error::new(feeless_check.span(), msg))
398						},
399					}
400
401					for (feeless_arg, arg) in feeless_check.inputs.iter().skip(1).zip(args.iter()) {
402						let feeless_arg_type =
403							if let syn::Pat::Type(syn::PatType { ty, .. }) = feeless_arg.clone() {
404								if let syn::Type::Reference(pat) = *ty {
405									pat.elem.clone()
406								} else {
407									let msg = "Invalid pallet::call, feeless_if closure argument must be a reference";
408									return Err(syn::Error::new(ty.span(), msg))
409								}
410							} else {
411								let msg = "Invalid pallet::call, feeless_if closure argument must be a type ascription pattern";
412								return Err(syn::Error::new(feeless_arg.span(), msg))
413							};
414
415						if feeless_arg_type != arg.2 {
416							let msg =
417								"Invalid pallet::call, feeless_if closure argument must have \
418								a reference to the same type as the dispatchable function argument";
419							return Err(syn::Error::new(feeless_arg.span(), msg))
420						}
421					}
422
423					let valid_return = match &feeless_check.output {
424						syn::ReturnType::Type(_, type_) => match *(type_.clone()) {
425							syn::Type::Path(syn::TypePath { path, .. }) => path.is_ident("bool"),
426							_ => false,
427						},
428						_ => false,
429					};
430					if !valid_return {
431						let msg = "Invalid pallet::call, feeless_if closure must return `bool`";
432						return Err(syn::Error::new(feeless_check.output.span(), msg))
433					}
434				}
435
436				methods.push(CallVariantDef {
437					name: method.sig.ident.clone(),
438					weight,
439					call_index: final_index,
440					explicit_call_index,
441					args,
442					docs,
443					attrs: method.attrs.clone(),
444					cfg_attrs,
445					feeless_check,
446					return_type,
447				});
448			} else {
449				let msg = "Invalid pallet::call, only method accepted";
450				return Err(syn::Error::new(item.span(), msg))
451			}
452		}
453
454		Ok(Self {
455			index,
456			attr_span,
457			instances,
458			methods,
459			where_clause: item_impl.generics.where_clause.clone(),
460			docs: get_doc_literals(&item_impl.attrs),
461			inherited_call_weight,
462		})
463	}
464}
