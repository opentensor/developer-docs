76	pub fn try_from(mut item: syn::ItemMod, dev_mode: bool) -> syn::Result<Self> {
77		let frame_system = generate_access_from_frame_or_crate("frame-system")?;
78		let frame_support = generate_access_from_frame_or_crate("frame-support")?;
79
80		let item_span = item.span();
81		let items = &mut item
82			.content
83			.as_mut()
84			.ok_or_else(|| {
85				let msg = "Invalid pallet definition, expected mod to be inlined.";
86				syn::Error::new(item_span, msg)
87			})?
88			.1;
89
90		let mut config = None;
91		let mut pallet_struct = None;
92		let mut hooks = None;
93		let mut call = None;
94		let mut tasks = None;
95		let mut task_enum = None;
96		let mut error = None;
97		let mut event = None;
98		let mut origin = None;
99		let mut inherent = None;
100		let mut genesis_config = None;
101		let mut genesis_build = None;
102		let mut validate_unsigned = None;
103		let mut extra_constants = None;
104		let mut storages = vec![];
105		let mut type_values = vec![];
106		let mut composites: Vec<CompositeDef> = vec![];
107
108		for (index, item) in items.iter_mut().enumerate() {
109			let pallet_attr: Option<PalletAttr> = helper::take_first_item_pallet_attr(item)?;
110
111			match pallet_attr {
112				Some(PalletAttr::Config(_, with_default)) if config.is_none() =>
113					config = Some(config::ConfigDef::try_from(
114						&frame_system,
115						index,
116						item,
117						with_default,
118					)?),
119				Some(PalletAttr::Pallet(span)) if pallet_struct.is_none() => {
120					let p = pallet_struct::PalletStructDef::try_from(span, index, item)?;
121					pallet_struct = Some(p);
122				},
123				Some(PalletAttr::Hooks(span)) if hooks.is_none() => {
124					let m = hooks::HooksDef::try_from(span, item)?;
125					hooks = Some(m);
126				},
127				Some(PalletAttr::RuntimeCall(cw, span)) if call.is_none() =>
128					call = Some(call::CallDef::try_from(span, index, item, dev_mode, cw)?),
129				Some(PalletAttr::Tasks(_)) if tasks.is_none() => {
130					let item_tokens = item.to_token_stream();
131					// `TasksDef::parse` needs to know if attr was provided so we artificially
132					// re-insert it here
133					tasks = Some(syn::parse2::<tasks::TasksDef>(quote::quote! {
134						#[pallet::tasks_experimental]
135						#item_tokens
136					})?);
137
138					// replace item with a no-op because it will be handled by the expansion of tasks
139					*item = syn::Item::Verbatim(quote::quote!());
140				}
141				Some(PalletAttr::TaskCondition(span)) => return Err(syn::Error::new(
142					span,
143					"`#[pallet::task_condition]` can only be used on items within an `impl` statement."
144				)),
145				Some(PalletAttr::TaskIndex(span)) => return Err(syn::Error::new(
146					span,
147					"`#[pallet::task_index]` can only be used on items within an `impl` statement."
148				)),
149				Some(PalletAttr::TaskList(span)) => return Err(syn::Error::new(
150					span,
151					"`#[pallet::task_list]` can only be used on items within an `impl` statement."
152				)),
153				Some(PalletAttr::RuntimeTask(_)) if task_enum.is_none() =>
154					task_enum = Some(syn::parse2::<tasks::TaskEnumDef>(item.to_token_stream())?),
155				Some(PalletAttr::Error(span)) if error.is_none() =>
156					error = Some(error::ErrorDef::try_from(span, index, item)?),
157				Some(PalletAttr::RuntimeEvent(span)) if event.is_none() =>
158					event = Some(event::EventDef::try_from(span, index, item)?),
159				Some(PalletAttr::GenesisConfig(_)) if genesis_config.is_none() => {
160					let g = genesis_config::GenesisConfigDef::try_from(index, item)?;
161					genesis_config = Some(g);
162				},
163				Some(PalletAttr::GenesisBuild(span)) if genesis_build.is_none() => {
164					let g = genesis_build::GenesisBuildDef::try_from(span, item)?;
165					genesis_build = Some(g);
166				},
167				Some(PalletAttr::RuntimeOrigin(_)) if origin.is_none() =>
168					origin = Some(origin::OriginDef::try_from(item)?),
169				Some(PalletAttr::Inherent(_)) if inherent.is_none() =>
170					inherent = Some(inherent::InherentDef::try_from(item)?),
171				Some(PalletAttr::Storage(span)) =>
172					storages.push(storage::StorageDef::try_from(span, index, item, dev_mode)?),
173				Some(PalletAttr::ValidateUnsigned(_)) if validate_unsigned.is_none() => {
174					let v = validate_unsigned::ValidateUnsignedDef::try_from(item)?;
175					validate_unsigned = Some(v);
176				},
177				Some(PalletAttr::TypeValue(span)) =>
178					type_values.push(type_value::TypeValueDef::try_from(span, index, item)?),
179				Some(PalletAttr::ExtraConstants(_)) =>
180					extra_constants =
181						Some(extra_constants::ExtraConstantsDef::try_from(item)?),
182				Some(PalletAttr::Composite(span)) => {
183					let composite =
184						composite::CompositeDef::try_from(span, &frame_support, item)?;
185					if composites.iter().any(|def| {
186						match (&def.composite_keyword, &composite.composite_keyword) {
187							(
188								CompositeKeyword::FreezeReason(_),
189								CompositeKeyword::FreezeReason(_),
190							) |
191							(CompositeKeyword::HoldReason(_), CompositeKeyword::HoldReason(_)) |
192							(CompositeKeyword::LockId(_), CompositeKeyword::LockId(_)) |
193							(
194								CompositeKeyword::SlashReason(_),
195								CompositeKeyword::SlashReason(_),
196							) => true,
197							_ => false,
198						}
199					}) {
200						let msg = format!(
201							"Invalid duplicated `{}` definition",
202							composite.composite_keyword
203						);
204						return Err(syn::Error::new(composite.composite_keyword.span(), &msg))
205					}
206					composites.push(composite);
207				},
208				Some(attr) => {
209					let msg = "Invalid duplicated attribute";
210					return Err(syn::Error::new(attr.span(), msg))
211				},
212				None => (),
213			}
214		}
215
216		if genesis_config.is_some() != genesis_build.is_some() {
217			let msg = format!(
218				"`#[pallet::genesis_config]` and `#[pallet::genesis_build]` attributes must be \
219				either both used or both not used, instead genesis_config is {} and genesis_build \
220				is {}",
221				genesis_config.as_ref().map_or("unused", |_| "used"),
222				genesis_build.as_ref().map_or("unused", |_| "used"),
223			);
224			return Err(syn::Error::new(item_span, msg))
225		}
226
227		Self::resolve_tasks(&item_span, &mut tasks, &mut task_enum, items)?;
228
229		let def = Def {
230			item,
231			config: config
232				.ok_or_else(|| syn::Error::new(item_span, "Missing `#[pallet::config]`"))?,
233			pallet_struct: pallet_struct
234				.ok_or_else(|| syn::Error::new(item_span, "Missing `#[pallet::pallet]`"))?,
235			hooks,
236			call,
237			tasks,
238			task_enum,
239			extra_constants,
240			genesis_config,
241			genesis_build,
242			validate_unsigned,
243			error,
244			event,
245			origin,
246			inherent,
247			storages,
248			composites,
249			type_values,
250			frame_system,
251			frame_support,
252			dev_mode,
253		};
254
255		def.check_instance_usage()?;
256		def.check_event_usage()?;
257
258		Ok(def)
259	}

261	/// Performs extra logic checks necessary for the `#[pallet::tasks_experimental]` feature.
262	fn resolve_tasks(
263		item_span: &proc_macro2::Span,
264		tasks: &mut Option<tasks::TasksDef>,
265		task_enum: &mut Option<tasks::TaskEnumDef>,
266		items: &mut Vec<syn::Item>,
267	) -> syn::Result<()> {
268		// fallback for manual (without macros) definition of tasks impl
269		Self::resolve_manual_tasks_impl(tasks, task_enum, items)?;
270
271		// fallback for manual (without macros) definition of task enum
272		Self::resolve_manual_task_enum(tasks, task_enum, items)?;
273
274		// ensure that if `task_enum` is specified, `tasks` is also specified
275		match (&task_enum, &tasks) {
276			(Some(_), None) =>
277				return Err(syn::Error::new(
278					*item_span,
279					"Missing `#[pallet::tasks_experimental]` impl",
280				)),
281			(None, Some(tasks)) =>
282				if tasks.tasks_attr.is_none() {
283					return Err(syn::Error::new(
284						tasks.item_impl.impl_token.span(),
285						"A `#[pallet::tasks_experimental]` attribute must be attached to your `Task` impl if the \
286						task enum has been omitted",
287					))
288				} else {
289				},
290			_ => (),
291		}
292
293		Ok(())
294	}

296	/// Tries to locate task enum based on the tasks impl target if attribute is not specified
297	/// but impl is present. If one is found, `task_enum` is set appropriately.
298	fn resolve_manual_task_enum(
299		tasks: &Option<tasks::TasksDef>,
300		task_enum: &mut Option<tasks::TaskEnumDef>,
301		items: &mut Vec<syn::Item>,
302	) -> syn::Result<()> {
303		let (None, Some(tasks)) = (&task_enum, &tasks) else { return Ok(()) };
304		let syn::Type::Path(type_path) = &*tasks.item_impl.self_ty else { return Ok(()) };
305		let type_path = type_path.path.segments.iter().collect::<Vec<_>>();
306		let (Some(seg), None) = (type_path.get(0), type_path.get(1)) else { return Ok(()) };
307		let mut result = None;
308		for item in items {
309			let syn::Item::Enum(item_enum) = item else { continue };
310			if item_enum.ident == seg.ident {
311				result = Some(syn::parse2::<tasks::TaskEnumDef>(item_enum.to_token_stream())?);
312				// replace item with a no-op because it will be handled by the expansion of
313				// `task_enum`. We use a no-op instead of simply removing it from the vec
314				// so that any indices collected by `Def::try_from` remain accurate
315				*item = syn::Item::Verbatim(quote::quote!());
316				break
317			}
318		}
319		*task_enum = result;
320		Ok(())
321	}

323	/// Tries to locate a manual tasks impl (an impl implementing a trait whose last path segment is
324	/// `Task`) in the event that one has not been found already via the attribute macro
325	pub fn resolve_manual_tasks_impl(
326		tasks: &mut Option<tasks::TasksDef>,
327		task_enum: &Option<tasks::TaskEnumDef>,
328		items: &Vec<syn::Item>,
329	) -> syn::Result<()> {
330		let None = tasks else { return Ok(()) };
331		let mut result = None;
332		for item in items {
333			let syn::Item::Impl(item_impl) = item else { continue };
334			let Some((_, path, _)) = &item_impl.trait_ else { continue };
335			let Some(trait_last_seg) = path.segments.last() else { continue };
336			let syn::Type::Path(target_path) = &*item_impl.self_ty else { continue };
337			let target_path = target_path.path.segments.iter().collect::<Vec<_>>();
338			let (Some(target_ident), None) = (target_path.get(0), target_path.get(1)) else {
339				continue
340			};
341			let matches_task_enum = match task_enum {
342				Some(task_enum) => task_enum.item_enum.ident == target_ident.ident,
343				None => true,
344			};
345			if trait_last_seg.ident == "Task" && matches_task_enum {
346				result = Some(syn::parse2::<tasks::TasksDef>(item_impl.to_token_stream())?);
347				break
348			}
349		}
350		*tasks = result;
351		Ok(())
352	}

354	/// Check that usage of trait `Event` is consistent with the definition, i.e. it is declared
355	/// and trait defines type RuntimeEvent, or not declared and no trait associated type.
356	fn check_event_usage(&self) -> syn::Result<()> {
357		match (self.config.has_event_type, self.event.is_some()) {
358			(true, false) => {
359				let msg = "Invalid usage of RuntimeEvent, `Config` contains associated type `RuntimeEvent`, \
360					but enum `Event` is not declared (i.e. no use of `#[pallet::event]`). \
361					Note that type `RuntimeEvent` in trait is reserved to work alongside pallet event.";
362				Err(syn::Error::new(proc_macro2::Span::call_site(), msg))
363			},
364			(false, true) => {
365				let msg = "Invalid usage of RuntimeEvent, `Config` contains no associated type \
366					`RuntimeEvent`, but enum `Event` is declared (in use of `#[pallet::event]`). \
367					An RuntimeEvent associated type must be declare on trait `Config`.";
368				Err(syn::Error::new(proc_macro2::Span::call_site(), msg))
369			},
370			_ => Ok(()),
371		}
372	}

374	/// Check that usage of trait `Config` is consistent with the definition, i.e. it is used with
375	/// instance iff it is defined with instance.
376	fn check_instance_usage(&self) -> syn::Result<()> {
377		let mut instances = vec![];
378		instances.extend_from_slice(&self.pallet_struct.instances[..]);
379		instances.extend(&mut self.storages.iter().flat_map(|s| s.instances.clone()));
380		if let Some(call) = &self.call {
381			instances.extend_from_slice(&call.instances[..]);
382		}
383		if let Some(hooks) = &self.hooks {
384			instances.extend_from_slice(&hooks.instances[..]);
385		}
386		if let Some(event) = &self.event {
387			instances.extend_from_slice(&event.instances[..]);
388		}
389		if let Some(error) = &self.error {
390			instances.extend_from_slice(&error.instances[..]);
391		}
392		if let Some(inherent) = &self.inherent {
393			instances.extend_from_slice(&inherent.instances[..]);
394		}
395		if let Some(origin) = &self.origin {
396			instances.extend_from_slice(&origin.instances[..]);
397		}
398		if let Some(genesis_config) = &self.genesis_config {
399			instances.extend_from_slice(&genesis_config.instances[..]);
400		}
401		if let Some(genesis_build) = &self.genesis_build {
402			genesis_build.instances.as_ref().map(|i| instances.extend_from_slice(&i));
403		}
404		if let Some(extra_constants) = &self.extra_constants {
405			instances.extend_from_slice(&extra_constants.instances[..]);
406		}
407
408		let mut errors = instances.into_iter().filter_map(|instances| {
409			if instances.has_instance == self.config.has_instance {
410				return None
411			}
412			let msg = if self.config.has_instance {
413				"Invalid generic declaration, trait is defined with instance but generic use none"
414			} else {
415				"Invalid generic declaration, trait is defined without instance but generic use \
416						some"
417			};
418			Some(syn::Error::new(instances.span, msg))
419		});
420
421		if let Some(mut first_error) = errors.next() {
422			for error in errors {
423				first_error.combine(error)
424			}
425			Err(first_error)
426		} else {
427			Ok(())
428		}
429	}

488/// Some generic kind for type which can be not generic, or generic over config,
489/// or generic over config and instance, but not generic only over instance.
490pub enum GenericKind {
491	None,
492	Config,
493	ConfigAndInstance,
494}
495
496impl GenericKind {
497	/// Return Err if it is only generics over instance but not over config.
498	pub fn from_gens(has_config: bool, has_instance: bool) -> Result<Self, ()> {
499		match (has_config, has_instance) {
500			(false, false) => Ok(GenericKind::None),
501			(true, false) => Ok(GenericKind::Config),
502			(true, true) => Ok(GenericKind::ConfigAndInstance),
503			(false, true) => Err(()),
504		}
505	}
506
507	/// Return the generic to be used when using the type.
508	///
509	/// Depending on its definition it can be: ``, `T` or `T, I`
510	pub fn type_use_gen(&self, span: proc_macro2::Span) -> proc_macro2::TokenStream {
511		match self {
512			GenericKind::None => quote::quote!(),
513			GenericKind::Config => quote::quote_spanned!(span => T),
514			GenericKind::ConfigAndInstance => quote::quote_spanned!(span => T, I),
515		}
516	}
517
518	/// Return the generic to be used in `impl<..>` when implementing on the type.
519	pub fn type_impl_gen(&self, span: proc_macro2::Span) -> proc_macro2::TokenStream {
520		match self {
521			GenericKind::None => quote::quote!(),
522			GenericKind::Config => quote::quote_spanned!(span => T: Config),
523			GenericKind::ConfigAndInstance => {
524				quote::quote_spanned!(span => T: Config<I>, I: 'static)
525			},
526		}
527	}
528
529	/// Return whereas the type has some generic.
530	pub fn is_generic(&self) -> bool {
531		match self {
532			GenericKind::None => false,
533			GenericKind::Config | GenericKind::ConfigAndInstance => true,
534		}
535	}
536}

564/// Parse attributes for item in pallet module
565/// syntax must be `pallet::` (e.g. `#[pallet::config]`)
566enum PalletAttr {
567	Config(proc_macro2::Span, bool),
568	Pallet(proc_macro2::Span),
569	Hooks(proc_macro2::Span),
570	/// A `#[pallet::call]` with optional attributes to specialize the behaviour.
571	///
572	/// # Attributes
573	///
574	/// Each attribute `attr` can take the form of `#[pallet::call(attr = …)]` or
575	/// `#[pallet::call(attr(…))]`. The possible attributes are:
576	///
577	/// ## `weight`
578	///
579	/// Can be used to reduce the repetitive weight annotation in the trivial case. It accepts one
580	/// argument that is expected to be an implementation of the `WeightInfo` or something that
581	/// behaves syntactically equivalent. This allows to annotate a `WeightInfo` for all the calls.
582	/// Now each call does not need to specify its own `#[pallet::weight]` but can instead use the
583	/// one from the `#[pallet::call]` definition. So instead of having to write it on each call:
584	///
585	/// ```ignore
586	/// #[pallet::call]
587	/// impl<T: Config> Pallet<T> {
588	///     #[pallet::weight(T::WeightInfo::create())]
589	///     pub fn create(
590	/// ```
591	/// you can now omit it on the call itself, if the name of the weigh function matches the call:
592	///
593	/// ```ignore
594	/// #[pallet::call(weight = <T as crate::Config>::WeightInfo)]
595	/// impl<T: Config> Pallet<T> {
596	///     pub fn create(
597	/// ```
598	///
599	/// It is possible to use this syntax together with instantiated pallets by using `Config<I>`
600	/// instead.
601	///
602	/// ### Dev Mode
603	///
604	/// Normally the `dev_mode` sets all weights of calls without a `#[pallet::weight]` annotation
605	/// to zero. Now when there is a `weight` attribute on the `#[pallet::call]`, then that is used
606	/// instead of the zero weight. So to say: it works together with `dev_mode`.
607	RuntimeCall(Option<InheritedCallWeightAttr>, proc_macro2::Span),
608	Error(proc_macro2::Span),
609	Tasks(proc_macro2::Span),
610	TaskList(proc_macro2::Span),
611	TaskCondition(proc_macro2::Span),
612	TaskIndex(proc_macro2::Span),
613	RuntimeTask(proc_macro2::Span),
614	RuntimeEvent(proc_macro2::Span),
615	RuntimeOrigin(proc_macro2::Span),
616	Inherent(proc_macro2::Span),
617	Storage(proc_macro2::Span),
618	GenesisConfig(proc_macro2::Span),
619	GenesisBuild(proc_macro2::Span),
620	ValidateUnsigned(proc_macro2::Span),
621	TypeValue(proc_macro2::Span),
622	ExtraConstants(proc_macro2::Span),
623	Composite(proc_macro2::Span),
624}

626impl PalletAttr {
627	fn span(&self) -> proc_macro2::Span {
628		match self {
629			Self::Config(span, _) => *span,
630			Self::Pallet(span) => *span,
631			Self::Hooks(span) => *span,
632			Self::Tasks(span) => *span,
633			Self::TaskCondition(span) => *span,
634			Self::TaskIndex(span) => *span,
635			Self::TaskList(span) => *span,
636			Self::Error(span) => *span,
637			Self::RuntimeTask(span) => *span,
638			Self::RuntimeCall(_, span) => *span,
639			Self::RuntimeEvent(span) => *span,
640			Self::RuntimeOrigin(span) => *span,
641			Self::Inherent(span) => *span,
642			Self::Storage(span) => *span,
643			Self::GenesisConfig(span) => *span,
644			Self::GenesisBuild(span) => *span,
645			Self::ValidateUnsigned(span) => *span,
646			Self::TypeValue(span) => *span,
647			Self::ExtraConstants(span) => *span,
648			Self::Composite(span) => *span,
649		}
650	}

653impl syn::parse::Parse for PalletAttr {
654	fn parse(input: syn::parse::ParseStream) -> syn::Result<Self> {
655		input.parse::<syn::Token![#]>()?;
656		let content;
657		syn::bracketed!(content in input);
658		content.parse::<keyword::pallet>()?;
659		content.parse::<syn::Token![::]>()?;
660
661		let lookahead = content.lookahead1();
662		if lookahead.peek(keyword::config) {
663			let span = content.parse::<keyword::config>()?.span();
664			let with_default = content.peek(syn::token::Paren);
665			if with_default {
666				let inside_config;
667				let _paren = syn::parenthesized!(inside_config in content);
668				inside_config.parse::<keyword::with_default>()?;
669			}
670			Ok(PalletAttr::Config(span, with_default))
671		} else if lookahead.peek(keyword::pallet) {
672			Ok(PalletAttr::Pallet(content.parse::<keyword::pallet>()?.span()))
673		} else if lookahead.peek(keyword::hooks) {
674			Ok(PalletAttr::Hooks(content.parse::<keyword::hooks>()?.span()))
675		} else if lookahead.peek(keyword::call) {
676			let span = content.parse::<keyword::call>().expect("peeked").span();
677			let attr = match content.is_empty() {
678				true => None,
679				false => Some(InheritedCallWeightAttr::parse(&content)?),
680			};
681			Ok(PalletAttr::RuntimeCall(attr, span))
682		} else if lookahead.peek(keyword::tasks_experimental) {
683			Ok(PalletAttr::Tasks(content.parse::<keyword::tasks_experimental>()?.span()))
684		} else if lookahead.peek(keyword::task_enum) {
685			Ok(PalletAttr::RuntimeTask(content.parse::<keyword::task_enum>()?.span()))
686		} else if lookahead.peek(keyword::task_condition) {
687			Ok(PalletAttr::TaskCondition(content.parse::<keyword::task_condition>()?.span()))
688		} else if lookahead.peek(keyword::task_index) {
689			Ok(PalletAttr::TaskIndex(content.parse::<keyword::task_index>()?.span()))
690		} else if lookahead.peek(keyword::task_list) {
691			Ok(PalletAttr::TaskList(content.parse::<keyword::task_list>()?.span()))
692		} else if lookahead.peek(keyword::error) {
693			Ok(PalletAttr::Error(content.parse::<keyword::error>()?.span()))
694		} else if lookahead.peek(keyword::event) {
695			Ok(PalletAttr::RuntimeEvent(content.parse::<keyword::event>()?.span()))
696		} else if lookahead.peek(keyword::origin) {
697			Ok(PalletAttr::RuntimeOrigin(content.parse::<keyword::origin>()?.span()))
698		} else if lookahead.peek(keyword::inherent) {
699			Ok(PalletAttr::Inherent(content.parse::<keyword::inherent>()?.span()))
700		} else if lookahead.peek(keyword::storage) {
701			Ok(PalletAttr::Storage(content.parse::<keyword::storage>()?.span()))
702		} else if lookahead.peek(keyword::genesis_config) {
703			Ok(PalletAttr::GenesisConfig(content.parse::<keyword::genesis_config>()?.span()))
704		} else if lookahead.peek(keyword::genesis_build) {
705			Ok(PalletAttr::GenesisBuild(content.parse::<keyword::genesis_build>()?.span()))
706		} else if lookahead.peek(keyword::validate_unsigned) {
707			Ok(PalletAttr::ValidateUnsigned(content.parse::<keyword::validate_unsigned>()?.span()))
708		} else if lookahead.peek(keyword::type_value) {
709			Ok(PalletAttr::TypeValue(content.parse::<keyword::type_value>()?.span()))
710		} else if lookahead.peek(keyword::extra_constants) {
711			Ok(PalletAttr::ExtraConstants(content.parse::<keyword::extra_constants>()?.span()))
712		} else if lookahead.peek(keyword::composite_enum) {
713			Ok(PalletAttr::Composite(content.parse::<keyword::composite_enum>()?.span()))
714		} else {
715			Err(lookahead.error())
716		}
717	}
718}

720/// The optional weight annotation on a `#[pallet::call]` like `#[pallet::call(weight($type))]`.
721#[derive(Clone)]
722pub struct InheritedCallWeightAttr {
723	pub typename: syn::Type,
724}
725
726impl syn::parse::Parse for InheritedCallWeightAttr {
727	// Parses `(weight($type))` or `(weight = $type)`.
728	fn parse(input: syn::parse::ParseStream) -> syn::Result<Self> {
729		let content;
730		syn::parenthesized!(content in input);
731		content.parse::<keyword::weight>()?;
732		let lookahead = content.lookahead1();
733
734		let buffer = if lookahead.peek(syn::token::Paren) {
735			let inner;
736			syn::parenthesized!(inner in content);
737			inner
738		} else if lookahead.peek(syn::Token![=]) {
739			content.parse::<syn::Token![=]>().expect("peeked");
740			content
741		} else {
742			return Err(lookahead.error())
743		};
744
745		Ok(Self { typename: buffer.parse()? })
746	}
