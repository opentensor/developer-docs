53/// Counter to generate a relatively unique identifier for macros. This is necessary because
54/// declarative macros gets hoisted to the crate root, which shares the namespace with other pallets
55/// containing the very same macros.
56struct Counter(u64);
57
58impl Counter {
59	fn inc(&mut self) -> u64 {
60		let ret = self.0;
61		self.0 += 1;
62		ret
63	}
64}

66/// Get the value from the given environment variable set by cargo.
67///
68/// The value is parsed into the requested destination type.
69fn get_cargo_env_var<T: FromStr>(version_env: &str) -> std::result::Result<T, ()> {
70	let version = std::env::var(version_env)
71		.unwrap_or_else(|_| panic!("`{}` is always set by cargo; qed", version_env));
72
73	T::from_str(&version).map_err(drop)
74}

82/// Construct a runtime, with the given name and the given pallets.
83///
84/// NOTE: A new version of this macro is available at `frame_support::runtime`. This macro will
85/// soon be deprecated. Please use the new macro instead.
86///
87/// The parameters here are specific types for `Block`, `NodeBlock`, and `UncheckedExtrinsic`
88/// and the pallets that are used by the runtime.
89/// `Block` is the block type that is used in the runtime and `NodeBlock` is the block type
90/// that is used in the node. For instance they can differ in the extrinsics type.
91///
92/// # Example:
93///
94/// ```ignore
95/// construct_runtime!(
96///     pub enum Runtime where
97///         Block = Block,
98///         NodeBlock = node::Block,
99///         UncheckedExtrinsic = UncheckedExtrinsic
100///     {
101///         System: frame_system::{Pallet, Call, Event<T>, Config<T>} = 0,
102///         Test: path::to::test::{Pallet, Call} = 1,
103///
104///         // Pallets with instances.
105///         Test2_Instance1: test2::<Instance1>::{Pallet, Call, Storage, Event<T, I>, Config<T, I>, Origin<T, I>},
106///         Test2_DefaultInstance: test2::{Pallet, Call, Storage, Event<T>, Config<T>, Origin<T>} = 4,
107///
108///         // Pallets declared with `pallet` attribute macro: no need to define the parts
109///         Test3_Instance1: test3::<Instance1>,
110///         Test3_DefaultInstance: test3,
111///
112///         // with `exclude_parts` keyword some part can be excluded.
113///         Test4_Instance1: test4::<Instance1> exclude_parts { Call, Origin },
114///         Test4_DefaultInstance: test4 exclude_parts { Storage },
115///
116///         // with `use_parts` keyword, a subset of the pallet parts can be specified.
117///         Test4_Instance1: test4::<Instance1> use_parts { Pallet, Call},
118///         Test4_DefaultInstance: test4 use_parts { Pallet },
119///     }
120/// )
121/// ```
122///
123/// Each pallet is declared as such:
124/// * `Identifier`: name given to the pallet that uniquely identifies it.
125///
126/// * `:`: colon separator
127///
128/// * `path::to::pallet`: identifiers separated by colons which declare the path to a pallet
129///   definition.
130///
131/// * `::<InstanceN>` optional: specify the instance of the pallet to use. If not specified it will
132///   use the default instance (or the only instance in case of non-instantiable pallets).
133///
134/// * `::{ Part1, Part2<T>, .. }` optional if pallet declared with `frame_support::pallet`: Comma
135///   separated parts declared with their generic. If a pallet is declared with
136///   `frame_support::pallet` macro then the parts can be automatically derived if not explicitly
137///   provided. We provide support for the following module parts in a pallet:
138///
139///   - `Pallet` - Required for all pallets
140///   - `Call` - If the pallet has callable functions
141///   - `Storage` - If the pallet uses storage
142///   - `Event` or `Event<T>` (if the event is generic) - If the pallet emits events
143///   - `Origin` or `Origin<T>` (if the origin is generic) - If the pallet has instantiable origins
144///   - `Config` or `Config<T>` (if the config is generic) - If the pallet builds the genesis
145///     storage with `GenesisConfig`
146///   - `Inherent` - If the pallet provides/can check inherents.
147///   - `ValidateUnsigned` - If the pallet validates unsigned extrinsics.
148///
149///   It is important to list these parts here to export them correctly in the metadata or to make
150/// the pallet usable in the runtime.
151///
152/// * `exclude_parts { Part1, Part2 }` optional: comma separated parts without generics. I.e. one of
153///   `Pallet`, `Call`, `Storage`, `Event`, `Origin`, `Config`, `Inherent`, `ValidateUnsigned`. It
154///   is incompatible with `use_parts`. This specifies the part to exclude. In order to select
155///   subset of the pallet parts.
156///
157///   For example excluding the part `Call` can be useful if the runtime doesn't want to make the
158///   pallet calls available.
159///
160/// * `use_parts { Part1, Part2 }` optional: comma separated parts without generics. I.e. one of
161///   `Pallet`, `Call`, `Storage`, `Event`, `Origin`, `Config`, `Inherent`, `ValidateUnsigned`. It
162///   is incompatible with `exclude_parts`. This specifies the part to use. In order to select a
163///   subset of the pallet parts.
164///
165///   For example not using the part `Call` can be useful if the runtime doesn't want to make the
166///   pallet calls available.
167///
168/// * `= $n` optional: number to define at which index the pallet variants in `OriginCaller`, `Call`
169///   and `Event` are encoded, and to define the ModuleToIndex value.
170///
171///   if `= $n` is not given, then index is resolved in the same way as fieldless enum in Rust
172///   (i.e. incrementally from previous index):
173///   ```nocompile
174///   pallet1 .. = 2,
175///   pallet2 .., // Here pallet2 is given index 3
176///   pallet3 .. = 0,
177///   pallet4 .., // Here pallet4 is given index 1
178///   ```
179///
180/// # Note
181///
182/// The population of the genesis storage depends on the order of pallets. So, if one of your
183/// pallets depends on another pallet, the pallet that is depended upon needs to come before
184/// the pallet depending on it.
185///
186/// # Type definitions
187///
188/// * The macro generates a type alias for each pallet to their `Pallet`. E.g. `type System =
189///   frame_system::Pallet<Runtime>`
190#[proc_macro]
191pub fn construct_runtime(input: TokenStream) -> TokenStream {
192	construct_runtime::construct_runtime(input)
193}

267/// Execute the annotated function in a new storage transaction.
268///
269/// The return type of the annotated function must be `Result`. All changes to storage performed
270/// by the annotated function are discarded if it returns `Err`, or committed if `Ok`.
271///
272/// # Example
273///
274/// ```nocompile
275/// #[transactional]
276/// fn value_commits(v: u32) -> result::Result<u32, &'static str> {
277/// 	Value::set(v);
278/// 	Ok(v)
279/// }
280///
281/// #[transactional]
282/// fn value_rollbacks(v: u32) -> result::Result<u32, &'static str> {
283/// 	Value::set(v);
284/// 	Err("nah")
285/// }
286/// ```
287#[proc_macro_attribute]
288pub fn transactional(attr: TokenStream, input: TokenStream) -> TokenStream {
289	transactional::transactional(attr, input).unwrap_or_else(|e| e.to_compile_error().into())
290}

318/// Derive [`Debug`], if `std` is enabled it uses `frame_support::DebugNoBound`, if `std` is not
319/// enabled it just returns `"<wasm:stripped>"`.
320/// This behaviour is useful to prevent bloating the runtime WASM blob from unneeded code.
321#[proc_macro_derive(RuntimeDebugNoBound)]
322pub fn derive_runtime_debug_no_bound(input: TokenStream) -> TokenStream {
323	if cfg!(any(feature = "std", feature = "try-runtime")) {
324		no_bound::debug::derive_debug_no_bound(input)
325	} else {
326		let input = syn::parse_macro_input!(input as syn::DeriveInput);
327
328		let name = &input.ident;
329		let (impl_generics, ty_generics, where_clause) = input.generics.split_for_impl();
330
331		quote::quote!(
332			const _: () = {
333				impl #impl_generics ::core::fmt::Debug for #name #ty_generics #where_clause {
334					fn fmt(&self, fmt: &mut ::core::fmt::Formatter) -> core::fmt::Result {
335						fmt.write_str("<wasm:stripped>")
336					}
337				}
338			};
339		)
340		.into()
341	}
342}

462/// This attribute can be used to derive a full implementation of a trait based on a local partial
463/// impl and an external impl containing defaults that can be overridden in the local impl.
464///
465/// For a full end-to-end example, see [below](#use-case-auto-derive-test-pallet-config-traits).
466///
467/// # Usage
468///
469/// The attribute should be attached to an impl block (strictly speaking a `syn::ItemImpl`) for
470/// which we want to inject defaults in the event of missing trait items in the block.
471///
472/// The attribute minimally takes a single `default_impl_path` argument, which should be the module
473/// path to an impl registered via [`#[register_default_impl]`](`macro@register_default_impl`) that
474/// contains the default trait items we want to potentially inject, with the general form:
475///
476/// ```ignore
477/// #[derive_impl(default_impl_path)]
478/// impl SomeTrait for SomeStruct {
479///     ...
480/// }
481/// ```
482///
483/// Optionally, a `disambiguation_path` can be specified as follows by providing `as path::here`
484/// after the `default_impl_path`:
485///
486/// ```ignore
487/// #[derive_impl(default_impl_path as disambiguation_path)]
488/// impl SomeTrait for SomeStruct {
489///     ...
490/// }
491/// ```
492///
493/// The `disambiguation_path`, if specified, should be the path to a trait that will be used to
494/// qualify all default entries that are injected into the local impl. For example if your
495/// `default_impl_path` is `some::path::TestTraitImpl` and your `disambiguation_path` is
496/// `another::path::DefaultTrait`, any items injected into the local impl will be qualified as
497/// `<some::path::TestTraitImpl as another::path::DefaultTrait>::specific_trait_item`.
498///
499/// If you omit the `as disambiguation_path` portion, the `disambiguation_path` will internally
500/// default to `A` from the `impl A for B` part of the default impl. This is useful for scenarios
501/// where all of the relevant types are already in scope via `use` statements.
502///
503/// In case the `default_impl_path` is scoped to a different module such as
504/// `some::path::TestTraitImpl`, the same scope is assumed for the `disambiguation_path`, i.e.
505/// `some::A`. This enables the use of `derive_impl` attribute without having to specify the
506/// `disambiguation_path` in most (if not all) uses within FRAME's context.
507///
508/// Conversely, the `default_impl_path` argument is required and cannot be omitted.
509///
510/// Optionally, `no_aggregated_types` can be specified as follows:
511///
512/// ```ignore
513/// #[derive_impl(default_impl_path as disambiguation_path, no_aggregated_types)]
514/// impl SomeTrait for SomeStruct {
515///     ...
516/// }
517/// ```
518///
519/// If specified, this indicates that the aggregated types (as denoted by impl items
520/// attached with [`#[inject_runtime_type]`]) should not be injected with the respective concrete
521/// types. By default, all such types are injected.
522///
523/// You can also make use of `#[pallet::no_default]` on specific items in your default impl that you
524/// want to ensure will not be copied over but that you nonetheless want to use locally in the
525/// context of the foreign impl and the pallet (or context) in which it is defined.
526///
527/// ## Use-Case Example: Auto-Derive Test Pallet Config Traits
528///
529/// The `#[derive_imp(..)]` attribute can be used to derive a test pallet `Config` based on an
530/// existing pallet `Config` that has been marked with
531/// [`#[pallet::config(with_default)]`](`macro@config`) (which under the hood, generates a
532/// `DefaultConfig` trait in the pallet in which the macro was invoked).
533///
534/// In this case, the `#[derive_impl(..)]` attribute should be attached to an `impl` block that
535/// implements a compatible `Config` such as `frame_system::Config` for a test/mock runtime, and
536/// should receive as its first argument the path to a `DefaultConfig` impl that has been registered
537/// via [`#[register_default_impl]`](`macro@register_default_impl`), and as its second argument, the
538/// path to the auto-generated `DefaultConfig` for the existing pallet `Config` we want to base our
539/// test config off of.
540///
541/// The following is what the `basic` example pallet would look like with a default testing config:
542///
543/// ```ignore
544/// #[derive_impl(frame_system::config_preludes::TestDefaultConfig as frame_system::pallet::DefaultConfig)]
545/// impl frame_system::Config for Test {
546///     // These are all defined by system as mandatory.
547///     type BaseCallFilter = frame_support::traits::Everything;
548///     type RuntimeEvent = RuntimeEvent;
549///     type RuntimeCall = RuntimeCall;
550///     type RuntimeOrigin = RuntimeOrigin;
551///     type OnSetCode = ();
552///     type PalletInfo = PalletInfo;
553///     type Block = Block;
554///     // We decide to override this one.
555///     type AccountData = pallet_balances::AccountData<u64>;
556/// }
557/// ```
558///
559/// where `TestDefaultConfig` was defined and registered as follows:
560/// ```ignore
561/// pub struct TestDefaultConfig;
562///
563/// #[register_default_impl(TestDefaultConfig)]
564/// impl DefaultConfig for TestDefaultConfig {
565///     type Version = ();
566///     type BlockWeights = ();
567///     type BlockLength = ();
568///     type DbWeight = ();
569///     type Nonce = u64;
570///     type BlockNumber = u64;
571///     type Hash = sp_core::hash::H256;
572///     type Hashing = sp_runtime::traits::BlakeTwo256;
573///     type AccountId = AccountId;
574///     type Lookup = IdentityLookup<AccountId>;
575///     type BlockHashCount = frame_support::traits::ConstU64<10>;
576///     type AccountData = u32;
577///     type OnNewAccount = ();
578///     type OnKilledAccount = ();
579///     type SystemWeightInfo = ();
580///     type SS58Prefix = ();
581///     type MaxConsumers = frame_support::traits::ConstU32<16>;
582/// }
583/// ```
584///
585/// The above call to `derive_impl` would expand to roughly the following:
586/// ```ignore
587/// impl frame_system::Config for Test {
588///     use frame_system::config_preludes::TestDefaultConfig;
589///     use frame_system::pallet::DefaultConfig;
590///
591///     type BaseCallFilter = frame_support::traits::Everything;
592///     type RuntimeEvent = RuntimeEvent;
593///     type RuntimeCall = RuntimeCall;
594///     type RuntimeOrigin = RuntimeOrigin;
595///     type OnSetCode = ();
596///     type PalletInfo = PalletInfo;
597///     type Block = Block;
598///     type AccountData = pallet_balances::AccountData<u64>;
599///     type Version = <TestDefaultConfig as DefaultConfig>::Version;
600///     type BlockWeights = <TestDefaultConfig as DefaultConfig>::BlockWeights;
601///     type BlockLength = <TestDefaultConfig as DefaultConfig>::BlockLength;
602///     type DbWeight = <TestDefaultConfig as DefaultConfig>::DbWeight;
603///     type Nonce = <TestDefaultConfig as DefaultConfig>::Nonce;
604///     type BlockNumber = <TestDefaultConfig as DefaultConfig>::BlockNumber;
605///     type Hash = <TestDefaultConfig as DefaultConfig>::Hash;
606///     type Hashing = <TestDefaultConfig as DefaultConfig>::Hashing;
607///     type AccountId = <TestDefaultConfig as DefaultConfig>::AccountId;
608///     type Lookup = <TestDefaultConfig as DefaultConfig>::Lookup;
609///     type BlockHashCount = <TestDefaultConfig as DefaultConfig>::BlockHashCount;
610///     type OnNewAccount = <TestDefaultConfig as DefaultConfig>::OnNewAccount;
611///     type OnKilledAccount = <TestDefaultConfig as DefaultConfig>::OnKilledAccount;
612///     type SystemWeightInfo = <TestDefaultConfig as DefaultConfig>::SystemWeightInfo;
613///     type SS58Prefix = <TestDefaultConfig as DefaultConfig>::SS58Prefix;
614///     type MaxConsumers = <TestDefaultConfig as DefaultConfig>::MaxConsumers;
615/// }
616/// ```
617///
618/// You can then use the resulting `Test` config in test scenarios.
619///
620/// Note that items that are _not_ present in our local `DefaultConfig` are automatically copied
621/// from the foreign trait (in this case `TestDefaultConfig`) into the local trait impl (in this
622/// case `Test`), unless the trait item in the local trait impl is marked with
623/// [`#[pallet::no_default]`](`macro@no_default`), in which case it cannot be overridden, and any
624/// attempts to do so will result in a compiler error.
625///
626/// See `frame/examples/default-config/tests.rs` for a runnable end-to-end example pallet that makes
627/// use of `derive_impl` to derive its testing config.
628///
629/// See [here](`macro@config`) for more information and caveats about the auto-generated
630/// `DefaultConfig` trait.
631///
632/// ## Optional Conventions
633///
634/// Note that as an optional convention, we encourage creating a `config_preludes` module inside of
635/// your pallet. This is the convention we follow for `frame_system`'s `TestDefaultConfig` which, as
636/// shown above, is located at `frame_system::config_preludes::TestDefaultConfig`. This is just a
637/// suggested convention -- there is nothing in the code that expects modules with these names to be
638/// in place, so there is no imperative to follow this pattern unless desired.
639///
640/// In `config_preludes`, you can place types named like:
641///
642/// * `TestDefaultConfig`
643/// * `ParachainDefaultConfig`
644/// * `SolochainDefaultConfig`
645///
646/// Signifying in which context they can be used.
647///
648/// # Advanced Usage
649///
650/// ## Expansion
651///
652/// The `#[derive_impl(default_impl_path as disambiguation_path)]` attribute will expand to the
653/// local impl, with any extra items from the foreign impl that aren't present in the local impl
654/// also included. In the case of a colliding trait item, the version of the item that exists in the
655/// local impl will be retained. All imported items are qualified by the `disambiguation_path`, as
656/// discussed above.
657///
658/// ## Handling of Unnamed Trait Items
659///
660/// Items that lack a `syn::Ident` for whatever reason are first checked to see if they exist,
661/// verbatim, in the local/destination trait before they are copied over, so you should not need to
662/// worry about collisions between identical unnamed items.
663#[import_tokens_attr_verbatim {
664    format!(
665        "{}::macro_magic",
666        match generate_access_from_frame_or_crate("frame-support") {
667            Ok(path) => Ok(path),
668            Err(_) => generate_access_from_frame_or_crate("polkadot-sdk-frame"),
669        }
670        .expect("Failed to find either `frame-support` or `polkadot-sdk-frame` in `Cargo.toml` dependencies.")
671        .to_token_stream()
672        .to_string()
673    )
674}]

708/// Attach this attribute to an impl statement that you want to use with
709/// [`#[derive_impl(..)]`](`macro@derive_impl`).
710///
711/// You must also provide an identifier/name as the attribute's argument. This is the name you
712/// must provide to [`#[derive_impl(..)]`](`macro@derive_impl`) when you import this impl via
713/// the `default_impl_path` argument. This name should be unique at the crate-level.
714///
715/// ## Example
716///
717/// ```ignore
718/// pub struct ExampleTestDefaultConfig;
719///
720/// #[register_default_impl(ExampleTestDefaultConfig)]
721/// impl DefaultConfig for ExampleTestDefaultConfig {
722/// 	type Version = ();
723/// 	type BlockWeights = ();
724/// 	type BlockLength = ();
725/// 	...
726/// 	type SS58Prefix = ();
727/// 	type MaxConsumers = frame_support::traits::ConstU32<16>;
728/// }
729/// ```
730///
731/// ## Advanced Usage
732///
733/// This macro acts as a thin wrapper around macro_magic's `#[export_tokens]`. See the docs
734/// [here](https://docs.rs/macro_magic/latest/macro_magic/attr.export_tokens.html) for more
735/// info.
736///
737/// There are some caveats when applying a `use` statement to bring a
738/// `#[register_default_impl]` item into scope. If you have a `#[register_default_impl]`
739/// defined in `my_crate::submodule::MyItem`, it is currently not sufficient to do something
740/// like:
741///
742/// ```ignore
743/// use my_crate::submodule::MyItem;
744/// #[derive_impl(MyItem as Whatever)]
745/// ```
746///
747/// This will fail with a mysterious message about `__export_tokens_tt_my_item` not being
748/// defined.
749///
750/// You can, however, do any of the following:
751/// ```ignore
752/// // partial path works
753/// use my_crate::submodule;
754/// #[derive_impl(submodule::MyItem as Whatever)]
755/// ```
756/// ```ignore
757/// // full path works
758/// #[derive_impl(my_crate::submodule::MyItem as Whatever)]
759/// ```
760/// ```ignore
761/// // wild-cards work
762/// use my_crate::submodule::*;
763/// #[derive_impl(MyItem as Whatever)]
764/// ```
765#[proc_macro_attribute]
766pub fn register_default_impl(attrs: TokenStream, tokens: TokenStream) -> TokenStream {
767	// ensure this is a impl statement
768	let item_impl = syn::parse_macro_input!(tokens as ItemImpl);
769
770	// internally wrap macro_magic's `#[export_tokens]` macro
771	match macro_magic::mm_core::export_tokens_internal(
772		attrs,
773		item_impl.to_token_stream(),
774		true,
775		false,
776	) {
777		Ok(tokens) => tokens.into(),
778		Err(err) => err.to_compile_error().into(),
779	}
780}

782/// The optional attribute `#[inject_runtime_type]` can be attached to `RuntimeCall`,
783/// `RuntimeEvent`, `RuntimeOrigin` or `PalletInfo` in an impl statement that has
784/// `#[register_default_impl]` attached to indicate that this item is generated by
785/// `construct_runtime`.
786///
787/// Attaching this attribute to such an item ensures that the combined impl generated via
788/// [`#[derive_impl(..)]`](macro@derive_impl) will use the correct type auto-generated by
789/// `construct_runtime!`.
790#[doc = docify::embed!("examples/proc_main/inject_runtime_type.rs", derive_impl_works_with_runtime_type_injection)]
791///
792/// However, if `no_aggregated_types` is specified while using
793/// [`#[derive_impl(..)]`](macro@derive_impl), then these items are attached verbatim to the
794/// combined impl.
795#[doc = docify::embed!("examples/proc_main/inject_runtime_type.rs", derive_impl_works_with_no_aggregated_types)]
796#[proc_macro_attribute]
797pub fn inject_runtime_type(_: TokenStream, tokens: TokenStream) -> TokenStream {
798	let item = tokens.clone();
799	let item = syn::parse_macro_input!(item as TraitItemType);
800	if item.ident != "RuntimeCall" &&
801		item.ident != "RuntimeEvent" &&
802		item.ident != "RuntimeTask" &&
803		item.ident != "RuntimeOrigin" &&
804		item.ident != "RuntimeHoldReason" &&
805		item.ident != "RuntimeFreezeReason" &&
806		item.ident != "RuntimeParameters" &&
807		item.ident != "PalletInfo"
808	{
809		return syn::Error::new_spanned(
810			item,
811			"`#[inject_runtime_type]` can only be attached to `RuntimeCall`, `RuntimeEvent`, \
812			`RuntimeTask`, `RuntimeOrigin`, `RuntimeParameters` or `PalletInfo`",
813		)
814		.to_compile_error()
815		.into()
816	}
817	tokens
818}

1178	let tokens_clone = tokens.clone();
1179	// ensure this can only be attached to a module
1180	let _mod = parse_macro_input!(tokens_clone as ItemMod);
1181

1189///
1190/// ---
1191///
1192/// **Rust-Analyzer users**: See the documentation of the Rust item in
1193/// `frame_support::pallet_macros::import_section`.
1194#[import_tokens_attr {
1195    format!(
1196        "{}::macro_magic",
1197        match generate_access_from_frame_or_crate("frame-support") {
1198            Ok(path) => Ok(path),
1199            Err(_) => generate_access_from_frame_or_crate("polkadot-sdk-frame"),
1200        }
1201        .expect("Failed to find either `frame-support` or `polkadot-sdk-frame` in `Cargo.toml` dependencies.")
1202        .to_token_stream()
1203        .to_string()
1204    )
1205}]
1206#[proc_macro_attribute]
1207pub fn import_section(attr: TokenStream, tokens: TokenStream) -> TokenStream {
1208	let foreign_mod = parse_macro_input!(attr as ItemMod);
1209	let mut internal_mod = parse_macro_input!(tokens as ItemMod);
1210
1211	// check that internal_mod is a pallet module
1212	if !internal_mod.attrs.iter().any(|attr| {
1213		if let Some(last_seg) = attr.path().segments.last() {
1214			last_seg.ident == "pallet"
1215		} else {
1216			false
1217		}
1218	}) {
1219		return Error::new(
1220			internal_mod.ident.span(),
1221			"`#[import_section]` can only be applied to a valid pallet module",
1222		)
1223		.to_compile_error()
1224		.into()
1225	}
1226
1227	if let Some(ref mut content) = internal_mod.content {
1228		if let Some(foreign_content) = foreign_mod.content {
1229			content.1.extend(foreign_content.1);
1230		}
1231	}
1232
1233	quote! {
1234		#internal_mod
1235	}
1236	.into()
1237}

1240///
1241/// # Example:
1242#[doc = docify::embed!("examples/proc_main/runtime.rs", runtime_macro)]
1243///
1244/// # Supported Attributes:
1245///
1246/// ## Legacy Ordering
1247///
1248/// An optional attribute can be defined as #[frame_support::runtime(legacy_ordering)] to
1249/// ensure that the order of hooks is same as the order of pallets (and not based on the
1250/// pallet_index). This is to support legacy runtimes and should be avoided for new ones.
1251///
1252/// # Note
1253///
1254/// The population of the genesis storage depends on the order of pallets. So, if one of your
1255/// pallets depends on another pallet, the pallet that is depended upon needs to come before
1256/// the pallet depending on it.
1257///
1258/// # Type definitions
1259///
1260/// * The macro generates a type alias for each pallet to their `Pallet`. E.g. `type System =
1261///   frame_system::Pallet<Runtime>`
1262#[proc_macro_attribute]
1263pub fn runtime(attr: TokenStream, item: TokenStream) -> TokenStream {
1264	runtime::runtime(attr, item)
1265}
