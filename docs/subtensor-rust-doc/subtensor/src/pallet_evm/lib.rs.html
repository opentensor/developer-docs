18//! # EVM Pallet
19//!
20//! The EVM pallet allows unmodified EVM code to be executed in a Substrate-based blockchain.
21//! - [`evm::Config`]
22//!
23//! ## EVM Engine
24//!
25//! The EVM pallet uses [`SputnikVM`](https://github.com/rust-blockchain/evm) as the underlying EVM engine.
26//! The engine is overhauled so that it's [`modular`](https://github.com/corepaper/evm).
27//!
28//! ## Execution Lifecycle
29//!
30//! There are a separate set of accounts managed by the EVM pallet. Substrate based accounts can call the EVM Pallet
31//! to deposit or withdraw balance from the Substrate base-currency into a different balance managed and used by
32//! the EVM pallet. Once a user has populated their balance, they can create and call smart contracts using this pallet.
33//!
34//! There's one-to-one mapping from Substrate accounts and EVM external accounts that is defined by a conversion function.
35//!
36//! ## EVM Pallet vs Ethereum Network
37//!
38//! The EVM pallet should be able to produce nearly identical results compared to the Ethereum mainnet,
39//! including gas cost and balance changes.
40//!
41//! Observable differences include:
42//!
43//! - The available length of block hashes may not be 256 depending on the configuration of the System pallet
44//! in the Substrate runtime.
45//! - Difficulty and coinbase, which do not make sense in this pallet and is currently hard coded to zero.
46//!
47//! We currently do not aim to make unobservable behaviors, such as state root, to be the same. We also don't aim to follow
48//! the exact same transaction / receipt format. However, given one Ethereum transaction and one Substrate account's
49//! private key, one should be able to convert any Ethereum transaction into a transaction compatible with this pallet.
50//!
51//! The gas configurations are configurable. Right now, a pre-defined London hard fork configuration option is provided.

224			origin: OriginFor<T>,
225			source: H160,
226			target: H160,
227			input: Vec<u8>,
228			value: U256,
229			gas_limit: u64,
230			max_fee_per_gas: U256,
231			max_priority_fee_per_gas: Option<U256>,
232			nonce: Option<U256>,
233			access_list: Vec<(H160, Vec<H256>)>,
234		) -> DispatchResultWithPostInfo {
235			T::CallOrigin::ensure_address_origin(&source, origin)?;
236
237			let is_transactional = true;
238			let validate = true;
239			let info = match T::Runner::call(
240				source,
241				target,
242				input,
243				value,
244				gas_limit,
245				Some(max_fee_per_gas),
246				max_priority_fee_per_gas,
247				nonce,
248				access_list,
249				is_transactional,
250				validate,
251				None,
252				None,
253				T::config(),
254			) {
255				Ok(info) => info,
256				Err(e) => {
257					return Err(DispatchErrorWithPostInfo {
258						post_info: PostDispatchInfo {
259							actual_weight: Some(e.weight),
260							pays_fee: Pays::Yes,
261						},
262						error: e.error.into(),
263					})
264				}
265			};
266
267			match info.exit_reason {
268				ExitReason::Succeed(_) => {
269					Pallet::<T>::deposit_event(Event::<T>::Executed { address: target });
270				}
271				_ => {
272					Pallet::<T>::deposit_event(Event::<T>::ExecutedFailed { address: target });
273				}
274			};
275
276			Ok(PostDispatchInfo {
277				actual_weight: {
278					let mut gas_to_weight = T::GasWeightMapping::gas_to_weight(
279						info.used_gas.standard.unique_saturated_into(),
280						true,
281					);
282					if let Some(weight_info) = info.weight_info {
283						if let Some(proof_size_usage) = weight_info.proof_size_usage {
284							*gas_to_weight.proof_size_mut() = proof_size_usage;
285						}
286					}
287					Some(gas_to_weight)
288				},
289				pays_fee: Pays::No,
290			})
291		}

571	#[pallet::genesis_build]
572	impl<T: Config> BuildGenesisConfig for GenesisConfig<T>
573	where
574		U256: UniqueSaturatedInto<BalanceOf<T>>,
575	{
576		fn build(&self) {
577			const MAX_ACCOUNT_NONCE: usize = 100;
578
579			for (address, account) in &self.accounts {
580				let account_id = T::AddressMapping::into_account_id(*address);
581
582				// ASSUME: in one single EVM transaction, the nonce will not increase more than
583				// `u128::max_value()`.
584				for _ in 0..min(
585					MAX_ACCOUNT_NONCE,
586					UniqueSaturatedInto::<usize>::unique_saturated_into(account.nonce),
587				) {
588					frame_system::Pallet::<T>::inc_account_nonce(&account_id);
589				}
590
591				let _ = T::Currency::deposit_creating(
592					&account_id,
593					account.balance.unique_saturated_into(),
594				);
595
596				Pallet::<T>::create_account(*address, account.code.clone());
597
598				for (index, value) in &account.storage {
599					<AccountStorages<T>>::insert(address, index, value);
600				}
601			}
602
603			<WhitelistedCreators<T>>::put(self.whitelisted.clone());
604		}
605	}

661pub trait EnsureAddressOrigin<OuterOrigin> {
662	/// Success return type.
663	type Success;
664
665	/// Perform the origin check.
666	fn ensure_address_origin(
667		address: &H160,
668		origin: OuterOrigin,
669	) -> Result<Self::Success, BadOrigin> {

806/// A mapping function that converts Ethereum gas to Substrate weight
807pub trait GasWeightMapping {
808	fn gas_to_weight(gas: u64, without_base_weight: bool) -> Weight;
809	fn weight_to_gas(weight: Weight) -> u64;
810}
811
812pub struct FixedGasWeightMapping<T>(core::marker::PhantomData<T>);
813impl<T: Config> GasWeightMapping for FixedGasWeightMapping<T> {
814	fn gas_to_weight(gas: u64, without_base_weight: bool) -> Weight {
815		let mut weight = T::WeightPerGas::get().saturating_mul(gas);
816		if without_base_weight {
817			weight = weight.saturating_sub(
818				T::BlockWeights::get()
819					.get(frame_support::dispatch::DispatchClass::Normal)
820					.base_extrinsic,
821			);
822		}
823		// Apply a gas to proof size ratio based on BlockGasLimit
824		let ratio = T::GasLimitPovSizeRatio::get();
825		if ratio > 0 {
826			let proof_size = gas.saturating_div(ratio);
827			*weight.proof_size_mut() = proof_size;
828		}
829
830		weight
831	}
832	fn weight_to_gas(weight: Weight) -> u64 {
833		weight.div(T::WeightPerGas::get().ref_time()).ref_time()
834	}

863	/// Remove an account.
864	pub fn remove_account(address: &H160) {
865		if <AccountCodes<T>>::contains_key(address) {
866			// Remember to call `dec_sufficients` when clearing Suicided.
867			<Suicided<T>>::insert(address, ());
868
869			// In theory, we can always have pre-EIP161 contracts, so we
870			// make sure the account nonce is at least one.
871			let account_id = T::AddressMapping::into_account_id(*address);
872			frame_system::Pallet::<T>::inc_account_nonce(&account_id);
873		}
874
875		<AccountCodes<T>>::remove(address);
876		<AccountCodesMetadata<T>>::remove(address);
877
878		if T::SuicideQuickClearLimit::get() > 0 {
879			#[allow(deprecated)]
880			let res = <AccountStorages<T>>::remove_prefix(address, Some(T::SuicideQuickClearLimit::get()));
881
882			match res {
883				KillStorageResult::AllRemoved(_) => {
884					<Suicided<T>>::remove(address);
885
886					let account_id = T::AddressMapping::into_account_id(*address);
887					let _ = frame_system::Pallet::<T>::dec_sufficients(&account_id);
888				}
889				KillStorageResult::SomeRemaining(_) => (),
890			}
891		}
892	}

894	/// Create an account.
895	pub fn create_account(address: H160, code: Vec<u8>) {
896		if <Suicided<T>>::contains_key(address) {
897			// This branch should never trigger, because when Suicided
898			// contains an address, then its nonce will be at least one,
899			// which causes CreateCollision error in EVM, but we add it
900			// here for safeguard.
901			return;
902		}
903
904		if code.is_empty() {
905			return;
906		}
907
908		if !<AccountCodes<T>>::contains_key(address) {
909			let account_id = T::AddressMapping::into_account_id(address);
910			let _ = frame_system::Pallet::<T>::inc_sufficients(&account_id);
911		}
912
913		// Update metadata.
914		let meta = CodeMetadata::from_code(&code);
915		<AccountCodesMetadata<T>>::insert(address, meta);
916
917		<AccountCodes<T>>::insert(address, code);
918	}

920	/// Get the account metadata (hash and size) from storage if it exists,

928
929		// If code is empty we return precomputed hash for empty code.
930		// We don't store it as this address could get code deployed in the future.

939		}

942
943		<AccountCodesMetadata<T>>::insert(address, meta);

976/// Handle withdrawing, refunding and depositing of transaction fees.
977/// Similar to `OnChargeTransaction` of `pallet_transaction_payment`
978pub trait OnChargeEVMTransaction<T: Config> {
979	type LiquidityInfo: Default;
980
981	/// Before the transaction is executed the payment of the transaction fees
982	/// need to be secured.
983	fn withdraw_fee(who: &H160, fee: EvmBalance) -> Result<Self::LiquidityInfo, Error<T>>;
984
985	/// After the transaction was executed the actual fee can be calculated.
986	/// This function should refund any overpaid fees and optionally deposit
987	/// the corrected amount, and handles the base fee rationing using the provided
988	/// `OnUnbalanced` implementation.
989	/// Returns the `NegativeImbalance` - if any - produced by the priority fee.
990	fn correct_and_deposit_fee(
991		who: &H160,
992		corrected_fee: EvmBalance,
993		base_fee: EvmBalance,
994		already_withdrawn: Self::LiquidityInfo,
995	) -> Self::LiquidityInfo;
996
997	/// Introduced in EIP1559 to handle the priority tip.
998	fn pay_priority_fee(tip: Self::LiquidityInfo);
999}

1001/// Implements the transaction payment for a pallet implementing the `Currency`
1002/// trait (eg. the pallet_balances) using an unbalance handler (implementing
1003/// `OnUnbalanced`).
1004/// Similar to `CurrencyAdapter` of `pallet_transaction_payment`
1005pub struct EVMCurrencyAdapter<C, OU>(core::marker::PhantomData<(C, OU)>);
1006
1007impl<T, C, OU> OnChargeEVMTransaction<T> for EVMCurrencyAdapter<C, OU>
1008where
1009	T: Config,
1010	C: Currency<<T as frame_system::Config>::AccountId>,
1011	C::PositiveImbalance: Imbalance<
1012		<C as Currency<<T as frame_system::Config>::AccountId>>::Balance,
1013		Opposite = C::NegativeImbalance,
1014	>,
1015	C::NegativeImbalance: Imbalance<
1016		<C as Currency<<T as frame_system::Config>::AccountId>>::Balance,
1017		Opposite = C::PositiveImbalance,
1018	>,
1019	OU: OnUnbalanced<NegativeImbalanceOf<C, T>>,
1020	U256: UniqueSaturatedInto<<C as Currency<<T as frame_system::Config>::AccountId>>::Balance>,
1021{
1022	// Kept type as Option to satisfy bound of Default
1023	type LiquidityInfo = Option<NegativeImbalanceOf<C, T>>;
1024
1025	fn withdraw_fee(who: &H160, fee: EvmBalance) -> Result<Self::LiquidityInfo, Error<T>> {
1026		if fee.0.is_zero() {
1027			return Ok(None);
1028		}
1029		let account_id = T::AddressMapping::into_account_id(*who);
1030
1031		// Recalculate fee decimals using BalanceConverter
1032		let fee_sub =
1033			T::BalanceConverter::into_substrate_balance(fee).ok_or(Error::<T>::FeeOverflow)?;
1034
1035		let imbalance = C::withdraw(
1036			&account_id,
1037			fee_sub.0.unique_saturated_into(),
1038			WithdrawReasons::FEE,
1039			ExistenceRequirement::AllowDeath,
1040		)
1041		.map_err(|_| Error::<T>::BalanceLow)?;
1042		Ok(Some(imbalance)) // Returns substrate balance
1043	}
1044
1045	fn correct_and_deposit_fee(
1046		who: &H160,
1047		corrected_fee: EvmBalance,
1048		base_fee: EvmBalance,
1049		already_withdrawn: Self::LiquidityInfo, // Expects substrate balance
1050	) -> Self::LiquidityInfo {
1051		if let Some(paid) = already_withdrawn {
1052			let account_id = T::AddressMapping::into_account_id(*who);
1053
1054			// Convert corrected fee into substrate balance
1055			let corrected_fee_sub = T::BalanceConverter::into_substrate_balance(corrected_fee)
1056				.unwrap_or(SubstrateBalance::from(0u64));
1057
1058			// Calculate how much refund we should return
1059			let refund_amount = paid
1060				.peek()
1061				.saturating_sub(corrected_fee_sub.0.unique_saturated_into());
1062
1063			// refund to the account that paid the fees. If this fails, the
1064			// account might have dropped below the existential balance. In
1065			// that case we don't refund anything.
1066			let refund_imbalance = C::deposit_into_existing(&account_id, refund_amount)
1067				.unwrap_or_else(|_| C::PositiveImbalance::zero());
1068
1069			// Make sure this works with 0 ExistentialDeposit
1070			// https://github.com/paritytech/substrate/issues/10117
1071			// If we tried to refund something, the account still empty and the ED is set to 0,
1072			// we call `make_free_balance_be` with the refunded amount.
1073			let refund_imbalance = if C::minimum_balance().is_zero()
1074				&& refund_amount > C::Balance::zero()
1075				&& C::total_balance(&account_id).is_zero()
1076			{
1077				// Known bug: Substrate tried to refund to a zeroed AccountData, but
1078				// interpreted the account to not exist.
1079				match C::make_free_balance_be(&account_id, refund_amount) {
1080					SignedImbalance::Positive(p) => p,
1081					_ => C::PositiveImbalance::zero(),
1082				}
1083			} else {
1084				refund_imbalance
1085			};
1086
1087			// merge the imbalance caused by paying the fees and refunding parts of it again.
1088			let adjusted_paid = paid
1089				.offset(refund_imbalance)
1090				.same()
1091				.unwrap_or_else(|_| C::NegativeImbalance::zero());
1092
1093			// Convert base fee into substrate balance
1094			let base_fee_sub = T::BalanceConverter::into_substrate_balance(base_fee)
1095				.unwrap_or(SubstrateBalance::from(0u64));
1096
1097			let (base_fee, tip) = adjusted_paid.split(base_fee_sub.0.unique_saturated_into());
1098			// Handle base fee. Can be either burned, rationed, etc ...
1099			OU::on_unbalanced(base_fee);
1100			return Some(tip); // Returns substrate balance
1101		}
1102		None
1103	}
1104
1105	fn pay_priority_fee(tip: Self::LiquidityInfo) {
1106		// Expects substrate balance
1107		// Default Ethereum behaviour: issue the tip to the block author.
1108		if let Some(tip) = tip {
1109			let account_id = T::AddressMapping::into_account_id(<Pallet<T>>::find_author());
1110			let _ = C::deposit_into_existing(&account_id, tip.peek());
1111		}
1112	}
1113}
