66	/// Execute an already validated EVM operation.
67	fn execute<'config, 'precompiles, F, R>(
68		source: H160,
69		value: U256,
70		gas_limit: u64,
71		max_fee_per_gas: Option<U256>,
72		max_priority_fee_per_gas: Option<U256>,
73		config: &'config evm::Config,
74		precompiles: &'precompiles T::PrecompilesType,
75		is_transactional: bool,
76		weight_limit: Option<Weight>,
77		proof_size_base_cost: Option<u64>,
78		f: F,
79	) -> Result<ExecutionInfoV2<R>, RunnerError<Error<T>>>
80	where
81		F: FnOnce(
82			&mut StackExecutor<
83				'config,
84				'precompiles,
85				SubstrateStackState<'_, 'config, T>,
86				T::PrecompilesType,
87			>,
88		) -> (ExitReason, R),
89		R: Default,
90	{
91		let (base_fee, weight) = T::FeeCalculator::min_gas_price();
92
93		#[cfg(feature = "forbid-evm-reentrancy")]
94		if IN_EVM.with(|in_evm| in_evm.replace(true)) {
95			return Err(RunnerError {
96				error: Error::<T>::Reentrancy,
97				weight,
98			});
99		}
100
101		let res = Self::execute_inner(
102			source,
103			value,
104			gas_limit,
105			max_fee_per_gas,
106			max_priority_fee_per_gas,
107			config,
108			precompiles,
109			is_transactional,
110			f,
111			base_fee,
112			weight,
113			weight_limit,
114			proof_size_base_cost,
115		);
116
117		// Set IN_EVM to false
118		// We should make sure that this line is executed whatever the execution path.
119		#[cfg(feature = "forbid-evm-reentrancy")]
120		let _ = IN_EVM.with(|in_evm| in_evm.take());
121
122		res
123	}

125	// Execute an already validated EVM operation.
126	fn execute_inner<'config, 'precompiles, F, R>(
127		source: H160,
128		value: U256,
129		mut gas_limit: u64,
130		max_fee_per_gas: Option<U256>,
131		max_priority_fee_per_gas: Option<U256>,
132		config: &'config evm::Config,
133		precompiles: &'precompiles T::PrecompilesType,
134		is_transactional: bool,
135		f: F,
136		base_fee: U256,
137		weight: Weight,
138		weight_limit: Option<Weight>,
139		proof_size_base_cost: Option<u64>,
140	) -> Result<ExecutionInfoV2<R>, RunnerError<Error<T>>>
141	where
142		F: FnOnce(
143			&mut StackExecutor<
144				'config,
145				'precompiles,
146				SubstrateStackState<'_, 'config, T>,
147				T::PrecompilesType,
148			>,
149		) -> (ExitReason, R),
150		R: Default,
151	{
152		// Used to record the external costs in the evm through the StackState implementation
153		let maybe_weight_info =
154			WeightInfo::new_from_weight_limit(weight_limit, proof_size_base_cost).map_err(
155				|_| RunnerError {
156					error: Error::<T>::GasLimitTooLow,
157					weight,
158				},
159			)?;
160		// The precompile check is only used for transactional invocations. However, here we always
161		// execute the check, because the check has side effects.
162		match precompiles.is_precompile(source, gas_limit) {
163			IsPrecompileResult::Answer { extra_cost, .. } => {
164				gas_limit = gas_limit.saturating_sub(extra_cost);
165			}
166			IsPrecompileResult::OutOfGas => {
167				return Ok(ExecutionInfoV2 {
168					exit_reason: ExitError::OutOfGas.into(),
169					value: Default::default(),
170					used_gas: fp_evm::UsedGas {
171						standard: gas_limit.into(),
172						effective: gas_limit.into(),
173					},
174					weight_info: maybe_weight_info,
175					logs: Default::default(),
176				})
177			}
178		};
179
180		// Only check the restrictions of EIP-3607 if the source of the EVM operation is from an external transaction.
181		// If the source of this EVM operation is from an internal call, like from `eth_call` or `eth_estimateGas` RPC,
182		// we will skip the checks for the EIP-3607.
183		//
184		// EIP-3607: https://eips.ethereum.org/EIPS/eip-3607
185		// Do not allow transactions for which `tx.sender` has any code deployed.
186		if is_transactional && !<AccountCodes<T>>::get(source).is_empty() {
187			return Err(RunnerError {
188				error: Error::<T>::TransactionMustComeFromEOA,
189				weight,
190			});
191		}
192
193		let total_fee_per_gas = if is_transactional {
194			match (max_fee_per_gas, max_priority_fee_per_gas) {
195				// Zero max_fee_per_gas for validated transactional calls exist in XCM -> EVM
196				// because fees are already withdrawn in the xcm-executor.
197				(Some(max_fee), _) if max_fee.is_zero() => U256::zero(),
198				// With no tip, we pay exactly the base_fee
199				(Some(_), None) => base_fee,
200				// With tip, we include as much of the tip on top of base_fee that we can, never
201				// exceeding max_fee_per_gas
202				(Some(max_fee_per_gas), Some(max_priority_fee_per_gas)) => {
203					let actual_priority_fee_per_gas = max_fee_per_gas
204						.saturating_sub(base_fee)
205						.min(max_priority_fee_per_gas);
206
207					base_fee.saturating_add(actual_priority_fee_per_gas)
208				}
209				_ => {
210					return Err(RunnerError {
211						error: Error::<T>::GasPriceTooLow,
212						weight,
213					})
214				}
215			}
216		} else {
217			// Gas price check is skipped for non-transactional calls or creates
218			Default::default()
219		};
220
221		// After eip-1559 we make sure the account can pay both the evm execution and priority fees.
222		let total_fee =
223			total_fee_per_gas
224				.checked_mul(U256::from(gas_limit))
225				.ok_or(RunnerError {
226					error: Error::<T>::FeeOverflow,
227					weight,
228				})?;
229
230		// Deduct fee from the `source` account. Returns `None` if `total_fee` is Zero.
231		// === Note: This fee gets converted to substrate decimals in `withdraw_fee` ===
232		let fee = T::OnChargeTransaction::withdraw_fee(&source, EvmBalance::new(total_fee))
233			.map_err(|e| RunnerError { error: e, weight })?;
234
235		// Execute the EVM call.
236		let vicinity = Vicinity {
237			gas_price: base_fee,
238			origin: source,
239		};
240
241		let metadata = StackSubstateMetadata::new(gas_limit, config);
242		let state = SubstrateStackState::new(&vicinity, metadata, maybe_weight_info);
243		let mut executor = StackExecutor::new_with_precompiles(state, config, precompiles);
244
245		let (reason, retv) = f(&mut executor);
246
247		// Post execution.
248		let used_gas = executor.used_gas();
249		let effective_gas = match executor.state().weight_info() {
250			Some(weight_info) => U256::from(core::cmp::max(
251				used_gas,
252				weight_info
253					.proof_size_usage
254					.unwrap_or_default()
255					.saturating_mul(T::GasLimitPovSizeRatio::get()),
256			)),
257			_ => used_gas.into(),
258		};
259		let actual_fee = effective_gas.saturating_mul(total_fee_per_gas);
260		let actual_base_fee = effective_gas.saturating_mul(base_fee);
261
262		log::debug!(
263			target: "evm",
264			"Execution {:?} [source: {:?}, value: {}, gas_limit: {}, actual_fee: {}, used_gas: {}, effective_gas: {}, base_fee: {}, total_fee_per_gas: {}, is_transactional: {}]",
265			reason,
266			source,
267			value,
268			gas_limit,
269			actual_fee,
270			used_gas,
271			effective_gas,
272			base_fee,
273			total_fee_per_gas,
274			is_transactional
275		);
276		// The difference between initially withdrawn and the actual cost is refunded.
277		//
278		// Considered the following request:
279		// +-----------+---------+--------------+
280		// | Gas_limit | Max_Fee | Max_Priority |
281		// +-----------+---------+--------------+
282		// |        20 |      10 |            6 |
283		// +-----------+---------+--------------+
284		//
285		// And execution:
286		// +----------+----------+
287		// | Gas_used | Base_Fee |
288		// +----------+----------+
289		// |        5 |        2 |
290		// +----------+----------+
291		//
292		// Initially withdrawn 10 * 20 = 200.
293		// Actual cost (2 + 6) * 5 = 40.
294		// Refunded 200 - 40 = 160.
295		// Tip 5 * 6 = 30.
296		// Burned 200 - (160 + 30) = 10. Which is equivalent to gas_used * base_fee.
297		// === Note: we expect acutal_fee and actual_base_fee to be in EVM decimals. but `fee` should be in substrate decimals already ===
298		// === Note: `actual_priority_fee` gets converted to substrate decimals in `correct_and_deposit_fee` ===
299		let actual_priority_fee = T::OnChargeTransaction::correct_and_deposit_fee(
300			&source,
301			// Actual fee after evm execution, including tip.
302			EvmBalance::new(actual_fee),
303			// Base fee.
304			EvmBalance::new(actual_base_fee),
305			// Fee initially withdrawn.
306			fee,
307		);
308		// === Note: `actual_priority_fee` is already in substrate decimals ===
309		T::OnChargeTransaction::pay_priority_fee(actual_priority_fee);
310
311		let state = executor.into_state();
312
313		for address in &state.substate.deletes {
314			log::debug!(
315				target: "evm",
316				"Deleting account at {:?}",
317				address
318			);
319			Pallet::<T>::remove_account(address)
320		}
321
322		for log in &state.substate.logs {
323			log::trace!(
324				target: "evm",
325				"Inserting log for {:?}, topics ({}) {:?}, data ({}): {:?}]",
326				log.address,
327				log.topics.len(),
328				log.topics,
329				log.data.len(),
330				log.data
331			);
332			Pallet::<T>::deposit_event(Event::<T>::Log {
333				log: Log {
334					address: log.address,
335					topics: log.topics.clone(),
336					data: log.data.clone(),
337				},
338			});
339		}
340
341		Ok(ExecutionInfoV2 {
342			value: retv,
343			exit_reason: reason,
344			used_gas: fp_evm::UsedGas {
345				standard: used_gas.into(),
346				effective: effective_gas,
347			},
348			weight_info: state.weight_info(),
349			logs: state.substate.logs,
350		})
351	}
352}

354impl<T: Config> RunnerT<T> for Runner<T>

360	fn validate(
361		source: H160,
362		target: Option<H160>,
363		input: Vec<u8>,
364		value: U256,
365		gas_limit: u64,
366		max_fee_per_gas: Option<U256>,
367		max_priority_fee_per_gas: Option<U256>,
368		nonce: Option<U256>,
369		access_list: Vec<(H160, Vec<H256>)>,
370		is_transactional: bool,
371		weight_limit: Option<Weight>,
372		proof_size_base_cost: Option<u64>,
373		evm_config: &evm::Config,
374	) -> Result<(), RunnerError<Self::Error>> {
375		let (base_fee, mut weight) = T::FeeCalculator::min_gas_price();
376		let (source_account, inner_weight) = Pallet::<T>::account_basic(&source);
377		weight = weight.saturating_add(inner_weight);
378
379		let _ = fp_evm::CheckEvmTransaction::<Self::Error>::new(
380			fp_evm::CheckEvmTransactionConfig {
381				evm_config,
382				block_gas_limit: T::BlockGasLimit::get(),
383				base_fee,
384				chain_id: T::ChainId::get(),
385				is_transactional,
386			},
387			fp_evm::CheckEvmTransactionInput {
388				chain_id: Some(T::ChainId::get()),
389				to: target,
390				input,
391				nonce: nonce.unwrap_or(source_account.nonce),
392				gas_limit: gas_limit.into(),
393				gas_price: None,
394				max_fee_per_gas,
395				max_priority_fee_per_gas,
396				value,
397				access_list,
398			},
399			weight_limit,
400			proof_size_base_cost,
401		)
402		.validate_in_block_for(&source_account)
403		.and_then(|v| v.with_base_fee())
404		.and_then(|v| v.with_balance_for(&source_account))
405		.map_err(|error| RunnerError { error, weight })?;
406		Ok(())
407	}

459		source: H160,

461		value: U256,
462		gas_limit: u64,
463		max_fee_per_gas: Option<U256>,

481			}

483			Self::validate(
484				source,
485				None,
486				init.clone(),

488				gas_limit,
489				max_fee_per_gas,

491				nonce,
492				access_list.clone(),
493				is_transactional,
494				weight_limit,
495				proof_size_base_cost,
496				config,
497			)?;
498		}
499		let precompiles = T::PrecompilesValue::get();
500		Self::execute(
501			source,
502			value,
503			gas_limit,
504			max_fee_per_gas,
505			max_priority_fee_per_gas,
506			config,
507			&precompiles,
508			is_transactional,
509			weight_limit,
510			proof_size_base_cost,
511			|executor| {
512				let address = executor.create_address(evm::CreateScheme::Legacy { caller: source });
513				T::OnCreate::on_create(source, address);
514				let (reason, _) =
515					executor.transact_create(source, value, init, gas_limit, access_list);
516				(reason, address)
517			},
518		)
519	}

521	fn create2(
522		source: H160,
523		init: Vec<u8>,
524		salt: H256,
525		value: U256,
526		gas_limit: u64,
527		max_fee_per_gas: Option<U256>,

539		if validate {
540			if !disable_whitelist_check && !whitelist.contains(&source) {
541				return Err(RunnerError {
542					error: Error::<T>::NotAllowed,
543					weight: Weight::zero(),
544				});
545			}
546
547			Self::validate(
548				source,
549				None,
550				init.clone(),
551				value,
552				gas_limit,
553				max_fee_per_gas,
554				max_priority_fee_per_gas,
555				nonce,
556				access_list.clone(),
557				is_transactional,
558				weight_limit,
559				proof_size_base_cost,
560				config,
561			)?;
562		}
563		let precompiles = T::PrecompilesValue::get();
564		let code_hash = H256::from(sp_io::hashing::keccak_256(&init));
565		Self::execute(
566			source,
567			value,
568			gas_limit,
569			max_fee_per_gas,
570			max_priority_fee_per_gas,
571			config,

573			is_transactional,
574			weight_limit,
575			proof_size_base_cost,
576			|executor| {
577				let address = executor.create_address(evm::CreateScheme::Create2 {
578					caller: source,
579					code_hash,
580					salt,
581				});
582				T::OnCreate::on_create(source, address);
583				let (reason, _) =
584					executor.transact_create2(source, value, init, salt, gas_limit, access_list);
585				(reason, address)
586			},
587		)
588	}

591struct SubstrateStackSubstate<'config> {
592	metadata: StackSubstateMetadata<'config>,
593	deletes: BTreeSet<H160>,
594	logs: Vec<Log>,
595	parent: Option<Box<SubstrateStackSubstate<'config>>>,
596}
597
598impl<'config> SubstrateStackSubstate<'config> {
599	pub fn metadata(&self) -> &StackSubstateMetadata<'config> {
600		&self.metadata
601	}
602
603	pub fn metadata_mut(&mut self) -> &mut StackSubstateMetadata<'config> {
604		&mut self.metadata
605	}
606
607	pub fn enter(&mut self, gas_limit: u64, is_static: bool) {
608		let mut entering = Self {
609			metadata: self.metadata.spit_child(gas_limit, is_static),
610			parent: None,
611			deletes: BTreeSet::new(),
612			logs: Vec::new(),
613		};
614		mem::swap(&mut entering, self);
615
616		self.parent = Some(Box::new(entering));
617
618		sp_io::storage::start_transaction();
619	}
620
621	pub fn exit_commit(&mut self) -> Result<(), ExitError> {
622		let mut exited = *self.parent.take().expect("Cannot commit on root substate");
623		mem::swap(&mut exited, self);
624
625		self.metadata.swallow_commit(exited.metadata)?;
626		self.logs.append(&mut exited.logs);
627		self.deletes.append(&mut exited.deletes);
628
629		sp_io::storage::commit_transaction();
630		Ok(())
631	}
632
633	pub fn exit_revert(&mut self) -> Result<(), ExitError> {
634		let mut exited = *self.parent.take().expect("Cannot discard on root substate");
635		mem::swap(&mut exited, self);
636		self.metadata.swallow_revert(exited.metadata)?;
637
638		sp_io::storage::rollback_transaction();
639		Ok(())
640	}
641
642	pub fn exit_discard(&mut self) -> Result<(), ExitError> {
643		let mut exited = *self.parent.take().expect("Cannot discard on root substate");
644		mem::swap(&mut exited, self);
645		self.metadata.swallow_discard(exited.metadata)?;
646
647		sp_io::storage::rollback_transaction();
648		Ok(())
649	}

651	pub fn deleted(&self, address: H160) -> bool {
652		if self.deletes.contains(&address) {
653			return true;
654		}
655
656		if let Some(parent) = self.parent.as_ref() {
657			return parent.deleted(address);
658		}
659
660		false
661	}

663	pub fn set_deleted(&mut self, address: H160) {
664		self.deletes.insert(address);
665	}

668		self.logs.push(Log {
669			address,
670			topics,
671			data,
672		});
673	}

675	fn recursive_is_cold<F: Fn(&Accessed) -> bool>(&self, f: &F) -> bool {
676		let local_is_accessed = self.metadata.accessed().as_ref().map(f).unwrap_or(false);
677		if local_is_accessed {
678			false
679		} else {
680			self.parent
681				.as_ref()
682				.map(|p| p.recursive_is_cold(f))
683				.unwrap_or(true)
684		}
685	}

974	fn record_external_operation(&mut self, op: evm::ExternalOperation) -> Result<(), ExitError> {
975		let size_limit: u64 = self
976			.metadata()
977			.gasometer()
978			.config()
979			.create_contract_limit
980			.unwrap_or_default() as u64;
981		let (weight_info, recorded) = self.info_mut();
982
983		if let Some(weight_info) = weight_info {
984			match op {
985				ExternalOperation::AccountBasicRead => {
986					weight_info.try_record_proof_size_or_fail(ACCOUNT_BASIC_PROOF_SIZE)?
987				}
988				ExternalOperation::AddressCodeRead(address) => {
989					let maybe_record = !recorded.account_codes.contains(&address);
990					// Skip if the address has been already recorded this block
991					if maybe_record {
992						// First we record account emptiness check.
993						// Transfers to EOAs with standard 21_000 gas limit are able to
994						// pay for this pov size.
995						weight_info.try_record_proof_size_or_fail(IS_EMPTY_CHECK_PROOF_SIZE)?;
996						if <AccountCodes<T>>::decode_len(address).unwrap_or(0) == 0 {
997							return Ok(());
998						}
999
1000						weight_info
1001							.try_record_proof_size_or_fail(ACCOUNT_CODES_METADATA_PROOF_SIZE)?;
1002						if let Some(meta) = <AccountCodesMetadata<T>>::get(address) {
1003							weight_info.try_record_proof_size_or_fail(meta.size)?;
1004						} else if let Some(remaining_proof_size) =
1005							weight_info.remaining_proof_size()
1006						{
1007							let pre_size = remaining_proof_size.min(size_limit);
1008							weight_info.try_record_proof_size_or_fail(pre_size)?;
1009
1010							let actual_size = Pallet::<T>::account_code_metadata(address).size;
1011							if actual_size > pre_size {
1012								return Err(ExitError::OutOfGas);
1013							}
1014							// Refund unused proof size
1015							weight_info.refund_proof_size(pre_size.saturating_sub(actual_size));
1016						}
1017						recorded.account_codes.push(address);
1018					}
1019				}
1020				ExternalOperation::IsEmpty => {
1021					weight_info.try_record_proof_size_or_fail(IS_EMPTY_CHECK_PROOF_SIZE)?
1022				}
1023				ExternalOperation::Write(_) => {
1024					weight_info.try_record_proof_size_or_fail(WRITE_PROOF_SIZE)?
1025				}
1026			};
1027		}
1028		Ok(())
1029	}

1031	fn record_external_dynamic_opcode_cost(
1032		&mut self,
1033		opcode: Opcode,
1034		_gas_cost: GasCost,
1035		target: evm::gasometer::StorageTarget,
1036	) -> Result<(), ExitError> {
1037		// If account code or storage slot is in the overlay it is already accounted for and early exit
1038		let accessed_storage: Option<AccessedStorage> = match target {
1039			StorageTarget::Address(address) => {
1040				if self.recorded().account_codes.contains(&address) {
1041					return Ok(());
1042				} else {
1043					Some(AccessedStorage::AccountCodes(address))
1044				}
1045			}
1046			StorageTarget::Slot(address, index) => {
1047				if self
1048					.recorded()
1049					.account_storages
1050					.contains_key(&(address, index))
1051				{
1052					return Ok(());
1053				} else {
1054					Some(AccessedStorage::AccountStorages((address, index)))
1055				}
1056			}
1057			_ => None,
1058		};
1059
1060		let size_limit: u64 = self
1061			.metadata()
1062			.gasometer()
1063			.config()
1064			.create_contract_limit
1065			.unwrap_or_default() as u64;
1066		let (weight_info, recorded) = self.info_mut();
1067
1068		if let Some(weight_info) = weight_info {
1069			// proof_size_limit is None indicates no need to record proof size, return directly.
1070			if weight_info.proof_size_limit.is_none() {
1071				return Ok(());
1072			};
1073
1074			let mut record_account_codes_proof_size =
1075				|address: H160, empty_check: bool| -> Result<(), ExitError> {
1076					let mut base_size = ACCOUNT_CODES_METADATA_PROOF_SIZE;
1077					if empty_check {
1078						base_size = base_size.saturating_add(IS_EMPTY_CHECK_PROOF_SIZE);
1079					}
1080					weight_info.try_record_proof_size_or_fail(base_size)?;
1081
1082					if let Some(meta) = <AccountCodesMetadata<T>>::get(address) {
1083						weight_info.try_record_proof_size_or_fail(meta.size)?;
1084					} else if let Some(remaining_proof_size) = weight_info.remaining_proof_size() {
1085						let pre_size = remaining_proof_size.min(size_limit);
1086						weight_info.try_record_proof_size_or_fail(pre_size)?;
1087
1088						let actual_size = Pallet::<T>::account_code_metadata(address).size;
1089						if actual_size > pre_size {
1090							return Err(ExitError::OutOfGas);
1091						}
1092						// Refund unused proof size
1093						weight_info.refund_proof_size(pre_size.saturating_sub(actual_size));
1094					}
1095
1096					Ok(())
1097				};
1098
1099			// Proof size is fixed length for writes (a 32-byte hash in a merkle trie), and
1100			// the full key/value for reads. For read and writes over the same storage, the full value
1101			// is included.
1102			// For cold reads involving code (call, callcode, staticcall and delegatecall):
1103			//	- We depend on https://github.com/paritytech/frontier/pull/893
1104			//	- Try to get the cached size or compute it on the fly
1105			//	- We record the actual size after caching, refunding the difference between it and the initially deducted
1106			//	contract size limit.
1107			match opcode {
1108				Opcode::BALANCE => {
1109					weight_info.try_record_proof_size_or_fail(ACCOUNT_BASIC_PROOF_SIZE)?;
1110				}
1111				Opcode::EXTCODESIZE | Opcode::EXTCODECOPY | Opcode::EXTCODEHASH => {
1112					if let Some(AccessedStorage::AccountCodes(address)) = accessed_storage {
1113						record_account_codes_proof_size(address, false)?;
1114						recorded.account_codes.push(address);
1115					}
1116				}
1117				Opcode::CALLCODE | Opcode::CALL | Opcode::DELEGATECALL | Opcode::STATICCALL => {
1118					if let Some(AccessedStorage::AccountCodes(address)) = accessed_storage {
1119						record_account_codes_proof_size(address, true)?;
1120						recorded.account_codes.push(address);
1121					}
1122				}
1123				Opcode::SLOAD => {
1124					if let Some(AccessedStorage::AccountStorages((address, index))) =
1125						accessed_storage
1126					{
1127						weight_info.try_record_proof_size_or_fail(ACCOUNT_STORAGE_PROOF_SIZE)?;
1128						recorded.account_storages.insert((address, index), true);
1129					}
1130				}
1131				Opcode::SSTORE => {
1132					if let Some(AccessedStorage::AccountStorages((address, index))) =
1133						accessed_storage
1134					{
1135						let size = WRITE_PROOF_SIZE.saturating_add(ACCOUNT_STORAGE_PROOF_SIZE);
1136						weight_info.try_record_proof_size_or_fail(size)?;
1137						recorded.account_storages.insert((address, index), true);
1138					}
1139				}
1140				Opcode::CREATE | Opcode::CREATE2 => {
1141					weight_info.try_record_proof_size_or_fail(WRITE_PROOF_SIZE)?;
1142				}
1143				// When calling SUICIDE a target account will receive the self destructing
1144				// address's balance. We need to account for both:
1145				//	- Target basic account read
1146				//	- 5 bytes of `decode_len`
1147				Opcode::SUICIDE => {
1148					weight_info.try_record_proof_size_or_fail(IS_EMPTY_CHECK_PROOF_SIZE)?;
1149				}
1150				// Rest of dynamic opcodes that do not involve proof size recording, do nothing
1151				_ => return Ok(()),
1152			};
1153		}
1154
1155		Ok(())
1156	}

1159		&mut self,
1160		ref_time: Option<u64>,
1161		proof_size: Option<u64>,
1162		_storage_growth: Option<u64>,
1163	) -> Result<(), ExitError> {
1164		let weight_info = if let (Some(weight_info), _) = self.info_mut() {
1165			weight_info
1166		} else {
1167			return Ok(());
1168		};
1169
1170		if let Some(amount) = ref_time {
1171			weight_info.try_record_ref_time_or_fail(amount)?;
1172		}
1173		if let Some(amount) = proof_size {
1174			weight_info.try_record_proof_size_or_fail(amount)?;
1175		}

1177	}

1181			if let Some(amount) = ref_time {
1182				weight_info.refund_ref_time(amount);
1183			}
1184			if let Some(amount) = proof_size {
1185				weight_info.refund_proof_size(amount);
1186			}
1187		}
