1use std::time::{Duration, Instant};
2
3/// Limits the amount of time spent on a certain type of work in a cycle
4///
5/// The limiter works dynamically: For a sampled subset of cycles it measures
6/// the time that is approximately required for fulfilling 1 work item, and
7/// calculates the amount of allowed work items per cycle.
8/// The estimates are smoothed over all cycles where the exact duration is measured.
9///
10/// In cycles where no measurement is performed the previously determined work limit
11/// is used.
12///
13/// For the limiter the exact definition of a work item does not matter.
14/// It could for example track the amount of transmitted bytes per cycle,
15/// or the amount of transmitted datagrams per cycle.
16/// It will however work best if the required time to complete a work item is
17/// constant.
18#[derive(Debug)]
19pub struct WorkLimiter {
20    /// Whether to measure the required work time, or to use the previous estimates
21    mode: Mode,
22    /// The current cycle number
23    cycle: u16,
24    /// The time the cycle started - only used in measurement mode
25    start_time: Instant,
26    /// How many work items have been completed in the cycle
27    completed: usize,
28    /// The amount of work items which are allowed for a cycle
29    allowed: usize,
30    /// The desired cycle time
31    desired_cycle_time: Duration,
32    /// The estimated and smoothed time per work item in nanoseconds
33    smoothed_time_per_work_item_nanos: f64,
34    /// Retrieves the current time for unit-test purposes
35    #[cfg(test)]
36    get_time: fn() -> Instant,
37}
38
39impl WorkLimiter {
40    pub fn new(desired_cycle_time: Duration) -> Self {
41        Self {
42            mode: Mode::Measure,
43            cycle: 0,
44            start_time: Instant::now(),
45            completed: 0,
46            allowed: 0,
47            desired_cycle_time,
48            smoothed_time_per_work_item_nanos: 0.0,
49            #[cfg(test)]
50            get_time: std::time::Instant::now,
51        }
52    }
53
54    /// Starts one work cycle
55    pub fn start_cycle(&mut self) {
56        self.completed = 0;
57        if let Mode::Measure = self.mode {
58            self.start_time = self.now();
59        }
60    }
61
62    /// Returns whether more work can be performed inside the `desired_cycle_time`
63    ///
64    /// Requires that previous work was tracked using `record_work`.
65    pub fn allow_work(&mut self) -> bool {
66        match self.mode {
67            Mode::Measure => (self.now() - self.start_time) < self.desired_cycle_time,
68            Mode::HistoricData => self.completed < self.allowed,
69        }
70    }
71
72    /// Records that `work` additional work items have been completed inside the cycle
73    ///
74    /// Must be called between `start_cycle` and `finish_cycle`.
75    pub fn record_work(&mut self, work: usize) {
76        self.completed += work;
77    }
78
79    /// Finishes one work cycle
80    ///
81    /// For cycles where the exact duration is measured this will update the estimates
82    /// for the time per work item and the limit of allowed work items per cycle.
83    /// The estimate is updated using the same exponential averaging (smoothing)
84    /// mechanism which is used for determining QUIC path rtts: The last value is
85    /// weighted by 1/8, and the previous average by 7/8.
86    pub fn finish_cycle(&mut self) {
87        // If no work was done in the cycle drop the measurement, it won't be useful
88        if self.completed == 0 {
89            return;
90        }
91
92        if let Mode::Measure = self.mode {
93            let elapsed = self.now() - self.start_time;
94
95            let time_per_work_item_nanos = (elapsed.as_nanos()) as f64 / self.completed as f64;
96
97            // Calculate the time per work item. We set this to at least 1ns to avoid
98            // dividing by 0 when calculating the allowed amount of work items.
99            self.smoothed_time_per_work_item_nanos = if self.allowed == 0 {
100                // Initial estimate
101                time_per_work_item_nanos
102            } else {
103                // Smoothed estimate
104                (7.0 * self.smoothed_time_per_work_item_nanos + time_per_work_item_nanos) / 8.0
105            }
106            .max(1.0);
107
108            // Allow at least 1 work item in order to make progress
109            self.allowed = (((self.desired_cycle_time.as_nanos()) as f64
110                / self.smoothed_time_per_work_item_nanos) as usize)
111                .max(1);
112        }
113
114        self.cycle = self.cycle.wrapping_add(1);
115        self.mode = match self.cycle % SAMPLING_INTERVAL {
116            0 => Mode::Measure,
117            _ => Mode::HistoricData,
118        };
119    }
120
121    #[cfg(not(test))]
122    fn now(&self) -> Instant {
123        Instant::now()
124    }
125
126    #[cfg(test)]
127    fn now(&self) -> Instant {
128        (self.get_time)()
129    }
130}
131
132/// We take a measurement sample once every `SAMPLING_INTERVAL` cycles
133const SAMPLING_INTERVAL: u16 = 256;
134
135#[derive(Debug, Clone, Copy, PartialEq, Eq)]
136enum Mode {
137    Measure,
138    HistoricData,
139}

141#[cfg(test)]
142mod tests {
143    use super::*;
144    use std::cell::RefCell;
145
146    #[test]
147    fn limit_work() {
148        const CYCLE_TIME: Duration = Duration::from_millis(500);
149        const BATCH_WORK_ITEMS: usize = 12;
150        const BATCH_TIME: Duration = Duration::from_millis(100);
151
152        const EXPECTED_INITIAL_BATCHES: usize =
153            (CYCLE_TIME.as_nanos() / BATCH_TIME.as_nanos()) as usize;
154        const EXPECTED_ALLOWED_WORK_ITEMS: usize = EXPECTED_INITIAL_BATCHES * BATCH_WORK_ITEMS;
155
156        let mut limiter = WorkLimiter::new(CYCLE_TIME);
157        limiter.get_time = get_time;
158        reset_time();
159
160        // The initial cycle is measuring
161        limiter.start_cycle();
162        let mut initial_batches = 0;
163        while limiter.allow_work() {
164            limiter.record_work(BATCH_WORK_ITEMS);
165            advance_time(BATCH_TIME);
166            initial_batches += 1;
167        }
168        limiter.finish_cycle();
169
170        assert_eq!(initial_batches, EXPECTED_INITIAL_BATCHES);
171        assert_eq!(limiter.allowed, EXPECTED_ALLOWED_WORK_ITEMS);
172        let initial_time_per_work_item = limiter.smoothed_time_per_work_item_nanos;
173
174        // The next cycles are using historic data
175        const BATCH_SIZES: [usize; 4] = [1, 2, 3, 5];
176        for &batch_size in &BATCH_SIZES {
177            limiter.start_cycle();
178            let mut allowed_work = 0;
179            while limiter.allow_work() {
180                limiter.record_work(batch_size);
181                allowed_work += batch_size;
182            }
183            limiter.finish_cycle();
184
185            assert_eq!(allowed_work, EXPECTED_ALLOWED_WORK_ITEMS);
186        }
187
188        // After `SAMPLING_INTERVAL`, we get into measurement mode again
189        for _ in 0..(SAMPLING_INTERVAL as usize - BATCH_SIZES.len() - 1) {
190            limiter.start_cycle();
191            limiter.record_work(1);
192            limiter.finish_cycle();
193        }
194
195        // We now do more work per cycle, and expect the estimate of allowed
196        // work items to go up
197        const BATCH_WORK_ITEMS_2: usize = 96;
198        const TIME_PER_WORK_ITEMS_2_NANOS: f64 =
199            CYCLE_TIME.as_nanos() as f64 / (EXPECTED_INITIAL_BATCHES * BATCH_WORK_ITEMS_2) as f64;
200
201        let expected_updated_time_per_work_item =
202            (initial_time_per_work_item * 7.0 + TIME_PER_WORK_ITEMS_2_NANOS) / 8.0;
203        let expected_updated_allowed_work_items =
204            (CYCLE_TIME.as_nanos() as f64 / expected_updated_time_per_work_item) as usize;
205
206        limiter.start_cycle();
207        let mut initial_batches = 0;
208        while limiter.allow_work() {
209            limiter.record_work(BATCH_WORK_ITEMS_2);
210            advance_time(BATCH_TIME);
211            initial_batches += 1;
212        }
213        limiter.finish_cycle();
214
215        assert_eq!(initial_batches, EXPECTED_INITIAL_BATCHES);
216        assert_eq!(limiter.allowed, expected_updated_allowed_work_items);
217    }

219    thread_local! {
220        /// Mocked time
221        pub static TIME: RefCell<Instant> = RefCell::new(Instant::now());
222    }
223
224    fn reset_time() {
225        TIME.with(|t| {
226            *t.borrow_mut() = Instant::now();
227        })
228    }
229
230    fn get_time() -> Instant {
231        TIME.with(|t| *t.borrow())
232    }
233
234    fn advance_time(duration: Duration) {
235        TIME.with(|t| {
236            *t.borrow_mut() += duration;
237        })
238    }
