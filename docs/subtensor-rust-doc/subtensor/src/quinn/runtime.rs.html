15/// Abstracts I/O and timer operations for runtime independence
16pub trait Runtime: Send + Sync + Debug + 'static {
17    /// Construct a timer that will expire at `i`
18    fn new_timer(&self, i: Instant) -> Pin<Box<dyn AsyncTimer>>;
19    /// Drive `future` to completion in the background
20    fn spawn(&self, future: Pin<Box<dyn Future<Output = ()> + Send>>);
21    /// Convert `t` into the socket type used by this runtime
22    fn wrap_udp_socket(&self, t: std::net::UdpSocket) -> io::Result<Box<dyn AsyncUdpSocket>>;
23}
24
25/// Abstract implementation of an async timer for runtime independence
26pub trait AsyncTimer: Send + Debug + 'static {
27    /// Update the timer to expire at `i`
28    fn reset(self: Pin<&mut Self>, i: Instant);
29    /// Check whether the timer has expired, and register to be woken if not
30    fn poll(self: Pin<&mut Self>, cx: &mut Context) -> Poll<()>;
31}
32
33/// Abstract implementation of a UDP socket for runtime independence
34pub trait AsyncUdpSocket: Send + Debug + 'static {
35    /// Send UDP datagrams from `transmits`, or register to be woken if sending may succeed in the
36    /// future
37    fn poll_send(
38        &mut self,
39        state: &UdpState,
40        cx: &mut Context,
41        transmits: &[Transmit],
42    ) -> Poll<Result<usize, io::Error>>;
43
44    /// Receive UDP datagrams, or register to be woken if receiving may succeed in the future
45    fn poll_recv(
46        &self,
47        cx: &mut Context,
48        bufs: &mut [IoSliceMut<'_>],
49        meta: &mut [RecvMeta],
50    ) -> Poll<io::Result<usize>>;
51
52    /// Look up the local IP address and port used by this socket
53    fn local_addr(&self) -> io::Result<SocketAddr>;
54}

56/// Automatically select an appropriate runtime from those enabled at compile time
57///
58/// If `runtime-tokio` is enabled and this function is called from within a Tokio runtime context,
59/// then `TokioRuntime` is returned. Otherwise, if `runtime-async-std` is enabled, `AsyncStdRuntime`
60/// is returned. Otherwise, `None` is returned.
61pub fn default_runtime() -> Option<Arc<dyn Runtime>> {
62    #[cfg(feature = "runtime-tokio")]
63    {
64        if ::tokio::runtime::Handle::try_current().is_ok() {
65            return Some(Arc::new(TokioRuntime));
66        }
67    }
68
69    #[cfg(feature = "runtime-async-std")]
70    {
71        return Some(Arc::new(AsyncStdRuntime));
72    }
73
74    #[cfg(not(feature = "runtime-async-std"))]
75    None
76}
