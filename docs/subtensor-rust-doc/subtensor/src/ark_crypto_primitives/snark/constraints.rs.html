29/// This implements constraints for SNARK verifiers.
30pub trait SNARKGadget<F: PrimeField, ConstraintF: PrimeField, S: SNARK<F>> {
31    type ProcessedVerifyingKeyVar: AllocVar<S::ProcessedVerifyingKey, ConstraintF> + Clone;
32    type VerifyingKeyVar: AllocVar<S::VerifyingKey, ConstraintF>
33        + ToBytesGadget<ConstraintF>
34        + Clone;
35    type InputVar: AllocVar<Vec<F>, ConstraintF> + FromFieldElementsGadget<F, ConstraintF> + Clone;
36    type ProofVar: AllocVar<S::Proof, ConstraintF> + Clone;
37
38    /// Information about the R1CS constraints required to check proofs relative
39    /// a given verification key. In the context of a LPCP-based pairing-based SNARK
40    /// like that of [[Groth16]](https://eprint.iacr.org/2016/260),
41    /// this is independent of the R1CS matrices,
42    /// whereas for more "complex" SNARKs like [[Marlin]](https://eprint.iacr.org/2019/1047),
43    /// this can encode information about the highest degree of polynomials
44    /// required to verify proofs.
45    type VerifierSize: PartialOrd + Clone + fmt::Debug;
46
47    /// Returns information about the R1CS constraints required to check proofs relative
48    /// to the verification key `circuit_vk`.
49    fn verifier_size(circuit_vk: &S::VerifyingKey) -> Self::VerifierSize;
50
51    /// Optionally allocates `S::Proof` in `cs` without performing
52    /// additional checks, such as subgroup membership checks. Use this *only*
53    /// if you know it is safe to do so. Such "safe" scenarios can include
54    /// the case where `proof` is a public input (`mode == AllocationMode::Input`),
55    /// and the corresponding checks are performed by the SNARK verifier outside
56    /// the circuit.  Another example is the when `mode == AllocationMode::Constant`.
57    ///
58    /// The default implementation does not omit such checks, and just invokes
59    /// `Self::ProofVar::new_variable`.
60    #[tracing::instrument(target = "r1cs", skip(cs, f))]
61    fn new_proof_unchecked<T: Borrow<S::Proof>>(
62        cs: impl Into<Namespace<ConstraintF>>,
63        f: impl FnOnce() -> Result<T, SynthesisError>,
64        mode: AllocationMode,
65    ) -> Result<Self::ProofVar, SynthesisError> {
66        Self::ProofVar::new_variable(cs, f, mode)
67    }
68
69    /// Optionally allocates `S::VerifyingKey` in `cs` without performing
70    /// additional checks, such as subgroup membership checks. Use this *only*
71    /// if you know it is safe to do so. Such "safe" scenarios can include
72    /// the case where `vk` is a public input (`mode == AllocationMode::Input`),
73    /// and the corresponding checks are performed by the SNARK verifier outside
74    /// the circuit. Another example is the when `mode == AllocationMode::Constant`.
75    ///
76    /// The default implementation does not omit such checks, and just invokes
77    /// `Self::VerifyingKeyVar::new_variable`.
78    #[tracing::instrument(target = "r1cs", skip(cs, f))]
79    fn new_verification_key_unchecked<T: Borrow<S::VerifyingKey>>(
80        cs: impl Into<Namespace<ConstraintF>>,
81        f: impl FnOnce() -> Result<T, SynthesisError>,
82        mode: AllocationMode,
83    ) -> Result<Self::VerifyingKeyVar, SynthesisError> {
84        Self::VerifyingKeyVar::new_variable(cs, f, mode)
85    }
86
87    fn verify_with_processed_vk(
88        circuit_pvk: &Self::ProcessedVerifyingKeyVar,
89        x: &Self::InputVar,
90        proof: &Self::ProofVar,
91    ) -> Result<Boolean<ConstraintF>, SynthesisError>;
92
93    fn verify(
94        circuit_vk: &Self::VerifyingKeyVar,
95        x: &Self::InputVar,
96        proof: &Self::ProofVar,
97    ) -> Result<Boolean<ConstraintF>, SynthesisError>;
98}

149impl<F: PrimeField, CF: PrimeField> AllocVar<Vec<F>, CF> for BooleanInputVar<F, CF> {
150    fn new_variable<T: Borrow<Vec<F>>>(
151        cs: impl Into<Namespace<CF>>,
152        f: impl FnOnce() -> Result<T, SynthesisError>,
153        mode: AllocationMode,
154    ) -> Result<Self, SynthesisError> {
155        if mode == AllocationMode::Input {
156            Self::new_input(cs, f)
157        } else {
158            let ns = cs.into();
159            let cs = ns.cs();
160
161            let t = f()?;
162            let obj = t.borrow();
163
164            // convert the elements into booleans (little-endian)
165            let mut res = Vec::<Vec<Boolean<CF>>>::new();
166            for elem in obj.iter() {
167                let mut bits = elem.into_bigint().to_bits_le();
168                bits.truncate(F::MODULUS_BIT_SIZE as usize);
169
170                let mut booleans = Vec::<Boolean<CF>>::new();
171                for bit in bits.iter() {
172                    booleans.push(Boolean::new_variable(ns!(cs, "bit"), || Ok(*bit), mode)?);
173                }
174
175                res.push(booleans);
176            }
177
178            Ok(Self {
179                val: res,
180                _snark_field_: PhantomData,
181            })
182        }
183    }
184
185    fn new_input<T: Borrow<Vec<F>>>(
186        cs: impl Into<Namespace<CF>>,
187        f: impl FnOnce() -> Result<T, SynthesisError>,
188    ) -> Result<Self, SynthesisError> {
189        let ns = cs.into();
190        let cs = ns.cs();
191
192        let obj = f()?;
193
194        // Step 1: obtain the bits of the F field elements (little-endian)
195        let mut src_bits = Vec::<bool>::new();
196        for elem in obj.borrow().iter() {
197            let mut bits = elem.into_bigint().to_bits_le();
198            bits.truncate(F::MODULUS_BIT_SIZE as usize);
199            for _ in bits.len()..F::MODULUS_BIT_SIZE as usize {
200                bits.push(false);
201            }
202            bits.reverse();
203
204            src_bits.append(&mut bits);
205        }
206
207        // Step 2: repack the bits as CF field elements
208        // Deciding how many bits can be embedded,
209        //  if CF has the same number of bits as F, but is larger,
210        //  then it is okay to put the entire field element in.
211        let capacity = if CF::MODULUS_BIT_SIZE == F::MODULUS_BIT_SIZE {
212            let fq = <CF as PrimeField>::MODULUS;
213            let fr = <F as PrimeField>::MODULUS;
214
215            let fq_u64: &[u64] = fq.as_ref();
216            let fr_u64: &[u64] = fr.as_ref();
217
218            let mut fq_not_smaller_than_fr = true;
219            for (left, right) in fq_u64.iter().zip(fr_u64.iter()).rev() {
220                if left < right {
221                    fq_not_smaller_than_fr = false;
222                    break;
223                }
224
225                if left > right {
226                    break;
227                }
228            }
229
230            if fq_not_smaller_than_fr {
231                CF::MODULUS_BIT_SIZE as usize
232            } else {
233                CF::MODULUS_BIT_SIZE as usize - 1
234            }
235        } else {
236            CF::MODULUS_BIT_SIZE as usize - 1
237        };
238
239        // Step 3: allocate the CF field elements as input
240        let mut src_booleans = Vec::<Boolean<CF>>::new();
241        for chunk in src_bits.chunks(capacity) {
242            let elem = CF::from_bigint(<CF as PrimeField>::BigInt::from_bits_be(chunk)).unwrap(); // big endian
243
244            let elem_gadget = FpVar::<CF>::new_input(ns!(cs, "input"), || Ok(elem))?;
245
246            let mut booleans = elem_gadget.to_bits_le()?;
247            booleans.truncate(chunk.len());
248            booleans.reverse();
249
250            src_booleans.append(&mut booleans);
251        }
252
253        // Step 4: unpack them back to bits
254        let res = src_booleans
255            .chunks(F::MODULUS_BIT_SIZE as usize)
256            .map(|f| {
257                let mut res = f.to_vec();
258                res.reverse();
259                res
260            })
261            .collect::<Vec<Vec<Boolean<CF>>>>();
262
263        Ok(Self {
264            val: res,
265            _snark_field_: PhantomData,
266        })
267    }
268}

270impl<F: PrimeField, CF: PrimeField> FromFieldElementsGadget<F, CF> for BooleanInputVar<F, CF> {
271    fn repack_input(src: &Vec<F>) -> Vec<CF> {
272        // Step 1: obtain the bits of the F field elements
273        let mut src_bits = Vec::<bool>::new();
274        for (_, elem) in src.iter().enumerate() {
275            let mut bits = elem.into_bigint().to_bits_le();
276            bits.truncate(F::MODULUS_BIT_SIZE as usize);
277            for _ in bits.len()..F::MODULUS_BIT_SIZE as usize {
278                bits.push(false);
279            }
280            bits.reverse();
281
282            src_bits.append(&mut bits);
283        }
284
285        // Step 2: repack the bits as CF field elements
286        // Deciding how many bits can be embedded.
287        let capacity = if CF::MODULUS_BIT_SIZE == F::MODULUS_BIT_SIZE {
288            let fq = <CF as PrimeField>::MODULUS;
289            let fr = <F as PrimeField>::MODULUS;
290
291            let fq_u64: &[u64] = fq.as_ref();
292            let fr_u64: &[u64] = fr.as_ref();
293
294            let mut fq_not_smaller_than_fr = true;
295            for (left, right) in fq_u64.iter().zip(fr_u64.iter()).rev() {
296                if left < right {
297                    fq_not_smaller_than_fr = false;
298                    break;
299                }
300
301                if left > right {
302                    break;
303                }
304            }
305
306            if fq_not_smaller_than_fr {
307                CF::MODULUS_BIT_SIZE as usize
308            } else {
309                CF::MODULUS_BIT_SIZE as usize - 1
310            }
311        } else {
312            CF::MODULUS_BIT_SIZE as usize - 1
313        };
314
315        // Step 3: directly pack the bits
316        let mut dest = Vec::<CF>::new();
317        for chunk in src_bits.chunks(capacity) {
318            let elem = CF::from_bigint(<CF as PrimeField>::BigInt::from_bits_be(chunk)).unwrap(); // big endian
319            dest.push(elem);
320        }
321
322        dest
323    }
324
325    fn from_field_elements(src: &Vec<FpVar<CF>>) -> Result<Self, SynthesisError> {
326        // Step 1: obtain the booleans of the CF field variables
327        let mut src_booleans = Vec::<Boolean<CF>>::new();
328        for elem in src.iter() {
329            let mut bits = elem.to_bits_le()?;
330            bits.reverse();
331            src_booleans.extend_from_slice(&bits);
332        }
333
334        // Step 2: repack the bits as F field elements
335        // Deciding how many bits can be embedded.
336        let capacity = if CF::MODULUS_BIT_SIZE == F::MODULUS_BIT_SIZE {
337            let fq = <CF as PrimeField>::MODULUS;
338            let fr = <F as PrimeField>::MODULUS;
339
340            let fq_u64: &[u64] = fq.as_ref();
341            let fr_u64: &[u64] = fr.as_ref();
342
343            let mut fr_not_smaller_than_fq = true;
344            for (left, right) in fr_u64.iter().zip(fq_u64.iter()).rev() {
345                if left < right {
346                    fr_not_smaller_than_fq = false;
347                    break;
348                }
349
350                if left > right {
351                    break;
352                }
353            }
354
355            if fr_not_smaller_than_fq {
356                F::MODULUS_BIT_SIZE as usize
357            } else {
358                F::MODULUS_BIT_SIZE as usize - 1
359            }
360        } else {
361            F::MODULUS_BIT_SIZE as usize - 1
362        };
363
364        // Step 3: group them based on the used capacity of F
365        let res = src_booleans
366            .chunks(capacity)
367            .map(|x| {
368                let mut res = x.to_vec();
369                res.reverse();
370                res
371            })
372            .collect::<Vec<Vec<Boolean<CF>>>>();
373        Ok(Self {
374            val: res,
375            _snark_field_: PhantomData,
376        })
377    }

425impl<F, CF> AllocVar<Vec<F>, CF> for NonNativeFieldInputVar<F, CF>
426where
427    F: PrimeField,
428    CF: PrimeField,
429{
430    fn new_variable<T: Borrow<Vec<F>>>(
431        cs: impl Into<Namespace<CF>>,
432        f: impl FnOnce() -> Result<T, SynthesisError>,
433        mode: AllocationMode,
434    ) -> Result<Self, SynthesisError> {
435        if mode == AllocationMode::Input {
436            Self::new_input(cs, f)
437        } else {
438            // directly allocate them as nonnative field elements
439
440            let ns = cs.into();
441            let cs = ns.cs();
442
443            let t = f()?;
444            let obj = t.borrow();
445            let mut allocated = Vec::<NonNativeFieldVar<F, CF>>::new();
446
447            for elem in obj.iter() {
448                let elem_allocated = NonNativeFieldVar::<F, CF>::new_variable(
449                    ns!(cs, "allocating element"),
450                    || Ok(elem),
451                    mode,
452                )?;
453                allocated.push(elem_allocated);
454            }
455
456            Ok(Self { val: allocated })
457        }
458    }
459
460    fn new_input<T: Borrow<Vec<F>>>(
461        cs: impl Into<Namespace<CF>>,
462        f: impl FnOnce() -> Result<T, SynthesisError>,
463    ) -> Result<Self, SynthesisError> {
464        // allocate the nonnative field elements by squeezing the bits like in BooleanInputVar
465
466        let ns = cs.into();
467        let cs = ns.cs();
468
469        let optimization_type = match cs.optimization_goal() {
470            OptimizationGoal::None => OptimizationType::Constraints,
471            OptimizationGoal::Constraints => OptimizationType::Constraints,
472            OptimizationGoal::Weight => OptimizationType::Weight,
473        };
474
475        let params = get_params(
476            F::MODULUS_BIT_SIZE as usize,
477            CF::MODULUS_BIT_SIZE as usize,
478            optimization_type,
479        );
480
481        let obj = f()?;
482
483        // Step 1: use BooleanInputVar to allocate the values as bits
484        // This is to make sure that we are using as few elements as possible
485        let boolean_allocation =
486            BooleanInputVar::new_input(ns!(cs, "boolean"), || Ok(obj.borrow()))?;
487
488        // Step 2: allocating the nonnative field elements as witnesses
489        let mut field_allocation = Vec::<AllocatedNonNativeFieldVar<F, CF>>::new();
490
491        for elem in obj.borrow().iter() {
492            let mut elem_allocated = AllocatedNonNativeFieldVar::<F, CF>::new_witness(
493                ns!(cs, "allocating element"),
494                || Ok(elem),
495            )?;
496
497            // due to the consistency check below
498            elem_allocated.is_in_the_normal_form = true;
499            elem_allocated.num_of_additions_over_normal_form = CF::zero();
500
501            field_allocation.push(elem_allocated);
502        }
503
504        // Step 3: check consistency
505        for (field_bits, field_elem) in boolean_allocation.val.iter().zip(field_allocation.iter()) {
506            let mut field_bits = field_bits.clone();
507            field_bits.reverse();
508
509            let bit_per_top_limb =
510                F::MODULUS_BIT_SIZE as usize - (params.num_limbs - 1) * params.bits_per_limb;
511            let bit_per_non_top_limb = params.bits_per_limb;
512
513            // must use lc to save computation
514            for (j, limb) in field_elem.limbs.iter().enumerate() {
515                let bits_slice = if j == 0 {
516                    field_bits[0..bit_per_top_limb].to_vec()
517                } else {
518                    field_bits[bit_per_top_limb + (j - 1) * bit_per_non_top_limb
519                        ..bit_per_top_limb + j * bit_per_non_top_limb]
520                        .to_vec()
521                };
522
523                let mut bit_sum = FpVar::<CF>::zero();
524                let mut cur = CF::one();
525
526                for bit in bits_slice.iter().rev() {
527                    bit_sum += <FpVar<CF> as From<Boolean<CF>>>::from((*bit).clone()) * cur;
528                    cur.double_in_place();
529                }
530
531                limb.enforce_equal(&bit_sum)?;
532            }
533        }
534
535        let mut wrapped_field_allocation = Vec::<NonNativeFieldVar<F, CF>>::new();
536        for field_gadget in field_allocation.iter() {
537            wrapped_field_allocation.push(NonNativeFieldVar::Var(field_gadget.clone()));
538        }
539        Ok(Self {
540            val: wrapped_field_allocation,
541        })
542    }
543}

545impl<F, CF> FromFieldElementsGadget<F, CF> for NonNativeFieldInputVar<F, CF>
546where
547    F: PrimeField,
548    CF: PrimeField,
549{
550    fn repack_input(src: &Vec<F>) -> Vec<CF> {
551        BooleanInputVar::repack_input(src)
552    }
553
554    fn from_field_elements(src: &Vec<FpVar<CF>>) -> Result<Self, SynthesisError> {
555        let cs = src.cs();
556
557        if cs == ConstraintSystemRef::None {
558            // Step 1: use BooleanInputVar to convert them into booleans
559            let boolean_allocation = BooleanInputVar::<F, CF>::from_field_elements(src)?;
560
561            // Step 2: construct the nonnative field gadgets from bits
562            let mut field_allocation = Vec::<NonNativeFieldVar<F, CF>>::new();
563
564            // reconstruct the field elements and check consistency
565            for field_bits in boolean_allocation.val.iter() {
566                let mut field_bits = field_bits.clone();
567                field_bits.resize(F::MODULUS_BIT_SIZE as usize, Boolean::<CF>::Constant(false));
568
569                let mut cur = F::one();
570
571                let mut value = F::zero();
572                for bit in field_bits.iter().rev() {
573                    if bit.value().unwrap_or_default() {
574                        value += &cur;
575                    }
576                    cur.double_in_place();
577                }
578
579                field_allocation.push(NonNativeFieldVar::Constant(value));
580            }
581
582            Ok(Self {
583                val: field_allocation,
584            })
585        } else {
586            let optimization_type = match cs.optimization_goal() {
587                OptimizationGoal::None => OptimizationType::Constraints,
588                OptimizationGoal::Constraints => OptimizationType::Constraints,
589                OptimizationGoal::Weight => OptimizationType::Weight,
590            };
591
592            let params = get_params(
593                F::MODULUS_BIT_SIZE as usize,
594                CF::MODULUS_BIT_SIZE as usize,
595                optimization_type,
596            );
597
598            // Step 1: use BooleanInputVar to convert them into booleans
599            let boolean_allocation = BooleanInputVar::<F, CF>::from_field_elements(src)?;
600
601            // Step 2: construct the nonnative field gadgets from bits
602            let mut field_allocation = Vec::<NonNativeFieldVar<F, CF>>::new();
603
604            // reconstruct the field elements and check consistency
605            for field_bits in boolean_allocation.val.iter() {
606                let mut field_bits = field_bits.clone();
607                field_bits.resize(F::MODULUS_BIT_SIZE as usize, Boolean::<CF>::Constant(false));
608                field_bits.reverse();
609
610                let mut limbs = Vec::<FpVar<CF>>::new();
611
612                let bit_per_top_limb =
613                    F::MODULUS_BIT_SIZE as usize - (params.num_limbs - 1) * params.bits_per_limb;
614                let bit_per_non_top_limb = params.bits_per_limb;
615
616                // must use lc to save computation
617                for j in 0..params.num_limbs {
618                    let bits_slice = if j == 0 {
619                        field_bits[0..bit_per_top_limb].to_vec()
620                    } else {
621                        field_bits[bit_per_top_limb + (j - 1) * bit_per_non_top_limb
622                            ..bit_per_top_limb + j * bit_per_non_top_limb]
623                            .to_vec()
624                    };
625
626                    let mut lc = LinearCombination::<CF>::zero();
627                    let mut cur = CF::one();
628
629                    let mut limb_value = CF::zero();
630                    for bit in bits_slice.iter().rev() {
631                        lc = &lc + bit.lc() * cur;
632                        if bit.value().unwrap_or_default() {
633                            limb_value += &cur;
634                        }
635                        cur.double_in_place();
636                    }
637
638                    let limb = AllocatedFp::<CF>::new_witness(ns!(cs, "limb"), || Ok(limb_value))?;
639                    lc = lc - limb.variable;
640                    cs.enforce_constraint(lc!(), lc!(), lc).unwrap();
641
642                    limbs.push(FpVar::from(limb));
643                }
644
645                field_allocation.push(NonNativeFieldVar::<F, CF>::Var(
646                    AllocatedNonNativeFieldVar::<F, CF> {
647                        cs: cs.clone(),
