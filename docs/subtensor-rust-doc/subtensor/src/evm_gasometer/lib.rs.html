47	( $inner:expr, $e:expr ) => {
48		match $e {

92	pub fn gas_cost(&self, cost: GasCost, gas: u64) -> Result<u64, ExitError> {
93		match self.inner.as_ref() {

110	#[inline]
111	/// Remaining gas.
112	pub fn gas(&self) -> u64 {
113		match self.inner.as_ref() {
114			Ok(inner) => self.gas_limit - inner.used_gas - inner.memory_gas,
115			Err(_) => 0,
116		}
117	}

119	#[inline]
120	/// Total used gas.
121	pub fn total_used_gas(&self) -> u64 {
122		match self.inner.as_ref() {
123			Ok(inner) => inner.used_gas + inner.memory_gas,
124			Err(_) => self.gas_limit,
125		}
126	}

132			Ok(inner) => inner.refunded_gas,
133			Err(_) => 0,
134		}
135	}

137	/// Explicitly fail the gasometer with out of gas. Return `OutOfGas` error.

143	#[inline]
144	/// Record an explicit cost.
145	pub fn record_cost(&mut self, cost: u64) -> Result<(), ExitError> {
146		event!(RecordCost {
147			cost,
148			snapshot: self.snapshot(),
149		});
150		log_gas!(self, "Record cost {}", cost);
151
152		let all_gas_cost = self.total_used_gas() + cost;
153		if self.gas_limit < all_gas_cost {
154			self.inner = Err(ExitError::OutOfGas);
155			return Err(ExitError::OutOfGas);
156		}
157
158		self.inner_mut()?.used_gas += cost;
159		Ok(())
160	}

182	/// Record opcode gas cost.
183	pub fn record_dynamic_cost(
184		&mut self,
185		cost: GasCost,
186		memory: Option<MemoryCost>,
187	) -> Result<(), ExitError> {
188		let gas = self.gas();
189
190		let memory_gas = match memory {
191			Some(memory) => try_or_fail!(self.inner, self.inner_mut()?.memory_gas(memory)),
192			None => self.inner_mut()?.memory_gas,
193		};
194		let gas_cost = try_or_fail!(self.inner, self.inner_mut()?.gas_cost(cost, gas));
195		let gas_refund = self.inner_mut()?.gas_refund(cost);
196		let used_gas = self.inner_mut()?.used_gas;
197
198		event!(RecordDynamicCost {
199			gas_cost,
200			memory_gas,
201			gas_refund,
202			snapshot: self.snapshot(),
203		});
204
205		let all_gas_cost = memory_gas + used_gas + gas_cost;
206		if self.gas_limit < all_gas_cost {
207			self.inner = Err(ExitError::OutOfGas);
208			return Err(ExitError::OutOfGas);
209		}
210
211		log_gas!(
212			self,
213			"Record dynamic cost {} - memory_gas {} - gas_refund {}",
214			gas_cost,
215			memory_gas,
216			gas_refund
217		);
218
219		let after_gas = self.gas_limit - all_gas_cost;
220		try_or_fail!(self.inner, self.inner_mut()?.extra_check(cost, after_gas));
221
222		self.inner_mut()?.used_gas += gas_cost;
223		self.inner_mut()?.memory_gas = memory_gas;
224		self.inner_mut()?.refunded_gas += gas_refund;
225
226		Ok(())
227	}

242	/// Record transaction cost.
243	pub fn record_transaction(&mut self, cost: TransactionCost) -> Result<(), ExitError> {
244		let gas_cost = match cost {
245			TransactionCost::Call {
246				zero_data_len,
247				non_zero_data_len,
248				access_list_address_len,
249				access_list_storage_len,
250			} => {
251				#[deny(clippy::let_and_return)]
252				let cost = self.config.gas_transaction_call
253					+ zero_data_len as u64 * self.config.gas_transaction_zero_data
254					+ non_zero_data_len as u64 * self.config.gas_transaction_non_zero_data
255					+ access_list_address_len as u64 * self.config.gas_access_list_address
256					+ access_list_storage_len as u64 * self.config.gas_access_list_storage_key;
257
258				log_gas!(
259					self,
260					"Record Call {} [gas_transaction_call: {}, zero_data_len: {}, non_zero_data_len: {}, access_list_address_len: {}, access_list_storage_len: {}]",
261					cost,
262					self.config.gas_transaction_call,
263					zero_data_len,
264					non_zero_data_len,
265					access_list_address_len,
266					access_list_storage_len
267				);
268
269				cost
270			}
271			TransactionCost::Create {
272				zero_data_len,
273				non_zero_data_len,
274				access_list_address_len,
275				access_list_storage_len,
276				initcode_cost,
277			} => {
278				let mut cost = self.config.gas_transaction_create
279					+ zero_data_len as u64 * self.config.gas_transaction_zero_data
280					+ non_zero_data_len as u64 * self.config.gas_transaction_non_zero_data
281					+ access_list_address_len as u64 * self.config.gas_access_list_address
282					+ access_list_storage_len as u64 * self.config.gas_access_list_storage_key;
283				if self.config.max_initcode_size.is_some() {
284					cost += initcode_cost;
285				}
286
287				log_gas!(
288					self,
289					"Record Create {} [gas_transaction_create: {}, zero_data_len: {}, non_zero_data_len: {}, access_list_address_len: {}, access_list_storage_len: {}, initcode_cost: {}]",
290					cost,
291					self.config.gas_transaction_create,
292					zero_data_len,
293					non_zero_data_len,
294					access_list_address_len,
295					access_list_storage_len,
296					initcode_cost
297				);
298				cost
299			}
300		};
301
302		event!(RecordTransaction {
303			cost: gas_cost,
304			snapshot: self.snapshot(),
305		});
306
307		if self.gas() < gas_cost {
308			self.inner = Err(ExitError::OutOfGas);
309			return Err(ExitError::OutOfGas);
310		}
311
312		self.inner_mut()?.used_gas += gas_cost;
313		Ok(())
314	}

327/// Calculate the call transaction cost.
328#[allow(clippy::naive_bytecount)]
329pub fn call_transaction_cost(data: &[u8], access_list: &[(H160, Vec<H256>)]) -> TransactionCost {
330	let zero_data_len = data.iter().filter(|v| **v == 0).count();
331	let non_zero_data_len = data.len() - zero_data_len;
332	let (access_list_address_len, access_list_storage_len) = count_access_list(access_list);
333
334	TransactionCost::Call {
335		zero_data_len,
336		non_zero_data_len,
337		access_list_address_len,
338		access_list_storage_len,
339	}
340}

342/// Calculate the create transaction cost.
343#[allow(clippy::naive_bytecount)]
344pub fn create_transaction_cost(data: &[u8], access_list: &[(H160, Vec<H256>)]) -> TransactionCost {
345	let zero_data_len = data.iter().filter(|v| **v == 0).count();
346	let non_zero_data_len = data.len() - zero_data_len;
347	let (access_list_address_len, access_list_storage_len) = count_access_list(access_list);
348	let initcode_cost = init_code_cost(data);
349
350	TransactionCost::Create {
351		zero_data_len,
352		non_zero_data_len,
353		access_list_address_len,
354		access_list_storage_len,
355		initcode_cost,
356	}
357}

359pub fn init_code_cost(data: &[u8]) -> u64 {
360	// As per EIP-3860:
361	// > We define initcode_cost(initcode) to equal INITCODE_WORD_COST * ceil(len(initcode) / 32).
362	// where INITCODE_WORD_COST is 2.
363	2 * ((data.len() as u64 + 31) / 32)
364}
365
366/// Counts the number of addresses and storage keys in the access list
367fn count_access_list(access_list: &[(H160, Vec<H256>)]) -> (usize, usize) {
368	let access_list_address_len = access_list.len();
369	let access_list_storage_len = access_list.iter().map(|(_, keys)| keys.len()).sum();
370
371	(access_list_address_len, access_list_storage_len)
372}

497/// Calculate the opcode cost.
498#[allow(clippy::nonminimal_bool)]
499pub fn dynamic_opcode_cost<H: Handler>(
500	address: H160,
501	opcode: Opcode,
502	stack: &Stack,
503	is_static: bool,
504	config: &Config,
505	handler: &mut H,
506) -> Result<(GasCost, StorageTarget, Option<MemoryCost>), ExitError> {
507	let mut storage_target = StorageTarget::None;
508	let gas_cost = match opcode {
509		Opcode::RETURN => GasCost::Zero,
510
511		Opcode::MLOAD | Opcode::MSTORE | Opcode::MSTORE8 => GasCost::VeryLow,
512
513		Opcode::REVERT if config.has_revert => GasCost::Zero,
514		Opcode::REVERT => GasCost::Invalid(opcode),
515
516		Opcode::CHAINID if config.has_chain_id => GasCost::Base,
517		Opcode::CHAINID => GasCost::Invalid(opcode),
518
519		Opcode::SHL | Opcode::SHR | Opcode::SAR if config.has_bitwise_shifting => GasCost::VeryLow,
520		Opcode::SHL | Opcode::SHR | Opcode::SAR => GasCost::Invalid(opcode),
521
522		Opcode::SELFBALANCE if config.has_self_balance => GasCost::Low,
523		Opcode::SELFBALANCE => GasCost::Invalid(opcode),
524
525		Opcode::BASEFEE if config.has_base_fee => GasCost::Base,
526		Opcode::BASEFEE => GasCost::Invalid(opcode),
527
528		Opcode::EXTCODESIZE => {
529			let target = stack.peek(0)?.into();
530			storage_target = StorageTarget::Address(target);
531			GasCost::ExtCodeSize {
532				target_is_cold: handler.is_cold(target, None)?,
533			}
534		}
535		Opcode::BALANCE => {
536			let target = stack.peek(0)?.into();
537			storage_target = StorageTarget::Address(target);
538			GasCost::Balance {
539				target_is_cold: handler.is_cold(target, None)?,
540			}
541		}
542		Opcode::BLOCKHASH => GasCost::BlockHash,
543
544		Opcode::EXTCODEHASH if config.has_ext_code_hash => {
545			let target = stack.peek(0)?.into();
546			storage_target = StorageTarget::Address(target);
547			GasCost::ExtCodeHash {
548				target_is_cold: handler.is_cold(target, None)?,
549			}
550		}
551		Opcode::EXTCODEHASH => GasCost::Invalid(opcode),
552
553		Opcode::CALLCODE => {
554			let target = stack.peek(1)?.into();
555			storage_target = StorageTarget::Address(target);
556			GasCost::CallCode {
557				value: U256::from_big_endian(&stack.peek(2)?[..]),
558				gas: U256::from_big_endian(&stack.peek(0)?[..]),
559				target_is_cold: handler.is_cold(target, None)?,
560				target_exists: {
561					handler.record_external_operation(evm_core::ExternalOperation::IsEmpty)?;
562					handler.exists(target)
563				},
564			}
565		}
566		Opcode::STATICCALL => {
567			let target = stack.peek(1)?.into();
568			storage_target = StorageTarget::Address(target);
569			GasCost::StaticCall {
570				gas: U256::from_big_endian(&stack.peek(0)?[..]),
571				target_is_cold: handler.is_cold(target, None)?,
572				target_exists: {
573					handler.record_external_operation(evm_core::ExternalOperation::IsEmpty)?;
574					handler.exists(target)
575				},
576			}
577		}
578		Opcode::SHA3 => GasCost::Sha3 {
579			len: U256::from_big_endian(&stack.peek(1)?[..]),
580		},
581		Opcode::EXTCODECOPY => {
582			let target = stack.peek(0)?.into();
583			storage_target = StorageTarget::Address(target);
584			GasCost::ExtCodeCopy {
585				target_is_cold: handler.is_cold(target, None)?,
586				len: U256::from_big_endian(&stack.peek(3)?[..]),
587			}
588		}
589		Opcode::CALLDATACOPY | Opcode::CODECOPY => GasCost::VeryLowCopy {
590			len: U256::from_big_endian(&stack.peek(2)?[..]),
591		},
592		Opcode::EXP => GasCost::Exp {
593			power: U256::from_big_endian(&stack.peek(1)?[..]),
594		},
595		Opcode::SLOAD => {
596			let index = stack.peek(0)?;
597			storage_target = StorageTarget::Slot(address, index);
598			GasCost::SLoad {
599				target_is_cold: handler.is_cold(address, Some(index))?,
600			}
601		}
602
603		Opcode::DELEGATECALL if config.has_delegate_call => {
604			let target = stack.peek(1)?.into();
605			storage_target = StorageTarget::Address(target);
606			GasCost::DelegateCall {
607				gas: U256::from_big_endian(&stack.peek(0)?[..]),
608				target_is_cold: handler.is_cold(target, None)?,
609				target_exists: {
610					handler.record_external_operation(evm_core::ExternalOperation::IsEmpty)?;
611					handler.exists(target)
612				},
613			}
614		}
615		Opcode::DELEGATECALL => GasCost::Invalid(opcode),
616
617		Opcode::RETURNDATASIZE if config.has_return_data => GasCost::Base,
618		Opcode::RETURNDATACOPY if config.has_return_data => GasCost::VeryLowCopy {
619			len: U256::from_big_endian(&stack.peek(2)?[..]),
620		},
621		Opcode::RETURNDATASIZE | Opcode::RETURNDATACOPY => GasCost::Invalid(opcode),
622
623		Opcode::SSTORE if !is_static => {
624			let index = stack.peek(0)?;
625			let value = stack.peek(1)?;
626			storage_target = StorageTarget::Slot(address, index);
627
628			GasCost::SStore {
629				original: handler.original_storage(address, index),
630				current: handler.storage(address, index),
631				new: value,
632				target_is_cold: handler.is_cold(address, Some(index))?,
633			}
634		}
635		Opcode::LOG0 if !is_static => GasCost::Log {
636			n: 0,
637			len: U256::from_big_endian(&stack.peek(1)?[..]),
638		},
639		Opcode::LOG1 if !is_static => GasCost::Log {
640			n: 1,
641			len: U256::from_big_endian(&stack.peek(1)?[..]),
642		},
643		Opcode::LOG2 if !is_static => GasCost::Log {
644			n: 2,
645			len: U256::from_big_endian(&stack.peek(1)?[..]),
646		},
647		Opcode::LOG3 if !is_static => GasCost::Log {
648			n: 3,
649			len: U256::from_big_endian(&stack.peek(1)?[..]),
650		},
651		Opcode::LOG4 if !is_static => GasCost::Log {
652			n: 4,
653			len: U256::from_big_endian(&stack.peek(1)?[..]),
654		},
655		Opcode::CREATE if !is_static => GasCost::Create,
656		Opcode::CREATE2 if !is_static && config.has_create2 => GasCost::Create2 {
657			len: U256::from_big_endian(&stack.peek(2)?[..]),
658		},
659		Opcode::SUICIDE if !is_static => {
660			let target = stack.peek(0)?.into();
661			storage_target = StorageTarget::Address(target);
662			GasCost::Suicide {
663				value: handler.balance(address),
664				target_is_cold: handler.is_cold(target, None)?,
665				target_exists: {
666					handler.record_external_operation(evm_core::ExternalOperation::IsEmpty)?;
667					handler.exists(target)
668				},
669				already_removed: handler.deleted(address),
670			}
671		}
672		Opcode::CALL
673			if !is_static
674				|| (is_static && U256::from_big_endian(&stack.peek(2)?[..]) == U256::zero()) =>
675		{
676			let target = stack.peek(1)?.into();
677			storage_target = StorageTarget::Address(target);
678			GasCost::Call {
679				value: U256::from_big_endian(&stack.peek(2)?[..]),
680				gas: U256::from_big_endian(&stack.peek(0)?[..]),
681				target_is_cold: handler.is_cold(target, None)?,
682				target_exists: {
683					handler.record_external_operation(evm_core::ExternalOperation::IsEmpty)?;
684					handler.exists(target)
685				},
686			}
687		}
688
689		Opcode::PUSH0 if config.has_push0 => GasCost::Base,
690
691		_ => GasCost::Invalid(opcode),
692	};
693
694	let memory_cost = match opcode {
695		Opcode::SHA3
696		| Opcode::RETURN
697		| Opcode::REVERT
698		| Opcode::LOG0
699		| Opcode::LOG1
700		| Opcode::LOG2
701		| Opcode::LOG3
702		| Opcode::LOG4 => Some(MemoryCost {
703			offset: U256::from_big_endian(&stack.peek(0)?[..]),
704			len: U256::from_big_endian(&stack.peek(1)?[..]),
705		}),
706
707		Opcode::CODECOPY | Opcode::CALLDATACOPY | Opcode::RETURNDATACOPY => Some(MemoryCost {
708			offset: U256::from_big_endian(&stack.peek(0)?[..]),
709			len: U256::from_big_endian(&stack.peek(2)?[..]),
710		}),
711
712		Opcode::EXTCODECOPY => Some(MemoryCost {
713			offset: U256::from_big_endian(&stack.peek(1)?[..]),
714			len: U256::from_big_endian(&stack.peek(3)?[..]),
715		}),
716
717		Opcode::MLOAD | Opcode::MSTORE => Some(MemoryCost {
718			offset: U256::from_big_endian(&stack.peek(0)?[..]),
719			len: U256::from(32),
720		}),
721
722		Opcode::MSTORE8 => Some(MemoryCost {
723			offset: U256::from_big_endian(&stack.peek(0)?[..]),
724			len: U256::from(1),
725		}),
726
727		Opcode::CREATE | Opcode::CREATE2 => Some(MemoryCost {
728			offset: U256::from_big_endian(&stack.peek(1)?[..]),
729			len: U256::from_big_endian(&stack.peek(2)?[..]),
730		}),
731
732		Opcode::CALL | Opcode::CALLCODE => Some(
733			MemoryCost {
734				offset: U256::from_big_endian(&stack.peek(3)?[..]),
735				len: U256::from_big_endian(&stack.peek(4)?[..]),
736			}
737			.join(MemoryCost {
738				offset: U256::from_big_endian(&stack.peek(5)?[..]),
739				len: U256::from_big_endian(&stack.peek(6)?[..]),
740			}),
741		),
742
743		Opcode::DELEGATECALL | Opcode::STATICCALL => Some(
744			MemoryCost {
745				offset: U256::from_big_endian(&stack.peek(2)?[..]),
746				len: U256::from_big_endian(&stack.peek(3)?[..]),
747			}
748			.join(MemoryCost {
749				offset: U256::from_big_endian(&stack.peek(4)?[..]),
750				len: U256::from_big_endian(&stack.peek(5)?[..]),
751			}),
752		),
753
754		_ => None,
755	};
756
757	Ok((gas_cost, storage_target, memory_cost))
758}

760/// Holds the gas consumption for a Gasometer instance.
761#[derive(Clone, Debug)]
762struct Inner<'config> {
763	memory_gas: u64,
764	used_gas: u64,
765	refunded_gas: i64,
766	config: &'config Config,
767}
768
769impl<'config> Inner<'config> {
770	fn memory_gas(&self, memory: MemoryCost) -> Result<u64, ExitError> {
771		let from = memory.offset;
772		let len = memory.len;
773
774		if len == U256::zero() {
775			return Ok(self.memory_gas);
776		}
777
778		let end = from.checked_add(len).ok_or(ExitError::OutOfGas)?;
779
780		if end > U256::from(usize::MAX) {
781			return Err(ExitError::OutOfGas);
782		}
783		let end = end.as_usize();
784
785		let rem = end % 32;
786		let new = if rem == 0 { end / 32 } else { end / 32 + 1 };
787
788		Ok(max(self.memory_gas, memory::memory_gas(new)?))
789	}
790
791	fn extra_check(&self, cost: GasCost, after_gas: u64) -> Result<(), ExitError> {
792		match cost {
793			GasCost::Call { gas, .. } => costs::call_extra_check(gas, after_gas, self.config),
794			GasCost::CallCode { gas, .. } => costs::call_extra_check(gas, after_gas, self.config),
795			GasCost::DelegateCall { gas, .. } => {
796				costs::call_extra_check(gas, after_gas, self.config)
797			}
798			GasCost::StaticCall { gas, .. } => costs::call_extra_check(gas, after_gas, self.config),
799			_ => Ok(()),
800		}
801	}
802
803	/// Returns the gas cost numerical value.
804	fn gas_cost(&self, cost: GasCost, gas: u64) -> Result<u64, ExitError> {
805		Ok(match cost {
806			GasCost::Call {
807				value,
808				target_is_cold,
809				target_exists,
810				..
811			} => costs::call_cost(
812				value,
813				target_is_cold,
814				true,
815				true,
816				!target_exists,
817				self.config,
818			),
819			GasCost::CallCode {
820				value,
821				target_is_cold,
822				target_exists,
823				..
824			} => costs::call_cost(
825				value,
826				target_is_cold,
827				true,
828				false,
829				!target_exists,
830				self.config,
831			),
832			GasCost::DelegateCall {
833				target_is_cold,
834				target_exists,
835				..
836			} => costs::call_cost(
837				U256::zero(),
838				target_is_cold,
839				false,
840				false,
841				!target_exists,
842				self.config,
843			),
844			GasCost::StaticCall {
845				target_is_cold,
846				target_exists,
847				..
848			} => costs::call_cost(
849				U256::zero(),
850				target_is_cold,
851				false,
852				true,
853				!target_exists,
854				self.config,
855			),
856
857			GasCost::Suicide {
858				value,
859				target_is_cold,
860				target_exists,
861				..
862			} => costs::suicide_cost(value, target_is_cold, target_exists, self.config),
863			GasCost::SStore {
864				original,
865				current,
866				new,
867				target_is_cold,
868			} => costs::sstore_cost(original, current, new, gas, target_is_cold, self.config)?,
869
870			GasCost::Sha3 { len } => costs::sha3_cost(len)?,
871			GasCost::Log { n, len } => costs::log_cost(n, len)?,
872			GasCost::VeryLowCopy { len } => costs::verylowcopy_cost(len)?,
873			GasCost::Exp { power } => costs::exp_cost(power, self.config)?,
874			GasCost::Create => consts::G_CREATE,
875			GasCost::Create2 { len } => costs::create2_cost(len)?,
876			GasCost::SLoad { target_is_cold } => costs::sload_cost(target_is_cold, self.config),
877
878			GasCost::Zero => consts::G_ZERO,
879			GasCost::Base => consts::G_BASE,
880			GasCost::VeryLow => consts::G_VERYLOW,
881			GasCost::Low => consts::G_LOW,
882			GasCost::Invalid(opcode) => return Err(ExitError::InvalidCode(opcode)),
883
884			GasCost::ExtCodeSize { target_is_cold } => {
885				costs::address_access_cost(target_is_cold, self.config.gas_ext_code, self.config)
886			}
887			GasCost::ExtCodeCopy {
888				target_is_cold,
889				len,
890			} => costs::extcodecopy_cost(len, target_is_cold, self.config)?,
891			GasCost::Balance { target_is_cold } => {
892				costs::address_access_cost(target_is_cold, self.config.gas_balance, self.config)
893			}
894			GasCost::BlockHash => consts::G_BLOCKHASH,
895			GasCost::ExtCodeHash { target_is_cold } => costs::address_access_cost(
896				target_is_cold,
897				self.config.gas_ext_code_hash,
898				self.config,
899			),
900		})
901	}
902
903	fn gas_refund(&self, cost: GasCost) -> i64 {
904		match cost {
905			_ if self.config.estimate => 0,
906
907			GasCost::SStore {
908				original,
909				current,
910				new,
911				..
912			} => costs::sstore_refund(original, current, new, self.config),
913			GasCost::Suicide {
914				already_removed, ..
915			} if !self.config.decrease_clears_refund => costs::suicide_refund(already_removed),
916			_ => 0,
917		}
918	}
919}

1098		/// Number of addresses in transaction access list (see EIP-2930)

1104	},

1107impl MemoryCost {
1108	/// Join two memory cost together.
1109	pub fn join(self, other: MemoryCost) -> MemoryCost {
1110		if self.len == U256::zero() {
1111			return other;
1112		}
1113
1114		if other.len == U256::zero() {
1115			return self;
1116		}
1117
1118		let self_end = self.offset.saturating_add(self.len);
1119		let other_end = other.offset.saturating_add(other.len);
1120
1121		if self_end >= other_end {
1122			self
1123		} else {
1124			other
1125		}
1126	}
1127}
