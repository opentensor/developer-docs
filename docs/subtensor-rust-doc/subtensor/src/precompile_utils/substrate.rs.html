38#[derive(Debug)]
39pub enum TryDispatchError {
40	Evm(ExitError),
41	Substrate(DispatchError),
42}
43
44impl From<TryDispatchError> for PrecompileFailure {
45	fn from(f: TryDispatchError) -> PrecompileFailure {
46		match f {
47			TryDispatchError::Evm(e) => PrecompileFailure::Error { exit_status: e },
48			TryDispatchError::Substrate(e) => {
49				revert(alloc::format!("Dispatched call failed with error: {e:?}"))
50			}
51		}
52	}
53}

55/// Helper functions requiring a Substrate runtime.
56/// This runtime must of course implement `pallet_evm::Config`.
57#[derive(Clone, Copy, Debug)]
58pub struct RuntimeHelper<Runtime>(PhantomData<Runtime>);
59
60impl<Runtime> RuntimeHelper<Runtime>
61where
62	Runtime: pallet_evm::Config,
63	Runtime::RuntimeCall: Dispatchable<PostInfo = PostDispatchInfo> + GetDispatchInfo,
64{
65	#[inline(always)]
66	pub fn record_weight_v2_cost(
67		handle: &mut impl PrecompileHandle,
68		weight: Weight,
69	) -> Result<(), ExitError> {
70		// Make sure there is enough gas.
71		let remaining_gas = handle.remaining_gas();
72		let required_gas = Runtime::GasWeightMapping::weight_to_gas(weight);
73		if required_gas > remaining_gas {
74			return Err(ExitError::OutOfGas);
75		}
76
77		// Make sure there is enough remaining weight
78		// TODO: record ref time when precompile will be benchmarked
79		handle.record_external_cost(None, Some(weight.proof_size()), None)
80	}
81
82	#[inline(always)]
83	pub fn refund_weight_v2_cost(
84		handle: &mut impl PrecompileHandle,
85		weight: Weight,
86		maybe_actual_weight: Option<Weight>,
87	) -> Result<u64, ExitError> {
88		// Refund weights and compute used weight them record used gas
89		// TODO: refund ref time when precompile will be benchmarked
90		let used_weight = if let Some(actual_weight) = maybe_actual_weight {
91			let refund_weight = weight.checked_sub(&actual_weight).unwrap_or_default();
92			handle.refund_external_cost(None, Some(refund_weight.proof_size()));
93			actual_weight
94		} else {
95			weight
96		};
97		let used_gas = Runtime::GasWeightMapping::weight_to_gas(used_weight);
98		handle.record_cost(used_gas)?;
99		Ok(used_gas)
100	}
101
102	/// Try to dispatch a Substrate call.
103	/// Return an error if there are not enough gas, or if the call fails.
104	/// If successful returns the used gas using the Runtime GasWeightMapping.
105	pub fn try_dispatch<Call>(
106		handle: &mut impl PrecompileHandle,
107		origin: <Runtime::RuntimeCall as Dispatchable>::RuntimeOrigin,
108		call: Call,
109	) -> Result<PostDispatchInfo, TryDispatchError>
110	where
111		Runtime::RuntimeCall: From<Call>,
112	{
113		let call = Runtime::RuntimeCall::from(call);
114		let dispatch_info = call.get_dispatch_info();
115
116		Self::record_weight_v2_cost(handle, dispatch_info.weight).map_err(TryDispatchError::Evm)?;
117
118		// Dispatch call.
119		// It may be possible to not record gas cost if the call returns Pays::No.
120		// However while Substrate handle checking weight while not making the sender pay for it,
121		// the EVM doesn't. It seems this safer to always record the costs to avoid unmetered
122		// computations.
123		let post_dispatch_info = using_precompile_handle(handle, || call.dispatch(origin))
124			.map_err(|e| TryDispatchError::Substrate(e.error))?;
125
126		Self::refund_weight_v2_cost(
127			handle,
128			dispatch_info.weight,
129			post_dispatch_info.actual_weight,
130		)
131		.map_err(TryDispatchError::Evm)?;
132
133		Ok(post_dispatch_info)
134	}
135}

137impl<Runtime> RuntimeHelper<Runtime>
138where
139	Runtime: pallet_evm::Config,
140{
141	/// Cost of a Substrate DB write in gas.
142	pub fn db_write_gas_cost() -> u64 {
143		<Runtime as pallet_evm::Config>::GasWeightMapping::weight_to_gas(
144			<Runtime as frame_system::Config>::DbWeight::get().writes(1),
145		)
146	}
147
148	/// Cost of a Substrate DB read in gas.
149	pub fn db_read_gas_cost() -> u64 {
150		<Runtime as pallet_evm::Config>::GasWeightMapping::weight_to_gas(
151			<Runtime as frame_system::Config>::DbWeight::get().reads(1),
152		)
153	}
154}
