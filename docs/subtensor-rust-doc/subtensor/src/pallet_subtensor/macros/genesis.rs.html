1use frame_support::pallet_macros::pallet_section;

3/// A [`pallet_section`] that defines the errors for a pallet.
4/// This can later be imported into the pallet using [`import_section`].
5#[pallet_section]
6mod genesis {
7
8    #[pallet::genesis_build]
9    impl<T: Config> BuildGenesisConfig for GenesisConfig<T> {
10        fn build(&self) {
11            // Set initial total issuance from balances
12            TotalIssuance::<T>::put(self.balances_issuance);
13
14            // Get the root network uid.
15            let root_netuid: u16 = 0;
16
17            // Set the root network as added.
18            NetworksAdded::<T>::insert(root_netuid, true);
19
20            // Increment the number of total networks.
21            TotalNetworks::<T>::mutate(|n| *n = n.saturating_add(1));
22
23            // Set the number of validators to 1.
24            SubnetworkN::<T>::insert(root_netuid, 0);
25
26            // Set the maximum number to the number of senate members.
27            MaxAllowedUids::<T>::insert(root_netuid, 64u16);
28
29            // Set the maximum number to the number of validators to all members.
30            MaxAllowedValidators::<T>::insert(root_netuid, 64u16);
31
32            // Set the min allowed weights to zero, no weights restrictions.
33            MinAllowedWeights::<T>::insert(root_netuid, 0);
34
35            // Set the max weight limit to infitiy, no weight restrictions.
36            MaxWeightsLimit::<T>::insert(root_netuid, u16::MAX);
37
38            // Add default root tempo.
39            Tempo::<T>::insert(root_netuid, 100);
40
41            // Set the root network as open.
42            NetworkRegistrationAllowed::<T>::insert(root_netuid, true);
43
44            // Set target registrations for validators as 1 per block.
45            TargetRegistrationsPerInterval::<T>::insert(root_netuid, 1);
46
47            for net in 1..2 {
48                let netuid: u16 = net as u16;
49                let hotkey = DefaultAccount::<T>::get();
50                SubnetMechanism::<T>::insert(netuid, 1); // Make dynamic.
51                Owner::<T>::insert(hotkey.clone(), hotkey.clone());
52                SubnetAlphaIn::<T>::insert(netuid, 10_000_000_000);
53                SubnetTAO::<T>::insert(netuid, 10_000_000_000);
54                NetworksAdded::<T>::insert(netuid, true);
55                TotalNetworks::<T>::mutate(|n| *n = n.saturating_add(1));
56                SubnetworkN::<T>::insert(netuid, 0);
57                MaxAllowedUids::<T>::insert(netuid, 256u16);
58                MaxAllowedValidators::<T>::insert(netuid, 64u16);
59                MinAllowedWeights::<T>::insert(netuid, 0);
60                MaxWeightsLimit::<T>::insert(netuid, u16::MAX);
61                Tempo::<T>::insert(netuid, 100);
62                NetworkRegistrationAllowed::<T>::insert(netuid, true);
63                SubnetOwner::<T>::insert(netuid, hotkey.clone());
64                SubnetLocked::<T>::insert(netuid, 1);
65                LargestLocked::<T>::insert(netuid, 1);
66                Alpha::<T>::insert(
67                    // Lock the initial funds making this key the owner.
68                    (hotkey.clone(), hotkey.clone(), netuid),
69                    U64F64::saturating_from_num(1_000_000_000),
70                );
71                TotalHotkeyAlpha::<T>::insert(hotkey.clone(), netuid, 1_000_000_000);
72                TotalHotkeyShares::<T>::insert(
73                    hotkey.clone(),
74                    netuid,
75                    U64F64::saturating_from_num(1_000_000_000),
76                );
77                // TotalColdkeyAlpha::<T>::insert(hotkey.clone(), netuid, 1_000_000_000);
78                SubnetAlphaOut::<T>::insert(netuid, 1_000_000_000);
79                let mut staking_hotkeys = StakingHotkeys::<T>::get(hotkey.clone());
80                if !staking_hotkeys.contains(&hotkey) {
81                    staking_hotkeys.push(hotkey.clone());
82                    StakingHotkeys::<T>::insert(hotkey.clone(), staking_hotkeys.clone());
83                }
84
85                let block_number = Pallet::<T>::get_current_block_as_u64();
86                for next_uid_s in 0..1 {
87                    let next_uid: u16 = next_uid_s as u16;
88                    SubnetworkN::<T>::insert(netuid, next_uid.saturating_add(1));
89                    Rank::<T>::mutate(netuid, |v| v.push(0));
90                    Trust::<T>::mutate(netuid, |v| v.push(0));
91                    Active::<T>::mutate(netuid, |v| v.push(true));
92                    Emission::<T>::mutate(netuid, |v| v.push(0));
93                    Consensus::<T>::mutate(netuid, |v| v.push(0));
94                    Incentive::<T>::mutate(netuid, |v| v.push(0));
95                    Dividends::<T>::mutate(netuid, |v| v.push(0));
96                    LastUpdate::<T>::mutate(netuid, |v| v.push(block_number));
97                    PruningScores::<T>::mutate(netuid, |v| v.push(0));
98                    ValidatorTrust::<T>::mutate(netuid, |v| v.push(0));
99                    ValidatorPermit::<T>::mutate(netuid, |v| v.push(false));
100                    Keys::<T>::insert(netuid, next_uid, hotkey.clone()); // Make hotkey - uid association.
101                    Uids::<T>::insert(netuid, hotkey.clone(), next_uid); // Make uid - hotkey association.
102                    BlockAtRegistration::<T>::insert(netuid, next_uid, block_number); // Fill block at registration.
103                    IsNetworkMember::<T>::insert(hotkey.clone(), netuid, true); // Fill network is member.
104                }
105            }
106        }
107    }
108}
