11    impl<T: Config> Hooks<BlockNumberFor<T>> for Pallet<T> {
12        // ---- Called on the initialization of this pallet. (the order of on_finalize calls is determined in the runtime)
13        //
14        // # Args:
15        // 	* 'n': (BlockNumberFor<T>):
16        // 		- The number of the block we are initializing.
17        fn on_initialize(_block_number: BlockNumberFor<T>) -> Weight {
18            let block_step_result = Self::block_step();
19            match block_step_result {
20                Ok(_) => {
21                    // --- If the block step was successful, return the weight.
22                    log::debug!("Successfully ran block step.");
23                    Weight::from_parts(110_634_229_000_u64, 0)
24                        .saturating_add(T::DbWeight::get().reads(8304_u64))
25                        .saturating_add(T::DbWeight::get().writes(110_u64))
26                }
27                Err(e) => {
28                    // --- If the block step was unsuccessful, return the weight anyway.
29                    log::error!("Error while stepping block: {:?}", e);
30                    Weight::from_parts(110_634_229_000_u64, 0)
31                        .saturating_add(T::DbWeight::get().reads(8304_u64))
32                        .saturating_add(T::DbWeight::get().writes(110_u64))
33                }
34            }
35        }
36
37        // ---- Called on the finalization of this pallet. The code weight must be taken into account prior to the execution of this macro.
38        //
39        // # Args:
40        // 	* 'n': (BlockNumberFor<T>):
41        // 		- The number of the block we are finalizing.
42        fn on_finalize(_block_number: BlockNumberFor<T>) {
43            // Self::do_on_finalize(block_number);
44        }

46        fn on_runtime_upgrade() -> frame_support::weights::Weight {
47            // --- Migrate storage
48            let mut weight = frame_support::weights::Weight::from_parts(0, 0);
49
50            // Hex encoded foundation coldkey
51            let hex = hex_literal::hex![
52                "feabaafee293d3b76dae304e2f9d885f77d2b17adab9e17e921b321eccd61c77"
53            ];
54            weight = weight
55                // Initializes storage version (to 1)
56                .saturating_add(migrations::migrate_to_v1_separate_emission::migrate_to_v1_separate_emission::<T>())
57                // Storage version v1 -> v2
58                .saturating_add(migrations::migrate_to_v2_fixed_total_stake::migrate_to_v2_fixed_total_stake::<T>())
59                // Doesn't check storage version. TODO: Remove after upgrade
60                .saturating_add(migrations::migrate_create_root_network::migrate_create_root_network::<T>())
61                // Storage version v2 -> v3
62                .saturating_add(migrations::migrate_transfer_ownership_to_foundation::migrate_transfer_ownership_to_foundation::<T>(
63                    hex,
64                ))
65                // Storage version v3 -> v4
66                .saturating_add(migrations::migrate_delete_subnet_21::migrate_delete_subnet_21::<T>())
67                // Storage version v4 -> v5
68                .saturating_add(migrations::migrate_delete_subnet_3::migrate_delete_subnet_3::<T>())
69                // Doesn't check storage version. TODO: Remove after upgrade
70                // Storage version v5 -> v6
71                .saturating_add(migrations::migrate_total_issuance::migrate_total_issuance::<T>(false))
72                // Populate OwnedHotkeys map for coldkey swap. Doesn't update storage vesion.
73                // Storage version v6 -> v7
74                .saturating_add(migrations::migrate_populate_owned_hotkeys::migrate_populate_owned::<T>())
75                // Migrate Delegate Ids on chain
76                .saturating_add(migrations::migrate_chain_identity::migrate_set_hotkey_identities::<T>())
77                // Migrate Commit-Reval 2.0
78                .saturating_add(migrations::migrate_commit_reveal_v2::migrate_commit_reveal_2::<T>())
79                // Migrate to RAO
80                .saturating_add(migrations::migrate_rao::migrate_rao::<T>())
81				// Fix the IsNetworkMember map to be consistent with other storage maps
82				.saturating_add(migrations::migrate_fix_is_network_member::migrate_fix_is_network_member::<T>())
83				.saturating_add(migrations::migrate_subnet_volume::migrate_subnet_volume::<T>())
84                // Upgrade identities to V2
85                .saturating_add(migrations::migrate_identities_v2::migrate_identities_to_v2::<T>())
86				// Set the min burn across all subnets to a new minimum
87				.saturating_add(migrations::migrate_set_min_burn::migrate_set_min_burn::<T>())
88				// Set the min difficulty across all subnets to a new minimum
89				.saturating_add(migrations::migrate_set_min_difficulty::migrate_set_min_difficulty::<T>())
90                // Remove Stake map entries
91				.saturating_add(migrations::migrate_remove_stake_map::migrate_remove_stake_map::<T>())
92                // Remove unused maps entries
93				.saturating_add(migrations::migrate_remove_unused_maps_and_values::migrate_remove_unused_maps_and_values::<T>())
94                // Set last emission block number for all existed subnets before start call feature applied
95                .saturating_add(migrations::migrate_set_first_emission_block_number::migrate_set_first_emission_block_number::<T>())
96                // Remove all zero value entries in TotalHotkeyAlpha
97                .saturating_add(migrations::migrate_remove_zero_total_hotkey_alpha::migrate_remove_zero_total_hotkey_alpha::<T>())
98                // Wipe existing items to prevent bad decoding for new type
99                .saturating_add(migrations::migrate_upgrade_revealed_commitments::migrate_upgrade_revealed_commitments::<T>())
100                // Set subtoken enabled for all existed subnets

102                // Remove all entries in TotalHotkeyColdkeyStakesThisInterval

104                // Wipe the deprecated RateLimit storage item in the commitments pallet
105                .saturating_add(migrations::migrate_remove_commitments_rate_limit::migrate_remove_commitments_rate_limit::<T>());
106
107            weight
108                // Remove all entries in orphaned storage items
109                .saturating_add(
110                    migrations::migrate_orphaned_storage_items::migrate_orphaned_storage_items::<T>(
111                    ),
112                )
113                // Reset bonds moving average
114                .saturating_add(migrations::migrate_reset_bonds_moving_average::migrate_reset_bonds_moving_average::<T>())
115                // Reset max burn
116                .saturating_add(migrations::migrate_reset_max_burn::migrate_reset_max_burn::<T>())
117                // Migrate ColdkeySwapScheduled structure to new format
118                .saturating_add(migrations::migrate_coldkey_swap_scheduled::migrate_coldkey_swap_scheduled::<T>());
119            weight
120        }
