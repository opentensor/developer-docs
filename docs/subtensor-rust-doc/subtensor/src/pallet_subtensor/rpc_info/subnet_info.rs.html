134    pub fn get_subnets_info() -> Vec<Option<SubnetInfo<T::AccountId>>> {
135        let mut subnet_netuids = Vec::<u16>::new();
136        let mut max_netuid: u16 = 0;
137        for (netuid, added) in <NetworksAdded<T> as IterableStorageMap<u16, bool>>::iter() {
138            if added {
139                subnet_netuids.push(netuid);
140                if netuid > max_netuid {
141                    max_netuid = netuid;
142                }
143            }
144        }
145
146        let mut subnets_info = Vec::<Option<SubnetInfo<T::AccountId>>>::new();
147        for netuid_ in 0..=max_netuid {
148            if subnet_netuids.contains(&netuid_) {
149                subnets_info.push(Self::get_subnet_info(netuid_));
150            }
151        }
152
153        subnets_info
154    }
