39impl<T: Config> Pallet<T> {
40    /// Retrieves the emission history for a list of hotkeys across all subnets.
41    ///
42    /// This function iterates over all subnets and collects the last emission value
43    /// for each hotkey in the provided list. The result is a vector of vectors, where
44    /// each inner vector contains the emission values for a specific subnet.
45    ///
46    /// # Arguments
47    ///
48    /// * `hotkeys` - A vector of hotkeys (account IDs) for which the emission history is to be retrieved.
49    ///
50    /// # Returns
51    ///
52    /// * `Vec<Vec<Compact<u64>>>` - A vector of vectors containing the emission history for each hotkey across all subnets.
53    pub fn get_emissions_history(hotkeys: Vec<T::AccountId>) -> Vec<Vec<Compact<u64>>> {
54        let mut result: Vec<Vec<Compact<u64>>> = vec![];
55        for netuid in Self::get_all_subnet_netuids() {
56            let mut hotkeys_emissions: Vec<Compact<u64>> = vec![];
57            for hotkey in hotkeys.clone() {
58                let last_emission: Compact<u64> =
59                    LastHotkeyEmissionOnNetuid::<T>::get(hotkey.clone(), netuid).into();
60                hotkeys_emissions.push(last_emission);
61            }
62            result.push(hotkeys_emissions.clone());
63        }
64        result
65    }
66
67    /// Retrieves the state of a specific subnet.
68    ///
69    /// This function gathers various metrics and data points for a given subnet, identified by its `netuid`.
70    /// It collects information such as hotkeys, coldkeys, block at registration, active status, validator permits,
71    /// pruning scores, last updates, emissions, dividends, incentives, consensus, trust, rank, local stake, global stake,
72    /// stake weight, and emission history.
73    ///
74    /// # Arguments
75    ///
76    /// * `netuid` - The unique identifier of the subnet for which the state is to be retrieved.
77    ///
78    /// # Returns
79    ///
80    /// * `Option<SubnetState<T::AccountId>>` - An optional `SubnetState` struct containing the collected data for the subnet.
81    ///   Returns `None` if the subnet does not exist.
82    pub fn get_subnet_state(netuid: u16) -> Option<SubnetState<T::AccountId>> {
83        if !Self::if_subnet_exist(netuid) {
84            return None;
85        }
86        let n: u16 = Self::get_subnetwork_n(netuid);
87        let mut hotkeys: Vec<T::AccountId> = vec![];
88        let mut coldkeys: Vec<T::AccountId> = vec![];
89        let mut block_at_registration: Vec<Compact<u64>> = vec![];
90        // let mut identities: Vec<ChainIdentityOf> = vec![];
91        for uid in 0..n {
92            let hotkey = Keys::<T>::get(netuid, uid);
93            let coldkey = Owner::<T>::get(hotkey.clone());
94            hotkeys.push(hotkey);
95            coldkeys.push(coldkey);
96            block_at_registration.push(BlockAtRegistration::<T>::get(netuid, uid).into());
97            // identities.push( Identities::<T>::get( coldkey.clone() ) );
98        }
99        let active: Vec<bool> = Active::<T>::get(netuid);
100        let validator_permit: Vec<bool> = ValidatorPermit::<T>::get(netuid);
101        let pruning_score: Vec<Compact<u16>> = PruningScores::<T>::get(netuid)
102            .into_iter()
103            .map(Compact::from)
104            .collect();
105        let last_update: Vec<Compact<u64>> = LastUpdate::<T>::get(netuid)
106            .into_iter()
107            .map(Compact::from)
108            .collect();
109        let emission: Vec<Compact<u64>> = Emission::<T>::get(netuid)
110            .into_iter()
111            .map(Compact::from)
112            .collect();
113        let dividends: Vec<Compact<u16>> = Dividends::<T>::get(netuid)
114            .into_iter()
115            .map(Compact::from)
116            .collect();
117        let incentives: Vec<Compact<u16>> = Incentive::<T>::get(netuid)
118            .into_iter()
119            .map(Compact::from)
120            .collect();
121        let consensus: Vec<Compact<u16>> = Consensus::<T>::get(netuid)
122            .into_iter()
123            .map(Compact::from)
124            .collect();
125        let trust: Vec<Compact<u16>> = Trust::<T>::get(netuid)
126            .into_iter()
127            .map(Compact::from)
128            .collect();
129        let rank: Vec<Compact<u16>> = Rank::<T>::get(netuid)
130            .into_iter()
131            .map(Compact::from)
132            .collect();
133        let (total_stake_fl, alpha_stake_fl, tao_stake_fl): (
134            Vec<I64F64>,
135            Vec<I64F64>,
136            Vec<I64F64>,
137        ) = Self::get_stake_weights_for_network(netuid);
138        let alpha_stake: Vec<Compact<u64>> = alpha_stake_fl
139            .iter()
140            .map(|xi| Compact::from(fixed64_to_u64(*xi)))
141            .collect::<Vec<Compact<u64>>>();
142        let tao_stake: Vec<Compact<u64>> = tao_stake_fl
143            .iter()
144            .map(|xi| Compact::from(fixed64_to_u64(*xi)))
145            .collect::<Vec<Compact<u64>>>();
146        let total_stake: Vec<Compact<u64>> = total_stake_fl
147            .iter()
148            .map(|xi| Compact::from(fixed64_to_u64(*xi)))
149            .collect::<Vec<Compact<u64>>>();
150        let emission_history: Vec<Vec<Compact<u64>>> = Self::get_emissions_history(hotkeys.clone());
151        Some(SubnetState {
152            netuid: netuid.into(),
153            hotkeys,
154            coldkeys,
155            active,
156            validator_permit,
157            pruning_score,
158            last_update,
159            emission,
160            dividends,
161            incentives,
162            consensus,
163            trust,
164            rank,
165            block_at_registration,
166            alpha_stake,
167            tao_stake,
168            total_stake,
169            emission_history,
170        })
171    }
