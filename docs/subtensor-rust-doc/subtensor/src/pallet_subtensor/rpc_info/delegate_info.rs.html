12pub struct DelegateInfo<AccountId: TypeInfo + Encode + Decode> {
13    pub delegate_ss58: AccountId,
14    pub take: Compact<u16>,
15    pub nominators: Vec<(AccountId, Vec<(Compact<u16>, Compact<u64>)>)>, // map of nominator_ss58 to netuid and stake amount
16    pub owner_ss58: AccountId,

18    pub validator_permits: Vec<Compact<u16>>, // Vec of netuid this delegate has validator permit on

23impl<T: Config> Pallet<T> {
24    fn return_per_1000_tao(
25        take: Compact<u16>,
26        total_stake: U64F64,
27        emissions_per_day: U64F64,
28    ) -> U64F64 {
29        // Get the take as a percentage and subtract it from 1 for remainder.
30        let without_take: U64F64 = U64F64::saturating_from_num(1)
31            .saturating_sub(U64F64::saturating_from_num(take.0).safe_div(u16::MAX.into()));
32
33        if total_stake > U64F64::saturating_from_num(0) {
34            emissions_per_day
35                .saturating_mul(without_take)
36                // Divide by 1000 TAO for return per 1k
37                .safe_div(total_stake.safe_div(U64F64::saturating_from_num(1000.0 * 1e9)))
38        } else {
39            U64F64::saturating_from_num(0)
40        }
41    }

52    fn get_delegate_by_existing_account(
53        delegate: AccountIdOf<T>,
54        skip_nominators: bool,
55    ) -> DelegateInfo<T::AccountId> {
56        let mut nominators = Vec::<(T::AccountId, Vec<(Compact<u16>, Compact<u64>)>)>::new();
57        let mut nominator_map = BTreeMap::<T::AccountId, Vec<(Compact<u16>, Compact<u64>)>>::new();
58
59        if !skip_nominators {
60            let mut alpha_share_pools = vec![];
61            for netuid in Self::get_all_subnet_netuids() {
62                let alpha_share_pool = Self::get_alpha_share_pool(delegate.clone(), netuid);
63                alpha_share_pools.push(alpha_share_pool);
64            }
65
66            for ((nominator, netuid), alpha_stake) in Alpha::<T>::iter_prefix((delegate.clone(),)) {
67                if alpha_stake == 0 {
68                    continue;
69                }
70
71                if let Some(alpha_share_pool) = alpha_share_pools.get(netuid as usize) {
72                    let coldkey_stake = alpha_share_pool.get_value_from_shares(alpha_stake);
73
74                    nominator_map
75                        .entry(nominator.clone())
76                        .or_insert(Vec::new())
77                        .push((netuid.into(), coldkey_stake.into()));
78                }
79            }
80
81            for (nominator, stakes) in nominator_map {
82                nominators.push((nominator, stakes));
83            }
84        }
85
86        let registrations = Self::get_registered_networks_for_hotkey(&delegate.clone());
87        let mut validator_permits = Vec::<Compact<u16>>::new();
88        let mut emissions_per_day: U64F64 = U64F64::saturating_from_num(0);
89
90        for netuid in registrations.iter() {
91            if let Ok(uid) = Self::get_uid_for_net_and_hotkey(*netuid, &delegate.clone()) {
92                let validator_permit = Self::get_validator_permit_for_uid(*netuid, uid);
93                if validator_permit {
94                    validator_permits.push((*netuid).into());
95                }
96
97                let emission: U64F64 = Self::get_emission_for_uid(*netuid, uid).into();
98                let tempo: U64F64 = Self::get_tempo(*netuid).into();
99                if tempo > U64F64::saturating_from_num(0) {
100                    let epochs_per_day: U64F64 = U64F64::saturating_from_num(7200).safe_div(tempo);
101                    emissions_per_day =
102                        emissions_per_day.saturating_add(emission.saturating_mul(epochs_per_day));
103                }
104            }
105        }
106
107        let owner = Self::get_owning_coldkey_for_hotkey(&delegate.clone());
108        let take: Compact<u16> = <Delegates<T>>::get(delegate.clone()).into();
109
110        let total_stake: U64F64 =
111            Self::get_stake_for_hotkey_on_subnet(&delegate.clone(), Self::get_root_netuid()).into();
112
113        let return_per_1000: U64F64 =
114            Self::return_per_1000_tao(take, total_stake, emissions_per_day);
115
116        DelegateInfo {
117            delegate_ss58: delegate.clone(),
118            take,
119            nominators,
120            owner_ss58: owner.clone(),
121            registrations: registrations.iter().map(|x| x.into()).collect(),
122            validator_permits,
123            return_per_1000: return_per_1000.saturating_to_num::<u64>().into(),
124            total_daily_return: emissions_per_day.saturating_to_num::<u64>().into(),
125        }
126    }

128    pub fn get_delegate(delegate: T::AccountId) -> Option<DelegateInfo<T::AccountId>> {
129        // Check delegate exists
130        if !<Delegates<T>>::contains_key(delegate.clone()) {
131            return None;
132        }
133
134        let delegate_info = Self::get_delegate_by_existing_account(delegate.clone(), false);
135        Some(delegate_info)
136    }

138    /// get all delegates info from storage
139    ///
140    pub fn get_delegates() -> Vec<DelegateInfo<T::AccountId>> {
141        let mut delegates = Vec::<DelegateInfo<T::AccountId>>::new();
142        for delegate in <Delegates<T> as IterableStorageMap<T::AccountId, u16>>::iter_keys() {
143            let delegate_info = Self::get_delegate_by_existing_account(delegate.clone(), false);
144            delegates.push(delegate_info);
145        }
146
147        delegates
148    }

150    /// get all delegate info and staked token amount for a given delegatee account
151    ///
152    pub fn get_delegated(
153        delegatee: T::AccountId,
154    ) -> Vec<(DelegateInfo<T::AccountId>, (Compact<u16>, Compact<u64>))> {
155        let mut delegates: Vec<(DelegateInfo<T::AccountId>, (Compact<u16>, Compact<u64>))> =
156            Vec::new();
157        for delegate in <Delegates<T> as IterableStorageMap<T::AccountId, u16>>::iter_keys() {
158            // Staked to this delegate, so add to list
159            for (netuid, _) in Alpha::<T>::iter_prefix((delegate.clone(), delegatee.clone())) {
160                let delegate_info = Self::get_delegate_by_existing_account(delegate.clone(), true);
161                delegates.push((
162                    delegate_info,
163                    (
164                        netuid.into(),
165                        Self::get_stake_for_hotkey_and_coldkey_on_subnet(
166                            &delegate, &delegatee, netuid,
167                        )
168                        .into(),
169                    ),
170                ));
171            }
172        }
173
174        delegates
175    }
