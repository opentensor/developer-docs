17        // ======================
18        // == Active & updated ==
19        // ======================
20
21        // Get current block.
22        let current_block: u64 = Self::get_current_block_as_u64();
23        log::trace!("current_block: {:?}", current_block);
24
25        // Get tempo.
26        let tempo: u64 = Self::get_tempo(netuid).into();
27        log::trace!("tempo: {:?}", tempo);
28
29        // Get activity cutoff.
30        let activity_cutoff: u64 = Self::get_activity_cutoff(netuid) as u64;
31        log::trace!("activity_cutoff: {:?}", activity_cutoff);
32
33        // Last update vector.
34        let last_update: Vec<u64> = Self::get_last_update(netuid);
35        log::trace!("Last update: {:?}", &last_update);
36
37        // Inactive mask.
38        let inactive: Vec<bool> = last_update
39            .iter()
40            .map(|updated| updated.saturating_add(activity_cutoff) < current_block)
41            .collect();
42        log::trace!("Inactive: {:?}", inactive.clone());
43
44        // Logical negation of inactive.
45        let active: Vec<bool> = inactive.iter().map(|&b| !b).collect();
46
47        // Block at registration vector (block when each neuron was most recently registered).
48        let block_at_registration: Vec<u64> = Self::get_block_at_registration(netuid);
49        log::trace!("Block at registration: {:?}", &block_at_registration);
50
51        // Outdated matrix, outdated_ij=True if i has last updated (weights) after j has last registered.
52        let outdated: Vec<Vec<bool>> = last_update
53            .iter()
54            .map(|updated| {
55                block_at_registration
56                    .iter()
57                    .map(|registered| updated <= registered)
58                    .collect()
59            })
60            .collect();
61        log::trace!("Outdated: {:?}", &outdated);
62
63        // Recently registered matrix, recently_ij=True if last_tempo was *before* j was last registered.
64        // Mask if: the last tempo block happened *before* the registration block
65        // ==> last_tempo <= registered
66        let last_tempo: u64 = current_block.saturating_sub(tempo);
67        let recently_registered: Vec<bool> = block_at_registration
68            .iter()
69            .map(|registered| last_tempo <= *registered)
70            .collect();
71        log::trace!("Recently registered: {:?}", &recently_registered);
72
73        // ===========
74        // == Stake ==
75        // ===========
76
77        let hotkeys: Vec<(u16, T::AccountId)> =
78            <Keys<T> as IterableStorageDoubleMap<u16, u16, T::AccountId>>::iter_prefix(netuid)
79                .collect();
80        log::trace!("hotkeys: {:?}", &hotkeys);
81
82        // Access network stake as normalized vector.
83        let (total_stake, _alpha_stake, _tao_stake): (Vec<I64F64>, Vec<I64F64>, Vec<I64F64>) =
84            Self::get_stake_weights_for_network(netuid);
85
86        // Get the minimum stake required.
87        let min_stake = Self::get_stake_threshold();
88
89        // Set stake of validators that doesn't meet the staking threshold to 0 as filter.
90        let mut filtered_stake: Vec<I64F64> = total_stake
91            .iter()
92            .map(|&s| {
93                if fixed64_to_u64(s) < min_stake {
94                    return I64F64::from(0);
95                }
96                s
97            })
98            .collect();
99        log::debug!("Filtered stake: {:?}", &filtered_stake);
100
101        inplace_normalize_64(&mut filtered_stake);
102        let stake: Vec<I32F32> = vec_fixed64_to_fixed32(filtered_stake);
103        log::trace!("S: {:?}", &stake);
104
105        // =======================
106        // == Validator permits ==
107        // =======================
108
109        // Get validator permits.
110        let validator_permits: Vec<bool> = Self::get_validator_permit(netuid);
111        log::trace!("validator_permits: {:?}", validator_permits);
112
113        // Logical negation of validator_permits.
114        let validator_forbids: Vec<bool> = validator_permits.iter().map(|&b| !b).collect();
115
116        // Get max allowed validators.
117        let max_allowed_validators: u16 = Self::get_max_allowed_validators(netuid);
118        log::trace!("max_allowed_validators: {:?}", max_allowed_validators);
119
120        // Get new validator permits.
121        let new_validator_permits: Vec<bool> =
122            is_topk_nonzero(&stake, max_allowed_validators as usize);
123        log::trace!("new_validator_permits: {:?}", new_validator_permits);
124
125        // ==================
126        // == Active Stake ==
127        // ==================
128
129        let mut active_stake: Vec<I32F32> = stake.clone();
130
131        // Remove inactive stake.
132        inplace_mask_vector(&inactive, &mut active_stake);
133
134        // Remove non-validator stake.
135        inplace_mask_vector(&validator_forbids, &mut active_stake);
136
137        // Normalize active stake.
138        inplace_normalize(&mut active_stake);
139        log::trace!("S: {:?}", &active_stake);
140
141        // =============
142        // == Weights ==
143        // =============
144
145        // Get owner uid.
146        let owner_uid: Option<u16> = Self::get_owner_uid(netuid);
147
148        // Access network weights row unnormalized.
149        let mut weights: Vec<Vec<I32F32>> = Self::get_weights(netuid);
150        log::trace!("W: {:?}", &weights);
151
152        // Mask weights that are not from permitted validators.
153        inplace_mask_rows(&validator_forbids, &mut weights);
154        log::trace!("W (permit): {:?}", &weights);
155
156        // Remove self-weight by masking diagonal; keep owner_uid self-weight.
157        if let Some(owner_uid) = owner_uid {
158            inplace_mask_diag_except_index(&mut weights, owner_uid);
159        } else {
160            inplace_mask_diag(&mut weights);
161        }
162
163        inplace_mask_diag(&mut weights);
164        log::trace!("W (permit+diag): {:?}", &weights);
165
166        // Mask outdated weights: remove weights referring to deregistered neurons.
167        inplace_mask_matrix(&outdated, &mut weights);
168        log::trace!("W (permit+diag+outdate): {:?}", &weights);
169
170        // Normalize remaining weights.
171        inplace_row_normalize(&mut weights);
172        log::trace!("W (mask+norm): {:?}", &weights);
173
174        // ================================
175        // == Consensus, Validator Trust ==
176        // ================================
177
178        // Compute preranks: r_j = SUM(i) w_ij * s_i
179        let preranks: Vec<I32F32> = matmul(&weights, &active_stake);
180
181        // Consensus majority ratio, e.g. 51%.
182        let kappa: I32F32 = Self::get_float_kappa(netuid);
183        // Calculate consensus as stake-weighted median of weights.
184        let consensus: Vec<I32F32> = weighted_median_col(&active_stake, &weights, kappa);
185        // Clip weights at majority consensus.
186        let mut clipped_weights: Vec<Vec<I32F32>> = weights.clone();
187        inplace_col_clip(&mut clipped_weights, &consensus);
188        // Calculate validator trust as sum of clipped weights set by validator.
189        let validator_trust: Vec<I32F32> = row_sum(&clipped_weights);
190
191        // ====================================
192        // == Ranks, Server Trust, Incentive ==
193        // ====================================
194
195        // Compute ranks: r_j = SUM(i) w_ij * s_i
196        let mut ranks: Vec<I32F32> = matmul(&clipped_weights, &active_stake);
197
198        // Compute server trust: ratio of rank after vs. rank before.
199        let trust: Vec<I32F32> = vecdiv(&ranks, &preranks);
200
201        inplace_normalize(&mut ranks);
202        let incentive: Vec<I32F32> = ranks.clone();
203        log::trace!("I: {:?}", &incentive);
204
205        // =========================
206        // == Bonds and Dividends ==
207        // =========================
208
209        // Get validator bonds penalty in [0, 1].
210        let bonds_penalty: I32F32 = Self::get_float_bonds_penalty(netuid);
211        // Calculate weights for bonds, apply bonds penalty to weights.
212        // bonds_penalty = 0: weights_for_bonds = weights.clone()
213        // bonds_penalty = 1: weights_for_bonds = clipped_weights.clone()
214        let weights_for_bonds: Vec<Vec<I32F32>> =
215            interpolate(&weights, &clipped_weights, bonds_penalty);
216
217        let mut dividends: Vec<I32F32>;
218        let mut ema_bonds: Vec<Vec<I32F32>>;
219        if Yuma3On::<T>::get(netuid) {
220            // Access network bonds.
221            let mut bonds: Vec<Vec<I32F32>> = Self::get_bonds_fixed_proportion(netuid);
222            inplace_mask_cols(&recently_registered, &mut bonds); // mask outdated bonds
223            log::trace!("B: {:?}", &bonds);
224
225            // Compute the Exponential Moving Average (EMA) of bonds.
226            ema_bonds = Self::compute_bonds(netuid, &weights_for_bonds, &bonds, &consensus);
227            log::trace!("emaB: {:?}", &ema_bonds);
228
229            // Normalize EMA bonds.
230            let mut ema_bonds_norm = ema_bonds.clone();
231            inplace_col_normalize(&mut ema_bonds_norm);
232            log::trace!("emaB norm: {:?}", &ema_bonds_norm);
233
234            // # === Dividend Calculation===
235            let total_bonds_per_validator: Vec<I32F32> =
236                row_sum(&mat_vec_mul(&ema_bonds_norm, &incentive));
237            log::trace!(
238                "total_bonds_per_validator: {:?}",
239                &total_bonds_per_validator
240            );
241
242            dividends = vec_mul(&total_bonds_per_validator, &active_stake);
243            inplace_normalize(&mut dividends);
244            log::trace!("D: {:?}", &dividends);
245        } else {
246            // original Yuma - liquid alpha disabled
247            // Access network bonds.
248            let mut bonds: Vec<Vec<I32F32>> = Self::get_bonds(netuid);
249            // Remove bonds referring to neurons that have registered since last tempo.
250            inplace_mask_cols(&recently_registered, &mut bonds); // mask recently registered bonds
251            inplace_col_normalize(&mut bonds); // sum_i b_ij = 1
252            log::trace!("B: {:?}", &bonds);
253
254            // Compute bonds delta column normalized.
255            let mut bonds_delta: Vec<Vec<I32F32>> = row_hadamard(&weights_for_bonds, &active_stake); // ΔB = W◦S
256            inplace_col_normalize(&mut bonds_delta); // sum_i b_ij = 1
257            log::trace!("ΔB: {:?}", &bonds_delta);
258
259            // Compute the Exponential Moving Average (EMA) of bonds.
260            ema_bonds = Self::compute_ema_bonds_normal(&bonds_delta, &bonds, netuid);
261            inplace_col_normalize(&mut ema_bonds); // sum_i b_ij = 1
262            log::trace!("emaB: {:?}", &ema_bonds);
263
264            // Compute dividends: d_i = SUM(j) b_ij * inc_j
265            dividends = matmul_transpose(&ema_bonds, &incentive);
266            inplace_normalize(&mut dividends);
267            log::trace!("Dividends: {:?}", &dividends);
268
269            // Column max-upscale EMA bonds for storage: max_i w_ij = 1.
270            inplace_col_max_upscale(&mut ema_bonds);
271        }
272
273        // =================================
274        // == Emission and Pruning scores ==
275        // =================================
276
277        // Compute emission scores.
278
279        // Compute normalized emission scores. range: I32F32(0, 1)
280        // Compute normalized emission scores. range: I32F32(0, 1)
281        let combined_emission: Vec<I32F32> = incentive
282            .iter()
283            .zip(dividends.clone())
284            .map(|(ii, di)| ii.saturating_add(di))
285            .collect();
286        let emission_sum: I32F32 = combined_emission.iter().sum();
287
288        let mut normalized_server_emission: Vec<I32F32> = incentive.clone(); // Servers get incentive.
289        let mut normalized_validator_emission: Vec<I32F32> = dividends.clone(); // Validators get dividends.
290        let mut normalized_combined_emission: Vec<I32F32> = combined_emission.clone();
291        // Normalize on the sum of incentive + dividends.
292        inplace_normalize_using_sum(&mut normalized_server_emission, emission_sum);
293        inplace_normalize_using_sum(&mut normalized_validator_emission, emission_sum);
294        inplace_normalize(&mut normalized_combined_emission);
295
296        // If emission is zero, replace emission with normalized stake.
297        if emission_sum == I32F32::from(0) {
298            // no weights set | outdated weights | self_weights
299            if is_zero(&active_stake) {
300                // no active stake
301                normalized_validator_emission.clone_from(&stake); // do not mask inactive, assumes stake is normalized
302                normalized_combined_emission.clone_from(&stake);
303            } else {
304                normalized_validator_emission.clone_from(&active_stake); // emission proportional to inactive-masked normalized stake
305                normalized_combined_emission.clone_from(&active_stake);
306            }
307        }
308
309        // Compute rao based emission scores. range: I96F32(0, rao_emission)
310        let float_rao_emission: I96F32 = I96F32::saturating_from_num(rao_emission);
311
312        let server_emission: Vec<I96F32> = normalized_server_emission
313            .iter()
314            .map(|se: &I32F32| I96F32::saturating_from_num(*se).saturating_mul(float_rao_emission))
315            .collect();
316        let server_emission: Vec<u64> = server_emission
317            .iter()
318            .map(|e: &I96F32| e.saturating_to_num::<u64>())
319            .collect();
320
321        let validator_emission: Vec<I96F32> = normalized_validator_emission
322            .iter()
323            .map(|ve: &I32F32| I96F32::saturating_from_num(*ve).saturating_mul(float_rao_emission))
324            .collect();
325        let validator_emission: Vec<u64> = validator_emission
326            .iter()
327            .map(|e: &I96F32| e.saturating_to_num::<u64>())
328            .collect();
329
330        // Used only to track combined emission in the storage.
331        let combined_emission: Vec<I96F32> = normalized_combined_emission
332            .iter()
333            .map(|ce: &I32F32| I96F32::saturating_from_num(*ce).saturating_mul(float_rao_emission))
334            .collect();
335        let combined_emission: Vec<u64> = combined_emission
336            .iter()
337            .map(|e: &I96F32| e.saturating_to_num::<u64>())
338            .collect();
339
340        log::trace!("nSE: {:?}", &normalized_server_emission);
341        log::trace!("SE: {:?}", &server_emission);
342        log::trace!("nVE: {:?}", &normalized_validator_emission);
343        log::trace!("VE: {:?}", &validator_emission);
344        log::trace!("nCE: {:?}", &normalized_combined_emission);
345        log::trace!("CE: {:?}", &combined_emission);
346
347        // Set pruning scores using combined emission scores.
348        let pruning_scores: Vec<I32F32> = normalized_combined_emission.clone();
349        log::trace!("P: {:?}", &pruning_scores);
350
351        // ===================
352        // == Value storage ==
353        // ===================
354        let cloned_emission: Vec<u64> = combined_emission.clone();
355        let cloned_stake_weight: Vec<u16> = stake
356            .iter()
357            .map(|xi| fixed_proportion_to_u16(*xi))
358            .collect::<Vec<u16>>();
359        let cloned_ranks: Vec<u16> = ranks
360            .iter()
361            .map(|xi| fixed_proportion_to_u16(*xi))
362            .collect::<Vec<u16>>();
363        let cloned_trust: Vec<u16> = trust
364            .iter()
365            .map(|xi| fixed_proportion_to_u16(*xi))
366            .collect::<Vec<u16>>();
367        let cloned_consensus: Vec<u16> = consensus
368            .iter()
369            .map(|xi| fixed_proportion_to_u16(*xi))
370            .collect::<Vec<u16>>();
371        let cloned_incentive: Vec<u16> = incentive
372            .iter()
373            .map(|xi| fixed_proportion_to_u16(*xi))
374            .collect::<Vec<u16>>();
375        let cloned_dividends: Vec<u16> = dividends
376            .iter()
377            .map(|xi| fixed_proportion_to_u16(*xi))
378            .collect::<Vec<u16>>();
379        let cloned_pruning_scores: Vec<u16> = vec_max_upscale_to_u16(&pruning_scores);
380        let cloned_validator_trust: Vec<u16> = validator_trust
381            .iter()
382            .map(|xi| fixed_proportion_to_u16(*xi))
383            .collect::<Vec<u16>>();
384        StakeWeight::<T>::insert(netuid, cloned_stake_weight.clone());
385        Active::<T>::insert(netuid, active.clone());
386        Emission::<T>::insert(netuid, cloned_emission);
387        Rank::<T>::insert(netuid, cloned_ranks);
388        Trust::<T>::insert(netuid, cloned_trust);
389        Consensus::<T>::insert(netuid, cloned_consensus);
390        Incentive::<T>::insert(netuid, cloned_incentive);
391        Dividends::<T>::insert(netuid, cloned_dividends);
392        PruningScores::<T>::insert(netuid, cloned_pruning_scores);
393        ValidatorTrust::<T>::insert(netuid, cloned_validator_trust);
394        ValidatorPermit::<T>::insert(netuid, new_validator_permits.clone());
395
396        new_validator_permits
397            .iter()
398            .zip(validator_permits)
399            .zip(ema_bonds)
400            .enumerate()
401            .for_each(|(i, ((new_permit, validator_permit), ema_bond))| {
402                // Set bonds only if uid retains validator permit, otherwise clear bonds.
403                if *new_permit {
404                    let new_bonds_row: Vec<(u16, u16)> = (0..n)
405                        .zip(vec_fixed_proportions_to_u16(ema_bond.clone()))
406                        .collect();
407                    Bonds::<T>::insert(netuid, i as u16, new_bonds_row);
408                } else if validator_permit {
409                    // Only overwrite the intersection.
410                    let new_empty_bonds_row: Vec<(u16, u16)> = vec![];
411                    Bonds::<T>::insert(netuid, i as u16, new_empty_bonds_row);
412                }
413            });
414
415        hotkeys
416            .into_iter()
417            .map(|(uid_i, hotkey)| {
418                (
419                    hotkey,
420                    server_emission[uid_i as usize],
421                    validator_emission[uid_i as usize],
422                )
423            })
424            .collect()
425    }

558        // Mask weights that are not from permitted validators.
559        weights = mask_rows_sparse(&validator_forbids, &weights);
560        log::trace!("Weights (permit): {:?}", &weights);
561
562        // Remove self-weight by masking diagonal; keep owner_uid self-weight.
563        if let Some(owner_uid) = owner_uid {
564            weights = mask_diag_sparse_except_index(&weights, owner_uid);
565        } else {
566            weights = mask_diag_sparse(&weights);
567        }
568        log::trace!("Weights (permit+diag): {:?}", &weights);
569
570        // Remove weights referring to deregistered neurons.
571        weights = vec_mask_sparse_matrix(
572            &weights,
573            &last_update,
574            &block_at_registration,
575            &|updated, registered| updated <= registered,
576        );
577        log::trace!("Weights (permit+diag+outdate): {:?}", &weights);
578
579        // Normalize remaining weights.
580        inplace_row_normalize_sparse(&mut weights);
581        log::trace!("Weights (mask+norm): {:?}", &weights);
582
583        // ================================
584        // == Consensus, Validator Trust ==
585        // ================================
586
587        // Compute preranks: r_j = SUM(i) w_ij * s_i
588        let preranks: Vec<I32F32> = matmul_sparse(&weights, &active_stake, n);
589        log::trace!("Ranks (before): {:?}", &preranks);
590
591        // Consensus majority ratio, e.g. 51%.
592        let kappa: I32F32 = Self::get_float_kappa(netuid);
593        // Calculate consensus as stake-weighted median of weights.
594        let consensus: Vec<I32F32> = weighted_median_col_sparse(&active_stake, &weights, n, kappa);
595        log::trace!("Consensus: {:?}", &consensus);
596
597        // Clip weights at majority consensus.
598        let clipped_weights: Vec<Vec<(u16, I32F32)>> = col_clip_sparse(&weights, &consensus);
599        log::trace!("Clipped Weights: {:?}", &clipped_weights);
600
601        // Calculate validator trust as sum of clipped weights set by validator.
602        let validator_trust: Vec<I32F32> = row_sum_sparse(&clipped_weights);
603        log::trace!("Validator Trust: {:?}", &validator_trust);
604
605        // =============================
606        // == Ranks, Trust, Incentive ==
607        // =============================
608
609        // Compute ranks: r_j = SUM(i) w_ij * s_i.
610        let mut ranks: Vec<I32F32> = matmul_sparse(&clipped_weights, &active_stake, n);
611        log::trace!("Ranks (after): {:?}", &ranks);
612
613        // Compute server trust: ratio of rank after vs. rank before.
614        let trust: Vec<I32F32> = vecdiv(&ranks, &preranks); // range: I32F32(0, 1)
615        log::trace!("Trust: {:?}", &trust);
616
617        inplace_normalize(&mut ranks); // range: I32F32(0, 1)
618        let incentive: Vec<I32F32> = ranks.clone();
619        log::trace!("Incentive (=Rank): {:?}", &incentive);
620
621        // =========================
622        // == Bonds and Dividends ==
623        // =========================
624
625        // Get validator bonds penalty in [0, 1].
626        let bonds_penalty: I32F32 = Self::get_float_bonds_penalty(netuid);
627        // Calculate weights for bonds, apply bonds penalty to weights.
628        // bonds_penalty = 0: weights_for_bonds = weights.clone()
629        // bonds_penalty = 1: weights_for_bonds = clipped_weights.clone()
630        let weights_for_bonds: Vec<Vec<(u16, I32F32)>> =
631            interpolate_sparse(&weights, &clipped_weights, n, bonds_penalty);
632
633        let mut dividends: Vec<I32F32>;
634        let mut ema_bonds: Vec<Vec<(u16, I32F32)>>;
635        if Yuma3On::<T>::get(netuid) {
636            // Access network bonds.
637            let mut bonds = Self::get_bonds_sparse_fixed_proportion(netuid);
638            log::trace!("Bonds: {:?}", &bonds);
639
640            // Remove bonds referring to neurons that have registered since last tempo.
641            // Mask if: the last tempo block happened *before* the registration block
642            // ==> last_tempo <= registered
643            let last_tempo: u64 = current_block.saturating_sub(tempo);
644            bonds = scalar_vec_mask_sparse_matrix(
645                &bonds,
646                last_tempo,
647                &block_at_registration,
648                &|last_tempo, registered| last_tempo <= registered,
649            );
650            log::trace!("Bonds: (mask) {:?}", &bonds);
651
652            // Compute the Exponential Moving Average (EMA) of bonds.
653            log::trace!("weights_for_bonds: {:?}", &weights_for_bonds);
654            ema_bonds = Self::compute_bonds_sparse(netuid, &weights_for_bonds, &bonds, &consensus);
655            log::trace!("emaB: {:?}", &ema_bonds);
656
657            // Normalize EMA bonds.
658            let mut ema_bonds_norm = ema_bonds.clone();
659            inplace_col_normalize_sparse(&mut ema_bonds_norm, n); // sum_i b_ij = 1
660            log::trace!("emaB norm: {:?}", &ema_bonds_norm);
661
662            // # === Dividend Calculation===
663            let total_bonds_per_validator: Vec<I32F32> =
664                row_sum_sparse(&mat_vec_mul_sparse(&ema_bonds_norm, &incentive));
665            log::trace!(
666                "total_bonds_per_validator: {:?}",
667                &total_bonds_per_validator
668            );
669
670            dividends = vec_mul(&total_bonds_per_validator, &active_stake);
671            inplace_normalize(&mut dividends);
672            log::trace!("Dividends: {:?}", &dividends);
673        } else {
674            // original Yuma - liquid alpha disabled
675            // Access network bonds.
676            let mut bonds: Vec<Vec<(u16, I32F32)>> = Self::get_bonds_sparse(netuid);
677            log::trace!("B: {:?}", &bonds);
678
679            // Remove bonds referring to neurons that have registered since last tempo.
680            // Mask if: the last tempo block happened *before* the registration block
681            // ==> last_tempo <= registered
682            let last_tempo: u64 = current_block.saturating_sub(tempo);
683            bonds = scalar_vec_mask_sparse_matrix(
684                &bonds,
685                last_tempo,
686                &block_at_registration,
687                &|last_tempo, registered| last_tempo <= registered,
688            );
689            log::trace!("B (outdatedmask): {:?}", &bonds);
690
691            // Normalize remaining bonds: sum_i b_ij = 1.
692            inplace_col_normalize_sparse(&mut bonds, n);
693            log::trace!("B (mask+norm): {:?}", &bonds);
694
695            // Compute bonds delta column normalized.
696            let mut bonds_delta: Vec<Vec<(u16, I32F32)>> =
697                row_hadamard_sparse(&weights_for_bonds, &active_stake); // ΔB = W◦S (outdated W masked)
698            log::trace!("ΔB: {:?}", &bonds_delta);
699
700            // Normalize bonds delta.
701            inplace_col_normalize_sparse(&mut bonds_delta, n); // sum_i b_ij = 1
702            log::trace!("ΔB (norm): {:?}", &bonds_delta);
703
704            // Compute the Exponential Moving Average (EMA) of bonds.
705            ema_bonds = Self::compute_ema_bonds_normal_sparse(&bonds_delta, &bonds, netuid);
706            // Normalize EMA bonds.
707            inplace_col_normalize_sparse(&mut ema_bonds, n); // sum_i b_ij = 1
708            log::trace!("Exponential Moving Average Bonds: {:?}", &ema_bonds);
709
710            // Compute dividends: d_i = SUM(j) b_ij * inc_j.
711            // range: I32F32(0, 1)
712            dividends = matmul_transpose_sparse(&ema_bonds, &incentive);
713            inplace_normalize(&mut dividends);
714            log::trace!("Dividends: {:?}", &dividends);
715
716            // Column max-upscale EMA bonds for storage: max_i w_ij = 1.
717            inplace_col_max_upscale_sparse(&mut ema_bonds, n);
718        }
719
720        // =================================
721        // == Emission and Pruning scores ==
722        // =================================
723
724        // Compute normalized emission scores. range: I32F32(0, 1)
725        let combined_emission: Vec<I32F32> = incentive
726            .iter()
727            .zip(dividends.clone())
728            .map(|(ii, di)| ii.saturating_add(di))
729            .collect();
730        let emission_sum: I32F32 = combined_emission.iter().sum();
731
732        let mut normalized_server_emission: Vec<I32F32> = incentive.clone(); // Servers get incentive.
733        let mut normalized_validator_emission: Vec<I32F32> = dividends.clone(); // Validators get dividends.
734        let mut normalized_combined_emission: Vec<I32F32> = combined_emission.clone();
735        // Normalize on the sum of incentive + dividends.
736        inplace_normalize_using_sum(&mut normalized_server_emission, emission_sum);
737        inplace_normalize_using_sum(&mut normalized_validator_emission, emission_sum);
738        inplace_normalize(&mut normalized_combined_emission);
739
740        // If emission is zero, replace emission with normalized stake.
741        if emission_sum == I32F32::from(0) {
742            // no weights set | outdated weights | self_weights
743            if is_zero(&active_stake) {
744                // no active stake
745                normalized_validator_emission.clone_from(&stake); // do not mask inactive, assumes stake is normalized
746                normalized_combined_emission.clone_from(&stake);
747            } else {
748                normalized_validator_emission.clone_from(&active_stake); // emission proportional to inactive-masked normalized stake
749                normalized_combined_emission.clone_from(&active_stake);
750            }
751        }
752
753        // Compute rao based emission scores. range: I96F32(0, rao_emission)
754        let float_rao_emission: I96F32 = I96F32::saturating_from_num(rao_emission);
755
756        let server_emission: Vec<I96F32> = normalized_server_emission
757            .iter()
758            .map(|se: &I32F32| I96F32::saturating_from_num(*se).saturating_mul(float_rao_emission))
759            .collect();
760        let server_emission: Vec<u64> = server_emission
761            .iter()
762            .map(|e: &I96F32| e.saturating_to_num::<u64>())
763            .collect();
764
765        let validator_emission: Vec<I96F32> = normalized_validator_emission
766            .iter()
767            .map(|ve: &I32F32| I96F32::saturating_from_num(*ve).saturating_mul(float_rao_emission))
768            .collect();
769        let validator_emission: Vec<u64> = validator_emission
770            .iter()
771            .map(|e: &I96F32| e.saturating_to_num::<u64>())
772            .collect();
773
774        // Only used to track emission in storage.
775        let combined_emission: Vec<I96F32> = normalized_combined_emission
776            .iter()
777            .map(|ce: &I32F32| I96F32::saturating_from_num(*ce).saturating_mul(float_rao_emission))
778            .collect();
779        let combined_emission: Vec<u64> = combined_emission
780            .iter()
781            .map(|e: &I96F32| e.saturating_to_num::<u64>())
782            .collect();
783
784        log::trace!(
785            "Normalized Server Emission: {:?}",
786            &normalized_server_emission
787        );
788        log::trace!("Server Emission: {:?}", &server_emission);
789        log::trace!(
790            "Normalized Validator Emission: {:?}",
791            &normalized_validator_emission
792        );
793        log::trace!("Validator Emission: {:?}", &validator_emission);
794        log::trace!(
795            "Normalized Combined Emission: {:?}",
796            &normalized_combined_emission
797        );
798        log::trace!("Combined Emission: {:?}", &combined_emission);
799
800        // Set pruning scores using combined emission scores.
801        let pruning_scores: Vec<I32F32> = normalized_combined_emission.clone();
802        log::trace!("Pruning Scores: {:?}", &pruning_scores);
803
804        // ===================
805        // == Value storage ==
806        // ===================
807        let cloned_stake_weight: Vec<u16> = stake
808            .iter()
809            .map(|xi| fixed_proportion_to_u16(*xi))
810            .collect::<Vec<u16>>();
811        let cloned_emission: Vec<u64> = combined_emission.clone();
812        let cloned_ranks: Vec<u16> = ranks
813            .iter()
814            .map(|xi| fixed_proportion_to_u16(*xi))
815            .collect::<Vec<u16>>();
816        let cloned_trust: Vec<u16> = trust
817            .iter()
818            .map(|xi| fixed_proportion_to_u16(*xi))
819            .collect::<Vec<u16>>();
820        let cloned_consensus: Vec<u16> = consensus
821            .iter()
822            .map(|xi| fixed_proportion_to_u16(*xi))
823            .collect::<Vec<u16>>();
824        let cloned_incentive: Vec<u16> = incentive
825            .iter()
826            .map(|xi| fixed_proportion_to_u16(*xi))
827            .collect::<Vec<u16>>();
828        let cloned_dividends: Vec<u16> = dividends
829            .iter()
830            .map(|xi| fixed_proportion_to_u16(*xi))
831            .collect::<Vec<u16>>();
832        let cloned_pruning_scores: Vec<u16> = vec_max_upscale_to_u16(&pruning_scores);
833        let cloned_validator_trust: Vec<u16> = validator_trust
834            .iter()
835            .map(|xi| fixed_proportion_to_u16(*xi))
836            .collect::<Vec<u16>>();
837        StakeWeight::<T>::insert(netuid, cloned_stake_weight.clone());
838        Active::<T>::insert(netuid, active.clone());
839        Emission::<T>::insert(netuid, cloned_emission);
840        Rank::<T>::insert(netuid, cloned_ranks);
841        Trust::<T>::insert(netuid, cloned_trust);
842        Consensus::<T>::insert(netuid, cloned_consensus);
843        Incentive::<T>::insert(netuid, cloned_incentive);
844        Dividends::<T>::insert(netuid, cloned_dividends);
845        PruningScores::<T>::insert(netuid, cloned_pruning_scores);
846        ValidatorTrust::<T>::insert(netuid, cloned_validator_trust);
847        ValidatorPermit::<T>::insert(netuid, new_validator_permits.clone());
848
849        new_validator_permits
850            .iter()
851            .zip(validator_permits)
852            .zip(ema_bonds)
853            .enumerate()
854            .for_each(|(i, ((new_permit, validator_permit), ema_bond))| {
855                // Set bonds only if uid retains validator permit, otherwise clear bonds.
856                if *new_permit {
857                    let new_bonds_row: Vec<(u16, u16)> = ema_bond
858                        .iter()
859                        .map(|(j, value)| (*j, fixed_proportion_to_u16(*value)))
860                        .collect();
861                    Bonds::<T>::insert(netuid, i as u16, new_bonds_row);
862                } else if validator_permit {
863                    // Only overwrite the intersection.
864                    let new_empty_bonds_row: Vec<(u16, u16)> = vec![];
865                    Bonds::<T>::insert(netuid, i as u16, new_empty_bonds_row);
866                }
867            });
868
869        // Emission tuples ( hotkeys, server_emission, validator_emission )
870        hotkeys
871            .into_iter()
872            .map(|(uid_i, hotkey)| {
873                (
874                    hotkey,
875                    server_emission[uid_i as usize],
876                    validator_emission[uid_i as usize],
877                )
878            })
879            .collect()
880    }

908    /// Output unnormalized sparse weights, input weights are assumed to be row max-upscaled in u16.

910        let n: usize = Self::get_subnetwork_n(netuid) as usize;
911        let mut weights: Vec<Vec<(u16, I32F32)>> = vec![vec![]; n];
912        for (uid_i, weights_i) in
913            <Weights<T> as IterableStorageDoubleMap<u16, u16, Vec<(u16, u16)>>>::iter_prefix(netuid)
914                .filter(|(uid_i, _)| *uid_i < n as u16)
915        {
916            for (uid_j, weight_ij) in weights_i.iter().filter(|(uid_j, _)| *uid_j < n as u16) {
917                weights
918                    .get_mut(uid_i as usize)
919                    .expect("uid_i is filtered to be less than n; qed")
920                    .push((*uid_j, I32F32::saturating_from_num(*weight_ij)));
921            }
922        }
923        weights
924    }

1007    /// Compute the Exponential Moving Average (EMA) of bonds using a normal alpha value for a sparse matrix.
1008    ///
1009    /// # Args:
1010    /// * `bonds_delta` - A vector of bond deltas.
1011    /// * `bonds` - A vector of bonds.
1012    /// * `netuid` - The network ID.
1013    ///
1014    /// # Returns:
1015    /// A vector of EMA bonds.
1016    pub fn compute_ema_bonds_normal_sparse(
1017        bonds_delta: &[Vec<(u16, I32F32)>],
1018        bonds: &[Vec<(u16, I32F32)>],
1019        netuid: u16,
1020    ) -> Vec<Vec<(u16, I32F32)>> {
1021        // Retrieve the bonds moving average for the given network ID and scale it down.
1022        let bonds_moving_average: I64F64 =
1023            I64F64::saturating_from_num(Self::get_bonds_moving_average(netuid))
1024                .safe_div(I64F64::saturating_from_num(1_000_000));
1025
1026        // Calculate the alpha value for the EMA calculation.
1027        // Alpha is derived by subtracting the scaled bonds moving average from 1.
1028        let alpha: I32F32 = I32F32::saturating_from_num(1)
1029            .saturating_sub(I32F32::saturating_from_num(bonds_moving_average));
1030
1031        // Compute the Exponential Moving Average (EMA) of bonds using the calculated alpha value.
1032        let ema_bonds = mat_ema_sparse(bonds_delta, bonds, alpha);
1033
1034        // Log the computed EMA bonds for debugging purposes.
1035        log::trace!("Exponential Moving Average Bonds Normal: {:?}", ema_bonds);
1036
1037        // Return the computed EMA bonds.
1038        ema_bonds
1039    }

1041    /// Compute the Exponential Moving Average (EMA) of bonds using a normal alpha value.
1042    ///
1043    /// # Args:
1044    /// * `bonds_delta` - A vector of bond deltas.
1045    /// * `bonds` - A vector of bonds.
1046    /// * `netuid` - The network ID.
1047    ///
1048    /// # Returns:
1049    /// A vector of EMA bonds.
1050    pub fn compute_ema_bonds_normal(
1051        bonds_delta: &[Vec<I32F32>],
1052        bonds: &[Vec<I32F32>],
1053        netuid: u16,
1054    ) -> Vec<Vec<I32F32>> {
1055        // Retrieve the bonds moving average for the given network ID and scale it down.
1056        let bonds_moving_average: I64F64 =
1057            I64F64::saturating_from_num(Self::get_bonds_moving_average(netuid))
1058                .safe_div(I64F64::saturating_from_num(1_000_000));
1059
1060        // Calculate the alpha value for the EMA calculation.
1061        // Alpha is derived by subtracting the scaled bonds moving average from 1.
1062        let alpha: I32F32 = I32F32::saturating_from_num(1)
1063            .saturating_sub(I32F32::saturating_from_num(bonds_moving_average));
1064
1065        // Compute the Exponential Moving Average (EMA) of bonds using the calculated alpha value.
1066        let ema_bonds = mat_ema(bonds_delta, bonds, alpha);
1067
1068        // Log the computed EMA bonds for debugging purposes.
1069        log::trace!("Exponential Moving Average Bonds Normal: {:?}", ema_bonds);
1070
1071        // Return the computed EMA bonds.
1072        ema_bonds
1073    }

1202    /// There is a separate alpha param for each validator-miner binding
1203    ///
1204    /// # Args:
1205    /// * `netuid` - The network ID.
1206    /// * `weights` - A vector of weights.
1207    /// * `bonds` - A vector of bonds.
1208    /// * `consensus` - A vector of consensus values.
1209    ///
1210    /// # Returns:
1211    /// A dense matrix of alphas
1212    pub fn compute_liquid_alpha_values_sparse(
1213        netuid: u16,
1214        weights: &[Vec<(u16, I32F32)>], // current epoch weights
1215        bonds: &[Vec<(u16, I32F32)>],   // previous epoch bonds
1216        consensus: &[I32F32],           // previous epoch consensus weights
1217    ) -> Vec<Vec<I32F32>> {
1218        assert!(weights.len() == bonds.len());
1219
1220        let alpha_sigmoid_steepness: I32F32 = Self::get_alpha_sigmoid_steepness(netuid);
1221        let (alpha_low, alpha_high): (I32F32, I32F32) = Self::get_alpha_values_32(netuid);
1222
1223        let mut alphas = Vec::with_capacity(consensus.len());
1224        let zero = I32F32::from_num(0.0);
1225
1226        // iterate over rows
1227        for (w_row, b_row) in weights.iter().zip(bonds.iter()) {
1228            let mut row_alphas = Vec::with_capacity(w_row.len());
1229            let mut w_iter = w_row.iter().peekable();
1230            let mut b_iter = b_row.iter().peekable();
1231            for (j_pos, consensus_val) in consensus.iter().enumerate() {
1232                let j = j_pos as u16;
1233
1234                let mut weight = zero;
1235                while let Some(&&(i, val)) = w_iter.peek() {
1236                    if i < j {
1237                        w_iter.next();
1238                    } else {
1239                        if i == j {
1240                            weight = val;
1241                        }
1242                        break;
1243                    }
1244                }
1245
1246                let mut bond = zero;
1247                while let Some(&&(i, val)) = b_iter.peek() {
1248                    if i < j {
1249                        b_iter.next();
1250                    } else {
1251                        if i == j {
1252                            bond = val;
1253                        }
1254                        break;
1255                    }
1256                }
1257
1258                let alpha = Self::alpha_sigmoid(
1259                    *consensus_val,
1260                    weight,
1261                    bond,
1262                    alpha_low,
1263                    alpha_high,
1264                    alpha_sigmoid_steepness,
1265                );
1266                row_alphas.push(alpha);
1267            }
1268            alphas.push(row_alphas);
1269        }
1270        alphas
1271    }

1273    /// Helper function to compute the alpha value using a sigmoid function.
1274    pub fn alpha_sigmoid(
1275        consensus: I32F32,
1276        weight: I32F32,
1277        bond: I32F32,
1278        alpha_low: I32F32,
1279        alpha_high: I32F32,
1280        alpha_sigmoid_steepness: I32F32,
1281    ) -> I32F32 {
1282        let zero = I32F32::from_num(0.0);
1283        let one = I32F32::from_num(1.0);
1284
1285        let diff_buy = clamp_value(weight.saturating_sub(consensus), zero, one);
1286        let diff_sell = clamp_value(bond.saturating_sub(weight), zero, one);
1287        let combined_diff = if weight >= bond { diff_buy } else { diff_sell };
1288
1289        // sigmoid = 1. / (1. + e^(-steepness * (combined_diff - 0.5)))
1290        let sigmoid = one.saturating_div(
1291            one.saturating_add(safe_exp(
1292                I32F32::from_num(-1).saturating_mul(
1293                    alpha_sigmoid_steepness
1294                        .saturating_mul(combined_diff.saturating_sub(I32F32::from_num(0.5))),
1295                ),
1296            )),
1297        );
1298        let alpha =
1299            alpha_low.saturating_add(sigmoid.saturating_mul(alpha_high.saturating_sub(alpha_low)));
1300
1301        clamp_value(alpha, alpha_low, alpha_high)
1302    }

1310        // Alpha is derived by subtracting the scaled bonds moving average from 1.

1312            I32F32::from_num(1).saturating_sub(I32F32::from_num(bonds_moving_average));

1318        netuid: u16,

1321    ) -> Result<(), DispatchError> {

1324
1325        // --- 2. Ensure the function caller is the subnet owner or root.
1326        Self::ensure_subnet_owner_or_root(origin, netuid)?;
1327
1328        // --- 3. Ensure liquid alpha is enabled
1329        ensure!(
1330            Self::get_liquid_alpha_enabled(netuid),
1331            Error::<T>::LiquidAlphaDisabled
1332        );
1333
1334        let max_u16: u32 = u16::MAX as u32; // 65535
1335        let min_alpha_high: u16 = (max_u16.saturating_mul(4).safe_div(5)) as u16; // 52428
1336
1337        // --- 4. Ensure alpha high is greater than the minimum
1338        ensure!(alpha_high >= min_alpha_high, Error::<T>::AlphaHighTooLow);
1339
1340        // -- 5. Ensure alpha low is within range
1341        ensure!(
1342            alpha_low > 0 && alpha_low < min_alpha_high,
1343            Error::<T>::AlphaLowOutOfRange
1344        );
1345
1346        AlphaValues::<T>::insert(netuid, (alpha_low, alpha_high));
1347
1348        log::debug!(
1349            "AlphaValuesSet( netuid: {:?}, AlphaLow: {:?}, AlphaHigh: {:?} ) ",
1350            netuid,
1351            alpha_low,
1352            alpha_high,
1353        );

1355    }
