7    /// Swaps the hotkey of a coldkey account.
8    ///
9    /// # Arguments
10    ///
11    /// * `origin` - The origin of the transaction, and also the coldkey account.
12    /// * `old_hotkey` - The old hotkey to be swapped.
13    /// * `new_hotkey` - The new hotkey to replace the old one.
14    ///
15    /// # Returns
16    ///
17    /// * `DispatchResultWithPostInfo` - The result of the dispatch.
18    ///
19    /// # Errors
20    ///
21    /// * `NonAssociatedColdKey` - If the coldkey does not own the old hotkey.
22    /// * `HotKeySetTxRateLimitExceeded` - If the transaction rate limit is exceeded.
23    /// * `NewHotKeyIsSameWithOld` - If the new hotkey is the same as the old hotkey.
24    /// * `HotKeyAlreadyRegisteredInSubNet` - If the new hotkey is already registered in the subnet.
25    /// * `NotEnoughBalanceToPaySwapHotKey` - If there is not enough balance to pay for the swap.
26    pub fn do_swap_hotkey(
27        origin: T::RuntimeOrigin,
28        old_hotkey: &T::AccountId,
29        new_hotkey: &T::AccountId,
30    ) -> DispatchResultWithPostInfo {
31        // 1. Ensure the origin is signed and get the coldkey
32        let coldkey = ensure_signed(origin)?;
33
34        // 2. Initialize the weight for this operation
35        let mut weight = T::DbWeight::get().reads(2);
36
37        // 3. Ensure the new hotkey is different from the old one
38        ensure!(old_hotkey != new_hotkey, Error::<T>::NewHotKeyIsSameWithOld);
39
40        // 4. Ensure the new hotkey is not already registered on any network
41        ensure!(
42            !Self::is_hotkey_registered_on_any_network(new_hotkey),
43            Error::<T>::HotKeyAlreadyRegisteredInSubNet
44        );
45
46        // 5. Update the weight for the checks above
47        weight.saturating_accrue(T::DbWeight::get().reads_writes(2, 0));
48
49        // 6. Ensure the coldkey owns the old hotkey
50        ensure!(
51            Self::coldkey_owns_hotkey(&coldkey, old_hotkey),
52            Error::<T>::NonAssociatedColdKey
53        );
54
55        // 7. Get the current block number
56        let block: u64 = Self::get_current_block_as_u64();
57
58        // 8. Ensure the transaction rate limit is not exceeded
59        ensure!(
60            !Self::exceeds_tx_rate_limit(Self::get_last_tx_block(&coldkey), block),
61            Error::<T>::HotKeySetTxRateLimitExceeded
62        );
63
64        // 9. Update the weight for reading the total networks
65        weight.saturating_accrue(
66            T::DbWeight::get().reads((TotalNetworks::<T>::get().saturating_add(1u16)) as u64),
67        );
68
69        // 10. Get the cost for swapping the key
70        let swap_cost = Self::get_key_swap_cost();
71        log::debug!("Swap cost: {:?}", swap_cost);
72
73        // 11. Ensure the coldkey has enough balance to pay for the swap
74        ensure!(
75            Self::can_remove_balance_from_coldkey_account(&coldkey, swap_cost),
76            Error::<T>::NotEnoughBalanceToPaySwapHotKey
77        );
78
79        // 12. Remove the swap cost from the coldkey's account
80        let actual_burn_amount = Self::remove_balance_from_coldkey_account(&coldkey, swap_cost)?;
81
82        // 13. Burn the tokens
83        Self::burn_tokens(actual_burn_amount);
84
85        // 14. Perform the hotkey swap
86        let _ = Self::perform_hotkey_swap(old_hotkey, new_hotkey, &coldkey, &mut weight);
87
88        // 15. Update the last transaction block for the coldkey
89        Self::set_last_tx_block(&coldkey, block);
90        weight.saturating_accrue(T::DbWeight::get().writes(1));
91
92        // 16. Emit an event for the hotkey swap
93        Self::deposit_event(Event::HotkeySwapped {
94            coldkey,
95            old_hotkey: old_hotkey.clone(),
96            new_hotkey: new_hotkey.clone(),
97        });
98
99        // 17. Return the weight of the operation
100        Ok(Some(weight).into())
101    }

103    /// Performs the hotkey swap operation, transferring all associated data and state from the old hotkey to the new hotkey.
104    ///
105    /// This function executes a series of steps to ensure a complete transfer of all relevant information:
106    /// 1. Swaps the owner of the hotkey.
107    /// 2. Updates the list of owned hotkeys for the coldkey.
108    /// 3. Transfers the total hotkey stake.
109    /// 4. Moves all stake-related data for the interval.
110    /// 5. Updates the last transaction block for the new hotkey.
111    /// 6. Transfers the delegate take information.
112    /// 7. Swaps Senate membership if applicable.
113    /// 8. Updates delegate information.
114    /// 9. For each subnet:
115    ///    - Updates network membership status.
116    ///    - Transfers UID and key information.
117    ///    - Moves Prometheus data.
118    ///    - Updates axon information.
119    ///    - Transfers weight commits.
120    ///    - Updates loaded emission data.
121    /// 10. Transfers all stake information, including updating staking hotkeys for each coldkey.
122    ///
123    /// Throughout the process, the function accumulates the computational weight of operations performed.
124    ///
125    /// # Arguments
126    /// * `old_hotkey` - The AccountId of the current hotkey to be replaced.
127    /// * `new_hotkey` - The AccountId of the new hotkey to replace the old one.
128    /// * `coldkey` - The AccountId of the coldkey that owns both hotkeys.
129    /// * `weight` - A mutable reference to the Weight, updated as operations are performed.
130    ///
131    /// # Returns
132    /// * `DispatchResult` - Ok(()) if the swap was successful, or an error if any operation failed.
133    ///
134    /// # Note
135    /// This function performs extensive storage reads and writes, which can be computationally expensive.
136    /// The accumulated weight should be carefully considered in the context of block limits.
137    pub fn perform_hotkey_swap(
138        old_hotkey: &T::AccountId,
139        new_hotkey: &T::AccountId,
140        coldkey: &T::AccountId,
141        weight: &mut Weight,
142    ) -> DispatchResult {
143        // 1. Swap owner.
144        // Owner( hotkey ) -> coldkey -- the coldkey that owns the hotkey.
145        Owner::<T>::remove(old_hotkey);
146        Owner::<T>::insert(new_hotkey, coldkey.clone());
147        weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 1));
148
149        // 2. Swap OwnedHotkeys.
150        // OwnedHotkeys( coldkey ) -> Vec<hotkey> -- the hotkeys that the coldkey owns.
151        let mut hotkeys = OwnedHotkeys::<T>::get(coldkey);
152        // Add the new key if needed.
153        if !hotkeys.contains(new_hotkey) {
154            hotkeys.push(new_hotkey.clone());
155        }
156        // Remove the old key.
157        hotkeys.retain(|hk| *hk != *old_hotkey);
158        OwnedHotkeys::<T>::insert(coldkey, hotkeys);
159        weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 1));
160
161        // 3. Swap total hotkey alpha for all subnets it exists on.
162        // TotalHotkeyAlpha( hotkey, netuid ) -> alpha -- the total alpha that the hotkey has on a specific subnet.
163        TotalHotkeyAlpha::<T>::iter_prefix(old_hotkey)
164            .drain()
165            .for_each(|(netuid, old_alpha)| {
166                let new_total_hotkey_alpha = TotalHotkeyAlpha::<T>::get(new_hotkey, netuid);
167                TotalHotkeyAlpha::<T>::insert(
168                    new_hotkey,
169                    netuid,
170                    old_alpha.saturating_add(new_total_hotkey_alpha),
171                );
172                weight.saturating_accrue(T::DbWeight::get().reads_writes(2, 2));
173            });
174
175        // 4. Swap total hotkey shares on all subnets it exists on.
176        // TotalHotkeyShares( hotkey, netuid ) -> alpha -- the total alpha that the hotkey has on a specific subnet.
177        TotalHotkeyShares::<T>::iter_prefix(old_hotkey)
178            .drain()
179            .for_each(|(netuid, old_shares)| {
180                let new_total_hotkey_shares = TotalHotkeyShares::<T>::get(new_hotkey, netuid);
181                TotalHotkeyShares::<T>::insert(
182                    new_hotkey,
183                    netuid,
184                    old_shares.saturating_add(new_total_hotkey_shares),
185                );
186                weight.saturating_accrue(T::DbWeight::get().reads_writes(2, 2));
187            });
188
189        // 5. Swap LastTxBlock
190        // LastTxBlock( hotkey ) --> u64 -- the last transaction block for the hotkey.
191        let last_tx_block: u64 = LastTxBlock::<T>::get(old_hotkey);
192        LastTxBlock::<T>::remove(old_hotkey);
193        LastTxBlock::<T>::insert(new_hotkey, last_tx_block);
194        weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
195
196        // 6. Swap LastTxBlockDelegateTake
197        // LastTxBlockDelegateTake( hotkey ) --> u64 -- the last transaction block for the hotkey delegate take.
198        let last_tx_block_delegate_take: u64 = LastTxBlockDelegateTake::<T>::get(old_hotkey);
199        LastTxBlockDelegateTake::<T>::remove(old_hotkey);
200        LastTxBlockDelegateTake::<T>::insert(new_hotkey, last_tx_block_delegate_take);
201        weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
202
203        // 7. Swap LastTxBlockChildKeyTake
204        // LastTxBlockChildKeyTake( hotkey ) --> u64 -- the last transaction block for the hotkey child key take.
205        let last_tx_block_child_key_take: u64 = LastTxBlockChildKeyTake::<T>::get(old_hotkey);
206        LastTxBlockChildKeyTake::<T>::remove(old_hotkey);
207        LastTxBlockChildKeyTake::<T>::insert(new_hotkey, last_tx_block_child_key_take);
208        weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
209
210        // 8. Swap Senate members.
211        // Senate( hotkey ) --> ?
212        if T::SenateMembers::is_member(old_hotkey) {
213            T::SenateMembers::swap_member(old_hotkey, new_hotkey).map_err(|e| e.error)?;
214            weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
215        }
216
217        // 9. Swap delegates.
218        // Delegates( hotkey ) -> take value -- the hotkey delegate take value.
219        if Delegates::<T>::contains_key(old_hotkey) {
220            let old_delegate_take = Delegates::<T>::get(old_hotkey);
221            Delegates::<T>::remove(old_hotkey);
222            Delegates::<T>::insert(new_hotkey, old_delegate_take);
223            weight.saturating_accrue(T::DbWeight::get().reads_writes(2, 2));
224        }
225
226        // 9. swap PendingHotkeyEmissionOnNetuid
227        // (DEPRECATED.)
228
229        // 10. Swap all subnet specific info.
230        let all_netuids: Vec<u16> = Self::get_all_subnet_netuids();
231        all_netuids.iter().for_each(|netuid| {
232            // 10.1 Remove the previous hotkey and insert the new hotkey from membership.
233            // IsNetworkMember( hotkey, netuid ) -> bool -- is the hotkey a subnet member.
234            let is_network_member: bool = IsNetworkMember::<T>::get(old_hotkey, netuid);
235            IsNetworkMember::<T>::remove(old_hotkey, netuid);
236            IsNetworkMember::<T>::insert(new_hotkey, netuid, is_network_member);
237            weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
238
239            // 10.2 Swap Uids + Keys.
240            // Keys( netuid, hotkey ) -> uid -- the uid the hotkey has in the network if it is a member.
241            // Uids( netuid, hotkey ) -> uid -- the uids that the hotkey has.
242            if is_network_member {
243                // 10.2.1 Swap the UIDS
244                if let Ok(old_uid) = Uids::<T>::try_get(netuid, old_hotkey) {
245                    Uids::<T>::remove(netuid, old_hotkey);
246                    Uids::<T>::insert(netuid, new_hotkey, old_uid);
247                    weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
248
249                    // 10.2.2 Swap the keys.
250                    Keys::<T>::insert(netuid, old_uid, new_hotkey.clone());
251                    weight.saturating_accrue(T::DbWeight::get().reads_writes(0, 1));
252                }
253            }
254
255            // 10.3 Swap Prometheus.
256            // Prometheus( netuid, hotkey ) -> prometheus -- the prometheus data that a hotkey has in the network.
257            if is_network_member {
258                if let Ok(old_prometheus_info) = Prometheus::<T>::try_get(netuid, old_hotkey) {
259                    Prometheus::<T>::remove(netuid, old_hotkey);
260                    Prometheus::<T>::insert(netuid, new_hotkey, old_prometheus_info);
261                    weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
262                }
263            }
264
265            // 10.4. Swap axons.
266            // Axons( netuid, hotkey ) -> axon -- the axon that the hotkey has.
267            if is_network_member {
268                if let Ok(old_axon_info) = Axons::<T>::try_get(netuid, old_hotkey) {
269                    Axons::<T>::remove(netuid, old_hotkey);
270                    Axons::<T>::insert(netuid, new_hotkey, old_axon_info);
271                    weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
272                }
273            }
274
275            // 10.5 Swap WeightCommits
276            // WeightCommits( hotkey ) --> Vec<u64> -- the weight commits for the hotkey.
277            if is_network_member {
278                if let Ok(old_weight_commits) = WeightCommits::<T>::try_get(netuid, old_hotkey) {
279                    WeightCommits::<T>::remove(netuid, old_hotkey);
280                    WeightCommits::<T>::insert(netuid, new_hotkey, old_weight_commits);
281                    weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
282                }
283            }
284
285            // 10.6. Swap the subnet loaded emission.
286            // LoadedEmission( netuid ) --> Vec<(hotkey, u64)> -- the loaded emission for the subnet.
287            if is_network_member {
288                if let Some(mut old_loaded_emission) = LoadedEmission::<T>::get(netuid) {
289                    for emission in old_loaded_emission.iter_mut() {
290                        if emission.0 == *old_hotkey {
291                            emission.0 = new_hotkey.clone();
292                        }
293                    }
294                    LoadedEmission::<T>::remove(netuid);
295                    LoadedEmission::<T>::insert(netuid, old_loaded_emission);
296                    weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
297                }
298            }
299
300            // 10.7. Swap neuron TLS certificates.
301            // NeuronCertificates( netuid, hotkey ) -> Vec<u8> -- the neuron certificate for the hotkey.
302            if is_network_member {
303                if let Ok(old_neuron_certificates) =
304                    NeuronCertificates::<T>::try_get(netuid, old_hotkey)
305                {
306                    NeuronCertificates::<T>::remove(netuid, old_hotkey);
307                    NeuronCertificates::<T>::insert(netuid, new_hotkey, old_neuron_certificates);
308                    weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
309                }
310            }
311        });
312
313        // 11. Swap Alpha
314        // Alpha( hotkey, coldkey, netuid ) -> alpha
315        let old_alpha_values: Vec<((T::AccountId, u16), U64F64)> =
316            Alpha::<T>::iter_prefix((old_hotkey,)).collect();
317        // Clear the entire old prefix here.
318        let _ = Alpha::<T>::clear_prefix((old_hotkey,), old_alpha_values.len() as u32, None);
319        weight.saturating_accrue(T::DbWeight::get().reads(old_alpha_values.len() as u64));
320        weight.saturating_accrue(T::DbWeight::get().writes(old_alpha_values.len() as u64));
321
322        // Insert the new alpha values.
323        for ((coldkey, netuid), alpha) in old_alpha_values {
324            let new_alpha = Alpha::<T>::get((new_hotkey, &coldkey, netuid));
325            Alpha::<T>::insert(
326                (new_hotkey, &coldkey, netuid),
327                new_alpha.saturating_add(alpha),
328            );
329            weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 1));
330
331            // Swap StakingHotkeys.
332            // StakingHotkeys( coldkey ) --> Vec<hotkey> -- the hotkeys that the coldkey stakes.
333            let mut staking_hotkeys = StakingHotkeys::<T>::get(&coldkey);
334            weight.saturating_accrue(T::DbWeight::get().reads(1));
335            if staking_hotkeys.contains(old_hotkey) {
336                staking_hotkeys.retain(|hk| *hk != *old_hotkey && *hk != *new_hotkey);
337                staking_hotkeys.push(new_hotkey.clone());
338                StakingHotkeys::<T>::insert(&coldkey, staking_hotkeys);
339                weight.saturating_accrue(T::DbWeight::get().writes(1));
340            }
341        }
342
343        // 12. Swap ChildKeys.
344        // ChildKeys( parent, netuid ) --> Vec<(proportion,child)> -- the child keys of the parent.
345        for netuid in Self::get_all_subnet_netuids() {
346            // Get the children of the old hotkey for this subnet
347            let my_children: Vec<(u64, T::AccountId)> = ChildKeys::<T>::get(old_hotkey, netuid);
348            // Remove the old hotkey's child entries
349            ChildKeys::<T>::remove(old_hotkey, netuid);
350            // Insert the same child entries for the new hotkey
351            ChildKeys::<T>::insert(new_hotkey, netuid, my_children.clone());
352            weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
353            for (_, child_key_i) in my_children {
354                // For each child, update their parent list
355                let mut child_parents: Vec<(u64, T::AccountId)> =
356                    ParentKeys::<T>::get(child_key_i.clone(), netuid);
357                for parent in child_parents.iter_mut() {
358                    // If the parent is the old hotkey, replace it with the new hotkey
359                    if parent.1 == *old_hotkey {
360                        parent.1 = new_hotkey.clone();
361                    }
362                }
363                // Update the child's parent list
364                ParentKeys::<T>::insert(child_key_i, netuid, child_parents);
365                weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 1));
366            }
367        }
368
369        // 13. Swap ParentKeys.
370        // ParentKeys( child, netuid ) --> Vec<(proportion,parent)> -- the parent keys of the child.
371        for netuid in Self::get_all_subnet_netuids() {
372            // Get the parents of the old hotkey for this subnet
373            let parents: Vec<(u64, T::AccountId)> = ParentKeys::<T>::get(old_hotkey, netuid);
374            // Remove the old hotkey's parent entries
375            ParentKeys::<T>::remove(old_hotkey, netuid);
376            // Insert the same parent entries for the new hotkey
377            ParentKeys::<T>::insert(new_hotkey, netuid, parents.clone());
378            weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
379            for (_, parent_key_i) in parents {
380                // For each parent, update their children list
381                let mut parent_children: Vec<(u64, T::AccountId)> =
382                    ChildKeys::<T>::get(parent_key_i.clone(), netuid);
383                for child in parent_children.iter_mut() {
384                    // If the child is the old hotkey, replace it with the new hotkey
385                    if child.1 == *old_hotkey {
386                        child.1 = new_hotkey.clone();
387                    }
388                }
389                // Update the parent's children list
390                ChildKeys::<T>::insert(parent_key_i, netuid, parent_children);
391                weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 1));
392            }
393        }
394
395        // 14. Swap PendingChildKeys.
396        // PendingChildKeys( netuid, parent ) --> Vec<(proportion,child), cool_down_block>
397        for netuid in Self::get_all_subnet_netuids() {
398            weight.saturating_accrue(T::DbWeight::get().reads(1));
399            if PendingChildKeys::<T>::contains_key(netuid, old_hotkey) {
400                let (children, cool_down_block) = PendingChildKeys::<T>::get(netuid, old_hotkey);
401                PendingChildKeys::<T>::remove(netuid, old_hotkey);
402                PendingChildKeys::<T>::insert(netuid, new_hotkey, (children, cool_down_block));
403                weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
404            }
405
406            // Also check for others with our hotkey as a child
407            for (hotkey, (children, cool_down_block)) in PendingChildKeys::<T>::iter_prefix(netuid)
408            {
409                weight.saturating_accrue(T::DbWeight::get().reads(1));
410
411                if let Some(potential_idx) =
412                    children.iter().position(|(_, child)| *child == *old_hotkey)
413                {
414                    let mut new_children = children.clone();
415                    let entry_to_remove = new_children.remove(potential_idx);
416                    new_children.push((entry_to_remove.0, new_hotkey.clone())); // Keep the proportion.
417
418                    PendingChildKeys::<T>::remove(netuid, hotkey.clone());
419                    PendingChildKeys::<T>::insert(netuid, hotkey, (new_children, cool_down_block));
420                    weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
421                }
422            }
423        }
424
425        // 15. Swap SubnetOwnerHotkey
426        // SubnetOwnerHotkey( netuid ) --> hotkey -- the hotkey that is the owner of the subnet.
427        for netuid in Self::get_all_subnet_netuids() {
428            if let Ok(old_subnet_owner_hotkey) = SubnetOwnerHotkey::<T>::try_get(netuid) {
429                weight.saturating_accrue(T::DbWeight::get().reads(1));
430                if old_subnet_owner_hotkey == *old_hotkey {
431                    SubnetOwnerHotkey::<T>::insert(netuid, new_hotkey);
432                    weight.saturating_accrue(T::DbWeight::get().writes(1));
433                }
434            }
435        }
436
437        // 16. Swap dividend records
438        TotalHotkeyAlphaLastEpoch::<T>::iter_prefix(old_hotkey)
439            .drain()
440            .for_each(|(netuid, old_alpha)| {
441                // 16.1 Swap TotalHotkeyAlphaLastEpoch
442                let new_total_hotkey_alpha =
443                    TotalHotkeyAlphaLastEpoch::<T>::get(new_hotkey, netuid);
444                TotalHotkeyAlphaLastEpoch::<T>::insert(
445                    new_hotkey,
446                    netuid,
447                    old_alpha.saturating_add(new_total_hotkey_alpha),
448                );
449                weight.saturating_accrue(T::DbWeight::get().reads_writes(2, 2));
450
451                // 16.2 Swap AlphaDividendsPerSubnet
452                let old_hotkey_alpha_dividends =
453                    AlphaDividendsPerSubnet::<T>::get(netuid, old_hotkey);
454                let new_hotkey_alpha_dividends =
455                    AlphaDividendsPerSubnet::<T>::get(netuid, new_hotkey);
456                AlphaDividendsPerSubnet::<T>::remove(netuid, old_hotkey);
457                AlphaDividendsPerSubnet::<T>::insert(
458                    netuid,
459                    new_hotkey,
460                    old_hotkey_alpha_dividends.saturating_add(new_hotkey_alpha_dividends),
461                );
462                weight.saturating_accrue(T::DbWeight::get().reads_writes(2, 2));
463
464                // 16.3 Swap TaoDividendsPerSubnet
465                let old_hotkey_tao_dividends = TaoDividendsPerSubnet::<T>::get(netuid, old_hotkey);
466                let new_hotkey_tao_dividends = TaoDividendsPerSubnet::<T>::get(netuid, new_hotkey);
467                TaoDividendsPerSubnet::<T>::remove(netuid, old_hotkey);
468                TaoDividendsPerSubnet::<T>::insert(
469                    netuid,
470                    new_hotkey,
471                    old_hotkey_tao_dividends.saturating_add(new_hotkey_tao_dividends),
472                );
473                weight.saturating_accrue(T::DbWeight::get().reads_writes(2, 2));
474            });
475
476        // Return successful after swapping all the relevant terms.
477        Ok(())
478    }

480    pub fn swap_senate_member(
481        old_hotkey: &T::AccountId,
482        new_hotkey: &T::AccountId,
483        weight: &mut Weight,
484    ) -> DispatchResult {
485        weight.saturating_accrue(T::DbWeight::get().reads(1));
486        if T::SenateMembers::is_member(old_hotkey) {
487            T::SenateMembers::swap_member(old_hotkey, new_hotkey).map_err(|e| e.error)?;
488            weight.saturating_accrue(T::DbWeight::get().reads_writes(1, 2));
489        }
490        Ok(())
491    }
