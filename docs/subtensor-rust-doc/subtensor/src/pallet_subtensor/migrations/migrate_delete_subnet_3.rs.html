23/// Migrates the storage to delete subnet 3
24///
25/// This function performs the following steps:
26/// 1. Checks if the migration is necessary
27/// 2. Removes all storage related to subnet 3
28/// 3. Updates the storage version
29///
30/// # Arguments
31///
32/// * `T` - The Config trait of the pallet
33///
34/// # Returns
35///
36/// * `Weight` - The computational weight of this operation
37///
38/// # Example
39///
40/// ```ignore
41/// let weight = migrate_delete_subnet_3::<T>();
42/// ```
43pub fn migrate_delete_subnet_3<T: Config>() -> Weight {
44    let new_storage_version = 5;
45
46    // Initialize weight counter
47    let mut weight = T::DbWeight::get().reads(1);
48
49    // Get current on-chain storage version
50    let onchain_version = Pallet::<T>::on_chain_storage_version();
51
52    // Only proceed if current version is less than the new version and subnet 3 exists
53    if onchain_version < new_storage_version && Pallet::<T>::if_subnet_exist(3) {
54        info!(
55            target: LOG_TARGET,
56            "Removing subnet 3. Current version: {:?}",
57            onchain_version
58        );
59
60        let netuid = 3;
61
62        // Remove network count
63        SubnetworkN::<T>::remove(netuid);
64
65        // Remove network modality storage
66        NetworkModality::<T>::remove(netuid);
67
68        // Remove netuid from added networks
69        NetworksAdded::<T>::remove(netuid);
70
71        // Decrement the network counter
72        TotalNetworks::<T>::mutate(|n| *n = n.saturating_sub(1));
73
74        // Remove network registration time
75        NetworkRegisteredAt::<T>::remove(netuid);
76
77        weight.saturating_accrue(T::DbWeight::get().writes(5));
78
79        // Remove incentive mechanism memory
80        let _ = Uids::<T>::clear_prefix(netuid, u32::MAX, None);
81        let _ = Keys::<T>::clear_prefix(netuid, u32::MAX, None);
82        let _ = Bonds::<T>::clear_prefix(netuid, u32::MAX, None);
83        let _ = Weights::<T>::clear_prefix(netuid, u32::MAX, None);
84
85        weight.saturating_accrue(T::DbWeight::get().writes(4));
86
87        // Remove various network-related parameters
88        Rank::<T>::remove(netuid);
89        Trust::<T>::remove(netuid);
90        Active::<T>::remove(netuid);
91        Emission::<T>::remove(netuid);
92        Incentive::<T>::remove(netuid);
93        Consensus::<T>::remove(netuid);
94        Dividends::<T>::remove(netuid);
95        PruningScores::<T>::remove(netuid);
96        LastUpdate::<T>::remove(netuid);
97        ValidatorPermit::<T>::remove(netuid);
98        ValidatorTrust::<T>::remove(netuid);
99
100        weight.saturating_accrue(T::DbWeight::get().writes(11));
101
102        // Erase network parameters
103        Tempo::<T>::remove(netuid);
104        Kappa::<T>::remove(netuid);
105        Difficulty::<T>::remove(netuid);
106        MaxAllowedUids::<T>::remove(netuid);
107        ImmunityPeriod::<T>::remove(netuid);
108        ActivityCutoff::<T>::remove(netuid);
109        MaxWeightsLimit::<T>::remove(netuid);
110        MinAllowedWeights::<T>::remove(netuid);
111        RegistrationsThisInterval::<T>::remove(netuid);
112        POWRegistrationsThisInterval::<T>::remove(netuid);
113        BurnRegistrationsThisInterval::<T>::remove(netuid);
114
115        weight.saturating_accrue(T::DbWeight::get().writes(12));
116
117        // Update storage version
118        StorageVersion::new(new_storage_version).put::<Pallet<T>>();
119        weight.saturating_accrue(T::DbWeight::get().writes(1));
120
121        weight
122    } else {
123        info!(target: LOG_TARGET, "Migration to v5 already completed or subnet 3 doesn't exist");
124        Weight::zero()
125    }
126}
