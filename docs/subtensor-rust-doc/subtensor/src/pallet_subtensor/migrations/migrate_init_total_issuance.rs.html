17pub(crate) fn migrate_init_total_issuance<T: Config>() -> Weight {
18    let subnets_len = crate::SubnetLocked::<T>::iter().count() as u64;
19
20    // Retrieve the total balance of all accounts
21    let total_account_balances = <<T as crate::Config>::Currency as fungible::Inspect<
22        <T as frame_system::Config>::AccountId,
23    >>::total_issuance();
24
25    // Get the total stake from the system
26    let prev_total_stake = crate::TotalStake::<T>::get();
27
28    // Calculate new total stake using the sum of all subnet TAO
29    let total_subnet_tao: u64 =
30        crate::SubnetTAO::<T>::iter().fold(0, |acc, (_, v)| acc.saturating_add(v));
31
32    let total_stake = total_subnet_tao;
33    // Update the total stake in storage
34    crate::TotalStake::<T>::put(total_stake);
35    log::info!(
36        "Subtensor Pallet Total Stake Updated: previous: {:?}, new: {:?}",
37        prev_total_stake,
38        total_stake
39    );
40    // Retrieve the previous total issuance for logging purposes
41    let prev_total_issuance = crate::TotalIssuance::<T>::get();
42
43    // Calculate the new total issuance
44    let new_total_issuance = total_account_balances.saturating_add(total_stake);
45
46    // Update the total issuance in storage
47    crate::TotalIssuance::<T>::put(new_total_issuance);
48
49    // Log the change in total issuance
50    log::info!(
51        "Subtensor Pallet Total Issuance Updated: previous: {:?}, new: {:?}",
52        prev_total_issuance,
53        new_total_issuance
54    );
55
56    // Return the weight of the operation
57    // We performed subnets_len + 5 reads and 1 write
58    <T as frame_system::Config>::DbWeight::get().reads_writes(subnets_len.saturating_add(5), 2)
59}

61pub mod initialise_total_issuance {
62    use frame_support::pallet_prelude::Weight;
63    use frame_support::traits::OnRuntimeUpgrade;
64
65    use crate::*;
66
67    pub struct Migration<T: Config>(PhantomData<T>);
68
69    impl<T: Config> OnRuntimeUpgrade for Migration<T> {
70        /// Performs the migration to initialize and update the total issuance.
71        ///
72        /// This function does the following:
73        /// 1. Calculates the total locked tokens across all subnets
74        /// 2. Retrieves the total account balances and total stake
75        /// 3. Computes and updates the new total issuance
76        ///
77        /// Returns the weight of the migration operation.
78        fn on_runtime_upgrade() -> Weight {
79            super::migrate_init_total_issuance::<T>()
80        }
81
82        /// Performs post-upgrade checks to ensure the migration was successful.
83        ///
84        /// This function is only compiled when the "try-runtime" feature is enabled.
85        #[cfg(feature = "try-runtime")]
86        fn post_upgrade(_state: Vec<u8>) -> Result<(), sp_runtime::TryRuntimeError> {
87            // Verify that all accounting invariants are satisfied after the migration
88            crate::Pallet::<T>::check_total_issuance()?;
89            Ok(())
90        }
91    }
92}
