9pub fn migrate_rao<T: Config>() -> Weight {
10    let migration_name = b"migrate_rao".to_vec();
11
12    // Initialize the weight with one read operation.
13    let mut weight = T::DbWeight::get().reads(1);
14
15    // Check if the migration has already run
16    if HasMigrationRun::<T>::get(&migration_name) {
17        log::info!(
18            "Migration '{:?}' has already run. Skipping.",
19            migration_name
20        );
21        return weight;
22    }
23    log::info!(
24        "Running migration '{}'",
25        String::from_utf8_lossy(&migration_name)
26    );
27
28    let netuids: Vec<u16> = <NetworksAdded<T> as IterableStorageMap<u16, bool>>::iter()
29        .map(|(netuid, _)| netuid)
30        .collect();
31    weight = weight.saturating_add(T::DbWeight::get().reads_writes(netuids.len() as u64, 0));
32
33    // Migrate all TAO to root.
34    // This migration has already run, leaving this only for reference for now, since this is a recent migration
35    // Stake::<T>::iter().for_each(|(hotkey, coldkey, stake)| {
36    //     // Increase SubnetTAO on root.
37    //     SubnetTAO::<T>::mutate(0, |total| {
38    //         *total = total.saturating_add(stake);
39    //     });
40    //     // Increase SubnetAlphaOut on root.
41    //     SubnetAlphaOut::<T>::mutate(0, |total| {
42    //         *total = total.saturating_add(stake);
43    //     });
44    //     // Set all the stake on root 0 subnet.
45    //     Alpha::<T>::mutate((hotkey.clone(), coldkey.clone(), 0), |total| {
46    //         *total = total.saturating_add(U64F64::saturating_from_num(stake))
47    //     });
48    //     TotalHotkeyShares::<T>::mutate(hotkey.clone(), 0, |total| {
49    //         *total = total.saturating_add(U64F64::saturating_from_num(stake))
50    //     });
51    //     // Set the total stake on the hotkey
52    //     TotalHotkeyAlpha::<T>::mutate(hotkey.clone(), 0, |total| {
53    //         *total = total.saturating_add(stake)
54    //     });
55    //     // 6 reads and 6 writes.
56    //     weight = weight.saturating_add(T::DbWeight::get().reads_writes(6, 6));
57    // });
58
59    // Convert subnets and give them lock.
60    // Set global weight to 18% from the start
61    // Set min lock
62    NetworkMinLockCost::<T>::set(1_000_000_000);
63    // Set tao weight.
64    TaoWeight::<T>::set(3_320_413_933_267_719_290);
65    for netuid in netuids.iter().clone() {
66        if *netuid == 0 {
67            // Give root a single RAO in pool to avoid any catestrophic division by zero.
68            SubnetAlphaIn::<T>::insert(netuid, 1_000_000_000);
69            SubnetMechanism::<T>::insert(netuid, 0); // Set to zero mechanism.
70            TokenSymbol::<T>::insert(netuid, Pallet::<T>::get_symbol_for_subnet(0));
71            continue;
72        }
73        let owner = SubnetOwner::<T>::get(netuid);
74        let lock = SubnetLocked::<T>::get(netuid);
75
76        // Put initial TAO from lock into subnet TAO and produce numerically equal amount of Alpha
77        // The initial TAO is the locked amount, with a minimum of 1 RAO and a cap of 100 TAO.
78        let pool_initial_tao = Pallet::<T>::get_network_min_lock();
79        if lock < pool_initial_tao {
80            let difference: u64 = pool_initial_tao.saturating_sub(lock);
81            TotalIssuance::<T>::mutate(|total| {
82                *total = total.saturating_add(difference);
83            });
84        }
85
86        let remaining_lock = lock.saturating_sub(pool_initial_tao);
87        // Refund the owner for the remaining lock.
88        // SubnetMovingPrice::<T>::insert(
89        //     netuid,
90        //     I96F32::from_num(EmissionValues::<T>::get(netuid))
91        //         .checked_div(I96F32::from_num(1_000_000_000))
92        //         .unwrap_or(I96F32::from_num(0.0)),
93        // );
94        Pallet::<T>::add_balance_to_coldkey_account(&owner, remaining_lock);
95        SubnetLocked::<T>::insert(netuid, 0); // Clear lock amount.
96        SubnetTAO::<T>::insert(netuid, pool_initial_tao);
97        TotalStake::<T>::mutate(|total| {
98            *total = total.saturating_add(pool_initial_tao);
99        }); // Increase total stake.
100        SubnetAlphaIn::<T>::insert(netuid, pool_initial_tao); // Set initial alpha to pool initial tao.
101        SubnetAlphaOut::<T>::insert(netuid, 0); // Set zero subnet alpha out.
102        SubnetMechanism::<T>::insert(netuid, 1); // Convert to dynamic immediately with initialization.
103
104        // Set the token symbol for this subnet using Self instead of Pallet::<T>
105        TokenSymbol::<T>::insert(netuid, Pallet::<T>::get_symbol_for_subnet(*netuid));
106
107        if let Ok(owner_coldkey) = SubnetOwner::<T>::try_get(netuid) {
108            // Set Owner as the coldkey.
109            SubnetOwnerHotkey::<T>::insert(netuid, owner_coldkey.clone());
110            // Associate the coldkey to coldkey.
111            Pallet::<T>::create_account_if_non_existent(&owner_coldkey, &owner_coldkey);
112
113            // Only register the owner coldkey if it's not already a hotkey on the subnet.
114            if !Uids::<T>::contains_key(*netuid, &owner_coldkey) {
115                // Register the owner_coldkey as neuron to the network.
116                let _neuron_uid: u16 = Pallet::<T>::register_neuron(*netuid, &owner_coldkey);
117            }
118            // Register the neuron immediately.
119            if !IdentitiesV2::<T>::contains_key(owner_coldkey.clone()) {
120                // Set the identitiy for the Owner coldkey if non existent.
121                let identity = ChainIdentityOfV2 {
122                    name: format!("Owner{}", netuid).as_bytes().to_vec(),
123                    url: Vec::new(),
124                    image: Vec::new(),
125                    github_repo: Vec::new(),
126                    discord: Vec::new(),
127                    description: Vec::new(),
128                    additional: Vec::new(),
129                };
130                // Validate the created identity and set it.
131                if Pallet::<T>::is_valid_identity(&identity) {
132                    IdentitiesV2::<T>::insert(owner_coldkey.clone(), identity.clone());
133                }
134            }
135        }
136
137        // HotkeyEmissionTempo::<T>::put(30); // same as subnet tempo // (DEPRECATED)
138        // Set the target stakes per interval to 10.
139        // TargetStakesPerInterval::<T>::put(10); (DEPRECATED)
140    }
141
142    // update `TotalIssuance`, because currency issuance (`T::Currency`) has changed due to lock
143    // refunds above
144    weight = weight.saturating_add(migrate_init_total_issuance::migrate_init_total_issuance::<T>());
145
146    // Mark the migration as completed
147    HasMigrationRun::<T>::insert(&migration_name, true);
148    weight = weight.saturating_add(T::DbWeight::get().writes(1));
149
150    log::info!(
151        "Migration '{:?}' completed. Storage version set to 7.",
152        String::from_utf8_lossy(&migration_name)
153    );
154
155    // Return the migration weight.
156    weight
157}
