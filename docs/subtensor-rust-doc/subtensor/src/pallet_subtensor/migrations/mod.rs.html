37pub(crate) fn migrate_storage<T: Config>(
38    migration_name: &'static str,
39    pallet_name: &'static str,
40    storage_name: &'static str,
41) -> Weight {
42    let migration_name_bytes = migration_name.as_bytes().to_vec();
43
44    let mut weight = T::DbWeight::get().reads(1);
45    if HasMigrationRun::<T>::get(&migration_name_bytes) {
46        log::info!(
47            "Migration '{:?}' has already run. Skipping.",
48            migration_name
49        );
50        return weight;
51    }
52
53    log::info!("Running migration '{}'", migration_name);
54
55    let pallet_name = twox_128(pallet_name.as_bytes());
56    let storage_name = twox_128(storage_name.as_bytes());
57    let prefix = [pallet_name, storage_name].concat();
58
59    // Remove all entries.
60    let removed_entries_count = match clear_prefix(&prefix, Some(u32::MAX)) {
61        KillStorageResult::AllRemoved(removed) => {
62            log::info!("Removed all entries from {:?}.", storage_name);
63
64            // Mark migration as completed
65            HasMigrationRun::<T>::insert(&migration_name_bytes, true);
66            weight = weight.saturating_add(T::DbWeight::get().writes(1));
67
68            removed as u64
69        }
70        KillStorageResult::SomeRemaining(removed) => {
71            log::info!("Failed to remove all entries from {:?}", storage_name);
72            removed as u64
73        }
74    };
75
76    weight = weight.saturating_add(T::DbWeight::get().writes(removed_entries_count as u64));
77
78    log::info!(
79        "Migration '{:?}' completed successfully. {:?} entries removed.",
80        migration_name,
81        removed_entries_count
82    );
83
84    weight
85}
