7    /// Executes the necessary operations for each block.
8    pub fn block_step() -> Result<(), &'static str> {
9        let block_number: u64 = Self::get_current_block_as_u64();
10        log::debug!("block_step for block: {:?} ", block_number);
11        // --- 1. Adjust difficulties.
12        Self::adjust_registration_terms_for_networks();
13        // --- 2. Get the current coinbase emission.
14        let block_emission: U96F32 =
15            U96F32::saturating_from_num(Self::get_block_emission().unwrap_or(0));
16        log::debug!("Block emission: {:?}", block_emission);
17        // --- 3. Run emission through network.
18        Self::run_coinbase(block_emission);
19        // --- 4. Set pending children on the epoch; but only after the coinbase has been run.
20        Self::try_set_pending_children(block_number);
21        // Return ok.
22        Ok(())
23    }
24
25    fn try_set_pending_children(block_number: u64) {
26        let subnets: Vec<u16> = Self::get_all_subnet_netuids();
27        for &netuid in subnets.iter() {
28            if Self::should_run_epoch(netuid, block_number) {
29                // Set pending children on the epoch.
30                Self::do_set_pending_children(netuid);
31            }
32        }
33    }

35    /// Adjusts the network difficulties/burns of every active network. Resetting state parameters.
36    ///
37    pub fn adjust_registration_terms_for_networks() {
38        log::debug!("adjust_registration_terms_for_networks");
39
40        // --- 1. Iterate through each network.
41        for (netuid, _) in <NetworksAdded<T> as IterableStorageMap<u16, bool>>::iter() {
42            // --- 2. Pull counters for network difficulty.
43            let last_adjustment_block: u64 = Self::get_last_adjustment_block(netuid);
44            let adjustment_interval: u16 = Self::get_adjustment_interval(netuid);
45            let current_block: u64 = Self::get_current_block_as_u64();
46            log::debug!(
47                "netuid: {:?} last_adjustment_block: {:?} adjustment_interval: {:?} current_block: {:?}",
48                netuid,
49                last_adjustment_block,
50                adjustment_interval,
51                current_block
52            );
53
54            // --- 3. Check if we are at the adjustment interval for this network.
55            // If so, we need to adjust the registration difficulty based on target and actual registrations.
56            if current_block.saturating_sub(last_adjustment_block) >= adjustment_interval as u64 {
57                log::debug!("interval reached.");
58
59                // --- 4. Get the current counters for this network w.r.t burn and difficulty values.
60                let current_burn: u64 = Self::get_burn_as_u64(netuid);
61                let current_difficulty: u64 = Self::get_difficulty_as_u64(netuid);
62                let registrations_this_interval: u16 =
63                    Self::get_registrations_this_interval(netuid);
64                let pow_registrations_this_interval: u16 =
65                    Self::get_pow_registrations_this_interval(netuid);
66                let burn_registrations_this_interval: u16 =
67                    Self::get_burn_registrations_this_interval(netuid);
68                let target_registrations_this_interval: u16 =
69                    Self::get_target_registrations_per_interval(netuid);
70                // --- 5. Adjust burn + pow
71                // There are six cases to consider. A, B, C, D, E, F
72                if registrations_this_interval > target_registrations_this_interval {
73                    #[allow(clippy::comparison_chain)]
74                    if pow_registrations_this_interval > burn_registrations_this_interval {
75                        // A. There are too many registrations this interval and most of them are pow registrations
76                        // this triggers an increase in the pow difficulty.
77                        // pow_difficulty ++
78                        Self::set_difficulty(
79                            netuid,
80                            Self::upgraded_difficulty(
81                                netuid,
82                                current_difficulty,
83                                registrations_this_interval,
84                                target_registrations_this_interval,
85                            ),
86                        );
87                    } else if pow_registrations_this_interval < burn_registrations_this_interval {
88                        // B. There are too many registrations this interval and most of them are burn registrations
89                        // this triggers an increase in the burn cost.
90                        // burn_cost ++
91                        Self::set_burn(
92                            netuid,
93                            Self::upgraded_burn(
94                                netuid,
95                                current_burn,
96                                registrations_this_interval,
97                                target_registrations_this_interval,
98                            ),
99                        );
100                    } else {
101                        // F. There are too many registrations this interval and the pow and burn registrations are equal
102                        // this triggers an increase in the burn cost and pow difficulty
103                        // burn_cost ++
104                        Self::set_burn(
105                            netuid,
106                            Self::upgraded_burn(
107                                netuid,
108                                current_burn,
109                                registrations_this_interval,
110                                target_registrations_this_interval,
111                            ),
112                        );
113                        // pow_difficulty ++
114                        Self::set_difficulty(
115                            netuid,
116                            Self::upgraded_difficulty(
117                                netuid,
118                                current_difficulty,
119                                registrations_this_interval,
120                                target_registrations_this_interval,
121                            ),
122                        );
123                    }
124                } else {
125                    // Not enough registrations this interval.
126                    #[allow(clippy::comparison_chain)]
127                    if pow_registrations_this_interval > burn_registrations_this_interval {
128                        // C. There are not enough registrations this interval and most of them are pow registrations
129                        // this triggers a decrease in the burn cost
130                        // burn_cost --
131                        Self::set_burn(
132                            netuid,
133                            Self::upgraded_burn(
134                                netuid,
135                                current_burn,
136                                registrations_this_interval,
137                                target_registrations_this_interval,
138                            ),
139                        );
140                    } else if pow_registrations_this_interval < burn_registrations_this_interval {
141                        // D. There are not enough registrations this interval and most of them are burn registrations
142                        // this triggers a decrease in the pow difficulty
143                        // pow_difficulty --
144                        Self::set_difficulty(
145                            netuid,
146                            Self::upgraded_difficulty(
147                                netuid,
148                                current_difficulty,
149                                registrations_this_interval,
150                                target_registrations_this_interval,
151                            ),
152                        );
153                    } else {
154                        // E. There are not enough registrations this interval and the pow and burn registrations are equal
155                        // this triggers a decrease in the burn cost and pow difficulty
156                        // burn_cost --
157                        Self::set_burn(
158                            netuid,
159                            Self::upgraded_burn(
160                                netuid,
161                                current_burn,
162                                registrations_this_interval,
163                                target_registrations_this_interval,
164                            ),
165                        );
166                        // pow_difficulty --
167                        Self::set_difficulty(
168                            netuid,
169                            Self::upgraded_difficulty(
170                                netuid,
171                                current_difficulty,
172                                registrations_this_interval,
173                                target_registrations_this_interval,
174                            ),
175                        );
176                    }
177                }
178
179                // --- 6. Drain all counters for this network for this interval.
180                Self::set_last_adjustment_block(netuid, current_block);
181                Self::set_registrations_this_interval(netuid, 0);
182                Self::set_pow_registrations_this_interval(netuid, 0);
183                Self::set_burn_registrations_this_interval(netuid, 0);
184            } else {
185                log::debug!("interval not reached.");
186            }
187
188            // --- 7. Drain block registrations for each network. Needed for registration rate limits.
189            Self::set_registrations_this_block(netuid, 0);
190        }
191    }

193    /// Calculates the upgraded difficulty by multiplying the current difficulty by the ratio ( reg_actual + reg_target / reg_target + reg_target )
194    /// We use U110F18 to avoid any overflows on u64. Also min_difficulty and max_difficulty bound the range.
195    ///
196    pub fn upgraded_difficulty(
197        netuid: u16,
198        current_difficulty: u64,
199        registrations_this_interval: u16,
200        target_registrations_per_interval: u16,
201    ) -> u64 {
202        let updated_difficulty: U110F18 = U110F18::saturating_from_num(current_difficulty)
203            .saturating_mul(U110F18::saturating_from_num(
204                registrations_this_interval.saturating_add(target_registrations_per_interval),
205            ))
206            .safe_div(U110F18::saturating_from_num(
207                target_registrations_per_interval.saturating_add(target_registrations_per_interval),
208            ));
209        let alpha: U110F18 = U110F18::saturating_from_num(Self::get_adjustment_alpha(netuid))
210            .safe_div(U110F18::saturating_from_num(u64::MAX));
211        let next_value: U110F18 = alpha
212            .saturating_mul(U110F18::saturating_from_num(current_difficulty))
213            .saturating_add(
214                U110F18::saturating_from_num(1.0)
215                    .saturating_sub(alpha)
216                    .saturating_mul(updated_difficulty),
217            );
218        if next_value >= U110F18::saturating_from_num(Self::get_max_difficulty(netuid)) {
219            Self::get_max_difficulty(netuid)
220        } else if next_value <= U110F18::saturating_from_num(Self::get_min_difficulty(netuid)) {
221            return Self::get_min_difficulty(netuid);
222        } else {
223            return next_value.saturating_to_num::<u64>();
224        }
225    }
