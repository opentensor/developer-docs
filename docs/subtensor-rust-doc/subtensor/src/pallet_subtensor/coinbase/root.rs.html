73    /// Retrieves weight matrix associated with the root network.
74    ///  Weights represent the preferences for each subnetwork.
75    ///
76    /// # Returns:
77    /// A 2D vector ('Vec<Vec<I32F32>>') where each entry [i][j] represents the weight of subnetwork
78    /// 'j' with according to the preferences of key. Validator 'i' within the root network.
79    ///
80    pub fn get_root_weights() -> Vec<Vec<I64F64>> {
81        // --- 0. The number of validators on the root network.
82        let n: usize = Self::get_num_root_validators() as usize;
83
84        // --- 1 The number of subnets to validate.
85        log::debug!("subnet size before cast: {:?}", Self::get_num_subnets());
86        let k: usize = Self::get_num_subnets() as usize;
87        log::debug!("n: {:?} k: {:?}", n, k);
88
89        // --- 2. Initialize a 2D vector with zeros to store the weights. The dimensions are determined
90        // by `n` (number of validators) and `k` (total number of subnets).
91        let mut weights: Vec<Vec<I64F64>> = vec![vec![I64F64::saturating_from_num(0.0); k]; n];
92        log::debug!("weights:\n{:?}\n", weights);
93
94        let subnet_list = Self::get_all_subnet_netuids();
95
96        // --- 3. Iterate over stored weights and fill the matrix.
97        for (uid_i, weights_i) in
98            <Weights<T> as IterableStorageDoubleMap<u16, u16, Vec<(u16, u16)>>>::iter_prefix(
99                Self::get_root_netuid(),
100            )
101        {
102            // --- 4. Iterate over each weight entry in `weights_i` to update the corresponding value in the
103            // initialized `weights` 2D vector. Here, `uid_j` represents a subnet, and `weight_ij` is the
104            // weight of `uid_i` with respect to `uid_j`.
105            for (netuid, weight_ij) in &weights_i {
106                let idx = uid_i as usize;
107                if let Some(weight) = weights.get_mut(idx) {
108                    if let Some((w, _)) = weight
109                        .iter_mut()
110                        .zip(&subnet_list)
111                        .find(|(_, subnet)| *subnet == netuid)
112                    {
113                        *w = I64F64::saturating_from_num(*weight_ij);
114                    }
115                }
116            }
117        }
118
119        // --- 5. Return the filled weights matrix.
120        weights
121    }

123    /// Registers a user's hotkey to the root network.
124    ///
125    /// This function is responsible for registering the hotkey of a user.
126    /// The root key with the least stake if pruned in the event of a filled network.
127    ///
128    /// # Arguments:
129    /// * 'origin': Represents the origin of the call.
130    /// * 'hotkey': The hotkey that the user wants to register to the root network.
131    ///
132    /// # Returns:
133    /// * 'DispatchResult': A result type indicating success or failure of the registration.
134    ///
135    pub fn do_root_register(origin: T::RuntimeOrigin, hotkey: T::AccountId) -> DispatchResult {
136        // --- 0. Get the unique identifier (UID) for the root network.
137        let root_netuid: u16 = Self::get_root_netuid();
138        let current_block_number: u64 = Self::get_current_block_as_u64();
139        ensure!(
140            Self::if_subnet_exist(root_netuid),
141            Error::<T>::RootNetworkDoesNotExist
142        );
143
144        // --- 1. Ensure that the call originates from a signed source and retrieve the caller's account ID (coldkey).
145        let coldkey = ensure_signed(origin)?;
146        log::debug!(
147            "do_root_register( coldkey: {:?}, hotkey: {:?} )",
148            coldkey,
149            hotkey
150        );
151
152        // --- 2. Ensure that the number of registrations in this block doesn't exceed the allowed limit.
153        ensure!(
154            Self::get_registrations_this_block(root_netuid)
155                < Self::get_max_registrations_per_block(root_netuid),
156            Error::<T>::TooManyRegistrationsThisBlock
157        );
158
159        // --- 3. Ensure that the number of registrations in this interval doesn't exceed thrice the target limit.
160        ensure!(
161            Self::get_registrations_this_interval(root_netuid)
162                < Self::get_target_registrations_per_interval(root_netuid).saturating_mul(3),
163            Error::<T>::TooManyRegistrationsThisInterval
164        );
165
166        // --- 4. Check if the hotkey is already registered. If so, error out.
167        ensure!(
168            !Uids::<T>::contains_key(root_netuid, &hotkey),
169            Error::<T>::HotKeyAlreadyRegisteredInSubNet
170        );
171
172        // --- 6. Create a network account for the user if it doesn't exist.
173        Self::create_account_if_non_existent(&coldkey, &hotkey);
174
175        // --- 7. Fetch the current size of the subnetwork.
176        let current_num_root_validators: u16 = Self::get_num_root_validators();
177
178        // Declare a variable to hold the root UID.
179        let subnetwork_uid: u16;
180
181        // --- 8. Check if the root net is below its allowed size.
182        // max allowed is senate size.
183        if current_num_root_validators < Self::get_max_root_validators() {
184            // --- 12.1.1 We can append to the subnetwork as it's not full.
185            subnetwork_uid = current_num_root_validators;
186
187            // --- 12.1.2 Add the new account and make them a member of the Senate.
188            Self::append_neuron(root_netuid, &hotkey, current_block_number);
189            log::debug!("add new neuron: {:?} on uid {:?}", hotkey, subnetwork_uid);
190        } else {
191            // --- 13.1.1 The network is full. Perform replacement.
192            // Find the neuron with the lowest stake value to replace.
193            let mut lowest_stake: u64 = u64::MAX;
194            let mut lowest_uid: u16 = 0;
195
196            // Iterate over all keys in the root network to find the neuron with the lowest stake.
197            for (uid_i, hotkey_i) in
198                <Keys<T> as IterableStorageDoubleMap<u16, u16, T::AccountId>>::iter_prefix(
199                    root_netuid,
200                )
201            {
202                let stake_i: u64 = Self::get_stake_for_hotkey_on_subnet(&hotkey_i, 0);
203                if stake_i < lowest_stake {
204                    lowest_stake = stake_i;
205                    lowest_uid = uid_i;
206                }
207            }
208            subnetwork_uid = lowest_uid;
209            let replaced_hotkey: T::AccountId =
210                Self::get_hotkey_for_net_and_uid(root_netuid, subnetwork_uid)?;
211
212            // --- 13.1.2 The new account has a higher stake than the one being replaced.
213            ensure!(
214                lowest_stake < Self::get_stake_for_hotkey_on_subnet(&hotkey, 0),
215                Error::<T>::StakeTooLowForRoot
216            );
217
218            // --- 13.1.3 The new account has a higher stake than the one being replaced.
219            // Replace the neuron account with new information.
220            Self::replace_neuron(root_netuid, lowest_uid, &hotkey, current_block_number);
221
222            log::debug!(
223                "replace neuron: {:?} with {:?} on uid {:?}",
224                replaced_hotkey,
225                hotkey,
226                subnetwork_uid
227            );
228        }
229
230        // --- 13. Join the Senate if eligible.
231        // Returns the replaced member, if any.
232        let _ = Self::join_senate_if_eligible(&hotkey)?;
233
234        // --- 14. Force all members on root to become a delegate.
235        if !Self::hotkey_is_delegate(&hotkey) {
236            Self::delegate_hotkey(&hotkey, 11_796); // 18% cut defaulted.
237        }
238
239        // --- 15. Update the registration counters for both the block and interval.
240        #[allow(clippy::arithmetic_side_effects)]
241        // note this RA + clippy false positive is a known substrate issue
242        RegistrationsThisInterval::<T>::mutate(root_netuid, |val| *val += 1);
243        #[allow(clippy::arithmetic_side_effects)]
244        // note this RA + clippy false positive is a known substrate issue
245        RegistrationsThisBlock::<T>::mutate(root_netuid, |val| *val += 1);
246
247        // --- 16. Log and announce the successful registration.
248        log::debug!(
249            "RootRegistered(netuid:{:?} uid:{:?} hotkey:{:?})",
250            root_netuid,
251            subnetwork_uid,
252            hotkey
253        );
254        Self::deposit_event(Event::NeuronRegistered(root_netuid, subnetwork_uid, hotkey));
255
256        // --- 17. Finish and return success.
257        Ok(())
258    }

260    // Checks if a hotkey should be a member of the Senate, and if so, adds them.
261    //
262    // This function is responsible for adding a hotkey to the Senate if they meet the requirements.
263    // The root key with the least stake is pruned in the event of a filled membership.
264    //
265    // # Arguments:
266    // * 'origin': Represents the origin of the call.
267    // * 'hotkey': The hotkey that the user wants to register to the root network.
268    //
269    // # Returns:
270    // * 'DispatchResult': A result type indicating success or failure of the registration.
271    //
272    pub fn do_adjust_senate(origin: T::RuntimeOrigin, hotkey: T::AccountId) -> DispatchResult {
273        // --- 0. Get the unique identifier (UID) for the root network.
274        let root_netuid: u16 = Self::get_root_netuid();
275        ensure!(
276            Self::if_subnet_exist(root_netuid),
277            Error::<T>::RootNetworkDoesNotExist
278        );
279
280        // --- 1. Ensure that the call originates from a signed source and retrieve the caller's account ID (coldkey).
281        let coldkey = ensure_signed(origin)?;
282        log::debug!(
283            "do_root_register( coldkey: {:?}, hotkey: {:?} )",
284            coldkey,
285            hotkey
286        );
287
288        // --- 2. Check if the hotkey is already registered to the root network. If not, error out.
289        ensure!(
290            Uids::<T>::contains_key(root_netuid, &hotkey),
291            Error::<T>::HotKeyNotRegisteredInSubNet
292        );
293
294        // --- 3. Create a network account for the user if it doesn't exist.
295        Self::create_account_if_non_existent(&coldkey, &hotkey);
296
297        // --- 4. Join the Senate if eligible.
298        // Returns the replaced member, if any.
299        let replaced = Self::join_senate_if_eligible(&hotkey)?;
300
301        if replaced.is_none() {
302            // Not eligible to join the Senate, or no replacement needed.
303            // Check if the hotkey is *now* a member of the Senate.
304            // Otherwise, error out.
305            ensure!(
306                T::SenateMembers::is_member(&hotkey),
307                Error::<T>::StakeTooLowForRoot, // Had less stake than the lowest stake incumbent.
308            );
309        }
310
311        // --- 5. Log and announce the successful Senate adjustment.
312        log::debug!(
313            "SenateAdjusted(old_hotkey:{:?} hotkey:{:?})",
314            replaced,
315            hotkey
316        );
317        Self::deposit_event(Event::SenateAdjusted {
318            old_member: replaced.cloned(),
319            new_member: hotkey,
320        });
321
322        // --- 6. Finish and return success.
323        Ok(())
324    }

326    // Checks if a hotkey should be a member of the Senate, and if so, adds them.
327    //
328    // # Arguments:
329    // * 'hotkey': The hotkey that the user wants to register to the root network.
330    //
331    // # Returns:
332    // * 'Result<Option<&T::AccountId>, Error<T>>': A result containing the replaced member, if any.
333    //
334    fn join_senate_if_eligible(hotkey: &T::AccountId) -> Result<Option<&T::AccountId>, Error<T>> {
335        // Get the root network UID.
336        let root_netuid: u16 = Self::get_root_netuid();
337
338        // --- 1. Check the hotkey is registered in the root network.
339        ensure!(
340            Uids::<T>::contains_key(root_netuid, hotkey),
341            Error::<T>::HotKeyNotRegisteredInSubNet
342        );
343
344        // --- 2. Verify the hotkey is NOT already a member of the Senate.
345        ensure!(
346            !T::SenateMembers::is_member(hotkey),
347            Error::<T>::HotKeyAlreadyRegisteredInSubNet
348        );
349
350        // --- 3. Grab the hotkey's stake.
351        let current_stake = Self::get_stake_for_hotkey_on_subnet(hotkey, Self::get_root_netuid());
352
353        // Add the hotkey to the Senate.
354        // If we're full, we'll swap out the lowest stake member.
355        let members = T::SenateMembers::members();
356        let last: Option<&T::AccountId> = None;
357        if (members.len() as u32) == T::SenateMembers::max_members() {
358            let mut sorted_members = members.clone();
359            sorted_members.sort_by(|a, b| {
360                let a_stake = Self::get_stake_for_hotkey_on_subnet(a, Self::get_root_netuid());
361                let b_stake = Self::get_stake_for_hotkey_on_subnet(b, Self::get_root_netuid());
362
363                b_stake.cmp(&a_stake)
364            });
365
366            if let Some(last) = sorted_members.last() {
367                let last_stake =
368                    Self::get_stake_for_hotkey_on_subnet(last, Self::get_root_netuid());
369
370                if last_stake < current_stake {
371                    // Swap the member with the lowest stake.
372                    T::SenateMembers::swap_member(last, hotkey)
373                        .map_err(|_| Error::<T>::CouldNotJoinSenate)?;
374                }
375            }
376        } else {
377            T::SenateMembers::add_member(hotkey).map_err(|_| Error::<T>::CouldNotJoinSenate)?;
378        }
379
380        // Return the swapped out member, if any.
381        Ok(last)
382    }

384    pub fn do_vote_root(
385        origin: T::RuntimeOrigin,
386        hotkey: &T::AccountId,
387        proposal: T::Hash,
388        index: u32,
389        approve: bool,
390    ) -> DispatchResultWithPostInfo {
391        // --- 1. Ensure that the caller has signed with their coldkey.
392        let coldkey = ensure_signed(origin.clone())?;
393
394        // --- 2. Ensure that the calling coldkey owns the associated hotkey.
395        ensure!(
396            Self::coldkey_owns_hotkey(&coldkey, hotkey),
397            Error::<T>::NonAssociatedColdKey
398        );
399
400        // --- 3. Ensure that the calling hotkey is a member of the senate.
401        ensure!(
402            T::SenateMembers::is_member(hotkey),
403            Error::<T>::NotSenateMember
404        );
405
406        // --- 4. Detects first vote of the member in the motion
407        let is_account_voting_first_time =
408            T::TriumvirateInterface::add_vote(hotkey, proposal, index, approve)?;
409
410        // --- 5. Calculate extrinsic weight
411        let members = T::SenateMembers::members();
412        let member_count = members.len() as u32;
413        let vote_weight = Weight::from_parts(20_528_275, 4980)
414            .saturating_add(Weight::from_parts(48_856, 0).saturating_mul(member_count.into()))
415            .saturating_add(T::DbWeight::get().reads(2_u64))
416            .saturating_add(T::DbWeight::get().writes(1_u64))
417            .saturating_add(Weight::from_parts(0, 128).saturating_mul(member_count.into()));
418
419        Ok((
420            Some(vote_weight),
421            if is_account_voting_first_time {
422                Pays::No
423            } else {
424                Pays::Yes
425            },
426        )
427            .into())
428    }

430    /// Facilitates the removal of a user's subnetwork.
431    ///
432    /// # Args:
433    /// * 'origin': ('T::RuntimeOrigin'): The calling origin. Must be signed.
434    /// * 'netuid': ('u16'): The unique identifier of the network to be removed.
435    ///
436    /// # Event:
437    /// * 'NetworkRemoved': Emitted when a network is successfully removed.
438    ///
439    /// # Raises:
440    /// * 'SubNetworkDoesNotExist': If the specified network does not exist.
441    /// * 'NotSubnetOwner': If the caller does not own the specified subnet.
442    ///
443    pub fn user_remove_network(coldkey: T::AccountId, netuid: u16) -> dispatch::DispatchResult {
444        // --- 1. Ensure this subnet exists.
445        ensure!(
446            Self::if_subnet_exist(netuid),
447            Error::<T>::SubNetworkDoesNotExist
448        );
449
450        // --- 2. Ensure the caller owns this subnet.
451        ensure!(
452            SubnetOwner::<T>::get(netuid) == coldkey,
453            Error::<T>::NotSubnetOwner
454        );
455
456        // --- 4. Remove the subnet identity if it exists.
457        if SubnetIdentitiesV2::<T>::take(netuid).is_some() {
458            Self::deposit_event(Event::SubnetIdentityRemoved(netuid));
459        }
460
461        // --- 5. Explicitly erase the network and all its parameters.
462        Self::remove_network(netuid);
463
464        // --- 6. Emit the NetworkRemoved event.
465        log::debug!("NetworkRemoved( netuid:{:?} )", netuid);
466        Self::deposit_event(Event::NetworkRemoved(netuid));
467
468        // --- 7. Return success.
469        Ok(())
470    }

472    /// Removes a network (identified by netuid) and all associated parameters.
473    ///
474    /// This function is responsible for cleaning up all the data associated with a network.
475    /// It ensures that all the storage values related to the network are removed, any
476    /// reserved balance is returned to the network owner, and the subnet identity is removed if it exists.
477    ///
478    /// # Args:
479    ///  * 'netuid': ('u16'): The unique identifier of the network to be removed.
480    ///
481    /// # Note:
482    /// This function does not emit any events, nor does it raise any errors. It silently
483    /// returns if any internal checks fail.
484    pub fn remove_network(netuid: u16) {
485        // --- 1. Return balance to subnet owner.
486        let owner_coldkey: T::AccountId = SubnetOwner::<T>::get(netuid);
487        let reserved_amount: u64 = Self::get_subnet_locked_balance(netuid);
488
489        // --- 2. Remove network count.
490        SubnetworkN::<T>::remove(netuid);
491
492        // --- 3. Remove network modality storage.
493        NetworkModality::<T>::remove(netuid);
494
495        // --- 4. Remove netuid from added networks.
496        NetworksAdded::<T>::remove(netuid);
497
498        // --- 5. Decrement the network counter.
499        TotalNetworks::<T>::mutate(|n: &mut u16| *n = n.saturating_sub(1));
500
501        // --- 6. Remove various network-related storages.
502        NetworkRegisteredAt::<T>::remove(netuid);
503
504        // --- 7. Remove incentive mechanism memory.
505        let _ = Uids::<T>::clear_prefix(netuid, u32::MAX, None);
506        let keys = Keys::<T>::iter_prefix(netuid).collect::<Vec<_>>();
507        let _ = Keys::<T>::clear_prefix(netuid, u32::MAX, None);
508        let _ = Bonds::<T>::clear_prefix(netuid, u32::MAX, None);
509
510        // --- 8. Removes the weights for this subnet (do not remove).
511        let _ = Weights::<T>::clear_prefix(netuid, u32::MAX, None);
512
513        // --- 9. Iterate over stored weights and fill the matrix.
514        for (uid_i, weights_i) in
515            <Weights<T> as IterableStorageDoubleMap<u16, u16, Vec<(u16, u16)>>>::iter_prefix(
516                Self::get_root_netuid(),
517            )
518        {
519            // Create a new vector to hold modified weights.
520            let mut modified_weights: Vec<(u16, u16)> = weights_i.clone();
521            // Iterate over each weight entry to potentially update it.
522            for (subnet_id, weight) in modified_weights.iter_mut() {
523                if subnet_id == &netuid {
524                    // If the condition matches, modify the weight
525                    *weight = 0; // Set weight to 0 for the matching subnet_id.
526                }
527            }
528            Weights::<T>::insert(Self::get_root_netuid(), uid_i, modified_weights);
529        }
530
531        // --- 10. Remove various network-related parameters.
532        Rank::<T>::remove(netuid);
533        Trust::<T>::remove(netuid);
534        Active::<T>::remove(netuid);
535        Emission::<T>::remove(netuid);
536        Incentive::<T>::remove(netuid);
537        Consensus::<T>::remove(netuid);
538        Dividends::<T>::remove(netuid);
539        PruningScores::<T>::remove(netuid);
540        LastUpdate::<T>::remove(netuid);
541        ValidatorPermit::<T>::remove(netuid);
542        ValidatorTrust::<T>::remove(netuid);
543
544        for (_uid, key) in keys {
545            IsNetworkMember::<T>::remove(key, netuid);
546        }
547
548        // --- 11. Erase network parameters.
549        Tempo::<T>::remove(netuid);
550        Kappa::<T>::remove(netuid);
551        Difficulty::<T>::remove(netuid);
552        MaxAllowedUids::<T>::remove(netuid);
553        ImmunityPeriod::<T>::remove(netuid);
554        ActivityCutoff::<T>::remove(netuid);
555        MaxWeightsLimit::<T>::remove(netuid);
556        MinAllowedWeights::<T>::remove(netuid);
557        RegistrationsThisInterval::<T>::remove(netuid);
558        POWRegistrationsThisInterval::<T>::remove(netuid);
559        BurnRegistrationsThisInterval::<T>::remove(netuid);
560
561        // --- 12. Add the balance back to the owner.
562        Self::add_balance_to_coldkey_account(&owner_coldkey, reserved_amount);
563        Self::set_subnet_locked_balance(netuid, 0);
564        SubnetOwner::<T>::remove(netuid);
565
566        // --- 13. Remove subnet identity if it exists.
567        if SubnetIdentitiesV2::<T>::contains_key(netuid) {
568            SubnetIdentitiesV2::<T>::remove(netuid);
569            Self::deposit_event(Event::SubnetIdentityRemoved(netuid));
570        }
571    }

573    #[allow(clippy::arithmetic_side_effects)]
574    /// This function calculates the lock cost for a network based on the last lock amount, minimum lock cost, last lock block, and current block.
575    /// The lock cost is calculated using the formula:
576    /// lock_cost = (last_lock * mult) - (last_lock / lock_reduction_interval) * (current_block - last_lock_block)
577    /// where:
578    /// - last_lock is the last lock amount for the network
579    /// - mult is the multiplier which increases lock cost each time a registration occurs
580    /// - last_lock_block is the block number at which the last lock occurred
581    /// - lock_reduction_interval the number of blocks before the lock returns to previous value.
582    /// - current_block is the current block number
583    /// - DAYS is the number of blocks in a day
584    /// - min_lock is the minimum lock cost for the network
585    ///
586    /// If the calculated lock cost is less than the minimum lock cost, the minimum lock cost is returned.
587    ///
588    /// # Returns:
589    ///  * 'u64':
590    ///     - The lock cost for the network.
591    ///
592    pub fn get_network_lock_cost() -> u64 {
593        let last_lock = Self::get_network_last_lock();
594        let min_lock = Self::get_network_min_lock();
595        let last_lock_block = Self::get_network_last_lock_block();
596        let current_block = Self::get_current_block_as_u64();
597        let lock_reduction_interval = Self::get_lock_reduction_interval();
598        let mult = if last_lock_block == 0 { 1 } else { 2 };
599
600        let mut lock_cost = last_lock.saturating_mul(mult).saturating_sub(
601            last_lock
602                .safe_div(lock_reduction_interval)
603                .saturating_mul(current_block.saturating_sub(last_lock_block)),
604        );
605
606        if lock_cost < min_lock {
607            lock_cost = min_lock;
608        }
609
610        log::debug!(
611            "last_lock: {:?}, min_lock: {:?}, last_lock_block: {:?}, lock_reduction_interval: {:?}, current_block: {:?}, mult: {:?} lock_cost: {:?}",
612            last_lock,
613            min_lock,
614            last_lock_block,
615            lock_reduction_interval,
616            current_block,
617            mult,
618            lock_cost
619        );
620
621        lock_cost
622    }

657    pub fn get_lock_reduction_interval() -> u64 {
658        let interval: I64F64 =
659            I64F64::saturating_from_num(NetworkLockReductionInterval::<T>::get());
660        let block_emission: I64F64 =
661            I64F64::saturating_from_num(Self::get_block_emission().unwrap_or(1_000_000_000));
662        let halving: I64F64 = block_emission
663            .checked_div(I64F64::saturating_from_num(1_000_000_000))
664            .unwrap_or(I64F64::saturating_from_num(0.0));
665        let halved_interval: I64F64 = interval.saturating_mul(halving);
666        halved_interval.saturating_to_num::<u64>()
667    }
