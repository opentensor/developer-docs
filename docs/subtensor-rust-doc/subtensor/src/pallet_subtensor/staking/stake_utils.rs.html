24    /// Calculates the price of alpha for a given subnet.
25    ///
26    /// This function determines the price of alpha by dividing the total TAO
27    /// reserves by the total alpha reserves (`SubnetAlphaIn`) for the specified subnet.
28    /// If the alpha reserves are zero, the function returns zero to avoid division by zero.
29    ///
30    /// # Arguments
31    /// * `netuid` - The unique identifier of the subnet.
32    ///
33    /// # Returns
34    /// * `I96F32` - The price of alpha for the specified subnet.
35    pub fn get_alpha_price(netuid: u16) -> U96F32 {
36        if netuid == Self::get_root_netuid() {
37            return U96F32::saturating_from_num(1.0); // Root.
38        }
39        if SubnetMechanism::<T>::get(netuid) == 0 {
40            return U96F32::saturating_from_num(1.0); // Stable
41        }
42        if SubnetAlphaIn::<T>::get(netuid) == 0 {
43            U96F32::saturating_from_num(0)
44        } else {
45            U96F32::saturating_from_num(SubnetTAO::<T>::get(netuid))
46                .checked_div(U96F32::saturating_from_num(SubnetAlphaIn::<T>::get(netuid)))
47                .unwrap_or(U96F32::saturating_from_num(0))
48        }
49    }

63    pub fn update_moving_price(netuid: u16) {
64        let blocks_since_start_call = U96F32::saturating_from_num({
65            // We expect FirstEmissionBlockNumber to be set earlier, and we take the block when
66            // `start_call` was called (first block before FirstEmissionBlockNumber).
67            let start_call_block = FirstEmissionBlockNumber::<T>::get(netuid)
68                .unwrap_or_default()
69                .saturating_sub(1);
70
71            Self::get_current_block_as_u64().saturating_sub(start_call_block)
72        });
73
74        // Use halving time hyperparameter. The meaning of this parameter can be best explained under
75        // the assumption of a constant price and SubnetMovingAlpha == 0.5: It is how many blocks it
76        // will take in order for the distance between current EMA of price and current price to shorten
77        // by half.
78        let halving_time = EMAPriceHalvingBlocks::<T>::get(netuid);
79        let current_ma_unsigned = U96F32::saturating_from_num(SubnetMovingAlpha::<T>::get());
80        let alpha: U96F32 = current_ma_unsigned.saturating_mul(blocks_since_start_call.safe_div(
81            blocks_since_start_call.saturating_add(U96F32::saturating_from_num(halving_time)),
82        ));
83        // Because alpha = b / (b + h), where b and h > 0, alpha < 1, so 1 - alpha > 0.
84        // We can use unsigned type here: U96F32
85        let one_minus_alpha: U96F32 = U96F32::saturating_from_num(1.0).saturating_sub(alpha);
86        let current_price: U96F32 = alpha
87            .saturating_mul(Self::get_alpha_price(netuid).min(U96F32::saturating_from_num(1.0)));
88        let current_moving: U96F32 =
89            one_minus_alpha.saturating_mul(Self::get_moving_alpha_price(netuid));
90        // Convert batch to signed I96F32 to avoid migration of SubnetMovingPrice for now``
91        let new_moving: I96F32 =
92            I96F32::saturating_from_num(current_price.saturating_add(current_moving));
93        SubnetMovingPrice::<T>::insert(netuid, new_moving);
94    }

96    /// Retrieves the global global weight as a normalized value between 0 and 1.
97    ///
98    /// This function performs the following steps:
99    /// 1. Fetches the global weight from storage using the TaoWeight storage item.
100    /// 2. Converts the retrieved u64 value to a fixed-point number (U96F32).
101    /// 3. Normalizes the weight by dividing it by the maximum possible u64 value.
102    /// 4. Returns the normalized weight as an U96F32 fixed-point number.
103    ///
104    /// The normalization ensures that the returned value is always between 0 and 1,
105    /// regardless of the actual stored weight value.
106    ///
107    /// # Returns
108    /// * `U96F32` - The normalized global global weight as a fixed-point number between 0 and 1.
109    ///
110    /// # Note
111    /// This function uses saturating division to prevent potential overflow errors.
112    pub fn get_tao_weight() -> U96F32 {
113        // Step 1: Fetch the global weight from storage
114        let stored_weight = TaoWeight::<T>::get();
115
116        // Step 2: Convert the u64 weight to U96F32
117        let weight_fixed = U96F32::saturating_from_num(stored_weight);
118
119        // Step 3: Normalize the weight by dividing by u64::MAX
120        // This ensures the result is always between 0 and 1
121        weight_fixed.safe_div(U96F32::saturating_from_num(u64::MAX))
122    }

145    /// Calculates the weighted combination of alpha and global tao for a single hotkey onet a subnet.
146    ///
147    pub fn get_stake_weights_for_hotkey_on_subnet(
148        hotkey: &T::AccountId,
149        netuid: u16,
150    ) -> (I64F64, I64F64, I64F64) {
151        // Retrieve the global tao weight.
152        let tao_weight = I64F64::saturating_from_num(Self::get_tao_weight());
153        log::debug!("tao_weight: {:?}", tao_weight);
154
155        // Step 1: Get stake of hotkey (neuron)
156        let alpha_stake =
157            I64F64::saturating_from_num(Self::get_inherited_for_hotkey_on_subnet(hotkey, netuid));
158        log::debug!("alpha_stake: {:?}", alpha_stake);
159
160        // Step 2: Get the global tao stake for the hotkey
161        let tao_stake = I64F64::saturating_from_num(Self::get_tao_inherited_for_hotkey_on_subnet(
162            hotkey, netuid,
163        ));
164        log::debug!("tao_stake: {:?}", tao_stake);
165
166        // Step 3: Combine alpha and tao stakes
167        let total_stake = alpha_stake.saturating_add(tao_stake.saturating_mul(tao_weight));
168        log::debug!("total_stake: {:?}", total_stake);
169
170        (total_stake, alpha_stake, tao_stake)
171    }

173    /// Calculates the weighted combination of alpha and global tao for hotkeys on a subnet.
174    ///
175    pub fn get_stake_weights_for_network(netuid: u16) -> (Vec<I64F64>, Vec<I64F64>, Vec<I64F64>) {
176        // Retrieve the global tao weight.
177        let tao_weight: I64F64 = I64F64::saturating_from_num(Self::get_tao_weight());
178        log::debug!("tao_weight: {:?}", tao_weight);
179
180        // Step 1: Get subnetwork size
181        let n: u16 = Self::get_subnetwork_n(netuid);
182
183        // Step 2: Get stake of all hotkeys (neurons) ordered by uid
184        let alpha_stake: Vec<I64F64> = (0..n)
185            .map(|uid| {
186                if Keys::<T>::contains_key(netuid, uid) {
187                    let hotkey: T::AccountId = Keys::<T>::get(netuid, uid);
188                    I64F64::saturating_from_num(Self::get_inherited_for_hotkey_on_subnet(
189                        &hotkey, netuid,
190                    ))
191                } else {
192                    I64F64::saturating_from_num(0)
193                }
194            })
195            .collect();
196        log::debug!("alpha_stake: {:?}", alpha_stake);
197
198        // Step 3: Calculate the global tao stake vector.
199        // Initialize a vector to store global tao stakes for each neuron.
200        let tao_stake: Vec<I64F64> = (0..n)
201            .map(|uid| {
202                if Keys::<T>::contains_key(netuid, uid) {
203                    let hotkey: T::AccountId = Keys::<T>::get(netuid, uid);
204                    I64F64::saturating_from_num(Self::get_tao_inherited_for_hotkey_on_subnet(
205                        &hotkey, netuid,
206                    ))
207                } else {
208                    I64F64::saturating_from_num(0)
209                }
210            })
211            .collect();
212        log::trace!("tao_stake: {:?}", tao_stake);
213
214        // Step 4: Combine alpha and root tao stakes.
215        // Calculate the weighted average of alpha and global tao stakes for each neuron.
216        let total_stake: Vec<I64F64> = alpha_stake
217            .iter()
218            .zip(tao_stake.iter())
219            .map(|(alpha_i, tao_i)| alpha_i.saturating_add(tao_i.saturating_mul(tao_weight)))
220            .collect();
221        log::trace!("total_stake: {:?}", total_stake);
222
223        (total_stake, alpha_stake, tao_stake)
224    }

226    /// Calculates the total inherited stake (alpha) held by a hotkey on a network, considering child/parent relationships.
227    ///
228    /// This function performs the following steps:
229    /// 1. Retrieves the initial alpha (stake) for the hotkey on the specified subnet.
230    /// 2. Retrieves the list of children and parents for the hotkey on the subnet.
231    /// 3. Calculates the alpha allocated to children:
232    ///    a. For each child, computes the proportion of alpha to be allocated.
233    ///    b. Accumulates the total alpha allocated to all children.
234    /// 4. Calculates the alpha received from parents:
235    ///    a. For each parent, retrieves the parent's stake on the subnet.
236    ///    b. Computes the proportion of the parent's stake to be inherited.
237    ///    c. Accumulates the total alpha inherited from all parents.
238    /// 5. Computes the final inherited alpha by adjusting the initial alpha:
239    ///    a. Subtracts the alpha allocated to children.
240    ///    b. Adds the alpha inherited from parents.
241    /// 6. Returns the final inherited alpha value.
242    ///
243    /// # Arguments
244    /// * `hotkey` - AccountId of the hotkey whose total inherited stake is to be calculated.
245    /// * `netuid` - Network unique identifier specifying the subnet context.
246    ///
247    /// # Returns
248    /// * `u64`: The total inherited alpha for the hotkey on the subnet after considering the
249    ///   stakes allocated to children and inherited from parents.
250    ///
251    /// # Note
252    /// This function uses saturating arithmetic to prevent overflows.
253    pub fn get_tao_inherited_for_hotkey_on_subnet(hotkey: &T::AccountId, netuid: u16) -> u64 {
254        let initial_tao: U96F32 = U96F32::saturating_from_num(
255            Self::get_stake_for_hotkey_on_subnet(hotkey, Self::get_root_netuid()),
256        );
257
258        // Initialize variables to track alpha allocated to children and inherited from parents.
259        let mut tao_to_children: U96F32 = U96F32::saturating_from_num(0);
260        let mut tao_from_parents: U96F32 = U96F32::saturating_from_num(0);
261
262        // Step 2: Retrieve the lists of parents and children for the hotkey on the subnet.
263        let parents: Vec<(u64, T::AccountId)> = Self::get_parents(hotkey, netuid);
264        let children: Vec<(u64, T::AccountId)> = Self::get_children(hotkey, netuid);
265        log::trace!(
266            "Parents for hotkey {:?} on subnet {}: {:?}",
267            hotkey,
268            netuid,
269            parents
270        );
271        log::trace!(
272            "Children for hotkey {:?} on subnet {}: {:?}",
273            hotkey,
274            netuid,
275            children
276        );
277
278        // Step 3: Calculate the total tao allocated to children.
279        for (proportion, _) in children {
280            // Convert the proportion to a normalized value between 0 and 1.
281            let normalized_proportion: U96F32 = U96F32::saturating_from_num(proportion)
282                .safe_div(U96F32::saturating_from_num(u64::MAX));
283            log::trace!(
284                "Normalized proportion for child: {:?}",
285                normalized_proportion
286            );
287
288            // Calculate the amount of tao to be allocated to this child.
289            let tao_proportion_to_child: U96F32 =
290                U96F32::saturating_from_num(initial_tao).saturating_mul(normalized_proportion);
291            log::trace!("Tao proportion to child: {:?}", tao_proportion_to_child);
292
293            // Add this child's allocation to the total tao allocated to children.
294            tao_to_children = tao_to_children.saturating_add(tao_proportion_to_child);
295        }
296        log::trace!("Total tao allocated to children: {:?}", tao_to_children);
297
298        // Step 4: Calculate the total tao inherited from parents.
299        for (proportion, parent) in parents {
300            // Retrieve the parent's total stake on this subnet.
301            let parent_tao: U96F32 = U96F32::saturating_from_num(
302                Self::get_stake_for_hotkey_on_subnet(&parent, Self::get_root_netuid()),
303            );
304            log::trace!(
305                "Parent tao for parent {:?} on subnet {}: {:?}",
306                parent,
307                netuid,
308                parent_tao
309            );
310
311            // Convert the proportion to a normalized value between 0 and 1.
312            let normalized_proportion: U96F32 = U96F32::saturating_from_num(proportion)
313                .safe_div(U96F32::saturating_from_num(u64::MAX));
314            log::trace!(
315                "Normalized proportion from parent: {:?}",
316                normalized_proportion
317            );
318
319            // Calculate the amount of tao to be inherited from this parent.
320            let tao_proportion_from_parent: U96F32 =
321                U96F32::saturating_from_num(parent_tao).saturating_mul(normalized_proportion);
322            log::trace!(
323                "Tao proportion from parent: {:?}",
324                tao_proportion_from_parent
325            );
326
327            // Add this parent's contribution to the total tao inherited from parents.
328            tao_from_parents = tao_from_parents.saturating_add(tao_proportion_from_parent);
329        }
330        log::trace!("Total tao inherited from parents: {:?}", tao_from_parents);
331
332        // Step 5: Calculate the final inherited tao for the hotkey.
333        let finalized_tao: U96F32 = initial_tao
334            .saturating_sub(tao_to_children) // Subtract tao allocated to children
335            .saturating_add(tao_from_parents); // Add tao inherited from parents
336        log::trace!(
337            "Finalized tao for hotkey {:?} on subnet {}: {:?}",
338            hotkey,
339            netuid,
340            finalized_tao
341        );
342
343        // Step 6: Return the final inherited tao value.
344        finalized_tao.saturating_to_num::<u64>()
345    }

348        // Step 1: Retrieve the initial total stake (alpha) for the hotkey on the specified subnet.

375            "Children for hotkey {:?} on subnet {}: {:?}",

405                U96F32::saturating_from_num(Self::get_stake_for_hotkey_on_subnet(&parent, netuid));

417                "Normalized proportion from parent: {:?}",
418                normalized_proportion

424            log::trace!(
425                "Alpha proportion from parent: {:?}",
426                alpha_proportion_from_parent
427            );
428
429            // Add this parent's contribution to the total alpha inherited from parents.

435        );

444            netuid,

446        );

449        finalized_alpha.saturating_to_num::<u64>()
450    }

633    /// Calculates Some(Alpha) returned from pool by staking operation
634    /// if liquidity allows that. If not, returns None.
635    ///
636    /// If new alpha_reserve is about to drop below DefaultMinimumPoolLiquidity,
637    /// then don't do it.
638    ///
639    pub fn sim_swap_tao_for_alpha(netuid: u16, tao: u64) -> Option<u64> {
640        // Step 1: Get the mechanism type for the subnet (0 for Stable, 1 for Dynamic)
641        let mechanism_id: u16 = SubnetMechanism::<T>::get(netuid);
642        // Step 2: Initialized vars.
643        if mechanism_id == 1 {
644            // Step 3.a.1: Dynamic mechanism calculations
645            let tao_reserves: U110F18 = U110F18::saturating_from_num(SubnetTAO::<T>::get(netuid));
646            let alpha_reserves: U110F18 =
647                U110F18::saturating_from_num(SubnetAlphaIn::<T>::get(netuid));
648            // Step 3.a.2: Compute constant product k = alpha * tao
649            let k: U110F18 = alpha_reserves.saturating_mul(tao_reserves);
650
651            // Calculate new alpha reserve
652            let new_alpha_reserves: U110F18 =
653                k.safe_div(tao_reserves.saturating_add(U110F18::saturating_from_num(tao)));
654
655            // Step 3.a.3: Calculate alpha staked using the constant product formula
656            // alpha_stake_recieved = current_alpha - (k / (current_tao + new_tao))
657            if new_alpha_reserves >= DefaultMinimumPoolLiquidity::<T>::get() {
658                Some(
659                    alpha_reserves
660                        .saturating_sub(new_alpha_reserves)
661                        .saturating_to_num::<u64>(),
662                )
663            } else {
664                None
665            }
666        } else {
667            // Step 3.b.1: Stable mechanism, just return the value 1:1
668            Some(tao)
669        }
670    }

672    /// Calculates Some(Tao) returned from pool by unstaking operation
673    /// if liquidity allows that. If not, returns None.
674    ///
675    /// If new tao_reserve is about to drop below DefaultMinimumPoolLiquidity,
676    /// then don't do it.
677    ///
678    pub fn sim_swap_alpha_for_tao(netuid: u16, alpha: u64) -> Option<u64> {
679        // Step 1: Get the mechanism type for the subnet (0 for Stable, 1 for Dynamic)
680        let mechanism_id: u16 = SubnetMechanism::<T>::get(netuid);
681        // Step 2: Swap alpha and attain tao
682        if mechanism_id == 1 {
683            // Step 3.a.1: Dynamic mechanism calculations
684            let tao_reserves: U110F18 = U110F18::saturating_from_num(SubnetTAO::<T>::get(netuid));
685            let alpha_reserves: U110F18 =
686                U110F18::saturating_from_num(SubnetAlphaIn::<T>::get(netuid));
687            // Step 3.a.2: Compute constant product k = alpha * tao
688            let k: U110F18 = alpha_reserves.saturating_mul(tao_reserves);
689
690            // Calculate new tao reserve
691            let new_tao_reserves: U110F18 = k
692                .checked_div(alpha_reserves.saturating_add(U110F18::saturating_from_num(alpha)))
693                .unwrap_or(U110F18::saturating_from_num(0));
694
695            // Step 3.a.3: Calculate alpha staked using the constant product formula
696            // tao_recieved = tao_reserves - (k / (alpha_reserves + new_tao))
697            if new_tao_reserves >= DefaultMinimumPoolLiquidity::<T>::get() {
698                Some(
699                    tao_reserves
700                        .saturating_sub(new_tao_reserves)
701                        .saturating_to_num::<u64>(),
702                )
703            } else {
704                None
705            }
706        } else {
707            // Step 3.b.1: Stable mechanism, just return the value 1:1
708            Some(alpha)
709        }
710    }

713    ///
714    /// Updates TaoIn, AlphaIn, and AlphaOut

734            SubnetVolume::<T>::mutate(netuid, |total| {
735                *total = total.saturating_add(tao.into());
736            });
737            // Step 9. Return the alpha received.
738            alpha
739        } else {
740            0
741        }

936        );

1034    /// Validate stake transition user input

1078            Self::hotkey_account_exists(origin_hotkey),
1079            Error::<T>::HotKeyAccountNotExists
1080        );
1081
1082        // Ensure that the destination hotkey account exists
1083        ensure!(
1084            Self::hotkey_account_exists(destination_hotkey),
1085            Error::<T>::HotKeyAccountNotExists
1086        );
1087

1090            origin_hotkey,
1091            origin_coldkey,
1092            origin_netuid,
1093        );
1094        ensure!(
1095            alpha_amount <= origin_alpha,
1096            Error::<T>::NotEnoughStakeToWithdraw
1097        );
1098
1099        // Ensure that the stake amount to be removed is above the minimum in tao equivalent.
1100        let tao_equivalent_result = Self::sim_swap_alpha_for_tao(origin_netuid, alpha_amount);
1101        if let Some(tao_equivalent) = tao_equivalent_result {
1102            ensure!(
1103                tao_equivalent > DefaultMinStake::<T>::get(),
1104                Error::<T>::AmountTooLow

1106        } else {
1107            return Err(Error::<T>::InsufficientLiquidity);
1108        }

1111        // slippage over desired

1113            if !allow_partial {
1114                ensure!(alpha_amount <= max_amount, Error::<T>::SlippageTooHigh);
1115            }
1116        }
1117
1118        let expected_alpha =
1119            Self::sim_swap_tao_for_alpha(destination_netuid, tao_equivalent_result.unwrap_or(0))
1120                .unwrap_or(0);
1121
1122        // Ensure that the amount being staked to the new hotkey is precise enough

1145    pub(crate) fn calculate_staking_fee(
1146        origin: Option<(&T::AccountId, u16)>,
1147        _origin_coldkey: &T::AccountId,
1148        destination: Option<(&T::AccountId, u16)>,
1149        _destination_coldkey: &T::AccountId,
1150        alpha_estimate: U96F32,
1151    ) -> u64 {
1152        match origin {
1153            // If origin is defined, we are removing/moving stake
1154            Some((origin_hotkey, origin_netuid)) => {
1155                if let Some((_destination_hotkey, destination_netuid)) = destination {
1156                    // This is a stake move/swap/transfer
1157                    if destination_netuid == origin_netuid {
1158                        // If destination is on the same subnet, use the default fee
1159                        return DefaultStakingFee::<T>::get();
1160                    }
1161                }
1162
1163                if origin_netuid == Self::get_root_netuid()
1164                    || SubnetMechanism::<T>::get(origin_netuid) == 0
1165                {
1166                    // If the origin netuid is root, or the subnet mechanism is 0, use the default fee
1167                    DefaultStakingFee::<T>::get()
1168                } else {
1169                    // Otherwise, calculate the fee based on the alpha estimate
1170                    // Here we are using TotalHotkeyAlphaLastEpoch, which is exactly the value that
1171                    // was used to calculate AlphaDividendsPerSubnet
1172                    let tao_estimate = U96F32::saturating_from_num(
1173                        Self::sim_swap_alpha_for_tao(
1174                            origin_netuid,
1175                            alpha_estimate.saturating_to_num::<u64>(),
1176                        )
1177                        .unwrap_or(0),
1178                    );
1179                    let mut fee = tao_estimate
1180                        .saturating_mul(
1181                            U96F32::saturating_from_num(AlphaDividendsPerSubnet::<T>::get(
1182                                origin_netuid,
1183                                &origin_hotkey,
1184                            ))
1185                            .safe_div(U96F32::saturating_from_num(
1186                                TotalHotkeyAlphaLastEpoch::<T>::get(&origin_hotkey, origin_netuid),
1187                            )),
1188                        )
1189                        .saturating_to_num::<u64>();
1190
1191                    // 0.005% per epoch matches to 44% annual in compound interest. Do not allow the fee
1192                    // to be lower than that. (1.00005^(365*20) ~= 1.44)
1193                    let apr_20_percent = U96F32::saturating_from_num(0.00005);
1194                    fee = fee.max(
1195                        tao_estimate
1196                            .saturating_mul(apr_20_percent)
1197                            .saturating_to_num::<u64>(),
1198                    );
1199
1200                    // We should at least get DefaultStakingFee anyway
1201                    fee.max(DefaultStakingFee::<T>::get())
1202                }
1203            }
1204            // If origin is not defined, we are adding stake; use default fee
1205            None => DefaultStakingFee::<T>::get(),
1206        }
1207    }
