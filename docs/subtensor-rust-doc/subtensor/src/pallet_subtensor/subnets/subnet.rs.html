68    /// Finds the next available mechanism ID.
69    ///
70    /// This function iterates through possible mechanism IDs starting from 0
71    /// until it finds an ID that is not currently in use.
72    ///
73    /// # Returns
74    /// * `u16` - The next available mechanism ID.
75    pub fn get_next_netuid() -> u16 {
76        let mut next_netuid = 1; // do not allow creation of root
77        let netuids: Vec<u16> = Self::get_all_subnet_netuids();
78        loop {
79            if !netuids.contains(&next_netuid) {
80                break next_netuid;
81            }
82            next_netuid = next_netuid.saturating_add(1);
83        }
84    }

112    /// Facilitates user registration of a new subnetwork.
113    ///
114    /// # Args:
115    /// * 'origin': ('T::RuntimeOrigin'): The calling origin. Must be signed.
116    /// * `identity` (`Option<SubnetIdentityOf>`): Optional identity to be associated with the new subnetwork.
117    ///
118    /// # Event:
119    /// * 'NetworkAdded': Emitted when a new network is successfully added.
120    ///
121    /// # Raises:
122    /// * 'TxRateLimitExceeded': If the rate limit for network registration is exceeded.
123    /// * 'NotEnoughBalanceToStake': If there isn't enough balance to stake for network registration.
124    /// * 'BalanceWithdrawalError': If an error occurs during balance withdrawal for network registration.
125    /// * `SubnetIdentitySet(netuid)`: Emitted when a custom identity is set for a new subnetwork.
126    /// * `SubnetIdentityRemoved(netuid)`: Emitted when the identity of a removed network is also deleted.
127    ///
128    pub fn do_register_network(
129        origin: T::RuntimeOrigin,
130        hotkey: &T::AccountId,
131        mechid: u16,
132        identity: Option<SubnetIdentityOfV2>,
133    ) -> DispatchResult {
134        // --- 1. Ensure the caller is a signed user.
135        let coldkey = ensure_signed(origin)?;
136
137        // --- 2. Ensure the hotkey does not exist or is owned by the coldkey.
138        ensure!(
139            !Self::hotkey_account_exists(hotkey) || Self::coldkey_owns_hotkey(&coldkey, hotkey),
140            Error::<T>::NonAssociatedColdKey
141        );
142
143        // --- 3. Ensure the mechanism is Dynamic.
144        ensure!(mechid == 1, Error::<T>::MechanismDoesNotExist);
145
146        // --- 4. Rate limit for network registrations.
147        let current_block = Self::get_current_block_as_u64();
148        let last_lock_block = Self::get_network_last_lock_block();
149        ensure!(
150            current_block.saturating_sub(last_lock_block) >= NetworkRateLimit::<T>::get(),
151            Error::<T>::NetworkTxRateLimitExceeded
152        );
153
154        // --- 5. Calculate and lock the required tokens.
155        let lock_amount: u64 = Self::get_network_lock_cost();
156        log::debug!("network lock_amount: {:?}", lock_amount);
157        ensure!(
158            Self::can_remove_balance_from_coldkey_account(&coldkey, lock_amount),
159            Error::<T>::NotEnoughBalanceToStake
160        );
161
162        // --- 5. Determine the netuid to register.
163        let netuid_to_register: u16 = Self::get_next_netuid();
164
165        // --- 6. Perform the lock operation.
166        let actual_tao_lock_amount: u64 =
167            Self::remove_balance_from_coldkey_account(&coldkey, lock_amount)?;
168        log::debug!("actual_tao_lock_amount: {:?}", actual_tao_lock_amount);
169
170        // --- 7. Set the lock amount for use to determine pricing.
171        Self::set_network_last_lock(actual_tao_lock_amount);
172
173        // --- 8. Set initial and custom parameters for the network.
174        let default_tempo = DefaultTempo::<T>::get();
175        Self::init_new_network(netuid_to_register, default_tempo);
176        log::debug!("init_new_network: {:?}", netuid_to_register);
177
178        // --- 9 . Add the caller to the neuron set.
179        Self::create_account_if_non_existent(&coldkey, hotkey);
180        Self::append_neuron(netuid_to_register, hotkey, current_block);
181        log::debug!(
182            "Appended neuron for netuid {:?}, hotkey: {:?}",
183            netuid_to_register,
184            hotkey
185        );
186
187        // --- 10. Set the mechanism.
188        SubnetMechanism::<T>::insert(netuid_to_register, mechid);
189        log::debug!(
190            "SubnetMechanism for netuid {:?} set to: {:?}",
191            netuid_to_register,
192            mechid
193        );
194
195        // --- 11. Set the creation terms.
196        NetworkLastRegistered::<T>::set(current_block);
197        NetworkRegisteredAt::<T>::insert(netuid_to_register, current_block);
198
199        // --- 14. Init the pool by putting the lock as the initial alpha.
200        TokenSymbol::<T>::insert(
201            netuid_to_register,
202            Self::get_symbol_for_subnet(netuid_to_register),
203        ); // Set subnet token symbol.
204
205        // Put initial TAO from lock into subnet TAO and produce numerically equal amount of Alpha
206        // The initial TAO is the locked amount, with a minimum of 1 RAO and a cap of 100 TAO.
207        let pool_initial_tao = Self::get_network_min_lock();
208        let actual_tao_lock_amount_less_pool_tao =
209            actual_tao_lock_amount.saturating_sub(pool_initial_tao);
210        SubnetTAO::<T>::insert(netuid_to_register, pool_initial_tao);
211        SubnetAlphaIn::<T>::insert(netuid_to_register, pool_initial_tao);
212        SubnetOwner::<T>::insert(netuid_to_register, coldkey.clone());
213        SubnetOwnerHotkey::<T>::insert(netuid_to_register, hotkey.clone());
214
215        if actual_tao_lock_amount_less_pool_tao > 0 {
216            Self::burn_tokens(actual_tao_lock_amount_less_pool_tao);
217        }
218
219        if actual_tao_lock_amount > 0 && pool_initial_tao > 0 {
220            // Record in TotalStake the initial TAO in the pool.
221            Self::increase_total_stake(pool_initial_tao);
222        }
223
224        // --- 15. Add the identity if it exists
225        if let Some(identity_value) = identity {
226            ensure!(
227                Self::is_valid_subnet_identity(&identity_value),
228                Error::<T>::InvalidIdentity
229            );
230
231            SubnetIdentitiesV2::<T>::insert(netuid_to_register, identity_value);
232            Self::deposit_event(Event::SubnetIdentitySet(netuid_to_register));
233        }
234
235        // --- 16. Emit the NetworkAdded event.
236        log::info!(
237            "NetworkAdded( netuid:{:?}, mechanism:{:?} )",
238            netuid_to_register,
239            mechid
240        );
241        Self::deposit_event(Event::NetworkAdded(netuid_to_register, mechid));
242
243        // --- 17. Return success.
244        Ok(())
245    }

247    /// Sets initial and custom parameters for a new network.
248    pub fn init_new_network(netuid: u16, tempo: u16) {
249        // --- 1. Set network to 0 size.
250        SubnetworkN::<T>::insert(netuid, 0);
251
252        // --- 2. Set this network uid to alive.
253        NetworksAdded::<T>::insert(netuid, true);
254
255        // --- 3. Fill tempo memory item.
256        Tempo::<T>::insert(netuid, tempo);
257
258        // --- 4 Fill modality item.
259        NetworkModality::<T>::insert(netuid, 0);
260
261        // --- 5. Increase total network count.
262        TotalNetworks::<T>::mutate(|n| *n = n.saturating_add(1));
263
264        // --- 6. Set all default values **explicitly**.
265        Self::set_network_registration_allowed(netuid, true);
266        Self::set_max_allowed_uids(netuid, 256);
267        Self::set_max_allowed_validators(netuid, 64);
268        Self::set_min_allowed_weights(netuid, 1);
269        Self::set_max_weight_limit(netuid, u16::MAX);
270        Self::set_adjustment_interval(netuid, 360);
271        Self::set_target_registrations_per_interval(netuid, 1);
272        Self::set_adjustment_alpha(netuid, 17_893_341_751_498_265_066); // 18_446_744_073_709_551_615 * 0.97 = 17_893_341_751_498_265_066
273        Self::set_immunity_period(netuid, 5000);
274        Self::set_min_difficulty(netuid, u64::MAX);
275        Self::set_max_difficulty(netuid, u64::MAX);
276
277        // Make network parameters explicit.
278        if !Tempo::<T>::contains_key(netuid) {
279            Tempo::<T>::insert(netuid, Tempo::<T>::get(netuid));
280        }
281        if !Kappa::<T>::contains_key(netuid) {
282            Kappa::<T>::insert(netuid, Kappa::<T>::get(netuid));
283        }
284        if !Difficulty::<T>::contains_key(netuid) {
285            Difficulty::<T>::insert(netuid, Difficulty::<T>::get(netuid));
286        }
287        if !MaxAllowedUids::<T>::contains_key(netuid) {
288            MaxAllowedUids::<T>::insert(netuid, MaxAllowedUids::<T>::get(netuid));
289        }
290        if !ImmunityPeriod::<T>::contains_key(netuid) {
291            ImmunityPeriod::<T>::insert(netuid, ImmunityPeriod::<T>::get(netuid));
292        }
293        if !ActivityCutoff::<T>::contains_key(netuid) {
294            ActivityCutoff::<T>::insert(netuid, ActivityCutoff::<T>::get(netuid));
295        }
296        if !MaxWeightsLimit::<T>::contains_key(netuid) {
297            MaxWeightsLimit::<T>::insert(netuid, MaxWeightsLimit::<T>::get(netuid));
298        }
299        if !MinAllowedWeights::<T>::contains_key(netuid) {
300            MinAllowedWeights::<T>::insert(netuid, MinAllowedWeights::<T>::get(netuid));
301        }
302        if !RegistrationsThisInterval::<T>::contains_key(netuid) {
303            RegistrationsThisInterval::<T>::insert(
304                netuid,
305                RegistrationsThisInterval::<T>::get(netuid),
306            );
307        }
308        if !POWRegistrationsThisInterval::<T>::contains_key(netuid) {
309            POWRegistrationsThisInterval::<T>::insert(
310                netuid,
311                POWRegistrationsThisInterval::<T>::get(netuid),
312            );
313        }
314        if !BurnRegistrationsThisInterval::<T>::contains_key(netuid) {
315            BurnRegistrationsThisInterval::<T>::insert(
316                netuid,
317                BurnRegistrationsThisInterval::<T>::get(netuid),
318            );
319        }
320    }

322    /// Execute the start call for a subnet.
323    ///
324    /// This function is used to trigger the start call process for a subnet identified by `netuid`.
325    /// It ensures that the subnet exists, the caller is the subnet owner,
326    /// and the last emission block number has not been set yet.
327    /// It then sets the last emission block number to the current block number.
328    ///
329    /// # Parameters
330    ///
331    /// * `origin`: The origin of the call, which is used to ensure the caller is the subnet owner.
332    /// * `netuid`: The unique identifier of the subnet for which the start call process is being initiated.
333    ///
334    /// # Raises
335    ///
336    /// * `Error::<T>::SubNetworkDoesNotExist`: If the subnet does not exist.
337    /// * `DispatchError::BadOrigin`: If the caller is not the subnet owner.
338    /// * `Error::<T>::FirstEmissionBlockNumberAlreadySet`: If the last emission block number has already been set.
339    ///
340    /// # Returns
341    ///
342    /// * `DispatchResult`: A result indicating the success or failure of the operation.
343    pub fn do_start_call(origin: T::RuntimeOrigin, netuid: u16) -> DispatchResult {
344        ensure!(
345            Self::if_subnet_exist(netuid),
346            Error::<T>::SubNetworkDoesNotExist
347        );
348        Self::ensure_subnet_owner(origin, netuid)?;
349        ensure!(
350            FirstEmissionBlockNumber::<T>::get(netuid).is_none(),
351            Error::<T>::FirstEmissionBlockNumberAlreadySet
352        );
353
354        let registration_block_number = NetworkRegisteredAt::<T>::get(netuid);
355        let current_block_number = Self::get_current_block_as_u64();
356
357        ensure!(
358            current_block_number
359                >= registration_block_number.saturating_add(T::DurationOfStartCall::get()),
360            Error::<T>::NeedWaitingMoreBlocksToStarCall
361        );
362        let next_block_number = current_block_number.saturating_add(1);
363
364        FirstEmissionBlockNumber::<T>::insert(netuid, next_block_number);
365        SubtokenEnabled::<T>::insert(netuid, true);
366        Self::deposit_event(Event::FirstEmissionBlockNumberSet(
367            netuid,
368            next_block_number,
369        ));
370        Ok(())
371    }

373    /// Sets or updates the hotkey account associated with the owner of a specific subnet.
374    ///
375    /// This function allows either the root origin or the current subnet owner to set or update
376    /// the hotkey for a given subnet. The subnet must already exist. To prevent abuse, the call is
377    /// rate-limited to once per configured interval (default: one week) per subnet.
378    ///
379    /// # Parameters
380    /// - `origin`: The dispatch origin of the call. Must be either root or the current owner of the subnet.
381    /// - `netuid`: The unique identifier of the subnet whose owner hotkey is being set.
382    /// - `hotkey`: The new hotkey account to associate with the subnet owner.
383    ///
384    /// # Returns
385    /// - `DispatchResult`: Returns `Ok(())` if the hotkey was successfully set, or an appropriate error otherwise.
386    ///
387    /// # Errors
388    /// - `Error::SubnetNotExists`: If the specified subnet does not exist.
389    /// - `Error::TxRateLimitExceeded`: If the function is called more frequently than the allowed rate limit.
390    ///
391    /// # Access Control
392    /// Only callable by:
393    /// - Root origin, or
394    /// - The coldkey account that owns the subnet.
395    ///
396    /// # Storage
397    /// - Updates [`SubnetOwnerHotkey`] for the given `netuid`.
398    /// - Reads and updates [`LastRateLimitedBlock`] for rate-limiting.
399    /// - Reads [`DefaultSetSNOwnerHotkeyRateLimit`] to determine the interval between allowed updates.
400    ///
401    /// # Rate Limiting
402    /// This function is rate-limited to one call per subnet per interval (e.g., one week).
403    pub fn do_set_sn_owner_hotkey(
404        origin: T::RuntimeOrigin,
405        netuid: u16,
406        hotkey: &T::AccountId,
407    ) -> DispatchResult {
408        // Ensure the caller is either root or subnet owner.
409        Self::ensure_subnet_owner_or_root(origin, netuid)?;
410
411        // Ensure that the subnet exists.
412        ensure!(Self::if_subnet_exist(netuid), Error::<T>::SubnetNotExists);
413
414        // Rate limit: 1 call per week
415        ensure!(
416            Self::passes_rate_limit_on_subnet(
417                &TransactionType::SetSNOwnerHotkey,
418                hotkey, // ignored
419                netuid, // Specific to a subnet.
420            ),
421            Error::<T>::TxRateLimitExceeded
422        );
423
424        // Set last transaction block
425        let current_block = Self::get_current_block_as_u64();
426        Self::set_last_transaction_block_on_subnet(
427            hotkey,
428            netuid,
429            &TransactionType::SetSNOwnerHotkey,
430            current_block,
431        );
432
433        // Insert/update the hotkey
434        SubnetOwnerHotkey::<T>::insert(netuid, hotkey);
435
436        // Return success.
437        Ok(())
438    }
