4    /// ---- The implementation for the extrinsic serve_axon which sets the ip endpoint information for a uid on a network.
5    ///
6    /// # Args:
7    /// * 'origin': (<T as frame_system::Config>RuntimeOrigin):
8    ///     - The signature of the caller.
9    ///
10    /// * 'netuid' (u16):
11    ///     - The u16 network identifier.
12    ///
13    /// * 'version' (u64):
14    ///     - The bittensor version identifier.
15    ///
16    /// * 'ip' (u64):
17    ///     - The endpoint ip information as a u128 encoded integer.
18    ///
19    /// * 'port' (u16):
20    ///     - The endpoint port information as a u16 encoded integer.
21    ///
22    /// * 'ip_type' (u8):
23    ///     - The endpoint ip version as a u8, 4 or 6.
24    ///
25    /// * 'protocol' (u8):
26    ///     - UDP:1 or TCP:0
27    ///
28    /// * 'placeholder1' (u8):
29    ///     - Placeholder for further extra params.
30    ///
31    /// * 'placeholder2' (u8):
32    ///     - Placeholder for further extra params.
33    ///
34    /// * 'certificate' (Option<Vec<u8>>):
35    ///     - Certificate for mutual Tls connection between neurons
36    ///
37    /// # Event:
38    /// * AxonServed;
39    ///     - On successfully serving the axon info.
40    ///
41    /// # Raises:
42    /// * 'SubNetworkDoesNotExist':
43    ///     - Attempting to set weights on a non-existent network.
44    ///
45    /// * 'NotRegistered':
46    ///     - Attempting to set weights from a non registered account.
47    ///
48    /// * 'InvalidIpType':
49    ///     - The ip type is not 4 or 6.
50    ///
51    /// * 'InvalidIpAddress':
52    ///     - The numerically encoded ip address does not resolve to a proper ip.
53    ///
54    /// * 'ServingRateLimitExceeded':
55    ///     - Attempting to set prometheus information withing the rate limit min.
56    ///
57    pub fn do_serve_axon(
58        origin: T::RuntimeOrigin,
59        netuid: u16,
60        version: u32,
61        ip: u128,
62        port: u16,
63        ip_type: u8,
64        protocol: u8,
65        placeholder1: u8,
66        placeholder2: u8,
67        certificate: Option<Vec<u8>>,
68    ) -> dispatch::DispatchResult {
69        // We check the callers (hotkey) signature.
70        let hotkey_id = ensure_signed(origin)?;
71
72        // Validate user input
73        Self::validate_serve_axon(
74            &hotkey_id,
75            netuid,
76            version,
77            ip,
78            port,
79            ip_type,
80            protocol,
81            placeholder1,
82            placeholder2,
83        )?;
84
85        // Check+insert certificate
86        if let Some(certificate) = certificate {
87            if let Ok(certificate) = NeuronCertificateOf::try_from(certificate) {
88                NeuronCertificates::<T>::insert(netuid, hotkey_id.clone(), certificate)
89            }
90        }
91
92        // We insert the axon meta.
93        let mut prev_axon = Self::get_axon_info(netuid, &hotkey_id);
94        prev_axon.block = Self::get_current_block_as_u64();
95        prev_axon.version = version;
96        prev_axon.ip = ip;
97        prev_axon.port = port;
98        prev_axon.ip_type = ip_type;
99        prev_axon.protocol = protocol;
100        prev_axon.placeholder1 = placeholder1;
101        prev_axon.placeholder2 = placeholder2;
102
103        // Validate axon data with delegate func
104        let axon_validated = Self::validate_axon_data(&prev_axon);
105        ensure!(
106            axon_validated.is_ok(),
107            axon_validated.err().unwrap_or(Error::<T>::InvalidPort)
108        );
109
110        Axons::<T>::insert(netuid, hotkey_id.clone(), prev_axon);
111
112        // We deposit axon served event.
113        log::debug!("AxonServed( hotkey:{:?} ) ", hotkey_id.clone());
114        Self::deposit_event(Event::AxonServed(netuid, hotkey_id));
115
116        // Return is successful dispatch.
117        Ok(())
118    }

124    ///     - The signature of the caller.

134    ///
135    /// * 'port' (u16):
136    ///     - The prometheus port information as a u16 encoded integer.
137    ///
138    /// * 'ip_type' (u8):
139    ///     - The prometheus ip version as a u8, 4 or 6.
140    ///
141    /// # Event:
142    /// * PrometheusServed;
143    ///     - On successfully serving the axon info.
144    ///
145    /// # Raises:
146    /// * 'SubNetworkDoesNotExist':
147    ///     - Attempting to set weights on a non-existent network.
148    ///
149    /// * 'NotRegistered':
150    ///     - Attempting to set weights from a non registered account.
151    ///
152    /// * 'InvalidIpType':
153    ///     - The ip type is not 4 or 6.
154    ///
155    /// * 'InvalidIpAddress':
156    ///     - The numerically encoded ip address does not resolve to a proper ip.
157    ///
158    /// * 'ServingRateLimitExceeded':
159    ///     - Attempting to set prometheus information withing the rate limit min.
160    ///
161    pub fn do_serve_prometheus(
162        origin: T::RuntimeOrigin,
163        netuid: u16,
164        version: u32,
165        ip: u128,
166        port: u16,
167        ip_type: u8,
168    ) -> dispatch::DispatchResult {
169        // We check the callers (hotkey) signature.
170        let hotkey_id = ensure_signed(origin)?;
171
172        // Check the ip signature validity.
173        ensure!(Self::is_valid_ip_type(ip_type), Error::<T>::InvalidIpType);
174        ensure!(
175            Self::is_valid_ip_address(ip_type, ip),
176            Error::<T>::InvalidIpAddress
177        );
178
179        // Ensure the hotkey is registered somewhere.
180        ensure!(
181            Self::is_hotkey_registered_on_any_network(&hotkey_id),
182            Error::<T>::HotKeyNotRegisteredInNetwork
183        );
184
185        // We get the previous axon info assoicated with this ( netuid, uid )
186        let mut prev_prometheus = Self::get_prometheus_info(netuid, &hotkey_id);
187        let current_block: u64 = Self::get_current_block_as_u64();
188        ensure!(
189            Self::prometheus_passes_rate_limit(netuid, &prev_prometheus, current_block),
190            Error::<T>::ServingRateLimitExceeded
191        );
192
193        // We insert the prometheus meta.
194        prev_prometheus.block = Self::get_current_block_as_u64();
195        prev_prometheus.version = version;
196        prev_prometheus.ip = ip;
197        prev_prometheus.port = port;
198        prev_prometheus.ip_type = ip_type;
199
200        // Validate prometheus data with delegate func
201        let prom_validated = Self::validate_prometheus_data(&prev_prometheus);
202        ensure!(
203            prom_validated.is_ok(),
204            prom_validated.err().unwrap_or(Error::<T>::InvalidPort)
205        );
206
207        // Insert new prometheus data
208        Prometheus::<T>::insert(netuid, hotkey_id.clone(), prev_prometheus);
209
210        // We deposit prometheus served event.
211        log::debug!("PrometheusServed( hotkey:{:?} ) ", hotkey_id.clone());
212        Self::deposit_event(Event::PrometheusServed(netuid, hotkey_id));
213
214        // Return is successful dispatch.
215        Ok(())
216    }

222    pub fn axon_passes_rate_limit(
223        netuid: u16,
224        prev_axon_info: &AxonInfoOf,
225        current_block: u64,
226    ) -> bool {
227        let rate_limit: u64 = Self::get_serving_rate_limit(netuid);
228        let last_serve = prev_axon_info.block;
229        rate_limit == 0 || last_serve == 0 || current_block.saturating_sub(last_serve) >= rate_limit
230    }

279    pub fn is_valid_ip_address(ip_type: u8, addr: u128) -> bool {
280        if !Self::is_valid_ip_type(ip_type) {
281            return false;
282        }
283        if addr == 0 {
284            return false;
285        }
286        if ip_type == 4 {
287            if addr == 0 {
288                return false;
289            }
290            if addr >= u32::MAX as u128 {
291                return false;
292            }
293            if addr == 0x7f000001 {
294                return false;
295            } // Localhost
296        }
297        if ip_type == 6 {
298            if addr == 0x0 {
299                return false;
300            }
301            if addr == u128::MAX {
302                return false;
303            }
304            if addr == 1 {
305                return false;
306            } // IPv6 localhost
307        }
308        true
309    }

329    pub fn validate_serve_axon(
330        hotkey_id: &T::AccountId,
331        netuid: u16,
332        version: u32,
333        ip: u128,
334        port: u16,
335        ip_type: u8,
336        protocol: u8,
337        placeholder1: u8,
338        placeholder2: u8,
339    ) -> Result<(), Error<T>> {
340        // Ensure the hotkey is registered somewhere.
341        ensure!(
342            Self::is_hotkey_registered_on_any_network(hotkey_id),
343            Error::<T>::HotKeyNotRegisteredInNetwork
344        );
345
346        // Check the ip signature validity.
347        ensure!(Self::is_valid_ip_type(ip_type), Error::<T>::InvalidIpType);
348        ensure!(
349            Self::is_valid_ip_address(ip_type, ip),
350            Error::<T>::InvalidIpAddress
351        );
352
353        // Get the previous axon information.
354        let mut prev_axon = Self::get_axon_info(netuid, hotkey_id);
355        let current_block: u64 = Self::get_current_block_as_u64();
356        ensure!(
357            Self::axon_passes_rate_limit(netuid, &prev_axon, current_block),
358            Error::<T>::ServingRateLimitExceeded
359        );
360
361        // Validate axon data with delegate func
362        prev_axon.block = Self::get_current_block_as_u64();
363        prev_axon.version = version;
364        prev_axon.ip = ip;
365        prev_axon.port = port;
366        prev_axon.ip_type = ip_type;
367        prev_axon.protocol = protocol;
368        prev_axon.placeholder1 = placeholder1;
369        prev_axon.placeholder2 = placeholder2;
370
371        let axon_validated = Self::validate_axon_data(&prev_axon);
372        ensure!(
373            axon_validated.is_ok(),
374            axon_validated.err().unwrap_or(Error::<T>::InvalidPort)
375        );
376
377        Ok(())
378    }
