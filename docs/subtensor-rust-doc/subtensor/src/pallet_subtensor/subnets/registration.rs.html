10    pub fn register_neuron(netuid: u16, hotkey: &T::AccountId) -> u16 {
11        // Init param
12        let neuron_uid: u16;
13        let block_number: u64 = Self::get_current_block_as_u64();
14        let current_subnetwork_n: u16 = Self::get_subnetwork_n(netuid);
15
16        if current_subnetwork_n < Self::get_max_allowed_uids(netuid) {
17            // No replacement required, the uid appends the subnetwork.
18            // We increment the subnetwork count here but not below.
19            neuron_uid = current_subnetwork_n;
20
21            // Expand subnetwork with new account.
22            Self::append_neuron(netuid, hotkey, block_number);
23            log::debug!("add new neuron account");
24        } else {
25            // Replacement required.
26            // We take the neuron with the lowest pruning score here.
27            neuron_uid = Self::get_neuron_to_prune(netuid);
28
29            // Replace the neuron account with the new info.
30            Self::replace_neuron(netuid, neuron_uid, hotkey, block_number);
31            log::debug!("prune neuron");
32        }
33
34        // Return the UID of the neuron.
35        neuron_uid
36    }

38    /// ---- The implementation for the extrinsic do_burned_registration: registering by burning TAO.
39    ///
40    /// # Args:
41    /// * 'origin': (<T as frame_system::Config>RuntimeOrigin):
42    ///     - The signature of the calling coldkey.
43    ///       Burned registers can only be created by the coldkey.
44    ///
45    /// * 'netuid' (u16):
46    ///     - The u16 network identifier.
47    ///
48    /// * 'hotkey' ( T::AccountId ):
49    ///     - Hotkey to be registered to the network.
50    ///
51    /// # Event:
52    /// * NeuronRegistered;
53    ///     - On successfully registereing a uid to a neuron slot on a subnetwork.
54    ///
55    /// # Raises:
56    /// * 'SubNetworkDoesNotExist':
57    ///     - Attempting to registed to a non existent network.
58    ///
59    /// * 'TooManyRegistrationsThisBlock':
60    ///     - This registration exceeds the total allowed on this network this block.
61    ///
62    /// * 'HotKeyAlreadyRegisteredInSubNet':
63    ///     - The hotkey is already registered on this network.
64    ///
65    pub fn do_burned_registration(
66        origin: T::RuntimeOrigin,
67        netuid: u16,
68        hotkey: T::AccountId,
69    ) -> DispatchResult {
70        // --- 1. Check that the caller has signed the transaction. (the coldkey of the pairing)
71        let coldkey = ensure_signed(origin)?;
72        log::debug!(
73            "do_registration( coldkey:{:?} netuid:{:?} hotkey:{:?} )",
74            coldkey,
75            netuid,
76            hotkey
77        );
78
79        // --- 2. Ensure the passed network is valid.
80        ensure!(
81            netuid != Self::get_root_netuid(),
82            Error::<T>::RegistrationNotPermittedOnRootSubnet
83        );
84        ensure!(
85            Self::if_subnet_exist(netuid),
86            Error::<T>::SubNetworkDoesNotExist
87        );
88
89        // --- 3. Ensure the passed network allows registrations.
90        ensure!(
91            Self::get_network_registration_allowed(netuid),
92            Error::<T>::SubNetRegistrationDisabled
93        );
94
95        // --- 4. Ensure we are not exceeding the max allowed registrations per block.
96        ensure!(
97            Self::get_registrations_this_block(netuid)
98                < Self::get_max_registrations_per_block(netuid),
99            Error::<T>::TooManyRegistrationsThisBlock
100        );
101
102        // --- 4. Ensure we are not exceeding the max allowed registrations per interval.
103        ensure!(
104            Self::get_registrations_this_interval(netuid)
105                < Self::get_target_registrations_per_interval(netuid).saturating_mul(3),
106            Error::<T>::TooManyRegistrationsThisInterval
107        );
108
109        // --- 4. Ensure that the key is not already registered.
110        ensure!(
111            !Uids::<T>::contains_key(netuid, &hotkey),
112            Error::<T>::HotKeyAlreadyRegisteredInSubNet
113        );
114
115        // --- 7. Ensure the callers coldkey has enough stake to perform the transaction.
116        let registration_cost = Self::get_burn_as_u64(netuid);
117        ensure!(
118            Self::can_remove_balance_from_coldkey_account(&coldkey, registration_cost),
119            Error::<T>::NotEnoughBalanceToStake
120        );
121
122        // If the network account does not exist we will create it here.
123        Self::create_account_if_non_existent(&coldkey, &hotkey);
124
125        // --- 8. Ensure that the pairing is correct.
126        ensure!(
127            Self::coldkey_owns_hotkey(&coldkey, &hotkey),
128            Error::<T>::NonAssociatedColdKey
129        );
130
131        // --- 9. Possibly there are no neuron slots at all.
132        ensure!(
133            Self::get_max_allowed_uids(netuid) != 0,
134            Error::<T>::NoNeuronIdAvailable
135        );
136
137        // --- 10. Ensure the remove operation from the coldkey is a success.
138        let actual_burn_amount =
139            Self::remove_balance_from_coldkey_account(&coldkey, registration_cost)?;
140
141        // Tokens are swapped and then burned.
142        let burned_alpha: u64 = Self::swap_tao_for_alpha(netuid, actual_burn_amount);
143        SubnetAlphaOut::<T>::mutate(netuid, |total| *total = total.saturating_sub(burned_alpha));
144
145        // Actually perform the registration.
146        let neuron_uid: u16 = Self::register_neuron(netuid, &hotkey);
147
148        // --- 14. Record the registration and increment block and interval counters.
149        BurnRegistrationsThisInterval::<T>::mutate(netuid, |val| val.saturating_inc());
150        RegistrationsThisInterval::<T>::mutate(netuid, |val| val.saturating_inc());
151        RegistrationsThisBlock::<T>::mutate(netuid, |val| val.saturating_inc());
152        Self::increase_rao_recycled(netuid, Self::get_burn_as_u64(netuid));
153
154        // --- 15. Deposit successful event.
155        log::debug!(
156            "NeuronRegistered( netuid:{:?} uid:{:?} hotkey:{:?}  ) ",
157            netuid,
158            neuron_uid,
159            hotkey
160        );
161        Self::deposit_event(Event::NeuronRegistered(netuid, neuron_uid, hotkey));
162
163        // --- 16. Ok and done.
164        Ok(())
165    }

167    /// ---- The implementation for the extrinsic do_registration.
168    ///
169    /// # Args:
170    /// *'origin': (<T as frame_system::Config>RuntimeOrigin):
171    ///     - The signature of the calling hotkey.
172    ///
173    /// *'netuid' (u16):
174    ///     - The u16 network identifier.
175    ///
176    /// *'block_number' ( u64 ):
177    ///     - Block hash used to prove work done.
178    ///
179    /// *'nonce' ( u64 ):
180    ///     - Positive integer nonce used in POW.
181    ///
182    /// *'work' ( Vec<u8> ):
183    ///     - Vector encoded bytes representing work done.
184    ///
185    /// *'hotkey' ( T::AccountId ):
186    ///     - Hotkey to be registered to the network.
187    ///
188    /// *'coldkey' ( T::AccountId ):
189    ///     - Associated coldkey account.
190    ///
191    /// # Event:
192    /// *NeuronRegistered;
193    ///     - On successfully registereing a uid to a neuron slot on a subnetwork.
194    ///
195    /// # Raises:
196    /// *'SubNetworkDoesNotExist':
197    ///     - Attempting to registed to a non existent network.
198    ///
199    /// *'TooManyRegistrationsThisBlock':
200    ///     - This registration exceeds the total allowed on this network this block.
201    ///
202    /// *'HotKeyAlreadyRegisteredInSubNet':
203    ///     - The hotkey is already registered on this network.
204    ///
205    /// *'InvalidWorkBlock':
206    ///     - The work has been performed on a stale, future, or non existent block.
207    ///
208    /// *'InvalidDifficulty':
209    ///     - The work does not match the difficutly.
210    ///
211    /// *'InvalidSeal':
212    ///     - The seal is incorrect.
213    ///
214    pub fn do_registration(
215        origin: T::RuntimeOrigin,
216        netuid: u16,
217        block_number: u64,
218        nonce: u64,
219        work: Vec<u8>,
220        hotkey: T::AccountId,
221        coldkey: T::AccountId,
222    ) -> DispatchResult {
223        // --- 1. Check that the caller has signed the transaction.
224        let signing_origin = ensure_signed(origin)?;
225        log::debug!(
226            "do_registration( origin:{:?} netuid:{:?} hotkey:{:?}, coldkey:{:?} )",
227            signing_origin,
228            netuid,
229            hotkey,
230            coldkey
231        );
232
233        ensure!(
234            signing_origin == hotkey,
235            Error::<T>::TransactorAccountShouldBeHotKey
236        );
237
238        // --- 2. Ensure the passed network is valid.
239        ensure!(
240            netuid != Self::get_root_netuid(),
241            Error::<T>::RegistrationNotPermittedOnRootSubnet
242        );
243        ensure!(
244            Self::if_subnet_exist(netuid),
245            Error::<T>::SubNetworkDoesNotExist
246        );
247
248        // --- 3. Ensure the passed network allows registrations.
249        ensure!(
250            Self::get_network_pow_registration_allowed(netuid),
251            Error::<T>::SubNetRegistrationDisabled
252        );
253
254        // --- 4. Ensure we are not exceeding the max allowed registrations per block.
255        ensure!(
256            Self::get_registrations_this_block(netuid)
257                < Self::get_max_registrations_per_block(netuid),
258            Error::<T>::TooManyRegistrationsThisBlock
259        );
260
261        // --- 5. Ensure we are not exceeding the max allowed registrations per interval.
262        ensure!(
263            Self::get_registrations_this_interval(netuid)
264                < Self::get_target_registrations_per_interval(netuid).saturating_mul(3),
265            Error::<T>::TooManyRegistrationsThisInterval
266        );
267
268        // --- 6. Ensure that the key is not already registered.
269        ensure!(
270            !Uids::<T>::contains_key(netuid, &hotkey),
271            Error::<T>::HotKeyAlreadyRegisteredInSubNet
272        );
273
274        // --- 7. Ensure the passed block number is valid, not in the future or too old.
275        // Work must have been done within 3 blocks (stops long range attacks).
276        let current_block_number: u64 = Self::get_current_block_as_u64();
277        ensure!(
278            block_number <= current_block_number,
279            Error::<T>::InvalidWorkBlock
280        );
281        ensure!(
282            current_block_number.saturating_sub(block_number) < 3,
283            Error::<T>::InvalidWorkBlock
284        );
285
286        // --- 8. Ensure the supplied work passes the difficulty.
287        let difficulty: U256 = Self::get_difficulty(netuid);
288        let work_hash: H256 = Self::vec_to_hash(work.clone());
289        ensure!(
290            Self::hash_meets_difficulty(&work_hash, difficulty),
291            Error::<T>::InvalidDifficulty
292        ); // Check that the work meets difficulty.
293
294        // --- 7. Check Work is the product of the nonce, the block number, and hotkey. Add this as used work.
295        let seal: H256 = Self::create_seal_hash(block_number, nonce, &hotkey);
296        ensure!(seal == work_hash, Error::<T>::InvalidSeal);
297        UsedWork::<T>::insert(work.clone(), current_block_number);
298
299        // DEPRECATED --- 8. Ensure that the key passes the registration requirement
300        // ensure!(
301        //     Self::passes_network_connection_requirement(netuid, &hotkey),
302        //     Error::<T>::DidNotPassConnectedNetworkRequirement
303        // );
304
305        // --- 9. If the network account does not exist we will create it here.
306        Self::create_account_if_non_existent(&coldkey, &hotkey);
307
308        // --- 10. Ensure that the pairing is correct.
309        ensure!(
310            Self::coldkey_owns_hotkey(&coldkey, &hotkey),
311            Error::<T>::NonAssociatedColdKey
312        );
313
314        // Possibly there is no neuron slots at all.
315        ensure!(
316            Self::get_max_allowed_uids(netuid) != 0,
317            Error::<T>::NoNeuronIdAvailable
318        );
319
320        // Actually perform the registration.
321        let neuron_uid: u16 = Self::register_neuron(netuid, &hotkey);
322
323        // --- 12. Record the registration and increment block and interval counters.
324        POWRegistrationsThisInterval::<T>::mutate(netuid, |val| val.saturating_inc());
325        RegistrationsThisInterval::<T>::mutate(netuid, |val| val.saturating_inc());
326        RegistrationsThisBlock::<T>::mutate(netuid, |val| val.saturating_inc());
327
328        // --- 13. Deposit successful event.
329        log::debug!(
330            "NeuronRegistered( netuid:{:?} uid:{:?} hotkey:{:?}  ) ",
331            netuid,
332            neuron_uid,
333            hotkey
334        );
335        Self::deposit_event(Event::NeuronRegistered(netuid, neuron_uid, hotkey));
336
337        // --- 14. Ok and done.
338        Ok(())
339    }

341    pub fn do_faucet(
342        origin: T::RuntimeOrigin,
343        block_number: u64,
344        nonce: u64,
345        work: Vec<u8>,
346    ) -> DispatchResult {
347        // --- 0. Ensure the faucet is enabled.
348        // ensure!(AllowFaucet::<T>::get(), Error::<T>::FaucetDisabled);
349
350        // --- 1. Check that the caller has signed the transaction.
351        let coldkey = ensure_signed(origin)?;
352        log::debug!("do_faucet( coldkey:{:?} )", coldkey);
353
354        // --- 2. Ensure the passed block number is valid, not in the future or too old.
355        // Work must have been done within 3 blocks (stops long range attacks).
356        let current_block_number: u64 = Self::get_current_block_as_u64();
357        ensure!(
358            block_number <= current_block_number,
359            Error::<T>::InvalidWorkBlock
360        );
361        ensure!(
362            current_block_number.saturating_sub(block_number) < 3,
363            Error::<T>::InvalidWorkBlock
364        );
365
366        // --- 3. Ensure the supplied work passes the difficulty.
367        let difficulty: U256 = U256::from(1_000_000); // Base faucet difficulty.
368        let work_hash: H256 = Self::vec_to_hash(work.clone());
369        ensure!(
370            Self::hash_meets_difficulty(&work_hash, difficulty),
371            Error::<T>::InvalidDifficulty
372        ); // Check that the work meets difficulty.
373
374        // --- 4. Check Work is the product of the nonce, the block number, and hotkey. Add this as used work.
375        let seal: H256 = Self::create_seal_hash(block_number, nonce, &coldkey);
376        ensure!(seal == work_hash, Error::<T>::InvalidSeal);
377        UsedWork::<T>::insert(work.clone(), current_block_number);
378
379        // --- 5. Add Balance via faucet.
380        let balance_to_add: u64 = 1_000_000_000_000;
381        Self::coinbase(100_000_000_000); // We are creating tokens here from the coinbase.
382
383        Self::add_balance_to_coldkey_account(&coldkey, balance_to_add);
384
385        // --- 6. Deposit successful event.
386        log::debug!(
387            "Faucet( coldkey:{:?} amount:{:?} ) ",
388            coldkey,
389            balance_to_add
390        );
391        Self::deposit_event(Event::Faucet(coldkey, balance_to_add));
392
393        // --- 7. Ok and done.
394        Ok(())
395    }

404    /// Determine which peer to prune from the network by finding the element with the lowest pruning score out of
405    /// immunity period. If there is a tie for lowest pruning score, the neuron registered earliest is pruned.
406    /// If all neurons are in immunity period, the neuron with the lowest pruning score is pruned. If there is a tie for
407    /// the lowest pruning score, the immune neuron registered earliest is pruned.
408    /// Ties for earliest registration are broken by the neuron with the lowest uid.
409    pub fn get_neuron_to_prune(netuid: u16) -> u16 {
410        let mut min_score: u16 = u16::MAX;
411        let mut min_score_in_immunity: u16 = u16::MAX;
412        let mut earliest_registration: u64 = u64::MAX;
413        let mut earliest_registration_in_immunity: u64 = u64::MAX;
414        let mut uid_to_prune: u16 = 0;
415        let mut uid_to_prune_in_immunity: u16 = 0;
416
417        // This boolean is used instead of checking if min_score == u16::MAX, to avoid the case
418        // where all non-immune neurons have pruning score u16::MAX
419        // This may be unlikely in practice.
420        let mut found_non_immune = false;
421
422        let neurons_n = Self::get_subnetwork_n(netuid);
423        if neurons_n == 0 {
424            return 0; // If there are no neurons in this network.
425        }
426
427        for neuron_uid in 0..neurons_n {
428            // Do not deregister the owner's hotkey from the `SubnetOwnerHotkey` map
429            if let Ok(hotkey) = Self::get_hotkey_for_net_and_uid(netuid, neuron_uid) {
430                if let Ok(top_sn_owner_hotkey) = SubnetOwnerHotkey::<T>::try_get(netuid) {
431                    if top_sn_owner_hotkey == hotkey {
432                        continue;
433                    }
434                }
435            }
436
437            let pruning_score: u16 = Self::get_pruning_score_for_uid(netuid, neuron_uid);
438            let block_at_registration: u64 =
439                Self::get_neuron_block_at_registration(netuid, neuron_uid);
440            let is_immune = Self::get_neuron_is_immune(netuid, neuron_uid);
441
442            if is_immune {
443                // if the immune neuron has a lower pruning score than the minimum for immune neurons,
444                // or, if the pruning scores are equal and the immune neuron was registered earlier than the current minimum for immune neurons,
445                // then update the minimum pruning score and the uid to prune for immune neurons
446                if pruning_score < min_score_in_immunity
447                    || (pruning_score == min_score_in_immunity
448                        && block_at_registration < earliest_registration_in_immunity)
449                {
450                    min_score_in_immunity = pruning_score;
451                    earliest_registration_in_immunity = block_at_registration;
452                    uid_to_prune_in_immunity = neuron_uid;
453                }
454            } else {
455                found_non_immune = true;
456                // if the non-immune neuron has a lower pruning score than the minimum for non-immune neurons,
457                // or, if the pruning scores are equal and the non-immune neuron was registered earlier than the current minimum for non-immune neurons,
458                // then update the minimum pruning score and the uid to prune for non-immune neurons
459                if pruning_score < min_score
460                    || (pruning_score == min_score && block_at_registration < earliest_registration)
461                {
462                    min_score = pruning_score;
463                    earliest_registration = block_at_registration;
464                    uid_to_prune = neuron_uid;
465                }
466            }
467        }
468
469        if found_non_immune {
470            Self::set_pruning_score_for_uid(netuid, uid_to_prune, u16::MAX);
471            uid_to_prune
472        } else {
473            Self::set_pruning_score_for_uid(netuid, uid_to_prune_in_immunity, u16::MAX);
474            uid_to_prune_in_immunity
475        }
476    }

478    /// Determine whether the given hash satisfies the given difficulty.
479    /// The test is done by multiplying the two together. If the product
480    /// overflows the bounds of U256, then the product (and thus the hash)
481    /// was too high.
482    pub fn hash_meets_difficulty(hash: &H256, difficulty: U256) -> bool {
483        let bytes: &[u8] = hash.as_bytes();
484        let num_hash: U256 = U256::from(bytes);
485        let (value, overflowed) = num_hash.overflowing_mul(difficulty);
486
487        log::trace!(
488            target: LOG_TARGET,
489            "Difficulty: hash: {:?}, hash_bytes: {:?}, hash_as_num: {:?}, difficulty: {:?}, value: {:?} overflowed: {:?}",
490            hash,
491            bytes,
492            num_hash,
493            difficulty,
494            value,
495            overflowed
496        );
497        !overflowed
498    }

500    pub fn get_block_hash_from_u64(block_number: u64) -> H256 {
501        let block_number: BlockNumberFor<T> = TryInto::<BlockNumberFor<T>>::try_into(block_number)
502            .ok()
503            .expect("convert u64 to block number.");
504        let block_hash_at_number: <T as frame_system::Config>::Hash =
505            system::Pallet::<T>::block_hash(block_number);
506        let vec_hash: Vec<u8> = block_hash_at_number.as_ref().to_vec();
507        let deref_vec_hash: &[u8] = &vec_hash; // c: &[u8]
508        let real_hash: H256 = H256::from_slice(deref_vec_hash);
509
510        log::trace!(
511            target: LOG_TARGET,
512            "block_number: {:?}, vec_hash: {:?}, real_hash: {:?}",
513            block_number,
514            vec_hash,
515            real_hash
516        );
517
518        real_hash
519    }

527    pub fn hash_block_and_hotkey(block_hash_bytes: &[u8; 32], hotkey: &T::AccountId) -> H256 {
528        let binding = hotkey.encode();
529        // Safe because Substrate guarantees that all AccountId types are at least 32 bytes
530        let (hotkey_bytes, _) = binding.split_at(32);
531        let mut full_bytes = [0u8; 64];
532        let (first_half, second_half) = full_bytes.split_at_mut(32);
533        first_half.copy_from_slice(block_hash_bytes);
534        second_half.copy_from_slice(hotkey_bytes);
535        let keccak_256_seal_hash_vec: [u8; 32] = keccak_256(&full_bytes[..]);
536
537        H256::from_slice(&keccak_256_seal_hash_vec)
538    }

540    pub fn hash_hotkey_to_u64(hotkey: &T::AccountId) -> u64 {
541        let binding = hotkey.encode();
542        let (hotkey_bytes, _) = binding.split_at(32);
543        let mut full_bytes = [0u8; 64];
544        // Copy the hotkey_bytes into the first half of full_bytes
545        full_bytes[..32].copy_from_slice(hotkey_bytes);
546        let keccak_256_seal_hash_vec: [u8; 32] = keccak_256(&full_bytes[..]);
547        let hash_u64: u64 = u64::from_le_bytes(
548            keccak_256_seal_hash_vec[0..8]
549                .try_into()
550                .unwrap_or_default(),
551        );
552        hash_u64
553    }

555    pub fn create_seal_hash(block_number_u64: u64, nonce_u64: u64, hotkey: &T::AccountId) -> H256 {
556        let nonce = nonce_u64.to_le_bytes();
557        let block_hash_at_number: H256 = Self::get_block_hash_from_u64(block_number_u64);
558        let block_hash_bytes: &[u8; 32] = block_hash_at_number.as_fixed_bytes();
559        let binding = Self::hash_block_and_hotkey(block_hash_bytes, hotkey);
560        let block_and_hotkey_hash_bytes: &[u8; 32] = binding.as_fixed_bytes();
561
562        let mut full_bytes = [0u8; 40];
563        let (first_chunk, second_chunk) = full_bytes.split_at_mut(8);
564        first_chunk.copy_from_slice(&nonce);
565        second_chunk.copy_from_slice(block_and_hotkey_hash_bytes);
566        let sha256_seal_hash_vec: [u8; 32] = sha2_256(&full_bytes[..]);
567        let keccak_256_seal_hash_vec: [u8; 32] = keccak_256(&sha256_seal_hash_vec);
568        let seal_hash: H256 = H256::from_slice(&keccak_256_seal_hash_vec);
569
570        log::trace!(
571            "\n hotkey:{:?} \nblock_number: {:?}, \nnonce_u64: {:?}, \nblock_hash: {:?}, \nfull_bytes: {:?}, \nsha256_seal_hash_vec: {:?},  \nkeccak_256_seal_hash_vec: {:?}, \nseal_hash: {:?}",
572            hotkey,
573            block_number_u64,
574            nonce_u64,
575            block_hash_at_number,
576            full_bytes,
577            sha256_seal_hash_vec,
578            keccak_256_seal_hash_vec,
579            seal_hash
580        );
581
582        seal_hash
583    }

585    /// Helper function for creating nonce and work.
586    pub fn create_work_for_block_number(
587        netuid: u16,
588        block_number: u64,
589        start_nonce: u64,
590        hotkey: &T::AccountId,
591    ) -> (u64, Vec<u8>) {
592        let difficulty: U256 = Self::get_difficulty(netuid);
593        let mut nonce: u64 = start_nonce;
594        let mut work: H256 = Self::create_seal_hash(block_number, nonce, hotkey);
595        while !Self::hash_meets_difficulty(&work, difficulty) {
596            nonce.saturating_inc();
597            work = Self::create_seal_hash(block_number, nonce, hotkey);
598        }
599        let vec_work: Vec<u8> = Self::hash_to_vec(work);
600        (nonce, vec_work)
601    }
