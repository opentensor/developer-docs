9const MESSAGE_PREFIX: &str = "\x19Ethereum Signed Message:\n";

11impl<T: Config> Pallet<T> {
12    pub(crate) fn hash_message_eip191<M: AsRef<[u8]>>(message: M) -> [u8; 32] {
13        let msg_len = message.as_ref().len().to_string();
14        keccak_256(
15            &[
16                MESSAGE_PREFIX.as_bytes(),
17                msg_len.as_bytes(),
18                message.as_ref(),
19            ]
20            .concat(),
21        )
22    }

24    /// Associate an EVM key with a hotkey.
25    ///
26    /// This function accepts a Signature, which is a signed message containing the hotkey concatenated with
27    /// the hashed block number. It will then attempt to recover the EVM key from the signature and compare it
28    /// with the `evm_key` parameter, and ensures that they match.
29    ///
30    /// The EVM key is expected to sign the message according to this formula to produce the signature:
31    /// ```text
32    /// keccak_256(hotkey ++ keccak_256(block_number))
33    /// ```
34    ///
35    /// # Arguments
36    ///
37    /// * `origin` - The origin of the call, which should be the coldkey that owns the hotkey.
38    /// * `netuid` - The unique identifier for the subnet that the hotkey belongs to.
39    /// * `hotkey` - The hotkey associated with the `origin` coldkey.
40    /// * `evm_key` - The EVM address to associate with the `hotkey`.
41    /// * `block_number` - The block number used in the `signature`.
42    /// * `signature` - A signed message by the `evm_key` containing the `hotkey` and the hashed `block_number`.
43    pub fn do_associate_evm_key(
44        origin: T::RuntimeOrigin,
45        netuid: u16,
46        hotkey: T::AccountId,
47        evm_key: H160,
48        block_number: u64,
49        mut signature: Signature,
50    ) -> dispatch::DispatchResult {
51        let coldkey = ensure_signed(origin)?;
52
53        ensure!(
54            Self::get_owning_coldkey_for_hotkey(&hotkey) == coldkey,
55            Error::<T>::NonAssociatedColdKey
56        );
57
58        // Normalize the v value to 0 or 1
59        if signature.0[64] >= 27 {
60            signature.0[64] = signature.0[64].saturating_sub(27);
61        }
62
63        let uid = Self::get_uid_for_net_and_hotkey(netuid, &hotkey)?;
64
65        let block_hash = keccak_256(block_number.encode().as_ref());
66        let message = [hotkey.encode().as_ref(), block_hash.as_ref()].concat();
67        let public = signature
68            .recover_prehashed(&Self::hash_message_eip191(message))
69            .ok_or(Error::<T>::InvalidIdentity)?;
70        let secp_pubkey = libsecp256k1::PublicKey::parse_compressed(&public.0)
71            .map_err(|_| Error::<T>::UnableToRecoverPublicKey)?;
72        let uncompressed = secp_pubkey.serialize();
73        let hashed_evm_key = H160::from_slice(&keccak_256(&uncompressed[1..])[12..]);
74
75        ensure!(
76            evm_key == hashed_evm_key,
77            Error::<T>::InvalidRecoveredPublicKey
78        );
79
80        let current_block_number = Self::get_current_block_as_u64();
81
82        AssociatedEvmAddress::<T>::insert(netuid, uid, (evm_key, current_block_number));
83
84        Self::deposit_event(Event::EvmKeyAssociated {
85            netuid,
86            hotkey,
87            evm_key,
88            block_associated: current_block_number,
89        });
90
91        Ok(())
92    }

94    pub fn uid_lookup(netuid: u16, evm_key: H160, limit: u16) -> Vec<(u16, u64)> {
95        let mut ret_val = AssociatedEvmAddress::<T>::iter_prefix(netuid)
96            .take(limit as usize)
97            .filter_map(|(uid, (stored_evm_key, block_associated))| {
98                if stored_evm_key != evm_key {
99                    return None;
100                }
101
102                Some((uid, block_associated))
103            })
104            .collect::<Vec<(u16, u64)>>();
105        ret_val.sort_by(|(_, block1), (_, block2)| block1.cmp(block2));
106        ret_val
107    }
