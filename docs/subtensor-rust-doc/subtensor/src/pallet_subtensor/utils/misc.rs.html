13    pub fn ensure_subnet_owner_or_root(
14        o: T::RuntimeOrigin,
15        netuid: u16,
16    ) -> Result<(), DispatchError> {
17        let coldkey = ensure_signed_or_root(o);
18        match coldkey {
19            Ok(Some(who)) if SubnetOwner::<T>::get(netuid) == who => Ok(()),
20            Ok(Some(_)) => Err(DispatchError::BadOrigin),
21            Ok(None) => Ok(()),
22            Err(x) => Err(x.into()),
23        }
24    }

250    pub fn do_take_checks(coldkey: &T::AccountId, hotkey: &T::AccountId) -> Result<(), Error<T>> {
251        // Ensure we are delegating a known key.
252        ensure!(
253            Self::hotkey_account_exists(hotkey),
254            Error::<T>::HotKeyAccountNotExists
255        );
256
257        // Ensure that the coldkey is the owner.
258        ensure!(
259            Self::coldkey_owns_hotkey(coldkey, hotkey),
260            Error::<T>::NonAssociatedColdKey
261        );
262
263        Ok(())
264    }

443    pub fn get_neuron_is_immune(netuid: u16, uid: u16) -> bool {
444        let registered_at = Self::get_neuron_block_at_registration(netuid, uid);
445        let current_block = Self::get_current_block_as_u64();

448    }

689        let (alpha_low, alpha_high): (u16, u16) = AlphaValues::<T>::get(netuid);
690        let converted_low =
691            I32F32::saturating_from_num(alpha_low).safe_div(I32F32::saturating_from_num(u16::MAX));
692        let converted_high =
693            I32F32::saturating_from_num(alpha_high).safe_div(I32F32::saturating_from_num(u16::MAX));
694
695        (converted_low, converted_high)
696    }
