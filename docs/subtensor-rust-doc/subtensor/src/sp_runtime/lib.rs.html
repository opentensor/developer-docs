18//! # Substrate Runtime Primitives.
19//!
20//! This crate, among other things, contains a large library of types and utilities that are used in
21//! the Substrate runtime, but are not particularly `FRAME`-oriented.
22//!
23//! ## Block, Header and Extrinsics
24//!
25//! Most notable, this crate contains some of the types and trait that enable important
26//! communication between the client and the runtime. This includes:
27//!
28//! - A set of traits to declare what any block/header/extrinsic type should provide.
29//! 	- [`traits::Block`], [`traits::Header`], [`traits::Extrinsic`]
30//! - A set of types that implement these traits, whilst still providing a high degree of
31//!   configurability via generics.
32//! 	- [`generic::Block`], [`generic::Header`], [`generic::UncheckedExtrinsic`] and
33//!    [`generic::CheckedExtrinsic`]
34//!
35//! ## Runtime API Types
36//!
37//! This crate also contains some types that are often used in conjuncture with Runtime APIs. Most
38//! notable:
39//!
40//! - [`ApplyExtrinsicResult`], and [`DispatchOutcome`], which dictate how the client and runtime
41//!   communicate about the success or failure of an extrinsic.
42//! - [`transaction_validity`], which dictates how the client and runtime communicate about the
43//!  validity of an extrinsic while still in the transaction-queue.

138/// An abstraction over justification for a block's validity under a consensus algorithm.
139///
140/// Essentially a finality proof. The exact formulation will vary between consensus
141/// algorithms. In the case where there are multiple valid proofs, inclusion within
142/// the block itself would allow swapping justifications to change the block's hash
143/// (and thus fork the chain). Sending a `Justification` alongside a block instead
144/// bypasses this problem.
145///
146/// Each justification is provided as an encoded blob, and is tagged with an ID
147/// to identify the consensus engine that generated the proof (we might have
148/// multiple justifications from different engines for the same block).
149pub type Justification = (ConsensusEngineId, EncodedJustification);
150
151/// The encoded justification specific to a consensus engine.
152pub type EncodedJustification = Vec<u8>;
153
154/// Collection of justifications for a given block, multiple justifications may
155/// be provided by different consensus engines for the same block.
156#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
157#[derive(Debug, Clone, PartialEq, Eq, Encode, Decode)]
158pub struct Justifications(Vec<Justification>);
159
160impl Justifications {
161	/// Return an iterator over the justifications.
162	pub fn iter(&self) -> impl Iterator<Item = &Justification> {
163		self.0.iter()
164	}
165
166	/// Append a justification. Returns false if a justification with the same
167	/// `ConsensusEngineId` already exists, in which case the justification is
168	/// not inserted.
169	pub fn append(&mut self, justification: Justification) -> bool {
170		if self.get(justification.0).is_some() {
171			return false
172		}
173		self.0.push(justification);
174		true
175	}
176
177	/// Return the encoded justification for the given consensus engine, if it
178	/// exists.
179	pub fn get(&self, engine_id: ConsensusEngineId) -> Option<&EncodedJustification> {
180		self.iter().find(|j| j.0 == engine_id).map(|j| &j.1)
181	}
182
183	/// Remove the encoded justification for the given consensus engine, if it exists.
184	pub fn remove(&mut self, engine_id: ConsensusEngineId) {
185		self.0.retain(|j| j.0 != engine_id)
186	}
187
188	/// Return a copy of the encoded justification for the given consensus
189	/// engine, if it exists.
190	pub fn into_justification(self, engine_id: ConsensusEngineId) -> Option<EncodedJustification> {
191		self.into_iter().find(|j| j.0 == engine_id).map(|j| j.1)
192	}
193}
194
195impl IntoIterator for Justifications {
196	type Item = Justification;
197	type IntoIter = alloc::vec::IntoIter<Self::Item>;
198
199	fn into_iter(self) -> Self::IntoIter {
200		self.0.into_iter()
201	}
202}
203
204impl From<Justification> for Justifications {
205	fn from(justification: Justification) -> Self {
206		Self(vec![justification])
207	}
208}

242#[cfg(feature = "std")]
243impl BuildStorage for sp_core::storage::Storage {
244	fn assimilate_storage(&self, storage: &mut sp_core::storage::Storage) -> Result<(), String> {
245		storage.top.extend(self.top.iter().map(|(k, v)| (k.clone(), v.clone())));
246		for (k, other_map) in self.children_default.iter() {
247			let k = k.clone();
248			if let Some(map) = storage.children_default.get_mut(&k) {
249				map.data.extend(other_map.data.iter().map(|(k, v)| (k.clone(), v.clone())));
250				if !map.child_info.try_update(&other_map.child_info) {
251					return Err("Incompatible child info update".to_string())
252				}
253			} else {
254				storage.children_default.insert(k, other_map.clone());
255			}
256		}
257		Ok(())
258	}
259}

262impl BuildStorage for () {
263	fn assimilate_storage(&self, _: &mut sp_core::storage::Storage) -> Result<(), String> {
264		Err("`assimilate_storage` not implemented for `()`".into())
265	}
266}

447impl Verify for MultiSignature {
448	type Signer = MultiSigner;
449	fn verify<L: Lazy<[u8]>>(&self, mut msg: L, signer: &AccountId32) -> bool {
450		let who: [u8; 32] = *signer.as_ref();
451		match self {
452			Self::Ed25519(sig) => sig.verify(msg, &who.into()),
453			Self::Sr25519(sig) => sig.verify(msg, &who.into()),
454			Self::Ecdsa(sig) => {
455				let m = sp_io::hashing::blake2_256(msg.get());
456				sp_io::crypto::secp256k1_ecdsa_recover_compressed(sig.as_ref(), &m)
457					.map_or(false, |pubkey| sp_io::hashing::blake2_256(&pubkey) == who)
458			},
459		}
460	}
461}

463/// Signature verify that can work with any known signature types..
464#[derive(Eq, PartialEq, Clone, Default, Encode, Decode, RuntimeDebug, TypeInfo)]
465#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
466pub struct AnySignature(H512);
467
468impl Verify for AnySignature {
469	type Signer = sr25519::Public;
470	fn verify<L: Lazy<[u8]>>(&self, mut msg: L, signer: &sr25519::Public) -> bool {
471		let msg = msg.get();
472		sr25519::Signature::try_from(self.0.as_fixed_bytes().as_ref())
473			.map(|s| s.verify(msg, signer))
474			.unwrap_or(false) ||
475			ed25519::Signature::try_from(self.0.as_fixed_bytes().as_ref())
476				.map(|s| match ed25519::Public::from_slice(signer.as_ref()) {
477					Err(()) => false,
478					Ok(signer) => s.verify(msg, &signer),
479				})
480				.unwrap_or(false)
481	}
482}

784/// This type specifies the outcome of dispatching a call to a module.
785///
786/// In case of failure an error specific to the module is returned.
787///
788/// Failure of the module call dispatching doesn't invalidate the extrinsic and it is still included
789/// in the block, therefore all state changes performed by the dispatched call are still persisted.
790///
791/// For example, if the dispatching of an extrinsic involves inclusion fee payment then these
792/// changes are going to be preserved even if the call dispatched failed.
793pub type DispatchOutcome = Result<(), DispatchError>;
794
795/// The result of applying of an extrinsic.
796///
797/// This type is typically used in the context of `BlockBuilder` to signal that the extrinsic
798/// in question cannot be included.
799///
800/// A block containing extrinsics that have a negative inclusion outcome is invalid. A negative
801/// result can only occur during the block production, where such extrinsics are detected and
802/// removed from the block that is being created and the transaction pool.
803///
804/// To rehash: every extrinsic in a valid block must return a positive `ApplyExtrinsicResult`.
805///
806/// Examples of reasons preventing inclusion in a block:
807/// - More block weight is required to process the extrinsic than is left in the block being built.
808///   This doesn't necessarily mean that the extrinsic is invalid, since it can still be included in
809///   the next block if it has enough spare weight available.
810/// - The sender doesn't have enough funds to pay the transaction inclusion fee. Including such a
811///   transaction in the block doesn't make sense.
812/// - The extrinsic supplied a bad signature. This transaction won't become valid ever.
813pub type ApplyExtrinsicResult =
814	Result<DispatchOutcome, transaction_validity::TransactionValidityError>;
815
816/// Same as `ApplyExtrinsicResult` but augmented with `PostDispatchInfo` on success.
817pub type ApplyExtrinsicResultWithInfo<T> =
818	Result<DispatchResultWithInfo<T>, transaction_validity::TransactionValidityError>;

823/// Verify a signature on an encoded value in a lazy manner. This can be
824/// an optimization if the signature scheme has an "unsigned" escape hash.
825pub fn verify_encoded_lazy<V: Verify, T: codec::Encode>(
826	sig: &V,
827	item: &T,
828	signer: &<V::Signer as IdentifyAccount>::AccountId,
829) -> bool {
830	// The `Lazy<T>` trait expresses something like `X: FnMut<Output = for<'a> &'a T>`.
831	// unfortunately this is a lifetime relationship that can't
832	// be expressed without generic associated types, better unification of HRTBs in type position,
833	// and some kind of integration into the Fn* traits.
834	struct LazyEncode<F> {
835		inner: F,
836		encoded: Option<Vec<u8>>,
837	}
838
839	impl<F: Fn() -> Vec<u8>> traits::Lazy<[u8]> for LazyEncode<F> {
840		fn get(&mut self) -> &[u8] {
841			self.encoded.get_or_insert_with(&self.inner).as_slice()
842		}
843	}
844
845	sig.verify(LazyEncode { inner: || item.encode(), encoded: None }, signer)
846}

848/// Checks that `$x` is equal to `$y` with an error rate of `$error`.
849///
850/// # Example
851///
852/// ```rust
853/// # fn main() {
854/// sp_runtime::assert_eq_error_rate!(10, 10, 0);
855/// sp_runtime::assert_eq_error_rate!(10, 11, 1);
856/// sp_runtime::assert_eq_error_rate!(12, 10, 2);
857/// # }
858/// ```
859///
860/// ```rust,should_panic
861/// # fn main() {
862/// sp_runtime::assert_eq_error_rate!(12, 10, 1);
863/// # }
864/// ```
865#[macro_export]
866#[cfg(feature = "std")]
867macro_rules! assert_eq_error_rate {
868	($x:expr, $y:expr, $error:expr $(,)?) => {
869		assert!(
870			($x >= $crate::Saturating::saturating_sub($y, $error)) &&
871				($x <= $crate::Saturating::saturating_add($y, $error)),
872			"{:?} != {:?} (with error rate {:?})",
873			$x,
874			$y,
875			$error,
876		);
877	};
878}
879
880/// Same as [`assert_eq_error_rate`], but intended to be used with floating point number, or
881/// generally those who do not have over/underflow potentials.
882#[macro_export]
883#[cfg(feature = "std")]
884macro_rules! assert_eq_error_rate_float {
885	($x:expr, $y:expr, $error:expr $(,)?) => {
886		assert!(
887			($x >= $y - $error) && ($x <= $y + $error),
888			"{:?} != {:?} (with error rate {:?})",
889			$x,
890			$y,
891			$error,
892		);
893	};
894}

952/// Utility function to declare string literals backed by an array of length N.
953///
954/// The input can be shorter than N, in that case the end of the array is padded with zeros.
955///
956/// [`str_array`] is useful when converting strings that end up in the storage as fixed size arrays
957/// or in const contexts where static data types have strings that could also end up in the storage.
958///
959/// # Example
960///
961/// ```rust
962/// # use sp_runtime::str_array;
963/// const MY_STR: [u8; 6] = str_array("data");
964/// assert_eq!(MY_STR, *b"data\0\0");
965/// ```
966pub const fn str_array<const N: usize>(s: &str) -> [u8; N] {
967	debug_assert!(s.len() <= N, "String literal doesn't fit in array");
968	let mut i = 0;
969	let mut arr = [0; N];
970	let s = s.as_bytes();
971	while i < s.len() {
972		arr[i] = s[i];
973		i += 1;
974	}
975	arr
976}
