16/*!
17This module contains constants.
18
19The constants have the maximum precision possible for a fixed-point
20number, and are rounded down at that precision.
21
22# Examples
23
24```rust
25use substrate_fixed::{consts, types::I4F28};
26let tau = I4F28::from_num(consts::TAU);
27println!("τ = 2π with eight binary places is {:.8b}", tau);
28assert_eq!(format!("{:.8b}", tau), "110.01001000");
29println!("τ = 2π with six decimal places is {:.8}", tau);
30assert_eq!(format!("{:.8}", tau), "6.28318531");
31```
32*/

301    fn compare_parse<F: Fixed>(f: F, s: &str)
302    where
303        F::Bits: TryFrom<u8>,
304    {
305        let sf = F::from_str(s).unwrap();
306        let f_plus = f + F::from_bits(F::Bits::try_from(1).ok().unwrap());
307        assert!(f <= sf && sf <= f_plus);
308    }
