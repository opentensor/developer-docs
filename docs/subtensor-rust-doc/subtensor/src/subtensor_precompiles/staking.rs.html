1// The goal of staking precompile is to allow interaction between EVM users and smart contracts and
2// subtensor staking functionality, namely add_stake, and remove_stake extrinsicsk, as well as the
3// staking state.
4//
5// Additional requirement is to preserve compatibility with Ethereum indexers, which requires
6// no balance transfers from EVM accounts without a corresponding transaction that can be
7// parsed by an indexer.
8//
9// Implementation of add_stake:
10//   - User transfers balance that will be staked to the precompile address with a payable
11//     method addStake. This method also takes hotkey public key (bytes32) of the hotkey
12//     that the stake should be assigned to.
13//   - Precompile transfers the balance back to the signing address, and then invokes
14//     do_add_stake from subtensor pallet with signing origin that mmatches to HashedAddressMapping
15//     of the message sender, which will effectively withdraw and stake balance from the message
16//     sender.
17//   - Precompile checks the result of do_add_stake and, in case of a failure, reverts the transaction,
18//     and leaves the balance on the message sender account.
19//
20// Implementation of remove_stake:
21//   - User involkes removeStake method and specifies hotkey public key (bytes32) of the hotkey
22//     to remove stake from, and the amount to unstake.
23//   - Precompile calls do_remove_stake method of the subtensor pallet with the signing origin of message
24//     sender, which effectively unstakes the specified amount and credits it to the message sender
25//   - Precompile checks the result of do_remove_stake and, in case of a failure, reverts the transaction.
26//

373    #[precompile::public("addStake(bytes32,uint256)")]
374    #[precompile::payable]
375    fn add_stake(handle: &mut impl PrecompileHandle, address: H256, netuid: U256) -> EvmResult<()> {
376        let account_id = handle.caller_account_id::<R>();
377        let amount = handle.context().apparent_value;
378
379        if !amount.is_zero() {
380            Self::transfer_back_to_caller(&account_id, amount)?;
381        }
382
383        let amount_sub = handle.try_convert_apparent_value::<R>()?;
384        let hotkey = R::AccountId::from(address.0);
385        let netuid = try_u16_from_u256(netuid)?;
386        let call = pallet_subtensor::Call::<R>::add_stake {
387            hotkey,
388            netuid,
389            amount_staked: amount_sub.unique_saturated_into(),
390        };
391
392        handle.try_dispatch_runtime_call::<R, _>(call, RawOrigin::Signed(account_id))
393    }

507    fn transfer_back_to_caller(
508        account_id: &<R as frame_system::Config>::AccountId,
509        amount: U256,
510    ) -> Result<(), PrecompileFailure> {
511        let amount = EvmBalance::new(amount);
512        let amount_sub =
513            <R as pallet_evm::Config>::BalanceConverter::into_substrate_balance(amount)
514                .ok_or(ExitError::OutOfFund)?;
515
516        // Create a transfer call from the smart contract to the caller
517        let value = amount_sub
518            .into_u64_saturating()
519            .try_into()
520            .map_err(|_| ExitError::Other("Failed to convert u64 to Balance".into()))?;
521        let transfer_call = <R as frame_system::Config>::RuntimeCall::from(
522            pallet_balances::Call::<R>::transfer_allow_death {
523                dest: account_id.clone().into(),
524                value,
525            },
526        );
527
528        // Execute the transfer
529        let transfer_result = transfer_call.dispatch(RawOrigin::Signed(Self::account_id()).into());
530
531        if let Err(dispatch_error) = transfer_result {
532            log::error!(
533                "Transfer back to caller failed. Error: {:?}",
534                dispatch_error
535            );
536            return Err(PrecompileFailure::Error {
537                exit_status: ExitError::Other("Transfer back to caller failed".into()),
538            });
539        }
540
541        Ok(())
542    }

545fn try_u16_from_u256(value: U256) -> Result<u16, PrecompileFailure> {
546    value.try_into().map_err(|_| PrecompileFailure::Error {
547        exit_status: ExitError::Other("the value is outside of u16 bounds".into()),
548    })
