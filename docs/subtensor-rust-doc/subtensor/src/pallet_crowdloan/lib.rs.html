53/// A struct containing the information about a crowdloan.
54#[freeze_struct("5db9538284491545")]
55#[derive(Encode, Decode, Eq, PartialEq, Ord, PartialOrd, RuntimeDebug, TypeInfo, MaxEncodedLen)]
56pub struct CrowdloanInfo<AccountId, Balance, BlockNumber, Call> {
57    /// The creator of the crowdloan.
58    pub creator: AccountId,
59    /// The initial deposit of the crowdloan from the creator.
60    pub deposit: Balance,
61    /// Minimum contribution to the crowdloan.
62    pub min_contribution: Balance,
63    /// The end block of the crowdloan.
64    pub end: BlockNumber,
65    /// The cap to raise.
66    pub cap: Balance,
67    /// The account holding the funds for this crowdloan. Derived on chain but put here for ease of use.
68    pub funds_account: AccountId,
69    /// The amount raised so far.
70    pub raised: Balance,
71    /// The optional target address to transfer the raised funds to, if not
72    /// provided, it means the funds will be transferred from on chain logic
73    /// inside the provided call to dispatch.
74    pub target_address: Option<AccountId>,
75    /// The optional call to dispatch when the crowdloan is finalized.
76    pub call: Option<Call>,
77    /// Whether the crowdloan has been finalized.
78    pub finalized: bool,
79    /// The number of contributors to the crowdloan.
80    pub contributors_count: u32,
81}

289    #[pallet::call]
290    impl<T: Config> Pallet<T> {
291        /// Create a crowdloan that will raise funds up to a maximum cap and if successful,
292        /// will transfer funds to the target address if provided and dispatch the call
293        /// (using creator origin).
294        ///
295        /// The initial deposit will be transfered to the crowdloan account and will be refunded
296        /// in case the crowdloan fails to raise the cap. Additionally, the creator will pay for
297        /// the execution of the call.
298        ///
299        /// The dispatch origin for this call must be _Signed_.
300        ///
301        /// Parameters:
302        /// - `deposit`: The initial deposit from the creator.
303        /// - `min_contribution`: The minimum contribution required to contribute to the crowdloan.
304        /// - `cap`: The maximum amount of funds that can be raised.
305        /// - `end`: The block number at which the crowdloan will end.
306        /// - `call`: The call to dispatch when the crowdloan is finalized.
307        /// - `target_address`: The address to transfer the raised funds to if provided.
308        #[pallet::call_index(0)]
309        #[pallet::weight({
310			let di = call.as_ref().map(|c| c.get_dispatch_info());
311			let inner_call_weight = match di {
312				Some(di) => di.weight,
313				None => Weight::zero(),
314			};
315			let base_weight = T::WeightInfo::create();
316			(base_weight.saturating_add(inner_call_weight), Pays::Yes)
317		})]
318        pub fn create(
319            origin: OriginFor<T>,
320            #[pallet::compact] deposit: BalanceOf<T>,
321            #[pallet::compact] min_contribution: BalanceOf<T>,
322            #[pallet::compact] cap: BalanceOf<T>,
323            #[pallet::compact] end: BlockNumberFor<T>,
324            call: Option<Box<<T as Config>::RuntimeCall>>,
325            target_address: Option<T::AccountId>,
326        ) -> DispatchResult {
327            let creator = ensure_signed(origin)?;
328            let now = frame_system::Pallet::<T>::block_number();
329
330            // Ensure the deposit is at least the minimum deposit, cap is greater than deposit
331            // and the minimum contribution is greater than the absolute minimum contribution.
332            ensure!(
333                deposit >= T::MinimumDeposit::get(),
334                Error::<T>::DepositTooLow
335            );
336            ensure!(cap > deposit, Error::<T>::CapTooLow);
337            ensure!(
338                min_contribution >= T::AbsoluteMinimumContribution::get(),
339                Error::<T>::MinimumContributionTooLow
340            );
341
342            Self::ensure_valid_end(now, end)?;
343
344            // Ensure the creator has enough balance to pay the initial deposit
345            ensure!(
346                CurrencyOf::<T>::balance(&creator) >= deposit,
347                Error::<T>::InsufficientBalance
348            );
349
350            let crowdloan_id = NextCrowdloanId::<T>::get();
351            let next_crowdloan_id = crowdloan_id.checked_add(1).ok_or(Error::<T>::Overflow)?;
352            NextCrowdloanId::<T>::put(next_crowdloan_id);
353
354            // Derive the funds account and keep track of it
355            let funds_account = Self::funds_account(crowdloan_id);
356            frame_system::Pallet::<T>::inc_providers(&funds_account);
357
358            // If the call is provided, bound it and store it in the preimage storage
359            let call = if let Some(call) = call {
360                Some(T::Preimages::bound(*call)?)
361            } else {
362                None
363            };
364
365            let crowdloan = CrowdloanInfo {
366                creator: creator.clone(),
367                deposit,
368                min_contribution,
369                end,
370                cap,
371                funds_account,
372                raised: deposit,
373                target_address,
374                call,
375                finalized: false,
376                contributors_count: 1,
377            };
378            Crowdloans::<T>::insert(crowdloan_id, &crowdloan);
379
380            // Transfer the deposit to the funds account
381            CurrencyOf::<T>::transfer(
382                &creator,
383                &crowdloan.funds_account,
384                deposit,
385                Preservation::Expendable,
386            )?;
387
388            Contributions::<T>::insert(crowdloan_id, &creator, deposit);
389
390            Self::deposit_event(Event::<T>::Created {
391                crowdloan_id,
392                creator,
393                end,
394                cap,
395            });
396
397            Ok(())
398        }
399
400        /// Contribute to an active crowdloan.
401        ///
402        /// The contribution will be transfered to the crowdloan account and will be refunded
403        /// if the crowdloan fails to raise the cap. If the contribution would raise the amount above the cap,
404        /// the contribution will be set to the amount that is left to be raised.
405        ///
406        /// The dispatch origin for this call must be _Signed_.
407        ///
408        /// Parameters:
409        /// - `crowdloan_id`: The id of the crowdloan to contribute to.
410        /// - `amount`: The amount to contribute.
411        #[pallet::call_index(1)]
412        #[pallet::weight(T::WeightInfo::contribute())]
413        pub fn contribute(
414            origin: OriginFor<T>,
415            #[pallet::compact] crowdloan_id: CrowdloanId,
416            #[pallet::compact] amount: BalanceOf<T>,
417        ) -> DispatchResult {
418            let contributor = ensure_signed(origin)?;
419            let now = frame_system::Pallet::<T>::block_number();
420
421            let mut crowdloan = Self::ensure_crowdloan_exists(crowdloan_id)?;
422
423            // Ensure crowdloan has not ended and has not raised cap
424            ensure!(now < crowdloan.end, Error::<T>::ContributionPeriodEnded);
425            ensure!(crowdloan.raised < crowdloan.cap, Error::<T>::CapRaised);
426
427            // Ensure contribution is at least the minimum contribution
428            ensure!(
429                amount >= crowdloan.min_contribution,
430                Error::<T>::ContributionTooLow
431            );
432
433            // Ensure the crowdloan has not reached the maximum number of contributors
434            ensure!(
435                crowdloan.contributors_count < T::MaxContributors::get(),
436                Error::<T>::MaxContributorsReached
437            );
438
439            // Ensure contribution does not overflow the actual raised amount
440            // and it does not exceed the cap
441            let left_to_raise = crowdloan
442                .cap
443                .checked_sub(crowdloan.raised)
444                .ok_or(Error::<T>::Underflow)?;
445
446            // If the contribution would raise the amount above the cap,
447            // set the contribution to the amount that is left to be raised
448            let amount = amount.min(left_to_raise);
449
450            // Ensure contribution does not overflow the actual raised amount
451            crowdloan.raised = crowdloan
452                .raised
453                .checked_add(amount)
454                .ok_or(Error::<T>::Overflow)?;
455
456            // Compute the new total contribution and ensure it does not overflow, we
457            // also increment the contributor count if the contribution is new.
458            let contribution =
459                if let Some(contribution) = Contributions::<T>::get(crowdloan_id, &contributor) {
460                    contribution
461                        .checked_add(amount)
462                        .ok_or(Error::<T>::Overflow)?
463                } else {
464                    // We have a new contribution
465                    crowdloan.contributors_count = crowdloan
466                        .contributors_count
467                        .checked_add(1)
468                        .ok_or(Error::<T>::Overflow)?;
469                    amount
470                };
471
472            // Ensure contributor has enough balance to pay
473            ensure!(
474                CurrencyOf::<T>::balance(&contributor) >= amount,
475                Error::<T>::InsufficientBalance
476            );
477
478            CurrencyOf::<T>::transfer(
479                &contributor,
480                &crowdloan.funds_account,
481                amount,
482                Preservation::Expendable,
483            )?;
484
485            Contributions::<T>::insert(crowdloan_id, &contributor, contribution);
486            Crowdloans::<T>::insert(crowdloan_id, &crowdloan);
487
488            Self::deposit_event(Event::<T>::Contributed {
489                crowdloan_id,
490                contributor,
491                amount,
492            });
493
494            Ok(())
495        }
496
497        /// Withdraw a contribution from an active (not yet finalized or dissolved) crowdloan.
498        ///
499        /// Only contributions over the deposit can be withdrawn by the creator.
500        ///
501        /// The dispatch origin for this call must be _Signed_.
502        ///
503        /// Parameters:
504        /// - `crowdloan_id`: The id of the crowdloan to withdraw from.
505        #[pallet::call_index(2)]
506        #[pallet::weight(T::WeightInfo::withdraw())]
507        pub fn withdraw(
508            origin: OriginFor<T>,
509            #[pallet::compact] crowdloan_id: CrowdloanId,
510        ) -> DispatchResult {
511            let who = ensure_signed(origin)?;
512
513            let mut crowdloan = Self::ensure_crowdloan_exists(crowdloan_id)?;
514            ensure!(!crowdloan.finalized, Error::<T>::AlreadyFinalized);
515
516            // Ensure contributor has balance left in the crowdloan account
517            let mut amount = Contributions::<T>::get(crowdloan_id, &who).unwrap_or_else(Zero::zero);
518            ensure!(amount > Zero::zero(), Error::<T>::NoContribution);
519
520            if who == crowdloan.creator {
521                // Ensure the deposit is kept
522                amount = amount.saturating_sub(crowdloan.deposit);
523                ensure!(amount > Zero::zero(), Error::<T>::DepositCannotBeWithdrawn);
524                Contributions::<T>::insert(crowdloan_id, &who, crowdloan.deposit);
525            } else {
526                Contributions::<T>::remove(crowdloan_id, &who);
527                crowdloan.contributors_count = crowdloan
528                    .contributors_count
529                    .checked_sub(1)
530                    .ok_or(Error::<T>::Underflow)?;
531            }
532
533            CurrencyOf::<T>::transfer(
534                &crowdloan.funds_account,
535                &who,
536                amount,
537                Preservation::Expendable,
538            )?;
539
540            // Update the crowdloan raised amount to reflect the withdrawal.
541            crowdloan.raised = crowdloan.raised.saturating_sub(amount);
542            Crowdloans::<T>::insert(crowdloan_id, &crowdloan);
543
544            Self::deposit_event(Event::<T>::Withdrew {
545                contributor: who,
546                crowdloan_id,
547                amount,
548            });
549
550            Ok(())
551        }
552
553        /// Finalize a successful crowdloan.
554        ///
555        /// The call will transfer the raised amount to the target address if it was provided when the crowdloan was created
556        /// and dispatch the call that was provided using the creator origin. The CurrentCrowdloanId will be set to the
557        /// crowdloan id being finalized so the dispatched call can access it temporarily by accessing
558        /// the `CurrentCrowdloanId` storage item.
559        ///
560        /// The dispatch origin for this call must be _Signed_ and must be the creator of the crowdloan.
561        ///
562        /// Parameters:
563        /// - `crowdloan_id`: The id of the crowdloan to finalize.
564        #[pallet::call_index(3)]
565        #[pallet::weight(T::WeightInfo::finalize())]
566        pub fn finalize(
567            origin: OriginFor<T>,
568            #[pallet::compact] crowdloan_id: CrowdloanId,
569        ) -> DispatchResult {
570            let who = ensure_signed(origin)?;
571            let now = frame_system::Pallet::<T>::block_number();
572
573            let mut crowdloan = Self::ensure_crowdloan_exists(crowdloan_id)?;
574
575            // Ensure the origin is the creator of the crowdloan and the crowdloan has ended,
576            // raised the cap and is not finalized.
577            ensure!(who == crowdloan.creator, Error::<T>::InvalidOrigin);
578            ensure!(now >= crowdloan.end, Error::<T>::ContributionPeriodNotEnded);
579            ensure!(crowdloan.raised == crowdloan.cap, Error::<T>::CapNotRaised);
580            ensure!(!crowdloan.finalized, Error::<T>::AlreadyFinalized);
581
582            // If the target address is provided, transfer the raised amount to it.
583            if let Some(ref target_address) = crowdloan.target_address {
584                CurrencyOf::<T>::transfer(
585                    &crowdloan.funds_account,
586                    target_address,
587                    crowdloan.raised,
588                    Preservation::Expendable,
589                )?;
590            }
591
592            // If the call is provided, dispatch it.
593            if let Some(ref call) = crowdloan.call {
594                // Set the current crowdloan id so the dispatched call
595                // can access it temporarily
596                CurrentCrowdloanId::<T>::put(crowdloan_id);
597
598                // Retrieve the call from the preimage storage
599                let stored_call = match T::Preimages::peek(call) {
600                    Ok((call, _)) => call,
601                    Err(_) => {
602                        // If the call is not found, we drop it from the preimage storage
603                        // because it's not needed anymore
604                        T::Preimages::drop(call);
605                        return Err(Error::<T>::CallUnavailable)?;
606                    }
607                };
608
609                // Dispatch the call with creator origin
610                stored_call
611                    .dispatch(frame_system::RawOrigin::Signed(who).into())
612                    .map(|_| ())
613                    .map_err(|e| e.error)?;
614
615                // Clear the current crowdloan id
616                CurrentCrowdloanId::<T>::kill();
617            }
618
619            crowdloan.finalized = true;
620            Crowdloans::<T>::insert(crowdloan_id, &crowdloan);
621
622            Self::deposit_event(Event::<T>::Finalized { crowdloan_id });
623
624            Ok(())
625        }
626
627        /// Refund a failed crowdloan.
628        ///
629        /// The call will try to refund all contributors (excluding the creator) up to the limit defined by the `RefundContributorsLimit`.
630        /// If the limit is reached, the call will stop and the crowdloan will be marked as partially refunded.
631        /// It may be needed to dispatch this call multiple times to refund all contributors.
632        ///
633        /// The dispatch origin for this call must be _Signed_ and doesn't need to be the creator of the crowdloan.
634        ///
635        /// Parameters:
636        /// - `crowdloan_id`: The id of the crowdloan to refund.
637        #[pallet::call_index(4)]
638        #[pallet::weight(T::WeightInfo::refund(T::RefundContributorsLimit::get()))]
639        pub fn refund(
640            origin: OriginFor<T>,
641            #[pallet::compact] crowdloan_id: CrowdloanId,
642        ) -> DispatchResultWithPostInfo {
643            let now = frame_system::Pallet::<T>::block_number();
644            ensure_signed(origin)?;
645
646            let mut crowdloan = Self::ensure_crowdloan_exists(crowdloan_id)?;
647
648            // Ensure the crowdloan has ended and is not finalized
649            ensure!(now >= crowdloan.end, Error::<T>::ContributionPeriodNotEnded);
650            ensure!(!crowdloan.finalized, Error::<T>::AlreadyFinalized);
651
652            let mut refunded_contributors: Vec<T::AccountId> = vec![];
653            let mut refund_count = 0;
654
655            // Assume everyone can be refunded
656            let mut all_refunded = true;
657
658            // We try to refund all contributors (excluding the creator)
659            let contributions = Contributions::<T>::iter_prefix(crowdloan_id)
660                .filter(|(contributor, _)| *contributor != crowdloan.creator);
661            for (contributor, amount) in contributions {
662                if refund_count >= T::RefundContributorsLimit::get() {
663                    // Not everyone can be refunded
664                    all_refunded = false;
665                    break;
666                }
667
668                CurrencyOf::<T>::transfer(
669                    &crowdloan.funds_account,
670                    &contributor,
671                    amount,
672                    Preservation::Expendable,
673                )?;
674
675                refunded_contributors.push(contributor);
676                crowdloan.raised = crowdloan.raised.saturating_sub(amount);
677                refund_count = refund_count.checked_add(1).ok_or(Error::<T>::Overflow)?;
678            }
679
680            crowdloan.contributors_count = crowdloan
681                .contributors_count
682                .checked_sub(refund_count)
683                .ok_or(Error::<T>::Underflow)?;
684            Crowdloans::<T>::insert(crowdloan_id, &crowdloan);
685
686            // Clear refunded contributors
687            for contributor in refunded_contributors {
688                Contributions::<T>::remove(crowdloan_id, &contributor);
689            }
690
691            if all_refunded {
692                Self::deposit_event(Event::<T>::AllRefunded { crowdloan_id });
693                // The loop didn't run fully, we refund the unused weights.
694                Ok(Some(T::WeightInfo::refund(refund_count)).into())
695            } else {
696                Self::deposit_event(Event::<T>::PartiallyRefunded { crowdloan_id });
697                // The loop ran fully, we don't refund anything.
698                Ok(().into())
699            }
700        }
701
702        /// Dissolve a crowdloan.
703        ///
704        /// The crowdloan will be removed from the storage.
705        /// All contributions must have been refunded before the crowdloan can be dissolved (except the creator's one).
706        ///
707        /// The dispatch origin for this call must be _Signed_ and must be the creator of the crowdloan.
708        ///
709        /// Parameters:
710        /// - `crowdloan_id`: The id of the crowdloan to dissolve.
711        #[pallet::call_index(5)]
712        #[pallet::weight(T::WeightInfo::dissolve())]
713        pub fn dissolve(
714            origin: OriginFor<T>,
715            #[pallet::compact] crowdloan_id: CrowdloanId,
716        ) -> DispatchResult {
717            let who = ensure_signed(origin)?;
718
719            let crowdloan = Self::ensure_crowdloan_exists(crowdloan_id)?;
720            ensure!(!crowdloan.finalized, Error::<T>::AlreadyFinalized);
721
722            // Only the creator can dissolve the crowdloan
723            ensure!(who == crowdloan.creator, Error::<T>::InvalidOrigin);
724
725            // It can only be dissolved if the raised amount is the creator's contribution,
726            // meaning there is no contributions or every contribution has been refunded
727            let creator_contribution = Contributions::<T>::get(crowdloan_id, &crowdloan.creator)
728                .ok_or(Error::<T>::NoContribution)?;
729            ensure!(
730                creator_contribution == crowdloan.raised,
731                Error::<T>::NotReadyToDissolve
732            );
733
734            // Refund the creator's contribution
735            CurrencyOf::<T>::transfer(
736                &crowdloan.funds_account,
737                &crowdloan.creator,
738                creator_contribution,
739                Preservation::Expendable,
740            )?;
741            Contributions::<T>::remove(crowdloan_id, &crowdloan.creator);
742
743            // Clear the call from the preimage storage
744            if let Some(call) = crowdloan.call {
745                T::Preimages::drop(&call);
746            }
747
748            // Remove the crowdloan
749            let _ = frame_system::Pallet::<T>::dec_providers(&crowdloan.funds_account).defensive();
750            Crowdloans::<T>::remove(crowdloan_id);
751
752            Self::deposit_event(Event::<T>::Dissolved { crowdloan_id });
753            Ok(())
754        }
755
756        /// Update the minimum contribution of a non-finalized crowdloan.
757        ///
758        /// The dispatch origin for this call must be _Signed_ and must be the creator of the crowdloan.
759        ///
760        /// Parameters:
761        /// - `crowdloan_id`: The id of the crowdloan to update the minimum contribution of.
762        /// - `new_min_contribution`: The new minimum contribution.
763        #[pallet::call_index(6)]
764        #[pallet::weight(T::WeightInfo::update_min_contribution())]
765        pub fn update_min_contribution(
766            origin: OriginFor<T>,
767            #[pallet::compact] crowdloan_id: CrowdloanId,
768            #[pallet::compact] new_min_contribution: BalanceOf<T>,
769        ) -> DispatchResult {
770            let who = ensure_signed(origin)?;
771
772            let mut crowdloan = Self::ensure_crowdloan_exists(crowdloan_id)?;
773            ensure!(!crowdloan.finalized, Error::<T>::AlreadyFinalized);
774
775            // Only the creator can update the min contribution.
776            ensure!(who == crowdloan.creator, Error::<T>::InvalidOrigin);
777
778            // The new min contribution should be greater than absolute minimum contribution.
779            ensure!(
780                new_min_contribution >= T::AbsoluteMinimumContribution::get(),
781                Error::<T>::MinimumContributionTooLow
782            );
783
784            crowdloan.min_contribution = new_min_contribution;
785            Crowdloans::<T>::insert(crowdloan_id, &crowdloan);
786
787            Self::deposit_event(Event::<T>::MinContributionUpdated {
788                crowdloan_id,
789                new_min_contribution,
790            });
791            Ok(())
792        }
793
794        /// Update the end block of a non-finalized crowdloan.
795        ///
796        /// The dispatch origin for this call must be _Signed_ and must be the creator of the crowdloan.
797        ///
798        /// Parameters:
799        /// - `crowdloan_id`: The id of the crowdloan to update the end block of.
800        /// - `new_end`: The new end block.
801        #[pallet::call_index(7)]
802        #[pallet::weight(T::WeightInfo::update_end())]
803        pub fn update_end(
804            origin: OriginFor<T>,
805            #[pallet::compact] crowdloan_id: CrowdloanId,
806            #[pallet::compact] new_end: BlockNumberFor<T>,
807        ) -> DispatchResult {
808            let who = ensure_signed(origin)?;
809            let now = frame_system::Pallet::<T>::block_number();
810
811            let mut crowdloan = Self::ensure_crowdloan_exists(crowdloan_id)?;
812            ensure!(!crowdloan.finalized, Error::<T>::AlreadyFinalized);
813
814            // Only the creator can update the min contribution.
815            ensure!(who == crowdloan.creator, Error::<T>::InvalidOrigin);
816
817            Self::ensure_valid_end(now, new_end)?;
818
819            crowdloan.end = new_end;
820            Crowdloans::<T>::insert(crowdloan_id, &crowdloan);
821
822            Self::deposit_event(Event::<T>::EndUpdated {
823                crowdloan_id,
824                new_end,
825            });
826            Ok(())
827        }
828
829        /// Update the cap of a non-finalized crowdloan.
830        ///
831        /// The dispatch origin for this call must be _Signed_ and must be the creator of the crowdloan.
832        ///
833        /// Parameters:
834        /// - `crowdloan_id`: The id of the crowdloan to update the cap of.
835        /// - `new_cap`: The new cap.
836        #[pallet::call_index(8)]
837        #[pallet::weight(T::WeightInfo::update_cap())]
838        pub fn update_cap(
839            origin: OriginFor<T>,
840            #[pallet::compact] crowdloan_id: CrowdloanId,
841            #[pallet::compact] new_cap: BalanceOf<T>,
842        ) -> DispatchResult {
843            let who = ensure_signed(origin)?;
844
845            // The cap can only be updated if the crowdloan has not been finalized.
846            let mut crowdloan = Self::ensure_crowdloan_exists(crowdloan_id)?;
847            ensure!(!crowdloan.finalized, Error::<T>::AlreadyFinalized);
848
849            // Only the creator can update the cap.
850            ensure!(who == crowdloan.creator, Error::<T>::InvalidOrigin);
851
852            // The new cap should be greater than the actual raised amount.
853            ensure!(new_cap >= crowdloan.raised, Error::<T>::CapTooLow);
854
855            crowdloan.cap = new_cap;
856            Crowdloans::<T>::insert(crowdloan_id, &crowdloan);
857
858            Self::deposit_event(Event::<T>::CapUpdated {
859                crowdloan_id,
860                new_cap,
861            });
862            Ok(())
863        }
864    }
865}
866
867impl<T: Config> Pallet<T> {
868    fn funds_account(id: CrowdloanId) -> T::AccountId {
869        T::PalletId::get().into_sub_account_truncating(id)
870    }
871
872    fn ensure_crowdloan_exists(crowdloan_id: CrowdloanId) -> Result<CrowdloanInfoOf<T>, Error<T>> {
873        Crowdloans::<T>::get(crowdloan_id).ok_or(Error::<T>::InvalidCrowdloanId)
874    }
875
876    // Ensure the provided end block is after the current block and the duration is
877    // between the minimum and maximum block duration
878    fn ensure_valid_end(now: BlockNumberFor<T>, end: BlockNumberFor<T>) -> Result<(), Error<T>> {
879        ensure!(now < end, Error::<T>::CannotEndInPast);
880        let block_duration = end.checked_sub(&now).ok_or(Error::<T>::Underflow)?;
881        ensure!(
882            block_duration >= T::MinimumBlockDuration::get(),
883            Error::<T>::BlockDurationTooShort
884        );
885        ensure!(
886            block_duration <= T::MaximumBlockDuration::get(),
887            Error::<T>::BlockDurationTooLong
888        );
889        Ok(())
890    }
891}
