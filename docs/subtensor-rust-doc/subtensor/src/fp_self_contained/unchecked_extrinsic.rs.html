35/// A extrinsic right from the external world. This is unchecked and so
36/// can contain a signature.
37#[derive(PartialEq, Eq, Clone, Encode, Decode, RuntimeDebug, TypeInfo)]
38pub struct UncheckedExtrinsic<Address, Call, Signature, Extra: SignedExtension>(
39	pub sp_runtime::generic::UncheckedExtrinsic<Address, Call, Signature, Extra>,
40);
41
42impl<Address, Call, Signature, Extra: SignedExtension>
43	UncheckedExtrinsic<Address, Call, Signature, Extra>
44{
45	/// New instance of a signed extrinsic aka "transaction".
46	pub fn new_signed(function: Call, signed: Address, signature: Signature, extra: Extra) -> Self {
47		Self(sp_runtime::generic::UncheckedExtrinsic::new_signed(
48			function, signed, signature, extra,
49		))
50	}
51
52	/// New instance of an unsigned extrinsic aka "inherent".
53	pub fn new_unsigned(function: Call) -> Self {
54		Self(sp_runtime::generic::UncheckedExtrinsic::new_unsigned(
55			function,
56		))
57	}
58}

60impl<Address, Call, Signature, Extra> Extrinsic
61	for UncheckedExtrinsic<Address, Call, Signature, Extra>
62where
63	Address: TypeInfo,
64	Call: SelfContainedCall + TypeInfo,
65	Signature: TypeInfo,
66	Extra: SignedExtension,
67{
68	type Call = Call;
69
70	type SignaturePayload = (Address, Signature, Extra);
71
72	fn is_signed(&self) -> Option<bool> {
73		if self.0.function.is_self_contained() {
74			Some(true)
75		} else {
76			self.0.is_signed()
77		}
78	}
79
80	fn new(function: Call, signed_data: Option<Self::SignaturePayload>) -> Option<Self> {
81		sp_runtime::generic::UncheckedExtrinsic::new(function, signed_data).map(Self)
82	}
83}

85impl<Address, AccountId, Call, Signature, Extra, Lookup> Checkable<Lookup>
86	for UncheckedExtrinsic<Address, Call, Signature, Extra>
87where
88	Address: Member + MaybeDisplay,
89	Call: Encode + Member + SelfContainedCall,
90	Signature: Member + traits::Verify,
91	<Signature as traits::Verify>::Signer: IdentifyAccount<AccountId = AccountId>,
92	Extra: SignedExtension<AccountId = AccountId>,
93	AccountId: Member + MaybeDisplay,
94	Lookup: traits::Lookup<Source = Address, Target = AccountId>,
95{
96	type Checked =
97		CheckedExtrinsic<AccountId, Call, Extra, <Call as SelfContainedCall>::SignedInfo>;
98
99	fn check(self, lookup: &Lookup) -> Result<Self::Checked, TransactionValidityError> {
100		if self.0.function.is_self_contained() {
101			if self.0.signature.is_some() {
102				return Err(TransactionValidityError::Invalid(
103					InvalidTransaction::BadProof,
104				));
105			}
106
107			let signed_info = self.0.function.check_self_contained().ok_or(
108				TransactionValidityError::Invalid(InvalidTransaction::BadProof),
109			)??;
110			Ok(CheckedExtrinsic {
111				signed: CheckedSignature::SelfContained(signed_info),
112				function: self.0.function,
113			})
114		} else {
115			let checked = Checkable::<Lookup>::check(self.0, lookup)?;
116			Ok(CheckedExtrinsic {
117				signed: match checked.signed {
118					Some((id, extra)) => CheckedSignature::Signed(id, extra),
119					None => CheckedSignature::Unsigned,
120				},
121				function: checked.function,
122			})
123		}
124	}
125
126	#[cfg(feature = "try-runtime")]
127	fn unchecked_into_checked_i_know_what_i_am_doing(
128		self,
129		lookup: &Lookup,
130	) -> Result<Self::Checked, TransactionValidityError> {
131		if self.0.function.is_self_contained() {
132			match self.0.function.check_self_contained() {
133				Some(signed_info) => Ok(CheckedExtrinsic {
134					signed: match signed_info {
135						Ok(info) => CheckedSignature::SelfContained(info),
136						_ => CheckedSignature::Unsigned,
137					},
138					function: self.0.function,
139				}),
140				None => Ok(CheckedExtrinsic {
141					signed: CheckedSignature::Unsigned,
142					function: self.0.function,
143				}),
144			}
145		} else {
146			let checked =
147				Checkable::<Lookup>::unchecked_into_checked_i_know_what_i_am_doing(self.0, lookup)?;
148			Ok(CheckedExtrinsic {
149				signed: match checked.signed {
150					Some((id, extra)) => CheckedSignature::Signed(id, extra),
151					None => CheckedSignature::Unsigned,
152				},
153				function: checked.function,
154			})
155		}
156	}
157}
