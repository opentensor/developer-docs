49/// check a header has been signed by the right key. If the slot is too far in the future, an error
50/// will be returned. If it's successful, returns the pre-header and the digest item
51/// containing the seal.
52///
53/// This digest item will always return `Some` when used with `as_aura_seal`.
54fn check_header<C, B: BlockT, P: Pair>(
55	client: &C,
56	slot_now: Slot,
57	header: B::Header,
58	hash: B::Hash,
59	authorities: &[AuthorityId<P>],
60	check_for_equivocation: CheckForEquivocation,
61) -> Result<CheckedHeader<B::Header, (Slot, DigestItem)>, Error<B>>
62where
63	P::Public: Codec,
64	P::Signature: Codec,
65	C: sc_client_api::backend::AuxStore,
66{
67	let check_result =
68		crate::standalone::check_header_slot_and_seal::<B, P>(slot_now, header, authorities);
69
70	match check_result {
71		Ok((header, slot, seal)) => {
72			let expected_author = crate::standalone::slot_author::<P>(slot, &authorities);
73			let should_equiv_check = check_for_equivocation.check_for_equivocation();
74			if let (true, Some(expected)) = (should_equiv_check, expected_author) {
75				if let Some(equivocation_proof) =
76					check_equivocation(client, slot_now, slot, &header, expected)
77						.map_err(Error::Client)?
78				{
79					info!(
80						target: LOG_TARGET,
81						"Slot author is equivocating at slot {} with headers {:?} and {:?}",
82						slot,
83						equivocation_proof.first_header.hash(),
84						equivocation_proof.second_header.hash(),
85					);
86				}
87			}
88
89			Ok(CheckedHeader::Checked(header, (slot, seal)))
90		},
91		Err(SealVerificationError::Deferred(header, slot)) =>
92			Ok(CheckedHeader::Deferred(header, slot)),
93		Err(SealVerificationError::Unsealed) => Err(Error::HeaderUnsealed(hash)),
94		Err(SealVerificationError::BadSeal) => Err(Error::HeaderBadSeal(hash)),
95		Err(SealVerificationError::BadSignature) => Err(Error::BadSignature(hash)),
96		Err(SealVerificationError::SlotAuthorNotFound) => Err(Error::SlotAuthorNotFound),
97		Err(SealVerificationError::InvalidPreDigest(e)) => Err(Error::from(e)),
98	}
99}

101/// A verifier for Aura blocks.
102pub struct AuraVerifier<C, P, CIDP, N> {
103	client: Arc<C>,
104	create_inherent_data_providers: CIDP,
105	check_for_equivocation: CheckForEquivocation,
106	telemetry: Option<TelemetryHandle>,
107	compatibility_mode: CompatibilityMode<N>,
108	_phantom: PhantomData<fn() -> P>,
109}
110
111impl<C, P, CIDP, N> AuraVerifier<C, P, CIDP, N> {
112	pub(crate) fn new(
113		client: Arc<C>,
114		create_inherent_data_providers: CIDP,
115		check_for_equivocation: CheckForEquivocation,
116		telemetry: Option<TelemetryHandle>,
117		compatibility_mode: CompatibilityMode<N>,
118	) -> Self {
119		Self {
120			client,
121			create_inherent_data_providers,
122			check_for_equivocation,
123			telemetry,
124			compatibility_mode,
125			_phantom: PhantomData,
126		}
127	}
128}

130impl<C, P, CIDP, N> AuraVerifier<C, P, CIDP, N>
131where
132	CIDP: Send,
133{
134	async fn check_inherents<B: BlockT>(
135		&self,
136		block: B,
137		at_hash: B::Hash,
138		inherent_data: sp_inherents::InherentData,
139		create_inherent_data_providers: CIDP::InherentDataProviders,
140	) -> Result<(), Error<B>>
141	where
142		C: ProvideRuntimeApi<B>,
143		C::Api: BlockBuilderApi<B>,
144		CIDP: CreateInherentDataProviders<B, ()>,
145	{
146		let inherent_res = self
147			.client
148			.runtime_api()
149			.check_inherents(at_hash, block, inherent_data)
150			.map_err(|e| Error::Client(e.into()))?;
151
152		if !inherent_res.ok() {
153			for (i, e) in inherent_res.into_errors() {
154				match create_inherent_data_providers.try_handle_error(&i, &e).await {
155					Some(res) => res.map_err(Error::Inherent)?,
156					None => return Err(Error::UnknownInherentError(i)),
157				}
158			}
159		}
160
161		Ok(())
162	}
163}

165#[async_trait::async_trait]
166impl<B: BlockT, C, P, CIDP> Verifier<B> for AuraVerifier<C, P, CIDP, NumberFor<B>>
167where
168	C: ProvideRuntimeApi<B> + Send + Sync + sc_client_api::backend::AuxStore,
169	C::Api: BlockBuilderApi<B> + AuraApi<B, AuthorityId<P>> + ApiExt<B>,
170	P: Pair,
171	P::Public: Codec + Debug,
172	P::Signature: Codec,
173	CIDP: CreateInherentDataProviders<B, ()> + Send + Sync,
174	CIDP::InherentDataProviders: InherentDataProviderExt + Send + Sync,
175{
176	async fn verify(
177		&self,
178		mut block: BlockImportParams<B>,
179	) -> Result<BlockImportParams<B>, String> {
180		// Skip checks that include execution, if being told so or when importing only state.
181		//
182		// This is done for example when gap syncing and it is expected that the block after the gap
183		// was checked/chosen properly, e.g. by warp syncing to this block using a finality proof.
184		// Or when we are importing state only and can not verify the seal.
185		if block.with_state() || block.state_action.skip_execution_checks() {
186			// When we are importing only the state of a block, it will be the best block.
187			block.fork_choice = Some(ForkChoiceStrategy::Custom(block.with_state()));
188
189			return Ok(block)
190		}
191
192		let hash = block.header.hash();
193		let parent_hash = *block.header.parent_hash();
194		let authorities = authorities(
195			self.client.as_ref(),
196			parent_hash,
197			*block.header.number(),
198			&self.compatibility_mode,
199		)
200		.map_err(|e| format!("Could not fetch authorities at {:?}: {}", parent_hash, e))?;
201
202		let create_inherent_data_providers = self
203			.create_inherent_data_providers
204			.create_inherent_data_providers(parent_hash, ())
205			.await
206			.map_err(|e| Error::<B>::Client(sp_blockchain::Error::Application(e)))?;
207
208		let mut inherent_data = create_inherent_data_providers
209			.create_inherent_data()
210			.await
211			.map_err(Error::<B>::Inherent)?;
212
213		let slot_now = create_inherent_data_providers.slot();
214
215		// we add one to allow for some small drift.
216		// FIXME #1019 in the future, alter this queue to allow deferring of
217		// headers
218		let checked_header = check_header::<C, B, P>(
219			&self.client,
220			slot_now + 1,
221			block.header,
222			hash,
223			&authorities[..],
224			self.check_for_equivocation,
225		)
226		.map_err(|e| e.to_string())?;
227		match checked_header {
228			CheckedHeader::Checked(pre_header, (slot, seal)) => {
229				// if the body is passed through, we need to use the runtime
230				// to check that the internally-set timestamp in the inherents
231				// actually matches the slot set in the seal.
232				if let Some(inner_body) = block.body.take() {
233					let new_block = B::new(pre_header.clone(), inner_body);
234
235					inherent_data.aura_replace_inherent_data(slot);
236
237					// skip the inherents verification if the runtime API is old or not expected to
238					// exist.
239					if self
240						.client
241						.runtime_api()
242						.has_api_with::<dyn BlockBuilderApi<B>, _>(parent_hash, |v| v >= 2)
243						.map_err(|e| e.to_string())?
244					{
245						self.check_inherents(
246							new_block.clone(),
247							parent_hash,
248							inherent_data,
249							create_inherent_data_providers,
250						)
251						.await
252						.map_err(|e| e.to_string())?;
253					}
254
255					let (_, inner_body) = new_block.deconstruct();
256					block.body = Some(inner_body);
257				}
258
259				trace!(target: LOG_TARGET, "Checked {:?}; importing.", pre_header);
260				telemetry!(
261					self.telemetry;
262					CONSENSUS_TRACE;
263					"aura.checked_and_importing";
264					"pre_header" => ?pre_header,
265				);
266
267				block.header = pre_header;
268				block.post_digests.push(seal);
269				block.fork_choice = Some(ForkChoiceStrategy::LongestChain);
270				block.post_hash = Some(hash);
271
272				Ok(block)
273			},
274			CheckedHeader::Deferred(a, b) => {
275				debug!(target: LOG_TARGET, "Checking {:?} failed; {:?}, {:?}.", hash, a, b);
276				telemetry!(
277					self.telemetry;
278					CONSENSUS_DEBUG;
279					"aura.header_too_far_in_future";
280					"hash" => ?hash,
281					"a" => ?a,
282					"b" => ?b,
283				);
284				Err(format!("Header {:?} rejected: too far in the future", hash))
285			},
286		}
287	}
288}

290/// Should we check for equivocation of a block author?
291#[derive(Debug, Clone, Copy)]
292pub enum CheckForEquivocation {
293	/// Yes, check for equivocation.
294	///
295	/// This is the default setting for this.
296	Yes,
297	/// No, don't check for equivocation.
298	No,
299}
300
301impl CheckForEquivocation {
302	/// Should we check for equivocation?
303	fn check_for_equivocation(self) -> bool {
304		matches!(self, Self::Yes)
305	}
306}
307
308impl Default for CheckForEquivocation {
309	fn default() -> Self {
310		Self::Yes
311	}
312}

314/// Parameters of [`import_queue`].
315pub struct ImportQueueParams<'a, Block: BlockT, I, C, S, CIDP> {
316	/// The block import to use.
317	pub block_import: I,
318	/// The justification import.
319	pub justification_import: Option<BoxJustificationImport<Block>>,
320	/// The client to interact with the chain.
321	pub client: Arc<C>,
322	/// Something that can create the inherent data providers.
323	pub create_inherent_data_providers: CIDP,
324	/// The spawner to spawn background tasks.
325	pub spawner: &'a S,
326	/// The prometheus registry.
327	pub registry: Option<&'a Registry>,
328	/// Should we check for equivocation?
329	pub check_for_equivocation: CheckForEquivocation,
330	/// Telemetry instance used to report telemetry metrics.
331	pub telemetry: Option<TelemetryHandle>,
332	/// Compatibility mode that should be used.
333	///
334	/// If in doubt, use `Default::default()`.
335	pub compatibility_mode: CompatibilityMode<NumberFor<Block>>,
336}

338/// Start an import queue for the Aura consensus algorithm.
339pub fn import_queue<P, Block, I, C, S, CIDP>(
340	ImportQueueParams {
341		block_import,
342		justification_import,
343		client,
344		create_inherent_data_providers,
345		spawner,
346		registry,
347		check_for_equivocation,
348		telemetry,
349		compatibility_mode,
350	}: ImportQueueParams<Block, I, C, S, CIDP>,
351) -> Result<DefaultImportQueue<Block>, sp_consensus::Error>
352where
353	Block: BlockT,
354	C::Api: BlockBuilderApi<Block> + AuraApi<Block, AuthorityId<P>> + ApiExt<Block>,
355	C: 'static
356		+ ProvideRuntimeApi<Block>
357		+ BlockOf
358		+ Send
359		+ Sync
360		+ AuxStore
361		+ UsageProvider<Block>
362		+ HeaderBackend<Block>,
363	I: BlockImport<Block, Error = ConsensusError> + Send + Sync + 'static,
364	P: Pair + 'static,
365	P::Public: Codec + Debug,
366	P::Signature: Codec,
367	S: sp_core::traits::SpawnEssentialNamed,
368	CIDP: CreateInherentDataProviders<Block, ()> + Sync + Send + 'static,
369	CIDP::InherentDataProviders: InherentDataProviderExt + Send + Sync,
370{
371	let verifier = build_verifier::<P, _, _, _>(BuildVerifierParams {
372		client,
373		create_inherent_data_providers,
374		check_for_equivocation,
375		telemetry,
376		compatibility_mode,
377	});
378
379	Ok(BasicQueue::new(verifier, Box::new(block_import), justification_import, spawner, registry))
380}

398/// Build the [`AuraVerifier`]
399pub fn build_verifier<P, C, CIDP, N>(
400	BuildVerifierParams {
401		client,
402		create_inherent_data_providers,
403		check_for_equivocation,
404		telemetry,
405		compatibility_mode,
406	}: BuildVerifierParams<C, CIDP, N>,

409		client,
410		create_inherent_data_providers,
411		check_for_equivocation,
412		telemetry,
413		compatibility_mode,
414	)
