19//! Aura (Authority-round) consensus in substrate.
20//!
21//! Aura works by having a list of authorities A who are expected to roughly
22//! agree on the current time. Time is divided up into discrete slots of t
23//! seconds each. For each slot s, the author of that slot is A[s % |A|].
24//!
25//! The author is allowed to issue one block but not more during that slot,
26//! and it will be built upon the longest valid chain that has been seen.
27//!
28//! Blocks from future steps will be either deferred or rejected depending on how
29//! far in the future they are.
30//!
31//! NOTE: Aura itself is designed to be generic over the crypto used.
32#![forbid(missing_docs, unsafe_code)]

75/// Run `AURA` in a compatibility mode.
76///
77/// This is required for when the chain was launched and later there
78/// was a consensus breaking change.
79#[derive(Debug, Clone)]
80pub enum CompatibilityMode<N> {
81	/// Don't use any compatibility mode.
82	None,
83	/// Call `initialize_block` before doing any runtime calls.
84	///
85	/// Previously the node would execute `initialize_block` before fetching the authorities
86	/// from the runtime. This behaviour changed in: <https://github.com/paritytech/substrate/pull/9132>
87	///
88	/// By calling `initialize_block` before fetching the authorities, on a block that
89	/// would enact a new validator set, the block would already be build/sealed by an
90	/// authority of the new set. With this mode disabled (the default) a block that enacts a new
91	/// set isn't sealed/built by an authority of the new set, however to make new nodes be able to
92	/// sync old chains this compatibility mode exists.
93	UseInitializeBlock {
94		/// The block number until this compatibility mode should be executed. The first runtime
95		/// call in the context of the `until` block (importing it/building it) will disable the
96		/// compatibility mode (i.e. at `until` the default rules will apply). When enabling this
97		/// compatibility mode the `until` block should be a future block on which all nodes will
98		/// have upgraded to a release that includes the updated compatibility mode configuration.
99		/// At `until` block there will be a hard fork when the authority set changes, between the
100		/// old nodes (running with `initialize_block`, i.e. without the compatibility mode
101		/// configuration) and the new nodes.
102		until: N,
103	},
104}
105
106impl<N> Default for CompatibilityMode<N> {
107	fn default() -> Self {
108		Self::None
109	}
110}

207		slot_duration,
208		select_chain,

326impl<B, C, E, I, P, Error, SO, L, BS> sc_consensus_slots::SimpleSlotWorker<B>
327	for AuraWorker<C, E, I, P, SO, L, BS, NumberFor<B>>
328where
329	B: BlockT,
330	C: ProvideRuntimeApi<B> + BlockOf + HeaderBackend<B> + Sync,
331	C::Api: AuraApi<B, AuthorityId<P>>,
332	E: Environment<B, Error = Error> + Send + Sync,
333	E::Proposer: Proposer<B, Error = Error>,
334	I: BlockImport<B> + Send + Sync + 'static,
335	P: Pair,
336	P::Public: AppPublic + Member,
337	P::Signature: TryFrom<Vec<u8>> + Member + Codec,
338	SO: SyncOracle + Send + Clone + Sync,
339	L: sc_consensus::JustificationSyncLink<B>,
340	BS: BackoffAuthoringBlocksStrategy<NumberFor<B>> + Send + Sync + 'static,
341	Error: std::error::Error + Send + From<ConsensusError> + 'static,
342{
343	type BlockImport = I;
344	type SyncOracle = SO;
345	type JustificationSyncLink = L;
346	type CreateProposer =
347		Pin<Box<dyn Future<Output = Result<E::Proposer, ConsensusError>> + Send + 'static>>;
348	type Proposer = E::Proposer;
349	type Claim = P::Public;
350	type AuxData = Vec<AuthorityId<P>>;
351
352	fn logging_target(&self) -> &'static str {
353		"aura"
354	}
355
356	fn block_import(&mut self) -> &mut Self::BlockImport {
357		&mut self.block_import
358	}
359
360	fn aux_data(&self, header: &B::Header, _slot: Slot) -> Result<Self::AuxData, ConsensusError> {
361		authorities(
362			self.client.as_ref(),
363			header.hash(),
364			*header.number() + 1u32.into(),
365			&self.compatibility_mode,
366		)
367	}
368
369	fn authorities_len(&self, authorities: &Self::AuxData) -> Option<usize> {
370		Some(authorities.len())
371	}
372
373	async fn claim_slot(
374		&mut self,
375		_header: &B::Header,
376		slot: Slot,
377		authorities: &Self::AuxData,
378	) -> Option<Self::Claim> {
379		crate::standalone::claim_slot::<P>(slot, authorities, &self.keystore).await
380	}
381
382	fn pre_digest_data(&self, slot: Slot, _claim: &Self::Claim) -> Vec<sp_runtime::DigestItem> {
383		vec![crate::standalone::pre_digest::<P>(slot)]
384	}
385
386	async fn block_import_params(
387		&self,
388		header: B::Header,
389		header_hash: &B::Hash,
390		body: Vec<B::Extrinsic>,
391		storage_changes: StorageChanges<B>,
392		public: Self::Claim,
393		_authorities: Self::AuxData,
394	) -> Result<sc_consensus::BlockImportParams<B>, ConsensusError> {
395		let signature_digest_item =
396			crate::standalone::seal::<_, P>(header_hash, &public, &self.keystore)?;
397
398		let mut import_block = BlockImportParams::new(BlockOrigin::Own, header);
399		import_block.post_digests.push(signature_digest_item);
400		import_block.body = Some(body);
401		import_block.state_action =
402			StateAction::ApplyChanges(sc_consensus::StorageChanges::Changes(storage_changes));
403		import_block.fork_choice = Some(ForkChoiceStrategy::LongestChain);
404
405		Ok(import_block)
406	}
407
408	fn force_authoring(&self) -> bool {
409		self.force_authoring
410	}
411
412	fn should_backoff(&self, slot: Slot, chain_head: &B::Header) -> bool {
413		if let Some(ref strategy) = self.backoff_authoring_blocks {
414			if let Ok(chain_head_slot) = find_pre_digest::<B, P::Signature>(chain_head) {
415				return strategy.should_backoff(
416					*chain_head.number(),
417					chain_head_slot,
418					self.client.info().finalized_number,
419					slot,
420					self.logging_target(),
421				)
422			}
423		}
424		false
425	}
426
427	fn sync_oracle(&mut self) -> &mut Self::SyncOracle {
428		&mut self.sync_oracle
429	}
430
431	fn justification_sync_link(&mut self) -> &mut Self::JustificationSyncLink {
432		&mut self.justification_sync_link
433	}
434
435	fn proposer(&mut self, block: &B::Header) -> Self::CreateProposer {
436		self.env
437			.init(block)
438			.map_err(|e| ConsensusError::ClientImport(format!("{:?}", e)))
439			.boxed()
440	}
441
442	fn telemetry(&self) -> Option<TelemetryHandle> {
443		self.telemetry.clone()
444	}
445
446	fn proposing_remaining_duration(&self, slot_info: &SlotInfo<B>) -> std::time::Duration {
447		let parent_slot = find_pre_digest::<B, P::Signature>(&slot_info.chain_head).ok();
448
449		sc_consensus_slots::proposing_remaining_duration(
450			parent_slot,
451			slot_info,
452			&self.block_proposal_slot_portion,
453			self.max_block_proposal_slot_portion.as_ref(),
454			sc_consensus_slots::SlotLenienceType::Exponential,
455			self.logging_target(),
456		)
457	}

460/// Aura Errors
461#[derive(Debug, thiserror::Error)]
462pub enum Error<B: BlockT> {
463	/// Multiple Aura pre-runtime headers
464	#[error("Multiple Aura pre-runtime headers")]
465	MultipleHeaders,
466	/// No Aura pre-runtime digest found
467	#[error("No Aura pre-runtime digest found")]
468	NoDigestFound,
469	/// Header is unsealed
470	#[error("Header {0:?} is unsealed")]
471	HeaderUnsealed(B::Hash),
472	/// Header has a bad seal
473	#[error("Header {0:?} has a bad seal")]
474	HeaderBadSeal(B::Hash),
475	/// Slot Author not found
476	#[error("Slot Author not found")]
477	SlotAuthorNotFound,
478	/// Bad signature
479	#[error("Bad signature on {0:?}")]
480	BadSignature(B::Hash),
481	/// Client Error
482	#[error(transparent)]
483	Client(sp_blockchain::Error),
484	/// Unknown inherent error for identifier
485	#[error("Unknown inherent error for identifier: {}", String::from_utf8_lossy(.0))]
486	UnknownInherentError(sp_inherents::InherentIdentifier),
487	/// Inherents Error
488	#[error("Inherent error: {0}")]
489	Inherent(sp_inherents::Error),
490}
491
492impl<B: BlockT> From<Error<B>> for String {
493	fn from(error: Error<B>) -> String {
494		error.to_string()
495	}
496}
497
498impl<B: BlockT> From<crate::standalone::PreDigestLookupError> for Error<B> {
499	fn from(e: crate::standalone::PreDigestLookupError) -> Self {

502			crate::standalone::PreDigestLookupError::NoDigestFound => Error::NoDigestFound,
503		}
504	}

507fn authorities<A, B, C>(
508	client: &C,
509	parent_hash: B::Hash,
510	context_block_number: NumberFor<B>,
511	compatibility_mode: &CompatibilityMode<NumberFor<B>>,
512) -> Result<Vec<A>, ConsensusError>
513where
514	A: Codec + Debug,
515	B: BlockT,
516	C: ProvideRuntimeApi<B>,
517	C::Api: AuraApi<B, A>,
518{
519	let runtime_api = client.runtime_api();
520
521	match compatibility_mode {
522		CompatibilityMode::None => {},
523		// Use `initialize_block` until we hit the block that should disable the mode.
524		CompatibilityMode::UseInitializeBlock { until } =>
525			if *until > context_block_number {
526				runtime_api
527					.initialize_block(
528						parent_hash,
529						&B::Header::new(
530							context_block_number,
531							Default::default(),
532							Default::default(),
533							parent_hash,
534							Default::default(),
535						),
536					)
537					.map_err(|_| ConsensusError::InvalidAuthoritiesSet)?;
538			},
539	}
540
541	runtime_api
542		.authorities(parent_hash)
543		.ok()
544		.ok_or(ConsensusError::InvalidAuthoritiesSet)
545}
