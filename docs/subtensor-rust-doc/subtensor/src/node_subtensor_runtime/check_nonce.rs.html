16/// Nonce check and increment to give replay protection for transactions.
17///
18/// # Transaction Validity
19///
20/// This extension affects `requires` and `provides` tags of validity, but DOES NOT
21/// set the `priority` field. Make sure that AT LEAST one of the signed extension sets
22/// some kind of priority upon validating transactions.
23#[freeze_struct("610b76f62cdb521e")]
24#[derive(Encode, Decode, Clone, Eq, PartialEq, TypeInfo)]
25#[scale_info(skip_type_params(T))]
26pub struct CheckNonce<T: Config>(#[codec(compact)] pub T::Nonce);
27
28impl<T: Config> CheckNonce<T> {
29    /// utility constructor. Used only in client/factory code.
30    pub fn from(nonce: T::Nonce) -> Self {
31        Self(nonce)
32    }
33}

47impl<T: Config> SignedExtension for CheckNonce<T>
48where
49    T::RuntimeCall: Dispatchable<Info = DispatchInfo>,
50{
51    type AccountId = T::AccountId;
52    type Call = T::RuntimeCall;
53    type AdditionalSigned = ();
54    type Pre = ();
55    const IDENTIFIER: &'static str = "CheckNonce";
56
57    fn additional_signed(&self) -> sp_std::result::Result<(), TransactionValidityError> {
58        Ok(())
59    }
60
61    fn pre_dispatch(
62        self,
63        who: &Self::AccountId,
64        _call: &Self::Call,
65        info: &DispatchInfoOf<Self::Call>,
66        _len: usize,
67    ) -> Result<(), TransactionValidityError> {
68        let mut account = frame_system::Account::<T>::get(who);
69        match info.pays_fee {
70            Pays::Yes => {
71                if account.providers.is_zero() && account.sufficients.is_zero() {
72                    // Nonce storage not paid for
73                    return Err(InvalidTransaction::Payment.into());
74                }
75            }
76            // not check providers and sufficients for Pays::No extrinsic
77            Pays::No => {}
78        }
79
80        if self.0 != account.nonce {
81            return Err(if self.0 < account.nonce {
82                InvalidTransaction::Stale
83            } else {
84                InvalidTransaction::Future
85            }
86            .into());
87        }
88        account.nonce.saturating_inc();
89        frame_system::Account::<T>::insert(who, account);
90        Ok(())
91    }
92
93    fn validate(
94        &self,
95        who: &Self::AccountId,
96        _call: &Self::Call,
97        info: &DispatchInfoOf<Self::Call>,
98        _len: usize,
99    ) -> TransactionValidity {
100        let account = frame_system::Account::<T>::get(who);
101        match info.pays_fee {
102            Pays::Yes => {
103                if account.providers.is_zero() && account.sufficients.is_zero() {
104                    // Nonce storage not paid for
105                    return Err(InvalidTransaction::Payment.into());
106                }
107            }
108            // not check providers and sufficients for Pays::No extrinsic
109            Pays::No => {}
110        }
111        if self.0 < account.nonce {
112            return InvalidTransaction::Stale.into();
113        }
114
115        let provides = vec![Encode::encode(&(who, self.0))];
116        let requires = if account.nonce < self.0 {
117            vec![Encode::encode(&(who, self.0.saturating_sub(One::one())))]
118        } else {
119            vec![]
120        };
121
122        Ok(ValidTransaction {
123            priority: 0,
124            requires,
125            provides,
126            longevity: TransactionLongevity::MAX,
127            propagate: true,
128        })
129    }
130}
