124impl frame_system::offchain::CreateSignedTransaction<pallet_drand::Call<Runtime>> for Runtime {
125    fn create_transaction<S: frame_system::offchain::AppCrypto<Self::Public, Self::Signature>>(
126        call: RuntimeCall,
127        public: <Signature as Verify>::Signer,
128        account: AccountId,
129        index: Index,
130    ) -> Option<(
131        RuntimeCall,
132        <UncheckedExtrinsic as sp_runtime::traits::Extrinsic>::SignaturePayload,
133    )> {
134        use sp_runtime::traits::StaticLookup;
135
136        let address = <Runtime as frame_system::Config>::Lookup::unlookup(account.clone());
137        let extra: SignedExtra = (
138            frame_system::CheckNonZeroSender::<Runtime>::new(),
139            frame_system::CheckSpecVersion::<Runtime>::new(),
140            frame_system::CheckTxVersion::<Runtime>::new(),
141            frame_system::CheckGenesis::<Runtime>::new(),
142            frame_system::CheckEra::<Runtime>::from(Era::Immortal),
143            check_nonce::CheckNonce::<Runtime>::from(index),
144            frame_system::CheckWeight::<Runtime>::new(),
145            pallet_transaction_payment::ChargeTransactionPayment::<Runtime>::from(0),
146            pallet_subtensor::SubtensorSignedExtension::<Runtime>::new(),
147            pallet_commitments::CommitmentsSignedExtension::<Runtime>::new(),
148            frame_metadata_hash_extension::CheckMetadataHash::<Runtime>::new(true),
149        );
150
151        let raw_payload = SignedPayload::new(call.clone(), extra.clone()).ok()?;
152        let signature = raw_payload.using_encoded(|payload| S::sign(payload, public))?;
153
154        let signature_payload = (address, signature, extra);
155
156        Some((call, signature_payload))
157    }
158}

167// Method used to calculate the fee of an extrinsic
168pub const fn deposit(items: u32, bytes: u32) -> Balance {
169    pub const ITEMS_FEE: Balance = 2_000 * 10_000;
170    pub const BYTES_FEE: Balance = 100 * 10_000;
171    (items as Balance)
172        .saturating_mul(ITEMS_FEE)
173        .saturating_add((bytes as Balance).saturating_mul(BYTES_FEE))
174}

249impl Contains<RuntimeCall> for NoNestingCallFilter {
250    fn contains(call: &RuntimeCall) -> bool {
251        match call {
252            RuntimeCall::Utility(inner) => {
253                let calls = match inner {
254                    pallet_utility::Call::force_batch { calls } => calls,
255                    pallet_utility::Call::batch { calls } => calls,
256                    pallet_utility::Call::batch_all { calls } => calls,
257                    _ => &Vec::new(),
258                };
259
260                !calls.iter().any(|call| {
261					matches!(call, RuntimeCall::Utility(inner) if matches!(inner, pallet_utility::Call::force_batch { .. } | pallet_utility::Call::batch_all { .. } | pallet_utility::Call::batch { .. }))
262				})
263            }
264            _ => true,
265        }
266    }

459/// Deduct the transaction fee from the Subtensor Pallet TotalIssuance when dropping the transaction
460/// fee.
461pub struct TransactionFeeHandler;
462impl
463    OnUnbalanced<
464        FungibleImbalance<
465            u64,
466            DecreaseIssuance<AccountId32, pallet_balances::Pallet<Runtime>>,
467            IncreaseIssuance<AccountId32, pallet_balances::Pallet<Runtime>>,
468        >,
469    > for TransactionFeeHandler
470{
471    fn on_nonzero_unbalanced(
472        credit: FungibleImbalance<
473            u64,
474            DecreaseIssuance<AccountId32, pallet_balances::Pallet<Runtime>>,
475            IncreaseIssuance<AccountId32, pallet_balances::Pallet<Runtime>>,
476        >,
477    ) {
478        let ti_before = pallet_subtensor::TotalIssuance::<Runtime>::get();
479        pallet_subtensor::TotalIssuance::<Runtime>::put(ti_before.saturating_sub(credit.peek()));
480        drop(credit);
481    }
482}

834    fn is_superset(&self, o: &Self) -> bool {
835        match (self, o) {
836            (x, y) if x == y => true,
837            (ProxyType::Any, _) => true,

844            (ProxyType::Transfer, ProxyType::SmallTransfer) => true,
845            _ => false,
846        }

872/// Used the compare the privilege of an origin inside the scheduler.
873pub struct OriginPrivilegeCmp;
874
875impl PrivilegeCmp<OriginCaller> for OriginPrivilegeCmp {
876    fn cmp_privilege(left: &OriginCaller, right: &OriginCaller) -> Option<Ordering> {
877        if left == right {
878            return Some(Ordering::Equal);
879        }
880
881        match (left, right) {
882            // Root is greater than anything.
883            (OriginCaller::system(frame_system::RawOrigin::Root), _) => Some(Ordering::Greater),
884            // Check which one has more yes votes.
885            (
886                OriginCaller::Triumvirate(pallet_collective::RawOrigin::Members(
887                    l_yes_votes,
888                    l_count,
889                )),
890                OriginCaller::Triumvirate(pallet_collective::RawOrigin::Members(
891                    r_yes_votes,
892                    r_count,
893                )), // Equivalent to (l_yes_votes / l_count).cmp(&(r_yes_votes / r_count))
894            ) => Some(
895                l_yes_votes
896                    .saturating_mul(*r_count)
897                    .cmp(&r_yes_votes.saturating_mul(*l_count)),
898            ),
899            // For every other origin we don't care, as they are not used for `ScheduleOrigin`.
900            _ => None,
901        }
902    }
903}

1272/// The difference between EVM decimals and Substrate decimals.
1273/// Substrate balances has 9 decimals, while EVM has 18, so the
1274/// difference factor is 9 decimals, or 10^9
1275const EVM_TO_SUBSTRATE_DECIMALS: u64 = 1_000_000_000_u64;

1279impl BalanceConverter for SubtensorEvmBalanceConverter {
1280    /// Convert from Substrate balance (u64) to EVM balance (U256)
1281    fn into_evm_balance(value: SubstrateBalance) -> Option<EvmBalance> {
1282        let value = value.into_u256();
1283        if let Some(evm_value) = value.checked_mul(U256::from(EVM_TO_SUBSTRATE_DECIMALS)) {
1284            // Ensure the result fits within the maximum U256 value
1285            if evm_value <= U256::MAX {
1286                Some(EvmBalance::new(evm_value))
1287            } else {
1288                // Log value too large
1289                log::debug!(
1290                    "SubtensorEvmBalanceConverter::into_evm_balance( {:?} ) larger than U256::MAX",
1291                    value
1292                );
1293                None
1294            }
1295        } else {
1296            // Log overflow
1297            log::debug!(
1298                "SubtensorEvmBalanceConverter::into_evm_balance( {:?} ) overflow",
1299                value
1300            );
1301            None
1302        }
1303    }
1304
1305    /// Convert from EVM balance (U256) to Substrate balance (u64)
1306    fn into_substrate_balance(value: EvmBalance) -> Option<SubstrateBalance> {
1307        let value = value.into_u256();
1308        if let Some(substrate_value) = value.checked_div(U256::from(EVM_TO_SUBSTRATE_DECIMALS)) {
1309            // Ensure the result fits within the TAO balance type (u64)
1310            if substrate_value <= U256::from(u64::MAX) {
1311                Some(SubstrateBalance::new(substrate_value))
1312            } else {
1313                // Log value too large
1314                log::debug!(
1315                    "SubtensorEvmBalanceConverter::into_substrate_balance( {:?} ) larger than u64::MAX",
1316                    value
1317                );
1318                None
1319            }
1320        } else {
1321            // Log overflow
1322            log::debug!(
1323                "SubtensorEvmBalanceConverter::into_substrate_balance( {:?} ) overflow",
1324                value
1325            );
