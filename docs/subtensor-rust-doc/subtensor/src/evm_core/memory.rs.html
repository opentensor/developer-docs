7/// A sequencial memory. It uses Rust's `Vec` for internal
8/// representation.
9#[derive(Clone, Debug)]
10pub struct Memory {
11	data: Vec<u8>,
12	effective_len: U256,
13	limit: usize,
14}
15
16impl Memory {
17	/// Create a new memory with the given limit.
18	pub fn new(limit: usize) -> Self {
19		Self {
20			data: Vec::new(),
21			effective_len: U256::zero(),
22			limit,
23		}
24	}
25
26	/// Memory limit.
27	pub fn limit(&self) -> usize {
28		self.limit
29	}
30
31	/// Get the length of the current memory range.
32	pub fn len(&self) -> usize {
33		self.data.len()
34	}
35
36	/// Get the effective length.
37	pub fn effective_len(&self) -> U256 {
38		self.effective_len
39	}
40
41	/// Return true if current effective memory range is zero.
42	pub fn is_empty(&self) -> bool {
43		self.len() == 0
44	}
45
46	/// Return the full memory.
47	pub fn data(&self) -> &Vec<u8> {
48		&self.data
49	}
50
51	/// Resize the memory, making it cover the memory region of `offset..(offset
52	/// + len)`, with 32 bytes as the step. If the length is zero, this function
53	/// does nothing.
54	pub fn resize_offset(&mut self, offset: U256, len: U256) -> Result<(), ExitError> {
55		if len == U256::zero() {
56			return Ok(());
57		}
58
59		if let Some(end) = offset.checked_add(len) {
60			self.resize_end(end)
61		} else {
62			Err(ExitError::InvalidRange)
63		}
64	}
65
66	/// Resize the memory, making it cover to `end`, with 32 bytes as the step.
67	pub fn resize_end(&mut self, end: U256) -> Result<(), ExitError> {
68		if end > self.effective_len {
69			let new_end = next_multiple_of_32(end).ok_or(ExitError::InvalidRange)?;
70			self.effective_len = new_end;
71		}
72
73		Ok(())
74	}
75
76	/// Get memory region at given offset.
77	///
78	/// ## Panics
79	///
80	/// Value of `size` is considered trusted. If they're too large,
81	/// the program can run out of memory, or it can overflow.
82	pub fn get(&self, offset: usize, size: usize) -> Vec<u8> {
83		let mut ret = Vec::new();
84		ret.resize(size, 0);
85
86		#[allow(clippy::needless_range_loop)]
87		for index in 0..size {
88			let position = offset + index;
89			if position >= self.data.len() {
90				break;
91			}
92
93			ret[index] = self.data[position];
94		}
95
96		ret
97	}
98
99	/// Set memory region at given offset. The offset and value is considered
100	/// untrusted.
101	pub fn set(
102		&mut self,
103		offset: usize,
104		value: &[u8],
105		target_size: Option<usize>,
106	) -> Result<(), ExitFatal> {
107		let target_size = target_size.unwrap_or(value.len());
108		if target_size == 0 {
109			return Ok(());
110		}
111
112		if offset
113			.checked_add(target_size)
114			.map(|pos| pos > self.limit)
115			.unwrap_or(true)
116		{
117			return Err(ExitFatal::NotSupported);
118		}
119
120		if self.data.len() < offset + target_size {
121			self.data.resize(offset + target_size, 0);
122		}
123
124		if target_size > value.len() {
125			self.data[offset..((value.len()) + offset)].clone_from_slice(value);
126			for index in (value.len())..target_size {
127				self.data[offset + index] = 0;
128			}
129		} else {
130			self.data[offset..(target_size + offset)].clone_from_slice(&value[..target_size]);
131		}
132
133		Ok(())
134	}
135
136	/// Copy `data` into the memory, of given `len`.
137	pub fn copy_large(
138		&mut self,
139		memory_offset: U256,
140		data_offset: U256,
141		len: U256,
142		data: &[u8],
143	) -> Result<(), ExitFatal> {
144		// Needed to pass ethereum test defined in
145		// https://github.com/ethereum/tests/commit/17f7e7a6c64bb878c1b6af9dc8371b46c133e46d
146		// (regardless of other inputs, a zero-length copy is defined to be a no-op).
147		// TODO: refactor `set` and `copy_large` (see
148		// https://github.com/rust-blockchain/evm/pull/40#discussion_r677180794)
149		if len.is_zero() {
150			return Ok(());
151		}
152
153		let memory_offset = if memory_offset > U256::from(usize::MAX) {
154			return Err(ExitFatal::NotSupported);
155		} else {
156			memory_offset.as_usize()
157		};
158
159		let ulen = if len > U256::from(usize::MAX) {
160			return Err(ExitFatal::NotSupported);
161		} else {
162			len.as_usize()
163		};
164
165		let data = if let Some(end) = data_offset.checked_add(len) {
166			if end > U256::from(usize::MAX) {
167				&[]
168			} else {
169				let data_offset = data_offset.as_usize();
170				let end = end.as_usize();
171
172				if data_offset > data.len() {
173					&[]
174				} else {
175					&data[data_offset..min(end, data.len())]
176				}
177			}
178		} else {
179			&[]
180		};
181
182		self.set(memory_offset, data, Some(ulen))
183	}
184}

186/// Rounds up `x` to the closest multiple of 32. If `x % 32 == 0` then `x` is returned.
187#[inline]
188fn next_multiple_of_32(x: U256) -> Option<U256> {
189	let r = x.low_u32().bitand(31).not().wrapping_add(1).bitand(31);
190	x.checked_add(r.into())
191}

193#[cfg(test)]
194mod tests {
195	use super::{next_multiple_of_32, U256};
196
197	#[test]
198	fn test_next_multiple_of_32() {
199		// next_multiple_of_32 returns x when it is a multiple of 32
200		for i in 0..32 {
201			let x = U256::from(i * 32);
202			assert_eq!(Some(x), next_multiple_of_32(x));
203		}
204
205		// next_multiple_of_32 rounds up to the nearest multiple of 32 when `x % 32 != 0`
206		for x in 0..1024 {
207			if x % 32 == 0 {
208				continue;
209			}
210			let next_multiple = x + 32 - (x % 32);
211			assert_eq!(
212				Some(U256::from(next_multiple)),
213				next_multiple_of_32(x.into())
214			);
215		}
216
217		// next_multiple_of_32 returns None when the next multiple of 32 is too big
218		let last_multiple_of_32 = U256::MAX & !U256::from(31);
219		for i in 0..63 {
220			let x = U256::MAX - U256::from(i);
221			if x > last_multiple_of_32 {
222				assert_eq!(None, next_multiple_of_32(x));
223			} else {
224				assert_eq!(Some(last_multiple_of_32), next_multiple_of_32(x));
225			}
226		}
227	}
