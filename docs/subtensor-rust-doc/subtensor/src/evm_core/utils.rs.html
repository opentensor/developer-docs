1use core::cmp::Ordering;
2use core::ops::{Div, Rem};
3use primitive_types::U256;
4
5#[derive(Copy, Clone, Eq, PartialEq, Debug)]
6pub enum Sign {
7	Plus,
8	Minus,
9	Zero,
10}
11
12const SIGN_BIT_MASK: U256 = U256([
13	0xffffffffffffffff,
14	0xffffffffffffffff,
15	0xffffffffffffffff,
16	0x7fffffffffffffff,
17]);
18
19#[derive(Copy, Clone, Eq, PartialEq, Debug)]
20pub struct I256(pub Sign, pub U256);
21
22impl I256 {
23	/// Zero value of I256.
24	pub fn zero() -> I256 {
25		I256(Sign::Zero, U256::zero())
26	}
27	/// Minimum value of I256.
28	pub fn min_value() -> I256 {
29		I256(Sign::Minus, (U256::MAX & SIGN_BIT_MASK) + U256::from(1u64))
30	}
31}
32
33impl Ord for I256 {
34	fn cmp(&self, other: &I256) -> Ordering {
35		match (self.0, other.0) {
36			(Sign::Zero, Sign::Zero) => Ordering::Equal,
37			(Sign::Zero, Sign::Plus) => Ordering::Less,
38			(Sign::Zero, Sign::Minus) => Ordering::Greater,
39			(Sign::Minus, Sign::Zero) => Ordering::Less,
40			(Sign::Minus, Sign::Plus) => Ordering::Less,
41			(Sign::Minus, Sign::Minus) => self.1.cmp(&other.1).reverse(),
42			(Sign::Plus, Sign::Minus) => Ordering::Greater,
43			(Sign::Plus, Sign::Zero) => Ordering::Greater,
44			(Sign::Plus, Sign::Plus) => self.1.cmp(&other.1),
45		}
46	}
47}
48
49impl PartialOrd for I256 {
50	fn partial_cmp(&self, other: &I256) -> Option<Ordering> {
51		Some(self.cmp(other))
52	}
53}
54
55impl Default for I256 {
56	fn default() -> I256 {
57		I256::zero()
58	}
59}
60
61impl From<U256> for I256 {
62	fn from(val: U256) -> I256 {
63		if val == U256::zero() {
64			I256::zero()
65		} else if val & SIGN_BIT_MASK == val {
66			I256(Sign::Plus, val)
67		} else {
68			I256(Sign::Minus, !val + U256::from(1u64))
69		}
70	}
71}
72
73impl From<I256> for U256 {
74	fn from(value: I256) -> U256 {
75		let sign = value.0;
76		if sign == Sign::Zero {
77			U256::zero()
78		} else if sign == Sign::Plus {
79			value.1
80		} else {
81			!value.1 + U256::from(1u64)
82		}
83	}
84}
85
86impl Div for I256 {
87	type Output = I256;
88
89	fn div(self, other: I256) -> I256 {
90		if other == I256::zero() {
91			return I256::zero();
92		}
93
94		if self == I256::min_value() && other.1 == U256::from(1u64) {
95			return I256::min_value();
96		}
97
98		let d = (self.1 / other.1) & SIGN_BIT_MASK;
99
100		if d == U256::zero() {
101			return I256::zero();
102		}
103
104		match (self.0, other.0) {
105			(Sign::Zero, Sign::Plus)
106			| (Sign::Plus, Sign::Zero)
107			| (Sign::Zero, Sign::Zero)
108			| (Sign::Plus, Sign::Plus)
109			| (Sign::Minus, Sign::Minus) => I256(Sign::Plus, d),
110			(Sign::Zero, Sign::Minus)
111			| (Sign::Plus, Sign::Minus)
112			| (Sign::Minus, Sign::Zero)
113			| (Sign::Minus, Sign::Plus) => I256(Sign::Minus, d),
114		}
115	}
116}
117
118impl Rem for I256 {
119	type Output = I256;
120
121	fn rem(self, other: I256) -> I256 {
122		let r = (self.1 % other.1) & SIGN_BIT_MASK;
123
124		if r == U256::zero() {
125			return I256::zero();
126		}
127
128		I256(self.0, r)
129	}
130}

132#[cfg(test)]
133mod tests {
134	use crate::utils::{Sign, I256};
135	use primitive_types::U256;
136	use std::num::Wrapping;
137
138	#[test]
139	fn div_i256() {
140		// Sanity checks based on i8. Notice that we need to use `Wrapping` here because
141		// Rust will prevent the overflow by default whereas the EVM does not.
142		assert_eq!(Wrapping(i8::MIN) / Wrapping(-1), Wrapping(i8::MIN));
143
144		assert_eq!(100i8 / -1, -100i8);
145		assert_eq!(100i8 / 2, 50i8);
146
147		// Now the same calculations based on i256
148		let one = I256(Sign::Zero, U256::from(1));
149		let one_hundred = I256(Sign::Zero, U256::from(100));
150		let fifty = I256(Sign::Plus, U256::from(50));
151		let two = I256(Sign::Zero, U256::from(2));
152		let neg_one_hundred = I256(Sign::Minus, U256::from(100));
153		let minus_one = I256(Sign::Minus, U256::from(1));
154		let max_value = I256(Sign::Plus, U256::from(2).pow(U256::from(255)) - 1);
155		let neg_max_value = I256(Sign::Minus, U256::from(2).pow(U256::from(255)) - 1);
156
157		assert_eq!(I256::min_value() / minus_one, I256::min_value());
158		assert_eq!(I256::min_value() / one, I256::min_value());
159		assert_eq!(max_value / one, max_value);
160		assert_eq!(max_value / minus_one, neg_max_value);
161
162		assert_eq!(one_hundred / minus_one, neg_one_hundred);
163		assert_eq!(one_hundred / two, fifty);
164	}
