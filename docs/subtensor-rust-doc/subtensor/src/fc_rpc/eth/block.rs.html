49	pub async fn block_by_hash(&self, hash: H256, full: bool) -> RpcResult<Option<RichBlock>> {
50		let BlockInfo {
51			block,
52			statuses,
53			substrate_hash,
54			base_fee,
55			..
56		} = self.block_info_by_eth_block_hash(hash).await?;
57
58		match (block, statuses) {
59			(Some(block), Some(statuses)) => {
60				let mut rich_block = rich_block_build(
61					block,
62					statuses.into_iter().map(Option::Some).collect(),
63					Some(hash),
64					full,
65					Some(base_fee),
66					false,
67				);
68
69				let substrate_hash = H256::from_slice(substrate_hash.as_ref());
70				if let Some(parent_hash) = self
71					.forced_parent_hashes
72					.as_ref()
73					.and_then(|parent_hashes| parent_hashes.get(&substrate_hash).cloned())
74				{
75					rich_block.inner.header.parent_hash = parent_hash
76				}
77
78				Ok(Some(rich_block))
79			}
80			_ => Ok(None),
81		}
82	}

84	pub async fn block_by_number(
85		&self,
86		number_or_hash: BlockNumberOrHash,
87		full: bool,
88	) -> RpcResult<Option<RichBlock>> {
89		let client = Arc::clone(&self.client);
90		let block_data_cache = Arc::clone(&self.block_data_cache);
91		let backend = Arc::clone(&self.backend);
92		let graph = Arc::clone(&self.graph);
93
94		match frontier_backend_client::native_block_id::<B, C>(
95			client.as_ref(),
96			backend.as_ref(),
97			Some(number_or_hash),
98		)
99		.await?
100		{
101			Some(id) => {
102				let substrate_hash = client
103					.expect_block_hash_from_id(&id)
104					.map_err(|_| internal_err(format!("Expect block number from id: {}", id)))?;
105
106				let block = block_data_cache.current_block(substrate_hash).await;
107				let statuses = block_data_cache
108					.current_transaction_statuses(substrate_hash)
109					.await;
110
111				let base_fee = client.runtime_api().gas_price(substrate_hash).ok();
112
113				match (block, statuses) {
114					(Some(block), Some(statuses)) => {
115						let hash = H256::from(keccak_256(&rlp::encode(&block.header)));
116						let mut rich_block = rich_block_build(
117							block,
118							statuses.into_iter().map(Option::Some).collect(),
119							Some(hash),
120							full,
121							base_fee,
122							false,
123						);
124
125						let substrate_hash = H256::from_slice(substrate_hash.as_ref());
126						if let Some(parent_hash) = self
127							.forced_parent_hashes
128							.as_ref()
129							.and_then(|parent_hashes| parent_hashes.get(&substrate_hash).cloned())
130						{
131							rich_block.inner.header.parent_hash = parent_hash
132						}
133
134						Ok(Some(rich_block))
135					}
136					_ => Ok(None),
137				}
138			}
139			None if number_or_hash == BlockNumberOrHash::Pending => {
140				let api = client.runtime_api();
141				let best_hash = client.info().best_hash;
142
143				// Get current in-pool transactions
144				let mut xts: Vec<<B as BlockT>::Extrinsic> = Vec::new();
145				// ready validated pool
146				xts.extend(
147					graph
148						.validated_pool()
149						.ready()
150						.map(|in_pool_tx| in_pool_tx.data().clone())
151						.collect::<Vec<<B as BlockT>::Extrinsic>>(),
152				);
153
154				// future validated pool
155				xts.extend(
156					graph
157						.validated_pool()
158						.futures()
159						.iter()
160						.map(|(_hash, extrinsic)| extrinsic.clone())
161						.collect::<Vec<<B as BlockT>::Extrinsic>>(),
162				);
163
164				let (block, statuses) = api
165					.pending_block(best_hash, xts)
166					.map_err(|_| internal_err(format!("Runtime access error at {}", best_hash)))?;
167
168				let base_fee = api.gas_price(best_hash).ok();
169
170				match (block, statuses) {
171					(Some(block), Some(statuses)) => Ok(Some(rich_block_build(
172						block,
173						statuses.into_iter().map(Option::Some).collect(),
174						None,
175						full,
176						base_fee,
177						true,
178					))),
179					_ => Ok(None),
180				}
181			}
182			None => Ok(None),
183		}
184	}

186	pub async fn block_transaction_count_by_hash(&self, hash: H256) -> RpcResult<Option<U256>> {
187		let blockinfo = self.block_info_by_eth_block_hash(hash).await?;
188		match blockinfo.block {
189			Some(block) => Ok(Some(U256::from(block.transactions.len()))),
190			None => Ok(None),
191		}
192	}

194	pub async fn block_transaction_count_by_number(
195		&self,
196		number_or_hash: BlockNumberOrHash,
197	) -> RpcResult<Option<U256>> {
198		if let BlockNumberOrHash::Pending = number_or_hash {
199			// get the pending transactions count
200			return Ok(Some(U256::from(
201				self.graph.validated_pool().ready().count(),
202			)));
203		}
204
205		let block_info = self.block_info_by_number(number_or_hash).await?;
206		match block_info.block {
207			Some(block) => Ok(Some(U256::from(block.transactions.len()))),
208			None => Ok(None),
209		}
210	}

212	pub async fn block_transaction_receipts(
213		&self,
214		number_or_hash: BlockNumberOrHash,
215	) -> RpcResult<Option<Vec<Receipt>>> {
216		let block_info = self.block_info_by_number(number_or_hash).await?;
217		let Some(statuses) = block_info.clone().statuses else {
218			return Ok(None);
219		};
220
221		let mut receipts = Vec::new();
222		let transactions: Vec<(H256, usize)> = statuses
223			.iter()
224			.map(|tx| (tx.transaction_hash, tx.transaction_index as usize))
225			.collect();
226		for (hash, index) in transactions {
227			if let Some(receipt) = self.transaction_receipt(&block_info, hash, index).await? {
228				receipts.push(receipt);
229			}
230		}
231
232		Ok(Some(receipts))
233	}
