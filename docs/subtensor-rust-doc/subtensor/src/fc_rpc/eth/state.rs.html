48	pub async fn balance(
49		&self,
50		address: H160,
51		number_or_hash: Option<BlockNumberOrHash>,
52	) -> RpcResult<U256> {
53		let number_or_hash = number_or_hash.unwrap_or(BlockNumberOrHash::Latest);
54		if number_or_hash == BlockNumberOrHash::Pending {
55			let (hash, api) = self
56				.pending_runtime_api()
57				.await
58				.map_err(|err| internal_err(format!("Create pending runtime api error: {err}")))?;
59			Ok(api
60				.account_basic(hash, address)
61				.map_err(|err| internal_err(format!("Fetch account balances failed: {err}")))?
62				.balance)
63		} else if let Ok(Some(id)) = frontier_backend_client::native_block_id::<B, C>(
64			self.client.as_ref(),
65			self.backend.as_ref(),
66			Some(number_or_hash),
67		)
68		.await
69		{
70			let substrate_hash = self
71				.client
72				.expect_block_hash_from_id(&id)
73				.map_err(|_| internal_err(format!("Expect block number from id: {id}")))?;
74
75			Ok(self
76				.client
77				.runtime_api()
78				.account_basic(substrate_hash, address)
79				.map_err(|err| internal_err(format!("Fetch account balances failed: {:?}", err)))?
80				.balance)
81		} else {
82			Ok(U256::zero())
83		}
84	}

119	pub async fn transaction_count(
120		&self,
121		address: H160,
122		number_or_hash: Option<BlockNumberOrHash>,
123	) -> RpcResult<U256> {
124		if let Some(BlockNumberOrHash::Pending) = number_or_hash {
125			let substrate_hash = self.client.info().best_hash;
126
127			let nonce = self
128				.client
129				.runtime_api()
130				.account_basic(substrate_hash, address)
131				.map_err(|err| internal_err(format!("Fetch account nonce failed: {err}")))?
132				.nonce;
133
134			let mut current_nonce = nonce;
135			let mut current_tag = (address, nonce).encode();
136			for tx in self.pool.ready() {
137				// since transactions in `ready()` need to be ordered by nonce
138				// it's fine to continue with current iterator.
139				if tx.provides().first() == Some(&current_tag) {
140					current_nonce = current_nonce.saturating_add(1.into());
141					current_tag = (address, current_nonce).encode();
142				}
143			}
144
145			return Ok(current_nonce);
146		}
147
148		let id = match frontier_backend_client::native_block_id::<B, C>(
149			self.client.as_ref(),
150			self.backend.as_ref(),
151			number_or_hash,
152		)
153		.await?
154		{
155			Some(id) => id,
156			None => return Ok(U256::zero()),
157		};
158
159		let substrate_hash = self
160			.client
161			.expect_block_hash_from_id(&id)
162			.map_err(|_| internal_err(format!("Expect block number from id: {id}")))?;
163
164		Ok(self
165			.client
166			.runtime_api()
167			.account_basic(substrate_hash, address)
168			.map_err(|err| internal_err(format!("Fetch account nonce failed: {err}")))?
169			.nonce)
170	}

172	pub async fn code_at(
173		&self,
174		address: H160,
175		number_or_hash: Option<BlockNumberOrHash>,
176	) -> RpcResult<Bytes> {
177		let number_or_hash = number_or_hash.unwrap_or(BlockNumberOrHash::Latest);
178		if number_or_hash == BlockNumberOrHash::Pending {
179			let (hash, api) = self
180				.pending_runtime_api()
181				.await
182				.map_err(|err| internal_err(format!("Create pending runtime api error: {err}")))?;
183			Ok(api
184				.account_code_at(hash, address)
185				.unwrap_or_default()
186				.into())
187		} else if let Ok(Some(id)) = frontier_backend_client::native_block_id::<B, C>(
188			self.client.as_ref(),
189			self.backend.as_ref(),
190			Some(number_or_hash),
191		)
192		.await
193		{
194			let substrate_hash = self
195				.client
196				.expect_block_hash_from_id(&id)
197				.map_err(|_| internal_err(format!("Expect block number from id: {id}")))?;
198			Ok(self
199				.storage_override
200				.account_code_at(substrate_hash, address)
201				.unwrap_or_default()
202				.into())
203		} else {
204			Ok(Bytes(vec![]))
205		}
206	}
