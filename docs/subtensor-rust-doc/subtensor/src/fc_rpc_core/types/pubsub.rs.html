34/// Subscription kind.
35#[derive(Clone, Debug, Eq, PartialEq, Hash, Deserialize)]
36#[serde(deny_unknown_fields)]
37#[serde(rename_all = "camelCase")]
38pub enum Kind {
39	/// New block headers subscription.
40	NewHeads,
41	/// Logs subscription.
42	Logs,
43	/// New Pending Transactions subscription.
44	NewPendingTransactions,
45	/// Node syncing status subscription.
46	Syncing,
47}
48
49/// Subscription kind.
50#[derive(Clone, Debug, Eq, PartialEq, Default, Hash)]
51pub enum Params {
52	/// No parameters passed.
53	#[default]
54	None,
55	/// Log parameters.
56	Logs(Filter),
57}
58
59impl<'a> Deserialize<'a> for Params {
60	fn deserialize<D>(deserializer: D) -> Result<Params, D::Error>
61	where
62		D: Deserializer<'a>,
63	{
64		let v: Value = Deserialize::deserialize(deserializer)?;
65
66		if v.is_null() {
67			return Ok(Params::None);
68		}
69
70		from_value(v)
71			.map(Params::Logs)
72			.map_err(|e| D::Error::custom(format!("Invalid Pub-Sub parameters: {}", e)))
73	}
74}

76/// Subscription result.
77#[derive(Clone, Debug, Eq, PartialEq)]
78pub enum PubSubResult {
79	/// New block header.
80	Header(Box<RichHeader>),
81	/// Log
82	Log(Box<Log>),
83	/// Transaction hash
84	TransactionHash(H256),
85	/// SyncStatus
86	SyncingStatus(PubSubSyncing),
87}
88
89impl PubSubResult {
90	pub fn header(block: EthereumBlock) -> Self {
91		Self::Header(Box::new(Rich {
92			inner: Header {
93				hash: Some(H256::from(keccak_256(&rlp::encode(&block.header)))),
94				parent_hash: block.header.parent_hash,
95				uncles_hash: block.header.ommers_hash,
96				author: block.header.beneficiary,
97				miner: Some(block.header.beneficiary),
98				state_root: block.header.state_root,
99				transactions_root: block.header.transactions_root,
100				receipts_root: block.header.receipts_root,
101				number: Some(block.header.number),
102				gas_used: block.header.gas_used,
103				gas_limit: block.header.gas_limit,
104				extra_data: Bytes(block.header.extra_data.clone()),
105				logs_bloom: block.header.logs_bloom,
106				timestamp: U256::from(block.header.timestamp),
107				difficulty: block.header.difficulty,
108				nonce: Some(block.header.nonce),
109				size: Some(U256::from(rlp::encode(&block.header).len() as u32)),
110			},
111			extra_info: BTreeMap::new(),
112		}))
113	}
114
115	pub fn logs(
116		block: EthereumBlock,
117		receipts: Vec<EthereumReceipt>,
118		params: &FilteredParams,
119	) -> impl Iterator<Item = Self> {
120		let block_number = block.header.number;
121		let block_hash = block.header.hash();
122
123		let mut logs: Vec<Log> = vec![];
124		let mut log_index: u32 = 0;
125		for (receipt_index, receipt) in receipts.into_iter().enumerate() {
126			let receipt_logs = match receipt {
127				EthereumReceipt::Legacy(d)
128				| EthereumReceipt::EIP2930(d)
129				| EthereumReceipt::EIP1559(d) => d.logs,
130			};
131
132			let transaction_hash: Option<H256> = if !receipt_logs.is_empty() {
133				Some(block.transactions[receipt_index].hash())
134			} else {
135				None
136			};
137
138			let mut transaction_log_index = 0;
139			for log in receipt_logs {
140				if params.is_not_filtered(block_number, block_hash, &log.address, &log.topics) {
141					logs.push(Log {
142						address: log.address,
143						topics: log.topics,
144						data: Bytes(log.data),
145						block_hash: Some(block_hash),
146						block_number: Some(block_number),
147						transaction_hash,
148						transaction_index: Some(U256::from(receipt_index)),
149						log_index: Some(U256::from(log_index)),
150						transaction_log_index: Some(U256::from(transaction_log_index)),
151						removed: false,
152					});
153				}
154				transaction_log_index += 1;
155				log_index += 1;
156			}
157		}
158		logs.into_iter().map(|log| Self::Log(Box::new(log)))
159	}
160
161	pub fn transaction_hash(tx: &EthereumTransaction) -> Self {
162		Self::TransactionHash(tx.hash())
163	}
164}

166impl Serialize for PubSubResult {
167	fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
168	where
169		S: Serializer,
170	{
171		match *self {
172			Self::Header(ref header) => header.serialize(serializer),
173			Self::Log(ref log) => log.serialize(serializer),
174			Self::TransactionHash(ref hash) => hash.serialize(serializer),
175			Self::SyncingStatus(ref sync) => sync.serialize(serializer),
176		}
177	}
178}

180#[derive(Clone, Debug, Eq, PartialEq, Serialize)]
181#[serde(untagged)]
182pub enum PubSubSyncing {
183	Synced(bool),
184	Syncing(SyncingStatus),
185}

187/// Pubsub sync status
188#[derive(Clone, Debug, Eq, PartialEq, Serialize)]
189#[serde(rename_all = "camelCase")]
190pub struct SyncingStatus {
191	pub starting_block: u64,
192	pub current_block: u64,
193	#[serde(default = "Default::default", skip_serializing_if = "Option::is_none")]
194	pub highest_block: Option<u64>,
195}
