26/// Block Transactions
27#[derive(Clone, Debug)]
28pub enum BlockTransactions {
29	/// Only hashes
30	Hashes(Vec<H256>),
31	/// Full transactions
32	Full(Vec<Transaction>),
33}
34
35impl Serialize for BlockTransactions {
36	fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
37	where
38		S: Serializer,
39	{
40		match *self {
41			BlockTransactions::Hashes(ref hashes) => hashes.serialize(serializer),
42			BlockTransactions::Full(ref ts) => ts.serialize(serializer),
43		}
44	}
45}
46
47/// Block representation
48#[derive(Clone, Debug, Serialize)]
49#[serde(rename_all = "camelCase")]
50pub struct Block {
51	/// Header of the block
52	#[serde(flatten)]
53	pub header: Header,
54	/// Total difficulty
55	pub total_difficulty: Option<U256>,
56	/// Uncles' hashes
57	pub uncles: Vec<H256>,
58	/// Transactions
59	pub transactions: BlockTransactions,
60	/// Size in bytes
61	pub size: Option<U256>,
62	/// Base Fee for post-EIP1559 blocks.
63	#[serde(skip_serializing_if = "Option::is_none")]
64	pub base_fee_per_gas: Option<U256>,
65}
66
67/// Block header representation.
68#[derive(Clone, Debug, Eq, PartialEq, Serialize)]
69#[serde(rename_all = "camelCase")]
70pub struct Header {
71	/// Hash of the block
72	pub hash: Option<H256>,
73	/// Hash of the parent
74	pub parent_hash: H256,
75	/// Hash of the uncles
76	#[serde(rename = "sha3Uncles")]
77	pub uncles_hash: H256,
78	/// Authors address
79	pub author: H160,
80	/// Alias of `author`
81	pub miner: Option<H160>,
82	/// State root hash
83	pub state_root: H256,
84	/// Transactions root hash
85	pub transactions_root: H256,
86	/// Transactions receipts root hash
87	pub receipts_root: H256,
88	/// Block number
89	pub number: Option<U256>,
90	/// Gas Used
91	pub gas_used: U256,
92	/// Gas Limit
93	pub gas_limit: U256,
94	/// Extra data
95	pub extra_data: Bytes,
96	/// Logs bloom
97	pub logs_bloom: H2048,
98	/// Timestamp
99	pub timestamp: U256,
100	/// Difficulty
101	pub difficulty: U256,
102	/// Nonce
103	pub nonce: Option<H64>,
104	/// Size in bytes
105	pub size: Option<U256>,
106}

114/// Value representation with additional info
115#[derive(Clone, Debug, Eq, PartialEq)]
116pub struct Rich<T> {
117	/// Standard value.
118	pub inner: T,
119	/// Engine-specific fields with additional description.
120	/// Should be included directly to serialized block object.
121	// TODO [ToDr] #[serde(skip_serializing)]
122	pub extra_info: BTreeMap<String, String>,
123}
124
125impl<T> Deref for Rich<T> {
126	type Target = T;
127	fn deref(&self) -> &Self::Target {
128		&self.inner
129	}
130}
131
132impl<T: Serialize> Serialize for Rich<T> {
133	fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
134	where
135		S: Serializer,
136	{
137		use serde_json::{to_value, Value};
138
139		let serialized = (to_value(&self.inner), to_value(&self.extra_info));
140		if let (Ok(Value::Object(mut value)), Ok(Value::Object(extras))) = serialized {
141			// join two objects
142			value.extend(extras);
143			// and serialize
144			value.serialize(serializer)
145		} else {
146			Err(S::Error::custom(
147				"Unserializable structures: expected objects",
148			))
149		}
150	}
151}
