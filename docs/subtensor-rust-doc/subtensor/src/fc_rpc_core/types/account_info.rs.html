19//! Return types for RPC calls
20
21use ethereum_types::{Address, Public, H160, H256, U256};
22use serde::Serialize;
23
24use crate::types::Bytes;
25
26/// Account information.
27#[derive(Clone, Debug, Default, Eq, PartialEq, Serialize)]
28pub struct AccountInfo {
29	/// Account name
30	pub name: String,
31}
32
33/// Data structure with proof for one single storage-entry
34#[derive(Clone, Debug, Default, Eq, PartialEq, Serialize)]
35#[serde(rename_all = "camelCase")]
36pub struct StorageProof {
37	pub key: U256,
38	pub value: U256,
39	pub proof: Vec<Bytes>,
40}
41
42/// Account information.
43#[derive(Clone, Debug, Default, Eq, PartialEq, Serialize)]
44#[serde(rename_all = "camelCase")]
45pub struct EthAccount {
46	pub address: H160,
47	pub balance: U256,
48	pub nonce: U256,
49	pub code_hash: H256,
50	pub storage_hash: H256,
51	pub account_proof: Vec<Bytes>,
52	pub storage_proof: Vec<StorageProof>,
53}
54
55/// Extended account information (used by `parity_allAccountInfo`).
56#[derive(Clone, Debug, Default, Eq, PartialEq, Serialize)]
57pub struct ExtAccountInfo {
58	/// Account name
59	pub name: String,
60	/// Account meta JSON
61	pub meta: String,
62	/// Account UUID (`None` for address book entries)
63	#[serde(skip_serializing_if = "Option::is_none")]
64	pub uuid: Option<String>,
65}
66
67/// account derived from a signature
68/// as well as information that tells if it is valid for
69/// the current chain
70#[derive(Clone, Debug, Serialize)]
71#[serde(rename_all = "camelCase")]
72pub struct RecoveredAccount {
73	/// address of the recovered account
74	pub address: Address,
75	/// public key of the recovered account
76	pub public_key: Public,
77	/// If the signature contains chain replay protection,
78	/// And the chain_id encoded within the signature
79	/// matches the current chain this would be true, otherwise false.
80	pub is_valid_for_current_chain: bool,
81}
