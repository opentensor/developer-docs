19use std::fmt;
20
21use serde::{
22	de::{Error, Visitor},
23	Deserialize, Deserializer,
24};
25
26/// Represents usize.
27#[derive(Debug, Eq, PartialEq)]
28pub struct Index(usize);
29
30impl Index {
31	/// Convert to usize
32	pub fn value(&self) -> usize {
33		self.0
34	}
35}
36
37impl<'a> Deserialize<'a> for Index {
38	fn deserialize<D>(deserializer: D) -> Result<Index, D::Error>
39	where
40		D: Deserializer<'a>,
41	{
42		deserializer.deserialize_any(IndexVisitor)
43	}
44}
45
46struct IndexVisitor;
47
48impl<'a> Visitor<'a> for IndexVisitor {
49	type Value = Index;
50
51	fn expecting(&self, formatter: &mut fmt::Formatter) -> fmt::Result {
52		write!(formatter, "a hex-encoded or decimal index")
53	}
54
55	fn visit_str<E>(self, value: &str) -> Result<Self::Value, E>
56	where
57		E: Error,
58	{
59		match value {
60			_ if value.starts_with("0x") => usize::from_str_radix(&value[2..], 16)
61				.map(Index)
62				.map_err(|e| Error::custom(format!("Invalid index: {}", e))),
63			_ => value
64				.parse::<usize>()
65				.map(Index)
66				.map_err(|e| Error::custom(format!("Invalid index: {}", e))),
67		}
68	}
69
70	fn visit_string<E>(self, value: String) -> Result<Self::Value, E>
71	where
72		E: Error,
73	{
74		self.visit_str(value.as_ref())
75	}
76
77	fn visit_u64<E>(self, value: u64) -> Result<Self::Value, E>
78	where
79		E: Error,
80	{
81		Ok(Index(value as usize))
82	}
83}

85#[cfg(test)]
86mod tests {
87	use super::*;
88	use serde_json;
89
90	#[test]
91	fn index_deserialization() {
92		let s = r#"["0xa", "10", 42]"#;
93		let deserialized: Vec<Index> = serde_json::from_str(s).unwrap();
94		assert_eq!(deserialized, vec![Index(10), Index(10), Index(42)]);
95	}
