25/// Transaction
26#[derive(Clone, Debug, Default, Eq, PartialEq, Serialize)]
27#[serde(rename_all = "camelCase")]
28pub struct Transaction {
29	/// EIP-2718 transaction type
30	#[serde(rename = "type")]
31	pub transaction_type: U256,
32	/// Hash
33	pub hash: H256,
34	/// Nonce
35	pub nonce: U256,
36	/// Block hash
37	pub block_hash: Option<H256>,
38	/// Block number
39	pub block_number: Option<U256>,
40	/// Transaction Index
41	pub transaction_index: Option<U256>,
42	/// Sender
43	pub from: H160,
44	/// Recipient
45	pub to: Option<H160>,
46	/// Transferred value
47	pub value: U256,
48	/// Gas
49	pub gas: U256,
50	/// Gas Price
51	#[serde(skip_serializing_if = "Option::is_none")]
52	pub gas_price: Option<U256>,
53	/// Max BaseFeePerGas the user is willing to pay.
54	#[serde(skip_serializing_if = "Option::is_none")]
55	pub max_fee_per_gas: Option<U256>,
56	/// The miner's tip.
57	#[serde(skip_serializing_if = "Option::is_none")]
58	pub max_priority_fee_per_gas: Option<U256>,
59	/// Data
60	pub input: Bytes,
61	/// Creates contract
62	pub creates: Option<H160>,
63	/// The network id of the transaction, if any.
64	#[serde(skip_serializing_if = "Option::is_none")]
65	pub chain_id: Option<U64>,
66	/// Pre-pay to warm storage access.
67	#[serde(skip_serializing_if = "Option::is_none")]
68	pub access_list: Option<Vec<AccessListItem>>,
69	/// The parity (0 for even, 1 for odd) of the y-value of the secp256k1 signature.
70	#[serde(skip_serializing_if = "Option::is_none")]
71	pub y_parity: Option<U256>,
72	/// The standardised V field of the signature.
73	///
74	/// For backwards compatibility, `v` is optionally provided as an alternative to `yParity`.
75	/// This field is DEPRECATED and all use of it should migrate to `yParity`.
76	#[serde(skip_serializing_if = "Option::is_none")]
77	pub v: Option<U256>,
78	/// The R field of the signature.
79	pub r: U256,
80	/// The S field of the signature.
81	pub s: U256,
82}
83
84impl BuildFrom for Transaction {
85	fn build_from(from: H160, transaction: &EthereumTransaction) -> Self {
86		let hash = transaction.hash();
87		match transaction {
88			EthereumTransaction::Legacy(t) => Self {
89				transaction_type: U256::from(0),
90				hash,
91				nonce: t.nonce,
92				block_hash: None,
93				block_number: None,
94				transaction_index: None,
95				from,
96				to: match t.action {
97					TransactionAction::Call(to) => Some(to),
98					TransactionAction::Create => None,
99				},
100				value: t.value,
101				gas: t.gas_limit,
102				gas_price: Some(t.gas_price),
103				max_fee_per_gas: None,
104				max_priority_fee_per_gas: None,
105				input: Bytes(t.input.clone()),
106				creates: None,
107				chain_id: t.signature.chain_id().map(U64::from),
108				access_list: None,
109				y_parity: None,
110				v: Some(U256::from(t.signature.v())),
111				r: U256::from(t.signature.r().as_bytes()),
112				s: U256::from(t.signature.s().as_bytes()),
113			},
114			EthereumTransaction::EIP2930(t) => Self {
115				transaction_type: U256::from(1),
116				hash,
117				nonce: t.nonce,
118				block_hash: None,
119				block_number: None,
120				transaction_index: None,
121				from,
122				to: match t.action {
123					TransactionAction::Call(to) => Some(to),
124					TransactionAction::Create => None,
125				},
126				value: t.value,
127				gas: t.gas_limit,
128				gas_price: Some(t.gas_price),
129				max_fee_per_gas: None,
130				max_priority_fee_per_gas: None,
131				input: Bytes(t.input.clone()),
132				creates: None,
133				chain_id: Some(U64::from(t.chain_id)),
134				access_list: Some(t.access_list.clone()),
135				y_parity: Some(U256::from(t.odd_y_parity as u8)),
136				v: Some(U256::from(t.odd_y_parity as u8)),
137				r: U256::from(t.r.as_bytes()),
138				s: U256::from(t.s.as_bytes()),
139			},
140			EthereumTransaction::EIP1559(t) => Self {
141				transaction_type: U256::from(2),
142				hash,
143				nonce: t.nonce,
144				block_hash: None,
145				block_number: None,
146				transaction_index: None,
147				from,
148				to: match t.action {
149					TransactionAction::Call(to) => Some(to),
150					TransactionAction::Create => None,
151				},
152				value: t.value,
153				gas: t.gas_limit,
154				// If transaction is not mined yet, gas price is considered just max fee per gas.
155				gas_price: Some(t.max_fee_per_gas),
156				max_fee_per_gas: Some(t.max_fee_per_gas),
157				max_priority_fee_per_gas: Some(t.max_priority_fee_per_gas),
158				input: Bytes(t.input.clone()),
159				creates: None,
160				chain_id: Some(U64::from(t.chain_id)),
161				access_list: Some(t.access_list.clone()),
162				y_parity: Some(U256::from(t.odd_y_parity as u8)),
163				v: Some(U256::from(t.odd_y_parity as u8)),
164				r: U256::from(t.r.as_bytes()),
165				s: U256::from(t.s.as_bytes()),
166			},
167		}
168	}
169}

171/// Local Transaction Status
172#[derive(Debug)]
173pub enum LocalTransactionStatus {
174	/// Transaction is pending
175	Pending,
176	/// Transaction is in future part of the queue
177	Future,
178	/// Transaction was mined.
179	Mined(Transaction),
180	/// Transaction was removed from the queue, but not mined.
181	Culled(Transaction),
182	/// Transaction was dropped because of limit.
183	Dropped(Transaction),
184	/// Transaction was replaced by transaction with higher gas price.
185	Replaced(Transaction, U256, H256),
186	/// Transaction never got into the queue.
187	Rejected(Transaction, String),
188	/// Transaction is invalid.
189	Invalid(Transaction),
190	/// Transaction was canceled.
191	Canceled(Transaction),
192}
193
194impl Serialize for LocalTransactionStatus {
195	fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
196	where
197		S: Serializer,
198	{
199		use self::LocalTransactionStatus::*;
200
201		let elems = match *self {
202			Pending | Future => 1,
203			Mined(..) | Culled(..) | Dropped(..) | Invalid(..) | Canceled(..) => 2,
204			Rejected(..) => 3,
205			Replaced(..) => 4,
206		};
207
208		let status = "status";
209		let transaction = "transaction";
210
211		let mut struc = serializer.serialize_struct("LocalTransactionStatus", elems)?;
212		match *self {
213			Pending => struc.serialize_field(status, "pending")?,
214			Future => struc.serialize_field(status, "future")?,
215			Mined(ref tx) => {
216				struc.serialize_field(status, "mined")?;
217				struc.serialize_field(transaction, tx)?;
218			}
219			Culled(ref tx) => {
220				struc.serialize_field(status, "culled")?;
221				struc.serialize_field(transaction, tx)?;
222			}
223			Dropped(ref tx) => {
224				struc.serialize_field(status, "dropped")?;
225				struc.serialize_field(transaction, tx)?;
226			}
227			Canceled(ref tx) => {
228				struc.serialize_field(status, "canceled")?;
229				struc.serialize_field(transaction, tx)?;
230			}
231			Invalid(ref tx) => {
232				struc.serialize_field(status, "invalid")?;
233				struc.serialize_field(transaction, tx)?;
234			}
235			Rejected(ref tx, ref reason) => {
236				struc.serialize_field(status, "rejected")?;
237				struc.serialize_field(transaction, tx)?;
238				struc.serialize_field("error", reason)?;
239			}
240			Replaced(ref tx, ref gas_price, ref hash) => {
241				struc.serialize_field(status, "replaced")?;
242				struc.serialize_field(transaction, tx)?;
243				struc.serialize_field("hash", hash)?;
244				struc.serialize_field("gasPrice", gas_price)?;
245			}
246		}
247
248		struc.end()
249	}
250}
