104/// RPC handlers that can perform RPC queries.
105#[derive(Clone)]
106pub struct RpcHandlers(Arc<RpcModule<()>>);
107
108impl RpcHandlers {
109	/// Create PRC handlers instance.
110	pub fn new(inner: Arc<RpcModule<()>>) -> Self {
111		Self(inner)
112	}
113
114	/// Starts an RPC query.
115	///
116	/// The query is passed as a string and must be valid JSON-RPC request object.
117	///
118	/// Returns a response and a stream if the call successful, fails if the
119	/// query could not be decoded as a JSON-RPC request object.
120	///
121	/// If the request subscribes you to events, the `stream` can be used to
122	/// retrieve the events.
123	pub async fn rpc_query(
124		&self,
125		json_query: &str,
126	) -> Result<(String, tokio::sync::mpsc::Receiver<String>), serde_json::Error> {
127		// Because `tokio::sync::mpsc::channel` is used under the hood
128		// it will panic if it's set to usize::MAX.
129		//
130		// This limit is used to prevent panics and is large enough.
131		const TOKIO_MPSC_MAX_SIZE: usize = tokio::sync::Semaphore::MAX_PERMITS;
132
133		self.0.raw_json_request(json_query, TOKIO_MPSC_MAX_SIZE).await
134	}
135
136	/// Provides access to the underlying `RpcModule`
137	pub fn handle(&self) -> Arc<RpcModule<()>> {
138		self.0.clone()
139	}
140}

162/// Builds a future that continuously polls the network.
163async fn build_network_future<
164	B: BlockT,
165	C: BlockchainEvents<B>
166		+ HeaderBackend<B>
167		+ BlockBackend<B>
168		+ HeaderMetadata<B, Error = sp_blockchain::Error>
169		+ ProofProvider<B>
170		+ Send
171		+ Sync
172		+ 'static,
173	H: sc_network_common::ExHashT,
174	N: NetworkBackend<B, <B as BlockT>::Hash>,
175>(
176	network: N,
177	client: Arc<C>,
178	sync_service: Arc<SyncingService<B>>,
179	announce_imported_blocks: bool,
180) {
181	let mut imported_blocks_stream = client.import_notification_stream().fuse();
182
183	// Stream of finalized blocks reported by the client.
184	let mut finality_notification_stream = client.finality_notification_stream().fuse();
185
186	let network_run = network.run().fuse();
187	pin_mut!(network_run);
188
189	loop {
190		futures::select! {
191			// List of blocks that the client has imported.
192			notification = imported_blocks_stream.next() => {
193				let notification = match notification {
194					Some(n) => n,
195					// If this stream is shut down, that means the client has shut down, and the
196					// most appropriate thing to do for the network future is to shut down too.
197					None => {
198						debug!("Block import stream has terminated, shutting down the network future.");
199						return
200					},
201				};
202
203				if announce_imported_blocks {
204					sync_service.announce_block(notification.hash, None);
205				}
206
207				if notification.is_new_best {
208					sync_service.new_best_block_imported(
209						notification.hash,
210						*notification.header.number(),
211					);
212				}
213			}
214
215			// List of blocks that the client has finalized.
216			notification = finality_notification_stream.select_next_some() => {
217				sync_service.on_block_finalized(notification.hash, notification.header);
218			}
219
220			// Drive the network. Shut down the network future if `NetworkWorker` has terminated.
221			_ = network_run => {
222				debug!("`NetworkWorker` has terminated, shutting down the network future.");
223				return
224			}
225		}
226	}
227}

229/// Builds a future that processes system RPC requests.
230pub async fn build_system_rpc_future<
231	B: BlockT,
232	C: BlockchainEvents<B>
233		+ HeaderBackend<B>
234		+ BlockBackend<B>
235		+ HeaderMetadata<B, Error = sp_blockchain::Error>
236		+ ProofProvider<B>
237		+ Send
238		+ Sync
239		+ 'static,
240	H: sc_network_common::ExHashT,
241>(
242	role: Role,
243	network_service: Arc<dyn NetworkService>,
244	sync_service: Arc<SyncingService<B>>,
245	client: Arc<C>,
246	mut rpc_rx: TracingUnboundedReceiver<sc_rpc::system::Request<B>>,
247	should_have_peers: bool,
248) {
249	// Current best block at initialization, to report to the RPC layer.
250	let starting_block = client.info().best_number;
251
252	loop {
253		// Answer incoming RPC requests.
254		let Some(req) = rpc_rx.next().await else {
255			debug!("RPC requests stream has terminated, shutting down the system RPC future.");
256			return
257		};
258
259		match req {
260			sc_rpc::system::Request::Health(sender) => match sync_service.peers_info().await {
261				Ok(info) => {
262					let _ = sender.send(sc_rpc::system::Health {
263						peers: info.len(),
264						is_syncing: sync_service.is_major_syncing(),
265						should_have_peers,
266					});
267				},
268				Err(_) => log::error!("`SyncingEngine` shut down"),
269			},
270			sc_rpc::system::Request::LocalPeerId(sender) => {
271				let _ = sender.send(network_service.local_peer_id().to_base58());
272			},
273			sc_rpc::system::Request::LocalListenAddresses(sender) => {
274				let peer_id = (network_service.local_peer_id()).into();
275				let p2p_proto_suffix = sc_network::multiaddr::Protocol::P2p(peer_id);
276				let addresses = network_service
277					.listen_addresses()
278					.iter()
279					.map(|addr| addr.clone().with(p2p_proto_suffix.clone()).to_string())
280					.collect();
281				let _ = sender.send(addresses);
282			},
283			sc_rpc::system::Request::Peers(sender) => match sync_service.peers_info().await {
284				Ok(info) => {
285					let _ = sender.send(
286						info.into_iter()
287							.map(|(peer_id, p)| sc_rpc::system::PeerInfo {
288								peer_id: peer_id.to_base58(),
289								roles: format!("{:?}", p.roles),
290								best_hash: p.best_hash,
291								best_number: p.best_number,
292							})
293							.collect(),
294					);
295				},
296				Err(_) => log::error!("`SyncingEngine` shut down"),
297			},
298			sc_rpc::system::Request::NetworkState(sender) => {
299				let network_state = network_service.network_state().await;
300				if let Ok(network_state) = network_state {
301					if let Ok(network_state) = serde_json::to_value(network_state) {
302						let _ = sender.send(network_state);
303					}
304				} else {
305					break
306				}
307			},
308			sc_rpc::system::Request::NetworkAddReservedPeer(peer_addr, sender) => {
309				let result = match MultiaddrWithPeerId::try_from(peer_addr) {
310					Ok(peer) => network_service.add_reserved_peer(peer),
311					Err(err) => Err(err.to_string()),
312				};
313				let x = result.map_err(sc_rpc::system::error::Error::MalformattedPeerArg);
314				let _ = sender.send(x);
315			},
316			sc_rpc::system::Request::NetworkRemoveReservedPeer(peer_id, sender) => {
317				let _ = match peer_id.parse::<PeerId>() {
318					Ok(peer_id) => {
319						network_service.remove_reserved_peer(peer_id);
320						sender.send(Ok(()))
321					},
322					Err(e) => sender.send(Err(sc_rpc::system::error::Error::MalformattedPeerArg(
323						e.to_string(),
324					))),
325				};
326			},
327			sc_rpc::system::Request::NetworkReservedPeers(sender) => {
328				let Ok(reserved_peers) = network_service.reserved_peers().await else {
329					break;
330				};
331
332				let _ =
333					sender.send(reserved_peers.iter().map(|peer_id| peer_id.to_base58()).collect());
334			},
335			sc_rpc::system::Request::NodeRoles(sender) => {
336				use sc_rpc::system::NodeRole;
337
338				let node_role = match role {
339					Role::Authority { .. } => NodeRole::Authority,
340					Role::Full => NodeRole::Full,
341				};
342
343				let _ = sender.send(vec![node_role]);
344			},
345			sc_rpc::system::Request::SyncState(sender) => {
346				use sc_rpc::system::SyncState;
347
348				match sync_service.status().await.map(|status| status.best_seen_block) {
349					Ok(best_seen_block) => {
350						let best_number = client.info().best_number;
351						let _ = sender.send(SyncState {
352							starting_block,
353							current_block: best_number,
354							highest_block: best_seen_block.unwrap_or(best_number),
355						});
356					},
357					Err(_) => log::error!("`SyncingEngine` shut down"),
358				}
359			},
360		}
361	}
362
363	debug!("`NetworkWorker` has terminated, shutting down the system RPC future.");
364}

367mod waiting {
368	pub struct Server(pub Option<sc_rpc_server::Server>);
369
370	impl Drop for Server {
371		fn drop(&mut self) {
372			if let Some(server) = self.0.take() {
373				// This doesn't not wait for the server to be stopped but fires the signal.
374				let _ = server.stop();
375			}
376		}
377	}

380/// Starts RPC servers.
381pub fn start_rpc_servers<R>(
382	rpc_configuration: &RpcConfiguration,
383	registry: Option<&Registry>,
384	tokio_handle: &Handle,
385	gen_rpc_module: R,
386	rpc_id_provider: Option<Box<dyn sc_rpc_server::SubscriptionIdProvider>>,
387) -> Result<Box<dyn std::any::Any + Send + Sync>, error::Error>
388where
389	R: Fn() -> Result<RpcModule<()>, Error>,
390{
391	let endpoints: Vec<sc_rpc_server::RpcEndpoint> = if let Some(endpoints) =
392		rpc_configuration.addr.as_ref()
393	{
394		endpoints.clone()
395	} else {
396		let ipv6 =
397			SocketAddr::V6(SocketAddrV6::new(Ipv6Addr::LOCALHOST, rpc_configuration.port, 0, 0));
398		let ipv4 = SocketAddr::V4(SocketAddrV4::new(Ipv4Addr::LOCALHOST, rpc_configuration.port));
399
400		vec![
401			sc_rpc_server::RpcEndpoint {
402				batch_config: rpc_configuration.batch_config,
403				cors: rpc_configuration.cors.clone(),
404				listen_addr: ipv4,
405				max_buffer_capacity_per_connection: rpc_configuration.message_buffer_capacity,
406				max_connections: rpc_configuration.max_connections,
407				max_payload_in_mb: rpc_configuration.max_request_size,
408				max_payload_out_mb: rpc_configuration.max_response_size,
409				max_subscriptions_per_connection: rpc_configuration.max_subs_per_conn,
410				rpc_methods: rpc_configuration.methods.into(),
411				rate_limit: rpc_configuration.rate_limit,
412				rate_limit_trust_proxy_headers: rpc_configuration.rate_limit_trust_proxy_headers,
413				rate_limit_whitelisted_ips: rpc_configuration.rate_limit_whitelisted_ips.clone(),
414				retry_random_port: true,
415				is_optional: false,
416			},
417			sc_rpc_server::RpcEndpoint {
418				batch_config: rpc_configuration.batch_config,
419				cors: rpc_configuration.cors.clone(),
420				listen_addr: ipv6,
421				max_buffer_capacity_per_connection: rpc_configuration.message_buffer_capacity,
422				max_connections: rpc_configuration.max_connections,
423				max_payload_in_mb: rpc_configuration.max_request_size,
424				max_payload_out_mb: rpc_configuration.max_response_size,
425				max_subscriptions_per_connection: rpc_configuration.max_subs_per_conn,
426				rpc_methods: rpc_configuration.methods.into(),
427				rate_limit: rpc_configuration.rate_limit,
428				rate_limit_trust_proxy_headers: rpc_configuration.rate_limit_trust_proxy_headers,
429				rate_limit_whitelisted_ips: rpc_configuration.rate_limit_whitelisted_ips.clone(),
430				retry_random_port: true,
431				is_optional: true,
432			},
433		]
434	};
435
436	let metrics = sc_rpc_server::RpcMetrics::new(registry)?;
437	let rpc_api = gen_rpc_module()?;
438
439	let server_config = sc_rpc_server::Config {
440		endpoints,
441		rpc_api,
442		metrics,
443		id_provider: rpc_id_provider,
444		tokio_handle: tokio_handle.clone(),
445	};
446
447	// TODO: https://github.com/paritytech/substrate/issues/13773
448	//
449	// `block_in_place` is a hack to allow callers to call `block_on` prior to
450	// calling `start_rpc_servers`.
451	match tokio::task::block_in_place(|| {
452		tokio_handle.block_on(sc_rpc_server::start_server(server_config))
453	}) {
454		Ok(server) => Ok(Box::new(waiting::Server(Some(server)))),
455		Err(e) => Err(Error::Application(e)),
456	}
457}

472/// Get transactions for propagation.
473///
474/// Function extracted to simplify the test and prevent creating `ServiceFactory`.
475fn transactions_to_propagate<Pool, B, H, E>(pool: &Pool) -> Vec<(H, B::Extrinsic)>
476where
477	Pool: TransactionPool<Block = B, Hash = H, Error = E>,
478	B: BlockT,
479	H: std::hash::Hash + Eq + sp_runtime::traits::Member + sp_runtime::traits::MaybeSerialize,
480	E: IntoPoolError + From<sc_transaction_pool_api::error::Error>,
481{
482	pool.ready()
483		.filter(|t| t.is_propagable())
484		.map(|t| {
485			let hash = t.hash().clone();
486			let ex: B::Extrinsic = t.data().clone();
487			(hash, ex)
488		})
489		.collect()
490}
491
492impl<B, H, C, Pool, E> sc_network_transactions::config::TransactionPool<H, B>
493	for TransactionPoolAdapter<C, Pool>
494where
495	C: HeaderBackend<B>
496		+ BlockBackend<B>
497		+ HeaderMetadata<B, Error = sp_blockchain::Error>
498		+ ProofProvider<B>
499		+ Send
500		+ Sync
501		+ 'static,
502	Pool: 'static + TransactionPool<Block = B, Hash = H, Error = E>,
503	B: BlockT,
504	H: std::hash::Hash + Eq + sp_runtime::traits::Member + sp_runtime::traits::MaybeSerialize,
505	E: 'static + IntoPoolError + From<sc_transaction_pool_api::error::Error>,
506{
507	fn transactions(&self) -> Vec<(H, B::Extrinsic)> {
508		transactions_to_propagate(&*self.pool)
509	}
510
511	fn hash_of(&self, transaction: &B::Extrinsic) -> H {
512		self.pool.hash_of(transaction)
513	}
514
515	fn import(&self, transaction: B::Extrinsic) -> TransactionImportFuture {
516		let encoded = transaction.encode();
517		let uxt = match Decode::decode(&mut &encoded[..]) {
518			Ok(uxt) => uxt,
519			Err(e) => {
520				debug!("Transaction invalid: {:?}", e);
521				return Box::pin(futures::future::ready(TransactionImport::Bad))
522			},
523		};
524
525		let import_future = self.pool.submit_one(
526			self.client.info().best_hash,
527			sc_transaction_pool_api::TransactionSource::External,
528			uxt,
529		);
530		Box::pin(async move {
531			match import_future.await {
532				Ok(_) => TransactionImport::NewGood,
533				Err(e) => match e.into_pool_error() {
534					Ok(sc_transaction_pool_api::error::Error::AlreadyImported(_)) =>
535						TransactionImport::KnownGood,
536					Ok(e) => {
537						debug!("Error adding transaction to the pool: {:?}", e);
538						TransactionImport::Bad
539					},
540					Err(e) => {
541						debug!("Error converting pool error: {}", e);
542						// it is not bad at least, just some internal node logic error, so peer is
543						// innocent.
544						TransactionImport::KnownGood
545					},
546				},
547			}
548		})
549	}
550
551	fn on_broadcasted(&self, propagations: HashMap<H, Vec<String>>) {
552		self.pool.on_broadcasted(propagations)
553	}
554
555	fn transaction(&self, hash: &H) -> Option<B::Extrinsic> {
556		self.pool.ready_transaction(hash).and_then(
557			// Only propagable transactions should be resolved for network service.
558			|tx| if tx.is_propagable() { Some(tx.data().clone()) } else { None },
559		)
560	}
561}
