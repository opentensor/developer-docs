100/// Construct a local keystore shareable container
101pub struct KeystoreContainer(Arc<LocalKeystore>);
102
103impl KeystoreContainer {
104	/// Construct KeystoreContainer
105	pub fn new(config: &KeystoreConfig) -> Result<Self, Error> {
106		let keystore = Arc::new(match config {
107			KeystoreConfig::Path { path, password } =>
108				LocalKeystore::open(path.clone(), password.clone())?,
109			KeystoreConfig::InMemory => LocalKeystore::in_memory(),
110		});
111
112		Ok(Self(keystore))
113	}
114
115	/// Returns a shared reference to a dynamic `Keystore` trait implementation.
116	pub fn keystore(&self) -> KeystorePtr {
117		self.0.clone()
118	}
119
120	/// Returns a shared reference to the local keystore .
121	pub fn local_keystore(&self) -> Arc<LocalKeystore> {
122		self.0.clone()
123	}
124}

181/// Create the initial parts of a full node.
182pub fn new_full_parts_with_genesis_builder<TBl, TRtApi, TExec, TBuildGenesisBlock>(
183	config: &Configuration,
184	telemetry: Option<TelemetryHandle>,
185	executor: TExec,
186	backend: Arc<TFullBackend<TBl>>,
187	genesis_block_builder: TBuildGenesisBlock,
188	enable_import_proof_recording: bool,
189) -> Result<TFullParts<TBl, TRtApi, TExec>, Error>
190where
191	TBl: BlockT,
192	TExec: CodeExecutor + RuntimeVersionOf + Clone,
193	TBuildGenesisBlock: BuildGenesisBlock<
194		TBl,
195		BlockImportOperation = <Backend<TBl> as sc_client_api::backend::Backend<TBl>>::BlockImportOperation
196	>,
197{
198	let keystore_container = KeystoreContainer::new(&config.keystore)?;
199
200	let task_manager = {
201		let registry = config.prometheus_config.as_ref().map(|cfg| &cfg.registry);
202		TaskManager::new(config.tokio_handle.clone(), registry)?
203	};
204
205	let chain_spec = &config.chain_spec;
206	let fork_blocks = get_extension::<ForkBlocks<TBl>>(chain_spec.extensions())
207		.cloned()
208		.unwrap_or_default();
209
210	let bad_blocks = get_extension::<BadBlocks<TBl>>(chain_spec.extensions())
211		.cloned()
212		.unwrap_or_default();
213
214	let client = {
215		let extensions = ExecutionExtensions::new(None, Arc::new(executor.clone()));
216
217		let wasm_runtime_substitutes = config
218			.chain_spec
219			.code_substitutes()
220			.into_iter()
221			.map(|(n, c)| {
222				let number = NumberFor::<TBl>::from_str(&n).map_err(|_| {
223					Error::Application(Box::from(format!(
224						"Failed to parse `{}` as block number for code substitutes. \
225						 In an old version the key for code substitute was a block hash. \
226						 Please update the chain spec to a version that is compatible with your node.",
227						n
228					)))
229				})?;
230				Ok((number, c))
231			})
232			.collect::<Result<std::collections::HashMap<_, _>, Error>>()?;
233
234		let client = new_client(
235			backend.clone(),
236			executor,
237			genesis_block_builder,
238			fork_blocks,
239			bad_blocks,
240			extensions,
241			Box::new(task_manager.spawn_handle()),
242			config.prometheus_config.as_ref().map(|config| config.registry.clone()),
243			telemetry,
244			ClientConfig {
245				offchain_worker_enabled: config.offchain_worker.enabled,
246				offchain_indexing_api: config.offchain_worker.indexing_enabled,
247				wasm_runtime_overrides: config.wasm_runtime_overrides.clone(),
248				no_genesis: config.no_genesis(),
249				wasm_runtime_substitutes,
250				enable_import_proof_recording,
251			},
252		)?;
253
254		client
255	};
256
257	Ok((client, backend, keystore_container, task_manager))
258}

271/// Creates a [`WasmExecutor`] according to [`ExecutorConfiguration`].
272pub fn new_wasm_executor<H: HostFunctions>(config: &ExecutorConfiguration) -> WasmExecutor<H> {
273	let strategy = config
274		.default_heap_pages
275		.map_or(DEFAULT_HEAP_ALLOC_STRATEGY, |p| HeapAllocStrategy::Static { extra_pages: p as _ });
276	WasmExecutor::<H>::builder()
277		.with_execution_method(config.wasm_method)
278		.with_onchain_heap_alloc_strategy(strategy)
279		.with_offchain_heap_alloc_strategy(strategy)
280		.with_max_runtime_instances(config.max_runtime_instances)
281		.with_runtime_cache_size(config.runtime_cache_size)
282		.build()
283}

375/// Spawn the tasks that are required to run a node.
376pub fn spawn_tasks<TBl, TBackend, TExPool, TRpc, TCl>(
377	params: SpawnTasksParams<TBl, TCl, TExPool, TRpc, TBackend>,
378) -> Result<RpcHandlers, Error>
379where
380	TCl: ProvideRuntimeApi<TBl>
381		+ HeaderMetadata<TBl, Error = sp_blockchain::Error>
382		+ Chain<TBl>
383		+ BlockBackend<TBl>
384		+ BlockIdTo<TBl, Error = sp_blockchain::Error>
385		+ ProofProvider<TBl>
386		+ HeaderBackend<TBl>
387		+ BlockchainEvents<TBl>
388		+ ExecutorProvider<TBl>
389		+ UsageProvider<TBl>
390		+ StorageProvider<TBl, TBackend>
391		+ CallApiAt<TBl>
392		+ Send
393		+ 'static,
394	<TCl as ProvideRuntimeApi<TBl>>::Api: sp_api::Metadata<TBl>
395		+ sp_transaction_pool::runtime_api::TaggedTransactionQueue<TBl>
396		+ sp_session::SessionKeys<TBl>
397		+ sp_api::ApiExt<TBl>,
398	TBl: BlockT,
399	TBl::Hash: Unpin,
400	TBl::Header: Unpin,
401	TBackend: 'static + sc_client_api::backend::Backend<TBl> + Send,
402	TExPool: MaintainedTransactionPool<Block = TBl, Hash = <TBl as BlockT>::Hash> + 'static,
403{
404	let SpawnTasksParams {
405		mut config,
406		task_manager,
407		client,
408		backend,
409		keystore,
410		transaction_pool,
411		rpc_builder,
412		network,
413		system_rpc_tx,
414		tx_handler_controller,
415		sync_service,
416		telemetry,
417	} = params;
418
419	let chain_info = client.usage_info().chain;
420
421	sp_session::generate_initial_session_keys(
422		client.clone(),
423		chain_info.best_hash,
424		config.dev_key_seed.clone().map(|s| vec![s]).unwrap_or_default(),
425		keystore.clone(),
426	)
427	.map_err(|e| Error::Application(Box::new(e)))?;
428
429	let sysinfo = sc_sysinfo::gather_sysinfo();
430	sc_sysinfo::print_sysinfo(&sysinfo);
431
432	let telemetry = telemetry
433		.map(|telemetry| {
434			init_telemetry(
435				config.network.node_name.clone(),
436				config.impl_name.clone(),
437				config.impl_version.clone(),
438				config.chain_spec.name().to_string(),
439				config.role.is_authority(),
440				network.clone(),
441				client.clone(),
442				telemetry,
443				Some(sysinfo),
444			)
445		})
446		.transpose()?;
447
448	info!("ðŸ“¦ Highest known block at #{}", chain_info.best_number);
449
450	let spawn_handle = task_manager.spawn_handle();
451
452	// Inform the tx pool about imported and finalized blocks.
453	spawn_handle.spawn(
454		"txpool-notifications",
455		Some("transaction-pool"),
456		sc_transaction_pool::notification_future(client.clone(), transaction_pool.clone()),
457	);
458
459	spawn_handle.spawn(
460		"on-transaction-imported",
461		Some("transaction-pool"),
462		propagate_transaction_notifications(
463			transaction_pool.clone(),
464			tx_handler_controller,
465			telemetry.clone(),
466		),
467	);
468
469	// Prometheus metrics.
470	let metrics_service =
471		if let Some(PrometheusConfig { port, registry }) = config.prometheus_config.clone() {
472			// Set static metrics.
473			let metrics = MetricsService::with_prometheus(
474				telemetry,
475				&registry,
476				config.role,
477				&config.network.node_name,
478				&config.impl_version,
479			)?;
480			spawn_handle.spawn(
481				"prometheus-endpoint",
482				None,
483				prometheus_endpoint::init_prometheus(port, registry).map(drop),
484			);
485
486			metrics
487		} else {
488			MetricsService::new(telemetry)
489		};
490
491	// Periodically updated metrics and telemetry updates.
492	spawn_handle.spawn(
493		"telemetry-periodic-send",
494		None,
495		metrics_service.run(
496			client.clone(),
497			transaction_pool.clone(),
498			network.clone(),
499			sync_service.clone(),
500		),
501	);
502
503	let rpc_id_provider = config.rpc.id_provider.take();
504
505	// jsonrpsee RPC
506	let gen_rpc_module = || {
507		gen_rpc_module(
508			task_manager.spawn_handle(),
509			client.clone(),
510			transaction_pool.clone(),
511			keystore.clone(),
512			system_rpc_tx.clone(),
513			config.impl_name.clone(),
514			config.impl_version.clone(),
515			config.chain_spec.as_ref(),
516			&config.state_pruning,
517			config.blocks_pruning,
518			backend.clone(),
519			&*rpc_builder,
520		)
521	};
522
523	let rpc_server_handle = start_rpc_servers(
524		&config.rpc,
525		config.prometheus_registry(),
526		&config.tokio_handle,
527		gen_rpc_module,
528		rpc_id_provider,
529	)?;
530	let in_memory_rpc = {
531		let mut module = gen_rpc_module()?;
532		module.extensions_mut().insert(DenyUnsafe::No);
533		module
534	};
535
536	let in_memory_rpc_handle = RpcHandlers::new(Arc::new(in_memory_rpc));
537
538	// Spawn informant task
539	spawn_handle.spawn(
540		"informant",
541		None,
542		sc_informant::build(client.clone(), network, sync_service.clone()),
543	);
544
545	task_manager.keep_alive((config.base_path, rpc_server_handle));
546
547	Ok(in_memory_rpc_handle)
548}

550/// Returns a future that forwards imported transactions to the transaction networking protocol.
551pub async fn propagate_transaction_notifications<Block, ExPool>(
552	transaction_pool: Arc<ExPool>,
553	tx_handler_controller: sc_network_transactions::TransactionsHandlerController<
554		<Block as BlockT>::Hash,
555	>,
556	telemetry: Option<TelemetryHandle>,
557) where
558	Block: BlockT,
559	ExPool: MaintainedTransactionPool<Block = Block, Hash = <Block as BlockT>::Hash>,
560{
561	// transaction notifications
562	transaction_pool
563		.import_notification_stream()
564		.for_each(move |hash| {
565			tx_handler_controller.propagate_transaction(hash);
566			let status = transaction_pool.status();
567			telemetry!(
568				telemetry;
569				SUBSTRATE_INFO;
570				"txpool.import";
571				"ready" => status.ready,
572				"future" => status.future,
573			);
574			ready(())
575		})
576		.await;
577}

579/// Initialize telemetry with provided configuration and return telemetry handle
580pub fn init_telemetry<Block, Client, Network>(
581	name: String,
582	implementation: String,
583	version: String,
584	chain: String,
585	authority: bool,
586	network: Network,
587	client: Arc<Client>,
588	telemetry: &mut Telemetry,
589	sysinfo: Option<sc_telemetry::SysInfo>,
590) -> sc_telemetry::Result<TelemetryHandle>
591where
592	Block: BlockT,
593	Client: BlockBackend<Block>,
594	Network: NetworkStateInfo,
595{
596	let genesis_hash = client.block_hash(Zero::zero()).ok().flatten().unwrap_or_default();
597	let connection_message = ConnectionMessage {
598		name,
599		implementation,
600		version,
601		target_os: sc_sysinfo::TARGET_OS.into(),
602		target_arch: sc_sysinfo::TARGET_ARCH.into(),
603		target_env: sc_sysinfo::TARGET_ENV.into(),
604		config: String::new(),
605		chain,
606		genesis_hash: format!("{:?}", genesis_hash),
607		authority,
608		startup_time: SystemTime::UNIX_EPOCH
609			.elapsed()
610			.map(|dur| dur.as_millis())
611			.unwrap_or(0)
612			.to_string(),
613		network_id: network.local_peer_id().to_base58(),
614		sysinfo,
615	};
616
617	telemetry.start_telemetry(connection_message)?;
618
619	Ok(telemetry.handle())
620}

622/// Generate RPC module using provided configuration
623pub fn gen_rpc_module<TBl, TBackend, TCl, TRpc, TExPool>(
624	spawn_handle: SpawnTaskHandle,
625	client: Arc<TCl>,
626	transaction_pool: Arc<TExPool>,
627	keystore: KeystorePtr,
628	system_rpc_tx: TracingUnboundedSender<sc_rpc::system::Request<TBl>>,
629	impl_name: String,
630	impl_version: String,
631	chain_spec: &dyn ChainSpec,
632	state_pruning: &Option<PruningMode>,
633	blocks_pruning: BlocksPruning,
634	backend: Arc<TBackend>,
635	rpc_builder: &(dyn Fn(SubscriptionTaskExecutor) -> Result<RpcModule<TRpc>, Error>),
636) -> Result<RpcModule<()>, Error>
637where
638	TBl: BlockT,
639	TCl: ProvideRuntimeApi<TBl>
640		+ BlockchainEvents<TBl>
641		+ HeaderBackend<TBl>
642		+ HeaderMetadata<TBl, Error = sp_blockchain::Error>
643		+ ExecutorProvider<TBl>
644		+ CallApiAt<TBl>
645		+ ProofProvider<TBl>
646		+ StorageProvider<TBl, TBackend>
647		+ BlockBackend<TBl>
648		+ Send
649		+ Sync
650		+ 'static,
651	TBackend: sc_client_api::backend::Backend<TBl> + 'static,
652	<TCl as ProvideRuntimeApi<TBl>>::Api: sp_session::SessionKeys<TBl> + sp_api::Metadata<TBl>,
653	TExPool: MaintainedTransactionPool<Block = TBl, Hash = <TBl as BlockT>::Hash> + 'static,
654	TBl::Hash: Unpin,
655	TBl::Header: Unpin,
656{
657	let system_info = sc_rpc::system::SystemInfo {
658		chain_name: chain_spec.name().into(),
659		impl_name,
660		impl_version,
661		properties: chain_spec.properties(),
662		chain_type: chain_spec.chain_type(),
663	};
664
665	let mut rpc_api = RpcModule::new(());
666	let task_executor = Arc::new(spawn_handle);
667
668	let (chain, state, child_state) = {
669		let chain = sc_rpc::chain::new_full(client.clone(), task_executor.clone()).into_rpc();
670		let (state, child_state) = sc_rpc::state::new_full(client.clone(), task_executor.clone());
671		let state = state.into_rpc();
672		let child_state = child_state.into_rpc();
673
674		(chain, state, child_state)
675	};
676
677	const MAX_TRANSACTION_PER_CONNECTION: usize = 16;
678
679	let transaction_broadcast_rpc_v2 = sc_rpc_spec_v2::transaction::TransactionBroadcast::new(
680		client.clone(),
681		transaction_pool.clone(),
682		task_executor.clone(),
683		MAX_TRANSACTION_PER_CONNECTION,
684	)
685	.into_rpc();
686
687	let transaction_v2 = sc_rpc_spec_v2::transaction::Transaction::new(
688		client.clone(),
689		transaction_pool.clone(),
690		task_executor.clone(),
691	)
692	.into_rpc();
693
694	let chain_head_v2 = sc_rpc_spec_v2::chain_head::ChainHead::new(
695		client.clone(),
696		backend.clone(),
697		task_executor.clone(),
698		// Defaults to sensible limits for the `ChainHead`.
699		sc_rpc_spec_v2::chain_head::ChainHeadConfig::default(),
700	)
701	.into_rpc();
702
703	// Part of the RPC v2 spec.
704	// An archive node that can respond to the `archive` RPC-v2 queries is a node with:
705	// - state pruning in archive mode: The storage of blocks is kept around
706	// - block pruning in archive mode: The block's body is kept around
707	let is_archive_node = state_pruning.as_ref().map(|sp| sp.is_archive()).unwrap_or(false) &&
708		blocks_pruning.is_archive();
709	let genesis_hash = client.hash(Zero::zero()).ok().flatten().expect("Genesis block exists; qed");
710	if is_archive_node {
711		let archive_v2 = sc_rpc_spec_v2::archive::Archive::new(
712			client.clone(),
713			backend.clone(),
714			genesis_hash,
715			// Defaults to sensible limits for the `Archive`.
716			sc_rpc_spec_v2::archive::ArchiveConfig::default(),
717		)
718		.into_rpc();
719		rpc_api.merge(archive_v2).map_err(|e| Error::Application(e.into()))?;
720	}
721
722	// ChainSpec RPC-v2.
723	let chain_spec_v2 = sc_rpc_spec_v2::chain_spec::ChainSpec::new(
724		chain_spec.name().into(),
725		genesis_hash,
726		chain_spec.properties(),
727	)
728	.into_rpc();
729
730	let author = sc_rpc::author::Author::new(
731		client.clone(),
732		transaction_pool,
733		keystore,
734		task_executor.clone(),
735	)
736	.into_rpc();
737
738	let system = sc_rpc::system::System::new(system_info, system_rpc_tx).into_rpc();
739
740	if let Some(storage) = backend.offchain_storage() {
741		let offchain = sc_rpc::offchain::Offchain::new(storage).into_rpc();
742
743		rpc_api.merge(offchain).map_err(|e| Error::Application(e.into()))?;
744	}
745
746	// Part of the RPC v2 spec.
747	rpc_api.merge(transaction_v2).map_err(|e| Error::Application(e.into()))?;
748	rpc_api
749		.merge(transaction_broadcast_rpc_v2)
750		.map_err(|e| Error::Application(e.into()))?;
751	rpc_api.merge(chain_head_v2).map_err(|e| Error::Application(e.into()))?;
752	rpc_api.merge(chain_spec_v2).map_err(|e| Error::Application(e.into()))?;
753
754	// Part of the old RPC spec.
755	rpc_api.merge(chain).map_err(|e| Error::Application(e.into()))?;
756	rpc_api.merge(author).map_err(|e| Error::Application(e.into()))?;
757	rpc_api.merge(system).map_err(|e| Error::Application(e.into()))?;
758	rpc_api.merge(state).map_err(|e| Error::Application(e.into()))?;
759	rpc_api.merge(child_state).map_err(|e| Error::Application(e.into()))?;
760	// Additional [`RpcModule`]s defined in the node to fit the specific blockchain
761	let extra_rpcs = rpc_builder(task_executor.clone())?;
762	rpc_api.merge(extra_rpcs).map_err(|e| Error::Application(e.into()))?;
763
764	Ok(rpc_api)
765}

767/// Parameters to pass into `build_network`.
768pub struct BuildNetworkParams<
769	'a,
770	TBl: BlockT,
771	TNet: NetworkBackend<TBl, <TBl as BlockT>::Hash>,
772	TExPool,
773	TImpQu,
774	TCl,
775> {
776	/// The service configuration.
777	pub config: &'a Configuration,
778	/// Full network configuration.
779	pub net_config: FullNetworkConfiguration<TBl, <TBl as BlockT>::Hash, TNet>,
780	/// A shared client returned by `new_full_parts`.
781	pub client: Arc<TCl>,
782	/// A shared transaction pool.
783	pub transaction_pool: Arc<TExPool>,
784	/// A handle for spawning tasks.
785	pub spawn_handle: SpawnTaskHandle,
786	/// An import queue.
787	pub import_queue: TImpQu,
788	/// A block announce validator builder.
789	pub block_announce_validator_builder:
790		Option<Box<dyn FnOnce(Arc<TCl>) -> Box<dyn BlockAnnounceValidator<TBl> + Send> + Send>>,
791	/// Optional warp sync config.
792	pub warp_sync_config: Option<WarpSyncConfig<TBl>>,
793	/// User specified block relay params. If not specified, the default
794	/// block request handler will be used.
795	pub block_relay: Option<BlockRelayParams<TBl, TNet>>,
796	/// Metrics.
797	pub metrics: NotificationMetrics,
798}
799
800/// Build the network service, the network status sinks and an RPC sender.
801pub fn build_network<TBl, TNet, TExPool, TImpQu, TCl>(
802	params: BuildNetworkParams<TBl, TNet, TExPool, TImpQu, TCl>,
803) -> Result<
804	(
805		Arc<dyn sc_network::service::traits::NetworkService>,
806		TracingUnboundedSender<sc_rpc::system::Request<TBl>>,
807		sc_network_transactions::TransactionsHandlerController<<TBl as BlockT>::Hash>,
808		NetworkStarter,
809		Arc<SyncingService<TBl>>,
810	),
811	Error,
812>
813where
814	TBl: BlockT,
815	TCl: ProvideRuntimeApi<TBl>
816		+ HeaderMetadata<TBl, Error = sp_blockchain::Error>
817		+ Chain<TBl>
818		+ BlockBackend<TBl>
819		+ BlockIdTo<TBl, Error = sp_blockchain::Error>
820		+ ProofProvider<TBl>
821		+ HeaderBackend<TBl>
822		+ BlockchainEvents<TBl>
823		+ 'static,
824	TExPool: TransactionPool<Block = TBl, Hash = <TBl as BlockT>::Hash> + 'static,
825	TImpQu: ImportQueue<TBl> + 'static,
826	TNet: NetworkBackend<TBl, <TBl as BlockT>::Hash>,
827{
828	let BuildNetworkParams {
829		config,
830		mut net_config,
831		client,
832		transaction_pool,
833		spawn_handle,
834		import_queue,
835		block_announce_validator_builder,
836		warp_sync_config,
837		block_relay,
838		metrics,
839	} = params;
840
841	if warp_sync_config.is_none() && config.network.sync_mode.is_warp() {
842		return Err("Warp sync enabled, but no warp sync provider configured.".into())
843	}
844
845	if client.requires_full_sync() {
846		match config.network.sync_mode {
847			SyncMode::LightState { .. } =>
848				return Err("Fast sync doesn't work for archive nodes".into()),
849			SyncMode::Warp => return Err("Warp sync doesn't work for archive nodes".into()),
850			SyncMode::Full => {},
851		}
852	}
853
854	let protocol_id = config.protocol_id();
855	let genesis_hash = client
856		.block_hash(0u32.into())
857		.ok()
858		.flatten()
859		.expect("Genesis block exists; qed");
860
861	let block_announce_validator = if let Some(f) = block_announce_validator_builder {
862		f(client.clone())
863	} else {
864		Box::new(DefaultBlockAnnounceValidator)
865	};
866
867	let (chain_sync_network_provider, chain_sync_network_handle) = NetworkServiceProvider::new();
868	let (mut block_server, block_downloader, block_request_protocol_config) = match block_relay {
869		Some(params) => (params.server, params.downloader, params.request_response_config),
870		None => {
871			// Custom protocol was not specified, use the default block handler.
872			// Allow both outgoing and incoming requests.
873			let params = BlockRequestHandler::new::<TNet>(
874				chain_sync_network_handle.clone(),
875				&protocol_id,
876				config.chain_spec.fork_id(),
877				client.clone(),
878				config.network.default_peers_set.in_peers as usize +
879					config.network.default_peers_set.out_peers as usize,
880			);
881			(params.server, params.downloader, params.request_response_config)
882		},
883	};
884	spawn_handle.spawn("block-request-handler", Some("networking"), async move {
885		block_server.run().await;
886	});
887
888	let (state_request_protocol_config, state_request_protocol_name) = {
889		let num_peer_hint = net_config.network_config.default_peers_set_num_full as usize +
890			net_config.network_config.default_peers_set.reserved_nodes.len();
891		// Allow both outgoing and incoming requests.
892		let (handler, protocol_config) = StateRequestHandler::new::<TNet>(
893			&protocol_id,
894			config.chain_spec.fork_id(),
895			client.clone(),
896			num_peer_hint,
897		);
898		let config_name = protocol_config.protocol_name().clone();
899
900		spawn_handle.spawn("state-request-handler", Some("networking"), handler.run());
901		(protocol_config, config_name)
902	};
903
904	let (warp_sync_protocol_config, warp_request_protocol_name) = match warp_sync_config.as_ref() {
905		Some(WarpSyncConfig::WithProvider(warp_with_provider)) => {
906			// Allow both outgoing and incoming requests.
907			let (handler, protocol_config) = WarpSyncRequestHandler::new::<_, TNet>(
908				protocol_id.clone(),
909				genesis_hash,
910				config.chain_spec.fork_id(),
911				warp_with_provider.clone(),
912			);
913			let config_name = protocol_config.protocol_name().clone();
914
915			spawn_handle.spawn("warp-sync-request-handler", Some("networking"), handler.run());
916			(Some(protocol_config), Some(config_name))
917		},
918		_ => (None, None),
919	};
920
921	let light_client_request_protocol_config = {
922		// Allow both outgoing and incoming requests.
923		let (handler, protocol_config) = LightClientRequestHandler::new::<TNet>(
924			&protocol_id,
925			config.chain_spec.fork_id(),
926			client.clone(),
927		);
928		spawn_handle.spawn("light-client-request-handler", Some("networking"), handler.run());
929		protocol_config
930	};
931
932	// install request handlers to `FullNetworkConfiguration`
933	net_config.add_request_response_protocol(block_request_protocol_config);
934	net_config.add_request_response_protocol(state_request_protocol_config);
935	net_config.add_request_response_protocol(light_client_request_protocol_config);
936
937	if let Some(config) = warp_sync_protocol_config {
938		net_config.add_request_response_protocol(config);
939	}
940
941	let bitswap_config = config.network.ipfs_server.then(|| {
942		let (handler, config) = TNet::bitswap_server(client.clone());
943		spawn_handle.spawn("bitswap-request-handler", Some("networking"), handler);
944
945		config
946	});
947
948	// create transactions protocol and add it to the list of supported protocols of
949	let peer_store_handle = net_config.peer_store_handle();
950	let (transactions_handler_proto, transactions_config) =
951		sc_network_transactions::TransactionsHandlerPrototype::new::<_, TBl, TNet>(
952			protocol_id.clone(),
953			genesis_hash,
954			config.chain_spec.fork_id(),
955			metrics.clone(),
956			Arc::clone(&peer_store_handle),
957		);
958	net_config.add_notification_protocol(transactions_config);
959
960	// Start task for `PeerStore`
961	let peer_store = net_config.take_peer_store();
962	let peer_store_handle = peer_store.handle();
963	spawn_handle.spawn("peer-store", Some("networking"), peer_store.run());
964
965	let (engine, sync_service, block_announce_config) = SyncingEngine::new(
966		Roles::from(&config.role),
967		client.clone(),
968		config.prometheus_config.as_ref().map(|config| config.registry.clone()).as_ref(),
969		metrics.clone(),
970		&net_config,
971		protocol_id.clone(),
972		&config.chain_spec.fork_id().map(ToOwned::to_owned),
973		block_announce_validator,
974		warp_sync_config,
975		chain_sync_network_handle,
976		import_queue.service(),
977		block_downloader,
978		state_request_protocol_name,
979		warp_request_protocol_name,
980		Arc::clone(&peer_store_handle),
981	)?;
982	let sync_service_import_queue = sync_service.clone();
983	let sync_service = Arc::new(sync_service);
984
985	let genesis_hash = client.hash(Zero::zero()).ok().flatten().expect("Genesis block exists; qed");
986	let network_params = sc_network::config::Params::<TBl, <TBl as BlockT>::Hash, TNet> {
987		role: config.role,
988		executor: {
989			let spawn_handle = Clone::clone(&spawn_handle);
990			Box::new(move |fut| {
991				spawn_handle.spawn("libp2p-node", Some("networking"), fut);
992			})
993		},
994		network_config: net_config,
995		genesis_hash,
996		protocol_id: protocol_id.clone(),
997		fork_id: config.chain_spec.fork_id().map(ToOwned::to_owned),
998		metrics_registry: config.prometheus_config.as_ref().map(|config| config.registry.clone()),
999		block_announce_config,
1000		bitswap_config,
1001		notification_metrics: metrics,
1002	};
1003
1004	let has_bootnodes = !network_params.network_config.network_config.boot_nodes.is_empty();
1005	let network_mut = TNet::new(network_params)?;
1006	let network = network_mut.network_service().clone();
1007
1008	let (tx_handler, tx_handler_controller) = transactions_handler_proto.build(
1009		network.clone(),
1010		sync_service.clone(),
1011		Arc::new(TransactionPoolAdapter { pool: transaction_pool, client: client.clone() }),
1012		config.prometheus_config.as_ref().map(|config| &config.registry),
1013	)?;
1014	spawn_handle.spawn_blocking(
1015		"network-transactions-handler",
1016		Some("networking"),
1017		tx_handler.run(),
1018	);
1019
1020	spawn_handle.spawn_blocking(
1021		"chain-sync-network-service-provider",
1022		Some("networking"),
1023		chain_sync_network_provider.run(Arc::new(network.clone())),
1024	);
1025	spawn_handle.spawn("import-queue", None, import_queue.run(Box::new(sync_service_import_queue)));
1026	spawn_handle.spawn_blocking("syncing", None, engine.run());
1027
1028	let (system_rpc_tx, system_rpc_rx) = tracing_unbounded("mpsc_system_rpc", 10_000);
1029	spawn_handle.spawn(
1030		"system-rpc-handler",
1031		Some("networking"),
1032		build_system_rpc_future::<_, _, <TBl as BlockT>::Hash>(
1033			config.role,
1034			network_mut.network_service(),
1035			sync_service.clone(),
1036			client.clone(),
1037			system_rpc_rx,
1038			has_bootnodes,
1039		),
1040	);
1041
1042	let future = build_network_future::<_, _, <TBl as BlockT>::Hash, _>(
1043		network_mut,
1044		client,
1045		sync_service.clone(),
1046		config.announce_block,
1047	);
1048
1049	// TODO: Normally, one is supposed to pass a list of notifications protocols supported by the
1050	// node through the `NetworkConfiguration` struct. But because this function doesn't know in
1051	// advance which components, such as GrandPa or Polkadot, will be plugged on top of the
1052	// service, it is unfortunately not possible to do so without some deep refactoring. To
1053	// bypass this problem, the `NetworkService` provides a `register_notifications_protocol`
1054	// method that can be called even after the network has been initialized. However, we want to
1055	// avoid the situation where `register_notifications_protocol` is called *after* the network
1056	// actually connects to other peers. For this reason, we delay the process of the network
1057	// future until the user calls `NetworkStarter::start_network`.
1058	//
1059	// This entire hack should eventually be removed in favour of passing the list of protocols
1060	// through the configuration.
1061	//
1062	// See also https://github.com/paritytech/substrate/issues/6827
1063	let (network_start_tx, network_start_rx) = oneshot::channel();
1064
1065	// The network worker is responsible for gathering all network messages and processing
1066	// them. This is quite a heavy task, and at the time of the writing of this comment it
1067	// frequently happens that this future takes several seconds or in some situations
1068	// even more than a minute until it has processed its entire queue. This is clearly an
1069	// issue, and ideally we would like to fix the network future to take as little time as
1070	// possible, but we also take the extra harm-prevention measure to execute the networking
1071	// future using `spawn_blocking`.
1072	spawn_handle.spawn_blocking("network-worker", Some("networking"), async move {
1073		if network_start_rx.await.is_err() {
1074			log::warn!(
1075				"The NetworkStart returned as part of `build_network` has been silently dropped"
1076			);
1077			// This `return` might seem unnecessary, but we don't want to make it look like
1078			// everything is working as normal even though the user is clearly misusing the API.
1079			return
1080		}
1081
1082		future.await
1083	});
1084
1085	Ok((
1086		network,
1087		system_rpc_tx,
1088		tx_handler_controller,
1089		NetworkStarter(network_start_tx),
1090		sync_service.clone(),
1091	))
1092}
