33/// Extra dependencies for Ethereum compatibility.
34pub struct EthDeps<P, A: ChainApi, CT, CIDP> {
35    /// The client instance to use.
36    pub client: Arc<FullClient>,
37    /// Transaction pool instance.
38    pub pool: Arc<P>,
39    /// Graph pool instance.
40    pub graph: Arc<Pool<A>>,
41    /// Ethereum transaction converter.
42    pub converter: Option<CT>,
43    /// The Node authority flag
44    pub is_authority: bool,
45    /// Whether to enable dev signer
46    pub enable_dev_signer: bool,
47    /// Network service
48    pub network: Arc<dyn NetworkService>,
49    /// Chain syncing service
50    pub sync: Arc<SyncingService<Block>>,
51    /// Frontier Backend.
52    pub frontier_backend: Arc<dyn fc_api::Backend<Block>>,
53    /// Ethereum data access overrides.
54    pub storage_override: Arc<dyn StorageOverride<Block>>,
55    /// Cache for Ethereum block data.
56    pub block_data_cache: Arc<EthBlockDataCacheTask<Block>>,
57    /// EthFilterApi pool.
58    pub filter_pool: Option<FilterPool>,
59    /// Maximum number of logs in a query.
60    pub max_past_logs: u32,
61    /// Fee history cache.
62    pub fee_history_cache: FeeHistoryCache,
63    /// Maximum fee history cache size.
64    pub fee_history_cache_limit: FeeHistoryCacheLimit,
65    /// Maximum allowed gas limit will be ` block.gas_limit * execute_gas_limit_multiplier` when
66    /// using eth_call/eth_estimateGas.
67    pub execute_gas_limit_multiplier: u64,
68    /// Mandated parent hashes for a given block hash.
69    pub forced_parent_hashes: Option<BTreeMap<H256, H256>>,
70    /// Something that can create the inherent data providers for pending state
71    pub pending_create_inherent_data_providers: CIDP,
72}

86/// Full client dependencies.
87pub struct FullDeps<P, A: ChainApi, CT, CIDP> {
88    /// The client instance to use.

90    /// Transaction pool instance.
91    pub pool: Arc<P>,

98/// Instantiate all full RPC extensions.
99pub fn create_full<P, A, CT, CIDP>(
100    deps: FullDeps<P, A, CT, CIDP>,
101    subscription_task_executor: SubscriptionTaskExecutor,
102    pubsub_notification_sinks: Arc<
103        fc_mapping_sync::EthereumBlockNotificationSinks<
104            fc_mapping_sync::EthereumBlockNotification<Block>,
105        >,
106    >,
107) -> Result<RpcModule<()>, Box<dyn std::error::Error + Send + Sync>>
108where
109    P: TransactionPool<Block = Block> + 'static,
110    A: ChainApi<Block = Block> + 'static,
111    CIDP: CreateInherentDataProviders<Block, ()> + Send + Clone + 'static,
112    CT: fp_rpc::ConvertTransaction<<Block as BlockT>::Extrinsic> + Send + Sync + Clone + 'static,
113{
114    use pallet_transaction_payment_rpc::{TransactionPayment, TransactionPaymentApiServer};
115    use sc_consensus_manual_seal::rpc::{ManualSeal, ManualSealApiServer};
116    use substrate_frame_rpc_system::{System, SystemApiServer};
117    use subtensor_custom_rpc::{SubtensorCustom, SubtensorCustomApiServer};
118
119    let mut module = RpcModule::new(());
120    let FullDeps {
121        client,
122        pool,
123        command_sink,
124        eth,
125    } = deps;
126
127    // Custom RPC methods for Paratensor
128    module.merge(SubtensorCustom::new(client.clone()).into_rpc())?;
129
130    module.merge(System::new(client.clone(), pool.clone()).into_rpc())?;
131    module.merge(TransactionPayment::new(client).into_rpc())?;
132
133    // Extend this RPC with a custom API by using the following syntax.
134    // `YourRpcStruct` should have a reference to a client, which is needed
135    // to call into the runtime.
136    // `module.merge(YourRpcTrait::into_rpc(YourRpcStruct::new(ReferenceToClient, ...)))?;`
137
138    if let Some(command_sink) = command_sink {
139        module.merge(
140            // We provide the rpc handler with the sending end of the channel to allow the rpc
141            // send EngineCommands to the background block authorship task.
142            ManualSeal::new(command_sink).into_rpc(),
143        )?;
144    }
145
146    // Ethereum compatibility RPCs
147    let module = create_eth::<_, _, _, _, DefaultEthConfig>(
148        module,
149        eth,
150        subscription_task_executor,
151        pubsub_notification_sinks,
152    )?;
153
154    Ok(module)
155}
