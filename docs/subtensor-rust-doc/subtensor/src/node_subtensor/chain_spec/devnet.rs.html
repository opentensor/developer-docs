65// Configure initial storage state for FRAME modules.
66#[allow(clippy::too_many_arguments)]
67fn devnet_genesis(
68    initial_authorities: Vec<(AuraId, GrandpaId)>,
69    root_key: AccountId,
70    _endowed_accounts: Vec<AccountId>,
71    _enable_println: bool,
72    _stakes: Vec<(AccountId, Vec<(AccountId, (u64, u16))>)>,
73    _balances: Vec<(AccountId, u64)>,
74    _balances_issuance: u64,
75) -> serde_json::Value {
76    serde_json::json!({
77        "balances": {
78            "balances": vec![(root_key.clone(), 1_000_000_000_000u128)],
79        },
80        "aura": {
81            "authorities": initial_authorities.iter().map(|x| (x.0.clone())).collect::<Vec<_>>(),
82        },
83        "grandpa": {
84            "authorities": initial_authorities
85                .iter()
86                .map(|x| (x.1.clone(), 1))
87                .collect::<Vec<_>>(),
88        },
89        "sudo": {
90            "key": Some(root_key),
91        },
92    })
93}
