7pub fn finney_mainnet_config() -> Result<ChainSpec, String> {
8    let path: PathBuf = std::path::PathBuf::from("./snapshot.json");
9    let wasm_binary = WASM_BINARY.ok_or("Development wasm not available".to_string())?;
10
11    // We mmap the file into memory first, as this is *a lot* faster than using
12    // `serde_json::from_reader`. See https://github.com/serde-rs/json/issues/160
13    let file = File::open(&path)
14        .map_err(|e| format!("Error opening genesis file `{}`: {}", path.display(), e))?;
15
16    // SAFETY: `mmap` is fundamentally unsafe since technically the file can change
17    //         underneath us while it is mapped; in practice it's unlikely to be a problem
18    let bytes = unsafe {
19        memmap2::Mmap::map(&file)
20            .map_err(|e| format!("Error mmaping genesis file `{}`: {}", path.display(), e))?
21    };
22
23    let old_state: ColdkeyHotkeys =
24        json::from_slice(&bytes).map_err(|e| format!("Error parsing genesis file: {}", e))?;
25
26    let mut processed_stakes: Vec<(
27        sp_runtime::AccountId32,
28        Vec<(sp_runtime::AccountId32, (u64, u16))>,
29    )> = Vec::new();
30    for (coldkey_str, hotkeys) in old_state.stakes.iter() {
31        let coldkey = <sr25519::Public as Ss58Codec>::from_ss58check(coldkey_str)
32            .map_err(|e| e.to_string())?;
33        let coldkey_account = sp_runtime::AccountId32::from(coldkey);
34
35        let mut processed_hotkeys: Vec<(sp_runtime::AccountId32, (u64, u16))> = Vec::new();
36
37        for (hotkey_str, amount_uid) in hotkeys.iter() {
38            let (amount, uid) = amount_uid;
39            let hotkey = <sr25519::Public as Ss58Codec>::from_ss58check(hotkey_str)
40                .map_err(|e| e.to_string())?;
41            let hotkey_account = sp_runtime::AccountId32::from(hotkey);
42
43            processed_hotkeys.push((hotkey_account, (*amount, *uid)));
44        }
45        processed_hotkeys.sort();
46
47        processed_stakes.push((coldkey_account, processed_hotkeys));
48    }
49
50    processed_stakes.sort();
51
52    let mut balances_issuance: u64 = 0;
53    let mut processed_balances: Vec<(sp_runtime::AccountId32, u64)> = Vec::new();
54    for (key_str, amount) in old_state.balances.iter() {
55        let key =
56            <sr25519::Public as Ss58Codec>::from_ss58check(key_str).map_err(|e| e.to_string())?;
57        let key_account = sp_runtime::AccountId32::from(key);
58
59        processed_balances.push((key_account, *amount));
60        balances_issuance = balances_issuance
61            .checked_add(*amount)
62            .ok_or("Balances issuance overflowed".to_string())?;
63    }
64
65    processed_balances.sort();
66
67    // Give front-ends necessary data to present to users
68    let mut properties = sc_service::Properties::new();

206// Configure initial storage state for FRAME modules.
207#[allow(clippy::too_many_arguments)]
208fn finney_genesis(
209    initial_authorities: Vec<(AuraId, GrandpaId)>,
210    _root_key: AccountId,
211    _endowed_accounts: Vec<AccountId>,
212    _enable_println: bool,
213    stakes: Vec<(AccountId, Vec<(AccountId, (u64, u16))>)>,
214    balances: Vec<(AccountId, u64)>,
215    balances_issuance: u64,
216) -> serde_json::Value {
217    serde_json::json!({
218        "balances": { "balances": balances.to_vec() },
219        "aura": { "authorities": initial_authorities.iter().map(|x| (x.0.clone())).collect::<Vec<_>>() },
220        "grandpa": { "authorities": initial_authorities
221                .iter()
222                .map(|x| (x.1.clone(), 1))
223                .collect::<Vec<_>>(),
224        },
225        "sudo": { "key": Some(<AccountId32 as Ss58Codec>::from_ss58check("5FCM3DBXWiGcwYYQtT8z4ZD93TqYpYxjaAfgv6aMStV1FTCT").unwrap()) },
226        "subtensorModule": {
227            "stakes": stakes,
228            "balancesIssuance": balances_issuance,
229        }
230    })
