49    /// Maximum number of logs in a query.

51    pub max_past_logs: u32,

53    /// Maximum fee history cache size.

59
60    /// Maximum allowed gas limit will be `block.gas_limit * execute_gas_limit_multiplier`

62    #[arg(long, default_value = "10")]
63    pub execute_gas_limit_multiplier: u64,
64
65    /// Size in bytes of the LRU cache for block data.
66    #[arg(long, default_value = "50")]
67    pub eth_log_block_cache: usize,
68
69    /// Size in bytes of the LRU cache for transactions statuses data.
70    #[arg(long, default_value = "50")]
71    pub eth_statuses_cache: usize,
72
73    /// Sets the frontier backend type (KeyValue or Sql)
74    #[arg(long, value_enum, ignore_case = true, default_value_t = BackendType::default())]
75    pub frontier_backend_type: BackendType,
76
77    // Sets the SQL backend's pool size.
78    #[arg(long, default_value = "100")]
79    pub frontier_sql_backend_pool_size: u32,
80
81    /// Sets the SQL backend's query timeout in number of VM ops.
82    #[arg(long, default_value = "10000000")]
83    pub frontier_sql_backend_num_ops_timeout: u32,
84
85    /// Sets the SQL backend's auxiliary thread limit.
86    #[arg(long, default_value = "4")]
87    pub frontier_sql_backend_thread_count: u32,
88
89    /// Sets the SQL backend's query timeout in number of VM ops.
90    /// Default value is 200MB.
91    #[arg(long, default_value = "209715200")]
92    pub frontier_sql_backend_cache_size: u64,

115#[allow(clippy::too_many_arguments)]
116pub async fn spawn_frontier_tasks(
117    task_manager: &TaskManager,
118    client: Arc<FullClient>,
119    backend: Arc<FullBackend>,
120    frontier_backend: Arc<FrontierBackend>,
121    filter_pool: Option<FilterPool>,
122    storage_override: Arc<dyn StorageOverride<Block>>,
123    fee_history_cache: FeeHistoryCache,
124    fee_history_cache_limit: FeeHistoryCacheLimit,
125    sync: Arc<SyncingService<Block>>,
126    pubsub_notification_sinks: Arc<
127        fc_mapping_sync::EthereumBlockNotificationSinks<
128            fc_mapping_sync::EthereumBlockNotification<Block>,
129        >,
130    >,
131) {
132    // Spawn main mapping sync worker background task.
133    match &*frontier_backend {
134        fc_db::Backend::KeyValue(b) => {
135            task_manager.spawn_essential_handle().spawn(
136                "frontier-mapping-sync-worker",
137                Some("frontier"),
138                fc_mapping_sync::kv::MappingSyncWorker::new(
139                    client.import_notification_stream(),
140                    Duration::new(6, 0),
141                    client.clone(),
142                    backend,
143                    storage_override.clone(),
144                    b.clone(),
145                    3,
146                    0u32,
147                    fc_mapping_sync::SyncStrategy::Normal,
148                    sync,
149                    pubsub_notification_sinks,
150                )
151                .for_each(|()| future::ready(())),
152            );
153        }
154        fc_db::Backend::Sql(b) => {
155            task_manager.spawn_essential_handle().spawn_blocking(
156                "frontier-mapping-sync-worker",
157                Some("frontier"),
158                fc_mapping_sync::sql::SyncWorker::run(
159                    client.clone(),
160                    backend,
161                    b.clone(),
162                    client.import_notification_stream(),
163                    fc_mapping_sync::sql::SyncWorkerConfig {
164                        read_notification_timeout: Duration::from_secs(30),
165                        check_indexed_blocks_interval: Duration::from_secs(60),
166                    },
167                    fc_mapping_sync::SyncStrategy::Parachain,
168                    sync,
169                    pubsub_notification_sinks,
170                ),
171            );
172        }
173    }
174
175    // Spawn Frontier EthFilterApi maintenance task.
176    if let Some(filter_pool) = filter_pool {
177        // Each filter is allowed to stay in the pool for 100 blocks.
178        const FILTER_RETAIN_THRESHOLD: u64 = 100;
179        task_manager.spawn_essential_handle().spawn(
180            "frontier-filter-pool",
181            Some("frontier"),
182            EthTask::filter_pool_task(client.clone(), filter_pool, FILTER_RETAIN_THRESHOLD),
183        );
184    }
185
186    // Spawn Frontier FeeHistory cache maintenance task.
187    task_manager.spawn_essential_handle().spawn(
188        "frontier-fee-history",
189        Some("frontier"),
190        EthTask::fee_history_task(
191            client,
192            storage_override,
193            fee_history_cache,
194            fee_history_cache_limit,
195        ),
196    );
197}

199fn extend_rpc_aet_api<P, A, CT, CIDP, EC>(
200    io: &mut RpcModule<()>,
201    deps: &EthDeps<P, A, CT, CIDP>,
202) -> Result<(), Box<dyn std::error::Error + Send + Sync>>
203where
204    P: TransactionPool<Block = Block> + 'static,
205    A: ChainApi<Block = Block> + 'static,
206    CT: ConvertTransaction<<Block as BlockT>::Extrinsic> + Send + Sync + Clone + 'static,
207    CIDP: CreateInherentDataProviders<Block, ()> + Send + Clone + 'static,
208    EC: EthConfig<Block, FullClient>,
209{
210    let mut signers = Vec::new();
211    if deps.enable_dev_signer {
212        signers.push(Box::new(EthDevSigner::new()) as Box<dyn EthSigner>);
213    }
214
215    io.merge(
216        Eth::<Block, FullClient, P, CT, FullBackend, A, CIDP, EC>::new(
217            deps.client.clone(),
218            deps.pool.clone(),
219            deps.graph.clone(),
220            deps.converter.clone(),
221            deps.sync.clone(),
222            signers,
223            deps.storage_override.clone(),
224            deps.frontier_backend.clone(),
225            deps.is_authority,
226            deps.block_data_cache.clone(),
227            deps.fee_history_cache.clone(),
228            deps.fee_history_cache_limit,
229            deps.execute_gas_limit_multiplier,
230            deps.forced_parent_hashes.clone(),
231            deps.pending_create_inherent_data_providers.clone(),
232            Some(Box::new(AuraConsensusDataProvider::new(
233                deps.client.clone(),
234            ))),
235        )
236        .replace_config::<EC>()
237        .into_rpc(),
238    )?;
239    Ok(())
240}

242fn extend_rpc_eth_filter<P, A, CT, CIDP>(
243    io: &mut RpcModule<()>,
244    deps: &EthDeps<P, A, CT, CIDP>,
245) -> Result<(), Box<dyn std::error::Error + Send + Sync>>
246where
247    P: TransactionPool<Block = Block> + 'static,
248    A: ChainApi<Block = Block> + 'static,
249    CT: ConvertTransaction<<Block as BlockT>::Extrinsic> + Send + Sync + Clone + 'static,
250    CIDP: CreateInherentDataProviders<Block, ()> + Send + Clone + 'static,
251{
252    if let Some(filter_pool) = deps.filter_pool.clone() {
253        io.merge(
254            EthFilter::new(
255                deps.client.clone(),
256                deps.frontier_backend.clone(),
257                deps.graph.clone(),
258                filter_pool,
259                500_usize, // max stored filters
260                deps.max_past_logs,
261                deps.block_data_cache.clone(),
262            )
263            .into_rpc(),
264        )?;
265    }
266    Ok(())
267}

358pub fn create_eth<P, A, CT, CIDP, EC>(
359    mut io: RpcModule<()>,
360    deps: EthDeps<P, A, CT, CIDP>,
361    subscription_task_executor: SubscriptionTaskExecutor,

369    P: TransactionPool<Block = Block> + 'static,
370    A: ChainApi<Block = Block> + 'static,
371    CT: ConvertTransaction<<Block as BlockT>::Extrinsic> + Send + Sync + Clone + 'static,
