44pub fn new_partial<BIQ>(
45    config: &Configuration,
46    eth_config: &EthConfiguration,
47    build_import_queue: BIQ,
48) -> Result<
49    PartialComponents<
50        FullClient,
51        FullBackend,
52        FullSelectChain,
53        BasicQueue<Block>,
54        FullPool<Block, FullClient>,
55        (
56            Option<Telemetry>,
57            BoxBlockImport<Block>,
58            GrandpaLinkHalf,
59            FrontierBackend,
60            Arc<dyn StorageOverride<Block>>,
61        ),
62    >,
63    ServiceError,
64>
65where
66    BIQ: FnOnce(
67        Arc<FullClient>,
68        &Configuration,
69        &EthConfiguration,
70        &TaskManager,
71        Option<TelemetryHandle>,
72        GrandpaBlockImport,
73    ) -> Result<(BasicQueue<Block>, BoxBlockImport<Block>), ServiceError>,
74{
75    let telemetry = config
76        .telemetry_endpoints
77        .clone()
78        .filter(|x| !x.is_empty())
79        .map(|endpoints| -> Result<_, sc_telemetry::Error> {
80            let worker = TelemetryWorker::new(16)?;
81            let telemetry = worker.handle().new_telemetry(endpoints);
82            Ok((worker, telemetry))
83        })
84        .transpose()?;
85
86    let executor = sc_service::new_wasm_executor::<HostFunctions>(&config.executor);
87    let (client, backend, keystore_container, task_manager) =
88        sc_service::new_full_parts::<Block, RuntimeApi, RuntimeExecutor>(
89            config,
90            telemetry.as_ref().map(|(_, telemetry)| telemetry.handle()),
91            executor,
92        )?;
93
94    let client = Arc::new(client);
95
96    let telemetry = telemetry.map(|(worker, telemetry)| {
97        task_manager
98            .spawn_handle()
99            .spawn("telemetry", None, worker.run());
100        telemetry
101    });
102
103    let select_chain = sc_consensus::LongestChain::new(backend.clone());
104    let (grandpa_block_import, grandpa_link) = sc_consensus_grandpa::block_import(
105        client.clone(),
106        GRANDPA_JUSTIFICATION_PERIOD,
107        &client,
108        select_chain.clone(),
109        telemetry.as_ref().map(|x| x.handle()),
110    )?;
111
112    let storage_override = Arc::new(StorageOverrideHandler::<_, _, _>::new(client.clone()));
113    let frontier_backend = match eth_config.frontier_backend_type {
114        BackendType::KeyValue => FrontierBackend::KeyValue(Arc::new(fc_db::kv::Backend::open(
115            Arc::clone(&client),
116            &config.database,
117            &db_config_dir(config),
118        )?)),
119        BackendType::Sql => {
120            let db_path = db_config_dir(config).join("sql");
121            std::fs::create_dir_all(&db_path).expect("failed creating sql db directory");
122            let backend = futures::executor::block_on(fc_db::sql::Backend::new(
123                fc_db::sql::BackendConfig::Sqlite(fc_db::sql::SqliteBackendConfig {
124                    path: Path::new("sqlite:///")
125                        .join(db_path)
126                        .join("frontier.db3")
127                        .to_str()
128                        .unwrap_or(""),
129                    create_if_missing: true,
130                    thread_count: eth_config.frontier_sql_backend_thread_count,
131                    cache_size: eth_config.frontier_sql_backend_cache_size,
132                }),
133                eth_config.frontier_sql_backend_pool_size,
134                std::num::NonZeroU32::new(eth_config.frontier_sql_backend_num_ops_timeout),
135                storage_override.clone(),
136            ))
137            .unwrap_or_else(|err| panic!("failed creating sql backend: {:?}", err));
138            FrontierBackend::Sql(Arc::new(backend))
139        }
140    };
141
142    let (import_queue, block_import) = build_import_queue(
143        client.clone(),
144        config,
145        eth_config,
146        &task_manager,
147        telemetry.as_ref().map(|x| x.handle()),
148        grandpa_block_import,
149    )?;
150
151    let transaction_pool = sc_transaction_pool::BasicPool::new_full(
152        config.transaction_pool.clone(),
153        config.role.is_authority().into(),
154        config.prometheus_registry(),
155        task_manager.spawn_essential_handle(),
156        client.clone(),
157    );
158
159    Ok(PartialComponents {
160        client,
161        backend,
162        keystore_container,
163        task_manager,
164        select_chain,
165        import_queue,
166        transaction_pool,
167        other: (
168            telemetry,
169            block_import,
170            grandpa_link,
171            frontier_backend,
172            storage_override,
173        ),
174    })
175}

177#[derive(Debug, thiserror::Error)]
178pub enum Error {
179    #[error("Multiple runtime Ethereum blocks, rejecting!")]
180    MultipleRuntimeLogs,
181    #[error("Runtime Ethereum block not found, rejecting!")]
182    NoRuntimeLog,
183    #[error("Cannot access the runtime at genesis, rejecting!")]
184    RuntimeApiCallFailed,
185}
186
187impl From<Error> for String {
188    fn from(error: Error) -> String {
189        error.to_string()
190    }
191}
192
193impl From<FindLogError> for Error {
194    fn from(error: FindLogError) -> Error {
195        match error {
196            FindLogError::NotFound => Error::NoRuntimeLog,
197            FindLogError::MultipleLogs => Error::MultipleRuntimeLogs,
198        }
199    }
200}

202impl From<Error> for ConsensusError {
203    fn from(error: Error) -> ConsensusError {
204        ConsensusError::ClientImport(error.to_string())
205    }

208pub struct ConditionalEVMBlockImport<B: BlockT, I, F, C> {
209    inner: I,
210    frontier_block_import: F,
211    client: Arc<C>,
212    _marker: PhantomData<B>,
213}
214
215impl<B, I, F, C> Clone for ConditionalEVMBlockImport<B, I, F, C>
216where
217    B: BlockT,
218    I: Clone + BlockImport<B>,
219    F: Clone + BlockImport<B>,
220{
221    fn clone(&self) -> Self {
222        ConditionalEVMBlockImport {
223            inner: self.inner.clone(),
224            frontier_block_import: self.frontier_block_import.clone(),
225            client: self.client.clone(),
226            _marker: PhantomData,
227        }
228    }
229}
230
231impl<B, I, F, C> ConditionalEVMBlockImport<B, I, F, C>
232where
233    B: BlockT,
234    I: BlockImport<B>,
235    I::Error: Into<ConsensusError>,
236    F: BlockImport<B>,
237    F::Error: Into<ConsensusError>,
238    C: ProvideRuntimeApi<B>,
239    C::Api: BlockBuilderApi<B> + EthereumRuntimeRPCApi<B>,
240{
241    pub fn new(inner: I, frontier_block_import: F, client: Arc<C>) -> Self {
242        Self {
243            inner,
244            frontier_block_import,
245            client,
246            _marker: PhantomData,
247        }
248    }
249}
250
251#[async_trait::async_trait]
252impl<B, I, F, C> BlockImport<B> for ConditionalEVMBlockImport<B, I, F, C>
253where
254    B: BlockT,
255    I: BlockImport<B> + Send + Sync,
256    I::Error: Into<ConsensusError>,
257    F: BlockImport<B> + Send + Sync,
258    F::Error: Into<ConsensusError>,
259    C: ProvideRuntimeApi<B> + Send + Sync,
260    C::Api: BlockBuilderApi<B> + EthereumRuntimeRPCApi<B>,
261{
262    type Error = ConsensusError;
263
264    async fn check_block(&self, block: BlockCheckParams<B>) -> Result<ImportResult, Self::Error> {
265        self.inner.check_block(block).await.map_err(Into::into)
266    }
267
268    async fn import_block(&self, block: BlockImportParams<B>) -> Result<ImportResult, Self::Error> {
269        // Import like Frontier, but fallback to grandpa import for errors
270        match ensure_log(block.header.digest()).map_err(Error::from) {
271            Ok(()) => self.inner.import_block(block).await.map_err(Into::into),
272            _ => self.inner.import_block(block).await.map_err(Into::into),
273        }
274    }
275}

277/// Build the import queue for the template runtime (aura + grandpa).
278pub fn build_aura_grandpa_import_queue(
279    client: Arc<FullClient>,
280    config: &Configuration,
281    _eth_config: &EthConfiguration,
282    task_manager: &TaskManager,
283    telemetry: Option<TelemetryHandle>,
284    grandpa_block_import: GrandpaBlockImport,
285) -> Result<(BasicQueue<Block>, BoxBlockImport<Block>), ServiceError>
286where
287    NumberFor<Block>: BlockNumberOps,
288{
289    let conditional_block_import = ConditionalEVMBlockImport::new(
290        grandpa_block_import.clone(),
291        FrontierBlockImport::new(grandpa_block_import.clone(), client.clone()),
292        client.clone(),
293    );
294
295    let slot_duration = sc_consensus_aura::slot_duration(&*client)?;
296    let create_inherent_data_providers = move |_, ()| async move {
297        let timestamp = sp_timestamp::InherentDataProvider::from_system_time();
298        let slot =
299            sp_consensus_aura::inherents::InherentDataProvider::from_timestamp_and_slot_duration(
300                *timestamp,
301                slot_duration,
302            );
303        Ok((slot, timestamp))
304    };
305
306    let import_queue = sc_consensus_aura::import_queue::<AuraPair, _, _, _, _, _>(
307        sc_consensus_aura::ImportQueueParams {
308            block_import: conditional_block_import.clone(),
309            justification_import: Some(Box::new(grandpa_block_import.clone())),
310            client,
311            create_inherent_data_providers,
312            spawner: &task_manager.spawn_essential_handle(),
313            registry: config.prometheus_registry(),
314            check_for_equivocation: Default::default(),
315            telemetry,
316            compatibility_mode: sc_consensus_aura::CompatibilityMode::None,
317        },
318    )
319    .map_err::<ServiceError, _>(Into::into)?;
320
321    Ok((import_queue, Box::new(conditional_block_import)))
322}

348/// Builds a new service for a full client.
349pub async fn new_full<NB>(
350    mut config: Configuration,
351    eth_config: EthConfiguration,
352    sealing: Option<Sealing>,
353) -> Result<TaskManager, ServiceError>
354where
355    NumberFor<Block>: BlockNumberOps,
356    NB: sc_network::NetworkBackend<Block, <Block as BlockT>::Hash>,
357{
358    let build_import_queue = if sealing.is_some() {
359        build_manual_seal_import_queue
360    } else {
361        build_aura_grandpa_import_queue
362    };
363
364    let PartialComponents {
365        client,
366        backend,
367        mut task_manager,
368        import_queue,
369        keystore_container,
370        select_chain,
371        transaction_pool,
372        other: (mut telemetry, block_import, grandpa_link, frontier_backend, storage_override),
373    } = new_partial(&config, &eth_config, build_import_queue)?;
374
375    let FrontierPartialComponents {
376        filter_pool,
377        fee_history_cache,
378        fee_history_cache_limit,
379    } = new_frontier_partial(&eth_config)?;
380
381    let maybe_registry = config.prometheus_config.as_ref().map(|cfg| &cfg.registry);
382    let mut net_config = sc_network::config::FullNetworkConfiguration::<_, _, NB>::new(
383        &config.network,
384        maybe_registry.cloned(),
385    );
386    let peer_store_handle = net_config.peer_store_handle();
387    let metrics = NB::register_notification_metrics(maybe_registry);
388
389    let grandpa_protocol_name = sc_consensus_grandpa::protocol_standard_name(
390        &client.block_hash(0u32)?.expect("Genesis block exists; qed"),
391        &config.chain_spec,
392    );
393
394    let (grandpa_protocol_config, grandpa_notification_service) =
395        sc_consensus_grandpa::grandpa_peers_set_config::<_, NB>(
396            grandpa_protocol_name.clone(),
397            metrics.clone(),
398            peer_store_handle,
399        );
400
401    let warp_sync_config = if sealing.is_some() {
402        None
403    } else {
404        net_config.add_notification_protocol(grandpa_protocol_config);
405        let warp_sync: Arc<dyn WarpSyncProvider<Block>> =
406            Arc::new(sc_consensus_grandpa::warp_proof::NetworkProvider::new(
407                backend.clone(),
408                grandpa_link.shared_authority_set().clone(),
409                Vec::new(),
410            ));
411        Some(WarpSyncConfig::WithProvider(warp_sync))
412    };
413
414    let (network, system_rpc_tx, tx_handler_controller, network_starter, sync_service) =
415        sc_service::build_network(sc_service::BuildNetworkParams {
416            config: &config,
417            net_config,
418            client: client.clone(),
419            transaction_pool: transaction_pool.clone(),
420            spawn_handle: task_manager.spawn_handle(),
421            import_queue,
422            block_announce_validator_builder: None,
423            warp_sync_config,
424            block_relay: None,
425            metrics,
426        })?;
427
428    if config.offchain_worker.enabled && config.role.is_authority() {
429        let public_keys = keystore_container
430            .keystore()
431            .sr25519_public_keys(pallet_drand::KEY_TYPE);
432
433        if public_keys.is_empty() {
434            match sp_keystore::Keystore::sr25519_generate_new(
435                &*keystore_container.keystore(),
436                pallet_drand::KEY_TYPE,
437                None,
438            ) {
439                Ok(_) => {
440                    log::debug!("Offchain worker key generated");
441                }
442                Err(e) => {
443                    log::error!("Failed to create SR25519 key for offchain worker: {:?}", e);
444                }
445            }
446        } else {
447            log::debug!("Offchain worker key already exists");
448        }
449
450        task_manager.spawn_essential_handle().spawn(
451            "offchain-workers-runner",
452            None,
453            sc_offchain::OffchainWorkers::new(sc_offchain::OffchainWorkerOptions {
454                runtime_api_provider: client.clone(),
455                is_validator: config.role.is_authority(),
456                keystore: Some(keystore_container.keystore()),
457                offchain_db: backend.offchain_storage(),
458                transaction_pool: Some(OffchainTransactionPoolFactory::new(
459                    transaction_pool.clone(),
460                )),
461                network_provider: Arc::new(network.clone()),
462                enable_http_requests: true,
463                custom_extensions: |_| vec![],
464            })
465            .run(client.clone(), task_manager.spawn_handle())
466            .boxed(),
467        );
468    }
469
470    let role = config.role;
471    let force_authoring = config.force_authoring;
472    let backoff_authoring_blocks =
473        Some(BackoffAuthoringOnFinalizedHeadLagging::<NumberFor<Block>> {
474            unfinalized_slack: 6u32,
475            ..Default::default()
476        });
477    let name = config.network.node_name.clone();
478    let frontier_backend = Arc::new(frontier_backend);
479    let enable_grandpa = !config.disable_grandpa && sealing.is_none();
480    let prometheus_registry = config.prometheus_registry().cloned();
481
482    // Channel for the rpc handler to communicate with the authorship task.
483    let (command_sink, commands_stream) = mpsc::channel(1000);
484
485    // Sinks for pubsub notifications.
486    // Everytime a new subscription is created, a new mpsc channel is added to the sink pool.
487    // The MappingSyncWorker sends through the channel on block import and the subscription emits a notification to the subscriber on receiving a message through this channel.
488    // This way we avoid race conditions when using native substrate block import notification stream.
489    let pubsub_notification_sinks: fc_mapping_sync::EthereumBlockNotificationSinks<
490        fc_mapping_sync::EthereumBlockNotification<Block>,
491    > = Default::default();
492    let pubsub_notification_sinks = Arc::new(pubsub_notification_sinks);
493
494    // for ethereum-compatibility rpc.
495    config.rpc.id_provider = Some(Box::new(fc_rpc::EthereumSubIdProvider));
496
497    let rpc_builder = {
498        let client = client.clone();
499        let pool = transaction_pool.clone();
500        let network = network.clone();
501        let sync_service = sync_service.clone();
502
503        let is_authority = role.is_authority();
504        let enable_dev_signer = eth_config.enable_dev_signer;
505        let max_past_logs = eth_config.max_past_logs;
506        let execute_gas_limit_multiplier = eth_config.execute_gas_limit_multiplier;
507        let filter_pool = filter_pool.clone();
508        let frontier_backend = frontier_backend.clone();
509        let pubsub_notification_sinks = pubsub_notification_sinks.clone();
510        let storage_override = storage_override.clone();
511        let fee_history_cache = fee_history_cache.clone();
512        let block_data_cache = Arc::new(fc_rpc::EthBlockDataCacheTask::new(
513            task_manager.spawn_handle(),
514            storage_override.clone(),
515            eth_config.eth_log_block_cache,
516            eth_config.eth_statuses_cache,
517            prometheus_registry.clone(),
518        ));
519
520        let slot_duration = sc_consensus_aura::slot_duration(&*client)?;
521        let pending_create_inherent_data_providers = move |_, ()| async move {
522            let current = sp_timestamp::InherentDataProvider::from_system_time();
523            let next_slot = current
524                .timestamp()
525                .as_millis()
526                .saturating_add(slot_duration.as_millis());
527            let timestamp = sp_timestamp::InherentDataProvider::new(next_slot.into());
528            let slot = sp_consensus_aura::inherents::InherentDataProvider::from_timestamp_and_slot_duration(
529				*timestamp,
530				slot_duration,
531			);
532            Ok((slot, timestamp))
533        };
534
535        Box::new(move |subscription_task_executor| {
536            let eth_deps = crate::rpc::EthDeps {
537                client: client.clone(),
538                pool: pool.clone(),
539                graph: pool.pool().clone(),
540                converter: Some(TransactionConverter::<Block>::default()),
541                is_authority,
542                enable_dev_signer,
543                network: network.clone(),
544                sync: sync_service.clone(),
545                frontier_backend: match &*frontier_backend {
546                    fc_db::Backend::KeyValue(b) => b.clone(),
547                    fc_db::Backend::Sql(b) => b.clone(),
548                },
549                storage_override: storage_override.clone(),
550                block_data_cache: block_data_cache.clone(),
551                filter_pool: filter_pool.clone(),
552                max_past_logs,
553                fee_history_cache: fee_history_cache.clone(),
554                fee_history_cache_limit,
555                execute_gas_limit_multiplier,
556                forced_parent_hashes: None,
557                pending_create_inherent_data_providers,
558            };
559            let deps = crate::rpc::FullDeps {
560                client: client.clone(),
561                pool: pool.clone(),
562                command_sink: if sealing.is_some() {
563                    Some(command_sink.clone())
564                } else {
565                    None
566                },
567                eth: eth_deps,
568            };
569            crate::rpc::create_full(
570                deps,
571                subscription_task_executor,
572                pubsub_notification_sinks.clone(),
573            )
574            .map_err(Into::into)
575        })
576    };
577
578    let _rpc_handlers = sc_service::spawn_tasks(sc_service::SpawnTasksParams {
579        config,
580        client: client.clone(),
581        backend: backend.clone(),
582        task_manager: &mut task_manager,
583        keystore: keystore_container.keystore(),
584        transaction_pool: transaction_pool.clone(),
585        rpc_builder,
586        network: network.clone(),
587        system_rpc_tx,
588        tx_handler_controller,
589        sync_service: sync_service.clone(),
590        telemetry: telemetry.as_mut(),
591    })?;
592
593    spawn_frontier_tasks(
594        &task_manager,
595        client.clone(),
596        backend,
597        frontier_backend,
598        filter_pool,
599        storage_override,
600        fee_history_cache,
601        fee_history_cache_limit,
602        sync_service.clone(),
603        pubsub_notification_sinks,
604    )
605    .await;
606
607    if role.is_authority() {
608        // manual-seal authorship
609        if let Some(sealing) = sealing {
610            run_manual_seal_authorship(
611                sealing,
612                client,
613                transaction_pool,
614                select_chain,
615                block_import,
616                &task_manager,
617                prometheus_registry.as_ref(),
618                telemetry.as_ref(),
619                commands_stream,
620            )?;
621
622            network_starter.start_network();
623            log::info!("Manual Seal Ready");
624            return Ok(task_manager);
625        }
626
627        let proposer_factory = sc_basic_authorship::ProposerFactory::new(
628            task_manager.spawn_handle(),
629            client.clone(),
630            transaction_pool.clone(),
631            prometheus_registry.as_ref(),
632            telemetry.as_ref().map(|x| x.handle()),
633        );
634
635        let slot_duration = sc_consensus_aura::slot_duration(&*client)?;
636        let create_inherent_data_providers = move |_, ()| async move {
637            let timestamp = sp_timestamp::InherentDataProvider::from_system_time();
638            let slot = sp_consensus_aura::inherents::InherentDataProvider::from_timestamp_and_slot_duration(
639				*timestamp,
640				slot_duration,
641			);
642            Ok((slot, timestamp))
643        };
644
645        let aura = sc_consensus_aura::start_aura::<AuraPair, _, _, _, _, _, _, _, _, _, _>(
646            sc_consensus_aura::StartAuraParams {
647                slot_duration,
648                client,
649                select_chain,
650                block_import,
651                proposer_factory,
652                sync_oracle: sync_service.clone(),
653                justification_sync_link: sync_service.clone(),
654                create_inherent_data_providers,
655                force_authoring,
656                backoff_authoring_blocks,
657                keystore: keystore_container.keystore(),
658                block_proposal_slot_portion: sc_consensus_aura::SlotProportion::new(2f32 / 3f32),
659                max_block_proposal_slot_portion: None,
660                telemetry: telemetry.as_ref().map(|x| x.handle()),
661                compatibility_mode: sc_consensus_aura::CompatibilityMode::None,
662            },
663        )?;
664        // the AURA authoring task is considered essential, i.e. if it
665        // fails we take down the service with it.
666        task_manager
667            .spawn_essential_handle()
668            .spawn_blocking("aura", Some("block-authoring"), aura);
669    }
670
671    if enable_grandpa {
672        // if the node isn't actively participating in consensus then it doesn't
673        // need a keystore, regardless of which protocol we use below.
674        let keystore = if role.is_authority() {
675            Some(keystore_container.keystore())
676        } else {
677            None
678        };
679
680        let grandpa_config = sc_consensus_grandpa::Config {
681            // FIXME #1578 make this available through chainspec
682            gossip_duration: Duration::from_millis(333),
683            justification_generation_period: GRANDPA_JUSTIFICATION_PERIOD,
684            name: Some(name),
685            observer_enabled: false,
686            keystore,
687            local_role: role,
688            telemetry: telemetry.as_ref().map(|x| x.handle()),
689            protocol_name: grandpa_protocol_name,
690        };
691
692        // start the full GRANDPA voter
693        // NOTE: non-authorities could run the GRANDPA observer protocol, but at
694        // this point the full voter should provide better guarantees of block
695        // and vote data availability than the observer. The observer has not
696        // been tested extensively yet and having most nodes in a network run it
697        // could lead to finality stalls.
698        let grandpa_voter =
699            sc_consensus_grandpa::run_grandpa_voter(sc_consensus_grandpa::GrandpaParams {
700                config: grandpa_config,
701                link: grandpa_link,
702                network,
703                sync: sync_service,
704                notification_service: grandpa_notification_service,
705                voting_rule: sc_consensus_grandpa::VotingRulesBuilder::default().build(),
706                prometheus_registry,
707                shared_voter_state: sc_consensus_grandpa::SharedVoterState::empty(),
708                telemetry: telemetry.as_ref().map(|x| x.handle()),
709                offchain_tx_pool_factory: OffchainTransactionPoolFactory::new(transaction_pool),
710            })?;
711
712        // the GRANDPA voter task is considered infallible, i.e.
713        // if it fails we take down the service with it.
714        task_manager
715            .spawn_essential_handle()
716            .spawn_blocking("grandpa-voter", None, grandpa_voter);
717    }
718
719    network_starter.start_network();
720    Ok(task_manager)
721}

763#[allow(clippy::too_many_arguments)]
764fn run_manual_seal_authorship(
765    sealing: Sealing,
766    client: Arc<FullClient>,
767    transaction_pool: Arc<FullPool<Block, FullClient>>,
768    select_chain: FullSelectChain,
769    block_import: BoxBlockImport<Block>,
770    task_manager: &TaskManager,
771    prometheus_registry: Option<&Registry>,
772    telemetry: Option<&Telemetry>,
773    commands_stream: mpsc::Receiver<
774        sc_consensus_manual_seal::rpc::EngineCommand<<Block as BlockT>::Hash>,
775    >,
776) -> Result<(), ServiceError> {
777    let proposer_factory = sc_basic_authorship::ProposerFactory::new(
778        task_manager.spawn_handle(),
779        client.clone(),
780        transaction_pool.clone(),
781        prometheus_registry,
782        telemetry.as_ref().map(|x| x.handle()),
783    );
784
785    thread_local!(static TIMESTAMP: RefCell<u64> = const { RefCell::new(0) });
786
787    /// Provide a mock duration starting at 0 in millisecond for timestamp inherent.
788    /// Each call will increment timestamp by slot_duration making Aura think time has passed.
789    struct MockTimestampInherentDataProvider;
790
791    #[async_trait::async_trait]
792    impl sp_inherents::InherentDataProvider for MockTimestampInherentDataProvider {
793        async fn provide_inherent_data(
794            &self,
795            inherent_data: &mut sp_inherents::InherentData,
796        ) -> Result<(), sp_inherents::Error> {
797            TIMESTAMP.with(|x| {
798                let mut x_ref = x.borrow_mut();
799                *x_ref = x_ref.saturating_add(subtensor_runtime_common::time::SLOT_DURATION);
800                inherent_data.put_data(sp_timestamp::INHERENT_IDENTIFIER, &*x.borrow())
801            })
802        }
803
804        async fn try_handle_error(
805            &self,
806            _identifier: &sp_inherents::InherentIdentifier,
807            _error: &[u8],
808        ) -> Option<Result<(), sp_inherents::Error>> {
809            // The pallet never reports error.
810            None
811        }
812    }
813
814    let create_inherent_data_providers =
815        move |_, ()| async move { Ok(MockTimestampInherentDataProvider) };
816
817    let manual_seal = match sealing {
818        Sealing::Manual => future::Either::Left(sc_consensus_manual_seal::run_manual_seal(
819            sc_consensus_manual_seal::ManualSealParams {
820                block_import,
821                env: proposer_factory,
822                client,
823                pool: transaction_pool,
824                commands_stream,
825                select_chain,
826                consensus_data_provider: None,
827                create_inherent_data_providers,
828            },
829        )),
830        Sealing::Instant => future::Either::Right(sc_consensus_manual_seal::run_instant_seal(
831            sc_consensus_manual_seal::InstantSealParams {
832                block_import,
833                env: proposer_factory,
834                client,
835                pool: transaction_pool,
836                select_chain,
837                consensus_data_provider: None,
838                create_inherent_data_providers,
839            },
840        )),
841    };
842
843    // we spawn the future on a background thread managed by service.
844    task_manager
845        .spawn_essential_handle()
846        .spawn_blocking("manual-seal", None, manual_seal);
847    Ok(())
848}
